fn as_char(self) -> char

Makes a char from self

§Example
use winnow::prelude::*;

assert_eq!('a'.as_char(), 'a');
assert_eq!(u8::MAX.as_char(), std::char::from_u32(u8::MAX as u32).unwrap());
