9/// Account information.
10#[derive(Clone, Debug, Default, PartialEq, Eq)]
11#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
12#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
13pub struct AccountInfo {
14    /// Account balance
15    pub balance: U256,
16    /// Account nonce
17    #[cfg_attr(feature = "serde", serde(with = "alloy_serde::quantity"))]
18    pub nonce: u64,
19    /// Account code
20    pub code: Bytes,
21}
22
23impl AccountInfo {
24    /// Returns true if the code hash is the Keccak256 hash of the empty string `""`.
25    #[inline]
26    pub fn is_empty_code_hash(&self) -> bool {
27        self.code.is_empty()
28    }
29
30    /// Returns the code hash of the account.
31    pub fn code_hash(&self) -> B256 {
32        if self.code.is_empty() {
33            KECCAK256_EMPTY
34        } else {
35            alloy_primitives::keccak256(&self.code)
36        }
37    }
38
39    /// Returns if an account is empty.
40    ///
41    /// An account is empty if the following conditions are met.
42    /// - code hash is the Keccak256 hash of the empty string `""`
43    /// - balance is zero
44    /// - nonce is zero
45    #[inline]
46    pub fn is_empty(&self) -> bool {
47        self.is_empty_code_hash() && self.balance.is_zero() && self.nonce == 0
48    }
49}

77/// Response for EIP-1186 account proof `eth_getProof`
78#[derive(Clone, Debug, Default, PartialEq, Eq)]
79#[cfg(feature = "serde")]
80#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
81#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
82pub struct EIP1186AccountProofResponse {
83    /// The account address.
84    pub address: Address,
85    /// The account balance.
86    pub balance: U256,
87    /// The hash of the code of the account.
88    pub code_hash: B256,
89    /// The account nonce.
90    #[cfg_attr(feature = "serde", serde(with = "alloy_serde::quantity"))]
91    pub nonce: u64,
92    /// The hash of the storage of the account.
93    pub storage_hash: B256,
94    /// The account proof.
95    pub account_proof: Vec<Bytes>,
96    /// The storage proof.
97    pub storage_proof: Vec<EIP1186StorageProof>,
98}
99
100#[cfg(feature = "serde")]
101impl EIP1186AccountProofResponse {
102    /// After `SpuriousDragon` empty account is defined as account with nonce == 0 && balance == 0
103    /// && bytecode = None (or hash is [`KECCAK_EMPTY`](alloy_consensus::constants::KECCAK_EMPTY)).
104    pub fn is_empty(&self) -> bool {
105        self.nonce == 0
106            && self.balance.is_zero()
107            && self.storage_hash.is_zero()
108            && self.code_hash == alloy_consensus::constants::KECCAK_EMPTY
109    }
110}
