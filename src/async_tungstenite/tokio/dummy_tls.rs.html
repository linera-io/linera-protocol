31/// Creates a WebSocket handshake from a request and a stream,
32/// upgrading the stream to TLS if required and using the given
33/// connector and WebSocket configuration.
34pub async fn client_async_tls_with_connector_and_config<R, S>(
35    request: R,
36    stream: S,
37    connector: Option<Connector>,
38    config: Option<WebSocketConfig>,
39) -> Result<(WebSocketStream<AutoStream<S>>, Response), Error>
40where
41    R: IntoClientRequest + Unpin,
42    S: 'static + tokio::io::AsyncRead + tokio::io::AsyncWrite + Unpin,
43    AutoStream<S>: Unpin,
44{
45    let request: Request = request.into_client_request()?;
46
47    let domain = domain(&request)?;
48
49    // Make sure we check domain and mode first. URL must be valid.
50    let mode = uri_mode(request.uri())?;
51
52    let stream = wrap_stream(stream, domain, connector, mode).await?;
53    client_async_with_config(request, stream, config).await
54}
