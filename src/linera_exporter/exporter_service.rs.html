17pub(crate) struct ExporterService {
18    block_processor_sender: UnboundedSender<BlockId>,
19}
20
21#[async_trait]
22impl NotifierService for ExporterService {
23    async fn notify_batch(
24        &self,
25        request: Request<NotificationBatch>,
26    ) -> Result<Response<()>, Status> {
27        for notification in request.into_inner().notifications {
28            let block_id =
29                match parse_notification(notification).map_err(|e| Status::from_error(e.into())) {
30                    Ok(block_id) => {
31                        tracing::debug!(
32                            ?block_id,
33                            "received new block notification from notifier service"
34                        );
35                        block_id
36                    }
37                    Err(error) => {
38                        // We assume errors when parsing are not critical and just log them.
39                        tracing::warn!(?error, "received bad notification");
40                        continue;
41                    }
42                };
43
44            #[cfg(with_metrics)]
45            crate::metrics::EXPORTER_NOTIFICATION_QUEUE_LENGTH.inc();
46            self.block_processor_sender
47                .send(block_id)
48                .expect("sender should never fail");
49        }
50
51        Ok(Response::new(()))
52    }
53}
54
55impl ExporterService {
56    pub(crate) fn new(sender: UnboundedSender<BlockId>) -> ExporterService {
57        ExporterService {
58            block_processor_sender: sender,
59        }
60    }
61
62    pub async fn run(
63        self,
64        cancellation_token: CancellationToken,
65        port: u16,
66    ) -> core::result::Result<(), ExporterError> {
67        info!("Linera exporter is running.");
68        self.start_notification_server(port, cancellation_token)
69            .await
70    }
71
72    async fn start_notification_server(
73        self,
74        port: u16,
75        cancellation_token: CancellationToken,
76    ) -> core::result::Result<(), ExporterError> {
77        let endpoint = get_address(port);
78        info!(
79            "Starting linera_exporter_service on endpoint = {}",
80            endpoint
81        );
82
83        let (health_reporter, health_service) = tonic_health::server::health_reporter();
84        health_reporter
85            .set_serving::<NotifierServiceServer<Self>>()
86            .await;
87
88        Server::builder()
89            .add_service(health_service)
90            .add_service(NotifierServiceServer::new(self))
91            .serve_with_shutdown(endpoint, cancellation_token.cancelled_owned())
92            .await
93            .expect("a running notification server");
94
95        Ok(())
96    }
97}

99fn parse_notification(notification: Notification) -> core::result::Result<BlockId, ExporterError> {
100    let chain_id = notification
101        .chain_id
102        .ok_or(BadNotificationKind::InvalidChainId { inner: None })?
103        .try_into()
104        .map_err(|err| BadNotificationKind::InvalidChainId { inner: Some(err) })?;
105
106    let reason = bincode::deserialize::<Reason>(&notification.reason)
107        .map_err(|err| BadNotificationKind::InvalidReason { inner: Some(err) })?;
108
109    if let Reason::NewBlock { height, hash } = reason {
110        return Ok(BlockId::new(chain_id, hash, height));
111    }
112
113    Err(BadNotificationKind::InvalidReason { inner: None })?
114}

116#[cfg(test)]
117mod test {
118    use linera_base::{crypto::CryptoHash, identifiers::ChainId, port::get_free_port};
119    use linera_core::worker::Notification;
120    use linera_rpc::grpc::api::notifier_service_client::NotifierServiceClient;
121    use linera_service::cli_wrappers::local_net::LocalNet;
122    use tokio::sync::mpsc::unbounded_channel;
123
124    use super::*;
125
126    #[test_log::test(tokio::test)]
127    async fn test_notification_server() -> anyhow::Result<()> {
128        let port = get_free_port().await?;
129        let endpoint = format!("127.0.0.1:{port}");
130        let cancellation_token = CancellationToken::new();
131        let (tx, mut rx) = unbounded_channel();
132        let server = ExporterService::new(tx);
133        let server_handle = tokio::spawn(server.run(cancellation_token.clone(), port));
134        LocalNet::ensure_grpc_server_has_started("test server", port as usize, "http").await?;
135
136        let mut client = NotifierServiceClient::connect(format!("http://{endpoint}")).await?;
137        let reason = Reason::NewBlock {
138            height: 4.into(),
139            hash: CryptoHash::test_hash("s"),
140        };
141        let request = Notification {
142            chain_id: ChainId::default(),
143            reason,
144        };
145
146        let notification_batch = linera_rpc::grpc::api::NotificationBatch {
147            notifications: vec![request.try_into().unwrap()],
148        };
149        assert!(client
150            .notify_batch(Request::new(notification_batch))
151            .await
152            .is_ok());
153
154        let expected_block_id =
155            BlockId::new(ChainId::default(), CryptoHash::test_hash("s"), 4.into());
156        assert!(rx
157            .recv()
158            .await
159            .is_some_and(|block_id| block_id == expected_block_id));
160
161        cancellation_token.cancel();
162        assert!(server_handle.await.is_ok());
163
164        Ok(())
165    }
