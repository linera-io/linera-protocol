33/// Options for running the linera block exporter.
34#[derive(clap::Parser, Debug, Clone)]
35#[command(
36    name = "Linera Exporter",
37    version = linera_version::VersionInfo::default_clap_str(),
38)]
39struct ExporterOptions {
40    /// Path to the TOML file describing the configuration for the block exporter.
41    #[arg(long)]
42    config_path: PathBuf,
43
44    /// Storage configuration for the blockchain history, chain states and binary blobs.
45    #[arg(long = "storage")]
46    storage_config: StorageConfig,
47
48    /// Common storage options.
49    #[command(flatten)]
50    common_storage_options: CommonStorageOptions,
51
52    /// Maximum number of threads to use for exporters
53    #[arg(long, default_value = "16")]
54    max_exporter_threads: usize,
55
56    /// Timeout in milliseconds for sending queries.
57    #[arg(long = "send-timeout-ms", default_value = "4000", value_parser = util::parse_millis)]
58    pub send_timeout: Duration,
59
60    /// Timeout in milliseconds for receiving responses.
61    #[arg(long = "recv-timeout-ms", default_value = "4000", value_parser = util::parse_millis)]
62    pub recv_timeout: Duration,
63
64    /// Delay increment for retrying to connect to a destination.
65    #[arg(
66        long = "retry-delay-ms",
67        default_value = "1000",
68        value_parser = util::parse_millis
69    )]
70    pub retry_delay: Duration,
71
72    /// Number of times to retry connecting to a destination.
73    #[arg(long, default_value = "10")]
74    pub max_retries: u32,
75}
76
77struct ExporterContext {
78    node_options: NodeOptions,
79    config: BlockExporterConfig,
80}
81
82#[async_trait]
83impl Runnable for ExporterContext {
84    type Output = Result<(), ExporterError>;
85
86    async fn run<S>(self, storage: S) -> Self::Output
87    where
88        S: Storage + Clone + Send + Sync + 'static,
89    {
90        let shutdown_notifier = CancellationToken::new();
91        tokio::spawn(listen_for_shutdown_signals(shutdown_notifier.clone()));
92
93        let (sender, handle) = start_block_processor_task(
94            storage,
95            ExporterCancellationSignal::new(shutdown_notifier.clone()),
96            self.config.limits,
97            self.node_options,
98            self.config.id,
99            self.config.destination_config,
100        )?;
101
102        let service = ExporterService::new(sender);
103        service
104            .run(shutdown_notifier, self.config.service_config.port)
105            .await?;
106        handle.join().unwrap()
107    }
108}
109
110impl ExporterContext {
111    fn new(node_options: NodeOptions, config: BlockExporterConfig) -> ExporterContext {
112        Self {
113            config,
114            node_options,
115        }
116    }
117}
118
119fn main() -> Result<()> {
120    linera_base::tracing::init("linera-exporter");
121    let options = <ExporterOptions as clap::Parser>::parse();
122    options.run()
123}
124
125impl ExporterOptions {
126    fn run(&self) -> anyhow::Result<()> {
127        let config_string = fs_err::read_to_string(&self.config_path)
128            .expect("Unable to read the configuration file");
129        let config: BlockExporterConfig =
130            toml::from_str(&config_string).expect("Invalid configuration file format");
131
132        let node_options = NodeOptions {
133            send_timeout: self.send_timeout,
134            recv_timeout: self.recv_timeout,
135            retry_delay: self.retry_delay,
136            max_retries: self.max_retries,
137        };
138
139        let context = ExporterContext::new(node_options, config);
140
141        let runtime = tokio::runtime::Builder::new_multi_thread()
142            .thread_name("block-exporter-worker")
143            .worker_threads(self.max_exporter_threads)
144            .enable_all()
145            .build()?;
146
147        let future = async {
148            let store_config = self
149                .storage_config
150                .add_common_storage_options(&self.common_storage_options)
151                .await
152                .unwrap();
153            store_config.run_with_storage(None, context).boxed().await
154        };
155
156        runtime.block_on(future)?.map_err(|e| e.into())
157    }
158}
