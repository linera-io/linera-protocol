34pub(crate) fn start_block_processor_task<S, F>(
35    storage: S,
36    shutdown_signal: F,
37    limits: LimitsConfig,
38    options: NodeOptions,
39    block_exporter_id: u32,
40    destination_config: DestinationConfig,
41) -> (
42    UnboundedSender<BlockId>,
43    std::thread::JoinHandle<Result<(), ExporterError>>,
44)
45where
46    S: Storage + Clone + Send + Sync + 'static,
47    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
48    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
49{
50    let (task_sender, queue_front) = unbounded_channel();
51    let new_block_queue = NewBlockQueue {
52        queue_rear: task_sender.clone(),
53        queue_front,
54    };
55    let handle = std::thread::spawn(move || {
56        start_block_processor(
57            storage,
58            shutdown_signal,
59            limits,
60            options,
61            block_exporter_id,
62            new_block_queue,
63            destination_config,
64        )
65    });
66
67    (task_sender, handle)
68}
69
70struct NewBlockQueue {
71    pub(crate) queue_rear: UnboundedSender<BlockId>,
72    pub(crate) queue_front: UnboundedReceiver<BlockId>,
73}
74
75impl NewBlockQueue {
76    async fn recv(&mut self) -> Option<BlockId> {
77        let block = self.queue_front.recv().await;
78        #[cfg(with_metrics)]
79        crate::metrics::EXPORTER_NOTIFICATION_QUEUE_LENGTH.dec();
80        block
81    }
82
83    fn push_back(&self, block_id: BlockId) {
84        self.queue_rear
85            .send(block_id)
86            .expect("sender should never fail");
87        #[cfg(with_metrics)]
88        crate::metrics::EXPORTER_NOTIFICATION_QUEUE_LENGTH.inc();
89    }
90}

92#[tokio::main(flavor = "current_thread")]
93async fn start_block_processor<S, F>(
94    storage: S,
95    shutdown_signal: F,
96    limits: LimitsConfig,
97    options: NodeOptions,
98    block_exporter_id: u32,
99    new_block_queue: NewBlockQueue,
100    destination_config: DestinationConfig,
101) -> Result<(), ExporterError>
102where
103    S: Storage + Clone + Send + Sync + 'static,
104    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
105    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
106{
107    let startup_destinations = destination_config
108        .destinations
109        .iter()
110        .map(|destination| destination.id())
111        .collect::<Vec<_>>();
112    let (block_processor_storage, mut exporter_storage) = BlockProcessorStorage::load(
113        storage.clone(),
114        block_exporter_id,
115        startup_destinations.clone(),
116        limits,
117    )
118    .await?;
119
120    let startup_committee_destinations = if destination_config.committee_destination {
121        // Load persisted committee destinations from storage if available
122        let persisted_committee_destinations =
123            load_persisted_committee_destinations(&storage, &block_processor_storage).await;
124
125        let latest_committee_destinations = persisted_committee_destinations.unwrap_or_default();
126        tracing::info!(
127            ?latest_committee_destinations,
128            "Init committee exporters from persisted state"
129        );
130        latest_committee_destinations
131            .into_iter()
132            .collect::<HashSet<_>>()
133    } else {
134        HashSet::new()
135    };
136
137    let tracker = ExportersTracker::new(
138        options,
139        limits.work_queue_size.into(),
140        shutdown_signal.clone(),
141        exporter_storage.clone()?,
142        destination_config.destinations.clone(),
143        startup_committee_destinations,
144    );
145
146    let mut block_processor = BlockProcessor::new(
147        tracker,
148        block_processor_storage,
149        new_block_queue,
150        destination_config.committee_destination,
151    );
152
153    block_processor
154        .run_with_shutdown(shutdown_signal, limits.persistence_period_ms)
155        .await?;
156
157    block_processor.pool_state().join_all().await;
158
159    Ok(())
160}

162/// Loads the persisted committee destinations from storage.
163async fn load_persisted_committee_destinations<S>(
164    storage: &S,
165    block_processor_storage: &BlockProcessorStorage<S>,
166) -> Option<Vec<DestinationId>>
167where
168    S: Storage + Clone + Send + Sync + 'static,
169{
170    // First, try to load from persisted committee blob ID
171    if let Some(committee_blob_id) = block_processor_storage.get_latest_committee_blob() {
172        tracing::info!(?committee_blob_id, "Found persisted committee blob ID");
173
174        return load_committee_from_blob(storage, committee_blob_id).await;
175    }
176
177    None
178}

180/// Loads the committee destinations from a specific blob ID.
181async fn load_committee_from_blob<S>(
182    storage: &S,
183    committee_blob_id: BlobId,
184) -> Option<Vec<DestinationId>>
185where
186    S: Storage + Clone + Send + Sync + 'static,
187{
188    let blob = match storage.read_blob(committee_blob_id).await {
189        Ok(Some(blob)) => blob,
190        Ok(None) => {
191            tracing::error!(?committee_blob_id, "Committee blob ID not found in storage");
192            return None;
193        }
194        Err(e) => {
195            tracing::error!(
196                ?committee_blob_id,
197                error = ?e,
198                "Failed to read committee blob"
199            );
200            return None;
201        }
202    };
203
204    let committee: Committee = match bcs::from_bytes(blob.bytes()) {
205        Ok(committee) => committee,
206        Err(e) => {
207            tracing::error!(
208                ?committee_blob_id,
209                error = ?e,
210                "Failed to deserialize committee blob"
211            );
212            return None;
213        }
214    };
215
216    let destinations: Vec<DestinationId> = committee
217        .validator_addresses()
218        .map(|(_, address)| DestinationId::validator(address.to_owned()))
219        .collect();
220
221    Some(destinations)
222}

485                destinations: vec![],

488
489        let mut single_validator = BTreeMap::new();

569            .await

571

575        sleep(Duration::from_secs(4)).await;
576        // The first validator should not receive the new confirmed certificate.
577        {
578            let pinned = dummy_validator.state.pin();
579            assert!(!pinned.contains(&third_notification.hash));
580        }
581        // We expect the first validator to receive the new confirmed certificate only once.
582        {
583            let pinned = dummy_validator.duplicate_blocks.pin();
584            assert!(pinned.get(&third_notification.hash).is_none());
585        }
586        // The second validator should receive the new confirmed certificate.
587        {
588            let pinned = second_dummy.state.pin();
589            assert!(pinned.contains(&third_notification.hash));
590        }
591        // We expect the second validator to receive the new confirmed certificate only once.
592        {
593            let pinned = second_dummy.duplicate_blocks.pin();
594            assert!(pinned.get(&third_notification.hash).is_none());
595        }
596
597        cancellation_token.cancel();
598        block_processor_handle.join().unwrap()?;
599        Ok(())
600    }

602    struct TestChain<S> {
603        chain_description: ChainDescription,
604        storage: S,
605    }

607    impl<S> TestChain<S> {
608        fn new(storage: S) -> Self {
609            let chain_description = ChainDescription::new(
610                ChainOrigin::Root(0),
611                InitialChainConfig {
612                    ownership: Default::default(),
613                    epoch: Default::default(),
614                    balance: Default::default(),
615                    application_permissions: Default::default(),
616                    min_active_epoch: Epoch::ZERO,
617                    max_active_epoch: Epoch::ZERO,
618                },
619                Timestamp::now(),
620            );
621            Self {
622                chain_description,
623                storage,
624            }
625        }
626
627        // Constructs a new block, with the blob containing the committee.
628        async fn publish_committee(
629            &self,
630            validators: &BTreeMap<Secp256k1PublicKey, ValidatorState>,
631            prev_block: Option<ConfirmedBlock>,
632        ) -> Result<ConfirmedBlockCertificate, ViewError>
633        where
634            S: Storage + Clone + Send + Sync + 'static,
635        {
636            let committee = Committee::new(validators.clone(), ResourceControlPolicy::testnet())?;
637            let chain_id = self.chain_description.id();
638            let chain_blob = Blob::new_chain_description(&self.chain_description);
639
640            let committee_blob = Blob::new(BlobContent::new_committee(bcs::to_bytes(&committee)?));
641            let proposed_block = if let Some(parent_block) = prev_block {
642                make_child_block(&parent_block).with_operation(Operation::System(Box::new(
643                    SystemOperation::Admin(AdminOperation::CreateCommittee {
644                        epoch: parent_block.epoch().try_add_one().unwrap(),
645                        blob_hash: committee_blob.id().hash,
646                    }),
647                )))
648            } else {
649                make_first_block(chain_id).with_operation(Operation::System(Box::new(
650                    SystemOperation::Admin(AdminOperation::CreateCommittee {
651                        epoch: Epoch::ZERO,
652                        blob_hash: committee_blob.id().hash,
653                    }),
654                )))
655            };
656            let blobs = vec![chain_blob, committee_blob];
657            let block = BlockExecutionOutcome {
658                blobs: vec![blobs.clone()],
659                ..Default::default()
660            }
661            .with(proposed_block);
662
663            let confirmed_block = ConfirmedBlock::new(block);
664            let certificate = ConfirmedBlockCertificate::new(confirmed_block, Round::Fast, vec![]);
665            self.storage
666                .write_blobs_and_certificate(blobs.as_ref(), &certificate)
667                .await?;
668
669            Ok(certificate)
670        }
671    }
