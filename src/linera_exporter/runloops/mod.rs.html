35pub(crate) fn start_block_processor_task<S, F>(
36    storage: S,
37    shutdown_signal: F,
38    limits: LimitsConfig,
39    options: NodeOptions,
40    block_exporter_id: u32,
41    clients_per_thread: usize,
42    destination_config: DestinationConfig,
43) -> Result<UnboundedSender<BlockId>, ExporterError>
44where
45    S: Storage + Clone + Send + Sync + 'static,
46    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
47    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
48{
49    let (task_sender, queue_front) = unbounded_channel();
50    let moved_task_sender = task_sender.clone();
51    let _handle = thread::spawn(move || {
52        start_block_processor(
53            storage,
54            shutdown_signal,
55            limits,
56            options,
57            block_exporter_id,
58            clients_per_thread,
59            moved_task_sender,
60            destination_config,
61            queue_front,
62        )
63    });
64
65    Ok(task_sender)
66}

68fn start_exporters<S, F>(
69    shutdown_signal: F,
70    options: NodeOptions,
71    work_queue_size: usize,
72    clients_per_thread: usize,
73    mut storage: ExporterStorage<S>,
74    destination_config: DestinationConfig,
75) -> Result<Vec<JoinHandle<()>>, ExporterError>
76where
77    S: Storage + Clone + Send + Sync + 'static,
78    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
79    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
80{
81    let number_of_threads = destination_config
82        .destinations
83        .len()
84        .div_ceil(clients_per_thread);
85    let threadpool = {
86        let mut pool = Vec::new();
87        for n in 0..number_of_threads {
88            let moved_signal = shutdown_signal.clone();
89            let moved_storage = storage.clone()?;
90            let destinations = destination_config
91                .destinations
92                .iter()
93                .enumerate()
94                .filter_map(|(i, dest)| {
95                    if i % number_of_threads == n {
96                        Some((i as u16, dest.clone()))
97                    } else {
98                        None
99                    }
100                })
101                .collect();
102
103            let handle = thread::spawn(move || {
104                start_exporter_tasks(
105                    moved_signal,
106                    options,
107                    work_queue_size,
108                    moved_storage,
109                    destinations,
110                )
111            });
112
113            pool.push(handle);
114        }
115
116        pool
117    };
118
119    Ok(threadpool)
120}

122#[tokio::main(flavor = "current_thread")]
123async fn start_exporter_tasks<S, F>(
124    shutdown_signal: F,
125    options: NodeOptions,
126    work_queue_size: usize,
127    mut storage: ExporterStorage<S>,
128    destinations: Vec<(DestinationId, Destination)>,
129) where
130    S: Storage + Clone + Send + Sync + 'static,
131    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
132    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
133{
134    let node_provider = GrpcNodeProvider::new(options);
135    let arced_node_provider = Arc::new(node_provider);
136    let mut set = JoinSet::new();
137    for (id, destination) in destinations {
138        spawn_exporter_task_on_set(
139            &mut set,
140            shutdown_signal.clone(),
141            id,
142            options,
143            work_queue_size,
144            destination,
145            storage.clone().unwrap(),
146            arced_node_provider.clone(),
147        );
148    }
149
150    while let Some(res) = set.join_next().await {
151        let _res = res.unwrap();
152    }
153}

155#[allow(clippy::too_many_arguments)]
156#[tokio::main(flavor = "current_thread")]
157async fn start_block_processor<S, F>(
158    storage: S,
159    shutdown_signal: F,
160    limits: LimitsConfig,
161    options: NodeOptions,
162    block_exporter_id: u32,
163    clients_per_thread: usize,
164    queue_rear: UnboundedSender<BlockId>,
165    destination_config: DestinationConfig,
166    queue_front: UnboundedReceiver<BlockId>,
167) -> Result<(), ExporterError>
168where
169    S: Storage + Clone + Send + Sync + 'static,
170    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
171    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
172{
173    let (block_processor_storage, exporter_storage) = BlockProcessorStorage::load(
174        storage.clone(),
175        block_exporter_id,
176        destination_config.destinations.len() as u16,
177        limits,
178    )
179    .await?;
180
181    let mut block_processor = BlockProcessor::new(block_processor_storage, queue_rear, queue_front);
182
183    let _pool = start_exporters(
184        shutdown_signal.clone(),
185        options,
186        limits.work_queue_size.into(),
187        clients_per_thread,
188        exporter_storage,
189        destination_config,
190    )
191    .unwrap();
192
193    block_processor
194        .run_with_shutdown(shutdown_signal, limits.persistence_period_ms)
195        .await
196        .unwrap();
197
198    Ok(())
199}

201#[allow(clippy::too_many_arguments)]
202fn spawn_exporter_task_on_set<F, S>(
203    set: &mut JoinSet<Result<(), anyhow::Error>>,
204    signal: F,
205    id: DestinationId,
206    options: NodeOptions,
207    work_queue_size: usize,
208    destination: Destination,
209    storage: ExporterStorage<S>,
210    node_provider: Arc<GrpcNodeProvider>,
211) where
212    S: Storage + Clone + Send + Sync + 'static,
213    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
214    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
215{
216    match destination.kind {
217        DestinationKind::Indexer => {
218            let exporter_task =
219                IndexerExporter::new(options, work_queue_size, storage, id, destination);
220
221            let _handle = set.spawn(exporter_task.run_with_shutdown(signal));
222        }
223
224        DestinationKind::Validator => {
225            let exporter_task =
226                ValidatorExporter::new(node_provider, id, storage, destination, work_queue_size);
227
228            let _handle = set.spawn(exporter_task.run_with_shutdown(signal));
229        }
230    }
231}

260        T: TestDestination + Clone + Send + 'static,
261    {
262        let port = get_free_port().await?;
263        let cancellation_token = CancellationToken::new();
264        tokio::spawn(destination.clone().start(port, cancellation_token.clone()));
265        LocalNet::ensure_grpc_server_has_started("test server", port as usize, "http").await?;
266
267        let signal = ExporterCancellationSignal::new(cancellation_token.clone());
268        let storage = DbStorage::<MemoryStore, _>::make_test_storage(None).await;
269        let destination_address = Destination {
270            port,
271            tls: TlsConfig::ClearText,
272            endpoint: "127.0.0.1".to_owned(),
273            kind: destination.kind(),
274        };
275
276        // make some blocks
277        let (notification, state) = make_simple_state_with_blobs(&storage).await;
278
279        let notifier = start_block_processor_task(
280            storage,
281            signal,
282            LimitsConfig::default(),
283            NodeOptions {
284                send_timeout: Duration::from_millis(4000),
285                recv_timeout: Duration::from_millis(4000),
286                retry_delay: Duration::from_millis(1000),
287                max_retries: 10,
288            },
289            0,
290            10,
291            DestinationConfig {
292                destinations: vec![destination_address],
293            },
294        )?;
295
296        assert!(
297            notifier.send(notification).is_ok(),
298            "notifier should work as long as there exists a receiver to receive notifications"
299        );
300
301        sleep(Duration::from_secs(4)).await;
302
303        for CanonicalBlock { blobs, block_hash } in state {
304            assert!(destination.state().contains(&block_hash));
305            for blob in blobs {
306                assert!(destination.blobs().contains(&blob));
307            }
308        }
309
310        cancellation_token.cancel();
311
312        Ok(())
313    }

316    async fn test_restart_persistence_and_faulty_destination_restart() -> Result<(), anyhow::Error>
317    {
318        let mut destinations = Vec::new();
319        let cancellation_token = CancellationToken::new();
320        let _dummy_indexer = spawn_dummy_indexer(&mut destinations, &cancellation_token).await?;
321        let faulty_indexer = spawn_faulty_indexer(&mut destinations, &cancellation_token).await?;
322        let _dummy_validator =
323            spawn_dummy_validator(&mut destinations, &cancellation_token).await?;
324        let faulty_validator =
325            spawn_faulty_validator(&mut destinations, &cancellation_token).await?;
326
327        let child = cancellation_token.child_token();
328        let signal = ExporterCancellationSignal::new(child.clone());
329        let storage = DbStorage::<MemoryStore, _>::make_test_storage(None).await;
330
331        let (notification, _state) = make_simple_state_with_blobs(&storage).await;
332
333        let notifier = start_block_processor_task(
334            storage.clone(),
335            signal,
336            LimitsConfig {
337                persistence_period_ms: 3000,
338                ..Default::default()
339            },
340            NodeOptions {
341                send_timeout: Duration::from_millis(4000),
342                recv_timeout: Duration::from_millis(4000),
343                retry_delay: Duration::from_millis(1000),
344                max_retries: 10,
345            },
346            0,
347            1,
348            DestinationConfig {
349                destinations: destinations.clone(),
350            },
351        )?;
352
353        assert!(
354            notifier.send(notification).is_ok(),
355            "notifier should work as long as there exists a receiver to receive notifications"
356        );
357
358        sleep(Duration::from_secs(4)).await;
359
360        child.cancel();
361
362        let context = storage.block_exporter_context(0).await?;
363        let (_, _, destination_states) =
364            BlockExporterStateView::initiate(context.clone(), 4).await?;
365        for i in 0..4 {
366            let state = destination_states.load_state(i);
367            if i % 2 == 0 {
368                assert_eq!(state, 2);
369            } else {
370                assert_eq!(state, 0);
371            }
372        }
373
374        faulty_indexer.unset_faulty();
375        faulty_validator.unset_faulty();
376
377        let child = cancellation_token.child_token();
378        let signal = ExporterCancellationSignal::new(child.clone());
379
380        // restart
381        let _notifier = start_block_processor_task(
382            storage.clone(),
383            signal,
384            LimitsConfig {
385                persistence_period_ms: 3000,
386                ..Default::default()
387            },
388            NodeOptions {
389                send_timeout: Duration::from_millis(4000),
390                recv_timeout: Duration::from_millis(4000),
391                retry_delay: Duration::from_millis(1000),
392                max_retries: 10,
393            },
394            0,
395            1,
396            DestinationConfig { destinations },
397        )?;
398
399        sleep(Duration::from_secs(4)).await;
400
401        child.cancel();
402
403        let (_, _, destination_states) =
404            BlockExporterStateView::initiate(context.clone(), 4).await?;
405        for i in 0..4 {
406            assert_eq!(destination_states.load_state(i), 2);
407        }
408
409        Ok(())
410    }
