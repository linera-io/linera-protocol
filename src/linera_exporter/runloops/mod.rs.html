29pub(crate) fn start_block_processor_task<S, F>(
30    storage: S,
31    shutdown_signal: F,
32    limits: LimitsConfig,
33    options: NodeOptions,
34    block_exporter_id: u32,
35    destination_config: DestinationConfig,
36) -> (
37    UnboundedSender<BlockId>,
38    std::thread::JoinHandle<Result<(), ExporterError>>,
39)
40where
41    S: Storage + Clone + Send + Sync + 'static,
42    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
43    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
44{
45    let (task_sender, queue_front) = unbounded_channel();
46    let new_block_queue = NewBlockQueue {
47        queue_rear: task_sender.clone(),
48        queue_front,
49    };
50    let handle = std::thread::spawn(move || {
51        start_block_processor(
52            storage,
53            shutdown_signal,
54            limits,
55            options,
56            block_exporter_id,
57            new_block_queue,
58            destination_config,
59        )
60    });
61
62    (task_sender, handle)
63}
64
65struct NewBlockQueue {
66    pub(crate) queue_rear: UnboundedSender<BlockId>,
67    pub(crate) queue_front: UnboundedReceiver<BlockId>,
68}
69
70impl NewBlockQueue {
71    async fn recv(&mut self) -> Option<BlockId> {
72        self.queue_front.recv().await
73    }
74
75    fn push_back(&self, block_id: BlockId) {
76        self.queue_rear
77            .send(block_id)
78            .expect("sender should never fail");
79    }
80}

82#[tokio::main(flavor = "current_thread")]
83async fn start_block_processor<S, F>(
84    storage: S,
85    shutdown_signal: F,
86    limits: LimitsConfig,
87    options: NodeOptions,
88    block_exporter_id: u32,
89    new_block_queue: NewBlockQueue,
90    destination_config: DestinationConfig,
91) -> Result<(), ExporterError>
92where
93    S: Storage + Clone + Send + Sync + 'static,
94    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
95    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
96{
97    let destination_ids = destination_config
98        .destinations
99        .iter()
100        .map(|destination| destination.id())
101        .collect::<Vec<_>>();
102    let (block_processor_storage, mut exporter_storage) =
103        BlockProcessorStorage::load(storage.clone(), block_exporter_id, destination_ids, limits)
104            .await?;
105
106    let tracker = ExportersTracker::new(
107        options,
108        limits.work_queue_size.into(),
109        shutdown_signal.clone(),
110        exporter_storage.clone()?,
111        destination_config.destinations.clone(),
112    );
113
114    let mut block_processor = BlockProcessor::new(
115        tracker,
116        block_processor_storage,
117        new_block_queue,
118        destination_config.committee_destination,
119    );
120
121    block_processor
122        .run_with_shutdown(shutdown_signal, limits.persistence_period_ms)
123        .await?;
124
125    block_processor.pool_state().join_all().await;
126
127    Ok(())
128}

354    async fn test_committee_destination() -> anyhow::Result<()> {
355        tracing::info!("Starting test_committee_destination test");
356
357        let storage = DbStorage::<MemoryDatabase, _>::make_test_storage(None).await;
358        let test_chain = TestChain::new(storage.clone());
359        let cancellation_token = CancellationToken::new();
360        let child = cancellation_token.child_token();
361        let signal = ExporterCancellationSignal::new(child.clone());
362
363        let mut destinations = vec![];
364        let dummy_validator = spawn_dummy_validator(&mut destinations, &cancellation_token).await?;
365        let destination = destinations[0].clone();
366        let validator_state = ValidatorState {
367            network_address: destination.address(),
368            votes: 0,
369            account_public_key: AccountPublicKey::test_key(0),
370        };
371        let (notifier, block_processor_handle) = start_block_processor_task(
372            storage.clone(),
373            signal,
374            LimitsConfig {
375                persistence_period_ms: 3000,
376                ..Default::default()
377            },
378            NodeOptions {
379                send_timeout: Duration::from_millis(4000),
380                recv_timeout: Duration::from_millis(4000),
381                retry_delay: Duration::from_millis(1000),
382                max_retries: 10,
383            },
384            0,
385            DestinationConfig {
386                committee_destination: true,
387                destinations: vec![],
388            },
389        );
390
391        let mut single_validator = BTreeMap::new();
392        single_validator.insert(Secp256k1PublicKey::test_key(0), validator_state);
393
394        let confirmed_certificate = test_chain
395            .publish_committee(&single_validator, None)
396            .await
397            .expect("Failed to publish committee");
398
399        let first_notification = BlockId::from_confirmed_block(confirmed_certificate.value());
400        notifier.send(first_notification)?;
401        sleep(Duration::from_secs(4)).await;
402
403        {
404            let pinned = dummy_validator.state.pin();
405            assert!(pinned.contains(&first_notification.hash));
406        }
407        // We expect the validator to receive the confired certificate only once.
408        {
409            let pinned = dummy_validator.duplicate_blocks.pin();
410            assert!(pinned.get(&first_notification.hash).is_none());
411        }
412
413        ///////////
414        // Add new validator to the committee.
415        ///////////
416        let second_dummy = spawn_dummy_validator(&mut destinations, &cancellation_token).await?;
417        let destination = destinations[1].clone();
418        let validator_state = ValidatorState {
419            network_address: destination.address(),
420            votes: 0,
421            account_public_key: AccountPublicKey::test_key(1),
422        };
423        let mut two_validators = single_validator.clone();
424        two_validators.insert(Secp256k1PublicKey::test_key(1), validator_state);
425        let add_validator_certificate = test_chain
426            .publish_committee(&two_validators, Some(confirmed_certificate.value().clone()))
427            .await
428            .expect("Failed to publish new committee");
429        let second_notification = BlockId::from_confirmed_block(add_validator_certificate.value());
430        notifier.send(second_notification)?;
431        sleep(Duration::from_secs(4)).await;
432
433        {
434            let pinned = second_dummy.state.pin();
435            assert!(pinned.contains(&second_notification.hash));
436        }
437        // We expect the new validator to receive the new confirmed certificate only once.
438        {
439            let pinned = second_dummy.duplicate_blocks.pin();
440            assert!(pinned.get(&second_notification.hash).is_none());
441        }
442        // The first certificate should not be duplicated.
443        {
444            let pinned = second_dummy.duplicate_blocks.pin();
445            assert!(pinned.get(&first_notification.hash).is_none());
446        }
447
448        // The first validator should receive the new committee as well.
449        {
450            let pinned = dummy_validator.state.pin();
451            assert!(pinned.contains(&second_notification.hash));
452        }
453        // We expect the first validator to receive the new confirmed certificate only once.
454        {
455            let pinned = dummy_validator.duplicate_blocks.pin();
456            assert!(pinned.get(&second_notification.hash).is_none());
457        }
458
459        ///////////
460        // Remove the validator from the committee.
461        ///////////
462
463        let mut new_validators = two_validators.clone();
464        new_validators.remove(&Secp256k1PublicKey::test_key(0));
465
466        let remove_validator_certificate = test_chain
467            .publish_committee(
468                &new_validators,
469                Some(add_validator_certificate.value().clone()),
470            )
471            .await
472            .expect("Failed to publish new committee");
473
474        let third_notification =
475            BlockId::from_confirmed_block(remove_validator_certificate.value());
476        notifier.send(third_notification)?;
477        sleep(Duration::from_secs(4)).await;
478        // The first validator should not receive the new confirmed certificate.
479        {
480            let pinned = dummy_validator.state.pin();
481            assert!(!pinned.contains(&third_notification.hash));
482        }
483        // We expect the first validator to receive the new confirmed certificate only once.
484        {
485            let pinned = dummy_validator.duplicate_blocks.pin();
486            assert!(pinned.get(&third_notification.hash).is_none());
487        }
488        // The second validator should receive the new confirmed certificate.
489        {
490            let pinned = second_dummy.state.pin();
491            assert!(pinned.contains(&third_notification.hash));
492        }
493        // We expect the second validator to receive the new confirmed certificate only once.
494        {
495            let pinned = second_dummy.duplicate_blocks.pin();
496            assert!(pinned.get(&third_notification.hash).is_none());
497        }
498
499        cancellation_token.cancel();
500        block_processor_handle.join().unwrap()?;
501        Ok(())
502    }

504    struct TestChain<S> {
505        chain_description: ChainDescription,
506        storage: S,
507    }

509    impl<S> TestChain<S> {
510        fn new(storage: S) -> Self {
511            let chain_description = ChainDescription::new(
512                ChainOrigin::Root(0),
513                InitialChainConfig {
514                    ownership: Default::default(),
515                    epoch: Default::default(),
516                    balance: Default::default(),
517                    application_permissions: Default::default(),
518                    min_active_epoch: Epoch::ZERO,
519                    max_active_epoch: Epoch::ZERO,
520                },
521                Timestamp::now(),
522            );
523            Self {
524                chain_description,
525                storage,
526            }
527        }
528
529        // Constructs a new block, with the blob containing the committee.
530        async fn publish_committee(
531            &self,
532            validators: &BTreeMap<Secp256k1PublicKey, ValidatorState>,
533            prev_block: Option<ConfirmedBlock>,
534        ) -> Result<ConfirmedBlockCertificate, ViewError>
535        where
536            S: Storage + Clone + Send + Sync + 'static,
537        {
538            let committee = Committee::new(validators.clone(), ResourceControlPolicy::testnet());
539            let chain_id = self.chain_description.id();
540            let chain_blob = Blob::new_chain_description(&self.chain_description);
541
542            let committee_blob = Blob::new(BlobContent::new_committee(bcs::to_bytes(&committee)?));
543            let proposed_block = if let Some(parent_block) = prev_block {
544                make_child_block(&parent_block).with_operation(Operation::System(Box::new(
545                    SystemOperation::Admin(AdminOperation::CreateCommittee {
546                        epoch: parent_block.epoch().try_add_one().unwrap(),
547                        blob_hash: committee_blob.id().hash,
548                    }),
549                )))
550            } else {
551                make_first_block(chain_id).with_operation(Operation::System(Box::new(
552                    SystemOperation::Admin(AdminOperation::CreateCommittee {
553                        epoch: Epoch::ZERO,
554                        blob_hash: committee_blob.id().hash,
555                    }),
556                )))
557            };
558            let blobs = vec![chain_blob, committee_blob];
559            let block = BlockExecutionOutcome {
560                blobs: vec![blobs.clone()],
561                ..Default::default()
562            }
563            .with(proposed_block);
564
565            let confirmed_block = ConfirmedBlock::new(block);
566            let certificate = ConfirmedBlockCertificate::new(confirmed_block, Round::Fast, vec![]);
567            self.storage
568                .write_blobs_and_certificate(blobs.as_ref(), &certificate)
569                .await?;
570
571            Ok(certificate)
572        }
573    }
