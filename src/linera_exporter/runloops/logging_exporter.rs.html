4use std::{future::IntoFuture, io::Write, path::Path};

6use linera_chain::types::CertificateValue;
7use tokio::select;

11/// A logging exporter that writes logs to a file.
12///
13/// This exporter does not track any state or process data; it simply logs messages to a specified file.
14/// It will export events as they occur, never exporting past ones,
15/// which can be useful for debugging and monitoring purposes.
16pub(crate) struct LoggingExporter {
17    file: std::fs::File,
18}
19
20impl LoggingExporter {
21    /// Creates a new `LoggingExporter` that logs to the specified file.
22    pub fn new(log_file: &Path) -> Self {
23        let file = std::fs::File::create(log_file).expect("Failed to create log file");
24        LoggingExporter { file }
25    }
26
27    pub(crate) async fn run_with_shutdown<S, F: IntoFuture<Output = ()>>(
28        self,
29        shutdown_signal: F,
30        storage: ExporterStorage<S>,
31    ) -> anyhow::Result<()>
32    where
33        S: linera_storage::Storage + Clone + Send + Sync + 'static,
34    {
35        let shutdown_signal_future = shutdown_signal.into_future();
36        let mut pinned_shutdown_signal = Box::pin(shutdown_signal_future);
37
38        select! {
39            _ = &mut pinned_shutdown_signal => {
40                tracing::info!("logging exporter shutdown signal received, exiting.");
41            }
42
43            _ = self.start_logger(storage) => {
44
45            }
46        }
47        Ok(())
48    }
49
50    async fn start_logger<S>(mut self, storage: ExporterStorage<S>) -> anyhow::Result<()>
51    where
52        S: linera_storage::Storage + Clone + Send + Sync + 'static,
53    {
54        let mut canonical_chain_height = storage.get_latest_index();
55        tracing::info!(
56            "starting logging exporter at height {}",
57            canonical_chain_height
58        );
59
60        loop {
61            if let Ok((block, blobs)) = storage.get_block_with_blobs(canonical_chain_height).await {
62                let inner = block.inner();
63                writeln!(
64                    self.file,
65                    "Block ID: {}, Chain: {}, Height: {}, State Hash: {}, Authenticated Owner: {}",
66                    inner.hash(),
67                    inner.chain_id(),
68                    inner.height(),
69                    inner.block().header.state_hash,
70                    inner
71                        .block()
72                        .header
73                        .authenticated_owner
74                        .map_or_else(|| "N/A".into(), |signer| signer.to_string()),
75                )?;
76                for blob in blobs {
77                    writeln!(self.file, "\tBlob ID: {}", blob.id(),)?;
78                }
79
80                canonical_chain_height += 1;
81            } else {
82                tokio::time::sleep(tokio::time::Duration::from_secs(1)).await;
83            }
84        }
85    }
86}
