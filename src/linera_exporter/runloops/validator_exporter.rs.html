30pub(crate) struct Exporter {
31    node_provider: Arc<GrpcNodeProvider>,
32    destination_id: DestinationId,
33    work_queue_size: usize,
34}
35
36impl Exporter {
37    pub(super) fn new(
38        destination_id: DestinationId,
39        node_provider: Arc<GrpcNodeProvider>,
40        work_queue_size: usize,
41    ) -> Self {
42        Self {
43            node_provider,
44            destination_id,
45            work_queue_size,
46        }
47    }
48
49    pub(super) async fn run_with_shutdown<S, F: IntoFuture<Output = ()>>(
50        self,
51        shutdown_signal: F,
52        mut storage: ExporterStorage<S>,
53    ) -> anyhow::Result<()>
54    where
55        S: Storage + Clone + Send + Sync + 'static,
56    {
57        let address = self.destination_id.address().to_owned();
58        let destination_state = storage.load_destination_state(&self.destination_id);
59
60        let node = self.node_provider.make_node(&address)?;
61
62        let (mut task_queue, task_receiver) = TaskQueue::new(
63            self.work_queue_size,
64            destination_state.load(Ordering::Acquire) as usize,
65            storage.clone()?,
66        );
67
68        let export_task = ExportTask::new(node, storage.clone()?, destination_state);
69
70        tokio::select! {
71
72            biased;
73
74            _ = shutdown_signal => {},
75
76            res = task_queue.run() => res?,
77
78            res = export_task.run(task_receiver) => res?,
79
80        };
81
82        Ok(())
83    }
84}

86struct ExportTask<S>
87where
88    S: Storage + Clone + Send + Sync + 'static,
89{
90    node: GrpcClient,
91    storage: ExporterStorage<S>,
92    destination_state: Arc<AtomicU64>,
93}
94
95impl<S> ExportTask<S>
96where
97    S: Storage + Clone + Send + Sync + 'static,
98{
99    fn new(
100        node: GrpcClient,
101        storage: ExporterStorage<S>,
102        destination_state: Arc<AtomicU64>,
103    ) -> ExportTask<S> {
104        ExportTask {
105            node,
106            storage,
107            destination_state,
108        }
109    }
110
111    async fn run(
112        &self,
113        mut receiver: Receiver<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>)>,
114    ) -> anyhow::Result<()> {
115        while let Some((block, blobs_ids)) = receiver.recv().await {
116            #[cfg(with_metrics)]
117            crate::metrics::VALIDATOR_EXPORTER_QUEUE_LENGTH
118                .with_label_values(&[self.node.address()])
119                .set(receiver.len() as i64);
120            match self.dispatch_block((*block).clone()).await {
121                Ok(_) => {}
122
123                Err(NodeError::BlobsNotFound(blobs_to_maybe_send)) => {
124                    let blobs = blobs_ids
125                        .into_iter()
126                        .filter(|id| blobs_to_maybe_send.contains(id))
127                        .collect();
128                    self.upload_blobs(blobs).await?;
129                    self.dispatch_block((*block).clone()).await?
130                }
131
132                Err(e) => Err(e)?,
133            }
134
135            self.increment_destination_state();
136        }
137
138        Ok(())
139    }
140
141    fn increment_destination_state(&self) {
142        let _ = self.destination_state.fetch_add(1, Ordering::Release);
143        #[cfg(with_metrics)]
144        crate::metrics::DESTINATION_STATE_COUNTER
145            .with_label_values(&[self.node.address()])
146            .inc();
147    }
148
149    async fn upload_blobs(&self, blobs: Vec<BlobId>) -> anyhow::Result<()> {
150        let tasks = blobs.iter().map(|id| async {
151            match self.storage.get_blob(*id).await {
152                Err(e) => Err(e),
153                Ok(blob) => {
154                    tracing::info!(
155                        "dispatching blob with id: {:#?} from linera exporter",
156                        blob.id()
157                    );
158                    #[cfg(with_metrics)]
159                    let start = linera_base::time::Instant::now();
160                    let result = self
161                        .node
162                        .upload_blob((*blob).clone().into())
163                        .await
164                        .map(|_| ())
165                        .map_err(|e| ExporterError::GenericError(e.into()));
166                    #[cfg(with_metrics)]
167                    crate::metrics::DISPATCH_BLOB_HISTOGRAM
168                        .with_label_values(&[self.node.address()])
169                        .observe(start.elapsed().as_secs_f64() * 1000.0);
170                    result
171                }
172            }
173        });
174
175        let _ = try_join_all(tasks).await?;
176
177        Ok(())
178    }
179
180    async fn dispatch_block(
181        &self,
182        certificate: ConfirmedBlockCertificate,
183    ) -> Result<(), NodeError> {
184        let delivery = CrossChainMessageDelivery::NonBlocking;
185        let block_id = BlockId::from_confirmed_block(certificate.value());
186        tracing::info!(?block_id, "dispatching block");
187        #[cfg(with_metrics)]
188        let start = linera_base::time::Instant::now();
189        match Box::pin(
190            self.node
191                .handle_confirmed_certificate(certificate, delivery),
192        )
193        .await
194        {
195            Ok(_) => {
196                #[cfg(with_metrics)]
197                crate::metrics::DISPATCH_BLOCK_HISTOGRAM
198                    .with_label_values(&[self.node.address()])
199                    .observe(start.elapsed().as_secs_f64() * 1000.0);
200            }
201            Err(e) => {
202                tracing::error!(error=%e, ?block_id, "error when dispatching block");
203                #[cfg(with_metrics)]
204                crate::metrics::DISPATCH_BLOCK_HISTOGRAM
205                    .with_label_values(&[self.node.address()])
206                    .observe(start.elapsed().as_secs_f64() * 1000.0);
207                Err(e)?
208            }
209        }
210
211        Ok(())
212    }
213}

215struct TaskQueue<S>
216where
217    S: Storage + Clone + Send + Sync + 'static,
218{
219    queue_size: usize,
220    start_height: usize,
221    storage: ExporterStorage<S>,
222    buffer: Sender<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>)>,
223}
224
225impl<S> TaskQueue<S>
226where
227    S: Storage + Clone + Send + Sync + 'static,
228{
229    #[expect(clippy::type_complexity)]
230    fn new(
231        queue_size: usize,
232        start_height: usize,
233        storage: ExporterStorage<S>,
234    ) -> (
235        TaskQueue<S>,
236        Receiver<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>)>,
237    ) {
238        let (sender, receiver) = tokio::sync::mpsc::channel(queue_size);
239
240        let queue = Self {
241            queue_size,
242            start_height,
243            storage,
244            buffer: sender,
245        };
246
247        (queue, receiver)
248    }
249
250    async fn run(&mut self) -> anyhow::Result<()> {
251        let mut index = self.start_height;
252        let mut futures = FuturesOrdered::new();
253        while futures.len() < self.queue_size {
254            futures.push_back(self.get_block_task(index));
255            index += 1;
256        }
257
258        while let Some(certificate) = futures.next().await.transpose()? {
259            let _ = self.buffer.send(certificate).await;
260            futures.push_back(self.get_block_task(index));
261            index += 1;
262        }
263
264        Ok(())
265    }
266
267    async fn get_block_task(
268        &self,
269        index: usize,
270    ) -> Result<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>), ExporterError> {
271        loop {
272            match self.storage.get_block_with_blob_ids(index).await {
273                Ok(block_with_blobs_ids) => return Ok(block_with_blobs_ids),
274                Err(ExporterError::UnprocessedBlock) => {
275                    tokio::time::sleep(Duration::from_secs(1)).await
276                }
277                Err(e) => return Err(e),
278            }
279        }
280    }
