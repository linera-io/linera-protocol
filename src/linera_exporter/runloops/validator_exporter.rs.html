30pub(crate) struct Exporter {
31    node_provider: Arc<GrpcNodeProvider>,
32    destination_id: DestinationId,
33    work_queue_size: usize,
34}
35
36impl Exporter {
37    pub(super) fn new(
38        destination_id: DestinationId,
39        node_provider: Arc<GrpcNodeProvider>,
40        work_queue_size: usize,
41    ) -> Self {
42        Self {
43            node_provider,
44            destination_id,
45            work_queue_size,
46        }
47    }
48
49    pub(super) async fn run_with_shutdown<S, F: IntoFuture<Output = ()>>(
50        self,
51        shutdown_signal: F,
52        mut storage: ExporterStorage<S>,
53    ) -> anyhow::Result<()>
54    where
55        S: Storage + Clone + Send + Sync + 'static,
56    {
57        let address = self.destination_id.address().to_owned();
58        let destination_state = storage.load_destination_state(&self.destination_id);
59
60        let node = self.node_provider.make_node(&address)?;
61
62        let (mut task_queue, task_receiver) = TaskQueue::new(
63            self.work_queue_size,
64            destination_state.load(Ordering::Acquire) as usize,
65            storage.clone()?,
66        );
67
68        let export_task = ExportTask::new(node, storage.clone()?, destination_state);
69
70        tokio::select! {
71
72            biased;
73
74            _ = shutdown_signal => {},
75
76            res = task_queue.run() => res?,
77
78            res = export_task.run(task_receiver) => res?,
79
80        };
81
82        Ok(())
83    }
84}

86struct ExportTask<S>
87where
88    S: Storage + Clone + Send + Sync + 'static,
89{
90    node: GrpcClient,
91    storage: ExporterStorage<S>,
92    destination_state: Arc<AtomicU64>,
93}
94
95impl<S> ExportTask<S>
96where
97    S: Storage + Clone + Send + Sync + 'static,
98{
99    fn new(
100        node: GrpcClient,
101        storage: ExporterStorage<S>,
102        destination_state: Arc<AtomicU64>,
103    ) -> ExportTask<S> {
104        ExportTask {
105            node,
106            storage,
107            destination_state,
108        }
109    }
110
111    async fn run(
112        &self,
113        mut receiver: Receiver<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>)>,
114    ) -> anyhow::Result<()> {
115        while let Some((block, blobs_ids)) = receiver.recv().await {
116            #[cfg(with_metrics)]
117            crate::metrics::VALIDATOR_EXPORTER_QUEUE_LENGTH
118                .with_label_values(&[self.node.address()])
119                .set(receiver.len() as i64);
120            match self.dispatch_block((*block).clone()).await {
121                Ok(_) => {}
122
123                Err(NodeError::BlobsNotFound(blobs_to_maybe_send)) => {
124                    let blobs = blobs_ids
125                        .into_iter()
126                        .filter(|id| blobs_to_maybe_send.contains(id))
127                        .collect();
128                    self.upload_blobs(blobs).await?;
129                    self.dispatch_block((*block).clone()).await?
130                }
131
132                Err(e) => Err(e)?,
133            }
134
135            self.increment_destination_state();
136        }
137
138        Ok(())
139    }
140
141    fn increment_destination_state(&self) {
142        self.destination_state.fetch_add(1, Ordering::Release);
143        #[cfg(with_metrics)]
144        crate::metrics::DESTINATION_STATE_COUNTER
145            .with_label_values(&[self.node.address()])
146            .inc();
147    }
148
149    async fn upload_blobs(&self, blobs: Vec<BlobId>) -> anyhow::Result<()> {
150        let tasks = blobs.iter().map(|id| async {
151            match self.storage.get_blob(*id).await {
152                Err(e) => Err(e),
153                Ok(blob) => {
154                    tracing::info!(
155                        destination=?self.node.address(),
156                        blob_id=?blob.id(),
157                        "dispatching blob",
158                    );
159                    #[cfg(with_metrics)]
160                    let start = linera_base::time::Instant::now();
161                    let result = self
162                        .node
163                        .upload_blob((*blob).clone().into())
164                        .await
165                        .map(|_| ())
166                        .map_err(|e| ExporterError::GenericError(e.into()));
167                    #[cfg(with_metrics)]
168                    crate::metrics::DISPATCH_BLOB_HISTOGRAM
169                        .with_label_values(&[self.node.address()])
170                        .observe(start.elapsed().as_secs_f64() * 1000.0);
171                    result
172                }
173            }
174        });
175
176        try_join_all(tasks).await?;
177
178        Ok(())
179    }
180
181    async fn dispatch_block(
182        &self,
183        certificate: ConfirmedBlockCertificate,
184    ) -> Result<(), NodeError> {
185        let delivery = CrossChainMessageDelivery::NonBlocking;
186        let block_id = BlockId::from_confirmed_block(certificate.value());
187        tracing::info!(destination=?self.node.address(), ?block_id, "dispatching block");
188        #[cfg(with_metrics)]
189        let start = linera_base::time::Instant::now();
190        match Box::pin(
191            self.node
192                .handle_confirmed_certificate(certificate, delivery),
193        )
194        .await
195        {
196            Ok(_) => {
197                #[cfg(with_metrics)]
198                crate::metrics::DISPATCH_BLOCK_HISTOGRAM
199                    .with_label_values(&[self.node.address()])
200                    .observe(start.elapsed().as_secs_f64() * 1000.0);
201            }
202            Err(e) => {
203                tracing::error!(error=%e, ?block_id, "error when dispatching block");
204                #[cfg(with_metrics)]
205                crate::metrics::DISPATCH_BLOCK_HISTOGRAM
206                    .with_label_values(&[self.node.address()])
207                    .observe(start.elapsed().as_secs_f64() * 1000.0);
208                Err(e)?
209            }
210        }
211
212        Ok(())
213    }
214}

216struct TaskQueue<S>
217where
218    S: Storage + Clone + Send + Sync + 'static,
219{
220    queue_size: usize,
221    start_height: usize,
222    storage: ExporterStorage<S>,
223    buffer: Sender<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>)>,
224}
225
226impl<S> TaskQueue<S>
227where
228    S: Storage + Clone + Send + Sync + 'static,
229{
230    #[expect(clippy::type_complexity)]
231    fn new(
232        queue_size: usize,
233        start_height: usize,
234        storage: ExporterStorage<S>,
235    ) -> (
236        TaskQueue<S>,
237        Receiver<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>)>,
238    ) {
239        let (sender, receiver) = tokio::sync::mpsc::channel(queue_size);
240
241        let queue = Self {
242            queue_size,
243            start_height,
244            storage,
245            buffer: sender,
246        };
247
248        (queue, receiver)
249    }
250
251    async fn run(&mut self) -> anyhow::Result<()> {
252        let mut index = self.start_height;
253        let mut futures = FuturesOrdered::new();
254        while futures.len() < self.queue_size {
255            futures.push_back(self.get_block_task(index));
256            index += 1;
257        }
258
259        while let Some(certificate) = futures.next().await.transpose()? {
260            // Ignore send errors; the receiver may have been dropped during shutdown.
261            self.buffer.send(certificate).await.ok();
262            futures.push_back(self.get_block_task(index));
263            index += 1;
264        }
265
266        Ok(())
267    }
268
269    async fn get_block_task(
270        &self,
271        index: usize,
272    ) -> Result<(Arc<ConfirmedBlockCertificate>, Vec<BlobId>), ExporterError> {
273        loop {
274            match self.storage.get_block_with_blob_ids(index).await {
275                Ok(block_with_blobs_ids) => return Ok(block_with_blobs_ids),
276                Err(ExporterError::UnprocessedBlock) => {
277                    tokio::time::sleep(Duration::from_secs(1)).await
278                }
279                Err(e) => return Err(e),
280            }
281        }
282    }
283}
