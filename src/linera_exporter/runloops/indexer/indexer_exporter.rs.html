28pub(crate) struct Exporter {
29    options: NodeOptions,
30    work_queue_size: usize,
31    destination_id: DestinationId,
32}
33
34impl Exporter {
35    pub(crate) fn new(
36        destination_id: DestinationId,
37        work_queue_size: usize,
38        options: NodeOptions,
39    ) -> Exporter {
40        Self {
41            options,
42            destination_id,
43            work_queue_size,
44        }
45    }
46
47    pub(crate) async fn run_with_shutdown<S, F: IntoFuture<Output = ()>>(
48        self,
49        shutdown_signal: F,
50        mut storage: ExporterStorage<S>,
51    ) -> anyhow::Result<()>
52    where
53        S: Storage + Clone + Send + Sync + 'static,
54    {
55        let shutdown_signal_future = shutdown_signal.into_future();
56        let mut pinned_shutdown_signal = Box::pin(shutdown_signal_future);
57
58        let address = self.destination_id.address();
59        let mut client = IndexerClient::new(address, self.options)?;
60        let destination_state = storage.load_destination_state(&self.destination_id);
61
62        tracing::info!(
63            start_index=&destination_state.load(Ordering::SeqCst),
64            indexer_address=%address,
65            "starting indexer exporter"
66        );
67
68        loop {
69            let (outgoing_stream, incoming_stream) =
70                client.setup_indexer_client(self.work_queue_size).await?;
71
72            let mut streamer = ExportTaskQueue::new(
73                self.work_queue_size,
74                destination_state.load(Ordering::Acquire) as usize,
75                outgoing_stream,
76                storage.clone()?,
77            );
78
79            let mut acknowledgement_task =
80                AcknowledgementTask::new(incoming_stream, destination_state.clone());
81
82            select! {
83
84                biased;
85
86                _ = &mut pinned_shutdown_signal => {break},
87
88                res = streamer.run() => {
89                    if let Err(error) = res {
90                        tracing::error!(?error, "exporter stream error. re-trying to establish a stream");
91                        client = IndexerClient::new(address, self.options)?;
92                        sleep(Duration::from_millis(500)).await;
93                    }
94                },
95
96                res = acknowledgement_task.run() => {
97                    match res {
98                        Err(error) => {
99                            tracing::error!(?error, "ack stream error. re-trying to establish a stream");
100                            client = IndexerClient::new(address, self.options)?;
101                        }
102
103                        Ok(_) => {
104                            tracing::error!("ack stream unexpectedly. retrying to establish a stream");
105                            client = IndexerClient::new(address, self.options)?;
106                        }
107                    }
108
109                    sleep(Duration::from_millis(500)).await;
110                },
111
112            }
113        }
114
115        Ok(())
116    }
117}

119struct AcknowledgementTask {
120    incoming: Streaming<()>,
121    destination_state: Arc<AtomicU64>,
122}
123
124impl AcknowledgementTask {
125    fn new(incoming: Streaming<()>, destination_state: Arc<AtomicU64>) -> Self {
126        Self {
127            incoming,
128            destination_state,
129        }
130    }
131
132    async fn run(&mut self) -> anyhow::Result<()> {
133        while self.incoming.message().await?.is_some() {
134            self.increment_destination_state();
135        }
136
137        Ok(())
138    }
139
140    fn increment_destination_state(&self) {
141        self.destination_state.fetch_add(1, Ordering::Release);
142    }
143}

145struct ExportTaskQueue<S>
146where
147    S: Storage + Clone + Send + Sync + 'static,
148{
149    queue_size: usize,
150    start_height: usize,
151    buffer: CanonicalBlockStream,
152    storage: ExporterStorage<S>,
153}
154
155impl<S> ExportTaskQueue<S>
156where
157    S: Storage + Clone + Send + Sync + 'static,
158{
159    fn new(
160        queue_size: usize,
161        start_height: usize,
162        sender: CanonicalBlockStream,
163        storage: ExporterStorage<S>,
164    ) -> ExportTaskQueue<S> {
165        Self {
166            queue_size,
167            start_height,
168            storage,
169            buffer: sender,
170        }
171    }
172
173    async fn run(&mut self) -> anyhow::Result<()> {
174        let mut index = self.start_height;
175        let mut futures = FuturesOrdered::new();
176        while futures.len() < self.queue_size {
177            futures.push_back(self.get_block_with_blobs_task(index));
178            index += 1;
179        }
180
181        while let Some((block, blobs)) = futures.next().await.transpose()? {
182            for blob in blobs {
183                tracing::info!(
184                    blob_id=?blob.id(),
185                    "dispatching blob"
186                );
187                self.buffer.send(blob.try_into().unwrap()).await?
188            }
189
190            let block_id = BlockId::from_confirmed_block(block.value());
191            tracing::info!(?block_id, "dispatching block");
192            self.buffer.send(block.try_into().unwrap()).await?;
193
194            futures.push_back(self.get_block_with_blobs_task(index));
195            index += 1;
196        }
197
198        Ok(())
199    }
200
201    async fn get_block_with_blobs_task(
202        &self,
203        index: usize,
204    ) -> Result<(Arc<ConfirmedBlockCertificate>, Vec<Arc<Blob>>), ExporterError> {
205        loop {
206            match self.storage.get_block_with_blobs(index).await {
207                Ok(res) => return Ok(res),
208                Err(ExporterError::UnprocessedBlock) => {
209                    tokio::time::sleep(Duration::from_secs(1)).await
210                }
211                Err(e) => return Err(e),
212            }
213        }
214    }
215}
216
217type CanonicalBlockStream = Sender<Element>;
