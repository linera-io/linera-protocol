28pub(crate) struct Exporter {
29    options: NodeOptions,
30    work_queue_size: usize,
31    destination_id: DestinationId,
32}
33
34impl Exporter {
35    pub(crate) fn new(
36        destination_id: DestinationId,
37        work_queue_size: usize,
38        options: NodeOptions,
39    ) -> Exporter {
40        Self {
41            options,
42            destination_id,
43            work_queue_size,
44        }
45    }
46
47    pub(crate) async fn run_with_shutdown<S, F: IntoFuture<Output = ()>>(
48        self,
49        shutdown_signal: F,
50        mut storage: ExporterStorage<S>,
51    ) -> anyhow::Result<()>
52    where
53        S: Storage + Clone + Send + Sync + 'static,
54    {
55        let shutdown_signal_future = shutdown_signal.into_future();
56        let mut pinned_shutdown_signal = Box::pin(shutdown_signal_future);
57
58        let address = self.destination_id.address();
59        let mut client = IndexerClient::new(address, self.options)?;
60        let destination_state = storage.load_destination_state(&self.destination_id);
61
62        tracing::info!(start_index=&destination_state.load(Ordering::SeqCst), indexer_address=%address, "starting indexer exporter");
63
64        loop {
65            let (outgoing_stream, incoming_stream) =
66                client.setup_indexer_client(self.work_queue_size).await?;
67            let mut streamer = ExportTaskQueue::new(
68                self.work_queue_size,
69                destination_state.load(Ordering::Acquire) as usize,
70                outgoing_stream,
71                storage.clone()?,
72            );
73
74            let mut acknowledgement_task =
75                AcknowledgementTask::new(incoming_stream, destination_state.clone());
76
77            select! {
78
79                biased;
80
81                _ = &mut pinned_shutdown_signal => {break},
82
83                res = streamer.run() => {
84                    if let Err(e) = res {
85                        tracing::error!("unexpected error: {e}, re-trying to establish a stream");
86                        sleep(Duration::from_secs(1)).await;
87                    }
88                },
89
90                res = acknowledgement_task.run() => {
91                    match res {
92                        Err(e) => {
93                            tracing::error!("unexpected error: {e}, re-trying to establish a stream");
94                        }
95
96                        Ok(_) => {
97                            tracing::error!("stream closed unexpectedly, retrying to establish a stream");
98                        }
99                    }
100
101                    sleep(Duration::from_secs(1)).await;
102                },
103
104            }
105        }
106
107        Ok(())
108    }
109}

111struct AcknowledgementTask {
112    incoming: Streaming<()>,
113    destination_state: Arc<AtomicU64>,
114}
115
116impl AcknowledgementTask {
117    fn new(incoming: Streaming<()>, destination_state: Arc<AtomicU64>) -> Self {
118        Self {
119            incoming,
120            destination_state,
121        }
122    }
123
124    async fn run(&mut self) -> anyhow::Result<()> {
125        while self.incoming.message().await?.is_some() {
126            self.increment_destination_state();
127        }
128
129        Ok(())
130    }
131
132    fn increment_destination_state(&self) {
133        let _ = self.destination_state.fetch_add(1, Ordering::Release);
134    }
135}

137struct ExportTaskQueue<S>
138where
139    S: Storage + Clone + Send + Sync + 'static,
140{
141    queue_size: usize,
142    start_height: usize,
143    buffer: CanonicalBlockStream,
144    storage: ExporterStorage<S>,
145}
146
147impl<S> ExportTaskQueue<S>
148where
149    S: Storage + Clone + Send + Sync + 'static,
150{
151    fn new(
152        queue_size: usize,
153        start_height: usize,
154        sender: CanonicalBlockStream,
155        storage: ExporterStorage<S>,
156    ) -> ExportTaskQueue<S> {
157        Self {
158            queue_size,
159            start_height,
160            storage,
161            buffer: sender,
162        }
163    }
164
165    async fn run(&mut self) -> anyhow::Result<()> {
166        let mut index = self.start_height;
167        let mut futures = FuturesOrdered::new();
168        while futures.len() < self.queue_size {
169            futures.push_back(self.get_block_with_blobs_task(index));
170            index += 1;
171        }
172
173        while let Some((block, blobs)) = futures.next().await.transpose()? {
174            for blob in blobs {
175                tracing::info!(
176                    blob_id=?blob.id(),
177                    "dispatching blob"
178                );
179                self.buffer.send(blob.try_into().unwrap()).await?
180            }
181
182            let block_id = BlockId::from_confirmed_block(block.value());
183            tracing::info!(?block_id, "dispatching block");
184            self.buffer.send(block.try_into().unwrap()).await?;
185
186            futures.push_back(self.get_block_with_blobs_task(index));
187            index += 1;
188        }
189
190        Ok(())
191    }
192
193    async fn get_block_with_blobs_task(
194        &self,
195        index: usize,
196    ) -> Result<(Arc<ConfirmedBlockCertificate>, Vec<Arc<Blob>>), ExporterError> {
197        loop {
198            match self.storage.get_block_with_blobs(index).await {
199                Ok(res) => return Ok(res),
200                Err(ExporterError::UnprocessedBlock) => {
201                    tokio::time::sleep(Duration::from_secs(1)).await
202                }
203                Err(e) => return Err(e),
204            }
205        }
206    }
207}
208
209type CanonicalBlockStream = Sender<Element>;
