20#[derive(thiserror::Error, Debug)]
21pub(crate) enum ExporterError {
22    #[error("received an invalid notification.")]
23    BadNotification(BadNotificationKind),
24
25    #[error("unable to load the exporter state: {0}")]
26    StateError(ViewError),
27
28    #[error("Missing certificate: {0}")]
29    ReadCertificateError(CryptoHash),
30
31    #[error("generic storage error: {0}")]
32    ViewError(#[from] ViewError),
33
34    #[error("block not processed by the block processor yet")]
35    UnprocessedBlock,
36
37    #[error("chain not yet processed by the block exporter")]
38    UnprocessedChain,
39
40    #[error("chain should be initialized from block zero")]
41    BadInitialization,
42
43    #[error("trying to re-initialize the chain: {0}")]
44    ChainAlreadyExists(ChainId),
45
46    #[error("unable to establish an underlying stream")]
47    SynchronizationFailed(Box<Status>),
48
49    #[error(transparent)]
50    GrpcError(#[from] GrpcError),
51
52    #[error("generic error: {0}")]
53    GenericError(Box<dyn std::error::Error + Send + Sync + 'static>),
54}

56#[derive(Debug)]
57pub(crate) enum BadNotificationKind {
58    InvalidChainId {
59        #[debug(skip_if = Option::is_none)]
60        inner: Option<GrpcProtoConversionError>,
61    },
62
63    InvalidReason {
64        #[debug(skip_if = Option::is_none)]
65        inner: Option<Box<ErrorKind>>,
66    },
67}
68
69impl From<BadNotificationKind> for ExporterError {
70    fn from(value: BadNotificationKind) -> Self {
71        ExporterError::BadNotification(value)
72    }
73}

97#[derive(Clone, Debug, Serialize, Deserialize)]
98pub(crate) struct LiteBlockId {
99    pub hash: CryptoHash,
100    pub height: BlockHeight,
101}
102
103impl BlockId {
104    pub(crate) fn new(chain_id: ChainId, hash: CryptoHash, height: BlockHeight) -> BlockId {
105        BlockId {
106            hash,
107            chain_id,
108            height,
109        }
110    }
111
112    pub(crate) fn from_incoming_bundle(incoming_bundle: &IncomingBundle) -> Self {
113        Self {
114            hash: incoming_bundle.bundle.certificate_hash,
115            chain_id: incoming_bundle.origin,
116            height: incoming_bundle.bundle.height,
117        }
118    }
119
120    pub(crate) fn from_confirmed_block(block: &ConfirmedBlock) -> BlockId {
121        BlockId::new(block.chain_id(), block.inner().hash(), block.height())
122    }
123}
124
125impl LiteBlockId {
126    pub(crate) fn new(height: BlockHeight, hash: CryptoHash) -> LiteBlockId {
127        LiteBlockId { hash, height }
128    }
129}
130
131impl From<BlockId> for LiteBlockId {
132    fn from(value: BlockId) -> Self {
133        LiteBlockId::new(value.height, value.hash)
134    }
135}

137#[derive(Clone)]
138pub(crate) struct ExporterCancellationSignal {
139    token: CancellationToken,
140}
141
142impl ExporterCancellationSignal {
143    pub(crate) fn new(token: CancellationToken) -> Self {
144        Self { token }
145    }
146}
147
148impl IntoFuture for ExporterCancellationSignal {
149    type Output = ();
150    type IntoFuture = WaitForCancellationFutureOwned;
151
152    fn into_future(self) -> Self::IntoFuture {
153        self.token.cancelled_owned()
154    }
155}
