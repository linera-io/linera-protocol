4//! GraphQL queries against the indexer's main API.
5
6// The `GraphQLQuery` derive macro generates public items (per-query modules,
7// `Variables` and `ResponseData` structs, etc.) that cannot carry doc comments,
8// so this module of generated bindings is exempted from the crate's `missing_docs`
9// policy. `expect` (rather than `allow`) flags this if the generated code ever stops
10// producing undocumented items. The hand-written query structs are still documented below.
11#![expect(missing_docs)]
12
13use graphql_client::GraphQLQuery;
14use linera_base::{crypto::CryptoHash, data_types::BlockHeight, identifiers::ChainId};
15
16/// GraphQL query returning the list of plugins registered with the indexer.
17#[derive(GraphQLQuery)]
18#[graphql(
19    schema_path = "gql/indexer_schema.graphql",
20    query_path = "gql/indexer_requests.graphql",
21    response_derives = "Debug, Serialize, Clone"
22)]
23pub struct Plugins;
24
25/// GraphQL query returning the indexer's current state (chain, block and height).
26#[derive(GraphQLQuery)]
27#[graphql(
28    schema_path = "gql/indexer_schema.graphql",
29    query_path = "gql/indexer_requests.graphql",
30    response_derives = "Debug, Serialize, Clone"
31)]
32pub struct State;
