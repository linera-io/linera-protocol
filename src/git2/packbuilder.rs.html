11/// Stages that are reported by the `PackBuilder` progress callback.
12pub enum PackBuilderStage {
13    /// Adding objects to the pack
14    AddingObjects,
15    /// Deltafication of the pack
16    Deltafication,
17}
18
19pub type ProgressCb<'a> = dyn FnMut(PackBuilderStage, u32, u32) -> bool + 'a;
20pub type ForEachCb<'a> = dyn FnMut(&[u8]) -> bool + 'a;

22/// A builder for creating a packfile
23pub struct PackBuilder<'repo> {
24    raw: *mut raw::git_packbuilder,
25    _progress: Option<Box<Box<ProgressCb<'repo>>>>,
26    _marker: marker::PhantomData<&'repo Repository>,
27}

29impl<'repo> PackBuilder<'repo> {
30    /// Insert a single object. For an optimal pack it's mandatory to insert
31    /// objects in recency order, commits followed by trees and blobs.
32    pub fn insert_object(&mut self, id: Oid, name: Option<&str>) -> Result<(), Error> {
33        let name = crate::opt_cstr(name)?;
34        unsafe {
35            try_call!(raw::git_packbuilder_insert(self.raw, id.raw(), name));
36        }
37        Ok(())
38    }
39
40    /// Insert a root tree object. This will add the tree as well as all
41    /// referenced trees and blobs.
42    pub fn insert_tree(&mut self, id: Oid) -> Result<(), Error> {
43        unsafe {
44            try_call!(raw::git_packbuilder_insert_tree(self.raw, id.raw()));
45        }
46        Ok(())
47    }
48
49    /// Insert a commit object. This will add a commit as well as the completed
50    /// referenced tree.
51    pub fn insert_commit(&mut self, id: Oid) -> Result<(), Error> {
52        unsafe {
53            try_call!(raw::git_packbuilder_insert_commit(self.raw, id.raw()));
54        }
55        Ok(())
56    }
57
58    /// Insert objects as given by the walk. Those commits and all objects they
59    /// reference will be inserted into the packbuilder.
60    pub fn insert_walk(&mut self, walk: &mut Revwalk<'_>) -> Result<(), Error> {
61        unsafe {
62            try_call!(raw::git_packbuilder_insert_walk(self.raw, walk.raw()));
63        }
64        Ok(())
65    }
66
67    /// Recursively insert an object and its referenced objects. Insert the
68    /// object as well as any object it references.
69    pub fn insert_recursive(&mut self, id: Oid, name: Option<&str>) -> Result<(), Error> {
70        let name = crate::opt_cstr(name)?;
71        unsafe {
72            try_call!(raw::git_packbuilder_insert_recur(self.raw, id.raw(), name));
73        }
74        Ok(())
75    }
76
77    /// Write the contents of the packfile to an in-memory buffer. The contents
78    /// of the buffer will become a valid packfile, even though there will be
79    /// no attached index.
80    pub fn write_buf(&mut self, buf: &mut Buf) -> Result<(), Error> {
81        unsafe {
82            try_call!(raw::git_packbuilder_write_buf(buf.raw(), self.raw));
83        }
84        Ok(())
85    }
86
87    /// Create the new pack and pass each object to the callback.
88    pub fn foreach<F>(&mut self, mut cb: F) -> Result<(), Error>
89    where
90        F: FnMut(&[u8]) -> bool,
91    {
92        let mut cb = &mut cb as &mut ForEachCb<'_>;
93        let ptr = &mut cb as *mut _;
94        let foreach: raw::git_packbuilder_foreach_cb = Some(foreach_c);
95        unsafe {
96            try_call!(raw::git_packbuilder_foreach(
97                self.raw,
98                foreach,
99                ptr as *mut _
100            ));
101        }
102        Ok(())
103    }
104
105    /// `progress` will be called with progress information during pack
106    /// building. Be aware that this is called inline with pack building
107    /// operations, so performance may be affected.
108    ///
109    /// There can only be one progress callback attached, this will replace any
110    /// existing one. See `unset_progress_callback` to remove the current
111    /// progress callback without attaching a new one.
112    pub fn set_progress_callback<F>(&mut self, progress: F) -> Result<(), Error>
113    where
114        F: FnMut(PackBuilderStage, u32, u32) -> bool + 'repo,
115    {
116        let mut progress = Box::new(Box::new(progress) as Box<ProgressCb<'_>>);
117        let ptr = &mut *progress as *mut _;
118        let progress_c: raw::git_packbuilder_progress = Some(progress_c);
119        unsafe {
120            try_call!(raw::git_packbuilder_set_callbacks(
121                self.raw,
122                progress_c,
123                ptr as *mut _
124            ));
125        }
126        self._progress = Some(progress);
127        Ok(())
128    }
129
130    /// Remove the current progress callback.  See `set_progress_callback` to
131    /// set the progress callback.
132    pub fn unset_progress_callback(&mut self) -> Result<(), Error> {
133        unsafe {
134            try_call!(raw::git_packbuilder_set_callbacks(
135                self.raw,
136                None,
137                ptr::null_mut()
138            ));
139            self._progress = None;
140        }
141        Ok(())
142    }

144    /// Set the number of threads to be used.
145    ///
146    /// Returns the number of threads to be used.
147    pub fn set_threads(&mut self, threads: u32) -> u32 {
148        unsafe { raw::git_packbuilder_set_threads(self.raw, threads) }
149    }

151    /// Get the total number of objects the packbuilder will write out.
152    pub fn object_count(&self) -> usize {
153        unsafe { raw::git_packbuilder_object_count(self.raw) }
154    }
155
156    /// Get the number of objects the packbuilder has already written out.
157    pub fn written(&self) -> usize {
158        unsafe { raw::git_packbuilder_written(self.raw) }
159    }
160
161    /// Get the packfile's hash. A packfile's name is derived from the sorted
162    /// hashing of all object names. This is only correct after the packfile
163    /// has been written.
164    #[deprecated = "use `name()` to retrieve the filename"]
165    #[allow(deprecated)]
166    pub fn hash(&self) -> Option<Oid> {
167        if self.object_count() == 0 {
168            unsafe { Some(Binding::from_raw(raw::git_packbuilder_hash(self.raw))) }
169        } else {
170            None
171        }
172    }
173
174    /// Get the unique name for the resulting packfile.
175    ///
176    /// The packfile's name is derived from the packfile's content. This is only
177    /// correct after the packfile has been written.
178    ///
179    /// Returns `None` if the packfile has not been written or if the name is
180    /// not valid utf-8.
181    pub fn name(&self) -> Option<&str> {
182        self.name_bytes().and_then(|s| str::from_utf8(s).ok())
183    }
184
185    /// Get the unique name for the resulting packfile, in bytes.
186    ///
187    /// The packfile's name is derived from the packfile's content. This is only
188    /// correct after the packfile has been written.
189    pub fn name_bytes(&self) -> Option<&[u8]> {
190        unsafe { crate::opt_bytes(self, raw::git_packbuilder_name(self.raw)) }
191    }

209    fn drop(&mut self) {
210        unsafe {

212            raw::git_packbuilder_free(self.raw);

217impl Binding for PackBuilderStage {
218    type Raw = raw::git_packbuilder_stage_t;
219    unsafe fn from_raw(raw: raw::git_packbuilder_stage_t) -> PackBuilderStage {
220        match raw {
221            raw::GIT_PACKBUILDER_ADDING_OBJECTS => PackBuilderStage::AddingObjects,
222            raw::GIT_PACKBUILDER_DELTAFICATION => PackBuilderStage::Deltafication,
223            _ => panic!("Unknown git diff binary kind"),
224        }
225    }
226    fn raw(&self) -> raw::git_packbuilder_stage_t {
227        match *self {
228            PackBuilderStage::AddingObjects => raw::GIT_PACKBUILDER_ADDING_OBJECTS,
229            PackBuilderStage::Deltafication => raw::GIT_PACKBUILDER_DELTAFICATION,
230        }
231    }
232}

234extern "C" fn foreach_c(buf: *const c_void, size: size_t, data: *mut c_void) -> c_int {
235    unsafe {
236        let buf = slice::from_raw_parts(buf as *const u8, size as usize);
237
238        let r = panic::wrap(|| {
239            let data = data as *mut &mut ForEachCb<'_>;
240            (*data)(buf)
241        });
242        if r == Some(true) {
243            0
244        } else {
245            -1
246        }
247    }
248}
249
250extern "C" fn progress_c(
251    stage: raw::git_packbuilder_stage_t,
252    current: c_uint,
253    total: c_uint,
254    data: *mut c_void,
255) -> c_int {
256    unsafe {
257        let stage = Binding::from_raw(stage);
258
259        let r = panic::wrap(|| {
260            let data = data as *mut Box<ProgressCb<'_>>;
261            (*data)(stage, current, total)
262        });
263        if r == Some(true) {
264            0
265        } else {
266            -1
267        }
268    }

275    fn pack_header(len: u8) -> Vec<u8> {
276        [].iter()
277            .chain(b"PACK") // signature
278            .chain(&[0, 0, 0, 2]) // version number
279            .chain(&[0, 0, 0, len]) // number of objects
280            .cloned()
281            .collect::<Vec<u8>>()
282    }
283
284    fn empty_pack_header() -> Vec<u8> {
285        pack_header(0)
286            .iter()
287            .chain(&[
288                0x02, 0x9d, 0x08, 0x82, 0x3b, // ^
289                0xd8, 0xa8, 0xea, 0xb5, 0x10, // | SHA-1 of the zero
290                0xad, 0x6a, 0xc7, 0x5c, 0x82, // | object pack header
291                0x3c, 0xfd, 0x3e, 0xd3, 0x1e,
292            ]) // v
293            .cloned()
294            .collect::<Vec<u8>>()
295    }
