24/// A system execution state, not represented as a view but as a simple struct.
25#[derive(Default, Debug, PartialEq, Eq, Clone)]
26pub struct SystemExecutionState {
27    /// The description of the chain, if it has been created.
28    pub description: Option<ChainDescription>,
29    /// The current epoch the chain is operating in.
30    pub epoch: Epoch,
31    /// The ID of the admin chain, if known.
32    pub admin_chain_id: Option<ChainId>,
33    /// The committees of validators, indexed by the epoch in which they are active.
34    pub committees: BTreeMap<Epoch, Committee>,
35    /// The ownership configuration of the chain.
36    pub ownership: ChainOwnership,
37    /// The chain's main balance.
38    pub balance: Amount,
39    /// The per-owner balances held on the chain.
40    #[debug(skip_if = BTreeMap::is_empty)]
41    pub balances: BTreeMap<AccountOwner, Amount>,
42    /// The latest timestamp recorded for the chain.
43    pub timestamp: Timestamp,
44    /// The set of blobs that have been used by the chain.
45    pub used_blobs: BTreeSet<BlobId>,
46    /// Whether the chain has been closed.
47    #[debug(skip_if = Not::not)]
48    pub closed: bool,
49    /// The application permissions configured on the chain.
50    pub application_permissions: ApplicationPermissions,
51    /// Additional blobs to make available to the chain's execution context.
52    #[debug(skip_if = Vec::is_empty)]
53    pub extra_blobs: Vec<Blob>,
54    /// The mock applications registered on the chain, indexed by their application ID.
55    #[debug(skip_if = BTreeMap::is_empty)]
56    pub mock_applications: BTreeMap<ApplicationId, MockApplication>,
57    /// Number of incoming message bundles executed so far.
58    pub num_incoming_bundles: u32,
59    /// Number of operations executed so far.
60    pub num_operations: u32,
61    /// Number of outgoing messages sent so far.
62    pub num_outgoing_messages: u32,
63}
64
65impl SystemExecutionState {
66    /// Creates a system execution state from a chain description, with dummy committees.
67    pub fn new(description: ChainDescription) -> Self {
68        let ownership = description.config().ownership.clone();
69        let balance = description.config().balance;
70        let epoch = description.config().epoch;
71        let admin_chain_id = Some(dummy_chain_description(0).id());
72        SystemExecutionState {
73            epoch,
74            description: Some(description),
75            admin_chain_id,
76            ownership,
77            balance,
78            committees: dummy_committees(),
79            ..SystemExecutionState::default()
80        }
81    }
82
83    /// Creates a dummy system execution state for the chain with the given index, returning it
84    /// together with its chain ID.
85    pub fn dummy_chain_state(index: u32) -> (Self, ChainId) {
86        let description = dummy_chain_description(index);
87        let chain_id = description.id();
88        (Self::new(description), chain_id)
89    }
90
91    /// Builds an execution state view from this state and returns its cryptographic hash.
92    pub async fn into_hash(self) -> CryptoHash {
93        let mut view = self.into_view().await;
94        view.crypto_hash_mut()
95            .await
96            .expect("hashing from memory should not fail")
97    }
98
99    /// Converts this state into an execution state view backed by an in-memory context.
100    pub async fn into_view(self) -> ExecutionStateView<MemoryContext<TestExecutionRuntimeContext>> {
101        let chain_id = self
102            .description
103            .as_ref()
104            .expect("Chain description should be set")
105            .into();
106        self.into_view_with(chain_id, ExecutionRuntimeConfig::default())
107            .await
108    }
109
110    /// Converts this state into an execution state view for the given chain ID and runtime
111    /// configuration.
112    pub async fn into_view_with(
113        self,
114        chain_id: ChainId,
115        execution_runtime_config: ExecutionRuntimeConfig,
116    ) -> ExecutionStateView<MemoryContext<TestExecutionRuntimeContext>> {
117        // Destructure, to make sure we don't miss any fields.
118        let SystemExecutionState {
119            description,
120            epoch,
121            admin_chain_id,
122            committees,
123            ownership,
124            balance,
125            balances,
126            timestamp,
127            used_blobs,
128            closed,
129            application_permissions,
130            extra_blobs,
131            mock_applications,
132            num_incoming_bundles,
133            num_operations,
134            num_outgoing_messages,
135        } = self;
136
137        let extra = TestExecutionRuntimeContext::new(chain_id, execution_runtime_config);
138        extra
139            .add_blobs(extra_blobs)
140            .await
141            .expect("Adding blobs to the `TestExecutionRuntimeContext` should not fail");
142        for (id, mock_application) in mock_applications {
143            extra
144                .user_contracts()
145                .pin()
146                .insert(id, mock_application.clone().into());
147            extra
148                .user_services()
149                .pin()
150                .insert(id, mock_application.into());
151        }
152
153        let mut committee_hashes = BTreeMap::new();
154        for (committee_epoch, committee) in committees {
155            let blob = Blob::new_committee(bcs::to_bytes(&committee).expect("BCS should succeed"));
156            let hash = blob.id().hash;
157            extra
158                .add_blobs([blob])
159                .await
160                .expect("Adding committee blobs should not fail");
161            committee_hashes.insert(committee_epoch, hash);
162        }
163
164        let context = MemoryContext::new_for_testing(extra);
165        let mut view = ExecutionStateView::load(context)
166            .await
167            .expect("Loading from memory should work");
168        view.system.description.set(description);
169        view.system.epoch.set(epoch);
170        view.system.admin_chain_id.set(admin_chain_id);
171        view.system
172            .committee_hash
173            .set(committee_hashes.get(&epoch).copied());
174        view.system.ownership.set(ownership);
175        view.system.balance.set(balance);
176        for (account_owner, balance) in balances {
177            view.system
178                .balances
179                .insert(&account_owner, balance)
180                .expect("insertion of balances should not fail");
181        }
182        for blob_id in used_blobs {
183            view.system
184                .used_blobs
185                .insert(&blob_id)
186                .expect("inserting blob IDs should not fail");
187        }
188        view.system.closed.set(closed);
189        view.system
190            .application_permissions
191            .set(application_permissions);
192        view.system.progress.set(ChainProgress {
193            timestamp,
194            num_incoming_bundles,
195            num_operations,
196            num_outgoing_messages,
197        });
198        view
199    }
200}

202impl RegisterMockApplication for SystemExecutionState {
203    async fn creator_chain_id(&self) -> ChainId {
204        self.description.as_ref().expect(
205            "Can't register applications on a system state with no associated `ChainDescription`",
206        ).into()
207    }
208
209    async fn register_mock_application_with(
210        &mut self,
211        description: ApplicationDescription,
212        contract: Blob,
213        service: Blob,
214    ) -> anyhow::Result<(ApplicationId, MockApplication)> {
215        let id = ApplicationId::from(&description);
216        let application = MockApplication::default();
217
218        self.extra_blobs.extend([
219            contract,
220            service,
221            Blob::new_application_description(&description),
222        ]);
223        self.mock_applications.insert(id, application.clone());
224
225        Ok((id, application))
226    }
