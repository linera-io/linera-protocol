4use alloy_primitives::U256;
5use linera_base::data_types::Amount;
6use serde::{Deserialize, Deserializer, Serialize, Serializer};
7
8#[derive(Eq, PartialEq, Ord, PartialOrd, Copy, Clone, Hash, Default, Debug)]
9/// An encapsulation of U256 in order to have a specific serialization
10pub struct AmountU256(U256);
11
12impl Serialize for AmountU256 {
13    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
14    where
15        S: Serializer,
16    {
17        use serde::ser::SerializeTuple;
18        let v: [u8; 32] = self.0.to_be_bytes();
19        let mut tuple = serializer.serialize_tuple(32)?;
20        for byte in &v {
21            tuple.serialize_element(byte)?;
22        }
23        tuple.end()
24    }
25}
26
27impl<'de> Deserialize<'de> for AmountU256 {
28    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
29    where
30        D: Deserializer<'de>,
31    {
32        let bytes: [u8; 32] = Deserialize::deserialize(deserializer)?;
33        let value = U256::from_be_bytes(bytes);
34        Ok(AmountU256(value))
35    }
36}
37
38impl From<Amount> for AmountU256 {
39    fn from(amount: Amount) -> AmountU256 {
40        AmountU256(amount.into())
41    }
42}

44#[cfg(test)]
45mod tests {
46    use linera_base::data_types::Amount;
47
48    use crate::evm::data_types::AmountU256;
49
50    #[test]
51    fn check_bcs_serialization() -> anyhow::Result<()> {
52        let value = 7837438347454859505557763535536363636;
53        let value = Amount::from_tokens(value);
54        let value = AmountU256::from(value);
55        let vec = bcs::to_bytes(&value)?;
56        assert_eq!(vec.len(), 32);
57        assert_eq!(value, bcs::from_bytes(&vec)?);
58        Ok(())
59    }
