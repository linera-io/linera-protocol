4/// A stream of events. This is returned from [Observable::observe](crate::Observable::observe).
5/// You will only start receiving events from the moment you call this. Any events in the observed
6/// object emitted before will not be delivered.
7///
8/// For pharos 0.4.0 on x64 Linux: `std::mem::size_of::<Events<_>>() == 16`
9//
10#[ derive( Debug ) ]
11//
12pub struct Events<Event> where Event: Clone + 'static + Send
13{
14	rx: Receiver<Event>,
15}
16
17
18impl<Event> Events<Event> where Event: Clone + 'static + Send
19{
20	pub(crate) fn new( config: ObserveConfig<Event> ) -> (Self, Sender<Event>)
21	{
22		let (tx, rx) = match config.channel
23		{
24			Channel::Bounded( queue_size ) =>
25			{
26				let (tx, rx) = mpsc::channel( queue_size - 1 );
27
28				( Sender::Bounded{ tx, filter: config.filter }, Receiver::Bounded{ rx } )
29			}
30
31			Channel::Unbounded =>
32			{
33				let (tx, rx) = mpsc::unbounded();
34
35				( Sender::Unbounded{ tx, filter: config.filter }, Receiver::Unbounded{ rx } )
36			}
37
38			_ => unreachable!(),
39		};
40
41
42		( Self{ rx }, tx )
43	}
44
45
46	/// Disconnect from the observable object. This way the sender will stop sending new events
47	/// and you can still continue to read any events that are still pending in the channel.
48	//
49	pub fn close( &mut self )
50	{
51		self.rx.close();
52	}
53}

85	// Verify whether this observer is still around.
86	//
87	pub(crate) fn is_closed( &self ) -> bool
88	{
89		match self
90		{
91			Sender::Bounded  { tx, .. } => tx.is_closed(),
92			Sender::Unbounded{ tx, .. } => tx.is_closed(),
93		}
94	}

97	/// Check whether this sender is interested in this event.
98	//
99	pub(crate) fn filter( &mut self, evt: &Event ) -> bool
100	{
101		match self
102		{
103			Sender::Bounded  { filter, .. } => Self::filter_inner( filter, evt ),
104			Sender::Unbounded{ filter, .. } => Self::filter_inner( filter, evt ),
105		}
106	}
107
108
109	fn filter_inner( filter: &mut Option<Filter<Event>>, evt: &Event ) -> bool
110	{
111		match filter
112		{
113			Some(f) => f.call(evt),
114			None    => true       ,
115		}
116	}

175impl<Event> Sink<Event> for Sender<Event> where Event: Clone + 'static + Send
176{
177	type Error = PharErr;
178
179
180	fn poll_ready( self: Pin<&mut Self>, cx: &mut Context<'_> ) -> Poll<Result<(), Self::Error>>
181	{
182		match self.get_mut()
183		{
184			Sender::Bounded  { tx, .. } => Pin::new( tx ).poll_ready( cx ).map_err( Into::into ),
185			Sender::Unbounded{ tx, .. } => Pin::new( tx ).poll_ready( cx ).map_err( Into::into ),
186		}
187	}
188
189
190	fn start_send( self: Pin<&mut Self>, item: Event ) -> Result<(), Self::Error>
191	{
192		match self.get_mut()
193		{
194			Sender::Bounded  { tx, .. } => Pin::new( tx ).start_send( item ).map_err( Into::into ),
195			Sender::Unbounded{ tx, .. } => Pin::new( tx ).start_send( item ).map_err( Into::into ),
196		}
197	}

200	// Note that on futures-rs bounded channels poll_flush has a problematic implementation.
201	// - it just calls poll_ready, which means it will be pending when the buffer is full. So
202	//   it will make SinkExt::send hang, bad!
203	// - it will swallow disconnected errors, so we don't get feedback allowing us to free slots.
204	//
205	// In principle channels are always flushed, because when the message is in the buffer, it's
206	// ready for the reader to read. So this should just be a noop.
207	//
208	// We compensate for the error swallowing by checking `is_closed`.
209	//
210	fn poll_flush( self: Pin<&mut Self>, _cx: &mut Context<'_> ) -> Poll<Result<(), Self::Error>>
211	{
212		match self.get_mut()
213		{
214			Sender::Bounded  { tx, .. } =>
215			{
216				if tx.is_closed() { Poll::Ready(Err( ErrorKind::Closed.into() ))}
217				else              { Poll::Ready(Ok ( ()                       ))}
218			}
219
220			Sender::Unbounded{ tx, .. } =>
221			{
222				if tx.is_closed() { Poll::Ready(Err( ErrorKind::Closed.into() ))}
223				else              { Poll::Ready(Ok ( ()                       ))}
224			}
225		}
226	}
227
228
229	fn poll_close( self: Pin<&mut Self>, cx: &mut Context<'_> ) -> Poll<Result<(), Self::Error>>
230	{
231		match self.get_mut()
232		{
233			Sender::Bounded  { tx, .. } => Pin::new( tx ).poll_close( cx ).map_err( Into::into ),
234			Sender::Unbounded{ tx, .. } => Pin::new( tx ).poll_close( cx ).map_err( Into::into ),
235		}
236	}
237}
