3/// Predicate for filtering events.
4///
5/// This is an enum because closures that capture variables from
6/// their environment need to be boxed. More often than not, an event will be a simple enum and
7/// the predicate will just match on the variant, so it would be wasteful to impose boxing in those
8/// cases, hence there is a function pointer variant which does not require boxing. This should
9/// be preferred where possible.
10///
11/// ```
12/// use pharos::*;
13///
14/// let a = 5;
15///
16/// // This closure captures the a variable from it's environment.
17/// // We can still use it as a filter by boxing it with `closure`.
18/// //
19/// // Note: it depends on the circumstances, but often enough, we need to
20/// // annotate the type of the event parameter to the predicate.
21/// //
22/// // For this example we use bool as event type for simplicity, but it works
23/// // just the same if that's an enum.
24/// //
25/// let predicate = move |_: &bool| { a; true };
26///
27/// let filter = Filter::Closure( Box::new(predicate) );
28///
29/// // This one does not capture anything, so it can be stored as a function pointer
30/// // without boxing.
31/// //
32/// let predicate = move |_: &bool| { true };
33///
34/// let filter = Filter::Pointer( predicate );
35///
36/// // You can also use actual functions as filters.
37/// //
38/// fn predicate_function( event: &bool ) -> bool { true }
39///
40/// let filter = Filter::Pointer( predicate_function );
41/// ```
42//
43pub enum Filter<Event>
44
45	where Event: Clone + 'static + Send ,
46
47{
48	/// A function pointer to a predicate to filter events.
49	//
50	Pointer( fn(&Event) -> bool ),
51
52	/// A boxed closure to a predicate to filter events.
53	//
54	Closure( Box<dyn FnMut(&Event) -> bool + Send> ),
55}
56
57
58impl<Event> Filter<Event>  where Event: Clone + 'static + Send
59{
60	/// Invoke the predicate.
61	//
62	pub(crate) fn call( &mut self, evt: &Event ) -> bool
63	{
64		match self
65		{
66			Self::Pointer(f) => f(evt),
67			Self::Closure(f) => f(evt),
68		}
69	}
70}
71
72
73impl<Event> fmt::Debug for Filter<Event>  where Event: Clone + 'static + Send
74{
75	fn fmt( &self, f: &mut fmt::Formatter<'_> ) -> fmt::Result
76	{
77		match self
78		{
79			Self::Pointer(_) => write!( f, "pharos::Filter<{}>::Pointer(_)", type_name::<Event>() ),
80			Self::Closure(_) => write!( f, "pharos::Filter<{}>::Closure(_)", type_name::<Event>() ),
81		}
82
83	}
84}
