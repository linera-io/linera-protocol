1//! Modified implementations of unstable libcore functions.

3use core::mem::{self, MaybeUninit};

5/// [`core::array::try_from_fn`]
6#[inline]
7pub(crate) fn try_from_fn<F, T, E, const N: usize>(mut cb: F) -> Result<[T; N], E>
8where
9    F: FnMut(usize) -> Result<T, E>,
10{
11    if N == 0 {
12        // SAFETY: An empty array is always inhabited and has no validity invariants.
13        return unsafe { Ok(mem::zeroed()) };
14    }
15
16    struct Guard<'a, T, const N: usize> {
17        array_mut: &'a mut [MaybeUninit<T>; N],
18        initialized: usize,
19    }
20
21    impl<T, const N: usize> Drop for Guard<'_, T, N> {
22        fn drop(&mut self) {
23            debug_assert!(self.initialized <= N);
24
25            // SAFETY: this slice will contain only initialized objects.
26            unsafe {
27                core::ptr::drop_in_place(slice_assume_init_mut(
28                    self.array_mut.get_unchecked_mut(..self.initialized),
29                ));
30            }
31        }
32    }
33
34    let mut array = uninit_array::<T, N>();
35    let mut guard = Guard { array_mut: &mut array, initialized: 0 };
36
37    for _ in 0..N {
38        // SAFETY: `guard.initialized` starts at 0, is increased by one in the
39        // loop and the loop is aborted once it reaches N (which is `array.len()`).
40        unsafe {
41            guard.array_mut.get_unchecked_mut(guard.initialized).write(cb(guard.initialized)?);
42        }
43        guard.initialized += 1;
44    }
45
46    mem::forget(guard);
47    // SAFETY: all elements are initialized.
48    Ok(unsafe { array_assume_init(array) })
49}

51/// `MaybeUninit::slice_assume_init_mut`
52#[inline(always)]
53unsafe fn slice_assume_init_mut<T>(slice: &mut [MaybeUninit<T>]) -> &mut [T] {
54    // SAFETY: similar to safety notes for `slice_get_ref`, but we have a
55    // mutable reference which is also guaranteed to be valid for writes.
56    unsafe { &mut *(slice as *mut [MaybeUninit<T>] as *mut [T]) }
57}
58
59/// `MaybeUninit::uninit_array`
60#[inline]
61pub(crate) fn uninit_array<T, const N: usize>() -> [MaybeUninit<T>; N] {
62    // SAFETY: An uninitialized `[MaybeUninit<_>; N]` is valid.
63    unsafe { MaybeUninit::<[MaybeUninit<T>; N]>::uninit().assume_init() }
64}
65
66/// [`MaybeUninit::array_assume_init`]
67#[inline]
68pub(crate) unsafe fn array_assume_init<T, const N: usize>(array: [MaybeUninit<T>; N]) -> [T; N] {
69    // SAFETY:
70    // * The caller guarantees that all elements of the array are initialized
71    // * `MaybeUninit<T>` and T are guaranteed to have the same layout
72    // * `MaybeUninit` does not drop, so there are no double-frees
73    // And thus the conversion is safe
74    unsafe { transpose(array).assume_init() }
75}
76
77/// [`MaybeUninit::transpose`]
78#[inline(always)]
79unsafe fn transpose<T, const N: usize>(array: [MaybeUninit<T>; N]) -> MaybeUninit<[T; N]> {
80    unsafe {
81        mem::transmute_copy::<[MaybeUninit<T>; N], MaybeUninit<[T; N]>>(&mem::ManuallyDrop::new(
82            &array,
83        ))
84    }
