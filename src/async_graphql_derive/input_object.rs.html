1use darling::ast::Data;
2use proc_macro::TokenStream;
3use quote::quote;
4use syn::{Error, ext::IdentExt};

6use crate::{
7    args::{self, RenameRuleExt, RenameTarget, TypeDirectiveLocation},
8    utils::{
9        GeneratorResult, gen_deprecation, gen_directive_calls, generate_default, get_crate_name,
10        get_rustdoc, visible_fn,
11    },
12};

14pub fn generate(object_args: &args::InputObject) -> GeneratorResult<TokenStream> {
15    let crate_name = get_crate_name(object_args.internal);
16    let (impl_generics, ty_generics, where_clause) = object_args.generics.split_for_impl();
17    let ident = &object_args.ident;
18    let inaccessible = object_args.inaccessible;
19    let tags = object_args
20        .tags
21        .iter()
22        .map(|tag| quote!(::std::string::ToString::to_string(#tag)))
23        .collect::<Vec<_>>();
24    let directives =
25        gen_directive_calls(&object_args.directives, TypeDirectiveLocation::InputObject);
26    let s = match &object_args.data {
27        Data::Struct(s) => s,
28        _ => {
29            return Err(
30                Error::new_spanned(ident, "InputObject can only be applied to an struct.").into(),
31            );
32        }
33    };
34
35    for field in &s.fields {
36        if field.ident.is_none() {
37            return Err(Error::new_spanned(ident, "All fields must be named.").into());
38        }
39    }
40
41    let gql_typename = if !object_args.name_type {
42        let name = object_args
43            .input_name
44            .clone()
45            .or_else(|| object_args.name.clone())
46            .unwrap_or_else(|| RenameTarget::Type.rename(ident.to_string()));
47
48        quote!(::std::borrow::Cow::Borrowed(#name))
49    } else {
50        quote!(<Self as #crate_name::TypeName>::type_name())
51    };
52
53    let desc = get_rustdoc(&object_args.attrs)?
54        .map(|s| quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#s)) })
55        .unwrap_or_else(|| quote! {::std::option::Option::None});
56
57    let mut get_fields = Vec::new();
58    let mut put_fields = Vec::new();
59    let mut fields = Vec::new();
60    let mut schema_fields = Vec::new();
61    let mut flatten_fields = Vec::new();
62    let mut federation_fields = Vec::new();
63
64    for field in &s.fields {
65        let ident = field.ident.as_ref().unwrap();
66        let ty = &field.ty;
67        let name = field.name.clone().unwrap_or_else(|| {
68            object_args
69                .rename_fields
70                .rename(ident.unraw().to_string(), RenameTarget::Field)
71        });
72        let inaccessible = field.inaccessible;
73        let tags = field
74            .tags
75            .iter()
76            .map(|tag| quote!(::std::string::ToString::to_string(#tag)))
77            .collect::<Vec<_>>();
78
79        let directive_invocations = gen_directive_calls(
80            &field.directives,
81            TypeDirectiveLocation::InputFieldDefinition,
82        );
83
84        if field.skip || field.skip_input {
85            get_fields.push(quote! {
86                let #ident: #ty = ::std::default::Default::default();
87            });
88            fields.push(ident);
89            continue;
90        }
91
92        federation_fields.push((ty, name.clone()));
93
94        let process_with = match field.process_with.as_ref() {
95            Some(fn_path) => quote! { #fn_path(&mut #ident); },
96            None => Default::default(),
97        };
98
99        let validators = field
100            .validator
101            .clone()
102            .unwrap_or_default()
103            .create_validators(
104                &crate_name,
105                quote!(&#ident),
106                Some(quote!(.map_err(#crate_name::InputValueError::propagate))),
107            )?;
108
109        if field.flatten {
110            flatten_fields.push((ident, ty));
111
112            schema_fields.push(quote! {
113                #crate_name::static_assertions_next::assert_impl_one!(#ty: #crate_name::InputObjectType);
114                <#ty as  #crate_name::InputType>::create_type_info(registry);
115                if let #crate_name::registry::MetaType::InputObject { input_fields, .. } =
116                    registry.create_fake_input_type::<#ty>() {
117                    fields.extend(input_fields);
118                }
119            });
120
121            get_fields.push(quote! {
122                #[allow(unused_mut)]
123                let mut #ident: #ty = #crate_name::InputType::parse(
124                    ::std::option::Option::Some(#crate_name::Value::Object(::std::clone::Clone::clone(&obj)))
125                ).map_err(#crate_name::InputValueError::propagate)?;
126                #process_with
127                #validators
128            });
129
130            fields.push(ident);
131
132            put_fields.push(quote! {
133                if let #crate_name::Value::Object(values) = #crate_name::InputType::to_value(&self.#ident) {
134                    map.extend(values);
135                }
136            });
137            continue;
138        }
139
140        let desc = get_rustdoc(&field.attrs)?
141            .map(|s| quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#s)) })
142            .unwrap_or_else(|| quote! {::std::option::Option::None});
143        let default = generate_default(&field.default, &field.default_with)?;
144        let schema_default = default
145            .as_ref()
146            .map(|value| {
147                quote! {
148                    ::std::option::Option::Some(::std::string::ToString::to_string(
149                        &<#ty as #crate_name::InputType>::to_value(&#value)
150                    ))
151                }
152            })
153            .unwrap_or_else(|| quote!(::std::option::Option::None));
154        let secret = field.secret;
155
156        if let Some(default) = default {
157            get_fields.push(quote! {
158                #[allow(non_snake_case)]
159                let #ident: #ty = {
160                    match obj.get(#name) {
161                        ::std::option::Option::Some(value) => {
162                            #[allow(unused_mut)]
163                            let mut #ident = #crate_name::InputType::parse(::std::option::Option::Some(::std::clone::Clone::clone(&value)))
164                                .map_err(#crate_name::InputValueError::propagate)?;
165                            #process_with
166                            #ident
167
168                        },
169                        ::std::option::Option::None => #default,
170                    }
171                };
172                #validators
173            });
174        } else {
175            get_fields.push(quote! {
176                #[allow(non_snake_case, unused_mut)]
177                let mut #ident: #ty = #crate_name::InputType::parse(obj.get(#name).cloned())
178                    .map_err(#crate_name::InputValueError::propagate)?;
179                #process_with
180                #validators
181            });
182        }
183
184        put_fields.push(quote! {
185            map.insert(
186                #crate_name::Name::new(#name),
187                #crate_name::InputType::to_value(&self.#ident)
188            );
189        });
190
191        fields.push(ident);
192
193        let visible = visible_fn(&field.visible);
194        let deprecation = gen_deprecation(&field.deprecation, &crate_name);
195
196        schema_fields.push(quote! {
197            fields.insert(::std::borrow::ToOwned::to_owned(#name), #crate_name::registry::MetaInputValue {
198                name: ::std::string::ToString::to_string(#name),
199                description: #desc,
200                ty: <#ty as #crate_name::InputType>::create_type_info(registry),
201                deprecation: #deprecation,
202                default_value: #schema_default,
203                visible: #visible,
204                inaccessible: #inaccessible,
205                tags: ::std::vec![ #(#tags),* ],
206                is_secret: #secret,
207                directive_invocations: ::std::vec![ #(#directive_invocations),* ],
208            });
209        })
210    }
211
212    if get_fields.is_empty() {
213        return Err(Error::new_spanned(
214            ident,
215            "A GraphQL Input Object type must define one or more input fields.",
216        )
217        .into());
218    }
219
220    let visible = visible_fn(&object_args.visible);
221
222    let get_federation_fields = {
223        let fields = federation_fields.into_iter().map(|(ty, name)| {
224            quote! {
225                if let ::std::option::Option::Some(fields) = <#ty as #crate_name::InputType>::federation_fields() {
226                    res.push(::std::format!("{} {}", #name, fields));
227                } else {
228                    res.push(::std::string::ToString::to_string(#name));
229                }
230            }
231        });
232        quote! {
233            let mut res = ::std::vec::Vec::new();
234            #(#fields)*
235            ::std::option::Option::Some(::std::format!("{{ {} }}", res.join(" ")))
236        }
237    };
238
239    let obj_validator = object_args
240        .validator
241        .as_ref()
242        .map(|expr| quote! { #crate_name::CustomValidator::check(&#expr, &obj)?; });
243
244    let expanded = if object_args.concretes.is_empty() {
245        quote! {
246            #[allow(clippy::all, clippy::pedantic)]
247            impl #crate_name::InputType for #ident {
248                type RawValueType = Self;
249
250                fn type_name() -> ::std::borrow::Cow<'static, ::std::primitive::str> {
251                    #gql_typename
252                }
253
254                fn create_type_info(registry: &mut #crate_name::registry::Registry) -> ::std::string::String {
255                    registry.create_input_type::<Self, _>(#crate_name::registry::MetaTypeId::InputObject, |registry| #crate_name::registry::MetaType::InputObject {
256                        name: ::std::borrow::Cow::into_owned(#gql_typename),
257                        description: #desc,
258                        input_fields: {
259                            let mut fields = #crate_name::indexmap::IndexMap::new();
260                            #(#schema_fields)*
261                            fields
262                        },
263                        visible: #visible,
264                        inaccessible: #inaccessible,
265                        tags: ::std::vec![ #(#tags),* ],
266                        rust_typename: ::std::option::Option::Some(::std::any::type_name::<Self>()),
267                        oneof: false,
268                        directive_invocations: ::std::vec![ #(#directives),* ],
269                    })
270                }
271
272                fn parse(value: ::std::option::Option<#crate_name::Value>) -> #crate_name::InputValueResult<Self> {
273                    if let ::std::option::Option::Some(#crate_name::Value::Object(obj)) = value {
274                        #(#get_fields)*
275                        let obj = Self { #(#fields),* };
276                        #obj_validator
277                        ::std::result::Result::Ok(obj)
278                    } else {
279                        ::std::result::Result::Err(#crate_name::InputValueError::expected_type(value.unwrap_or_default()))
280                    }
281                }
282
283                fn to_value(&self) -> #crate_name::Value {
284                    let mut map = #crate_name::indexmap::IndexMap::new();
285                    #(#put_fields)*
286                    #crate_name::Value::Object(map)
287                }
288
289                fn federation_fields() -> ::std::option::Option<::std::string::String> {
290                    #get_federation_fields
291                }
292
293                fn as_raw_value(&self) -> ::std::option::Option<&Self::RawValueType> {
294                    ::std::option::Option::Some(self)
295                }
296            }
297
298            impl #crate_name::InputObjectType for #ident {}
299        }
300    } else {
301        let mut code = Vec::new();
302
303        code.push(quote! {
304            #[allow(clippy::all, clippy::pedantic)]
305            impl #impl_generics #ident #ty_generics #where_clause {
306                fn __internal_create_type_info_input_object(registry: &mut #crate_name::registry::Registry, name: &str) -> ::std::string::String where Self: #crate_name::InputType {
307                    registry.create_input_type::<Self, _>(#crate_name::registry::MetaTypeId::InputObject, |registry| #crate_name::registry::MetaType::InputObject {
308                        name: ::std::borrow::ToOwned::to_owned(name),
309                        description: #desc,
310                        input_fields: {
311                            let mut fields = #crate_name::indexmap::IndexMap::new();
312                            #(#schema_fields)*
313                            fields
314                        },
315                        visible: #visible,
316                        inaccessible: #inaccessible,
317                        tags: ::std::vec![ #(#tags),* ],
318                        rust_typename: ::std::option::Option::Some(::std::any::type_name::<Self>()),
319                        oneof: false,
320                        directive_invocations: ::std::vec![ #(#directives),* ],
321                    })
322                }
323
324                fn __internal_parse(value: ::std::option::Option<#crate_name::Value>) -> #crate_name::InputValueResult<Self> where Self: #crate_name::InputType {
325                    if let ::std::option::Option::Some(#crate_name::Value::Object(obj)) = value {
326                        #(#get_fields)*
327                        let obj = Self { #(#fields),* };
328                        #obj_validator
329                        ::std::result::Result::Ok(obj)
330                    } else {
331                        ::std::result::Result::Err(#crate_name::InputValueError::expected_type(value.unwrap_or_default()))
332                    }
333                }
334
335                fn __internal_to_value(&self) -> #crate_name::Value where Self: #crate_name::InputType {
336                    let mut map = #crate_name::indexmap::IndexMap::new();
337                    #(#put_fields)*
338                    #crate_name::Value::Object(map)
339                }
340
341                fn __internal_federation_fields() -> ::std::option::Option<::std::string::String> where Self: #crate_name::InputType {
342                    #get_federation_fields
343                }
344            }
345        });
346
347        for concrete in &object_args.concretes {
348            let gql_typename = concrete.input_name.as_ref().unwrap_or(&concrete.name);
349            let params = &concrete.params.0;
350            let concrete_type = quote! { #ident<#(#params),*> };
351
352            let def_bounds = if !concrete.bounds.0.is_empty() {
353                let bounds = concrete.bounds.0.iter().map(|b| quote!(#b));
354                Some(quote!(<#(#bounds),*>))
355            } else {
356                None
357            };
358
359            let expanded = quote! {
360                #[allow(clippy::all, clippy::pedantic)]
361                impl #def_bounds #crate_name::InputType for #concrete_type {
362                    type RawValueType = Self;
363
364                    fn type_name() -> ::std::borrow::Cow<'static, ::std::primitive::str> {
365                        ::std::borrow::Cow::Borrowed(#gql_typename)
366                    }
367
368                    fn create_type_info(registry: &mut #crate_name::registry::Registry) -> ::std::string::String {
369                        Self::__internal_create_type_info_input_object(registry, #gql_typename)
370                    }
371
372                    fn parse(value: ::std::option::Option<#crate_name::Value>) -> #crate_name::InputValueResult<Self> {
373                        Self::__internal_parse(value)
374                    }
375
376                    fn to_value(&self) -> #crate_name::Value {
377                        self.__internal_to_value()
378                    }
379
380                    fn federation_fields() -> ::std::option::Option<::std::string::String> {
381                        Self::__internal_federation_fields()
382                    }
383
384                    fn as_raw_value(&self) -> ::std::option::Option<&Self::RawValueType> {
385                        ::std::option::Option::Some(self)
386                    }
387                }
388
389                impl #def_bounds #crate_name::InputObjectType for #concrete_type {}
390            };
391            code.push(expanded);
392        }
393        quote!(#(#code)*)
394    };
395
396    Ok(expanded.into())
397}
