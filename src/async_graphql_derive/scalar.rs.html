5use crate::{

13pub fn generate(
14    scalar_args: &args::Scalar,
15    item_impl: &mut ItemImpl,
16) -> GeneratorResult<TokenStream> {
17    let crate_name = get_crate_name(scalar_args.internal);
18    let boxed_trait = gen_boxed_trait(&crate_name);
19    let self_name = get_type_path_and_name(item_impl.self_ty.as_ref())?.1;
20    let gql_typename = if !scalar_args.name_type {
21        let name = scalar_args
22            .name
23            .clone()
24            .unwrap_or_else(|| RenameTarget::Type.rename(self_name.clone()));
25        quote!(::std::borrow::Cow::Borrowed(#name))
26    } else {
27        quote!(<Self as #crate_name::TypeName>::type_name())
28    };
29
30    let desc = if scalar_args.use_type_description {
31        quote! { ::std::option::Option::Some(::std::string::ToString::to_string(<Self as #crate_name::Description>::description())) }
32    } else {
33        get_rustdoc(&item_impl.attrs)?
34            .map(|s| quote!(::std::option::Option::Some(::std::string::ToString::to_string(#s))))
35            .unwrap_or_else(|| quote!(::std::option::Option::None))
36    };
37
38    let self_ty = &item_impl.self_ty;
39    let generic = &item_impl.generics;
40    let where_clause = &item_impl.generics.where_clause;
41    let visible = visible_fn(&scalar_args.visible);
42    let inaccessible = scalar_args.inaccessible;
43    let tags = scalar_args
44        .tags
45        .iter()
46        .map(|tag| quote!(::std::string::ToString::to_string(#tag)))
47        .collect::<Vec<_>>();
48    let requires_scopes = scalar_args
49        .requires_scopes
50        .iter()
51        .map(|scopes| quote!(::std::string::ToString::to_string(#scopes)))
52        .collect::<Vec<_>>();
53    let specified_by_url = match &scalar_args.specified_by_url {
54        Some(specified_by_url) => {
55            quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#specified_by_url)) }
56        }
57        None => quote! { ::std::option::Option::None },
58    };
59
60    let expanded = quote! {
61        #item_impl
62
63        #[allow(clippy::all, clippy::pedantic)]
64        impl #generic #crate_name::InputType for #self_ty #where_clause {
65            type RawValueType = Self;
66
67            fn type_name() -> ::std::borrow::Cow<'static, ::std::primitive::str> {
68                #gql_typename
69            }
70
71            fn create_type_info(registry: &mut #crate_name::registry::Registry) -> ::std::string::String {
72                registry.create_input_type::<#self_ty, _>(#crate_name::registry::MetaTypeId::Scalar, |_| #crate_name::registry::MetaType::Scalar {
73                    name: ::std::borrow::Cow::into_owned(#gql_typename),
74                    description: #desc,
75                    is_valid: ::std::option::Option::Some(::std::sync::Arc::new(|value| <#self_ty as #crate_name::ScalarType>::is_valid(value))),
76                    visible: #visible,
77                    inaccessible: #inaccessible,
78                    tags: ::std::vec![ #(#tags),* ],
79                    specified_by_url: #specified_by_url,
80                    directive_invocations: ::std::vec::Vec::new(),
81                    requires_scopes: ::std::vec![ #(#requires_scopes),* ],
82                })
83            }
84
85            fn parse(value: ::std::option::Option<#crate_name::Value>) -> #crate_name::InputValueResult<Self> {
86                <#self_ty as #crate_name::ScalarType>::parse(value.unwrap_or_default())
87            }
88
89            fn to_value(&self) -> #crate_name::Value {
90                <#self_ty as #crate_name::ScalarType>::to_value(self)
91            }
92
93            fn as_raw_value(&self) -> ::std::option::Option<&Self::RawValueType> {
94                ::std::option::Option::Some(self)
95            }
96        }
97
98        #[allow(clippy::all, clippy::pedantic)]
99        #boxed_trait
100        impl #generic #crate_name::OutputType for #self_ty #where_clause {
101            fn type_name() -> ::std::borrow::Cow<'static, ::std::primitive::str> {
102                #gql_typename
103            }
104
105            fn create_type_info(registry: &mut #crate_name::registry::Registry) -> ::std::string::String {
106                registry.create_output_type::<#self_ty, _>(#crate_name::registry::MetaTypeId::Scalar, |_| #crate_name::registry::MetaType::Scalar {
107                    name: ::std::borrow::Cow::into_owned(#gql_typename),
108                    description: #desc,
109                    is_valid: ::std::option::Option::Some(::std::sync::Arc::new(|value| <#self_ty as #crate_name::ScalarType>::is_valid(value))),
110                    visible: #visible,
111                    inaccessible: #inaccessible,
112                    tags: ::std::vec![ #(#tags),* ],
113                    specified_by_url: #specified_by_url,
114                    directive_invocations: ::std::vec::Vec::new(),
115                    requires_scopes: ::std::vec![ #(#requires_scopes),* ],
116                })
117            }
118
119            async fn resolve(
120                &self,
121                _: &#crate_name::ContextSelectionSet<'_>,
122                _field: &#crate_name::Positioned<#crate_name::parser::types::Field>
123            ) -> #crate_name::ServerResult<#crate_name::Value> {
124                ::std::result::Result::Ok(#crate_name::ScalarType::to_value(self))
125            }
126        }
127    };
128    Ok(expanded.into())
129}
