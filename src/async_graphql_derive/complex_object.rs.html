22pub fn generate(
23    object_args: &args::ComplexObject,
24    item_impl: &mut ItemImpl,
25) -> GeneratorResult<TokenStream> {
26    let crate_name = get_crate_name(object_args.internal);
27    let boxed_trait = gen_boxed_trait(&crate_name);
28    let (self_ty, _) = get_type_path_and_name(item_impl.self_ty.as_ref())?;
29    let generics = &item_impl.generics;
30    let where_clause = &item_impl.generics.where_clause;
31
32    let mut resolvers = Vec::new();
33    let mut schema_fields = Vec::new();
34
35    // Computation of the derived fields
36    let mut derived_impls = vec![];
37    for item in &mut item_impl.items {
38        if let ImplItem::Fn(method) = item {
39            let method_args: args::ComplexObjectField =
40                parse_graphql_attrs(&method.attrs)?.unwrap_or_default();
41
42            for derived in method_args.derived {
43                if derived.name.is_some() && derived.into.is_some() {
44                    let base_function_name = &method.sig.ident;
45                    let name = derived.name.unwrap();
46                    let with = derived.with;
47                    let into = Type::Verbatim(
48                        proc_macro2::TokenStream::from_str(&derived.into.unwrap()).unwrap(),
49                    );
50
51                    let mut new_impl = method.clone();
52                    new_impl.sig.ident = name;
53                    new_impl.sig.output =
54                        syn::parse2::<ReturnType>(quote! { -> #crate_name::Result<#into> })
55                            .expect("invalid result type");
56
57                    let should_create_context = new_impl
58                        .sig
59                        .inputs
60                        .iter()
61                        .nth(1)
62                        .map(|x| {
63                            if let FnArg::Typed(pat) = x {
64                                if let Type::Reference(TypeReference { elem, .. }) = &*pat.ty {
65                                    if let Type::Path(path) = elem.as_ref() {
66                                        return path.path.segments.last().unwrap().ident
67                                            != "Context";
68                                    }
69                                }
70                            };
71                            true
72                        })
73                        .unwrap_or(true);
74
75                    if should_create_context {
76                        let arg_ctx = syn::parse2::<FnArg>(quote! { ctx: &Context<'_> })
77                            .expect("invalid arg type");
78                        new_impl.sig.inputs.insert(1, arg_ctx);
79                    }
80
81                    let other_atts: Punctuated<Ident, Token![,]> = Punctuated::from_iter(
82                        new_impl
83                            .sig
84                            .inputs
85                            .iter()
86                            .filter_map(|x| match x {
87                                FnArg::Typed(pat) => match &*pat.pat {
88                                    Pat::Ident(ident) => Some(Ok(ident.ident.clone())),
89                                    _ => Some(Err(Error::new_spanned(
90                                        pat,
91                                        "Must be a simple argument",
92                                    ))),
93                                },
94                                FnArg::Receiver(_) => None,
95                            })
96                            .collect::<Result<Vec<Ident>, Error>>()?
97                            .into_iter(),
98                    );
99
100                    let new_block = match with {
101                        Some(with) => quote!({
102                            ::std::result::Result::Ok(#with(#self_ty::#base_function_name(&self, #other_atts).await?))
103                        }),
104                        None => quote!({
105                            {
106                                ::std::result::Result::Ok(#self_ty::#base_function_name(&self, #other_atts).await?.into())
107                            }
108                        }),
109                    };
110
111                    new_impl.block = syn::parse2::<Block>(new_block).expect("invalid block");
112
113                    derived_impls.push(ImplItem::Fn(new_impl));
114                }
115            }
116        }
117    }
118    item_impl.items.append(&mut derived_impls);
119
120    for item in &mut item_impl.items {
121        if let ImplItem::Fn(method) = item {
122            let method_args: args::ComplexObjectField =
123                parse_graphql_attrs(&method.attrs)?.unwrap_or_default();
124            if method_args.skip {
125                remove_graphql_attrs(&mut method.attrs);
126                continue;
127            }
128            let cfg_attrs = get_cfg_attrs(&method.attrs);
129
130            if method_args.flatten {
131                // Only used to inject the context placeholder if required.
132                extract_input_args::<args::Argument>(&crate_name, method)?;
133
134                let ty = match &method.sig.output {
135                    ReturnType::Type(_, ty) => OutputType::parse(ty)?,
136                    ReturnType::Default => {
137                        return Err(Error::new_spanned(
138                            &method.sig.output,
139                            "Flatten resolver must have a return type",
140                        )
141                        .into());
142                    }
143                };
144                let ty = ty.value_type();
145                let ident = &method.sig.ident;
146
147                schema_fields.push(quote! {
148                    #crate_name::static_assertions_next::assert_impl_one!(#ty: #crate_name::ObjectType);
149                    <#ty>::create_type_info(registry);
150                    if let #crate_name::registry::MetaType::Object { fields: obj_fields, .. } =
151                        registry.create_fake_output_type::<#ty>() {
152                        fields.extend(obj_fields);
153                    }
154                });
155
156                resolvers.push(quote! {
157                    #(#cfg_attrs)*
158                    if let ::std::option::Option::Some(value) = #crate_name::ContainerType::resolve_field(&self.#ident(ctx).await, ctx).await? {
159                        return ::std::result::Result::Ok(std::option::Option::Some(value));
160                    }
161                });
162
163                remove_graphql_attrs(&mut method.attrs);
164                continue;
165            }
166
167            let field_name = method_args.name.clone().unwrap_or_else(|| {
168                object_args
169                    .rename_fields
170                    .rename(method.sig.ident.unraw().to_string(), RenameTarget::Field)
171            });
172            let field_desc = get_rustdoc(&method.attrs)?
173                .map(|s| quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#s)) })
174                .unwrap_or_else(|| quote! {::std::option::Option::None});
175            let field_deprecation = gen_deprecation(&method_args.deprecation, &crate_name);
176            let external = method_args.external;
177            let shareable = method_args.shareable;
178            let directives = gen_directive_calls(
179                &method_args.directives,
180                TypeDirectiveLocation::FieldDefinition,
181            );
182            let override_from = match &method_args.override_from {
183                Some(from) => {
184                    quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#from)) }
185                }
186                None => quote! { ::std::option::Option::None },
187            };
188            let inaccessible = method_args.inaccessible;
189            let tags = method_args
190                .tags
191                .iter()
192                .map(|tag| quote!(::std::string::ToString::to_string(#tag)))
193                .collect::<Vec<_>>();
194            let requires_scopes = method_args
195                .requires_scopes
196                .iter()
197                .map(|scopes| quote!(::std::string::ToString::to_string(#scopes)))
198                .collect::<Vec<_>>();
199            let requires = match &method_args.requires {
200                Some(requires) => {
201                    quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#requires)) }
202                }
203                None => quote! { ::std::option::Option::None },
204            };
205            let provides = match &method_args.provides {
206                Some(provides) => {
207                    quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#provides)) }
208                }
209                None => quote! { ::std::option::Option::None },
210            };
211            let cache_control = {
212                let public = method_args.cache_control.is_public();
213                let max_age = if method_args.cache_control.no_cache {
214                    -1
215                } else {
216                    method_args.cache_control.max_age as i32
217                };
218                quote! {
219                    #crate_name::CacheControl {
220                        public: #public,
221                        max_age: #max_age,
222                    }
223                }
224            };
225
226            let args = extract_input_args::<args::Argument>(&crate_name, method)?;
227            let mut schema_args = Vec::new();
228            let mut use_params = Vec::new();
229            let mut get_params = Vec::new();
230
231            for (
232                ident,
233                ty,
234                args::Argument {
235                    name,
236                    desc,
237                    default,
238                    default_with,
239                    validator,
240                    process_with,
241                    visible,
242                    inaccessible,
243                    tags,
244                    secret,
245                    directives,
246                    deprecation,
247                    ..
248                },
249            ) in &args
250            {
251                let name = name.clone().unwrap_or_else(|| {
252                    object_args
253                        .rename_args
254                        .rename(ident.ident.unraw().to_string(), RenameTarget::Argument)
255                });
256                let desc = desc
257                    .as_ref()
258                    .map(|s| quote! {::std::option::Option::Some(::std::string::ToString::to_string(#s))})
259                    .unwrap_or_else(|| quote! {::std::option::Option::None});
260                let default = generate_default(default, default_with)?;
261                let schema_default = default
262                    .as_ref()
263                    .map(|value| {
264                        quote! {
265                            ::std::option::Option::Some(::std::string::ToString::to_string(
266                                &<#ty as #crate_name::InputType>::to_value(&#value)
267                            ))
268                        }
269                    })
270                    .unwrap_or_else(|| quote! {::std::option::Option::None});
271
272                let visible = visible_fn(visible);
273                let tags = tags
274                    .iter()
275                    .map(|tag| quote!(::std::string::ToString::to_string(#tag)))
276                    .collect::<Vec<_>>();
277                let directives =
278                    gen_directive_calls(directives, TypeDirectiveLocation::ArgumentDefinition);
279                let deprecation = gen_deprecation(deprecation, &crate_name);
280
281                schema_args.push(quote! {
282                        args.insert(::std::borrow::ToOwned::to_owned(#name), #crate_name::registry::MetaInputValue {
283                            name: ::std::string::ToString::to_string(#name),
284                            description: #desc,
285                            ty: <#ty as #crate_name::InputType>::create_type_info(registry),
286                            deprecation: #deprecation,
287                            default_value: #schema_default,
288                            visible: #visible,
289                            inaccessible: #inaccessible,
290                            tags: ::std::vec![ #(#tags),* ],
291                            is_secret: #secret,
292                            directive_invocations: ::std::vec![ #(#directives),* ],
293                        });
294                    });
295
296                let param_ident = &ident.ident;
297                use_params.push(quote! { #param_ident });
298
299                let default = match default {
300                    Some(default) => {
301                        quote! { ::std::option::Option::Some(|| -> #ty { #default }) }
302                    }
303                    None => quote! { ::std::option::Option::None },
304                };
305
306                let param_ident = &ident.ident;
307                let process_with = match process_with.as_ref() {
308                    Some(fn_path) => quote! { #fn_path(&mut #param_ident); },
309                    None => Default::default(),
310                };
311
312                let validators = validator.clone().unwrap_or_default().create_validators(
313                    &crate_name,
314                    quote!(&#ident),
315                    Some(quote!(.map_err(|err| err.into_server_error(__pos)))),
316                )?;
317
318                let mut non_mut_ident = ident.clone();
319                non_mut_ident.mutability = None;
320                get_params.push(quote! {
321                    #[allow(non_snake_case, unused_mut)]
322                    let (__pos, mut #non_mut_ident) = ctx.param_value::<#ty>(#name, #default)?;
323                    #process_with
324                    #validators
325                    #[allow(non_snake_case)]
326                    let #ident = #non_mut_ident;
327                });
328            }
329
330            let ty = match &method.sig.output {
331                ReturnType::Type(_, ty) => OutputType::parse(ty)?,
332                ReturnType::Default => {
333                    return Err(Error::new_spanned(
334                        &method.sig.output,
335                        "Resolver must have a return type",
336                    )
337                    .into());
338                }
339            };
340            let schema_ty = ty.value_type();
341            let visible = visible_fn(&method_args.visible);
342
343            let complexity = if let Some(complexity) = &method_args.complexity {
344                let (variables, expr) = parse_complexity_expr(complexity.clone())?;
345                let mut parse_args = Vec::new();
346                for variable in variables {
347                    if let Some((
348                        ident,
349                        ty,
350                        args::Argument {
351                            name,
352                            default,
353                            default_with,
354                            ..
355                        },
356                    )) = args
357                        .iter()
358                        .find(|(pat_ident, _, _)| pat_ident.ident == variable)
359                    {
360                        let default = match generate_default(default, default_with)? {
361                            Some(default) => {
362                                quote! { ::std::option::Option::Some(|| -> #ty { #default }) }
363                            }
364                            None => quote! { ::std::option::Option::None },
365                        };
366                        let name = name.clone().unwrap_or_else(|| {
367                            object_args
368                                .rename_args
369                                .rename(ident.ident.unraw().to_string(), RenameTarget::Argument)
370                        });
371                        parse_args.push(quote! {
372                            let #ident: #ty = __ctx.param_value(__variables_definition, __field, #name, #default)?;
373                        });
374                    }
375                }
376                quote! {
377                    Some(|__ctx, __variables_definition, __field, child_complexity| {
378                        #(#parse_args)*
379                        Ok(#expr)
380                    })
381                }
382            } else {
383                quote! { ::std::option::Option::None }
384            };
385
386            schema_fields.push(quote! {
387                #(#cfg_attrs)*
388                fields.push((#field_name.to_string(), #crate_name::registry::MetaField {
389                    name: ::std::borrow::ToOwned::to_owned(#field_name),
390                    description: #field_desc,
391                    args: {
392                        let mut args = #crate_name::indexmap::IndexMap::new();
393                        #(#schema_args)*
394                        args
395                    },
396                    ty: <#schema_ty as #crate_name::OutputType>::create_type_info(registry),
397                    deprecation: #field_deprecation,
398                    cache_control: #cache_control,
399                    external: #external,
400                    provides: #provides,
401                    requires: #requires,
402                    shareable: #shareable,
403                    inaccessible: #inaccessible,
404                    tags: ::std::vec![ #(#tags),* ],
405                    override_from: #override_from,
406                    visible: #visible,
407                    compute_complexity: #complexity,
408                    directive_invocations: ::std::vec![ #(#directives),* ],
409                    requires_scopes: ::std::vec![ #(#requires_scopes),* ],
410                }));
411            });
412
413            let field_ident = &method.sig.ident;
414            if let OutputType::Value(inner_ty) = &ty {
415                let block = &method.block;
416                let new_block = quote!({
417                    {
418                        ::std::result::Result::Ok(async move {
419                            let value:#inner_ty = #block;
420                            value
421                        }.await)
422                    }
423                });
424                method.block = syn::parse2::<Block>(new_block).expect("invalid block");
425                method.sig.output =
426                    syn::parse2::<ReturnType>(quote! { -> #crate_name::Result<#inner_ty> })
427                        .expect("invalid result type");
428            }
429
430            let resolve_obj = quote! {
431                {
432                    let res = self.#field_ident(ctx, #(#use_params),*).await;
433                    res.map_err(|err| ::std::convert::Into::<#crate_name::Error>::into(err).into_server_error(ctx.item.pos))
434                }
435            };
436
437            let guard_map_err = quote! {
438                .map_err(|err| err.into_server_error(ctx.item.pos))
439            };
440            let guard = match method_args.guard.as_ref().or(object_args.guard.as_ref()) {
441                Some(code) => Some(generate_guards(&crate_name, code, guard_map_err)?),
442                None => None,
443            };
444
445            resolvers.push(quote! {
446                #(#cfg_attrs)*
447                if ctx.item.node.name.node == #field_name {
448                    let f = async move {
449                        #(#get_params)*
450                        #guard
451                        #resolve_obj
452                    };
453                    let obj = f.await.map_err(|err| ctx.set_error_path(err))?;
454                    let ctx_obj = ctx.with_selection_set(&ctx.item.node.selection_set);
455                    return #crate_name::OutputType::resolve(&obj, &ctx_obj, ctx.item).await.map(::std::option::Option::Some);
456                }
457            });
458
459            remove_graphql_attrs(&mut method.attrs);
460        }
461    }
462
463    let expanded = quote! {
464        #item_impl
465
466        #[allow(clippy::all, clippy::pedantic)]
467        #boxed_trait
468        impl #generics #crate_name::ComplexObject for #self_ty #where_clause {
469            fn fields(registry: &mut #crate_name::registry::Registry) -> ::std::vec::Vec<(::std::string::String, #crate_name::registry::MetaField)> {
470                let mut fields = ::std::vec::Vec::new();
471                #(#schema_fields)*
472                fields
473            }
474
475            async fn resolve_field(&self, ctx: &#crate_name::Context<'_>) -> #crate_name::ServerResult<::std::option::Option<#crate_name::Value>> {
476                #(#resolvers)*
477                ::std::result::Result::Ok(::std::option::Option::None)
478            }
479        }
480    };
481
482    Ok(expanded.into())
