1use darling::ast::Data;
2use proc_macro::TokenStream;
3use proc_macro2::Span;
4use quote::quote;
5use syn::{Error, LitInt};
6
7use crate::{
8    args::{self, RenameTarget},
9    utils::{GeneratorResult, get_crate_name, get_rustdoc, visible_fn},
10};
11
12pub fn generate(object_args: &args::MergedSubscription) -> GeneratorResult<TokenStream> {
13    let crate_name = get_crate_name(object_args.internal);
14    let ident = &object_args.ident;
15    let (impl_generics, ty_generics, where_clause) = object_args.generics.split_for_impl();
16    let extends = object_args.extends;
17    let gql_typename = if !object_args.name_type {
18        let name = object_args
19            .name
20            .clone()
21            .unwrap_or_else(|| RenameTarget::Type.rename(ident.to_string()));
22        quote!(::std::borrow::Cow::Borrowed(#name))
23    } else {
24        quote!(<Self as #crate_name::TypeName>::type_name())
25    };
26
27    let desc = get_rustdoc(&object_args.attrs)?
28        .map(|s| quote! { ::std::option::Option::Some(::std::string::ToString::to_string(#s)) })
29        .unwrap_or_else(|| quote! {::std::option::Option::None});
30
31    let s = match &object_args.data {
32        Data::Struct(e) => e,
33        _ => {
34            return Err(Error::new_spanned(
35                ident,
36                "MergedSubscription can only be applied to an struct.",
37            )
38            .into());
39        }
40    };
41
42    let types: Vec<_> = s.fields.iter().map(|field| &field.ty).collect();
43
44    let create_field_stream: proc_macro2::TokenStream = (0..types.len())
45        .map(|i| {
46            let n = LitInt::new(&i.to_string(), Span::call_site());
47            quote!(.or_else(|| #crate_name::SubscriptionType::create_field_stream(&self.#n, ctx)))
48        })
49        .collect();
50
51    let merged_type = types.iter().fold(
52        quote!(#crate_name::MergedObjectTail),
53        |obj, ty| quote!(#crate_name::MergedObject::<#ty, #obj>),
54    );
55
56    let visible = visible_fn(&object_args.visible);
57    let expanded = quote! {
58        #[allow(clippy::all, clippy::pedantic)]
59        impl #impl_generics #crate_name::SubscriptionType for #ident #ty_generics #where_clause {
60            fn type_name() -> ::std::borrow::Cow<'static, ::std::primitive::str> {
61                #gql_typename
62            }
63
64            fn create_type_info(registry: &mut #crate_name::registry::Registry) -> ::std::string::String {
65                registry.create_subscription_type::<Self, _>(|registry| {
66                    let mut fields = ::std::default::Default::default();
67
68                    if let #crate_name::registry::MetaType::Object {
69                        fields: obj_fields,
70                        ..
71                    } = registry.create_fake_subscription_type::<#merged_type>() {
72                        fields = obj_fields;
73                    }
74
75                    #crate_name::registry::MetaType::Object {
76                        name: ::std::borrow::Cow::into_owned(#gql_typename),
77                        description: #desc,
78                        fields,
79                        cache_control: ::std::default::Default::default(),
80                        extends: #extends,
81                        keys: ::std::option::Option::None,
82                        visible: #visible,
83                        shareable: false,
84                        resolvable: true,
85                        inaccessible: false,
86                        interface_object: false,
87                        tags: ::std::default::Default::default(),
88                        is_subscription: true,
89                        rust_typename: ::std::option::Option::Some(::std::any::type_name::<Self>()),
90                        directive_invocations: ::std::default::Default::default(),
91                        requires_scopes: ::std::default::Default::default(),
92                    }
93                })
94            }
95
96            fn create_field_stream<'__life>(
97                &'__life self,
98                ctx: &'__life #crate_name::Context<'__life>
99            ) -> ::std::option::Option<::std::pin::Pin<::std::boxed::Box<dyn #crate_name::futures_util::stream::Stream<Item = #crate_name::Response> + ::std::marker::Send + '__life>>> {
100                ::std::option::Option::None #create_field_stream
101            }
102        }
103    };
104    Ok(expanded.into())
105}
