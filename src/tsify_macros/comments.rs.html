6/// Extract the documentation comments from a Vec of attributes
7pub fn extract_doc_comments(attrs: &[syn::Attribute]) -> Vec<String> {
8    attrs
9        .iter()
10        .filter_map(|a| {
11            // if the path segments include an ident of "doc" we know this
12            // this is a doc comment
13            if a.path().segments.iter().any(|s| s.ident == "doc") {
14                Some(a.to_token_stream().into_iter().filter_map(|t| match t {
15                    TokenTree::Group(group) => {
16                        // this will return the inner tokens of the group
17                        // which will be the doc comments
18                        Some(
19                            group
20                                .stream()
21                                .into_iter()
22                                .filter_map(|t| match t {
23                                    TokenTree::Literal(lit) => {
24                                        // this will always return the quoted string, we deal with
25                                        // that in the cli when we read in the comments
26                                        Some(lit.to_string())
27                                    }
28                                    _ => None,
29                                })
30                                .collect::<Vec<_>>()
31                                .join(""),
32                        )
33                    }
34                    _ => None,
35                }))
36            } else {
37                None
38            }
39        })
40        //Fold up the [[String]] iter we created into Vec<String>
41        .fold(vec![], |mut acc, a| {
42            acc.extend(a);
43            acc
44        })
45}

47/// Output extracted doc comments as Typescript doc comments.
48pub fn write_doc_comments(
49    f: &mut std::fmt::Formatter<'_>,
50    comments: &[String],
51) -> Result<(), std::fmt::Error> {
52    if comments.is_empty() {
53        return Ok(());
54    }
55
56    let comment = comments
57        .iter()
58        .map(|line| format!(" *{}\n", line.trim_matches('"')))
59        .collect::<Vec<_>>()
60        .join("");
61
62    write!(f, "{}", format_args!("/**\n{} */\n", comment))
63}

65/// Remove all comments from a `TsType::TypeLit`
66pub fn clean_comments(typ: &mut TsType) {
67    if let TsType::TypeLit(ref mut lit) = typ {
68        lit.members.iter_mut().for_each(|elem| {
69            elem.comments = vec![];
70            // Recurse
71            clean_comments(&mut elem.type_ann);
72        });
73    }
74}
