3/// Attributes that can be applied to a type decorated with `#[derive(Tsify)]`.
4/// E.g., through `#[tsify(into_wasm_abi)]`.
5#[derive(Debug, Default)]
6pub struct TsifyContainerAttrs {
7    pub type_override: Option<String>,
8    pub type_params: Option<Vec<String>>,
9    /// Implement `IntoWasmAbi` for the type.
10    pub into_wasm_abi: bool,
11    /// Implement `FromWasmAbi` for the type.
12    pub from_wasm_abi: bool,
13    /// Whether the type should be wrapped in a Typescript namespace.
14    pub namespace: bool,
15    /// Information about how the type should be serialized.
16    pub ty_config: TypeGenerationConfig,
17}
18
19/// Configuration affecting how Typescript types are generated.
20#[derive(Debug, Default)]
21pub struct TypeGenerationConfig {
22    /// Universal prefix for generated types
23    pub type_prefix: Option<String>,
24    /// Universal suffix for generated types
25    pub type_suffix: Option<String>,
26    /// Whether missing fields should be represented as null in Typescript
27    pub missing_as_null: bool,
28    /// Whether a hashmap should be represented as an object in Typescript
29    pub hashmap_as_object: bool,
30    /// Whether large number types should be represented as BigInts in Typescript
31    pub large_number_types_as_bigints: bool,
32}
33
34impl TypeGenerationConfig {
35    /// Format a type `name` adding a prefix and suffix if they are set.
36    pub fn format_name(&self, name: String) -> String {
37        let prefix = self.type_prefix.as_ref().map_or("", String::as_str);
38        let suffix = self.type_suffix.as_ref().map_or("", String::as_str);
39        format!("{}{}{}", prefix, name, suffix)
40    }
41}

43impl TsifyContainerAttrs {
44    pub fn from_derive_input(input: &syn::DeriveInput) -> syn::Result<Self> {
45        let mut attrs = Self {
46            type_override: None,
47            type_params: None,
48            into_wasm_abi: false,
49            from_wasm_abi: false,
50            namespace: false,
51            ty_config: TypeGenerationConfig::default(),
52        };
53
54        for attr in &input.attrs {
55            if !attr.path().is_ident("tsify") {
56                continue;
57            }
58
59            attr.parse_nested_meta(|meta| {
60                if meta.path.is_ident("type") {
61                    if attrs.type_override.is_some() {
62                        return Err(meta.error("duplicate attribute"));
63                    }
64                    let lit = meta.value()?.parse::<syn::LitStr>()?;
65                    attrs.type_override = Some(lit.value());
66                    return Ok(());
67                }
68
69                if meta.path.is_ident("type_params") {
70                    if attrs.type_params.is_some() {
71                        return Err(meta.error("duplicate attribute"));
72                    }
73                    let lit = meta.value()?.parse::<syn::LitStr>()?;
74                    attrs.type_params = Some(lit.value().split(',').map(|s| s.trim().to_string()).collect());
75                    return Ok(());
76                }
77
78                if meta.path.is_ident("into_wasm_abi") {
79                    if attrs.into_wasm_abi {
80                        return Err(meta.error("duplicate attribute"));
81                    }
82                    attrs.into_wasm_abi = true;
83                    return Ok(());
84                }
85
86                if meta.path.is_ident("from_wasm_abi") {
87                    if attrs.from_wasm_abi {
88                        return Err(meta.error("duplicate attribute"));
89                    }
90                    attrs.from_wasm_abi = true;
91                    return Ok(());
92                }
93
94                if meta.path.is_ident("namespace") {
95                    if !matches!(input.data, syn::Data::Enum(_)) {
96                        return Err(meta.error("#[tsify(namespace)] can only be used on enums"));
97                    }
98                    if attrs.namespace {
99                        return Err(meta.error("duplicate attribute"));
100                    }
101                    attrs.namespace = true;
102                    return Ok(());
103                }
104
105                if meta.path.is_ident("type_prefix") {
106                    if attrs.ty_config.type_prefix.is_some() {
107                        return Err(meta.error("duplicate attribute"));
108                    }
109                    let lit: syn::LitStr = meta.value()?.parse()?;
110                    attrs.ty_config.type_prefix = Some(lit.value());
111                    return Ok(());
112                }
113
114                if meta.path.is_ident("type_suffix") {
115                    if attrs.ty_config.type_suffix.is_some() {
116                        return Err(meta.error("duplicate attribute"));
117                    }
118                    let lit: syn::LitStr = meta.value()?.parse()?;
119                    attrs.ty_config.type_suffix = Some(lit.value());
120                    return Ok(());
121                }
122
123                if meta.path.is_ident("missing_as_null") {
124                    if attrs.ty_config.missing_as_null {
125                        return Err(meta.error("duplicate attribute"));
126                    }
127                    if cfg!(not(feature = "js")) {
128                        return Err(meta.error(
129                            "#[tsify(missing_as_null)] requires the `js` feature",
130                        ));
131                    }
132                    attrs.ty_config.missing_as_null = true;
133                    return Ok(());
134                }
135
136                if meta.path.is_ident("hashmap_as_object") {
137                    if attrs.ty_config.hashmap_as_object {
138                        return Err(meta.error("duplicate attribute"));
139                    }
140                    if cfg!(not(feature = "js")) {
141                        return Err(meta.error(
142                            "#[tsify(hashmap_as_object)] requires the `js` feature",
143                        ));
144                    }
145                    attrs.ty_config.hashmap_as_object = true;
146                    return Ok(());
147                }
148
149                if meta.path.is_ident("large_number_types_as_bigints") {
150                    if attrs.ty_config.large_number_types_as_bigints {
151                        return Err(meta.error("duplicate attribute"));
152                    }
153                    if cfg!(not(feature = "js")) {
154                        return Err(meta.error(
155                            "#[tsify(large_number_types_as_bigints)] requires the `js` feature",
156                        ));
157                    }
158                    attrs.ty_config.large_number_types_as_bigints = true;
159                    return Ok(());
160                }
161
162                Err(meta.error("unsupported tsify attribute, expected one of `type`, `type_params`, `into_wasm_abi`, `from_wasm_abi`, `namespace`, `type_prefix`, `type_suffix`, `missing_as_null`, `hashmap_as_object`, `large_number_types_as_bigints`"))
163            })?;
164        }
165
166        Ok(attrs)
167    }
168}

170#[derive(Debug, Default)]
171pub struct TsifyFieldAttrs {
172    pub type_override: Option<String>,
173    pub type_params: Option<Vec<String>>,
174    pub optional: bool,
175}
176
177impl TsifyFieldAttrs {
178    pub fn from_serde_field(field: &Field) -> syn::Result<Self> {
179        let mut attrs = Self {
180            type_override: None,
181            type_params: None,
182            optional: false,
183        };
184
185        for attr in &field.original.attrs {
186            if !attr.path().is_ident("tsify") {
187                continue;
188            }
189
190            attr.parse_nested_meta(|meta| {
191                if meta.path.is_ident("type") {
192                    if attrs.type_override.is_some() {
193                        return Err(meta.error("duplicate attribute"));
194                    }
195                    let lit = meta.value()?.parse::<syn::LitStr>()?;
196                    attrs.type_override = Some(lit.value());
197                    return Ok(());
198                }
199
200                if meta.path.is_ident("type_params") {
201                    if attrs.type_params.is_some() {
202                        return Err(meta.error("duplicate attribute"));
203                    }
204                    let lit = meta.value()?.parse::<syn::LitStr>()?;
205                    attrs.type_params = Some(
206                        lit.value()
207                            .split(',')
208                            .map(|s| s.trim().to_string())
209                            .collect(),
210                    );
211                    return Ok(());
212                }
213
214                if meta.path.is_ident("optional") {
215                    if attrs.optional {
216                        return Err(meta.error("duplicate attribute"));
217                    }
218                    attrs.optional = true;
219                    return Ok(());
220                }
221
222                Err(meta.error("unsupported tsify attribute, expected one of `type`, `type_params` or `optional`"))
223            })?;
224        }
225
226        if let Some(expr) = field.attrs.skip_serializing_if() {
227            let path = expr
228                .path
229                .segments
230                .iter()
231                .map(|segment| segment.ident.to_string())
232                .collect::<Vec<_>>()
233                .join("::");
234
235            attrs.optional |= &path == "Option::is_none";
236        }
237
238        Ok(attrs)
239    }
