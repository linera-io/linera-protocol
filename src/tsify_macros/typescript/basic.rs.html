28#[derive(Debug, Clone, PartialEq, Eq)]
29pub struct TsTypeElement {
30    pub key: String,
31    pub type_ann: TsType,
32    pub optional: bool,
33    pub comments: Vec<String>,
34}
35
36impl TsTypeElement {
37    pub fn to_string_with_indent(&self, indent: usize) -> String {
38        let out = self.to_string();
39        let indent_str = " ".repeat(indent);
40        out.split('\n')
41            .map(|line| format!("{}{}", indent_str, line))
42            .collect::<Vec<_>>()
43            .join("\n")
44    }
45}
46
47impl Display for TsTypeElement {
48    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        let key = &self.key;
50        let type_ann = &self.type_ann;
51
52        let optional_ann = if self.optional { "?" } else { "" };
53
54        write_doc_comments(f, &self.comments)?;
55
56        if is_js_ident(key) {
57            write!(f, "{key}{optional_ann}: {type_ann}")
58        } else {
59            write!(f, "\"{key}\"{optional_ann}: {type_ann}")
60        }
61    }
62}

76#[derive(Debug, Clone, PartialEq, Eq)]
77pub struct TsTypeLit {
78    pub members: Vec<TsTypeElement>,
79}
80
81impl TsTypeLit {
82    pub fn get_mut(&mut self, key: &String) -> Option<&mut TsTypeElement> {
83        self.members.iter_mut().find(|member| &member.key == key)
84    }
85
86    pub fn and(self, other: Self) -> Self {
87        let init = TsTypeLit { members: vec![] };
88
89        self.members
90            .into_iter()
91            .chain(other.members)
92            .fold(init, |mut acc, m| {
93                if let Some(acc_m) = acc.get_mut(&m.key) {
94                    let mut tmp = TsType::NULL;
95                    std::mem::swap(&mut acc_m.type_ann, &mut tmp);
96                    acc_m.type_ann = tmp.and(m.type_ann);
97                } else {
98                    acc.members.push(m)
99                }
100
101                acc
102            })
103    }
104}

112impl Display for TsTypeLit {
113    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
114        let members = self
115            .members
116            .iter()
117            .map(|elem| elem.to_string())
118            .collect::<Vec<_>>()
119            .join("; ");
120
121        if members.is_empty() {
122            write!(f, "{{}}")
123        } else {
124            write!(f, "{{ {members} }}")
125        }
126    }

129#[derive(Debug, Clone, PartialEq, Eq)]
130pub enum NullType {
131    Null,
132    Undefined,
133}
134
135impl NullType {
136    pub const fn new(config: &TypeGenerationConfig) -> Self {
137        if cfg!(feature = "js") && !config.missing_as_null {
138            Self::Undefined
139        } else {
140            Self::Null
141        }
142    }
143
144    pub const fn to_type(&self) -> TsType {
145        match self {
146            Self::Null => TsType::NULL,
147            Self::Undefined => TsType::UNDEFINED,
148        }
149    }
150}

152fn is_js_ident(string: &str) -> bool {
153    !string.is_empty()
154        && !string.starts_with(|c: char| c.is_ascii_digit())
155        && !string.contains(|c: char| !c.is_ascii_alphanumeric() && c != '_' && c != '$')
156}
