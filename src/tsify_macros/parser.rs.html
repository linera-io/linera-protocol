16enum ParsedFields {
17    Named(Vec<TsTypeElement>, Vec<TsType>),
18    Unnamed(Vec<TsType>),
19    Transparent(TsType),
20}
21
22impl From<ParsedFields> for TsType {
23    fn from(fields: ParsedFields) -> Self {
24        match fields {
25            ParsedFields::Named(members, extends) => {
26                let type_lit = TsType::from(TsTypeLit { members });
27
28                if extends.is_empty() {
29                    type_lit
30                } else {
31                    type_lit.and(TsType::Intersection(extends))
32                }
33            }
34            ParsedFields::Unnamed(elems) => TsType::Tuple(elems),
35            ParsedFields::Transparent(ty) => ty,
36        }
37    }
38}

40enum FieldsStyle {
41    Named,
42    Unnamed,
43}

45#[derive(Clone)]
46pub struct Parser<'a> {
47    pub container: &'a Container<'a>,
48}
49
50impl<'a> Parser<'a> {
51    pub fn new(container: &'a Container<'a>) -> Self {
52        Self { container }
53    }
54
55    pub fn parse(&self) -> Decl {
56        if let Some(decl) = &self.container.attrs.type_override {
57            self.create_type_alias_decl(TsType::Override {
58                type_override: decl.to_string(),
59                type_params: self
60                    .container
61                    .generics()
62                    .type_params()
63                    .map(|p| p.ident.to_string())
64                    .collect(),
65            })
66        } else {
67            match self.container.serde_data() {
68                Data::Struct(style, ref fields) => self.parse_struct(*style, fields),
69                Data::Enum(ref variants) => self.parse_enum(variants),
70            }
71        }
72    }
73
74    fn create_relevant_type_params(&self, type_ref_names: HashSet<&String>) -> Vec<String> {
75        self.container
76            .generics()
77            .type_params()
78            .map(|p| p.ident.to_string())
79            .filter(|t| type_ref_names.contains(t))
80            .collect()
81    }
82
83    fn create_type_alias_decl(&self, type_ann: TsType) -> Decl {
84        Decl::TsTypeAlias(TsTypeAliasDecl {
85            id: self.container.ident_str(),
86            export: true,
87            type_params: self
88                .container
89                .attrs
90                .type_params
91                .as_ref()
92                .cloned()
93                .unwrap_or_else(|| self.create_relevant_type_params(type_ann.type_ref_names())),
94            type_ann,
95            comments: extract_doc_comments(&self.container.serde_container.original.attrs),
96        })
97    }
98
99    fn create_decl(&self, members: Vec<TsTypeElement>, extends: Vec<TsType>) -> Decl {
100        // An interface can only extend an identifier/qualified-name with optional type arguments.
101        if extends.iter().all(|ty| ty.is_ref()) {
102            let mut type_ref_names: HashSet<&String> = HashSet::new();
103            members.iter().for_each(|member| {
104                type_ref_names.extend(member.type_ann.type_ref_names());
105            });
106            extends.iter().for_each(|ty| {
107                type_ref_names.extend(ty.type_ref_names());
108            });
109
110            let type_params = self
111                .container
112                .attrs
113                .type_params
114                .as_ref()
115                .cloned()
116                .unwrap_or_else(|| self.create_relevant_type_params(type_ref_names));
117
118            Decl::TsInterface(TsInterfaceDecl {
119                id: self.container.ident_str(),
120                type_params,
121                extends,
122                body: members,
123                comments: extract_doc_comments(&self.container.serde_container.original.attrs),
124            })
125        } else {
126            let extra = TsType::Intersection(
127                extends
128                    .into_iter()
129                    .map(|ty| match ty {
130                        TsType::Option(ty, _) => TsType::Union(vec![*ty, TsType::empty_type_lit()]),
131                        _ => ty,
132                    })
133                    .collect(),
134            );
135            let type_ann = TsType::TypeLit(TsTypeLit { members }).and(extra);
136
137            self.create_type_alias_decl(type_ann)
138        }
139    }

141    fn parse_struct(&self, style: Style, fields: &[Field]) -> Decl {
142        let parsed_fields = self.parse_fields(style, fields);
143        let tag_type = self.container.serde_attrs().tag();
144
145        match (tag_type, parsed_fields) {
146            (TagType::Internal { tag }, ParsedFields::Named(members, extends)) => {
147                let name = self.container.name();
148
149                let tag_field = TsTypeElement {
150                    key: tag.clone(),
151                    type_ann: TsType::Lit(name),
152                    optional: false,
153                    comments: vec![],
154                };
155
156                let mut vec = Vec::with_capacity(members.len() + 1);
157                vec.push(tag_field);
158                vec.extend(members);
159
160                self.create_decl(vec, extends)
161            }
162            (_, ParsedFields::Named(members, extends)) => self.create_decl(members, extends),
163            (_, parsed_fields) => self.create_type_alias_decl(parsed_fields.into()),
164        }
165    }
166
167    fn parse_fields(&self, style: Style, fields: &[Field]) -> ParsedFields {
168        let style = match style {
169            Style::Struct => FieldsStyle::Named,
170            Style::Newtype => return ParsedFields::Transparent(self.parse_field(&fields[0]).0),
171            Style::Tuple => FieldsStyle::Unnamed,
172            Style::Unit => {
173                return ParsedFields::Transparent(TsType::nullish(&self.container.attrs.ty_config))
174            }
175        };
176
177        let fields = fields
178            .iter()
179            .filter(|field| {
180                !field.attrs.skip_serializing()
181                    && !field.attrs.skip_deserializing()
182                    && !is_phantom(field.ty)
183            })
184            .collect::<Vec<_>>();
185
186        if fields.len() == 1 && self.container.transparent() {
187            return ParsedFields::Transparent(self.parse_field(fields[0]).0);
188        }
189
190        match style {
191            FieldsStyle::Named => {
192                let (members, flatten_fields) = self.parse_named_fields(fields);
193
194                ParsedFields::Named(members, flatten_fields)
195            }
196            FieldsStyle::Unnamed => {
197                let elems = fields
198                    .into_iter()
199                    .map(|field| self.parse_field(field).0)
200                    .collect();
201
202                ParsedFields::Unnamed(elems)
203            }
204        }
205    }
206
207    fn parse_field(&self, field: &Field) -> (TsType, Option<TsifyFieldAttrs>) {
208        let ts_attrs = match TsifyFieldAttrs::from_serde_field(field) {
209            Ok(attrs) => attrs,
210            Err(err) => {
211                self.container.syn_error(err);
212                return (TsType::NEVER, None);
213            }
214        };
215
216        let type_ann = TsType::from_syn_type(&self.container.attrs.ty_config, field.ty);
217
218        if let Some(t) = &ts_attrs.type_override {
219            let type_params = if let Some(params) = &ts_attrs.type_params {
220                params.clone()
221            } else {
222                let type_ref_names = type_ann.type_ref_names();
223                self.create_relevant_type_params(type_ref_names)
224            };
225            (
226                TsType::Override {
227                    type_override: t.clone(),
228                    type_params,
229                },
230                Some(ts_attrs),
231            )
232        } else {
233            (type_ann, Some(ts_attrs))
234        }
235    }
236
237    fn parse_named_fields(&self, fields: Vec<&Field>) -> (Vec<TsTypeElement>, Vec<TsType>) {
238        let (flatten_fields, members): (Vec<_>, Vec<_>) =
239            fields.into_iter().partition(|field| field.attrs.flatten());
240
241        let members = members
242            .into_iter()
243            .map(|field| {
244                let key = field.attrs.name().serialize_name().to_owned();
245                let (type_ann, field_attrs) = self.parse_field(field);
246
247                let optional = field_attrs.is_some_and(|attrs| attrs.optional);
248                let default_is_none = self.container.serde_attrs().default().is_none()
249                    && field.attrs.default().is_none();
250
251                let type_ann = if optional {
252                    match type_ann {
253                        TsType::Option(t, _) => *t,
254                        _ => type_ann,
255                    }
256                } else {
257                    type_ann
258                };
259
260                let comments = extract_doc_comments(&field.original.attrs);
261
262                TsTypeElement {
263                    key,
264                    type_ann,
265                    optional: optional || !default_is_none,
266                    comments,
267                }
268            })
269            .collect();
270
271        let flatten_fields = flatten_fields
272            .into_iter()
273            .map(|field| self.parse_field(field).0)
274            .collect();
275
276        (members, flatten_fields)
277    }

279    fn parse_enum(&self, variants: &[Variant]) -> Decl {
280        let members = variants
281            .iter()
282            .filter(|v| !v.attrs.skip_serializing() && !v.attrs.skip_deserializing())
283            .map(|variant| {
284                let decl = self.create_type_alias_decl(self.parse_variant(variant));
285                if let Decl::TsTypeAlias(mut type_alias) = decl {
286                    variant
287                        .attrs
288                        .name()
289                        .serialize_name()
290                        .clone_into(&mut type_alias.id);
291                    type_alias.comments = extract_doc_comments(&variant.original.attrs);
292
293                    type_alias
294                } else {
295                    panic!();
296                }
297            })
298            .collect::<Vec<_>>();
299
300        let type_ref_names = members
301            .iter()
302            .flat_map(|type_alias| type_alias.type_ann.type_ref_names())
303            .collect::<HashSet<_>>();
304
305        let relevant_type_params = self.create_relevant_type_params(type_ref_names);
306
307        Decl::TsEnum(TsEnumDecl {
308            id: self.container.ident_str(),
309            type_params: relevant_type_params,
310            members,
311            namespace: self.container.attrs.namespace,
312            comments: extract_doc_comments(&self.container.serde_container.original.attrs),
313        })
314    }
315
316    fn parse_variant(&self, variant: &Variant) -> TsType {
317        let tag_type = self.container.serde_attrs().tag();
318        let name = variant.attrs.name().serialize_name().to_owned();
319        // Checks for Newtype with a skip attribute and treats it as a Unit
320        let style = if matches!(variant.style, Style::Newtype)
321            && (variant.fields[0].attrs.skip_serializing()
322                || variant.fields[0].attrs.skip_deserializing()
323                || is_phantom(variant.fields[0].ty))
324        {
325            Style::Unit
326        } else {
327            variant.style
328        };
329        let type_ann: TsType = self.parse_fields(style, &variant.fields).into();
330        type_ann.with_tag_type(&self.container.attrs.ty_config, name, style, tag_type)
331    }
332}

334fn is_phantom(ty: &syn::Type) -> bool {
335    if let syn::Type::Path(syn::TypePath { path, .. }) = ty {
336        path.segments
337            .last()
338            .is_some_and(|path| path.ident == "PhantomData")
339    } else {
340        false
341    }
