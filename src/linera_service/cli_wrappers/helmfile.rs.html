4use std::path::Path;
5
6use anyhow::Result;
7use fs_extra::dir::CopyOptions;
8use linera_base::command::CommandExt;
9use tokio::process::Command;
10
11pub const DEFAULT_BLOCK_EXPORTER_PORT: u16 = 8882;
12
13pub struct HelmFile;
14
15impl HelmFile {
16    #[expect(clippy::too_many_arguments)]
17    pub async fn sync(
18        server_config_id: usize,
19        github_root: &Path,
20        num_proxies: usize,
21        num_shards: usize,
22        cluster_id: u32,
23        docker_image_name: String,
24        with_block_exporter: bool,
25        num_block_exporters: usize,
26        indexer_image_name: String,
27        explorer_image_name: String,
28        dual_store: bool,
29    ) -> Result<()> {
30        let chart_dir = format!("{}/kubernetes/linera-validator", github_root.display());
31
32        let temp_dir = tempfile::tempdir()?;
33        fs_extra::copy_items(&[&chart_dir], temp_dir.path(), &CopyOptions::new())?;
34
35        let mut command = Command::new("helmfile");
36        command.current_dir(temp_dir.path().join("linera-validator"));
37
38        if dual_store {
39            command.env(
40                "LINERA_HELMFILE_SET_STORAGE",
41                "dualrocksdbscylladb:/linera.db:spawn_blocking:tcp:scylla-client.scylla.svc.cluster.local:9042",
42            );
43            command.env("LINERA_HELMFILE_SET_DUAL_STORE", "true");
44        }
45
46        if with_block_exporter {
47            command.env("LINERA_HELMFILE_SET_EXPLORER_ENABLED", "true");
48            command.env(
49                "LINERA_HELMFILE_NUM_BLOCK_EXPORTERS",
50                num_block_exporters.to_string(),
51            );
52            command.env(
53                "LINERA_HELMFILE_BLOCK_EXPORTER_PORT",
54                DEFAULT_BLOCK_EXPORTER_PORT.to_string(),
55            );
56            command.env("LINERA_HELMFILE_INDEXER_IMAGE", indexer_image_name);
57            command.env("LINERA_HELMFILE_EXPLORER_IMAGE", explorer_image_name);
58        }
59
60        command
61            .env(
62                "LINERA_HELMFILE_SET_SERVER_CONFIG",
63                format!("working/server_{server_config_id}.json"),
64            )
65            .env("LINERA_HELMFILE_SET_NUM_PROXIES", num_proxies.to_string())
66            .env("LINERA_HELMFILE_SET_NUM_SHARDS", num_shards.to_string())
67            .env("LINERA_HELMFILE_LINERA_IMAGE", docker_image_name)
68            .env(
69                "LINERA_HELMFILE_SET_KUBE_CONTEXT",
70                format!("kind-{}", cluster_id),
71            )
72            .arg("sync")
73            .arg("--wait")
74            .spawn_and_wait()
75            .await
76    }
77}
