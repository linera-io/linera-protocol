51// Each destination has an ID and a configuration.
52#[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Eq, Hash)]
53pub struct DestinationId {
54    address: String,
55    kind: DestinationKind,
56}
57
58impl DestinationId {
59    /// Creates a new destination ID from the address and kind.
60    pub fn new(address: String, kind: DestinationKind) -> Self {
61        Self { address, kind }
62    }
63
64    pub fn validator(address: String) -> Self {
65        Self {
66            address,
67            kind: DestinationKind::Validator,
68        }
69    }
70
71    /// Returns the address of the destination.
72    pub fn address(&self) -> &str {
73        &self.address
74    }
75
76    /// Returns the kind of the destination.
77    pub fn kind(&self) -> DestinationKind {
78        self.kind
79    }
80}

82/// The uri to provide export services to.
83#[derive(Debug, Clone, PartialEq, Eq)]
84pub enum Destination {
85    Indexer {
86        /// The gRPC network protocol.
87        tls: TlsConfig,
88        /// The host name of the target destination (IP or hostname).
89        endpoint: String,
90        /// The port number of the target destination.
91        port: u16,
92    },
93    Validator {
94        /// The host name of the target destination (IP or hostname).
95        endpoint: String,
96        /// The port number of the target destination.
97        port: u16,
98    },
99    Logging {
100        /// The log file path.
101        file_name: String,
102    },
103    EvmChain {
104        /// The EVM JSON-RPC endpoint URL.
105        endpoint: String,
106        /// The deployed LightClient contract address (hex).
107        light_client_address: String,
108        /// Hex-encoded private key for signing EVM transactions.
109        private_key: String,
110    },
111}

113/// The description for the gRPC based destination.
114/// Discriminates the export mode and the client to use.
115#[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Eq, Copy, Hash)]
116pub enum DestinationKind {
117    /// The indexer description.
118    Indexer,
119    /// The validator description.
120    Validator,
121    /// The logging target.
122    Logging,
123    /// The EVM chain target.
124    EvmChain,
125}
126
127impl Serialize for Destination {
128    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
129    where
130        S: Serializer,
131    {
132        use serde::ser::SerializeMap;
133
134        match self {
135            Destination::Indexer {
136                tls,
137                endpoint,
138                port,
139            } => {
140                let mut map = serializer.serialize_map(Some(4))?;
141                map.serialize_entry("kind", "Indexer")?;
142                map.serialize_entry("tls", tls)?;
143                map.serialize_entry("endpoint", endpoint)?;
144                map.serialize_entry("port", port)?;
145                map.end()
146            }
147            Destination::Validator { endpoint, port } => {
148                let mut map = serializer.serialize_map(Some(3))?;
149                map.serialize_entry("kind", "Validator")?;
150                map.serialize_entry("endpoint", endpoint)?;
151                map.serialize_entry("port", port)?;
152                map.end()
153            }
154            Destination::Logging { file_name } => {
155                let mut map = serializer.serialize_map(Some(2))?;
156                map.serialize_entry("kind", "Logging")?;
157                map.serialize_entry("file_name", file_name)?;
158                map.end()
159            }
160            Destination::EvmChain {
161                endpoint,
162                light_client_address,
163                private_key,
164            } => {
165                let mut map = serializer.serialize_map(Some(4))?;
166                map.serialize_entry("kind", "EvmChain")?;
167                map.serialize_entry("endpoint", endpoint)?;
168                map.serialize_entry("light_client_address", light_client_address)?;
169                map.serialize_entry("private_key", private_key)?;
170                map.end()
171            }
172        }
173    }
174}
175
176struct DestinationVisitor;
177
178impl<'de> Visitor<'de> for DestinationVisitor {
179    type Value = Destination;
180
181    fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
182        formatter.write_str("a map with a 'kind' field")
183    }
184
185    fn visit_map<V>(self, mut map: V) -> Result<Destination, V::Error>
186    where
187        V: MapAccess<'de>,
188    {
189        let mut kind: Option<String> = None;
190        let mut tls: Option<TlsConfig> = None;
191        let mut endpoint: Option<String> = None;
192        let mut port: Option<u16> = None;
193        let mut file_name: Option<String> = None;
194        let mut light_client_address: Option<String> = None;
195        let mut private_key: Option<String> = None;
196
197        while let Some(key) = map.next_key::<String>()? {
198            match key.as_str() {
199                "kind" => {
200                    if kind.is_some() {
201                        return Err(V::Error::duplicate_field("kind"));
202                    }
203                    kind = Some(map.next_value()?);
204                }
205                "tls" => {
206                    if tls.is_some() {
207                        return Err(V::Error::duplicate_field("tls"));
208                    }
209                    tls = Some(map.next_value()?);
210                }
211                "endpoint" => {
212                    if endpoint.is_some() {
213                        return Err(V::Error::duplicate_field("endpoint"));
214                    }
215                    endpoint = Some(map.next_value()?);
216                }
217                "port" => {
218                    if port.is_some() {
219                        return Err(V::Error::duplicate_field("port"));
220                    }
221                    port = Some(map.next_value()?);
222                }
223                "file_name" => {
224                    if file_name.is_some() {
225                        return Err(V::Error::duplicate_field("file_name"));
226                    }
227                    file_name = Some(map.next_value()?);
228                }
229                "light_client_address" => {
230                    if light_client_address.is_some() {
231                        return Err(V::Error::duplicate_field("light_client_address"));
232                    }
233                    light_client_address = Some(map.next_value()?);
234                }
235                "private_key" => {
236                    if private_key.is_some() {
237                        return Err(V::Error::duplicate_field("private_key"));
238                    }
239                    private_key = Some(map.next_value()?);
240                }
241                _ => {
242                    // Ignore unknown fields
243                    let _: serde::de::IgnoredAny = map.next_value()?;
244                }
245            }
246        }
247
248        let kind = kind.ok_or_else(|| V::Error::missing_field("kind"))?;
249
250        match kind.as_str() {
251            "Indexer" => {
252                let tls = tls.ok_or_else(|| V::Error::missing_field("tls"))?;
253                let endpoint = endpoint.ok_or_else(|| V::Error::missing_field("endpoint"))?;
254                let port = port.ok_or_else(|| V::Error::missing_field("port"))?;
255                Ok(Destination::Indexer {
256                    tls,
257                    endpoint,
258                    port,
259                })
260            }
261            "Validator" => {
262                let endpoint = endpoint.ok_or_else(|| V::Error::missing_field("endpoint"))?;
263                let port = port.ok_or_else(|| V::Error::missing_field("port"))?;
264                Ok(Destination::Validator { endpoint, port })
265            }
266            "Logging" => {
267                let file_name = file_name.ok_or_else(|| V::Error::missing_field("file_name"))?;
268                Ok(Destination::Logging { file_name })
269            }
270            "EvmChain" => {
271                let endpoint = endpoint.ok_or_else(|| V::Error::missing_field("endpoint"))?;
272                let light_client_address = light_client_address
273                    .ok_or_else(|| V::Error::missing_field("light_client_address"))?;
274                let private_key =
275                    private_key.ok_or_else(|| V::Error::missing_field("private_key"))?;
276                Ok(Destination::EvmChain {
277                    endpoint,
278                    light_client_address,
279                    private_key,
280                })
281            }
282            _ => Err(V::Error::unknown_variant(
283                &kind,
284                &["Indexer", "Validator", "Logging", "EvmChain"],
285            )),
286        }
287    }
288}
289impl<'de> Deserialize<'de> for Destination {
290    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
291    where
292        D: Deserializer<'de>,
293    {
294        deserializer.deserialize_map(DestinationVisitor)
295    }
296}
297/// The configuration file to impose various limits
298/// on the resources used by the linera-exporter.
299#[derive(Serialize, Deserialize, Debug, Clone, Copy, PartialEq)]
300pub struct LimitsConfig {
301    /// Time period in milliseconds between periodic persistence
302    /// to the shared storage.
303    pub persistence_period_ms: u32,
304    /// Maximum size of the work queue i.e. maximum number
305    /// of blocks queued up for exports per destination.
306    pub work_queue_size: u16,
307    /// Maximum weight of the blob cache in megabytes.
308    pub blob_cache_weight_mb: u16,
309    /// Estimated number of elements for the blob cache.
310    pub blob_cache_items_capacity: u16,
311    /// Maximum weight of the block cache in megabytes.
312    pub block_cache_weight_mb: u16,
313    /// Estimated number of elements for the block cache.
314    pub block_cache_items_capacity: u16,
315    /// Maximum weight in megabytes for the combined
316    /// cache, consisting of small miscellaneous items.
317    pub auxiliary_cache_size_mb: u16,
318}
319
320impl Default for LimitsConfig {
321    fn default() -> Self {
322        Self {
323            persistence_period_ms: 299 * 1000,
324            work_queue_size: 256,
325            blob_cache_weight_mb: 1024,
326            blob_cache_items_capacity: 8192,
327            block_cache_weight_mb: 1024,
328            block_cache_items_capacity: 8192,
329            auxiliary_cache_size_mb: 1024,
330        }
331    }
332}

334impl Destination {
335    pub fn address(&self) -> String {
336        match &self {
337            Destination::Indexer {
338                tls,
339                endpoint,
340                port,
341            } => {
342                let tls = match tls {
343                    TlsConfig::ClearText => "http",
344                    TlsConfig::Tls => "https",
345                };
346
347                format!("{}://{}:{}", tls, endpoint, port)
348            }
349
350            Destination::Validator { endpoint, port } => {
351                format!("{}:{}:{}", "grpc", endpoint, port)
352            }
353
354            Destination::Logging { file_name } => file_name.to_string(),
355
356            Destination::EvmChain {
357                endpoint,
358                light_client_address,
359                ..
360            } => format!("evm://{}@{}", light_client_address, endpoint),
361        }
362    }
363
364    pub fn id(&self) -> DestinationId {
365        let kind = match self {
366            Destination::Indexer { .. } => DestinationKind::Indexer,
367            Destination::Validator { .. } => DestinationKind::Validator,
368            Destination::Logging { .. } => DestinationKind::Logging,
369            Destination::EvmChain { .. } => DestinationKind::EvmChain,
370        };
371        DestinationId {
372            address: self.address(),
373            kind,
374        }
375    }
376}
