4//! L4 bulk download: heavy read path via `download_certificates_by_heights`.
5//!
6//! Exercises a different I/O shape from L2/L3 (large batched payloads rather
7//! than many tiny RPCs). The default `auto` height range targets the most
8//! recent heights the candidate already has, so the only variable under test is
9//! the read path, not history availability.
10

14use linera_base::{data_types::BlockHeight, identifiers::ChainId};
15use linera_core::{data_types::ChainInfoQuery, node::ValidatorNode};
16use tokio::task::JoinSet;
17
18use super::{
19    latency::Samples,
20    progress::{Phase, Progress},
21    report::{BulkDownloadReport, BulkRun, PerChainBulk},
22    rpc::timed,
23};

25/// How many batches the `auto` range spans.
26const AUTO_BATCH_COUNT: u64 = 100;
27
28pub async fn run<N>(
29    node: &N,
30    chains: &[ChainId],
31    batch_size: u32,
32    concurrencies: &[usize],
33    height_range_arg: &str,
34    rpc_timeout: Duration,
35    progress: &Progress,
36) -> Result<BulkDownloadReport>
37where
38    N: ValidatorNode + Clone + Send + Sync + 'static,
39{
40    let phase = progress.phase("L5 bulk download", Some(0));
41    let mut per_chain = Vec::with_capacity(chains.len());
42    for &chain in chains {
43        // A timed-out tip query means we cannot pick a range; skip the chain
44        // (recorded as no runs) rather than hang or abort the whole layer.
45        let tip = match timed(
46            rpc_timeout,
47            node.handle_chain_info_query(ChainInfoQuery::new(chain)),
48        )
49        .await
50        {
51            Ok(info) => info.info.next_block_height.0,
52            Err(_) => {
53                per_chain.push(PerChainBulk {
54                    chain_id: chain.to_string(),
55                    runs: Vec::new(),
56                });
57                continue;
58            }
59        };
60        let (from, to) = resolve_range(height_range_arg, tip, batch_size)?;
61        let batches = (from..to).step_by(batch_size.max(1) as usize).count() as u64;
62        phase.inc_length(batches * concurrencies.len() as u64);
63
64        let mut runs = Vec::with_capacity(concurrencies.len());
65        for &concurrency in concurrencies {
66            runs.push(
67                run_one(
68                    node,
69                    chain,
70                    batch_size,
71                    concurrency.max(1),
72                    (from, to),
73                    rpc_timeout,
74                    &phase,
75                )
76                .await,
77            );
78        }
79        per_chain.push(PerChainBulk {
80            chain_id: chain.to_string(),
81            runs,
82        });
83    }
84    phase.finish_ok();
85    Ok(BulkDownloadReport { per_chain })
86}

88/// Run a single (chain, concurrency) batch download over `[from, to)` (`range`).
89async fn run_one<N>(
90    node: &N,
91    chain: ChainId,
92    batch_size: u32,
93    concurrency: usize,
94    range: (u64, u64),
95    rpc_timeout: Duration,
96    phase: &Phase,
97) -> BulkRun
98where
99    N: ValidatorNode + Clone + Send + Sync + 'static,
100{
101    let (from, to) = range;
102    let batch_starts: Vec<u64> = (from..to).step_by(batch_size.max(1) as usize).collect();
103    let mut set: JoinSet<(Result<u64, String>, u64, f64)> = JoinSet::new();
104    let mut samples = Samples::new();
105    let mut bytes_in: u64 = 0;
106    let mut certs_received: u64 = 0;
107
108    let started = Instant::now();
109    let mut next = 0;
110    // Keep at most `concurrency` batch requests in flight at once.
111    while next < batch_starts.len() || !set.is_empty() {
112        while set.len() < concurrency && next < batch_starts.len() {
113            let start_height = batch_starts[next];
114            next += 1;
115            let end_height = (start_height + batch_size as u64).min(to);
116            let heights: Vec<BlockHeight> = (start_height..end_height).map(BlockHeight).collect();
117            let node = node.clone();
118            set.spawn(async move {
119                let t0 = Instant::now();
120                let result = timed(
121                    rpc_timeout,
122                    node.download_certificates_by_heights(chain, heights),
123                )
124                .await;
125                let elapsed = t0.elapsed().as_secs_f64() * 1000.0;
126                match result {
127                    Ok(certs) => {
128                        let bytes: u64 = certs
129                            .iter()
130                            .map(|c| bcs::serialized_size(c).unwrap_or(0) as u64)
131                            .sum();
132                        (Ok(certs.len() as u64), bytes, elapsed)
133                    }
134                    Err(category) => (Err(category), 0, elapsed),
135                }
136            });
137        }
138        if let Some(joined) = set.join_next().await {
139            match joined {
140                Ok((Ok(count), bytes, ms)) => {
141                    samples.record_success(ms);
142                    certs_received += count;
143                    bytes_in += bytes;
144                }
145                Ok((Err(category), _, _)) => samples.record_error(category),
146                Err(e) => samples.record_error(format!("join: {e}")),
147            }
148            phase.inc(1);
149        }
150    }
151
152    let duration = started.elapsed().as_secs_f64().max(f64::EPSILON);
153    BulkRun {
154        concurrency,
155        batch_size,
156        heights_range: (from, to),
157        bytes_in,
158        certs_received,
159        duration_secs: duration,
160        mb_per_sec: (bytes_in as f64 / 1_048_576.0) / duration,
161        certs_per_sec: certs_received as f64 / duration,
162        latency_ms: samples.summary(),
163    }
164}

166/// Resolve the height range: `auto` targets the most recent
167/// `batch_size * AUTO_BATCH_COUNT` heights up to the tip; otherwise `FROM:TO`.
168fn resolve_range(arg: &str, tip: u64, batch_size: u32) -> Result<(u64, u64)> {
169    if arg == "auto" {
170        let span = batch_size as u64 * AUTO_BATCH_COUNT;
171        Ok((tip.saturating_sub(span), tip))
172    } else {
173        let (a, b) = arg
174            .split_once(':')
175            .ok_or_else(|| anyhow::anyhow!("--bulk-height-range must be `auto` or `FROM:TO`"))?;
176        let from: u64 = a.trim().parse()?;
177        let to: u64 = b.trim().parse()?;
178        if to < from {
179            anyhow::bail!("--bulk-height-range FROM must not exceed TO");
180        }
181        Ok((from, to))
182    }
183}

185#[cfg(test)]
186mod tests {
187    use super::resolve_range;
188
189    #[test]
190    fn auto_under_tip() {
191        assert_eq!(
192            resolve_range("auto", 50_000, 100).unwrap(),
193            (40_000, 50_000)
194        );
195    }
196
197    #[test]
198    fn auto_clamps_to_zero() {
199        assert_eq!(resolve_range("auto", 50, 100).unwrap(), (0, 50));
200    }
201
202    #[test]
203    fn explicit_range() {
204        assert_eq!(resolve_range("100:200", 999, 10).unwrap(), (100, 200));
205    }
206
207    #[test]
208    fn explicit_range_inverted_rejected() {
209        assert!(resolve_range("200:100", 999, 10).is_err());
210    }
211
212    #[test]
213    fn invalid_range_rejected() {
214        assert!(resolve_range("bad", 999, 10).is_err());
215    }
