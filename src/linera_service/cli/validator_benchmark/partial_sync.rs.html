4//! L6 partial sync (opt-in `--deep`): bounded write-path ingest measurement.
5//!
6//! Pushes at most `max_blocks` confirmed certificates starting at the
7//! candidate's current tip, mirroring the client-side push that
8//! `ChainClient::sync_validator` performs (including blob upload on
9//! `BlobsNotFound`) but capped to a fixed number of blocks. This is the only
10//! layer with a stateful side effect on the candidate, hence opt-in.
11
12use std::time::{Duration, Instant};
13
14use anyhow::Result;
15use linera_base::{data_types::BlockHeight, identifiers::ChainId};
16use linera_client::{chain_listener::ClientContext as _, client_context::ClientContext};
17use linera_core::node::{CrossChainMessageDelivery, NodeError, ValidatorNode};
18use linera_storage::Storage as _;
19
20use super::{progress::Progress, report::PartialSyncReport, rpc::timed};
21
22/// Compute the exclusive end height for a bounded sync, saturating on overflow.
23pub(super) fn end_height(candidate_tip: u64, max_blocks: u32, local_tip: u64) -> u64 {
24    candidate_tip
25        .saturating_add(max_blocks as u64)
26        .min(local_tip)
27}
28
29pub async fn run<N, Env>(
30    node: &N,
31    context: &ClientContext<Env>,
32    chain: ChainId,
33    max_blocks: u32,
34    rpc_timeout: Duration,
35    progress: &Progress,
36) -> Result<PartialSyncReport>
37where
38    N: ValidatorNode,
39    Env: linera_core::Environment,
40{
41    let mut report = PartialSyncReport {
42        chain_id: chain.to_string(),
43        from_height: 0,
44        to_height: 0,
45        blocks_attempted: 0,
46        blocks_accepted: 0,
47        bytes_in: 0,
48        duration_secs: 0.0,
49        blocks_per_sec: 0.0,
50    };
51
52    let chain_client = context.make_chain_client(chain).await?;
53
54    // The local sync can be heavy (it downloads the certificates to push); give
55    // it a visible spinner so it never looks frozen. NOTE: on a tall chain the
56    // wallet does not track, this can exceed --rpc-timeout-secs; raise it or use
57    // a tracked chain if the seed times out here.
58    let sync_phase = progress.phase("L2 partial sync: local state", None);
59    let local_tip = match timed(rpc_timeout, chain_client.synchronize_chain_state(chain)).await {
60        Ok(info) => {
61            sync_phase.finish_ok();
62            info.next_block_height.0
63        }
64        Err(category) => {
65            sync_phase.finish_fail();
66            anyhow::bail!("L2 partial sync: local sync failed ({category})");
67        }
68    };
69
70    // The candidate may not yet know the chain at all (BlobsNotFound -> 0).
71    let candidate_tip = match timed(
72        rpc_timeout,
73        node.handle_chain_info_query(linera_core::data_types::ChainInfoQuery::new(chain)),
74    )
75    .await
76    {
77        Ok(response) => response.info.next_block_height.0,
78        Err(_) => 0,
79    };
80
81    let from = candidate_tip;
82    let to = end_height(candidate_tip, max_blocks, local_tip);
83    report.from_height = from;
84    report.to_height = to;
85
86    if to <= from {
87        return Ok(report);
88    }
89
90    let phase = progress.phase("L2 partial sync", Some(to - from));
91    let heights: Vec<BlockHeight> = (from..to).map(BlockHeight).collect();
92    let storage = chain_client.storage_client();
93    // Certificates and blobs stay wrapped in the storage cache `Arc`, which is
94    // exactly what the validator node's handlers accept.
95    let certificates = storage
96        .read_certificates_by_heights(chain, &heights)
97        .await?
98        .into_iter()
99        .flatten();
100
101    let started = Instant::now();
102    for certificate in certificates {
103        report.blocks_attempted += 1;
104        phase.inc(1);
105        report.bytes_in += bcs::serialized_size(&*certificate).unwrap_or(0) as u64;
106
107        // First attempt. A timeout or non-blob error skips this block (the run
108        // keeps going); a `BlobsNotFound` triggers the blob-upload retry below.
109        // The typed error is needed here, so use a raw timeout rather than the
110        // string-categorizing `timed` helper.
111        let first = tokio::time::timeout(
112            rpc_timeout,
113            node.handle_confirmed_certificate(
114                certificate.clone(),
115                CrossChainMessageDelivery::NonBlocking,
116            ),
117        )
118        .await;
119        let missing = match first {
120            Ok(Ok(_)) => {
121                report.blocks_accepted += 1;
122                continue;
123            }
124            Ok(Err(NodeError::BlobsNotFound(missing_blob_ids))) => missing_blob_ids,
125            Ok(Err(_)) | Err(_) => continue,
126        };
127
128        let blobs = storage
129            .read_blobs(&missing)
130            .await?
131            .into_iter()
132            .flatten()
133            .map(|b| b.into_std())
134            .collect::<Vec<_>>();
135        if timed(rpc_timeout, node.upload_blobs(blobs)).await.is_err() {
136            continue;
137        }
138        if timed(
139            rpc_timeout,
140            node.handle_confirmed_certificate(certificate, CrossChainMessageDelivery::NonBlocking),
141        )
142        .await
143        .is_ok()
144        {
145            report.blocks_accepted += 1;
146        }
147    }
148
149    let duration = started.elapsed().as_secs_f64().max(f64::EPSILON);
150    report.duration_secs = duration;
151    report.blocks_per_sec = report.blocks_accepted as f64 / duration;
152    phase.finish_ok();
153    Ok(report)
154}

156#[cfg(test)]
157mod tests {
158    use super::end_height;
159
160    #[test]
161    fn caps_at_max_blocks() {
162        assert_eq!(end_height(1_000, 500, 10_000), 1_500);
163    }
164
165    #[test]
166    fn caps_at_local_tip() {
167        assert_eq!(end_height(1_000, 5_000, 1_200), 1_200);
168    }
169
170    #[test]
171    fn saturates_on_overflow() {
172        assert_eq!(end_height(u64::MAX - 1, 1_000, u64::MAX), u64::MAX);
173    }
174
175    #[test]
176    fn nothing_to_do_when_caught_up() {
177        assert_eq!(end_height(1_200, 1_000, 1_200), 1_200);
178    }
179}
