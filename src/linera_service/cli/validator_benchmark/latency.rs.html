4//! In-memory latency sample accumulator and percentile summary.
5//!
6//! Sample volumes per bucket are small (a few hundred to a few tens of
7//! thousands), so raw samples are kept in a `Vec` and percentiles are computed
8//! by sorting on demand. No HDR histogram dependency is needed.
9
10use std::collections::BTreeMap;
11
12use serde::{Deserialize, Serialize};
13
14/// Accumulates successful-request durations and categorized errors for a single
15/// measurement bucket (one layer / chain / concurrency level).
16#[derive(Debug, Default, Clone)]
17pub struct Samples {
18    durations_ms: Vec<f64>,
19    errors: BTreeMap<String, u64>,
20}
21
22impl Samples {
23    pub fn new() -> Self {
24        Self::default()
25    }
26
27    /// Record a successful request latency in milliseconds.
28    pub fn record_success(&mut self, ms: f64) {
29        self.durations_ms.push(ms);
30    }
31
32    /// Record a failed request under a category label (e.g. `timeout`).
33    pub fn record_error(&mut self, category: impl Into<String>) {
34        *self.errors.entry(category.into()).or_default() += 1;
35    }
36
37    /// Fold another bucket's samples and errors into this one.
38    ///
39    /// Used by the stress runner to merge per-task `Samples` without a shared
40    /// mutex (which would distort the latency measurement under contention).
41    pub fn merge_from(&mut self, other: Samples) {
42        let Samples {
43            mut durations_ms,
44            errors,
45        } = other;
46        self.durations_ms.append(&mut durations_ms);
47        for (category, count) in errors {
48            *self.errors.entry(category).or_default() += count;
49        }
50    }
51
52    /// Compute the distribution summary over the recorded samples.
53    pub fn summary(&self) -> LatencySummary {
54        let mut sorted = self.durations_ms.clone();
55        sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
56        let count = sorted.len();
57
58        // Nearest-rank percentile, `num/den` of the way through, via integer math
59        // to avoid a lossy float-to-index cast.
60        let percentile = |num: usize, den: usize| -> f64 {
61            if sorted.is_empty() {
62                return 0.0;
63            }
64            let rank = (count * num).div_ceil(den); // 1-based ceil(count * num / den)
65            let idx = rank.saturating_sub(1).min(count - 1);
66            sorted[idx]
67        };
68
69        let mean = if count == 0 {
70            0.0
71        } else {
72            sorted.iter().sum::<f64>() / count as f64
73        };
74        let stddev = if count < 2 {
75            0.0
76        } else {
77            let variance = sorted.iter().map(|x| (x - mean).powi(2)).sum::<f64>() / count as f64;
78            variance.sqrt()
79        };
80
81        LatencySummary {
82            count: count as u64,
83            min: sorted.first().copied().unwrap_or(0.0),
84            max: sorted.last().copied().unwrap_or(0.0),
85            mean,
86            stddev,
87            p50: percentile(50, 100),
88            p95: percentile(95, 100),
89            p99: percentile(99, 100),
90            errors: self
91                .errors
92                .iter()
93                .map(|(category, count)| ErrorBucket {
94                    category: category.clone(),
95                    count: *count,
96                })
97                .collect(),
98        }
99    }
100}
101
102/// Distribution summary for a measurement bucket.
103#[derive(Debug, Clone, Default, Serialize, Deserialize)]
104pub struct LatencySummary {
105    pub count: u64,
106    pub min: f64,
107    pub max: f64,
108    pub mean: f64,
109    pub stddev: f64,
110    pub p50: f64,
111    pub p95: f64,
112    pub p99: f64,
113    pub errors: Vec<ErrorBucket>,
114}
115
116/// A count of failures sharing a category label.
117#[derive(Debug, Clone, Serialize, Deserialize)]
118pub struct ErrorBucket {
119    pub category: String,
120    pub count: u64,
121}

123#[cfg(test)]
124mod tests {
125    use std::collections::BTreeMap;
126
127    use super::*;
128
129    #[test]
130    fn summary_of_known_sample_set() {
131        let mut s = Samples::new();
132        for i in 1..=100u64 {
133            s.record_success(i as f64);
134        }
135        let sum = s.summary();
136        assert_eq!(sum.count, 100);
137        assert_eq!(sum.min, 1.0);
138        assert_eq!(sum.max, 100.0);
139        assert!((sum.p50 - 50.0).abs() < 1.0);
140        assert!((sum.p95 - 95.0).abs() < 1.0);
141        assert!((sum.p99 - 99.0).abs() < 1.0);
142        assert!((sum.mean - 50.5).abs() < 0.01);
143    }
144
145    #[test]
146    fn errors_categorized() {
147        let mut s = Samples::new();
148        s.record_error("timeout");
149        s.record_error("timeout");
150        s.record_error("unavailable");
151        let sum = s.summary();
152        assert_eq!(sum.count, 0);
153        let by_cat: BTreeMap<_, _> = sum
154            .errors
155            .iter()
156            .map(|e| (e.category.clone(), e.count))
157            .collect();
158        assert_eq!(by_cat.get("timeout"), Some(&2));
159        assert_eq!(by_cat.get("unavailable"), Some(&1));
160    }
161
162    #[test]
163    fn empty_summary_is_zero() {
164        let s = Samples::new();
165        let sum = s.summary();
166        assert_eq!(sum.count, 0);
167        assert_eq!(sum.errors.len(), 0);
168        assert_eq!(sum.min, 0.0);
169        assert_eq!(sum.max, 0.0);
170    }
171
172    #[test]
173    fn merge_combines_samples_and_errors() {
174        let mut a = Samples::new();
175        a.record_success(1.0);
176        a.record_error("timeout");
177        let mut b = Samples::new();
178        b.record_success(2.0);
179        b.record_error("timeout");
180        b.record_error("other");
181        a.merge_from(b);
182        let sum = a.summary();
183        assert_eq!(sum.count, 2);
184        let by: BTreeMap<_, _> = sum
185            .errors
186            .iter()
187            .map(|e| (e.category.clone(), e.count))
188            .collect();
189        assert_eq!(by["timeout"], 2);
190        assert_eq!(by["other"], 1);
191    }
192}
