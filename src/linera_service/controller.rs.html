33/// An update message sent to a TaskProcessor to change its set of applications.
34#[derive(Debug)]
35pub struct Update {
36    pub application_ids: Vec<ApplicationId>,
37}

39struct ProcessorHandle {
40    update_sender: mpsc::UnboundedSender<Update>,
41}

43pub struct Controller<Ctx: ClientContext> {
44    chain_id: ChainId,
45    controller_id: ApplicationId,
46    context: Arc<Mutex<Ctx>>,
47    chain_client: ChainClient<Ctx::Environment>,
48    cancellation_token: CancellationToken,
49    notifications: NotificationStream,
50    operators: OperatorMap,
51    retry_delay: TimeDelta,
52    processors: BTreeMap<ChainId, ProcessorHandle>,
53    listened_local_chains: BTreeSet<ChainId>,
54    current_message_policies: BTreeMap<ChainId, MessagePolicy>,
55    command_sender: UnboundedSender<ListenerCommand>,
56    pending_services_notifications: BTreeMap<
57        ChainId,
58        (
59            HashMap<ManagedServiceId, PendingService>,

63}

100    pub async fn run(mut self) {
101        info!(
102            "Watching for notifications for controller chain {}",
103            self.chain_id
104        );
105        self.process_controller_state().await;
106        loop {
107            let pending_services_notifications: std::pin::Pin<
108                Box<dyn futures::Future<Output = (ChainId, Option<Notification>)> + Send>,
109            > = if !self.pending_services_notifications.is_empty() {
110                Box::pin(
111                    futures::future::select_all(
112                        self.pending_services_notifications.iter_mut().map(
113                            |(chain_id, (_, notifications))| {
114                                notifications.next().map(|result| (*chain_id, result))
115                            },
116                        ),
117                    )
118                    .map(|((chain_id, maybe_notification), _, _)| (chain_id, maybe_notification)),
119                )
120            } else {
121                Box::pin(futures::future::pending())
122            };
123            select! {
124                Some(notification) = self.notifications.next() => {
125                    if let Reason::NewBlock { .. } = notification.reason {
126                        debug!("Processing notification on controller chain {}", self.chain_id);
127                        self.process_controller_state().await;
128                    }
129                }
130                (chain_id, Some(notification)) = pending_services_notifications => {
131                    self.process_pending_service_notification(chain_id, notification).await;
132                }
133                _ = self.cancellation_token.cancelled().fuse() => {
134                    break;
135                }
136            }
137        }
138        debug!("Notification stream ended.");
139    }

141    async fn process_pending_service_notification(
142        &mut self,
143        chain_id: ChainId,
144        notification: Notification,
145    ) {
146        debug!(
147            "Processing notification on pending service chain {}",
148            chain_id
149        );
150        if let Reason::NewBlock { height, .. } = notification.reason {
151            let pending_services = &mut self
152                .pending_services_notifications
153                .get_mut(&chain_id)
154                .expect("the entry should exist")
155                .0;
156            for (service_id, pending_service) in &*pending_services {
157                if pending_service.start_block_height <= height {
158                    let bytes = bcs::to_bytes(&Operation::StartLocalService {
159                        service_id: *service_id,
160                    })
161                    .expect("bcs bytes");
162                    let operation = linera_execution::Operation::User {
163                        application_id: self.controller_id,
164                        bytes,
165                    };
166                    if let Err(e) = self
167                        .chain_client
168                        .execute_operations(vec![operation], vec![])
169                        .await
170                    {
171                        // TODO: handle leader timeouts
172                        error!("Failed to execute worker on-chain registration: {e}");
173                    }
174                }
175            }
176            pending_services
177                .retain(|_, pending_service| pending_service.start_block_height > height);
178            if pending_services.is_empty() {
179                let _ = self.pending_services_notifications.remove(&chain_id);
180            }
181        }
182    }

184    async fn process_controller_state(&mut self) {
185        let state = match self.query_controller_state().await {
186            Ok(state) => state,
187            Err(error) => {
188                error!("Error reading controller state: {error}");
189                return;
190            }
191        };
192        let Some(worker) = state.local_worker else {
193            // Worker needs to be registered.
194            self.register_worker().await;
195            return;
196        };
197        assert_eq!(
198            worker.owner,
199            self.chain_client
200                .preferred_owner()
201                .expect("The current wallet should own the chain being watched"),
202            "We should be registered with the current account owner."
203        );
204
205        // Subscribe to notifications on pending services chains - we need to know when
206        // they sync their blocks.
207        for (managed_service_id, (chain_id, pending_service)) in &state.local_pending_services {
208            // No need to subscribe twice.
209            if self.pending_services_notifications.contains_key(chain_id) {
210                continue;
211            }
212            let service_notifications = self
213                .chain_client
214                .subscribe_to(*chain_id)
215                .expect("client subscription");
216            self.pending_services_notifications
217                .entry(*chain_id)
218                .or_insert_with(|| (HashMap::new(), service_notifications))
219                .0
220                .insert(*managed_service_id, pending_service.clone());
221        }
222
223        // Build a map of ChainId -> Vec<ApplicationId> from local_services
224        let mut chain_apps: BTreeMap<ChainId, Vec<ApplicationId>> = BTreeMap::new();
225        for service in &state.local_services {
226            chain_apps
227                .entry(service.chain_id)
228                .or_default()
229                .push(service.application_id);
230        }
231
232        // The service chains should reject bundles that contain no messages sent by relevant apps
233        // and no system messages.
234        let mut message_policies: BTreeMap<_, _> = chain_apps
235            .iter()
236            .map(|(chain_id, apps)| {
237                let message_policy = MessagePolicy {
238                    reject_message_bundles_without_application_ids: Some(
239                        apps.iter()
240                            .map(|app_id| GenericApplicationId::User(*app_id))
241                            .chain(std::iter::once(GenericApplicationId::User(
242                                self.controller_id,
243                            )))
244                            .chain(std::iter::once(GenericApplicationId::System))
245                            .collect(),
246                    ),
247                    ..Default::default()
248                };
249                (*chain_id, message_policy)
250            })
251            .collect();
252        message_policies.extend(state.local_message_policy);
253
254        let message_policies_to_update: BTreeMap<_, _> = message_policies
255            .iter()
256            .filter(|(chain_id, message_policy)| {
257                self.current_message_policies.get(chain_id) != Some(*message_policy)
258            })
259            .map(|(chain_id, message_policy)| (*chain_id, message_policy.clone()))
260            .collect();
261
262        let old_chains: BTreeSet<_> = self.processors.keys().cloned().collect();
263
264        // Update or spawn processors for each chain
265        for (service_chain_id, application_ids) in chain_apps {
266            if let Err(err) = self
267                .update_or_spawn_processor(service_chain_id, application_ids)
268                .await
269            {
270                error!("Error updating or spawning processor: {err}");
271                return;
272            }
273        }
274
275        // Send empty updates to processors for chains no longer in the state
276        // This effectively tells them to stop processing applications
277        let active_chains: std::collections::BTreeSet<_> =
278            state.local_services.iter().map(|s| s.chain_id).collect();
279        let stale_chains: BTreeSet<_> = self
280            .processors
281            .keys()
282            .filter(|chain_id| !active_chains.contains(chain_id))
283            .cloned()
284            .collect();
285        for chain_id in &stale_chains {
286            if let Some(handle) = self.processors.get(chain_id) {
287                let update = Update {
288                    application_ids: Vec::new(),
289                };
290                if handle.update_sender.send(update).is_err() {
291                    // Processor has stopped, remove it
292                    self.processors.remove(chain_id);
293                }
294            }
295        }
296
297        // Collect local_chains from state
298        let local_chains: BTreeSet<_> = state.local_chains.iter().cloned().collect();
299
300        // Compute all chains we were listening to (processors + local_chains)
301        let old_listened: BTreeSet<_> = old_chains
302            .union(&self.listened_local_chains)
303            .cloned()
304            .collect();
305
306        // Compute all chains we want to listen to (active services + local_chains)
307        let desired_listened: BTreeSet<_> = active_chains.union(&local_chains).cloned().collect();
308
309        // New chains to listen (neither had processor nor were in listened_local_chains)
310        let owner = worker.owner;
311        let mut new_chains: BTreeMap<_, _> = desired_listened
312            .difference(&old_listened)
313            .map(|chain_id| (*chain_id, Some(owner)))
314            .collect();
315
316        // Follow the chains for pending services, so that they are synced.
317        new_chains.extend(
318            state
319                .local_pending_services
320                .iter()
321                .map(|(_, (chain_id, _))| *chain_id)
322                .collect::<BTreeSet<_>>()
323                .difference(&old_listened)
324                .map(|chain_id| (*chain_id, None)),
325        );
326
327        // Chains to stop listening (were listened but no longer needed)
328        let chains_to_stop: BTreeSet<_> = old_listened
329            .difference(&desired_listened)
330            .cloned()
331            .collect();
332
333        // Update listened_local_chains for next iteration
334        // These are local_chains that don't have services (not in active_chains)
335        self.listened_local_chains = local_chains.difference(&active_chains).cloned().collect();
336
337        if let Err(error) = self
338            .command_sender
339            .send(ListenerCommand::Listen(new_chains))
340        {
341            error!(%error, "error sending a command to chain listener");
342        }
343        if let Err(error) = self
344            .command_sender
345            .send(ListenerCommand::StopListening(chains_to_stop))
346        {
347            error!(%error, "error sending a command to chain listener");
348        }
349        // This may affect chains we are just starting to listen to, so this command has to be sent
350        // last.
351        if let Err(error) = self.command_sender.send(ListenerCommand::SetMessagePolicy(
352            message_policies_to_update,
353        )) {
354            error!(%error, "error sending a command to chain listener");
355        }
356        self.current_message_policies = message_policies;
357    }

359    async fn register_worker(&mut self) {
360        let capabilities = self.operators.keys().cloned().collect();
361        let command = WorkerCommand::RegisterWorker { capabilities };
362        let owner = self
363            .chain_client
364            .preferred_owner()
365            .expect("The current wallet should own the chain being watched");
366        let bytes =
367            bcs::to_bytes(&Operation::ExecuteWorkerCommand { owner, command }).expect("bcs bytes");
368        let operation = linera_execution::Operation::User {
369            application_id: self.controller_id,
370            bytes,
371        };
372        if let Err(e) = self
373            .chain_client
374            .execute_operations(vec![operation], vec![])
375            .await
376        {
377            // TODO: handle leader timeouts
378            error!("Failed to execute worker on-chain registration: {e}");
379        }
380    }

382    async fn update_or_spawn_processor(
383        &mut self,
384        service_chain_id: ChainId,
385        application_ids: Vec<ApplicationId>,
386    ) -> Result<(), anyhow::Error> {
387        if let Some(handle) = self.processors.get(&service_chain_id) {
388            // Processor exists, send update
389            let update = Update {
390                application_ids: application_ids.clone(),
391            };
392            if handle.update_sender.send(update).is_err() {
393                // Processor has stopped, remove and respawn
394                self.processors.remove(&service_chain_id);
395                self.spawn_processor(service_chain_id, application_ids)
396                    .await?;
397            }
398        } else {
399            // No processor for this chain, spawn one
400            self.spawn_processor(service_chain_id, application_ids)
401                .await?;
402        }
403        Ok(())
404    }

406    async fn spawn_processor(
407        &mut self,
408        service_chain_id: ChainId,
409        application_ids: Vec<ApplicationId>,
410    ) -> Result<(), anyhow::Error> {
411        info!(
412            "Spawning TaskProcessor for chain {} with applications {:?}",
413            service_chain_id, application_ids
414        );
415
416        let (update_sender, update_receiver) = mpsc::unbounded_channel();
417
418        let mut chain_client = self
419            .context
420            .lock()
421            .await
422            .make_chain_client(service_chain_id)
423            .await?;
424        // The processor may need to propose blocks with task results - for that, it will
425        // need the chain client to be configured with a preferred owner.
426        if let Some(owner) = self.chain_client.preferred_owner() {
427            chain_client.set_preferred_owner(owner);
428        }
429        let processor = TaskProcessor::new(
430            service_chain_id,
431            application_ids,
432            chain_client,
433            self.cancellation_token.child_token(),
434            self.operators.clone(),
435            self.retry_delay,
436            Some(update_receiver),
437        );
438
439        tokio::spawn(processor.run());
440
441        self.processors
442            .insert(service_chain_id, ProcessorHandle { update_sender });
443
444        Ok(())
445    }

447    async fn query_controller_state(&mut self) -> Result<LocalWorkerState, anyhow::Error> {
448        let query = "query { localWorkerState }";
449        let bytes = serde_json::to_vec(&json!({"query": query}))?;
450        let query = linera_execution::Query::User {
451            application_id: self.controller_id,
452            bytes,
453        };
454        let linera_execution::QueryOutcome {
455            response,
456            operations: _,
457        } = self.chain_client.query_application(query, None).await?;
458        let linera_execution::QueryResponse::User(response) = response else {
459            anyhow::bail!("cannot get a system response for a user query");
460        };
461        let mut response: serde_json::Value = serde_json::from_slice(&response)?;
462        let state = serde_json::from_value(response["data"]["localWorkerState"].take())?;
463        Ok(state)
464    }
