15/// A tag used for recovering the public key from a compact signature.
16#[derive(Copy, Clone, PartialEq, Eq, Debug)]
17pub enum RecoveryId {
18    /// Signature recovery ID 0
19    Zero,
20    /// Signature recovery ID 1
21    One,
22    /// Signature recovery ID 2
23    Two,
24    /// Signature recovery ID 3
25    Three,
26}
27
28impl TryFrom<i32> for RecoveryId {
29    type Error = Error;
30    #[inline]
31    fn try_from(id: i32) -> Result<RecoveryId, Error> {
32        match id {
33            0 => Ok(RecoveryId::Zero),
34            1 => Ok(RecoveryId::One),
35            2 => Ok(RecoveryId::Two),
36            3 => Ok(RecoveryId::Three),
37            _ => Err(Error::InvalidRecoveryId),
38        }
39    }
40}
41
42impl From<RecoveryId> for i32 {
43    #[inline]
44    fn from(val: RecoveryId) -> Self {
45        match val {
46            RecoveryId::Zero => 0,
47            RecoveryId::One => 1,
48            RecoveryId::Two => 2,
49            RecoveryId::Three => 3,
50        }
51    }
52}

54/// An ECDSA signature with a recovery ID for pubkey recovery.
55#[derive(Copy, Clone, PartialEq, Eq, Debug, Hash, Ord, PartialOrd)]
56pub struct RecoverableSignature(ffi::RecoverableSignature);
57
58impl RecoverableSignature {
59    #[inline]
60    /// Converts a compact-encoded byte slice to a signature. This
61    /// representation is nonstandard and defined by the libsecp256k1 library.
62    pub fn from_compact(data: &[u8], recid: RecoveryId) -> Result<RecoverableSignature, Error> {
63        if data.is_empty() {
64            return Err(Error::InvalidSignature);
65        }
66
67        let mut ret = ffi::RecoverableSignature::new();
68
69        unsafe {
70            if data.len() != 64 {
71                Err(Error::InvalidSignature)
72            } else if ffi::secp256k1_ecdsa_recoverable_signature_parse_compact(
73                super_ffi::secp256k1_context_no_precomp,
74                &mut ret,
75                data.as_c_ptr(),
76                recid.into(),
77            ) == 1
78            {
79                Ok(RecoverableSignature(ret))
80            } else {
81                Err(Error::InvalidSignature)
82            }
83        }
84    }
85
86    /// Obtains a raw pointer suitable for use with FFI functions.
87    #[inline]
88    #[deprecated(since = "0.25.0", note = "Use Self::as_c_ptr if you need to access the FFI layer")]
89    pub fn as_ptr(&self) -> *const ffi::RecoverableSignature { self.as_c_ptr() }
90
91    /// Obtains a raw mutable pointer suitable for use with FFI functions.
92    #[inline]
93    #[deprecated(
94        since = "0.25.0",
95        note = "Use Self::as_mut_c_ptr if you need to access the FFI layer"
96    )]
97    pub fn as_mut_ptr(&mut self) -> *mut ffi::RecoverableSignature { self.as_mut_c_ptr() }
98
99    #[inline]
100    /// Serializes the recoverable signature in compact format.
101    pub fn serialize_compact(&self) -> (RecoveryId, [u8; 64]) {
102        let mut ret = [0u8; 64];
103        let mut recid = RecoveryId::Zero.into();
104        unsafe {
105            let err = ffi::secp256k1_ecdsa_recoverable_signature_serialize_compact(
106                super_ffi::secp256k1_context_no_precomp,
107                ret.as_mut_c_ptr(),
108                &mut recid,
109                self.as_c_ptr(),
110            );
111            assert!(err == 1);
112        }
113        (recid.try_into().expect("ffi returned invalid RecoveryId!"), ret)
114    }
115
116    /// Converts a recoverable signature to a non-recoverable one (this is needed
117    /// for verification).
118    #[inline]
119    pub fn to_standard(&self) -> Signature {
120        unsafe {
121            let mut ret = super_ffi::Signature::new();
122            let err = ffi::secp256k1_ecdsa_recoverable_signature_convert(
123                super_ffi::secp256k1_context_no_precomp,
124                &mut ret,
125                self.as_c_ptr(),
126            );
127            assert!(err == 1);
128            Signature(ret)
129        }
130    }
131
132    /// Determines the public key for which this [`Signature`] is valid for `msg`. Requires a
133    /// verify-capable context.
134    #[inline]
135    #[cfg(feature = "global-context")]
136    pub fn recover(&self, msg: &Message) -> Result<key::PublicKey, Error> {
137        crate::SECP256K1.recover_ecdsa(msg, self)
138    }
139}

143    fn as_c_ptr(&self) -> *const Self::Target { &self.0 }
144
145    fn as_mut_c_ptr(&mut self) -> *mut Self::Target { &mut self.0 }
146}

150    #[inline]
151    fn from(sig: ffi::RecoverableSignature) -> RecoverableSignature { RecoverableSignature(sig) }

154impl<C: Signing> Secp256k1<C> {
155    fn sign_ecdsa_recoverable_with_noncedata_pointer(
156        &self,
157        msg: &Message,
158        sk: &key::SecretKey,
159        noncedata_ptr: *const super_ffi::types::c_void,
160    ) -> RecoverableSignature {
161        let mut ret = ffi::RecoverableSignature::new();
162        unsafe {
163            // We can assume the return value because it's not possible to construct
164            // an invalid signature from a valid `Message` and `SecretKey`
165            assert_eq!(
166                ffi::secp256k1_ecdsa_sign_recoverable(
167                    self.ctx.as_ptr(),
168                    &mut ret,
169                    msg.as_c_ptr(),
170                    sk.as_c_ptr(),
171                    super_ffi::secp256k1_nonce_function_rfc6979,
172                    noncedata_ptr
173                ),
174                1
175            );
176        }
177
178        RecoverableSignature::from(ret)
179    }
180
181    /// Constructs a signature for `msg` using the secret key `sk` and RFC6979 nonce
182    /// Requires a signing-capable context.
183    pub fn sign_ecdsa_recoverable(
184        &self,
185        msg: &Message,
186        sk: &key::SecretKey,
187    ) -> RecoverableSignature {
188        self.sign_ecdsa_recoverable_with_noncedata_pointer(msg, sk, ptr::null())
189    }
190
191    /// Constructs a signature for `msg` using the secret key `sk` and RFC6979 nonce
192    /// and includes 32 bytes of noncedata in the nonce generation via inclusion in
193    /// one of the hash operations during nonce generation. This is useful when multiple
194    /// signatures are needed for the same Message and SecretKey while still using RFC6979.
195    /// Requires a signing-capable context.
196    pub fn sign_ecdsa_recoverable_with_noncedata(
197        &self,
198        msg: &Message,
199        sk: &key::SecretKey,
200        noncedata: &[u8; 32],
201    ) -> RecoverableSignature {
202        let noncedata_ptr = noncedata.as_ptr() as *const super_ffi::types::c_void;
203        self.sign_ecdsa_recoverable_with_noncedata_pointer(msg, sk, noncedata_ptr)
204    }
205}

207impl<C: Verification> Secp256k1<C> {
208    /// Determines the public key for which `sig` is a valid signature for
209    /// `msg`. Requires a verify-capable context.
210    pub fn recover_ecdsa(
211        &self,
212        msg: &Message,
213        sig: &RecoverableSignature,
214    ) -> Result<key::PublicKey, Error> {
215        unsafe {
216            let mut pk = super_ffi::PublicKey::new();
217            if ffi::secp256k1_ecdsa_recover(
218                self.ctx.as_ptr(),
219                &mut pk,
220                sig.as_c_ptr(),
221                msg.as_c_ptr(),
222            ) != 1
223            {
224                return Err(Error::InvalidSignature);
225            }
226            Ok(key::PublicKey::from(pk))
227        }
228    }
229}

430    #[test]
431    fn test_recov_id_conversion_between_i32() {
432        assert!(RecoveryId::try_from(-1i32).is_err());
433        assert!(RecoveryId::try_from(0i32).is_ok());
434        assert!(RecoveryId::try_from(1i32).is_ok());
435        assert!(RecoveryId::try_from(2i32).is_ok());
436        assert!(RecoveryId::try_from(3i32).is_ok());
437        assert!(RecoveryId::try_from(4i32).is_err());
438        let id0 = RecoveryId::Zero;
439        assert_eq!(Into::<i32>::into(id0), 0i32);
440        let id1 = RecoveryId::One;
441        assert_eq!(Into::<i32>::into(id1), 1i32);
442    }
