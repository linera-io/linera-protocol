3//! This module provides an implementation of ElligatorSwift as well as a
4//! version of x-only ECDH using it (including compatibility with BIP324).
5//!
6//! `ElligatorSwift` is described in `https://eprint.iacr.org/2022/759` by
7//! Chavez-Saab, Rodriguez-Henriquez, and Tibouchi. It permits encoding
8//! uniformly chosen public keys as 64-byte arrays which are indistinguishable
9//! from uniformly random arrays.
10//!
11//! Let f be the function from pairs of field elements to point X coordinates,
12//! defined as follows (all operations modulo p = 2^256 - 2^32 - 977)
13//! f(u,t):
14//! - Let C = 0xa2d2ba93507f1df233770c2a797962cc61f6d15da14ecd47d8d27ae1cd5f852,
15//!   a square root of -3.
16//! - If u=0, set u=1 instead.
17//! - If t=0, set t=1 instead.
18//! - If u^3 + t^2 + 7 = 0, multiply t by 2.
19//! - Let X = (u^3 + 7 - t^2) / (2 * t)
20//! - Let Y = (X + t) / (C * u)
21//! - Return the first in [u + 4 * Y^2, (-X/Y - u) / 2, (X/Y - u) / 2] that is an
22//!   X coordinate on the curve (at least one of them is, for any u and t).
23//!
24//! Then an `ElligatorSwift` encoding of x consists of the 32-byte big-endian
25//! encodings of field elements u and t concatenated, where f(u,t) = x.
26//! The encoding algorithm is described in the paper, and effectively picks a
27//! uniformly random pair (u,t) among those which encode x.
28//!
29//! If the Y coordinate is relevant, it is given the same parity as t.
30//!
31//! Changes w.r.t. the paper:
32//! - The u=0, t=0, and u^3+t^2+7=0 conditions result in decoding to the point
33//!   at infinity in the paper. Here they are remapped to finite points.
34//! - The paper uses an additional encoding bit for the parity of y. Here the
35//!   parity of t is used (negating t does not affect the decoded x coordinate,
36//!   so this is possible).

47unsafe extern "C" fn hash_callback<F>(
48    output: *mut c_uchar,
49    x32: *const c_uchar,
50    ell_a64: *const c_uchar,
51    ell_b64: *const c_uchar,
52    hash_func: *mut c_void,
53) -> c_int
54where
55    F: FnMut([u8; 32], [u8; 64], [u8; 64]) -> ElligatorSwiftSharedSecret,
56{
57    let callback: &mut F = &mut *(hash_func as *mut F);
58    let mut x32_array = [0u8; 32];
59    let mut ell_a64_array = [0u8; 64];
60    let mut ell_b64_array = [0u8; 64];
61
62    // Copy the data into Rust slices
63    ptr::copy_nonoverlapping(x32, x32_array.as_mut_c_ptr(), 32);
64    ptr::copy_nonoverlapping(ell_a64, ell_a64_array.as_mut_c_ptr(), 64);
65    ptr::copy_nonoverlapping(ell_b64, ell_b64_array.as_mut_c_ptr(), 64);
66    // Call the hash function that was passed in through the `data` pointer
67    let secret = callback(x32_array, ell_a64_array, ell_b64_array);
68    // Copy the output from a [ElligatorSwiftSharedSecret] into the output pointer
69    ptr::copy_nonoverlapping(secret.0.as_ptr(), output, secret.0.len());
70    // Always returns 1
71    1
72}

74/// An encoding of an elliptic curvepoint such that a uniformly random on-curve
75/// point will be encoded as uniformly random bits.
76///
77/// This object holds two field elements u and t, which are the inputs to
78/// the `ElligatorSwift` encoding function.

106    /// Creates the Elligator Swift encoding from a secret key, using some aux_rand if defined.
107    /// This method is preferred instead of just decoding, because the private key offers extra
108    /// entropy.
109    /// # Example
110    /// ```
111    /// # #[cfg(feature = "alloc")] {
112    ///     use secp256k1::{ellswift::ElligatorSwift, PublicKey, Secp256k1, SecretKey};
113    ///     let secp = Secp256k1::new();
114    ///     let sk = SecretKey::from_slice(&[1; 32]).unwrap();
115    ///     let es = ElligatorSwift::from_seckey(&secp, sk, None);
116    /// # }
117    /// ```
118    pub fn from_seckey<C: Verification>(
119        secp: &Secp256k1<C>,
120        sk: SecretKey,
121        aux_rand: Option<[u8; 32]>,
122    ) -> ElligatorSwift {
123        let mut es_out = [0u8; constants::ELLSWIFT_ENCODING_SIZE];
124        let aux_rand_ptr = aux_rand.as_c_ptr();
125        unsafe {
126            let ret = ffi::secp256k1_ellswift_create(
127                secp.ctx().as_ptr(),
128                es_out.as_mut_c_ptr(),
129                sk.as_c_ptr(),
130                aux_rand_ptr,
131            );
132            debug_assert_eq!(ret, 1);
133        }
134        ElligatorSwift(ffi::ElligatorSwift::from_array(es_out))
135    }

151    /// Computes a shared secret only known by Alice and Bob. This is obtained by computing
152    /// the x-only Elliptic Curve Diffie-Hellman (ECDH) shared secret between Alice and Bob.
153    /// # Example
154    /// ```
155    /// # #[cfg(feature = "alloc")] {
156    ///     use secp256k1::{
157    ///         ellswift::{ElligatorSwift, ElligatorSwiftParty},
158    ///         PublicKey, SecretKey, XOnlyPublicKey, Secp256k1,
159    ///     };
160    ///     use core::str::FromStr;
161    ///
162    ///     let secp = Secp256k1::new();
163    ///
164    ///     let alice_sk = SecretKey::from_str("e714e76bdd67ad9f495683c37934148f4efc25ce3f01652c8a906498339e1f3a").unwrap();
165    ///     let bob_sk = SecretKey::from_str("b6c4b0e2f8c4359caf356a618cd1649d18790a1d67f7c2d1e4760e04c785db4f").unwrap();
166    ///
167    ///     let alice_es = ElligatorSwift::from_seckey(&secp, alice_sk, None);
168    ///     let bob_es = ElligatorSwift::from_seckey(&secp, bob_sk, None);
169    ///
170    ///     let alice_shared_secret = ElligatorSwift::shared_secret(alice_es, bob_es, alice_sk, ElligatorSwiftParty::A, None);
171    ///     let bob_shared_secret = ElligatorSwift::shared_secret(alice_es, bob_es, bob_sk, ElligatorSwiftParty::B, None);
172    ///
173    ///     assert_eq!(alice_shared_secret, bob_shared_secret);
174    /// # }
175    /// ```
176    pub fn shared_secret(
177        ellswift_a: ElligatorSwift,
178        ellswift_b: ElligatorSwift,
179        secret_key: SecretKey,
180        party: ElligatorSwiftParty,
181        data: Option<&[u8]>,
182    ) -> ElligatorSwiftSharedSecret {
183        let mut shared_secret = [0u8; 32];
184        unsafe {
185            let ret = ffi::secp256k1_ellswift_xdh(
186                ffi::secp256k1_context_no_precomp,
187                shared_secret.as_mut_c_ptr(),
188                ellswift_a.as_c_ptr(),
189                ellswift_b.as_c_ptr(),
190                secret_key.as_c_ptr(),
191                party.to_ffi_int(),
192                ffi::secp256k1_ellswift_xdh_hash_function_bip324,
193                data.as_c_ptr() as *mut c_void,
194            );
195            debug_assert_eq!(ret, 1);
196        }
197        ElligatorSwiftSharedSecret(shared_secret)
198    }

200    /// Computes a shared secret, just like `shared_secret`, but with a custom hash function
201    /// for computing the shared secret. For compatibility with other libraries, you should
202    /// use `shared_secret` instead, which is already compatible with BIP324.
203    /// The hash function takes three arguments: the shared point, and the `ElligatorSwift`
204    /// encodings of the two parties and returns a 32-byte shared secret.
205    pub fn shared_secret_with_hasher<F>(
206        ellswift_a: ElligatorSwift,
207        ellswift_b: ElligatorSwift,
208        secret_key: SecretKey,
209        party: ElligatorSwiftParty,
210        mut hash_function: F,
211    ) -> ElligatorSwiftSharedSecret
212    where
213        F: FnMut([u8; 32], [u8; 64], [u8; 64]) -> ElligatorSwiftSharedSecret,
214    {
215        let mut shared_secret = [0u8; 32];
216        let hashfp = hash_callback::<F>;
217        unsafe {
218            let ret = ffi::secp256k1_ellswift_xdh(
219                ffi::secp256k1_context_no_precomp,
220                shared_secret.as_mut_c_ptr(),
221                ellswift_a.0.as_c_ptr(),
222                ellswift_b.0.as_c_ptr(),
223                secret_key.as_c_ptr(),
224                party.to_ffi_int(),
225                Some(hashfp),
226                &mut hash_function as *mut F as *mut c_void,
227            );
228            debug_assert_eq!(ret, 1);
229        }
230        ElligatorSwiftSharedSecret(shared_secret)
231    }

233    /// Encodes a public key into an `ElligatorSwift` encoding
234    fn encode(pk: PublicKey) -> ElligatorSwift {
235        let mut ell_out = [0u8; constants::ELLSWIFT_ENCODING_SIZE];
236        unsafe {
237            let ret = ffi::secp256k1_ellswift_encode(
238                ffi::secp256k1_context_no_precomp,
239                ell_out.as_mut_c_ptr(),
240                pk.as_c_ptr(),
241                [0u8; 32].as_ptr(),
242            );
243            debug_assert_eq!(ret, 1);
244        }
245        ElligatorSwift(ffi::ElligatorSwift::from_array(ell_out))
246    }

248    /// Decodes an `ElligatorSwift` encoding into a [`PublicKey`].
249    pub(crate) fn decode(ell: ElligatorSwift) -> PublicKey {
250        unsafe {
251            let mut pk = ffi::PublicKey::new();
252
253            let ret = ffi::secp256k1_ellswift_decode(
254                ffi::secp256k1_context_no_precomp,
255                pk.as_mut_c_ptr(),
256                ell.as_c_ptr(),
257            );
258            debug_assert_eq!(ret, 1);
259            PublicKey::from(pk)
260        }
261    }

284/// Represents which party we are in the ECDH.
285///
286/// Here `A` is the initiator and `B` is the responder.
287///
288/// this context, "we" means the party that possesses the secret key passed to
289/// [`ElligatorSwift::shared_secret`].
290///
291/// This distinction is important because the different parties compute different
292/// hashes of the shared secret.
293#[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord, Hash)]
294pub enum ElligatorSwiftParty {
295    /// We are the initiator of the ECDH
296    A,
297    /// We are the responder of the ECDH
298    B,
299}
300
301impl ElligatorSwiftParty {
302    fn to_ffi_int(self) -> c_int {
303        match self {
304            ElligatorSwiftParty::A => 0,
305            ElligatorSwiftParty::B => 1,
306        }
307    }
308}

310impl FromStr for ElligatorSwift {
311    fn from_str(hex: &str) -> Result<Self, Self::Err> {
312        let mut ser = [0u8; 64];
313        let parsed = from_hex(hex, &mut ser);
314        match parsed {
315            Ok(64) => Ok(ElligatorSwift::from_array(ser)),
316            _ => Err(Error::InvalidEllSwift),
317        }
318    }
319    type Err = Error;
320}
