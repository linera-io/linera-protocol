43/// [`sol!`](sol_types::sol!) `macro_rules!` wrapper to set import attributes.
44///
45/// See [`sol!`](sol_types::sol!) for the actual macro documentation.
46#[cfg(all(not(doc), feature = "sol-types"))] // Show the actual macro in docs.
47#[macro_export]
48macro_rules! sol {
49    ($($t:tt)*) => {
50        $crate::sol_types::sol! {
51            #![sol(alloy_sol_types = $crate::sol_types, alloy_contract = $crate::contract)]
52            $($t)*
53        }
54    };
55}
