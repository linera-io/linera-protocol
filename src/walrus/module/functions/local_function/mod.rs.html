50    /// Construct a new `LocalFunction`.
51    ///
52    /// Validates the given function body and constructs the `Instr` IR at the
53    /// same time.
54    #[allow(clippy::too_many_arguments)]
55    pub(crate) fn parse(
56        module: &Module,
57        indices: &IndicesToIds,
58        id: FunctionId,
59        ty: TypeId,
60        args: Vec<LocalId>,
61        body: wasmparser::FunctionBody<'_>,
62        on_instr_pos: Option<&(dyn Fn(&usize) -> InstrLocId + Sync + Send + 'static)>,
63        mut validator: FuncValidator<ValidatorResources>,
64    ) -> Result<LocalFunction> {
65        let code_address_offset = module.funcs.code_section_offset;
66        let function_body_size = body.range().end - body.range().start;
67        let function_body_size_bit =
68            (std::mem::size_of::<usize>() as u32 * 8 - function_body_size.leading_zeros() - 1) / 7
69                + 1;
70        let mut func = LocalFunction {
71            builder: FunctionBuilder::without_entry(ty),
72            args,
73            instruction_mapping: Vec::new(),
74            original_range: Some(Range::<usize> {
75                start: body.range().start - code_address_offset - (function_body_size_bit as usize),
76                end: body.range().end - code_address_offset,
77            }),
78        };
79
80        let result: Vec<_> = module.types.get(ty).results().to_vec();
81        let result = result.into_boxed_slice();
82
83        let controls = &mut context::ControlStack::new();
84
85        let mut ctx = ValidationContext::new(module, indices, id, &mut func, controls);
86
87        let ty = module.types.find_for_function_entry(&result).expect(
88            "the function entry type should have already been created before parsing the body",
89        );
90        let entry = ctx.push_control_with_ty(BlockKind::FunctionEntry, ty);
91        let mut instruction_mapping = BTreeMap::new();
92        ctx.func.builder.entry = Some(entry);
93        let mut ops_reader = body.get_operators_reader()?;
94        while !ops_reader.eof() {
95            let pos = ops_reader.original_position();
96            let inst: Operator = ops_reader.read()?;
97            let loc = if let Some(ref on_instr_pos) = on_instr_pos {
98                on_instr_pos(&pos)
99            } else {
100                InstrLocId::new(pos as u32)
101            };
102            validator.op(pos, &inst)?;
103            append_instruction(&mut ctx, inst, loc);
104            instruction_mapping.insert(pos - code_address_offset, loc);
105        }
106        ctx.func.instruction_mapping = instruction_mapping.into_iter().collect();
107        ops_reader.finish()?;
108
109        debug_assert!(ctx.controls.is_empty());
110
111        Ok(func)
112    }

217    /// Emit this function's compact locals declarations.
218    pub(crate) fn emit_locals(
219        &self,
220        module: &Module,
221        indices: &crate::emit::IdsToIndices,
222    ) -> EmitLocalsResult {
223        let used_set = self.used_locals();
224        let mut used_locals = used_set.iter().cloned().collect::<Vec<_>>();
225        // Sort to ensure we assign local indexes deterministically, and
226        // everything is distinct so we can use a faster unstable sort.
227        used_locals.sort_unstable();
228
229        // NB: Use `BTreeMap` to make compilation deterministic by emitting
230        // types in the same order
231        let mut ty_to_locals = BTreeMap::new();
232        let args = self.args.iter().cloned().collect::<IdHashSet<_>>();
233
234        // Partition all locals by their type as we'll create at most one entry
235        // for each type. Skip all arguments to the function because they're
236        // handled separately.
237        for local in used_locals.iter() {
238            if !args.contains(local) {
239                let ty = module.locals.get(*local).ty();
240                ty_to_locals.entry(ty).or_insert_with(Vec::new).push(*local);
241            }
242        }
243
244        let mut local_map = IdHashMap::default();
245        local_map.reserve(used_locals.len());
246
247        // Allocate an index to all the function arguments, as these are all
248        // unconditionally used and are implicit locals in wasm.
249        let mut idx = 0;
250        for &arg in self.args.iter() {
251            local_map.insert(arg, idx);
252            idx += 1;
253        }
254
255        // Assign an index to all remaining locals
256        for (_, locals) in ty_to_locals.iter() {
257            for l in locals {
258                local_map.insert(*l, idx);
259                idx += 1;
260            }
261        }
262
263        // Use our type map to emit a compact representation of all locals now
264        (
265            ty_to_locals
266                .iter()
267                .map(|(ty, locals)| (locals.len() as u32, ty.to_wasmencoder_type(indices)))
268                .collect(),
269            used_set,
270            local_map,
271        )
272    }

286fn block_result_tys(ctx: &ValidationContext, ty: wasmparser::BlockType) -> Result<Box<[ValType]>> {
287    match ty {
288        wasmparser::BlockType::Type(ty) => ValType::from_wasmparser_type(ty, ctx.indices),
289        wasmparser::BlockType::FuncType(idx) => {
290            let ty = ctx.indices.get_type(idx)?;
291            Ok(ctx.module.types.results(ty).into())
292        }
293        wasmparser::BlockType::Empty => Ok(Box::new([])),
294    }
295}

297fn block_param_tys(ctx: &ValidationContext, ty: wasmparser::BlockType) -> Result<Box<[ValType]>> {
298    match ty {
299        wasmparser::BlockType::Type(_) => Ok([][..].into()),
300        wasmparser::BlockType::FuncType(idx) => {
301            let ty = ctx.indices.get_type(idx)?;
302            Ok(ctx.module.types.params(ty).into())
303        }
304        wasmparser::BlockType::Empty => Ok(Box::new([])),
305    }

308fn append_instruction(ctx: &mut ValidationContext, inst: Operator, loc: InstrLocId) {
309    // NB. there's a lot of `unwrap()` here in this function, and that's because
310    // the `Operator` was validated above to already be valid, so everything
311    // should succeed.
312    use crate::ir::ExtendedLoad::*;
313
314    log::trace!("validate instruction: {:?}", inst);
315
316    let const_ = |ctx: &mut ValidationContext, value| {
317        ctx.alloc_instr(Const { value }, loc);
318    };
319
320    let unop = |ctx: &mut ValidationContext, op| {
321        ctx.alloc_instr(Unop { op }, loc);
322    };
323    let binop = |ctx: &mut ValidationContext, op| {
324        ctx.alloc_instr(Binop { op }, loc);
325    };
326    let ternop = |ctx: &mut ValidationContext, op| {
327        ctx.alloc_instr(TernOp { op }, loc);
328    };
329
330    let mem_arg = |ctx: &mut ValidationContext, arg: &wasmparser::MemArg| -> (MemoryId, MemArg) {
331        (
332            ctx.indices.get_memory(arg.memory).unwrap(),
333            MemArg {
334                align: 1 << (arg.align as i32),
335                offset: arg.offset,
336            },
337        )
338    };
339
340    let load = |ctx: &mut ValidationContext, arg, kind| {
341        let (memory, arg) = mem_arg(ctx, &arg);
342        ctx.alloc_instr(Load { arg, kind, memory }, loc);
343    };
344
345    let store = |ctx: &mut ValidationContext, arg, kind| {
346        let (memory, arg) = mem_arg(ctx, &arg);
347        ctx.alloc_instr(Store { arg, kind, memory }, loc);
348    };
349
350    let atomicrmw = |ctx: &mut ValidationContext, arg, op, width| {
351        let (memory, arg) = mem_arg(ctx, &arg);
352        ctx.alloc_instr(
353            AtomicRmw {
354                arg,
355                memory,
356                op,
357                width,
358            },
359            loc,
360        );
361    };
362
363    let cmpxchg = |ctx: &mut ValidationContext, arg, width| {
364        let (memory, arg) = mem_arg(ctx, &arg);
365        ctx.alloc_instr(Cmpxchg { arg, memory, width }, loc);
366    };
367
368    let load_simd = |ctx: &mut ValidationContext, arg, kind| {
369        let (memory, arg) = mem_arg(ctx, &arg);
370        ctx.alloc_instr(LoadSimd { memory, arg, kind }, loc);
371    };
372    match inst {
373        Operator::Call { function_index } => {
374            let func = ctx.indices.get_func(function_index).unwrap();
375            ctx.alloc_instr(Call { func }, loc);
376        }
377        Operator::CallIndirect {
378            type_index,
379            table_index,
380        } => {
381            let type_id = ctx.indices.get_type(type_index).unwrap();
382            let table = ctx.indices.get_table(table_index).unwrap();
383            ctx.alloc_instr(CallIndirect { table, ty: type_id }, loc);
384        }
385        Operator::LocalGet { local_index } => {
386            let local = ctx.indices.get_local(ctx.func_id, local_index).unwrap();
387            ctx.alloc_instr(LocalGet { local }, loc);
388        }
389        Operator::LocalSet { local_index } => {
390            let local = ctx.indices.get_local(ctx.func_id, local_index).unwrap();
391            ctx.alloc_instr(LocalSet { local }, loc);
392        }
393        Operator::LocalTee { local_index } => {
394            let local = ctx.indices.get_local(ctx.func_id, local_index).unwrap();
395            ctx.alloc_instr(LocalTee { local }, loc);
396        }
397        Operator::GlobalGet { global_index } => {
398            let global = ctx.indices.get_global(global_index).unwrap();
399            ctx.alloc_instr(GlobalGet { global }, loc);
400        }
401        Operator::GlobalSet { global_index } => {
402            let global = ctx.indices.get_global(global_index).unwrap();
403            ctx.alloc_instr(GlobalSet { global }, loc);
404        }
405        Operator::I32Const { value } => const_(ctx, Value::I32(value)),
406        Operator::I64Const { value } => const_(ctx, Value::I64(value)),
407        Operator::F32Const { value } => const_(ctx, Value::F32(f32::from_bits(value.bits()))),
408        Operator::F64Const { value } => const_(ctx, Value::F64(f64::from_bits(value.bits()))),
409        Operator::V128Const { value } => {

415        Operator::I32Ne => binop(ctx, BinaryOp::I32Ne),
