10/// The id of a table.
11pub type TableId = Id<Table>;
12
13/// A table in the wasm.
14#[derive(Debug)]
15pub struct Table {
16    id: TableId,
17    /// Whether or not this is a 64-bit table.
18    pub table64: bool,
19    /// The initial size of this table
20    pub initial: u64,
21    /// The maximum size of this table
22    pub maximum: Option<u64>,
23    /// The type of the elements in this table
24    pub element_ty: RefType,
25    /// The initialization expression for the table elements.
26    /// Required for non-defaultable (non-nullable) element types.
27    /// If None, the table is initialized with null references.
28    pub init: Option<ConstExpr>,
29    /// Whether or not this table is imported, and if so what imports it.
30    pub import: Option<ImportId>,
31    /// Active data segments that will be used to initialize this memory.
32    pub elem_segments: IdHashSet<Element>,
33    /// The name of this table, used for debugging purposes in the `name`
34    /// custom section.
35    pub name: Option<String>,
36}
37
38impl Tombstone for Table {}
39
40impl Table {
41    /// Get this table's id.
42    pub fn id(&self) -> TableId {
43        self.id
44    }
45}

47/// The set of tables in this module.
48#[derive(Debug, Default)]
49pub struct ModuleTables {
50    /// The arena containing this module's tables.
51    arena: TombstoneArena<Table>,
52}
53
54impl ModuleTables {
55    /// Adds a new imported table to this list of tables
56    pub fn add_import(
57        &mut self,
58        table64: bool,
59        initial: u64,
60        maximum: Option<u64>,
61        element_ty: RefType,
62        import: ImportId,
63    ) -> TableId {
64        let id = self.arena.next_id();
65        self.arena.alloc(Table {
66            id,
67            table64,
68            initial,
69            maximum,
70            element_ty,
71            init: None,
72            import: Some(import),
73            elem_segments: Default::default(),
74            name: None,
75        })
76    }
77
78    /// Construct a new table, that does not originate from any of the input
79    /// wasm tables.
80    pub fn add_local(
81        &mut self,
82        table64: bool,
83        initial: u64,
84        maximum: Option<u64>,
85        element_ty: RefType,
86    ) -> TableId {
87        self.add_local_with_init(table64, initial, maximum, element_ty, None)
88    }
89
90    /// Construct a new table with an optional initializer expression.
91    /// The initializer is required for non-defaultable (non-nullable) element types.
92    pub fn add_local_with_init(
93        &mut self,
94        table64: bool,
95        initial: u64,
96        maximum: Option<u64>,
97        element_ty: RefType,
98        init: Option<ConstExpr>,
99    ) -> TableId {
100        let id = self.arena.next_id();
101        let id2 = self.arena.alloc(Table {
102            id,
103            table64,
104            initial,
105            maximum,
106            element_ty,
107            init,
108            import: None,
109            elem_segments: Default::default(),
110            name: None,
111        });
112        debug_assert_eq!(id, id2);
113        id
114    }
115
116    /// Returns the actual table associated with an ID
117    pub fn get(&self, table: TableId) -> &Table {
118        &self.arena[table]
119    }
120
121    /// Returns the actual table associated with an ID
122    pub fn get_mut(&mut self, table: TableId) -> &mut Table {
123        &mut self.arena[table]
124    }
125
126    /// Removes a table from this module.
127    ///
128    /// It is up to you to ensure that any potential references to the deleted
129    /// table are also removed, eg `call_indirect` expressions and exports, etc.
130    pub fn delete(&mut self, id: TableId) {
131        self.arena.delete(id);
132    }
133
134    /// Iterates over all tables in this section.
135    pub fn iter(&self) -> impl Iterator<Item = &Table> {
136        self.arena.iter().map(|p| p.1)
137    }
138
139    /// Finds a unique function table in a module.
140    ///
141    /// Modules produced by compilers like LLVM typically have one function
142    /// table for indirect function calls. This function will look for a single
143    /// function table inside this module, and return that if found. If no
144    /// function tables are present `None` will be returned
145    ///
146    /// # Errors
147    ///
148    /// Returns an error if there are two function tables in this module
149    pub fn main_function_table(&self) -> Result<Option<TableId>> {
150        let mut tables = self.iter().filter(|t| t.element_ty == RefType::FUNCREF);
151        let id = match tables.next() {
152            Some(t) => t.id(),
153            None => return Ok(None),
154        };
155        if tables.next().is_some() {
156            bail!("module contains more than one function table");
157        }
158        Ok(Some(id))
159    }
160
161    /// Iterates over all tables in this section.
162    pub fn iter_mut(&mut self) -> impl Iterator<Item = &mut Table> {
163        self.arena.iter_mut().map(|p| p.1)
164    }
165}

167impl Module {
168    /// Construct a new, empty set of tables for a module.
169    pub(crate) fn parse_tables(
170        &mut self,
171        section: wasmparser::TableSectionReader,
172        ids: &mut IndicesToIds,
173    ) -> Result<()> {
174        log::debug!("parse table section");
175        for t in section {
176            let t = t?;
177            let init = match t.init {
178                wasmparser::TableInit::RefNull => None,
179                wasmparser::TableInit::Expr(expr) => Some(ConstExpr::eval(&expr, ids)?),
180            };
181            let id = self.tables.add_local_with_init(
182                t.ty.table64,
183                t.ty.initial,
184                t.ty.maximum,
185                RefType::from_wasmparser(t.ty.element_type, ids, 0)?,
186                init,
187            );
188            ids.push_table(id);
189        }
190        Ok(())
191    }
192}

194impl Emit for ModuleTables {
195    fn emit(&self, cx: &mut EmitContext) {
196        log::debug!("emit table section");
197
198        let mut wasm_table_section = wasm_encoder::TableSection::new();
199
200        // Skip imported tables because those are emitted in the import section.
201        let tables = self.iter().filter(|t| t.import.is_none()).count();
202        if tables == 0 {
203            return;
204        }
205
206        for table in self.iter().filter(|t| t.import.is_none()) {
207            cx.indices.push_table(table.id());
208
209            let table_type = wasm_encoder::TableType {
210                table64: table.table64,
211                minimum: table.initial,
212                maximum: table.maximum,
213                element_type: table.element_ty.to_wasmencoder_ref_type(cx.indices),
214                shared: false,
215            };
216
217            match &table.init {
218                Some(init_expr) => {
219                    wasm_table_section
220                        .table_with_init(table_type, &init_expr.to_wasmencoder_type(cx));
221                }
222                None => {
223                    wasm_table_section.table(table_type);
224                }
225            }
226        }
227
228        cx.wasm_module.section(&wasm_table_section);
229    }
