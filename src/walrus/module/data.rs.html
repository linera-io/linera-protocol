10/// A passive element segment identifier
11pub type DataId = Id<Data>;
12
13/// A data segment.
14///
15/// Every data segment has an associated value. This value gets copied into a
16/// memory. It is either automatically copied into a specific memory at Wasm
17/// instantiation time (active data segments) or dynamically copied into a
18/// memory (or memories) via the `memory.init` instruction (passive data
19/// segments). See the `kind` member and `DataKind` type for more details on the
20/// active/passive distinction.
21#[derive(Debug)]
22pub struct Data {
23    id: DataId,
24    /// What kind of data segment is this? Passive or active?
25    pub kind: DataKind,
26    /// The data payload of this data segment.
27    pub value: Vec<u8>,
28    /// The name of this data, used for debugging purposes in the `name`
29    /// custom section.
30    pub name: Option<String>,
31}
32
33/// The kind of data segment: passive or active.
34#[derive(Debug)]
35pub enum DataKind {
36    /// An active data segment that is automatically initialized at some address
37    /// in a static memory.
38    Active {
39        /// The memory that this active data segment will be automatically
40        /// initialized in.
41        memory: MemoryId,
42        /// The memory offset where this active data segment will be automatically
43        /// initialized.
44        offset: ConstExpr,
45    },
46    /// A passive data segment that must be manually initialized at a dynamic
47    /// address via the `memory.init` instruction (perhaps multiple times in
48    /// multiple different memories) and then manually freed when it's no longer
49    /// needed via the `data.drop` instruction.
50    Passive,
51}
52
53impl Tombstone for Data {
54    fn on_delete(&mut self) {
55        self.value = Vec::new();
56    }
57}
58
59impl Data {
60    /// Returns the id of this passive data segment
61    pub fn id(&self) -> DataId {
62        self.id
63    }
64
65    /// Is this a passive data segment?
66    pub fn is_passive(&self) -> bool {
67        matches!(self.kind, DataKind::Passive)
68    }
69}

71/// All passive data sections of a wasm module, used to initialize memories via
72/// various instructions.
73#[derive(Debug, Default)]
74pub struct ModuleData {
75    arena: TombstoneArena<Data>,
76}
77
78impl ModuleData {
79    /// Get an element associated with an ID
80    pub fn get(&self, id: DataId) -> &Data {
81        &self.arena[id]
82    }
83
84    /// Get an element associated with an ID
85    pub fn get_mut(&mut self, id: DataId) -> &mut Data {
86        &mut self.arena[id]
87    }
88
89    /// Delete a passive data segment from this module.
90    ///
91    /// It is up to you to ensure that all references to the deleted segment are
92    /// removed, eg `memory.init` and `data.drop` expressions.
93    pub fn delete(&mut self, id: DataId) {
94        self.arena.delete(id);
95    }
96
97    /// Get a shared reference to this module's passive elements.
98    pub fn iter(&self) -> impl Iterator<Item = &Data> {
99        self.arena.iter().map(|(_, f)| f)
100    }
101
102    /// Add a data segment
103    pub fn add(&mut self, kind: DataKind, value: Vec<u8>) -> DataId {
104        let id = self.arena.next_id();
105        let id2 = self.arena.alloc(Data {
106            id,
107            kind,
108            value,
109            name: None,
110        });
111        debug_assert_eq!(id, id2);
112        id
113    }
114
115    // Note that this is inaccordance with the upstream bulk memory proposal to
116    // WebAssembly and isn't currently part of the WebAssembly standard.
117    pub(crate) fn emit_data_count(&self, cx: &mut EmitContext) {
118        #[cfg(feature = "parallel")]
119        use rayon::iter::ParallelIterator;
120
121        if self.arena.len() == 0 {
122            return;
123        }
124
125        let mut count = 0;
126        let mut any_passive = false;
127
128        for data in self.iter() {
129            cx.indices.set_data_index(data.id(), count);
130            count += 1;
131            any_passive |= data.is_passive();
132        }
133
134        // We only emit the `DataCount` section if there are passive data
135        // segments, or `data.drop` or `memory.init` instructions that use
136        // (passive or active) data segments. Yes! You can use `data.drop` and
137        // `memory.init` with active data segments, it just results in a runtime
138        // error.
139        //
140        // The key is that we don't want to generate this section for MVP Wasm,
141        // which has niether passive data segments, nor the `data.drop` and
142        // `memory.init` instructions.
143        let funcs = &cx.module.funcs;
144        if any_passive
145            || maybe_parallel!(funcs.(iter_local | par_iter_local))
146                .any(|(_, f)| !f.used_data_segments().is_empty())
147        {
148            cx.wasm_module
149                .section(&wasm_encoder::DataCountSection { count });
150        }
151    }
152}

154impl Module {
155    /// Called when we see the data section section to create an id for all data
156    /// indices
157    ///
158    /// Note that during function parsing all data indices less than `count` are
159    /// considered valid, and it's only afterwards that we discover whether
160    /// they're actually passive or not, and that property is checked during
161    /// validation.
162    pub(crate) fn reserve_data(&mut self, count: u32, ids: &mut IndicesToIds) {
163        log::debug!("reserving space for {} data segments", count);
164        for _ in 0..count {
165            ids.push_data(self.data.arena.alloc_with_id(|id| Data {
166                id,
167                // NB: We'll update the `value` and `kind` once we actually
168                // parse the data segments.
169                value: Vec::new(),
170                kind: DataKind::Passive,
171                name: None,
172            }));
173        }
174    }
175
176    /// Parses a raw wasm section into a fully-formed `ModuleData` instance.
177    pub(crate) fn parse_data(
178        &mut self,
179        section: wasmparser::DataSectionReader,
180        ids: &mut IndicesToIds,
181    ) -> Result<()> {
182        log::debug!("parse data section");
183        let preallocated = self.data.arena.len() > 0;
184        for (i, segment) in section.into_iter().enumerate() {
185            let segment = segment?;
186
187            // If we had the `DataCount` section, then we already pre-allocated
188            // a data segment. Otherwise, allocate one now.
189            let id = if preallocated {
190                ids.get_data(i as u32)?
191            } else {
192                let id = self.data.arena.alloc_with_id(|id| Data {
193                    id,
194                    value: Vec::new(),
195                    kind: DataKind::Passive,
196                    name: None,
197                });
198
199                ids.push_data(id);
200
201                id
202            };
203            let data = self.data.get_mut(id);
204
205            match segment.kind {
206                wasmparser::DataKind::Passive => {
207                    data.value = segment.data.to_vec();
208                    data.kind = DataKind::Passive;
209                }
210                wasmparser::DataKind::Active {
211                    memory_index,
212                    offset_expr,
213                } => {
214                    data.value = segment.data.to_vec();
215
216                    let memory_id = ids.get_memory(memory_index)?;
217                    let memory = self.memories.get_mut(memory_id);
218                    memory.data_segments.insert(data.id);
219
220                    let offset = ConstExpr::eval(&offset_expr, ids)
221                        .with_context(|| format!("failed to evaluate the offset of data {}", i))?;
222
223                    if memory.memory64 {
224                        match offset {
225                            ConstExpr::Value(Value::I64(_)) => {}
226                            ConstExpr::Global(global)
227                                if self.globals.get(global).ty == ValType::I64 => {}
228                            // Extended const expressions are already validated
229                            // by wasmparser to produce the correct type.
230                            ConstExpr::Extended(_) => {}
231                            _ => bail!(
232                                "data {} is active for 64-bit memory but has non-i64 offset",
233                                i
234                            ),
235                        }
236                    } else {
237                        match offset {
238                            ConstExpr::Value(Value::I32(_)) => {}
239                            ConstExpr::Global(global)
240                                if self.globals.get(global).ty == ValType::I32 => {}
241                            // Extended const expressions are already validated
242                            // by wasmparser to produce the correct type.
243                            ConstExpr::Extended(_) => {}
244                            _ => bail!(
245                                "data {} is active for 32-bit memory but has non-i32 offset",
246                                i
247                            ),
248                        }
249                    }
250                    data.kind = DataKind::Active {
251                        memory: memory_id,
252                        offset,
253                    }
254                }
255            }
256        }
257        Ok(())
258    }
259}

261impl Emit for ModuleData {
262    fn emit(&self, cx: &mut EmitContext) {
263        log::debug!("emit data section");
264
265        if self.arena.len() == 0 {
266            return;
267        }
268
269        // The encodings here are with respect to the bulk memory proposal, but
270        // should be backwards compatible with the current MVP WebAssembly spec
271        // so long as the only memory 0 is used.
272        let mut wasm_data_section = wasm_encoder::DataSection::new();
273        for data in self.iter() {
274            match &data.kind {
275                DataKind::Passive => {
276                    wasm_data_section.passive(data.value.clone());
277                }
278                DataKind::Active { memory, offset } => {
279                    wasm_data_section.active(
280                        cx.indices.get_memory_index(*memory),
281                        &offset.to_wasmencoder_type(cx),
282                        data.value.clone(),
283                    );
284                }
285            }
286        }
287        cx.wasm_module.section(&wasm_data_section);
288    }
