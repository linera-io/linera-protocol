10/// Set of all root used items in a wasm module.
11#[derive(Debug, Default)]
12pub struct Roots {
13    tables: Vec<TableId>,
14    funcs: Vec<FunctionId>,
15    globals: Vec<GlobalId>,
16    memories: Vec<MemoryId>,
17    tags: Vec<TagId>,
18    datas: Vec<DataId>,
19    elements: Vec<ElementId>,
20    used: Used,
21}
22
23impl Roots {
24    /// Creates a new set of empty roots.
25    pub fn new() -> Roots {
26        Roots::default()
27    }
28
29    /// Adds a new function to the set of roots
30    pub fn push_func(&mut self, func: FunctionId) -> &mut Roots {
31        if self.used.funcs.insert(func) {
32            log::trace!("function is used: {:?}", func);
33            self.funcs.push(func);
34        }
35        self
36    }
37
38    /// Adds a new table to the set of roots
39    pub fn push_table(&mut self, table: TableId) -> &mut Roots {
40        if self.used.tables.insert(table) {
41            log::trace!("table is used: {:?}", table);
42            self.tables.push(table);
43        }
44        self
45    }
46
47    /// Adds a new memory to the set of roots
48    pub fn push_memory(&mut self, memory: MemoryId) -> &mut Roots {
49        if self.used.memories.insert(memory) {
50            log::trace!("memory is used: {:?}", memory);
51            self.memories.push(memory);
52        }
53        self
54    }
55
56    /// Adds a new global to the set of roots
57    pub fn push_global(&mut self, global: GlobalId) -> &mut Roots {
58        if self.used.globals.insert(global) {
59            log::trace!("global is used: {:?}", global);
60            self.globals.push(global);
61        }
62        self
63    }
64
65    /// Adds a new tag to the set of roots
66    pub fn push_tag(&mut self, tag: TagId) -> &mut Roots {
67        if self.used.tags.insert(tag) {
68            log::trace!("tag is used: {:?}", tag);
69            self.tags.push(tag);
70        }
71        self
72    }
73
74    fn push_data(&mut self, data: DataId) -> &mut Roots {
75        if self.used.data.insert(data) {
76            log::trace!("data is used: {:?}", data);
77            self.datas.push(data);
78        }
79        self
80    }
81
82    fn push_element(&mut self, element: ElementId) -> &mut Roots {
83        if self.used.elements.insert(element) {
84            log::trace!("element is used: {:?}", element);
85            self.elements.push(element);
86        }
87        self
88    }
89
90    fn push_const_expr(&mut self, expr: &ConstExpr) {
91        match expr {
92            ConstExpr::Global(g) => {
93                self.push_global(*g);
94            }
95            ConstExpr::RefFunc(f) => {
96                self.push_func(*f);
97            }
98            ConstExpr::RefNull(ref_type) => {
99                mark_ref_type_used(&mut self.used, ref_type);
100            }
101            ConstExpr::Value(_) => {}
102            ConstExpr::Extended(ops) => {
103                for op in ops {
104                    match op {
105                        ConstOp::GlobalGet(g) => {
106                            self.push_global(*g);
107                        }
108                        ConstOp::RefFunc(f) => {
109                            self.push_func(*f);
110                        }
111                        ConstOp::StructNew(ty)
112                        | ConstOp::StructNewDefault(ty)
113                        | ConstOp::ArrayNew(ty)
114                        | ConstOp::ArrayNewDefault(ty) => {
115                            self.used.types.insert(*ty);
116                        }
117                        ConstOp::ArrayNewFixed { ty, .. } => {
118                            self.used.types.insert(*ty);
119                        }
120                        ConstOp::RefNull(ref_type) => {
121                            mark_ref_type_used(&mut self.used, ref_type);
122                        }
123                        _ => {}
124                    }
125                }
126            }
127        }
128    }
129}

131/// Finds the things within a module that are used.
132///
133/// This is useful for implementing something like a linker's `--gc-sections` so
134/// that our emitted `.wasm` binaries are small and don't contain things that
135/// are not used.
136#[derive(Debug, Default)]
137pub struct Used {
138    /// The module's used tables.
139    pub tables: IdHashSet<Table>,
140    /// The module's used types.
141    pub types: IdHashSet<Type>,
142    /// The module's used functions.
143    pub funcs: IdHashSet<Function>,
144    /// The module's used globals.
145    pub globals: IdHashSet<Global>,
146    /// The module's used memories.
147    pub memories: IdHashSet<Memory>,
148    /// The module's used tags.
149    pub tags: IdHashSet<Tag>,
150    /// The module's used passive element segments.
151    pub elements: IdHashSet<Element>,
152    /// The module's used passive data segments.
153    pub data: IdHashSet<Data>,
154}
155
156impl Used {
157    /// Construct a new `Used` set for the given module.
158    pub fn new(module: &Module) -> Used {
159        log::debug!("starting to calculate used set");
160        let mut stack = Roots::default();
161
162        // All exports are roots
163        for export in module.exports.iter() {
164            match export.item {
165                ExportItem::Function(f) => stack.push_func(f),
166                ExportItem::Table(t) => stack.push_table(t),
167                ExportItem::Memory(m) => stack.push_memory(m),
168                ExportItem::Global(g) => stack.push_global(g),
169                ExportItem::Tag(t) => stack.push_tag(t),
170            };
171        }
172
173        // The start function is an implicit root as well
174        if let Some(f) = module.start {
175            stack.push_func(f);
176        }
177
178        // Initialization of memories or tables is a side-effectful operation
179        // because they can be out-of-bounds, so keep all active segments.
180        for data in module.data.iter() {
181            if let DataKind::Active { .. } = &data.kind {
182                stack.push_data(data.id());
183            }
184        }
185        for elem in module.elements.iter() {
186            match elem.kind {
187                // Active segments are rooted because they initialize imported
188                // tables.
189                ElementKind::Active { table, .. } => {
190                    if module.tables.get(table).import.is_some() {
191                        stack.push_element(elem.id());
192                    }
193                }
194                // Declared segments can probably get gc'd but for now we're
195                // conservative and we root them
196                ElementKind::Declared => {
197                    stack.push_element(elem.id());
198                }
199                ElementKind::Passive => {}
200            }
201        }
202
203        // And finally ask custom sections for their roots
204        for (_id, section) in module.customs.iter() {
205            section.add_gc_roots(&mut stack);
206        }
207
208        // Iteratively visit all items until our stack is empty
209        while !stack.funcs.is_empty()
210            || !stack.tables.is_empty()
211            || !stack.memories.is_empty()
212            || !stack.globals.is_empty()
213            || !stack.tags.is_empty()
214            || !stack.datas.is_empty()
215            || !stack.elements.is_empty()
216        {
217            while let Some(f) = stack.funcs.pop() {
218                let func = module.funcs.get(f);
219                stack.used.types.insert(func.ty());
220
221                match &func.kind {
222                    FunctionKind::Local(func) => {
223                        let mut visitor = UsedVisitor {
224                            stack: &mut stack,
225                            local_ids: Vec::new(),
226                        };
227                        dfs_in_order(&mut visitor, func, func.entry_block());
228                        // Mark types referenced by used locals (non-arg locals
229                        // may have concrete ref types that need to survive GC).
230                        for local_id in visitor.local_ids {
231                            let ty = module.locals.get(local_id).ty();
232                            mark_val_type_used(&mut stack.used, &ty);
233                        }
234                    }
235                    FunctionKind::Import(_) => {}
236                    FunctionKind::Uninitialized(_) => unreachable!(),
237                }
238            }
239
240            while let Some(t) = stack.tables.pop() {
241                let table = module.tables.get(t);
242                mark_ref_type_used(&mut stack.used, &table.element_ty);
243                // Process the table's init expression for function/type refs.
244                if let Some(init) = &table.init {
245                    stack.push_const_expr(init);
246                }
247                for elem in table.elem_segments.iter() {
248                    stack.push_element(*elem);
249                }
250            }
251
252            while let Some(t) = stack.globals.pop() {
253                let global = module.globals.get(t);
254                mark_val_type_used(&mut stack.used, &global.ty);
255                match &global.kind {
256                    GlobalKind::Import(_) => {}
257                    GlobalKind::Local(expr) => {
258                        stack.push_const_expr(expr);
259                    }
260                }
261            }
262
263            while let Some(t) = stack.tags.pop() {
264                let tag = module.tags.get(t);
265                stack.used.types.insert(tag.ty);
266            }
267
268            while let Some(t) = stack.memories.pop() {
269                for data in &module.memories.get(t).data_segments {
270                    stack.push_data(*data);
271                }
272            }
273
274            while let Some(d) = stack.datas.pop() {
275                let d = module.data.get(d);
276                if let DataKind::Active { memory, offset } = &d.kind {
277                    stack.push_memory(*memory);
278                    stack.push_const_expr(offset);
279                }
280            }
281
282            while let Some(e) = stack.elements.pop() {
283                let e = module.elements.get(e);
284                if let ElementItems::Functions(function_ids) = &e.items {
285                    function_ids.iter().for_each(|f| {
286                        stack.push_func(*f);
287                    });
288                }
289                if let ElementItems::Expressions(ref_type, items) = &e.items {
290                    // Mark the element segment's declared type as used.
291                    mark_ref_type_used(&mut stack.used, ref_type);
292                    // Process all items regardless of element type.
293                    for item in items {
294                        stack.push_const_expr(item);
295                    }
296                }
297                if let ElementKind::Active { offset, table } = &e.kind {
298                    stack.push_const_expr(offset);
299                    stack.push_table(*table);
300                }
301            }
302        }
303
304        // Transitively mark all types referenced by used types. GC types
305        // (structs, arrays) and function types with ref-typed parameters can
306        // reference other types via HeapType::Concrete(TypeId). Without this
307        // transitive closure, the GC pass would delete types that are still
308        // needed during emission.
309        //
310        // Additionally, rec groups are atomic: if any member is used, all
311        // members must survive. We expand each newly-used type to include
312        // its entire rec group.
313        {
314            let mut type_stack: Vec<TypeId> = stack.used.types.iter().copied().collect();
315            let mut refs = Vec::new();
316            while let Some(type_id) = type_stack.pop() {
317                // Expand rec group: if this type belongs to a rec group,
318                // mark all sibling types in the group as used too.
319                if let Some(rg) = module.types.rec_group_for_type(type_id) {
320                    for &sibling in &rg.types {
321                        if stack.used.types.insert(sibling) {
322                            type_stack.push(sibling);
323                        }
324                    }
325                }
326
327                let ty = module.types.get(type_id);
328                refs.clear();
329                ty.referenced_types(&mut refs);
330                for &ref_id in &refs {
331                    if stack.used.types.insert(ref_id) {
332                        type_stack.push(ref_id);
333                    }
334                }
335            }
336        }
337
338        // Wabt seems to have weird behavior where a `data` segment, if present
339        // even if passive, requires a `memory` declaration. Our GC pass is
340        // pretty aggressive and if you have a passive data segment and only
341        // `data.drop` instructions you technically don't need the `memory`.
342        // Let's keep `wabt` passing though and just say that if there are data
343        // segments kept, but no memories, then we try to add the first memory,
344        // if any, to the used set.
345        if !stack.used.data.is_empty() && stack.used.memories.is_empty() {
346            if let Some(mem) = module.memories.iter().next() {
347                stack.used.memories.insert(mem.id());
348            }
349        }
350
351        stack.used
352    }
353}

355struct UsedVisitor<'a> {
356    stack: &'a mut Roots,
357    local_ids: Vec<LocalId>,
358}
359
360impl<'expr> Visitor<'expr> for UsedVisitor<'_> {
361    fn visit_function_id(&mut self, &func: &FunctionId) {
362        self.stack.push_func(func);
363    }
364
365    fn visit_memory_id(&mut self, &m: &MemoryId) {
366        self.stack.push_memory(m);
367    }
368
369    fn visit_global_id(&mut self, &g: &GlobalId) {
370        self.stack.push_global(g);
371    }
372
373    fn visit_table_id(&mut self, &t: &TableId) {
374        self.stack.push_table(t);
375    }
376
377    fn visit_type_id(&mut self, &t: &TypeId) {
378        self.stack.used.types.insert(t);
379    }
380
381    fn visit_local_id(&mut self, &id: &LocalId) {
382        self.local_ids.push(id);
383    }
384
385    fn visit_data_id(&mut self, &d: &DataId) {
386        self.stack.push_data(d);
387    }
388
389    fn visit_element_id(&mut self, &e: &ElementId) {
390        self.stack.push_element(e);
391    }
392
393    fn visit_tag_id(&mut self, &tag: &TagId) {
394        self.stack.push_tag(tag);
395    }
396
397    // RefType/ValType fields on Select and RefNull are still
398    // #[walrus(skip_visit)] so we handle them manually here.
399
400    fn visit_select(&mut self, instr: &Select) {
401        if let Some(ty) = &instr.ty {
402            mark_val_type_used(&mut self.stack.used, ty);
403        }
404    }
405
406    fn visit_ref_null(&mut self, instr: &RefNull) {
407        mark_ref_type_used(&mut self.stack.used, &instr.ty);
408    }
409}

411/// If the `RefType`'s heap type is concrete, mark that type as used.
412fn mark_ref_type_used(used: &mut Used, ref_type: &RefType) {
413    if let HeapType::Concrete(type_id) | HeapType::Exact(type_id) = ref_type.heap_type {
414        used.types.insert(type_id);
415    }
416}
417
418/// If the `ValType` is a ref type with a concrete heap type, mark it as used.
419fn mark_val_type_used(used: &mut Used, val_type: &crate::ValType) {
420    if let crate::ValType::Ref(ref_type) = val_type {
421        mark_ref_type_used(used, ref_type);
422    }
