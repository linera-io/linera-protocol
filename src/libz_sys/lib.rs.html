6// Macro for variances between zlib-ng in native mode and either zlib or zlib-ng in zlib compat
7// mode. Note in particular that zlib-ng in compat mode does *not* use the zng case.
8#[cfg(not(zng))]
9macro_rules! if_zng {
10    ($_zng:tt, $not_zng:tt) => {
11        $not_zng
12    };
13}
