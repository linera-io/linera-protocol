1//! This module is responsible for precomputing all possible replica lists for given replication strategies.
2//! Having the replicas precomputed allows to avoid costly calculations each time a request is sent.
3//! To get a replica list for a given token the driver can simply do a lookup in the precomputed data.
4//! Precomputing is realized in an efficient manner, we are leveraging an advantageous property
5//! of `SimpleStrategy` and datacenter-local `NetworkTopologyStrategy` (with some caveats described
6//! below) - replica lists for replication factor `n` are prefixes of replica lists for replication
7//! factor `n + 1`. This enables us to only compute replica lists for maximal replication factor
8//! found in replication strategies (in `NTS`, we are interested in maximal `rf`
9//! specified for each datacenter). It is a big optimization - other solution is
10//! to compute those lists for each strategy used in cluster.
11//!
12//! Notes on Network Topology Strategy precomputation:
13//! The optimization mentioned above works only if requested `replication factor` is <= `rack count`.
14

28/// Takes care of precomputing all possible replica lists for given replication strategies.
29/// Uses `ReplicationInfo` to calculate the replica lists for all tokens present in the ring,
30/// and stores them for future use.
31#[derive(Clone, Debug)]
32pub(crate) struct PrecomputedReplicas {
33    /// Precomputed replicas for SimpleStrategy, kept in a ring for efficient access.
34    global_replicas: PrecomputedReplicasRing,
35
36    /// Precomputed replicas for each datacenter, used in NetworkTopologyStrategy.
37    datacenter_replicas: HashMap<String, DatacenterPrecomputedReplicas>,
38}
39
40/// Keeps a precomputed replica list for each token present in the ring.
41#[derive(Debug, Clone)]
42struct PrecomputedReplicasRing {
43    /// Replica lists are kept in `TokenRing` to allow efficient access with a token as a key.
44    replicas_for_token: TokenRing<Replicas>,
45
46    /// Maximal replication factor up to which replica lists are computed.
47    max_rep_factor: usize,
48}
49
50#[derive(Clone, Debug)]
51struct DatacenterPrecomputedReplicas {
52    /// Holds replica lists that were computed using `replication factor` <= `rack count`.
53    /// Replica lists computed for `rf` = `n` are prefixes of replica lists computed for
54    /// `rf` = `rack count` (if `n` <= `rack count`), so they can be "compressed".
55    /// This compression relies on computing the longest lists only.
56    compressed_replica_ring: Option<PrecomputedReplicasRing>,
57
58    /// Holds the replica list computed for `replication factors` > `rack count`.
59    above_rack_count_replica_rings: HashMap<usize, TokenRing<Replicas>>,
60}
61
62impl DatacenterPrecomputedReplicas {
63    fn get_replica_ring_for_rf(&self, replication_factor: usize) -> Option<&TokenRing<Replicas>> {
64        if let Some(compressed) = &self.compressed_replica_ring {
65            if compressed.max_rep_factor >= replication_factor {
66                return Some(&compressed.replicas_for_token);
67            }
68        }
69
70        self.above_rack_count_replica_rings.get(&replication_factor)
71    }
72}
73
74impl PrecomputedReplicas {
75    /// Performs the replica precomputation and creates an instance of `PrecomputedReplicas`.
76    /// It extracts the maximal replication factor for which to compute from the given keyspace strategies.
77    /// The replicas are first precomputed for the global ring and then individually for each datacenter.
78    /// Uses the optimization specified in module description.
79    /// The computation could potentially take some time so it shouldn't be done in an async function.
80    pub(crate) fn compute<'a>(
81        replication_data: &ReplicationInfo,
82        keyspace_strategies: impl Iterator<Item = &'a Strategy>,
83    ) -> PrecomputedReplicas {
84        // Each ring will precompute for at least this RF
85        let min_precomputed_rep_factor: usize = 1;
86
87        // The maximum replication factor over those used with Simple strategy.
88        let mut max_global_repfactor: usize = min_precomputed_rep_factor;
89        let mut dc_repfactors: HashMap<&'a str, BTreeSet<usize>> = HashMap::new();
90
91        for strategy in keyspace_strategies {
92            match strategy {
93                Strategy::SimpleStrategy { replication_factor } => {
94                    max_global_repfactor = cmp::max(max_global_repfactor, *replication_factor)
95                }
96                Strategy::NetworkTopologyStrategy {
97                    datacenter_repfactors,
98                } => {
99                    for (dc_name, dc_repfactor) in datacenter_repfactors {
100                        let repfactors: &mut BTreeSet<usize> =
101                            dc_repfactors.entry(dc_name).or_default();
102
103                        repfactors.insert(*dc_repfactor);
104                    }
105                }
106                Strategy::LocalStrategy => {} // RF=1
107                Strategy::Other { .. } => {}  // Can't precompute for custom strategies
108            }
109        }
110
111        let global_replicas_iter = replication_data.get_global_ring().iter().map(|(token, _)| {
112            let cur_replicas: Replicas = replication_data
113                .simple_strategy_replicas(*token, max_global_repfactor)
114                .cloned()
115                .collect();
116            (*token, cur_replicas)
117        });
118        let global_replicas = PrecomputedReplicasRing {
119            replicas_for_token: TokenRing::new(global_replicas_iter),
120            max_rep_factor: max_global_repfactor,
121        };
122
123        let mut datacenter_replicas: HashMap<String, DatacenterPrecomputedReplicas> =
124            HashMap::new();
125        for (dc_name, repfactors) in dc_repfactors {
126            let dc_rep_data = match replication_data.get_datacenters().get(dc_name) {
127                Some(dc_rep_data) => dc_rep_data,
128                None => continue,
129            };
130
131            let rack_count = dc_rep_data.get_rack_count();
132            let compressed_replica_ring_rf = repfactors.range(..=rack_count).next_back();
133            let replica_ring_rf_above_rack_count = repfactors.range((rack_count + 1)..);
134
135            let produce_replica_ring_iter = |rf| {
136                let ring_iter = dc_rep_data.get_dc_ring().iter().map(|(token, _)| {
137                    let cur_replicas: Replicas = replication_data
138                        .nts_replicas_in_datacenter(*token, dc_name, rf)
139                        .cloned()
140                        .collect();
141                    (*token, cur_replicas)
142                });
143
144                TokenRing::new(ring_iter)
145            };
146
147            let compressed_replica_ring =
148                compressed_replica_ring_rf.map(|rf| PrecomputedReplicasRing {
149                    replicas_for_token: produce_replica_ring_iter(*rf),
150                    max_rep_factor: *rf,
151                });
152
153            let above_rack_count_replica_rings = replica_ring_rf_above_rack_count
154                .map(|rf| (*rf, produce_replica_ring_iter(*rf)))
155                .collect();
156
157            let dc_precomputed_replicas = DatacenterPrecomputedReplicas {
158                compressed_replica_ring,
159                above_rack_count_replica_rings,
160            };
161
162            datacenter_replicas.insert(dc_name.to_string(), dc_precomputed_replicas);
163        }
164
165        PrecomputedReplicas {
166            global_replicas,
167            datacenter_replicas,
168        }
169    }
170
171    /// Gets the precomputed replica list for a given SimpleStrategy.
172    /// When requested replication factor is larger than the maximal one (detected during the
173    /// precomputation stage), `None` is returned.
174    pub(crate) fn get_precomputed_simple_strategy_replicas(
175        &self,
176        token: Token,
177        replication_factor: usize,
178    ) -> Option<&[Arc<Node>]> {
179        if replication_factor > self.global_replicas.max_rep_factor {
180            return None;
181        }
182
183        let precomputed_token_replicas = self
184            .global_replicas
185            .replicas_for_token
186            .get_elem_for_token(token)?;
187        let result_len: usize = cmp::min(precomputed_token_replicas.len(), replication_factor);
188        Some(&precomputed_token_replicas[..result_len])
189    }
190
191    /// Gets the precomputed replica list for a given NetworkTopologyStrategy.
192    /// When requested replication factor is larger than the maximal one (detected during the
193    /// precomputation stage), `None` is returned.
194    /// If a provided datacenter does not exist, `None` is returned.
195    pub(crate) fn get_precomputed_network_strategy_replicas(
196        &self,
197        token: Token,
198        dc_name: &str,
199        dc_replication_factor: usize,
200    ) -> Option<&[Arc<Node>]> {
201        let precomputed_replicas_ring = self
202            .datacenter_replicas
203            .get(dc_name)?
204            .get_replica_ring_for_rf(dc_replication_factor)?;
205
206        let precomputed_replicas = precomputed_replicas_ring.get_elem_for_token(token)?;
207        let result_len: usize = cmp::min(precomputed_replicas.len(), dc_replication_factor);
208
209        Some(&precomputed_replicas[..result_len])
210    }
211}

228    #[tokio::test]
229    async fn test_simple_stategy() {
230        setup_tracing();
231        let mut metadata = mock_metadata_for_token_aware_tests();
232        metadata.keyspaces = [(
233            "SimpleStrategy{rf=2}".into(),
234            Ok(Keyspace {
235                strategy: Strategy::SimpleStrategy {
236                    replication_factor: 2,
237                },
238                tables: HashMap::new(),
239                views: HashMap::new(),
240                user_defined_types: HashMap::new(),
241            }),
242        )]
243        .iter()
244        .cloned()
245        .collect();
246
247        let ring = create_ring(&metadata);
248        let replication_info = ReplicationInfo::new(ring);
249        let precomputed_replicas = PrecomputedReplicas::compute(
250            &replication_info,
251            metadata
252                .keyspaces
253                .values()
254                .map(|keyspace| &keyspace.as_ref().unwrap().strategy),
255        );
256
257        let check = |token, replication_factor, expected_node_ids| {
258            let replicas = precomputed_replicas
259                .get_precomputed_simple_strategy_replicas(Token::new(token), replication_factor);
260
261            let ids: Vec<u16> = replicas
262                .unwrap()
263                .iter()
264                .map(|node| node.address.port())
265                .collect();
266
267            assert_eq!(ids, expected_node_ids);
268        };
269
270        check(160, 0, vec![]);
271        check(160, 1, vec![F]);
272        check(160, 2, vec![F, A]);
273        assert_eq!(
274            precomputed_replicas.get_precomputed_simple_strategy_replicas(Token::new(160), 3),
275            None
276        );
277
278        check(200, 1, vec![F]);
279        check(200, 2, vec![F, A]);
280
281        check(701, 1, vec![E]);
282        check(701, 2, vec![E, G]);
283    }
