16pub(crate) struct RequestSpan {
17    span: tracing::Span,
18    speculative_executions: AtomicUsize,
19}
20
21impl RequestSpan {
22    pub(crate) fn new_query(contents: &str) -> Self {
23        use tracing::field::Empty;
24
25        let span = trace_span!(
26            "Request",
27            kind = "unprepared",
28            contents = contents,
29            //
30            request_size = Empty,
31            result_size = Empty,
32            result_rows = Empty,
33            replicas = Empty,
34            shard = Empty,
35            speculative_executions = Empty,
36        );
37
38        Self {
39            span,
40            speculative_executions: 0.into(),
41        }
42    }
43
44    pub(crate) fn new_prepared<'ps, 'spec: 'ps>(
45        partition_key: Option<impl Iterator<Item = (&'ps [u8], &'ps ColumnSpec<'spec>)> + Clone>,
46        token: Option<Token>,
47        request_size: usize,
48    ) -> Self {
49        use tracing::field::Empty;
50
51        let span = trace_span!(
52            "Request",
53            kind = "prepared",
54            partition_key = Empty,
55            token = Empty,
56            //
57            request_size = request_size,
58            result_size = Empty,
59            result_rows = Empty,
60            replicas = Empty,
61            shard = Empty,
62            speculative_executions = Empty,
63        );
64
65        if let Some(partition_key) = partition_key {
66            span.record(
67                "partition_key",
68                tracing::field::display(
69                    format_args!("{}", partition_key_displayer(partition_key),),
70                ),
71            );
72        }
73        if let Some(token) = token {
74            span.record("token", token.value());
75        }
76
77        Self {
78            span,
79            speculative_executions: 0.into(),
80        }
81    }
82

104    pub(crate) fn record_shard_id(&self, conn: &Connection) {
105        if let Some(info) = conn.get_shard_info() {

110    pub(crate) fn record_raw_rows_fields(&self, raw_rows: &RawMetadataAndRawRows) {
111        self.span
112            .record("raw_result_size", raw_rows.metadata_and_rows_bytes_size());
113    }
114
115    pub(crate) fn record_result_fields(&self, query_result: &QueryResult) {
116        if let Some(raw_metadata_and_rows) = query_result.raw_metadata_and_rows() {
117            self.record_raw_rows_fields(raw_metadata_and_rows);
118        }
119    }
120
121    pub(crate) fn record_replicas<'a>(
122        &'a self,
123        replicas: impl Iterator<Item = (impl Borrow<Arc<Node>> + 'a, Shard)> + Clone,
124    ) {
125        struct Replica<N>(N, Shard);
126        impl<N> Display for Replica<N>
127        where
128            N: Borrow<Arc<Node>>,
129        {
130            fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
131                let Self(node, shard) = self;
132                write!(f, "{}-shard{}", node.borrow().address.ip(), shard)
133            }
134        }
135        self.span.record(
136            "replicas",
137            tracing::field::display(
138                replicas
139                    .map(|(node, shard)| Replica(node, shard))
140                    .format(", "),
141            ),
142        );
143    }
144
145    pub(crate) fn record_request_size(&self, size: usize) {
146        self.span.record("request_size", size);
147    }
148
149    pub(crate) fn inc_speculative_executions(&self) {
150        self.speculative_executions.fetch_add(1, Ordering::Relaxed);
151    }
152
153    pub(crate) fn span(&self) -> &tracing::Span {
154        &self.span
155    }
156}
157
158impl Drop for RequestSpan {
159    fn drop(&mut self) {
160        self.span.record(
161            "speculative_executions",
162            self.speculative_executions.load(Ordering::Relaxed),
163        );
164    }
165}

167fn partition_key_displayer<'ps, 'res, 'spec: 'ps>(
168    mut pk_values_iter: impl Iterator<Item = (&'ps [u8], &'ps ColumnSpec<'spec>)> + 'res + Clone,
169) -> impl Display + 'res {
170    std::iter::from_fn(move || {
171        pk_values_iter
172            .next()
173            .map(|(mut cell, spec)| deser_cql_value(spec.typ(), &mut cell))
174    })
175    .map(|c| match c {
176        Ok(c) => Either::Left(c),
177        Err(_) => Either::Right("<decoding error>"),
178    })
179    .format(", ")
180}
