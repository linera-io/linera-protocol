2//! and receive events from the cluster.

4use std::fmt::Write as _;

6use std::sync::Arc;

11use crate::client::pager::QueryPager;

17/// The single connection used to fetch metadata and receive events from the cluster.
18pub(super) struct ControlConnection {
19    conn: Arc<Connection>,
20    /// The custom server-side timeout set for requests executed on the control connection.
21    overriden_serverside_timeout: Option<Duration>,
22}
23
24impl ControlConnection {
25    pub(super) fn new(conn: Arc<Connection>) -> Self {
26        Self {
27            conn,
28            overriden_serverside_timeout: None,
29        }
30    }
31
32    /// Sets the custom server-side timeout set for requests executed on the control connection.
33    pub(super) fn override_serverside_timeout(self, overriden_timeout: Option<Duration>) -> Self {
34        Self {
35            overriden_serverside_timeout: overriden_timeout,
36            ..self
37        }
38    }
39
40    pub(super) fn get_connect_address(&self) -> SocketAddr {
41        self.conn.get_connect_address()
42    }
43
44    /// Returns true iff the target node is a ScyllaDB node (and not a, e.g., Cassandra node).
45    pub(super) fn is_to_scylladb(&self) -> bool {
46        self.conn.get_shard_info().is_some()
47    }
48
49    /// Appends the custom server-side timeout to the statement string, if such custom timeout
50    /// is provided and we are connected to ScyllaDB (since custom timeouts is ScyllaDB-only feature).
51    fn maybe_append_timeout_override(&self, statement: &mut Statement) {
52        if let Some(timeout) = self.overriden_serverside_timeout {
53            if self.is_to_scylladb() {
54                // SAFETY: io::fmt::Write impl for String is infallible.
55                write!(
56                    statement.contents,
57                    " USING TIMEOUT {}ms",
58                    timeout.as_millis()
59                )
60                .unwrap()
61            }
62        }
63    }
64
65    /// Executes a query and fetches its results over multiple pages, using
66    /// the asynchronous iterator interface.
67    pub(super) async fn query_iter(
68        &self,
69        mut statement: Statement,
70    ) -> Result<QueryPager, NextRowError> {
71        self.maybe_append_timeout_override(&mut statement);
72        Arc::clone(&self.conn).query_iter(statement).await
73    }
74
75    pub(crate) async fn prepare(
76        &self,
77        mut statement: Statement,
78    ) -> Result<PreparedStatement, RequestAttemptError> {
79        self.maybe_append_timeout_override(&mut statement);
80        self.conn.prepare(&statement).await
81    }
82
83    /// Executes a prepared statements and fetches its results over multiple pages, using
84    /// the asynchronous iterator interface.
85    pub(crate) async fn execute_iter(
86        &self,
87        prepared_statement: PreparedStatement,
88        values: SerializedValues,
89    ) -> Result<QueryPager, NextRowError> {
90        Arc::clone(&self.conn)
91            .execute_iter(prepared_statement, values)
92            .await
93    }
94}

97mod tests {
98    use std::collections::HashMap;
99    use std::net::SocketAddr;
100    use std::sync::Arc;
101    use std::time::Duration;

103    use scylla_proxy::{

119    /// Tests that ControlConnection enforces the provided custom timeout
120    /// iff ScyllaDB is the target node (else ignores the custom timeout).
121    #[tokio::test]
122    #[ntest::timeout(2000)]
123    async fn test_custom_timeouts() {
124        setup_tracing();
125
126        let proxy_addr = SocketAddr::new(scylla_proxy::get_exclusive_local_address(), 9042);
127
128        let (feedback_tx, mut feedback_rx) = mpsc::unbounded_channel();
129
130        let make_rules = |shard_info: Option<ShardInfo>| {
131            vec![
132                // OPTIONS -> SUPPORTED rule
133                RequestRule(
134                    Condition::RequestOpcode(RequestOpcode::Options),
135                    RequestReaction::forge_response(Arc::new(move |frame: RequestFrame| {
136                        ResponseFrame::forged_supported(frame.params, &{
137                            let mut options = HashMap::new();
138                            if let Some(shard_info) = shard_info.as_ref() {
139                                shard_info.add_to_options(&mut options);
140                            }
141                            options
142                        })
143                        .unwrap()
144                    })),
145                ),
146                // STARTUP -> READY rule
147                // REGISTER -> READY rule
148                RequestRule(
149                    Condition::or(
150                        Condition::RequestOpcode(RequestOpcode::Startup),
151                        Condition::RequestOpcode(RequestOpcode::Register),
152                    ),
153                    RequestReaction::forge_response(Arc::new(move |frame: RequestFrame| {
154                        ResponseFrame::forged_ready(frame.params)
155                    })),
156                ),
157                // Metadata query feedback rule
158                RequestRule(
159                    Condition::or(
160                        Condition::RequestOpcode(RequestOpcode::Query),
161                        Condition::RequestOpcode(RequestOpcode::Prepare),
162                    ),
163                    RequestReaction::forge()
164                        .server_error()
165                        .with_feedback_when_performed(feedback_tx),
166                ),
167            ]
168        };
169
170        let mut proxy = Proxy::builder()
171            .with_node(
172                Node::builder()
173                    .proxy_address(proxy_addr)
174                    .request_rules(make_rules.clone()(None))
175                    .build_dry_mode(),
176            )
177            .build()
178            .run()
179            .await
180            .unwrap();
181
182        const QUERY_STR: &str = "SELECT host_id FROM system.local";
183
184        fn expected_query_body(dur: Duration) -> String {
185            format!("{} USING TIMEOUT {}ms", QUERY_STR, dur.as_millis())
186        }
187
188        fn contains_subslice(slice: &[u8], subslice: &[u8]) -> bool {
189            slice
190                .windows(subslice.len())
191                .any(|window| window == subslice)
192        }
193
194        async fn assert_no_custom_timeout(
195            feedback_rx: &mut mpsc::UnboundedReceiver<(RequestFrame, Option<u16>)>,
196        ) {
197            let (frame, _) = feedback_rx.recv().await.unwrap();
198            let clause = "USING TIMEOUT";
199            assert!(
200                !contains_subslice(&frame.body, clause.as_bytes()),
201                "slice {:?} does contain subslice {:?}",
202                &frame.body,
203                clause,
204            );
205        }
206
207        async fn assert_custom_timeout(
208            feedback_rx: &mut mpsc::UnboundedReceiver<(RequestFrame, Option<u16>)>,
209            dur: Duration,
210        ) {
211            let (frame, _) = feedback_rx.recv().await.unwrap();
212            let expected = expected_query_body(dur);
213            assert!(
214                contains_subslice(&frame.body, expected.as_bytes()),
215                "slice {:?} does not contain subslice {:?}",
216                &frame.body,
217                expected,
218            );
219        }
220
221        async fn assert_custom_timeout_iff_scylladb(
222            feedback_rx: &mut mpsc::UnboundedReceiver<(RequestFrame, Option<u16>)>,
223            dur: Duration,
224            connected_to_scylladb: bool,
225        ) {
226            if connected_to_scylladb {
227                assert_custom_timeout(feedback_rx, dur).await;
228            } else {
229                assert_no_custom_timeout(feedback_rx).await;
230            }
231        }
232
233        async fn test_metadata_timeouts(
234            proxy_addr: SocketAddr,
235            feedback_rx: &mut mpsc::UnboundedReceiver<(RequestFrame, Option<u16>)>,
236        ) {
237            let (conn, _error_receiver) = open_connection(
238                &UntranslatedEndpoint::ContactPoint(ResolvedContactPoint {
239                    address: proxy_addr,
240                    datacenter: None,
241                }),
242                None,
243                &Default::default(),
244            )
245            .await
246            .unwrap();
247
248            let connected_to_scylladb = conn.get_shard_info().is_some();
249            let conn_with_default_timeout = ControlConnection::new(Arc::new(conn));
250
251            // No custom timeout set.
252            {
253                conn_with_default_timeout
254                    .query_iter(QUERY_STR.into())
255                    .await
256                    .unwrap_err();
257
258                assert_no_custom_timeout(feedback_rx).await;
259
260                conn_with_default_timeout
261                    .prepare(QUERY_STR.into())
262                    .await
263                    .unwrap_err();
264
265                assert_no_custom_timeout(feedback_rx).await;
266            }
267
268            // Custom timeout set, so it should be set in query strings iff the target node is ScyllaDB.
269            {
270                let custom_timeout = Duration::from_millis(2137);
271                let conn_with_custom_timeout =
272                    conn_with_default_timeout.override_serverside_timeout(Some(custom_timeout));
273
274                conn_with_custom_timeout
275                    .query_iter(QUERY_STR.into())
276                    .await
277                    .unwrap_err();
278
279                assert_custom_timeout_iff_scylladb(
280                    feedback_rx,
281                    custom_timeout,
282                    connected_to_scylladb,
283                )
284                .await;
285
286                conn_with_custom_timeout
287                    .prepare(QUERY_STR.into())
288                    .await
289                    .unwrap_err();
290
291                assert_custom_timeout_iff_scylladb(
292                    feedback_rx,
293                    custom_timeout,
294                    connected_to_scylladb,
295                )
296                .await;
297            }
298        }
299
300        // Simulated non-ScyllaDB node (no sharding info in SUPPORTED)
301        {
302            // Proxy starts without shards. No additional config needed.
303
304            test_metadata_timeouts(proxy_addr, &mut feedback_rx).await;
305        }
306
307        // Simulated ScyllaDB node (sharding info present in SUPPORTED)
308        {
309            proxy.running_nodes[0].change_request_rules(Some(make_rules(Some(ShardInfo {
310                shard: 2,
311                nr_shards: NonZeroU16::new(4).unwrap(),
312                msb_ignore: 1,
313            }))));
314
315            test_metadata_timeouts(proxy_addr, &mut feedback_rx).await;
316        }
317
318        let _ = proxy.finish().await;
319    }
