24/// Cluster manages up to date information and connections to database nodes.
25/// All state can be accessed by cloning Arc<ClusterState> in the `state` field
26pub(crate) struct Cluster {
27    // `ArcSwap<ClusterState>` is wrapped in `Arc` to support sharing cluster state
28    // between `Cluster` and `ClusterWorker`
29    state: Arc<ArcSwap<ClusterState>>,
30
31    refresh_channel: tokio::sync::mpsc::Sender<RefreshRequest>,
32    use_keyspace_channel: tokio::sync::mpsc::Sender<UseKeyspaceRequest>,
33
34    _worker_handle: RemoteHandle<()>,
35}

37/// Enables printing [Cluster] struct in a neat way, by skipping the rather useless
38/// print of channels state and printing [ClusterState] neatly.
39pub(crate) struct ClusterNeatDebug<'a>(pub(crate) &'a Cluster);
40impl std::fmt::Debug for ClusterNeatDebug<'_> {
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        let cluster = self.0;
43        f.debug_struct("Cluster")
44            .field("data", &ClusterStateNeatDebug(&cluster.state.load()))
45            .finish_non_exhaustive()
46    }
47}

49// Works in the background to keep the cluster updated
50struct ClusterWorker {
51    // Cluster state to keep updated:
52    cluster_state: Arc<ArcSwap<ClusterState>>,
53
54    // Cluster connections
55    metadata_reader: MetadataReader,
56    pool_config: PoolConfig,
57
58    // To listen for refresh requests
59    refresh_channel: tokio::sync::mpsc::Receiver<RefreshRequest>,
60
61    // Channel used to receive use keyspace requests
62    use_keyspace_channel: tokio::sync::mpsc::Receiver<UseKeyspaceRequest>,
63
64    // Channel used to receive server events
65    server_events_channel: tokio::sync::mpsc::Receiver<Event>,
66
67    // Channel used to receive signals that control connection is broken
68    control_connection_repair_channel: tokio::sync::broadcast::Receiver<()>,
69
70    // Channel used to receive info about new tablets from custom payload in responses
71    // sent by server.
72    tablets_channel: tokio::sync::mpsc::Receiver<(TableSpec<'static>, RawTablet)>,
73
74    // Keyspace send in "USE <keyspace name>" when opening each connection
75    used_keyspace: Option<VerifiedKeyspaceName>,
76
77    // The host filter determines towards which nodes we should open
78    // connections
79    host_filter: Option<Arc<dyn HostFilter>>,
80
81    // This value determines how frequently the cluster
82    // worker will refresh the cluster metadata
83    cluster_metadata_refresh_interval: Duration,
84
85    #[cfg(feature = "metrics")]
86    metrics: Arc<Metrics>,
87}

102    pub(crate) async fn new(
103        known_nodes: Vec<InternalKnownNode>,
104        pool_config: PoolConfig,
105        keyspaces_to_fetch: Vec<String>,
106        fetch_schema_metadata: bool,
107        metadata_request_serverside_timeout: Option<Duration>,
108        host_filter: Option<Arc<dyn HostFilter>>,
109        cluster_metadata_refresh_interval: Duration,
110        tablet_receiver: tokio::sync::mpsc::Receiver<(TableSpec<'static>, RawTablet)>,
111        #[cfg(feature = "metrics")] metrics: Arc<Metrics>,
112    ) -> Result<Cluster, NewSessionError> {
113        let (refresh_sender, refresh_receiver) = tokio::sync::mpsc::channel(32);
114        let (use_keyspace_sender, use_keyspace_receiver) = tokio::sync::mpsc::channel(32);
115        let (server_events_sender, server_events_receiver) = tokio::sync::mpsc::channel(32);
116        let (control_connection_repair_sender, control_connection_repair_receiver) =
117            tokio::sync::broadcast::channel(32);
118
119        let mut metadata_reader = MetadataReader::new(
120            known_nodes,
121            control_connection_repair_sender,
122            pool_config.connection_config.clone(),
123            metadata_request_serverside_timeout,
124            server_events_sender,
125            keyspaces_to_fetch,
126            fetch_schema_metadata,
127            &host_filter,
128            #[cfg(feature = "metrics")]
129            Arc::clone(&metrics),
130        )
131        .await?;
132
133        let metadata = metadata_reader.read_metadata(true).await?;
134        let cluster_state = ClusterState::new(
135            metadata,
136            &pool_config,
137            &HashMap::new(),
138            &None,
139            host_filter.as_deref(),
140            TabletsInfo::new(),
141            &HashMap::new(),
142            #[cfg(feature = "metrics")]
143            &metrics,
144        )
145        .await;
146        cluster_state.wait_until_all_pools_are_initialized().await;
147        let cluster_state: Arc<ArcSwap<ClusterState>> =
148            Arc::new(ArcSwap::from(Arc::new(cluster_state)));
149
150        let worker = ClusterWorker {
151            cluster_state: cluster_state.clone(),
152
153            metadata_reader,
154            pool_config,
155
156            refresh_channel: refresh_receiver,
157            server_events_channel: server_events_receiver,
158            control_connection_repair_channel: control_connection_repair_receiver,
159            tablets_channel: tablet_receiver,
160
161            use_keyspace_channel: use_keyspace_receiver,
162            used_keyspace: None,
163
164            host_filter,
165            cluster_metadata_refresh_interval,
166
167            #[cfg(feature = "metrics")]
168            metrics,
169        };
170
171        let (fut, worker_handle) = worker.work().remote_handle();
172        tokio::spawn(fut);
173
174        let result = Cluster {
175            state: cluster_state,
176            refresh_channel: refresh_sender,
177            use_keyspace_channel: use_keyspace_sender,
178            _worker_handle: worker_handle,
179        };
180
181        Ok(result)
182    }

188    pub(crate) async fn refresh_metadata(&self) -> Result<(), MetadataError> {
189        let (response_sender, response_receiver) = tokio::sync::oneshot::channel();
190
191        self.refresh_channel
192            .send(RefreshRequest {
193                response_chan: response_sender,
194            })
195            .await
196            .expect("Bug in Cluster::refresh_metadata sending");
197        // Other end of this channel is in ClusterWorker, can't be dropped while we have &self to Cluster with _worker_handle
198
199        response_receiver
200            .await
201            .expect("Bug in Cluster::refresh_metadata receiving")
202        // ClusterWorker always responds
203    }

224impl ClusterWorker {
225    pub(crate) async fn work(mut self) {
226        use tokio::time::Instant;
227
228        let control_connection_repair_duration = Duration::from_secs(1); // Attempt control connection repair every second
229        let mut last_refresh_time = Instant::now();
230        let mut control_connection_works = true;
231
232        loop {
233            let mut cur_request: Option<RefreshRequest> = None;
234
235            // Wait until it's time for the next refresh
236            let sleep_until: Instant = last_refresh_time
237                .checked_add(if control_connection_works {
238                    self.cluster_metadata_refresh_interval
239                } else {
240                    control_connection_repair_duration
241                })
242                .unwrap_or_else(Instant::now);
243
244            let mut tablets = Vec::new();
245
246            let sleep_future = tokio::time::sleep_until(sleep_until);
247            tokio::pin!(sleep_future);
248
249            tokio::select! {
250                _ = sleep_future => {},
251                recv_res = self.refresh_channel.recv() => {
252                    match recv_res {
253                        Some(request) => cur_request = Some(request),
254                        None => return, // If refresh_channel was closed then cluster was dropped, we can stop working
255                    }
256                }
257                tablets_count = self.tablets_channel.recv_many(&mut tablets, TABLET_CHANNEL_SIZE) => {
258                    tracing::trace!("Performing tablets update - received {} tablets", tablets_count);
259                    if tablets_count == 0 {
260                        // If the channel was closed then the cluster was dropped, we can stop working
261                        return;
262                    }
263                    // The current tablet implementation collects tablet feedback in a channel
264                    // and then clones the whole ClusterState, updates it with new tablets and replaces
265                    // the old ClusterState - this update procedure happens below.
266                    // This fits the general model of how ClusterState is handled in the driver:
267                    // - ClusterState remains a "simple" struct - without locks etc (apart from Node).
268                    // - Topology information update is similar to tablet update - it creates a new ClusterState
269                    //   and replaces the old one.
270                    // The disadvantage is that we need to have 2 copies of ClusterState, but this happens
271                    // anyway during topology update.
272                    //
273                    // An alternative solution would be to use some synchronization primitives to update tablet info
274                    // in place. This solution avoids ClusterState cloning but:
275                    // - ClusterState would be much more complicated
276                    // - Requires using locks in hot path (when sending request)
277                    // - Makes maintenance (which happens during topology update) more complicated and error-prone.
278                    //
279                    // I decided to stick with the approach that fits with the driver.
280                    // Apart from the reasons above, it is much easier to reason about concurrency etc
281                    // when reading the code in other parts of the driver.
282                    let mut new_cluster_state: ClusterState = self.cluster_state.load().as_ref().clone();
283                    new_cluster_state.update_tablets(tablets);
284                    self.update_cluster_state(Arc::new(new_cluster_state));
285
286                    continue;
287                }
288                recv_res = self.server_events_channel.recv() => {
289                    if let Some(event) = recv_res {
290                        debug!("Received server event: {:?}", event);
291                        match event {
292                            Event::TopologyChange(_) => (), // Refresh immediately
293                            Event::StatusChange(status) => {
294                                // If some node went down/up, update it's marker and refresh
295                                // later as planned.
296
297                                // TODO: Tracking status using events is unreliable because of
298                                // the possibility of losing events when control connection is broken.
299                                // Maybe a better thing to do here is to treat those events as hints?
300                                // What I mean by that?
301                                // - Don't store the status at all.
302                                // - When receiving down event, and the driver still sees the node
303                                //   as connected, then try to send a keepalive query to its connections.
304                                // - When receiving up event, and we have no connections to the node,
305                                //   then try to open new connections.
306
307                                match status {
308                                    StatusChangeEvent::Down(addr) => self.change_node_down_marker(addr, true),
309                                    StatusChangeEvent::Up(addr) => self.change_node_down_marker(addr, false),
310                                }
311                                continue;
312                            },
313                            _ => continue, // Don't go to refreshing
314                        }
315                    } else {
316                        // If server_events_channel was closed, than MetadataReader was dropped,
317                        // so we can probably stop working too
318                        return;
319                    }
320                }
321                recv_res = self.use_keyspace_channel.recv() => {
322                    match recv_res {
323                        Some(request) => {
324                            self.used_keyspace = Some(request.keyspace_name.clone());
325
326                            let cluster_state = self.cluster_state.load_full();
327                            let use_keyspace_future = Self::handle_use_keyspace_request(cluster_state, request);
328                            tokio::spawn(use_keyspace_future);
329                        },
330                        None => return, // If use_keyspace_channel was closed then cluster was dropped, we can stop working
331                    }
332
333                    continue; // Don't go to refreshing, wait for the next event
334                }
335                recv_res = self.control_connection_repair_channel.recv() => {
336                    match recv_res {
337                        Ok(()) => {
338                            // The control connection was broken. Acknowledge that and start attempting to reconnect.
339                            // The first reconnect attempt will be immediate (by attempting metadata refresh below),
340                            // and if it does not succeed, then `control_connection_works` will be set to `false`,
341                            // so subsequent attempts will be issued every second.
342                        }
343                        Err(tokio::sync::broadcast::error::RecvError::Lagged(_)) => {
344                            // This is very unlikely; we would have to have a lot of concurrent
345                            // control connections opened and broken at the same time.
346                            // The best we can do is ignoring this.
347                        }
348                        Err(tokio::sync::broadcast::error::RecvError::Closed) => {
349                            // If control_connection_repair_channel was closed then MetadataReader was dropped,
350                            // we can stop working.
351                            return;
352                        }
353                    }
354                }
355            }
356
357            // Perform the refresh
358            debug!("Requesting metadata refresh");
359            last_refresh_time = Instant::now();
360            let refresh_res = self.perform_refresh().await;
361
362            control_connection_works = refresh_res.is_ok();
363
364            // Send refresh result if there was a request
365            if let Some(request) = cur_request {
366                // We can ignore sending error - if no one waits for the response we can drop it
367                let _ = request.response_chan.send(refresh_res);
368            }
369        }
370    }

372    fn change_node_down_marker(&mut self, addr: SocketAddr, is_down: bool) {
373        let cluster_state = self.cluster_state.load_full();
374
375        // We need to iterate through the whole map here, but there will rarely more than ~100 nodes,
376        // and changes of down marker are infrequent enough to afford this. As an important tradeoff,
377        // we only keep one hashmap of known_peers, which is indexed by host IDs for node identification.
378        let node = match cluster_state
379            .known_peers
380            .values()
381            .find(|&peer| peer.address == NodeAddr::Translatable(addr))
382        {
383            Some(node) => node,
384            None => {
385                warn!("Unknown node address {}", addr);
386                return;
387            }
388        };
389
390        node.change_down_marker(is_down);
391    }

393    async fn handle_use_keyspace_request(
394        cluster_state: Arc<ClusterState>,
395        request: UseKeyspaceRequest,
396    ) {
397        let result = Self::send_use_keyspace(cluster_state, &request.keyspace_name).await;
398
399        // Don't care if nobody wants request result
400        let _ = request.response_chan.send(result);
401    }
402
403    async fn send_use_keyspace(
404        cluster_state: Arc<ClusterState>,
405        keyspace_name: &VerifiedKeyspaceName,
406    ) -> Result<(), UseKeyspaceError> {
407        let use_keyspace_futures = cluster_state
408            .known_peers
409            .values()
410            .map(|node| node.use_keyspace(keyspace_name.clone()));
411        let use_keyspace_results: Vec<Result<(), UseKeyspaceError>> =
412            join_all(use_keyspace_futures).await;
413
414        use_keyspace_result(use_keyspace_results.into_iter())
415    }
416
417    async fn perform_refresh(&mut self) -> Result<(), MetadataError> {
418        // Read latest Metadata
419        let metadata = self.metadata_reader.read_metadata(false).await?;
420        let cluster_state: Arc<ClusterState> = self.cluster_state.load_full();
421
422        let new_cluster_state = Arc::new(
423            ClusterState::new(
424                metadata,
425                &self.pool_config,
426                &cluster_state.known_peers,
427                &self.used_keyspace,
428                self.host_filter.as_deref(),
429                cluster_state.locator.tablets.clone(),
430                &cluster_state.keyspaces,
431                #[cfg(feature = "metrics")]
432                &self.metrics,
433            )
434            .await,
435        );
436
437        new_cluster_state
438            .wait_until_all_pools_are_initialized()
439            .await;
440
441        self.update_cluster_state(new_cluster_state);
442
443        Ok(())
444    }
445
446    fn update_cluster_state(&mut self, new_cluster_state: Arc<ClusterState>) {
447        self.cluster_state.store(new_cluster_state);
448    }

451/// Returns a result of use_keyspace operation, based on the query results
452/// returned from given node/connection.
453///
454/// This function assumes that `use_keyspace_results` iterator is NON-EMPTY!
455pub(crate) fn use_keyspace_result(
456    use_keyspace_results: impl Iterator<Item = Result<(), UseKeyspaceError>>,
457) -> Result<(), UseKeyspaceError> {
458    // If there was at least one Ok and the rest were broken connection errors we can return Ok
459    // keyspace name is correct and will be used on broken connection on the next reconnect
460
461    // If there were only broken connection errors then return broken connection error.
462    // If there was an error different than broken connection error return this error - something is wrong
463
464    let mut was_ok: bool = false;
465    let mut broken_conn_error: Option<UseKeyspaceError> = None;
466
467    for result in use_keyspace_results {
468        match result {
469            Ok(()) => was_ok = true,
470            Err(err) => match err {
471                UseKeyspaceError::RequestError(RequestAttemptError::BrokenConnectionError(_)) => {
472                    broken_conn_error = Some(err)
473                }
474                _ => return Err(err),
475            },
476        }
477    }
478
479    if was_ok {
480        return Ok(());
481    }
482
483    // We can unwrap conn_broken_error because use_keyspace_results must be nonempty
484    Err(broken_conn_error.unwrap())
