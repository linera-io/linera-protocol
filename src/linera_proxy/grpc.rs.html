102#[derive(Clone)]
103pub struct PrometheusMetricsMiddlewareService<T> {
104    service: T,
105}
106
107impl<S> Layer<S> for PrometheusMetricsMiddlewareLayer {
108    type Service = PrometheusMetricsMiddlewareService<S>;
109
110    fn layer(&self, service: S) -> Self::Service {
111        PrometheusMetricsMiddlewareService { service }
112    }
113}
114
115impl<S, Req> Service<Req> for PrometheusMetricsMiddlewareService<S>
116where
117    S::Future: Send + 'static,
118    S: Service<Req> + std::marker::Send,
119{
120    type Response = S::Response;
121    type Error = S::Error;
122    type Future = BoxFuture<'static, Result<S::Response, S::Error>>;
123
124    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
125        self.service.poll_ready(cx)
126    }
127
128    fn call(&mut self, request: Req) -> Self::Future {
129        #[cfg(with_metrics)]
130        let start = linera_base::time::Instant::now();
131        let future = self.service.call(request);
132        async move {
133            let response = future.await?;
134            #[cfg(with_metrics)]
135            {
136                metrics::PROXY_REQUEST_LATENCY
137                    .with_label_values(&[])
138                    .observe(start.elapsed().as_secs_f64() * 1000.0);
139                metrics::PROXY_REQUEST_COUNT.with_label_values(&[]).inc();
140            }
141            Ok(response)
142        }
143        .boxed()
144    }
145}

159impl<S> GrpcProxy<S>
160where
161    S: Storage + Clone + Send + Sync + 'static,
162{
163    pub fn new(
164        internal_config: ValidatorInternalNetworkConfig,
165        connect_timeout: Duration,
166        timeout: Duration,
167        tls: TlsConfig,
168        storage: S,
169        id: usize,
170    ) -> Self {
171        Self(Arc::new(GrpcProxyInner {
172            internal_config,
173            worker_connection_pool: GrpcConnectionPool::default()
174                .with_connect_timeout(connect_timeout)
175                .with_timeout(timeout),
176            notifier: ChannelNotifier::default(),
177            tls,
178            storage,
179            id,
180        }))
181    }
182
183    fn as_validator_node(&self) -> ValidatorNodeServer<Self> {
184        ValidatorNodeServer::new(self.clone())
185            .max_encoding_message_size(GRPC_MAX_MESSAGE_SIZE)
186            .max_decoding_message_size(GRPC_MAX_MESSAGE_SIZE)
187    }
188
189    fn config(&self) -> &ProxyConfig {
190        self.0
191            .internal_config
192            .proxies
193            .get(self.0.id)
194            .expect("No proxy config provided.")
195    }
196
197    fn as_notifier_service(&self) -> NotifierServiceServer<Self> {
198        NotifierServiceServer::new(self.clone())
199    }
200
201    fn public_address(&self) -> SocketAddr {
202        SocketAddr::from(([0, 0, 0, 0], self.config().public_port))
203    }
204
205    fn metrics_address(&self) -> SocketAddr {
206        SocketAddr::from(([0, 0, 0, 0], self.config().metrics_port))
207    }
208
209    fn internal_address(&self) -> SocketAddr {
210        SocketAddr::from(([0, 0, 0, 0], self.config().private_port))
211    }
212
213    fn shard_for(&self, proxyable: &impl GrpcProxyable) -> Option<ShardConfig> {
214        Some(
215            self.0
216                .internal_config
217                .get_shard_for(proxyable.chain_id()?)
218                .clone(),
219        )
220    }
221
222    fn worker_client_for_shard(
223        &self,
224        shard: &ShardConfig,
225    ) -> Result<ValidatorWorkerClient<Channel>> {
226        let address = shard.http_address();
227        let channel = self.0.worker_connection_pool.channel(address)?;
228        let client = ValidatorWorkerClient::new(channel)
229            .max_encoding_message_size(GRPC_MAX_MESSAGE_SIZE)
230            .max_decoding_message_size(GRPC_MAX_MESSAGE_SIZE);
231
232        Ok(client)
233    }
234
235    /// Runs the proxy. If either the public server or private server dies for whatever
236    /// reason we'll kill the proxy.
237    #[instrument(
238        name = "GrpcProxy::run",
239        skip_all,
240        fields(
241            public_address = %self.public_address(),
242            internal_address = %self.internal_address(),
243            metrics_address = %self.metrics_address(),
244        ),
245        err,
246    )]
247    pub async fn run(self, shutdown_signal: CancellationToken) -> Result<()> {
248        info!("Starting proxy");
249        let mut join_set = JoinSet::new();
250
251        #[cfg(with_metrics)]
252        monitoring_server::start_metrics(self.metrics_address(), shutdown_signal.clone());
253
254        let (health_reporter, health_service) = tonic_health::server::health_reporter();
255        health_reporter
256            .set_serving::<ValidatorNodeServer<GrpcProxy<S>>>()
257            .await;
258        let internal_server = join_set.spawn_task(
259            Server::builder()
260                .add_service(self.as_notifier_service())
261                .serve(self.internal_address())
262                .in_current_span(),
263        );
264        let reflection_service = tonic_reflection::server::Builder::configure()
265            .register_encoded_file_descriptor_set(linera_rpc::FILE_DESCRIPTOR_SET)
266            .build_v1()?;
267        let public_server = join_set.spawn_task(
268            self.public_server()?
269                .max_concurrent_streams(
270                    // we subtract one to make sure
271                    // that the value is not
272                    // interpreted as "not set"
273                    Some(u32::MAX - 1),
274                )
275                .layer(
276                    ServiceBuilder::new()
277                        .layer(PrometheusMetricsMiddlewareLayer)
278                        .into_inner(),
279                )
280                .layer(
281                    // enable
282                    // [CORS](https://developer.mozilla.org/en-US/docs/Web/HTTP/Guides/CORS)
283                    // for the proxy to originate anywhere
284                    tower_http::cors::CorsLayer::permissive(),
285                )
286                .layer(GrpcWebLayer::new())
287                .accept_http1(true)
288                .add_service(health_service)
289                .add_service(self.as_validator_node())
290                .add_service(reflection_service)
291                .serve_with_shutdown(self.public_address(), shutdown_signal.cancelled_owned())
292                .in_current_span(),
293        );
294
295        select! {
296            internal_res = internal_server => internal_res??,
297            public_res = public_server => public_res??,
298        }
299        Ok(())
300    }
301
302    /// Pre-configures the public server with no services attached.
303    /// If a certificate and key are defined, creates a TLS server.
304    fn public_server(&self) -> Result<Server> {
305        match self.0.tls {
306            TlsConfig::Tls => {
307                use linera_rpc::{CERT_PEM, KEY_PEM};
308                let identity = Identity::from_pem(CERT_PEM, KEY_PEM);
309                let tls_config = ServerTlsConfig::new().identity(identity);
310                Ok(Server::builder().tls_config(tls_config)?)
311            }
312            TlsConfig::ClearText => Ok(Server::builder()),
313        }
314    }
315
316    #[instrument(skip_all, fields(remote_addr = ?request.remote_addr(), chain_id = ?request.get_ref().chain_id()))]
317    fn worker_client<R>(
318        &self,
319        request: Request<R>,
320    ) -> Result<(ValidatorWorkerClient<Channel>, R), Status>
321    where
322        R: Debug + GrpcProxyable,
323    {
324        debug!("proxying request from {:?}", request.remote_addr());
325        let inner = request.into_inner();
326        let shard = self
327            .shard_for(&inner)
328            .ok_or_else(|| Status::not_found("could not find shard for message"))?;
329        let client = self
330            .worker_client_for_shard(&shard)
331            .map_err(|_| Status::internal("could not connect to shard"))?;
332        Ok((client, inner))
333    }

335    #[allow(clippy::result_large_err)]
336    fn log_and_return_proxy_request_outcome(
337        result: Result<Response<ChainInfoResult>, Status>,
338        method_name: &str,
339    ) -> Result<Response<ChainInfoResult>, Status> {
340        #![allow(unused_variables)]
341        match result {
342            Ok(chain_info_result) => {
343                #[cfg(with_metrics)]
344                metrics::PROXY_REQUEST_SUCCESS
345                    .with_label_values(&[method_name])
346                    .inc();
347                Ok(chain_info_result)
348            }
349            Err(status) => {
350                #[cfg(with_metrics)]
351                metrics::PROXY_REQUEST_ERROR
352                    .with_label_values(&[method_name])
353                    .inc();
354                Err(status)
355            }
356        }
357    }
358
359    /// Returns the appropriate gRPC status for the given [`ViewError`].
360    fn view_error_to_status(err: ViewError) -> Status {
361        let mut status = match &err {
362            ViewError::BcsError(_) => Status::invalid_argument(err.to_string()),
363            ViewError::StoreError { .. }
364            | ViewError::TokioJoinError(_)
365            | ViewError::TryLockError(_)
366            | ViewError::InconsistentEntries
367            | ViewError::PostLoadValuesError
368            | ViewError::IoError(_) => Status::internal(err.to_string()),
369            ViewError::KeyTooLong | ViewError::ArithmeticError(_) => {
370                Status::out_of_range(err.to_string())
371            }
372            ViewError::NotFound(_) | ViewError::MissingEntries => {
373                Status::not_found(err.to_string())
374            }
375        };
376        status.set_source(Arc::new(err));
377        status
378    }
379}

625    #[instrument(skip_all, err(Display), fields(method = "download_certificates"))]
626    async fn download_certificates(
627        &self,
628        request: Request<CertificatesBatchRequest>,
629    ) -> Result<Response<CertificatesBatchResponse>, Status> {
630        let hashes: Vec<linera_base::crypto::CryptoHash> = request
631            .into_inner()
632            .hashes
633            .into_iter()
634            .map(linera_base::crypto::CryptoHash::try_from)
635            .collect::<Result<Vec<linera_base::crypto::CryptoHash>, _>>()?;
636
637        // Use 70% of the max message size as a buffer capacity.
638        // Leave 30% as overhead.
639        let mut grpc_message_limiter: GrpcMessageLimiter<linera_chain::types::Certificate> =
640            GrpcMessageLimiter::new(GRPC_CHUNKED_MESSAGE_FILL_LIMIT);
641
642        let mut returned_certificates = vec![];
643
644        'outer: for batch in hashes.chunks(100) {
645            let certificates = self
646                .0
647                .storage
648                .read_certificates(batch.to_vec())
649                .await
650                .map_err(Self::view_error_to_status)?;
651            let certificates = match ResultReadCertificates::new(certificates, batch.to_vec()) {
652                ResultReadCertificates::Certificates(certificates) => certificates,
653                ResultReadCertificates::InvalidHashes(hashes) => {
654                    return Err(Status::not_found(format!("{:?}", hashes)))
655                }
656            };
657            for certificate in certificates {
658                if grpc_message_limiter.fits::<Certificate>(certificate.clone().into())? {
659                    returned_certificates.push(linera_chain::types::Certificate::from(certificate));
660                } else {
661                    break 'outer;
662                }
663            }
664        }
665
666        Ok(Response::new(CertificatesBatchResponse::try_from(
667            returned_certificates,
668        )?))
669    }

672        skip_all,
673        err(Display),
674        fields(method = "download_certificates_by_heights")
675    )]
676    async fn download_certificates_by_heights(
677        &self,
678        request: Request<api::DownloadCertificatesByHeightsRequest>,
679    ) -> Result<Response<CertificatesBatchResponse>, Status> {
680        let original_request: CertificatesByHeightRequest = request.into_inner().try_into()?;
681        let chain_info_request = ChainInfoQuery::new(original_request.chain_id)
682            .with_sent_certificate_hashes_by_heights(original_request.heights);
683
684        // Use handle_chain_info_query to get the certificate hashes
685        let chain_info_response = self
686            .handle_chain_info_query(Request::new(chain_info_request.try_into()?))
687            .await?;
688
689        // Extract the ChainInfoResult from the response
690        let chain_info_result = chain_info_response.into_inner();
691
692        // Extract the certificate hashes from the ChainInfo
693        let hashes = match chain_info_result.inner {
694            Some(api::chain_info_result::Inner::ChainInfoResponse(response)) => {
695                let chain_info: ChainInfo =
696                    bincode::deserialize(&response.chain_info).map_err(|e| {
697                        Status::internal(format!("Failed to deserialize ChainInfo: {}", e))
698                    })?;
699                chain_info.requested_sent_certificate_hashes
700            }
701            Some(api::chain_info_result::Inner::Error(error)) => {
702                let error =
703                    bincode::deserialize(&error).unwrap_or_else(|err| NodeError::GrpcError {
704                        error: format!("failed to unmarshal error message: {}", err),
705                    });
706                return Err(Status::internal(format!(
707                    "Chain info query failed: {error}"
708                )));
709            }
710            None => {
711                return Err(Status::internal("Empty chain info result"));
712            }
713        };
714
715        // Use download_certificates to get the actual certificates
716        let certificates_request = CertificatesBatchRequest {
717            hashes: hashes.into_iter().map(|h| h.into()).collect(),
718        };
719
720        self.download_certificates(Request::new(certificates_request))
721            .await
722    }

725    async fn download_raw_certificates_by_heights(
726        &self,
727        request: Request<api::DownloadCertificatesByHeightsRequest>,
728    ) -> Result<Response<api::RawCertificatesBatch>, Status> {
729        let original_request: CertificatesByHeightRequest = request.into_inner().try_into()?;
730        let chain_info_request = ChainInfoQuery::new(original_request.chain_id)
731            .with_sent_certificate_hashes_by_heights(original_request.heights);
732        // Use handle_chain_info_query to get the certificate hashes
733        let chain_info_response = self
734            .handle_chain_info_query(Request::new(chain_info_request.try_into()?))
735            .await?;
736        // Extract the ChainInfoResult from the response
737        let chain_info_result = chain_info_response.into_inner();
738        // Extract the certificate hashes from the ChainInfo
739        let hashes = match chain_info_result.inner {
740            Some(api::chain_info_result::Inner::ChainInfoResponse(response)) => {
741                let chain_info: ChainInfo =
742                    bincode::deserialize(&response.chain_info).map_err(|e| {
743                        Status::internal(format!("Failed to deserialize ChainInfo: {}", e))
744                    })?;
745                chain_info.requested_sent_certificate_hashes
746            }
747            Some(api::chain_info_result::Inner::Error(error)) => {
748                let error =
749                    bincode::deserialize(&error).unwrap_or_else(|err| NodeError::GrpcError {
750                        error: format!("failed to unmarshal error message: {}", err),
751                    });
752                return Err(Status::internal(format!(
753                    "Chain info query failed: {error}"
754                )));
755            }
756            None => {
757                return Err(Status::internal("Empty chain info result"));
758            }
759        };
760
761        // Use 70% of the max message size as a buffer capacity.
762        // Leave 30% as overhead.
763        let mut grpc_message_limiter: GrpcMessageLimiter<linera_chain::types::Certificate> =
764            GrpcMessageLimiter::new(GRPC_CHUNKED_MESSAGE_FILL_LIMIT);
765
766        let mut returned_certificates = vec![];
767
768        'outer: for batch in hashes.chunks(100) {
769            let certificates: Vec<(Vec<u8>, Vec<u8>)> = self
770                .0
771                .storage
772                .read_certificates_raw(batch.to_vec())
773                .await
774                .map_err(Self::view_error_to_status)?
775                .into_iter()
776                .collect();
777            for (lite_cert_bytes, confirmed_block_bytes) in certificates {
778                if grpc_message_limiter
779                    .fits_raw(lite_cert_bytes.len() + confirmed_block_bytes.len())
780                {
781                    returned_certificates.push(RawCertificate {
782                        lite_certificate: lite_cert_bytes,
783                        confirmed_block: confirmed_block_bytes,
784                    });
785                } else {
786                    break 'outer;
787                }
788            }
789        }
790
791        Ok(Response::new(RawCertificatesBatch {
792            certificates: returned_certificates,
793        }))
794    }

866/// A message limiter that keeps track of the remaining capacity in bytes.
867struct GrpcMessageLimiter<T> {
868    remaining: usize,
869    _phantom: PhantomData<T>,
870}
871
872impl<T> GrpcMessageLimiter<T> {
873    fn new(limit: usize) -> Self {
874        Self {
875            remaining: limit,
876            _phantom: PhantomData,
877        }
878    }
879
880    #[cfg(test)]
881    fn empty() -> Self {
882        Self::new(0)
883    }
884
885    // Returns true if the element, after serialising to proto bytes, fits within the remaining capacity.
886    fn fits<U>(&mut self, el: T) -> Result<bool, GrpcProtoConversionError>
887    where
888        U: TryFrom<T, Error = GrpcProtoConversionError> + Message,
889    {
890        let required = U::try_from(el).map(|proto| proto.encoded_len())?;
891        Ok(self.fits_raw(required))
892    }
893
894    /// Adds the given number of bytes to the remaining capacity.
895    ///
896    /// Returns whether we managed to fit the element.
897    fn fits_raw(&mut self, bytes_len: usize) -> bool {
898        if self.remaining < bytes_len {
899            return false;
900        }
901        self.remaining = self.remaining.saturating_sub(bytes_len);
902        true
903    }
904}
