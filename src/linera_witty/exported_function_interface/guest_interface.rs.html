4//! Representation of the interface a guest Wasm instance uses for a host function.
5//!
6//! The [`GuestInterface`] trait is implemented for the tuple pair of the flattened representation
7//! of the host function's parameters type and results type. The trait then maps the flattened host
8//! function signature into the interface guest Wasm instances expect. This involves selecting
9//! where the host function's parameters and results are stored and if the guest Wasm instance
10//! needs to provide additional parameters to the function with the addresses to store the
11//! parameters and/or results in memory.
12//!
13//! The [canonical ABI][flattening] describes a `MAX_FLAT_PARAMS` limit on the number of flat types
14//! that can be sent to the function through Wasm parameters. If more flat parameters are needed,
15//! then all of them are stored in memory instead, and the guest Wasm instance must provide a
16//! single parameter with the address in memory of where the parameters are stored. The same must
17//! be done if more flat result types than the `MAX_FLAT_RESULTS` limit defined by the [canonical
18//! ABI][flattening] need to be returned from the function. In that case, the guest Wasm instance
19//! is responsible for sending an additional parameter with an address in memory where the results
20//! can be stored. See [`ResultStorage`] for more information.
21//!
22//! [flattening]: https://github.com/WebAssembly/component-model/blob/main/design/mvp/CanonicalABI.md#flattening
23

34/// Representation of the interface a guest Wasm instance uses for a host function.
35///
36/// See the module level documentation for more information.
37pub trait GuestInterface {
38    /// The flattened layout of the host function's parameters type.
39    ///
40    /// This is used as a generic input type in all implementations, but is needed here so it can
41    /// be used in the constraints of the [`Self::lift_parameters`] method.
42    type FlatHostParameters: FlatLayout;
43
44    /// The flat types the guest Wasm instances use as parameters for the host function.
45    type FlatGuestParameters: FlatLayout;
46
47    /// How the host function's results type is sent back to the guest Wasm instance.
48    type ResultStorage: ResultStorage;
49
50    /// Lifts the parameters received from the guest Wasm instance into the parameters type the
51    /// host function expects.
52    ///
53    /// Returns the lifted host function's parameters together with a [`ResultStorage`]
54    /// implementation which is later used to lower the host function's results to where the guest
55    /// Wasm instance expects them to be. The [`ResultStorage`] is either the unit type `()`
56    /// indicating that the results should just be lowered and sent to the guest as the function's
57    /// return value, or a [`GuestPointer`] with the address of where the results should be stored
58    /// in memory.
59    fn lift_parameters<Instance, HostParameters>(
60        guest_parameters: Self::FlatGuestParameters,
61        memory: &Memory<'_, Instance>,
62    ) -> Result<(HostParameters, Self::ResultStorage), RuntimeError>
63    where
64        HostParameters: WitLoad,
65        HostParameters::Layout: Layout<Flat = Self::FlatHostParameters>,
66        Instance: InstanceWithMemory,
67        <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>;
68}
69
70/// Implements [`GuestInterface`] for the cases where the parameters and the results are sent
71/// directly as the function's parameters and results, without requiring any of them to be moved to
72/// memory.
73macro_rules! direct_interface {
74    ($( $types:ident ),* $(,)*) => {
75        direct_interface_with_result!($( $types ),* =>);
76        direct_interface_with_result!($( $types ),* => FlatResult);
77    };
78}
79
80/// Helper macro to [`direct_interface`] so that it can handle the two possible return signatures
81/// with all parameter combinations.
82macro_rules! direct_interface_with_result {
83    ($( $types:ident ),* => $( $flat_result:ident )?) => {
84        impl<$( $types, )* $( $flat_result )*> GuestInterface
85            for (HList![$( $types, )*], HList![$( $flat_result )*])
86        where
87            HList![$( $types, )*]: FlatLayout,
88            $( $flat_result: FlatType, )*
89        {
90            type FlatHostParameters = HList![$( $types, )*];
91            type FlatGuestParameters = HList![$( $types, )*];
92            type ResultStorage = ();
93
94            fn lift_parameters<Instance, HostParameters>(
95                guest_parameters: Self::FlatGuestParameters,
96                memory: &Memory<'_, Instance>,
97            ) -> Result<(HostParameters, Self::ResultStorage), RuntimeError>
98            where
99                HostParameters: WitLoad,
100                HostParameters::Layout: Layout<Flat = Self::FlatHostParameters>,
101                Instance: InstanceWithMemory,
102                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
103            {
104                let parameters = HostParameters::lift_from(guest_parameters, memory)?;
105
106                Ok((parameters, ()))
107            }
108        }
109    };
110}

114/// Implements [`GuestInterface`] for the cases where the parameters are sent directly as the
115/// function's parameters but the results are stored in memory instead.
116macro_rules! indirect_results {
117    ($( $types:ident ),*) => {
118        impl<$( $types, )* Y, Z, Tail> GuestInterface
119            for (HList![$( $types, )*], HList![Y, Z, ...Tail])
120        where
121            HList![$( $types, )*]: FlatLayout + Add<HList![i32]>,
122            <HList![$( $types, )*] as Add<HList![i32]>>::Output:
123                FlatLayout + Split<HList![$( $types, )*], Remainder = HList![i32]>,
124        {
125            type FlatHostParameters = HList![$( $types, )*];
126            type FlatGuestParameters = <Self::FlatHostParameters as Add<HList![i32]>>::Output;
127            type ResultStorage = GuestPointer;
128
129            fn lift_parameters<Instance, Parameters>(
130                guest_parameters: Self::FlatGuestParameters,
131                memory: &Memory<'_, Instance>,
132            ) -> Result<(Parameters, Self::ResultStorage), RuntimeError>
133            where
134                Parameters: WitLoad,
135                Parameters::Layout: Layout<Flat = Self::FlatHostParameters>,
136                Instance: InstanceWithMemory,
137                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
138            {
139                let (parameters_layout, result_storage_layout) = guest_parameters.split();
140                let parameters = Parameters::lift_from(parameters_layout, memory)?;
141                let result_storage = Self::ResultStorage::lift_from(result_storage_layout, memory)?;
142
143                Ok((parameters, result_storage))
144            }
145        }
146    };
147}

151/// Implements [`GuestInterface`] for the cases where the results are sent directly as the
152/// function's return value but the parameters are stored in memory instead.
153macro_rules! indirect_parameters {
154    (=> $( $flat_result:ident )? ) => {
155        impl<A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, Tail $(, $flat_result )*>
156            GuestInterface
157            for (
158                HList![A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, ...Tail],
159                HList![$( $flat_result )*],
160            )
161        where
162            HList![A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, ...Tail]: FlatLayout,
163            $( $flat_result: FlatType, )*
164        {
165            type FlatHostParameters =
166                HList![A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, ...Tail];
167            type FlatGuestParameters = HList![i32];
168            type ResultStorage = ();
169
170            fn lift_parameters<Instance, Parameters>(
171                guest_parameters: Self::FlatGuestParameters,
172                memory: &Memory<'_, Instance>,
173            ) -> Result<(Parameters, Self::ResultStorage), RuntimeError>
174            where
175                Parameters: WitLoad,
176                Parameters::Layout: Layout<Flat = Self::FlatHostParameters>,
177                Instance: InstanceWithMemory,
178                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
179            {
180                let parameters_location = GuestPointer::lift_from(guest_parameters, memory)?;
181                let parameters = Parameters::load(memory, parameters_location)?;
182
183                Ok((parameters, ()))
184            }
185        }
186    };
187}

192/// Implements [`GuestInterface`] for the cases where the parameters and the results need to be
193/// stored in memory.
194impl<A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, OtherParameters, Y, Z, OtherResults>
195    GuestInterface
196    for (
197        HList![A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, ...OtherParameters],
198        HList![Y, Z, ...OtherResults],
199    )
200where
201    HList![A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, ...OtherParameters]: FlatLayout,
202    HList![Y, Z, ...OtherResults]: FlatLayout,
203{
204    type FlatHostParameters =
205        HList![A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, ...OtherParameters];
206    type FlatGuestParameters = HList![i32, i32];
207    type ResultStorage = GuestPointer;
208
209    fn lift_parameters<Instance, Parameters>(
210        guest_parameters: Self::FlatGuestParameters,
211        memory: &Memory<'_, Instance>,
212    ) -> Result<(Parameters, Self::ResultStorage), RuntimeError>
213    where
214        Parameters: WitLoad,
215        Parameters::Layout: Layout<Flat = Self::FlatHostParameters>,
216        Instance: InstanceWithMemory,
217        <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
218    {
219        let (parameters_layout, result_storage_layout) = guest_parameters.split();
220        let parameters_location = GuestPointer::lift_from(parameters_layout, memory)?;
221        let parameters = Parameters::load(memory, parameters_location)?;
222        let result_storage = Self::ResultStorage::lift_from(result_storage_layout, memory)?;
223
224        Ok((parameters, result_storage))
225    }
