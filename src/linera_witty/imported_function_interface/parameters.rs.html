4//! Representation of the parameters of an imported function.
5//!
6//! The maximum number of parameters that can be used in a WIT function is defined by the
7//! [canonical ABI][flattening] as the `MAX_FLAT_PARAMS` constant (16). There is no equivalent
8//! constant defined in Witty. Instead, any attempt to use more than the limit should lead to a
9//! compiler error.
10//!
11//! The host parameters type is flattened and if it's made up of `MAX_FLAT_PARAMS` or less flat
12//! types, they are sent directly as the guest's function parameters. If there are more than
13//! `MAX_FLAT_PARAMS` flat types, then the host type is instead stored in a heap allocated region
14//! of memory, and the address to that region is sent as a parameter instead.
15//!
16//! [flattening]: https://github.com/WebAssembly/component-model/blob/main/design/mvp/CanonicalABI.md#flattening
17
18use frunk::HList;

20use crate::{

22    Runtime, RuntimeError, RuntimeMemory, WitStore, WitType,
23};

25/// Helper trait for converting from the host parameters to the guest parameters.
26///
27/// This trait is implemented for the intermediate flat layout of the host parameters type. This
28/// allows converting from the host parameters type into this flat layout and finally into the guest
29/// parameters type.
30pub trait FlatHostParameters: FlatLayout {
31    /// The actual parameters sent to the guest function.
32    type GuestParameters: FlatLayout;
33
34    /// Converts the `HostParameters` into the guest parameters that can be sent to the guest
35    /// function.
36    fn lower_parameters<HostParameters, Instance>(
37        parameters: HostParameters,
38        memory: &mut Memory<'_, Instance>,
39    ) -> Result<Self::GuestParameters, RuntimeError>
40    where
41        HostParameters: WitStore,
42        <HostParameters as WitType>::Layout: Layout<Flat = Self>,
43        Instance: InstanceWithMemory,
44        <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>;
45}
46
47/// Macro to implement [`FlatHostParameters`] for a parameters that are sent directly to the guest.
48///
49/// Each element in the flat layout type is sent as a separate parameter to the function.
50macro_rules! direct_parameters {
51    ($( $types:ident ),*) => {
52        impl<$( $types ),*> FlatHostParameters for HList![$( $types ),*]
53        where
54            $( $types: FlatType, )*
55        {
56            type GuestParameters = HList![$( $types ),*];
57
58            fn lower_parameters<Parameters, Instance>(
59                parameters: Parameters,
60                memory: &mut Memory<'_, Instance>,
61            ) -> Result<Self::GuestParameters, RuntimeError>
62            where
63                Parameters: WitStore,
64                <Parameters as WitType>::Layout: Layout<Flat = Self>,
65                Instance: InstanceWithMemory,
66                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
67            {
68                parameters.lower(memory)
69            }
70        }
71    };
72}
73
74repeat_macro!(direct_parameters => A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P);

76/// Implementation of [`FlatHostParameters`] for parameters that are sent to the guest through the
77/// heap.
78///
79/// The maximum number of parameters is defined by the [canonical
80/// ABI](https://github.com/WebAssembly/component-model/blob/main/design/mvp/CanonicalABI.md#flattening)
81/// as the `MAX_FLAT_PARAMS` constant. When more than `MAX_FLAT_PARAMS` (16) function parameters
82/// are needed, they are spilled over into a heap memory allocation and the only parameter sent as
83/// a function parameter is the address to that allocation.
84impl<A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, Tail> FlatHostParameters for HList![A, B, C, D, E, F, G, H, I, J, K, L, M, N, O, P, Q, ...Tail]
85where
86    A: FlatType,
87    B: FlatType,
88    C: FlatType,
89    D: FlatType,
90    E: FlatType,
91    F: FlatType,
92    G: FlatType,
93    H: FlatType,
94    I: FlatType,
95    J: FlatType,
96    K: FlatType,
97    L: FlatType,
98    M: FlatType,
99    N: FlatType,
100    O: FlatType,
101    P: FlatType,
102    Q: FlatType,
103    Tail: FlatLayout,
104{
105    type GuestParameters = HList![i32];
106
107    fn lower_parameters<Parameters, Instance>(
108        parameters: Parameters,
109        memory: &mut Memory<'_, Instance>,
110    ) -> Result<Self::GuestParameters, RuntimeError>
111    where
112        Parameters: WitStore,
113        <Parameters as WitType>::Layout: Layout<Flat = Self>,
114        Instance: InstanceWithMemory,
115        <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
116    {
117        let location =
118            memory.allocate(Parameters::SIZE, <Parameters::Layout as Layout>::ALIGNMENT)?;
119
120        parameters.store(memory, location)?;
121        location.lower(memory)
122    }
123}
