4//! A representation of either a [`FlatType`] or nothing, represented by the unit (`()`) type.
5
6use frunk::HCons;
7
8use super::flat_type::FlatType;
9use crate::{memory_layout::FlatLayout, Layout};
10
11/// A marker trait for [`FlatType`]s and the unit type, which uses no storage space.
12pub trait MaybeFlatType: Default + Sized {
13    /// Result of flattening the layout made up of the current element followed by `Tail`.
14    type Flatten<Tail: Layout>: FlatLayout;
15
16    /// Flattens a layout that starts with this [`MaybeFlatType`] followed by a provided `Tail`.
17    fn flatten<Tail>(self, tail: Tail) -> Self::Flatten<Tail>
18    where
19        Tail: Layout;
20}
21
22impl MaybeFlatType for () {
23    type Flatten<Tail: Layout> = Tail::Flat;
24
25    fn flatten<Tail>(self, tail: Tail) -> Self::Flatten<Tail>
26    where
27        Tail: Layout,
28    {
29        tail.flatten()
30    }
31}
32
33impl<AnyFlatType> MaybeFlatType for AnyFlatType
34where
35    AnyFlatType: FlatType,
36{
37    type Flatten<Tail: Layout> = HCons<Self, Tail::Flat>;
38
39    fn flatten<Tail>(self, tail: Tail) -> Self::Flatten<Tail>
40    where
41        Tail: Layout,
42    {
43        HCons {
44            head: self.flatten(),
45            tail: tail.flatten(),
46        }
47    }
48}
