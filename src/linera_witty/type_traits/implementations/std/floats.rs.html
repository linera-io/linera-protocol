4//! Implementations of the custom traits for float primitives.
5
6use std::borrow::Cow;
7
8use frunk::{hlist, hlist_pat, HList};
9
10use crate::{
11    GuestPointer, InstanceWithMemory, Layout, Memory, Runtime, RuntimeError, RuntimeMemory,
12    WitLoad, WitStore, WitType,
13};
14
15macro_rules! impl_wit_traits {
16    ($float:ty, $wit_name:literal, $size:expr) => {
17        impl WitType for $float {
18            const SIZE: u32 = $size;
19
20            type Layout = HList![$float];
21            type Dependencies = HList![];
22
23            fn wit_type_name() -> Cow<'static, str> {
24                $wit_name.into()
25            }
26
27            fn wit_type_declaration() -> Cow<'static, str> {
28                // Primitive types don't need to be declared
29                "".into()
30            }
31        }
32
33        impl WitLoad for $float {
34            fn load<Instance>(
35                memory: &Memory<'_, Instance>,
36                location: GuestPointer,
37            ) -> Result<Self, RuntimeError>
38            where
39                Instance: InstanceWithMemory,
40                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
41            {
42                let slice = memory.read(location, Self::SIZE)?;
43                let bytes = (*slice).try_into().expect("Incorrect number of bytes read");
44
45                Ok(Self::from_le_bytes(bytes))
46            }
47
48            fn lift_from<Instance>(
49                hlist_pat![value]: <Self::Layout as Layout>::Flat,
50                _memory: &Memory<'_, Instance>,
51            ) -> Result<Self, RuntimeError>
52            where
53                Instance: InstanceWithMemory,
54                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
55            {
56                Ok(value)
57            }
58        }
59
60        impl WitStore for $float {
61            fn store<Instance>(
62                &self,
63                memory: &mut Memory<'_, Instance>,
64                location: GuestPointer,
65            ) -> Result<(), RuntimeError>
66            where
67                Instance: InstanceWithMemory,
68                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
69            {
70                memory.write(location, &self.to_le_bytes())
71            }
72
73            fn lower<Instance>(
74                &self,
75                _memory: &mut Memory<'_, Instance>,
76            ) -> Result<<Self::Layout as Layout>::Flat, RuntimeError>
77            where
78                Instance: InstanceWithMemory,
79                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
80            {
81                Ok(hlist![*self])
82            }
83        }
84    };
85}
86
87impl_wit_traits!(f32, "float32", 4);
88impl_wit_traits!(f64, "float64", 8);
