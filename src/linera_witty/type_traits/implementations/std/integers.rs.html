4//! Implementations of the custom traits for integer primitives.
5
6use std::borrow::Cow;
7
8use frunk::{hlist, hlist_pat, HList};
9
10use crate::{
11    GuestPointer, InstanceWithMemory, Layout, Memory, Runtime, RuntimeError, RuntimeMemory,
12    WitLoad, WitStore, WitType,
13};
14
15macro_rules! impl_wit_traits {
16    ($integer:ty, $wit_name:literal, 1) => {
17        impl WitType for $integer {
18            const SIZE: u32 = 1;
19
20            type Layout = HList![$integer];
21            type Dependencies = HList![];
22
23            fn wit_type_name() -> Cow<'static, str> {
24                $wit_name.into()
25            }
26
27            fn wit_type_declaration() -> Cow<'static, str> {
28                "".into()
29            }
30        }
31
32        impl WitLoad for $integer {
33            fn load<Instance>(
34                memory: &Memory<'_, Instance>,
35                location: GuestPointer,
36            ) -> Result<Self, RuntimeError>
37            where
38                Instance: InstanceWithMemory,
39                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
40            {
41                let slice = memory.read(location, 1)?;
42                Ok(slice[0] as $integer)
43            }
44
45            fn lift_from<Instance>(
46                hlist_pat![value]: <Self::Layout as Layout>::Flat,
47                _memory: &Memory<'_, Instance>,
48            ) -> Result<Self, RuntimeError>
49            where
50                Instance: InstanceWithMemory,
51                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
52            {
53                Ok(value as $integer)
54            }
55        }
56
57        impl WitStore for $integer {
58            fn store<Instance>(
59                &self,
60                memory: &mut Memory<'_, Instance>,
61                location: GuestPointer,
62            ) -> Result<(), RuntimeError>
63            where
64                Instance: InstanceWithMemory,
65                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
66            {
67                memory.write(location, &[*self as u8])
68            }
69
70            fn lower<Instance>(
71                &self,
72                _memory: &mut Memory<'_, Instance>,
73            ) -> Result<<Self::Layout as Layout>::Flat, RuntimeError>
74            where
75                Instance: InstanceWithMemory,
76                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
77            {
78                Ok(hlist![*self as i32])
79            }
80        }
81    };
82
83    ($integer:ty, $wit_name:literal, $size:expr, $flat_type:ty) => {
84        impl_wit_traits!(
85            $integer,
86            $wit_name,
87            "",
88            $size,
89            ($integer),
90            ($flat_type),
91            self -> hlist![*self as $flat_type],
92            hlist_pat![value] => value as Self
93        );
94    };
95
96    (
97        $integer:ty,
98        $wit_name:literal,
99        $wit_declaration:literal,
100        $size:expr,
101        ($( $simple_types:ty ),*),
102        ($( $flat_types:ty ),*),
103        $this:ident -> $lower:expr,
104        $lift_pattern:pat => $lift:expr
105    ) => {
106        impl WitType for $integer {
107            const SIZE: u32 = $size;
108
109            type Layout = HList![$( $simple_types ),*];
110            type Dependencies = HList![];
111
112            fn wit_type_name() -> Cow<'static, str> {
113                $wit_name.into()
114            }
115
116            fn wit_type_declaration() -> Cow<'static, str> {
117                $wit_declaration.into()
118            }
119        }
120
121        impl WitLoad for $integer {
122            fn load<Instance>(
123                memory: &Memory<'_, Instance>,
124                location: GuestPointer,
125            ) -> Result<Self, RuntimeError>
126            where
127                Instance: InstanceWithMemory,
128                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
129            {
130                let slice = memory.read(location, Self::SIZE)?;
131                let bytes = (*slice).try_into().expect("Incorrect number of bytes read");
132
133                Ok(Self::from_le_bytes(bytes))
134            }
135
136            fn lift_from<Instance>(
137                $lift_pattern: <Self::Layout as Layout>::Flat,
138                _memory: &Memory<'_, Instance>,
139            ) -> Result<Self, RuntimeError>
140            where
141                Instance: InstanceWithMemory,
142                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
143            {
144                Ok($lift)
145            }
146        }
147
148        impl WitStore for $integer {
149            fn store<Instance>(
150                &self,
151                memory: &mut Memory<'_, Instance>,
152                location: GuestPointer,
153            ) -> Result<(), RuntimeError>
154            where
155                Instance: InstanceWithMemory,
156                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
157            {
158                memory.write(location, &self.to_le_bytes())
159            }
160
161            fn lower<Instance>(
162                &$this,
163                _memory: &mut Memory<'_, Instance>,
164            ) -> Result<<Self::Layout as Layout>::Flat, RuntimeError>
165            where
166                Instance: InstanceWithMemory,
167                <Instance::Runtime as Runtime>::Memory: RuntimeMemory<Instance>,
168            {
169                Ok($lower)
170            }
171        }
172    };
173}
174
175impl_wit_traits!(u8, "u8", 1);
176impl_wit_traits!(i8, "s8", 1);
177impl_wit_traits!(u16, "u16", 2, i32);
178impl_wit_traits!(i16, "s16", 2, i32);
179impl_wit_traits!(u32, "u32", 4, i32);
180impl_wit_traits!(i32, "s32", 4, i32);
181impl_wit_traits!(u64, "u64", 8, i64);
182impl_wit_traits!(i64, "s64", 8, i64);
183
184macro_rules! x128_lower {
185    ($this:ident) => {
186        hlist![
187            ($this & ((1 << 64) - 1)) as i64,
188            (($this >> 64) & ((1 << 64) - 1)) as i64,
189        ]
190    };
191}
192
193impl_wit_traits!(
194    u128,
195    "u128",
196    "    type u128 = tuple<u64, u64>;\n",
197    16,
198    (u64, u64),
199    (i64, i64),
200    self -> x128_lower!(self),
201    hlist_pat![least_significant_bytes, most_significant_bytes] => {
202        ((most_significant_bytes as Self) << 64)
203        | (least_significant_bytes as Self & ((1 << 64) - 1))
204    }
205);
206
207impl_wit_traits!(
208    i128,
209    "s128",
210    "    type s128 = tuple<s64, s64>;\n",
211    16,
212    (i64, i64),
213    (i64, i64),
214    self -> x128_lower!(self),
215    hlist_pat![least_significant_bytes, most_significant_bytes] => {
216        ((most_significant_bytes as Self) << 64)
217        | (least_significant_bytes as Self & ((1 << 64) - 1))
218    }
