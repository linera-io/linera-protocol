5/// Container-level arguments for the TransactionEnvelope derive macro.
6#[derive(Debug, FromDeriveInput)]
7#[darling(attributes(envelope))]
8pub(crate) struct EnvelopeArgs {
9    /// The identifier of the input enum.
10    pub ident: Ident,
11
12    /// The generics of the input enum.
13    pub generics: syn::Generics,
14
15    /// Custom name for the generated transaction type enum.
16    /// Defaults to `{EnumName}Type`.
17    #[darling(default)]
18    pub tx_type_name: Option<Ident>,
19
20    /// Custom path to the alloy_consensus crate.
21    /// Defaults to `::alloy_consensus`.
22    #[darling(default)]
23    pub alloy_consensus: Option<Path>,
24
25    /// Custom `cfg_attr` value for serde implementations.
26    #[darling(default)]
27    pub serde_cfg: Option<syn::Meta>,
28
29    /// Custom `cfg_attr` value for arbitrary implementations.
30    #[darling(default)]
31    pub arbitrary_cfg: Option<syn::Meta>,
32
33    /// The enum data (variants).
34    pub data: darling::ast::Data<EnvelopeVariant, ()>,
35}

37/// Variant of transaction envelope enum.
38#[derive(Debug, FromVariant)]
39#[darling(attributes(envelope), forward_attrs(serde))]
40pub(crate) struct EnvelopeVariant {
41    /// The identifier of the variant.
42    pub ident: Ident,
43
44    /// The fields of the variant.
45    pub fields: darling::ast::Fields<syn::Type>,
46
47    /// Kind of the variant.
48    #[darling(flatten)]
49    pub kind: VariantKind,
50
51    /// Forwarded attributes.
52    pub attrs: Vec<syn::Attribute>,
53}
54
55/// Kind of the envelope variant.
56#[derive(Debug, Clone, FromMeta)]
57pub(crate) enum VariantKind {
58    /// A standalone transaction with a type tag.
59    #[darling(rename = "ty")]
60    Typed(u8),
61    /// Flattened envelope.
62    #[darling(word, rename = "flatten")]
63    Flattened,
64}

66/// Processed variant information.
67#[derive(Debug, Clone)]
68pub(crate) struct ProcessedVariant {
69    /// The variant name.
70    pub name: Ident,
71    /// The inner type of the variant.
72    pub ty: Type,
73    /// The kind of variant.
74    pub kind: VariantKind,
75    /// The serde attributes for the variant.
76    pub serde_attrs: Option<TokenStream>,
77}
78
79impl ProcessedVariant {
80    /// Returns true if this is a legacy transaction variant (type 0).
81    pub(crate) const fn is_legacy(&self) -> bool {
82        matches!(self.kind, VariantKind::Typed(0))
83    }
84}
85
86/// Groups variants by their kind for easier code generation.
87pub(crate) struct GroupedVariants {
88    /// All variants.
89    pub all: Vec<ProcessedVariant>,
90    /// Only typed variants (with transaction type IDs).
91    pub typed: Vec<ProcessedVariant>,
92    /// Only flattened variants.
93    pub flattened: Vec<ProcessedVariant>,
94}
95
96impl GroupedVariants {
97    /// Create grouped variants from a list of processed variants.
98    pub(crate) fn from_args(args: EnvelopeArgs) -> darling::Result<Self> {
99        // Validate it's an enum
100        let variants = match args.data {
101            darling::ast::Data::Enum(variants) => variants,
102            _ => {
103                return Err(darling::Error::custom(
104                    "`TransactionEnvelope` can only be derived for enums",
105                )
106                .with_span(&args.ident));
107            }
108        };
109
110        let mut processed = Vec::new();
111        for variant in variants {
112            let EnvelopeVariant { ident, fields, kind, attrs } = variant;
113
114            let serde_attrs =
115                if let Some(attr) = attrs.into_iter().find(|attr| attr.path().is_ident("serde")) {
116                    if let syn::Meta::List(list) = attr.meta {
117                        Some(list.tokens.clone())
118                    } else {
119                        None
120                    }
121                } else {
122                    None
123                };
124
125            // Check that variant has exactly one unnamed field
126            let ty = match &fields.style {
127                darling::ast::Style::Tuple if fields.len() == 1 => fields.fields[0].clone(),
128                darling::ast::Style::Tuple => {
129                    return Err(darling::Error::custom(format!(
130                        "expected exactly one field, found {}",
131                        fields.len()
132                    ))
133                    .with_span(&ident))
134                }
135                _ => {
136                    return Err(darling::Error::custom(
137                        "TransactionEnvelope variants must have a single unnamed field",
138                    )
139                    .with_span(&ident))
140                }
141            };
142
143            processed.push(ProcessedVariant { name: ident, ty, kind, serde_attrs });
144        }
145
146        let typed =
147            processed.iter().filter(|v| matches!(v.kind, VariantKind::Typed(_))).cloned().collect();
148
149        let flattened = processed
150            .iter()
151            .filter(|v| matches!(v.kind, VariantKind::Flattened))
152            .cloned()
153            .collect();
154
155        Ok(Self { all: processed, typed, flattened })
156    }
157
158    /// Find the legacy variant if it exists.
159    pub(crate) fn legacy_variant(&self) -> Option<&ProcessedVariant> {
160        self.all.iter().find(|v| v.is_legacy())
161    }
162
163    /// Get all variant names.
164    pub(crate) fn variant_names(&self) -> Vec<&syn::Ident> {
165        self.all.iter().map(|v| &v.name).collect()
166    }
167
168    /// Get all variant types.
169    pub(crate) fn variant_types(&self) -> Vec<&syn::Type> {
170        self.all.iter().map(|v| &v.ty).collect()
171    }
172}
