21impl<'a, F: Function> Env<'a, F> {
22    pub fn try_allocating_regs_for_spilled_bundles(&mut self) {
23        trace!("allocating regs for spilled bundles");
24        for i in 0..self.spilled_bundles.len() {
25            let bundle = self.spilled_bundles[i]; // don't borrow self
26
27            if self.bundles[bundle].ranges.is_empty() {
28                continue;
29            }
30
31            let class = self.spillsets[self.bundles[bundle].spillset].class;
32            let hint = self.spillsets[self.bundles[bundle].spillset].reg_hint;
33
34            // This may be an empty-range bundle whose ranges are not
35            // sorted; sort all range-lists again here.
36            self.bundles[bundle]
37                .ranges
38                .sort_unstable_by_key(|entry| entry.range.from);
39
40            let mut success = false;
41            self.stats.spill_bundle_reg_probes += 1;
42            for preg in
43                RegTraversalIter::new(self.env, class, hint, PReg::invalid(), bundle.index(), None)
44            {
45                trace!("trying bundle {:?} to preg {:?}", bundle, preg);
46                let preg_idx = PRegIndex::new(preg.index());
47                if let AllocRegResult::Allocated(_) =
48                    self.try_to_allocate_bundle_to_reg(bundle, preg_idx, None)
49                {
50                    self.stats.spill_bundle_reg_success += 1;
51                    success = true;
52                    break;
53                }
54            }
55            if !success {
56                trace!(
57                    "spilling bundle {:?}: marking spillset {:?} as required",
58                    bundle,
59                    self.bundles[bundle].spillset
60                );
61                self.spillsets[self.bundles[bundle].spillset].required = true;
62            }
63        }
64    }

66    pub fn spillslot_can_fit_spillset(
67        &mut self,
68        spillslot: SpillSlotIndex,
69        spillset: SpillSetIndex,
70    ) -> bool {
71        !self.spillslots[spillslot.index()]
72            .ranges
73            .btree
74            .contains_key(&LiveRangeKey::from_range(&self.spillsets[spillset].range))
75    }
76
77    pub fn allocate_spillset_to_spillslot(
78        &mut self,
79        spillset: SpillSetIndex,
80        spillslot: SpillSlotIndex,
81    ) {
82        self.spillsets[spillset].slot = spillslot;
83
84        let res = self.spillslots[spillslot.index()].ranges.btree.insert(
85            LiveRangeKey::from_range(&self.spillsets[spillset].range),
86            spillset,
87        );
88
89        debug_assert!(res.is_none());
90    }

92    pub fn allocate_spillslots(&mut self) {
93        const MAX_ATTEMPTS: usize = 10;
94
95        for spillset in 0..self.spillsets.len() {
96            trace!("allocate spillslot: {}", spillset);
97            let spillset = SpillSetIndex::new(spillset);
98            if !self.spillsets[spillset].required {
99                continue;
100            }
101            let class = self.spillsets[spillset].class as usize;
102            // Try a few existing spillslots.
103            let mut i = self.slots_by_class[class].probe_start;
104            let mut success = false;
105            // Never probe the same element more than once: limit the
106            // attempt count to the number of slots in existence.
107            for _attempt in 0..core::cmp::min(self.slots_by_class[class].slots.len(), MAX_ATTEMPTS)
108            {
109                // Note: this indexing of `slots` is always valid
110                // because either the `slots` list is empty and the
111                // iteration limit above consequently means we don't
112                // run this loop at all, or else `probe_start` is
113                // in-bounds (because it is made so below when we add
114                // a slot, and it always takes on the last index `i`
115                // after this loop).
116                let spillslot = self.slots_by_class[class].slots[i];
117
118                if self.spillslot_can_fit_spillset(spillslot, spillset) {
119                    self.allocate_spillset_to_spillslot(spillset, spillslot);
120                    success = true;
121                    self.slots_by_class[class].probe_start = i;
122                    break;
123                }
124
125                i = self.slots_by_class[class].next_index(i);
126            }
127
128            if !success {
129                // Allocate a new spillslot.
130                let spillslot = SpillSlotIndex::new(self.spillslots.len());
131                self.spillslots.push(SpillSlotData {
132                    ranges: SpillSetRanges::new(),
133                    alloc: Allocation::none(),
134                    slots: self.func.spillslot_size(self.spillsets[spillset].class) as u32,
135                });
136                self.slots_by_class[class].slots.push(spillslot);
137                self.slots_by_class[class].probe_start = self.slots_by_class[class].slots.len() - 1;
138
139                self.allocate_spillset_to_spillslot(spillset, spillslot);
140            }
141        }
142
143        // Assign actual slot indices to spillslots.
144        for i in 0..self.spillslots.len() {
145            self.spillslots[i].alloc = self.allocate_spillslot(self.spillslots[i].slots);
146        }
147
148        trace!("spillslot allocator done");
149    }

151    pub fn allocate_spillslot(&mut self, size: u32) -> Allocation {
152        let mut offset = self.num_spillslots;
153        // Align up to `size`.
154        debug_assert!(size.is_power_of_two());
155        offset = (offset + size - 1) & !(size - 1);
156        let slot = if self.func.multi_spillslot_named_by_last_slot() {
157            offset + size - 1
158        } else {
159            offset
160        };
161        offset += size;
162        self.num_spillslots = offset;
163        Allocation::stack(SpillSlot::new(slot as usize))
164    }
