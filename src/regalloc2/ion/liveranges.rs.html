33/// A spill weight computed for a certain Use.
34#[derive(Clone, Copy, Debug)]
35pub struct SpillWeight(f32);
36
37#[inline(always)]
38pub fn spill_weight_from_constraint(
39    constraint: OperandConstraint,
40    loop_depth: usize,
41    is_def: bool,
42) -> SpillWeight {
43    // A bonus of 1000 for one loop level, 4000 for two loop levels,
44    // 16000 for three loop levels, etc. Avoids exponentiation.
45    let loop_depth = core::cmp::min(10, loop_depth);
46    let hot_bonus: f32 = (0..loop_depth).fold(1000.0, |a, _| a * 4.0);
47    let def_bonus: f32 = if is_def { 2000.0 } else { 0.0 };
48    let constraint_bonus: f32 = match constraint {
49        OperandConstraint::Any => 1000.0,
50        OperandConstraint::Reg | OperandConstraint::FixedReg(_) => 2000.0,
51        _ => 0.0,
52    };
53    SpillWeight(hot_bonus + def_bonus + constraint_bonus)
54}

56impl SpillWeight {
57    /// Convert a floating-point weight to a u16 that can be compactly
58    /// stored in a `Use`. We simply take the top 16 bits of the f32; this
59    /// is equivalent to the bfloat16 format
60    /// (https://en.wikipedia.org/wiki/Bfloat16_floating-point_format).
61    pub fn to_bits(self) -> u16 {
62        (self.0.to_bits() >> 15) as u16
63    }
64
65    /// Convert a value that was returned from
66    /// `SpillWeight::to_bits()` back into a `SpillWeight`. Note that
67    /// some precision may be lost when round-tripping from a spill
68    /// weight to packed bits and back.
69    pub fn from_bits(bits: u16) -> SpillWeight {
70        let x = f32::from_bits((bits as u32) << 15);
71        SpillWeight(x)
72    }
73
74    /// Get a zero spill weight.
75    pub fn zero() -> SpillWeight {
76        SpillWeight(0.0)
77    }
78
79    /// Convert to a raw floating-point value.
80    pub fn to_f32(self) -> f32 {
81        self.0
82    }
83
84    /// Create a `SpillWeight` from a raw floating-point value.
85    pub fn from_f32(x: f32) -> SpillWeight {
86        SpillWeight(x)
87    }
88
89    pub fn to_int(self) -> u32 {
90        self.0 as u32
91    }
92}

95    type Output = SpillWeight;
96    fn add(self, other: SpillWeight) -> Self {
97        SpillWeight(self.0 + other.0)
98    }

102    pub fn create_pregs_and_vregs(&mut self) {
103        // Create PRegs from the env.
104        self.pregs.resize(
105            PReg::NUM_INDEX,
106            PRegData {
107                allocations: LiveRangeSet::new(),
108                is_stack: false,
109            },
110        );
111        for &preg in &self.env.fixed_stack_slots {
112            self.pregs[preg.index()].is_stack = true;
113        }
114        for class in 0..self.preferred_victim_by_class.len() {
115            self.preferred_victim_by_class[class] = self.env.non_preferred_regs_by_class[class]
116                .last()
117                .or(self.env.preferred_regs_by_class[class].last())
118                .cloned()
119                .unwrap_or(PReg::invalid());
120        }
121        // Create VRegs from the vreg count.
122        for idx in 0..self.func.num_vregs() {
123            // We'll fill in the real details when we see the def.
124            self.vregs.add(
125                VReg::new(idx, RegClass::Int),
126                VRegData {
127                    ranges: smallvec![],
128                    blockparam: Block::invalid(),
129                    // We'll learn the RegClass as we scan the code.
130                    class: None,
131                },
132            );
133        }
134        // Create allocations too.
135        for inst in 0..self.func.num_insts() {
136            let start = self.allocs.len() as u32;
137            self.inst_alloc_offsets.push(start);
138            for _ in 0..self.func.inst_operands(Inst::new(inst)).len() {
139                self.allocs.push(Allocation::none());
140            }
141        }
142    }

144    /// Mark `range` as live for the given `vreg`.
145    ///
146    /// Returns the liverange that contains the given range.
147    pub fn add_liverange_to_vreg(
148        &mut self,
149        vreg: VRegIndex,
150        mut range: CodeRange,
151    ) -> LiveRangeIndex {
152        trace!("add_liverange_to_vreg: vreg {:?} range {:?}", vreg, range);
153
154        // Invariant: as we are building liveness information, we
155        // *always* process instructions bottom-to-top, and as a
156        // consequence, new liveranges are always created before any
157        // existing liveranges for a given vreg. We assert this here,
158        // then use it to avoid an O(n) merge step (which would lead
159        // to O(n^2) liveness construction cost overall).
160        //
161        // We store liveranges in reverse order in the `.ranges`
162        // array, then reverse them at the end of
163        // `compute_liveness()`.
164
165        if !self.vregs[vreg].ranges.is_empty() {
166            let last_range_index = self.vregs[vreg].ranges.last().unwrap().index;
167            let last_range = self.ranges[last_range_index].range;
168            if self.func.allow_multiple_vreg_defs() {
169                if last_range.contains(&range) {
170                    // Special case (may occur when multiple defs of pinned
171                    // physical regs occur): if this new range overlaps the
172                    // existing range, return it.
173                    return last_range_index;
174                }
175                // If this range's end falls in the middle of the last
176                // range, truncate it to be contiguous so we can merge
177                // below.
178                if range.to >= last_range.from && range.to <= last_range.to {
179                    range.to = last_range.from;
180                }
181            }
182            debug_assert!(
183                range.to <= last_range.from,
184                "range {:?}, last_range {:?}",
185                range,
186                last_range
187            );
188        }
189
190        if self.vregs[vreg].ranges.is_empty()
191            || range.to
192                < self.ranges[self.vregs[vreg].ranges.last().unwrap().index]
193                    .range
194                    .from
195        {
196            // Is not contiguous with previously-added (immediately
197            // following) range; create a new range.
198            let lr = self.ranges.add(range);
199            self.ranges[lr].vreg = vreg;
200            self.vregs[vreg]
201                .ranges
202                .push(LiveRangeListEntry { range, index: lr });
203            lr
204        } else {
205            // Is contiguous with previously-added range; just extend
206            // its range and return it.
207            let lr = self.vregs[vreg].ranges.last().unwrap().index;
208            debug_assert!(range.to == self.ranges[lr].range.from);
209            self.ranges[lr].range.from = range.from;
210            lr
211        }
212    }

214    pub fn insert_use_into_liverange(&mut self, into: LiveRangeIndex, mut u: Use) {
215        let operand = u.operand;
216        let constraint = operand.constraint();
217        let block = self.cfginfo.insn_block[u.pos.inst().index()];
218        let loop_depth = self.cfginfo.approx_loop_depth[block.index()] as usize;
219        let weight = spill_weight_from_constraint(
220            constraint,
221            loop_depth,
222            operand.kind() != OperandKind::Use,
223        );
224        u.weight = weight.to_bits();
225
226        trace!(
227            "insert use {:?} into lr {:?} with weight {:?}",
228            u,
229            into,
230            weight,
231        );
232
233        // N.B.: we do *not* update `requirement` on the range,
234        // because those will be computed during the multi-fixed-reg
235        // fixup pass later (after all uses are inserted).
236
237        self.ranges[into].uses.push(u);
238
239        // Update stats.
240        let range_weight = self.ranges[into].uses_spill_weight() + weight;
241        self.ranges[into].set_uses_spill_weight(range_weight);
242        trace!(
243            "  -> now range has weight {:?}",
244            self.ranges[into].uses_spill_weight(),
245        );
246    }

261    pub fn add_liverange_to_preg(&mut self, range: CodeRange, reg: PReg) {
262        trace!("adding liverange to preg: {:?} to {}", range, reg);
263        let preg_idx = PRegIndex::new(reg.index());
264        let res = self.pregs[preg_idx.index()]
265            .allocations
266            .btree

275    pub fn compute_liveness(&mut self) -> Result<(), RegAllocError> {
276        // Create initial LiveIn and LiveOut bitsets.
277        for _ in 0..self.func.num_blocks() {
278            self.liveins.push(IndexSet::new());
279            self.liveouts.push(IndexSet::new());
280        }
281
282        // Run a worklist algorithm to precisely compute liveins and
283        // liveouts.
284        let mut workqueue = VecDeque::new();
285        let mut workqueue_set = FxHashSet::default();
286        // Initialize workqueue with postorder traversal.
287        for &block in &self.cfginfo.postorder[..] {
288            workqueue.push_back(block);
289            workqueue_set.insert(block);
290        }
291
292        while let Some(block) = workqueue.pop_front() {
293            workqueue_set.remove(&block);
294            let insns = self.func.block_insns(block);
295
296            trace!("computing liveins for block{}", block.index());
297
298            self.stats.livein_iterations += 1;
299
300            let mut live = self.liveouts[block.index()].clone();
301            trace!(" -> initial liveout set: {:?}", live);
302
303            // Include outgoing blockparams in the initial live set.
304            if self.func.is_branch(insns.last()) {
305                for i in 0..self.func.block_succs(block).len() {
306                    for &param in self.func.branch_blockparams(block, insns.last(), i) {
307                        live.set(param.vreg(), true);
308                        self.observe_vreg_class(param);
309                    }
310                }
311            }
312
313            for inst in insns.iter().rev() {
314                for pos in &[OperandPos::Late, OperandPos::Early] {
315                    for op in self.func.inst_operands(inst) {
316                        if op.as_fixed_nonallocatable().is_some() {
317                            continue;
318                        }
319                        if op.pos() == *pos {
320                            let was_live = live.get(op.vreg().vreg());
321                            trace!("op {:?} was_live = {}", op, was_live);
322                            match op.kind() {
323                                OperandKind::Use => {
324                                    live.set(op.vreg().vreg(), true);
325                                }
326                                OperandKind::Def => {
327                                    live.set(op.vreg().vreg(), false);
328                                }
329                            }
330                            self.observe_vreg_class(op.vreg());
331                        }
332                    }
333                }
334            }
335            for &blockparam in self.func.block_params(block) {
336                live.set(blockparam.vreg(), false);
337                self.observe_vreg_class(blockparam);
338            }
339
340            for &pred in self.func.block_preds(block) {
341                if self.liveouts[pred.index()].union_with(&live) {
342                    if !workqueue_set.contains(&pred) {
343                        workqueue_set.insert(pred);
344                        workqueue.push_back(pred);
345                    }
346                }
347            }
348
349            trace!("computed liveins at block{}: {:?}", block.index(), live);
350            self.liveins[block.index()] = live;
351        }
352
353        // Check that there are no liveins to the entry block.
354        if !self.liveins[self.func.entry_block().index()].is_empty() {
355            trace!(
356                "non-empty liveins to entry block: {:?}",
357                self.liveins[self.func.entry_block().index()]
358            );
359            return Err(RegAllocError::EntryLivein);
360        }
361
362        Ok(())
363    }

365    pub fn build_liveranges(&mut self) {
366        // Create Uses and Defs referring to VRegs, and place the Uses
367        // in LiveRanges.
368        //
369        // We already computed precise liveouts and liveins for every
370        // block above, so we don't need to run an iterative algorithm
371        // here; instead, every block's computation is purely local,
372        // from end to start.
373
374        // Track current LiveRange for each vreg.
375        //
376        // Invariant: a stale range may be present here; ranges are
377        // only valid if `live.get(vreg)` is true.
378        let mut vreg_ranges: Vec<LiveRangeIndex> =
379            vec![LiveRangeIndex::invalid(); self.func.num_vregs()];
380
381        for i in (0..self.func.num_blocks()).rev() {
382            let block = Block::new(i);
383            let insns = self.func.block_insns(block);
384
385            self.stats.livein_blocks += 1;
386
387            // Init our local live-in set.
388            let mut live = self.liveouts[block.index()].clone();
389
390            // If the last instruction is a branch (rather than
391            // return), create blockparam_out entries.
392            if self.func.is_branch(insns.last()) {
393                for (i, &succ) in self.func.block_succs(block).iter().enumerate() {
394                    let blockparams_in = self.func.block_params(succ);
395                    let blockparams_out = self.func.branch_blockparams(block, insns.last(), i);
396                    for (&blockparam_in, &blockparam_out) in
397                        blockparams_in.iter().zip(blockparams_out)
398                    {
399                        let blockparam_out = VRegIndex::new(blockparam_out.vreg());
400                        let blockparam_in = VRegIndex::new(blockparam_in.vreg());
401                        self.blockparam_outs.push(BlockparamOut {
402                            to_vreg: blockparam_in,
403                            to_block: succ,
404                            from_block: block,
405                            from_vreg: blockparam_out,
406                        });
407
408                        // Include outgoing blockparams in the initial live set.
409                        live.set(blockparam_out.index(), true);
410                    }
411                }
412            }
413
414            // Initially, registers are assumed live for the whole block.
415            for vreg in live.iter() {
416                let range = CodeRange {
417                    from: self.cfginfo.block_entry[block.index()],
418                    to: self.cfginfo.block_exit[block.index()].next(),
419                };
420                trace!(
421                    "vreg {:?} live at end of block --> create range {:?}",
422                    VRegIndex::new(vreg),
423                    range
424                );
425                let lr = self.add_liverange_to_vreg(VRegIndex::new(vreg), range);
426                vreg_ranges[vreg] = lr;
427            }
428
429            // Create vreg data for blockparams.
430            for &param in self.func.block_params(block) {
431                self.vregs[param].blockparam = block;
432            }
433
434            // For each instruction, in reverse order, process
435            // operands and clobbers.
436            for inst in insns.iter().rev() {
437                // Mark clobbers with CodeRanges on PRegs.
438                for clobber in self.func.inst_clobbers(inst) {
439                    // Clobber range is at After point only: an
440                    // instruction can still take an input in a reg
441                    // that it later clobbers. (In other words, the
442                    // clobber is like a normal def that never gets
443                    // used.)
444                    let range = CodeRange {
445                        from: ProgPoint::after(inst),
446                        to: ProgPoint::before(inst.next()),
447                    };
448                    self.add_liverange_to_preg(range, clobber);
449                }
450
451                // Does the instruction have any input-reusing
452                // outputs? This is important below to establish
453                // proper interference wrt other inputs. We note the
454                // *vreg* that is reused, not the index.
455                let mut reused_input = None;
456                for op in self.func.inst_operands(inst) {
457                    if let OperandConstraint::Reuse(i) = op.constraint() {
458                        debug_assert!(self.func.inst_operands(inst)[i]
459                            .as_fixed_nonallocatable()
460                            .is_none());
461                        reused_input = Some(self.func.inst_operands(inst)[i].vreg());
462                        break;
463                    }
464                }
465
466                // Preprocess defs and uses. Specifically, if there
467                // are any fixed-reg-constrained defs at Late position
468                // and fixed-reg-constrained uses at Early position
469                // with the same preg, we need to (i) add a fixup move
470                // for the use, (ii) rewrite the use to have an Any
471                // constraint, and (ii) move the def to Early position
472                // to reserve the register for the whole instruction.
473                //
474                // We don't touch any fixed-early-def or fixed-late-use
475                // constraints: the only situation where the same physical
476                // register can be used multiple times in the same
477                // instruction is with an early-use and a late-def. Anything
478                // else is a user error.
479                let mut operand_rewrites: FxHashMap<usize, Operand> = FxHashMap::default();
480                let mut late_def_fixed: SmallVec<[PReg; 8]> = smallvec![];
481                for &operand in self.func.inst_operands(inst) {
482                    if let OperandConstraint::FixedReg(preg) = operand.constraint() {
483                        match (operand.pos(), operand.kind()) {
484                            (OperandPos::Late, OperandKind::Def) => {
485                                late_def_fixed.push(preg);
486                            }
487                            _ => {}
488                        }
489                    }
490                }
491                for (i, &operand) in self.func.inst_operands(inst).iter().enumerate() {
492                    if operand.as_fixed_nonallocatable().is_some() {
493                        continue;
494                    }
495                    if let OperandConstraint::FixedReg(preg) = operand.constraint() {
496                        match (operand.pos(), operand.kind()) {
497                            (OperandPos::Early, OperandKind::Use)
498                                if live.get(operand.vreg().vreg()) =>
499                            {
500                                // If we have a use constraint at the
501                                // Early point for a fixed preg, and
502                                // this preg is also constrained with
503                                // a *separate* def at Late or is
504                                // clobbered, and *if* the vreg is
505                                // live downward, we have to use the
506                                // multi-fixed-reg mechanism for a
507                                // fixup and rewrite here without the
508                                // constraint. See #53.
509                                //
510                                // We adjust the def liverange and Use
511                                // to an "early" position to reserve
512                                // the register, it still must not be
513                                // used by some other vreg at the
514                                // use-site.
515                                //
516                                // Note that we handle multiple
517                                // conflicting constraints for the
518                                // same vreg in a separate pass (see
519                                // `fixup_multi_fixed_vregs` below).
520                                if late_def_fixed.contains(&preg)
521                                    || self.func.inst_clobbers(inst).contains(preg)
522                                {
523                                    trace!(
524                                        concat!(
525                                            "-> operand {:?} is fixed to preg {:?}, ",
526                                            "is downward live, and there is also a ",
527                                            "def or clobber at this preg"
528                                        ),
529                                        operand,
530                                        preg
531                                    );
532                                    let pos = ProgPoint::before(inst);
533                                    self.multi_fixed_reg_fixups.push(MultiFixedRegFixup {
534                                        pos,
535                                        from_slot: i as u8,
536                                        to_slot: i as u8,
537                                        to_preg: PRegIndex::new(preg.index()),
538                                        vreg: VRegIndex::new(operand.vreg().vreg()),
539                                        level: FixedRegFixupLevel::Initial,
540                                    });
541
542                                    // We need to insert a reservation
543                                    // at the before-point to reserve
544                                    // the reg for the use too.
545                                    let range = CodeRange::singleton(pos);
546                                    self.add_liverange_to_preg(range, preg);
547
548                                    // Remove the fixed-preg
549                                    // constraint from the Use.
550                                    operand_rewrites.insert(
551                                        i,
552                                        Operand::new(
553                                            operand.vreg(),
554                                            OperandConstraint::Any,
555                                            operand.kind(),
556                                            operand.pos(),
557                                        ),
558                                    );
559                                }
560                            }
561                            _ => {}
562                        }
563                    }
564                }
565
566                // Process defs and uses.
567                for &cur_pos in &[InstPosition::After, InstPosition::Before] {
568                    for i in 0..self.func.inst_operands(inst).len() {
569                        // don't borrow `self`
570                        let operand = operand_rewrites
571                            .get(&i)
572                            .cloned()
573                            .unwrap_or(self.func.inst_operands(inst)[i]);
574                        let pos = match (operand.kind(), operand.pos()) {
575                            (OperandKind::Def, OperandPos::Early) => ProgPoint::before(inst),
576                            (OperandKind::Def, OperandPos::Late) => ProgPoint::after(inst),
577                            (OperandKind::Use, OperandPos::Late) => ProgPoint::after(inst),
578                            // If there are any reused inputs in this
579                            // instruction, and this is *not* the
580                            // reused vreg, force `pos` to
581                            // `After`. This ensures that we correctly
582                            // account for the interference between
583                            // the other inputs and the
584                            // input-that-is-reused/output.
585                            (OperandKind::Use, OperandPos::Early)
586                                if reused_input.is_some()
587                                    && reused_input.unwrap() != operand.vreg() =>
588                            {
589                                ProgPoint::after(inst)
590                            }
591                            (OperandKind::Use, OperandPos::Early) => ProgPoint::before(inst),
592                        };
593
594                        if pos.pos() != cur_pos {
595                            continue;
596                        }
597
598                        trace!(
599                            "processing inst{} operand at {:?}: {:?}",
600                            inst.index(),
601                            pos,
602                            operand
603                        );
604
605                        // If this is a "fixed non-allocatable
606                        // register" operand, set the alloc
607                        // immediately and then ignore the operand
608                        // hereafter.
609                        if let Some(preg) = operand.as_fixed_nonallocatable() {
610                            self.set_alloc(inst, i, Allocation::reg(preg));
611                            continue;
612                        }
613
614                        match operand.kind() {
615                            OperandKind::Def => {
616                                trace!("Def of {} at {:?}", operand.vreg(), pos);
617
618                                // Get or create the LiveRange.
619                                let mut lr = vreg_ranges[operand.vreg().vreg()];
620                                trace!(" -> has existing LR {:?}", lr);
621                                // If there was no liverange (dead def), create a trivial one.
622                                if !live.get(operand.vreg().vreg()) {
623                                    let from = pos;
624                                    // We want to we want to span
625                                    // until Before of the next
626                                    // inst. This ensures that early
627                                    // defs used for temps on an
628                                    // instruction are reserved across
629                                    // the whole instruction.
630                                    let to = ProgPoint::before(pos.inst().next());
631                                    lr = self.add_liverange_to_vreg(
632                                        VRegIndex::new(operand.vreg().vreg()),
633                                        CodeRange { from, to },
634                                    );
635                                    trace!(" -> invalid; created {:?}", lr);
636                                    vreg_ranges[operand.vreg().vreg()] = lr;
637                                    live.set(operand.vreg().vreg(), true);
638                                }
639                                // Create the use in the LiveRange.
640                                self.insert_use_into_liverange(lr, Use::new(operand, pos, i as u8));
641                                // If def (not mod), this reg is now dead,
642                                // scanning backward; make it so.
643                                if operand.kind() == OperandKind::Def {
644                                    // Trim the range for this vreg to start
645                                    // at `pos` if it previously ended at the
646                                    // start of this block (i.e. was not
647                                    // merged into some larger LiveRange due
648                                    // to out-of-order blocks).
649                                    if self.ranges[lr].range.from
650                                        == self.cfginfo.block_entry[block.index()]
651                                    {
652                                        trace!(" -> started at block start; trimming to {:?}", pos);
653                                        self.ranges[lr].range.from = pos;
654                                    }
655
656                                    self.ranges[lr].set_flag(LiveRangeFlag::StartsAtDef);
657
658                                    // Remove from live-set.
659                                    live.set(operand.vreg().vreg(), false);
660                                    vreg_ranges[operand.vreg().vreg()] = LiveRangeIndex::invalid();
661                                }
662                            }
663                            OperandKind::Use => {
664                                // Create/extend the LiveRange if it
665                                // doesn't already exist, and add the use
666                                // to the range.
667                                let mut lr = vreg_ranges[operand.vreg().vreg()];
668                                if !live.get(operand.vreg().vreg()) {
669                                    let range = CodeRange {
670                                        from: self.cfginfo.block_entry[block.index()],
671                                        to: pos.next(),
672                                    };
673                                    lr = self.add_liverange_to_vreg(
674                                        VRegIndex::new(operand.vreg().vreg()),
675                                        range,
676                                    );
677                                    vreg_ranges[operand.vreg().vreg()] = lr;
678                                }
679                                debug_assert!(lr.is_valid());
680
681                                trace!("Use of {:?} at {:?} -> {:?}", operand, pos, lr,);
682
683                                self.insert_use_into_liverange(lr, Use::new(operand, pos, i as u8));
684
685                                // Add to live-set.
686                                live.set(operand.vreg().vreg(), true);
687                            }
688                        }
689                    }
690                }
691            }
692
693            // Block parameters define vregs at the very beginning of
694            // the block. Remove their live vregs from the live set
695            // here.
696            for vreg in self.func.block_params(block) {
697                if live.get(vreg.vreg()) {
698                    live.set(vreg.vreg(), false);
699                } else {
700                    // Create trivial liverange if blockparam is dead.
701                    let start = self.cfginfo.block_entry[block.index()];
702                    self.add_liverange_to_vreg(
703                        VRegIndex::new(vreg.vreg()),
704                        CodeRange {
705                            from: start,
706                            to: start.next(),
707                        },
708                    );
709                }
710                // add `blockparam_ins` entries.
711                let vreg_idx = VRegIndex::new(vreg.vreg());
712                for &pred in self.func.block_preds(block) {
713                    self.blockparam_ins.push(BlockparamIn {
714                        to_vreg: vreg_idx,
715                        to_block: block,
716                        from_block: pred,
717                    });
718                }
719            }
720        }
721
722        // Make ranges in each vreg and uses in each range appear in
723        // sorted order. We built them in reverse order above, so this
724        // is a simple reversal, *not* a full sort.
725        //
726        // The ordering invariant is always maintained for uses and
727        // always for ranges in bundles (which are initialized later),
728        // but not always for ranges in vregs; those are sorted only
729        // when needed, here and then again at the end of allocation
730        // when resolving moves.
731
732        for vreg in &mut self.vregs {
733            vreg.ranges.reverse();
734            let mut last = None;
735            for entry in &mut vreg.ranges {
736                // Ranges may have been truncated above at defs. We
737                // need to update with the final range here.
738                entry.range = self.ranges[entry.index].range;
739                // Assert in-order and non-overlapping.
740                debug_assert!(last.is_none() || last.unwrap() <= entry.range.from);
741                last = Some(entry.range.to);
742            }
743        }
744
745        for range in &mut self.ranges {
746            range.uses.reverse();
747            debug_assert!(range.uses.windows(2).all(|win| win[0].pos <= win[1].pos));
748        }
749
750        self.blockparam_ins.sort_unstable_by_key(|x| x.key());
751        self.blockparam_outs.sort_unstable_by_key(|x| x.key());
752
753        self.stats.initial_liverange_count = self.ranges.len();
754        self.stats.blockparam_ins_count = self.blockparam_ins.len();
755        self.stats.blockparam_outs_count = self.blockparam_outs.len();
756    }
757
758    pub fn fixup_multi_fixed_vregs(&mut self) {
759        // Do a fixed-reg cleanup pass: if there are any LiveRanges with
760        // multiple uses at the same ProgPoint and there is
761        // more than one FixedReg constraint at that ProgPoint, we
762        // need to record all but one of them in a special fixup list
763        // and handle them later; otherwise, bundle-splitting to
764        // create minimal bundles becomes much more complex (we would
765        // have to split the multiple uses at the same progpoint into
766        // different bundles, which breaks invariants related to
767        // disjoint ranges and bundles).
768        let mut extra_clobbers: SmallVec<[(PReg, ProgPoint); 8]> = smallvec![];
769        for vreg in 0..self.vregs.len() {
770            let vreg = VRegIndex::new(vreg);
771            for range_idx in 0..self.vregs[vreg].ranges.len() {
772                let entry = self.vregs[vreg].ranges[range_idx];
773                let range = entry.index;
774                trace!("multi-fixed-reg cleanup: vreg {:?} range {:?}", vreg, range,);
775
776                // Find groups of uses that occur in at the same program point.
777                for uses in self.ranges[range].uses.linear_group_by_key_mut(|u| u.pos) {
778                    if uses.len() < 2 {
779                        continue;
780                    }
781
782                    // Search for conflicting constraints in the uses.
783                    let mut requires_reg = false;
784                    let mut num_fixed_reg = 0;
785                    let mut num_fixed_stack = 0;
786                    let mut first_reg_slot = None;
787                    let mut first_stack_slot = None;
788                    for u in uses.iter() {
789                        match u.operand.constraint() {
790                            OperandConstraint::Any => {
791                                first_reg_slot.get_or_insert(u.slot);
792                                first_stack_slot.get_or_insert(u.slot);
793                            }
794                            OperandConstraint::Reg | OperandConstraint::Reuse(_) => {
795                                first_reg_slot.get_or_insert(u.slot);
796                                requires_reg = true;
797                            }
798                            OperandConstraint::FixedReg(preg) => {
799                                if self.pregs[preg.index()].is_stack {
800                                    num_fixed_stack += 1;
801                                    first_stack_slot.get_or_insert(u.slot);
802                                } else {
803                                    requires_reg = true;
804                                    num_fixed_reg += 1;
805                                    first_reg_slot.get_or_insert(u.slot);
806                                }
807                            }
808                        }
809                    }
810
811                    // Fast path if there are no conflicts.
812                    if num_fixed_reg + num_fixed_stack <= 1
813                        && !(requires_reg && num_fixed_stack != 0)
814                    {
815                        continue;
816                    }
817
818                    // We pick one constraint (in order: FixedReg, Reg, FixedStack)
819                    // and then rewrite any incompatible constraints to Any.
820                    // This allows register allocation to succeed and we will
821                    // later insert moves to satisfy the rewritten constraints.
822                    let source_slot = if requires_reg {
823                        first_reg_slot.unwrap()
824                    } else {
825                        first_stack_slot.unwrap()
826                    };
827                    let mut first_preg = None;
828                    for u in uses.iter_mut() {
829                        if let OperandConstraint::FixedReg(preg) = u.operand.constraint() {
830                            let vreg_idx = VRegIndex::new(u.operand.vreg().vreg());
831                            let preg_idx = PRegIndex::new(preg.index());
832                            trace!(
833                                "at pos {:?}, vreg {:?} has fixed constraint to preg {:?}",
834                                u.pos,
835                                vreg_idx,
836                                preg_idx
837                            );
838
839                            // FixedStack is incompatible if there are any
840                            // Reg/FixedReg constraints. FixedReg is
841                            // incompatible if there already is a different
842                            // FixedReg constraint. If either condition is true,
843                            // we edit the constraint below; otherwise, we can
844                            // skip this edit.
845                            if !(requires_reg && self.pregs[preg.index()].is_stack)
846                                && *first_preg.get_or_insert(preg) == preg
847                            {
848                                continue;
849                            }
850
851                            trace!(" -> duplicate; switching to constraint Any");
852                            self.multi_fixed_reg_fixups.push(MultiFixedRegFixup {
853                                pos: u.pos,
854                                from_slot: source_slot,
855                                to_slot: u.slot,
856                                to_preg: preg_idx,
857                                vreg: vreg_idx,
858                                level: FixedRegFixupLevel::Secondary,
859                            });
860                            u.operand = Operand::new(
861                                u.operand.vreg(),
862                                OperandConstraint::Any,
863                                u.operand.kind(),
864                                u.operand.pos(),
865                            );
866                            trace!(" -> extra clobber {} at inst{}", preg, u.pos.inst().index());
867                            extra_clobbers.push((preg, u.pos));
868                        }
869                    }
870                }
871
872                for (clobber, pos) in extra_clobbers.drain(..) {
873                    let range = CodeRange {
874                        from: pos,
875                        to: pos.next(),
876                    };
877                    self.add_liverange_to_preg(range, clobber);
878                }
879            }
880        }
881    }
