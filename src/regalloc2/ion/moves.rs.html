35    pub fn is_start_of_block(&self, pos: ProgPoint) -> bool {
36        let block = self.cfginfo.insn_block[pos.inst().index()];
37        pos == self.cfginfo.block_entry[block.index()]
38    }
39    pub fn is_end_of_block(&self, pos: ProgPoint) -> bool {
40        let block = self.cfginfo.insn_block[pos.inst().index()];
41        pos == self.cfginfo.block_exit[block.index()]
42    }

45        let inst_allocs = &self.allocs[self.inst_alloc_offsets[inst.index()] as usize..];

54    pub fn get_alloc_for_range(&self, range: LiveRangeIndex) -> Allocation {
55        trace!("get_alloc_for_range: {:?}", range);
56        let bundle = self.ranges[range].bundle;
57        trace!(" -> bundle: {:?}", bundle);
58        let bundledata = &self.bundles[bundle];
59        trace!(" -> allocation {:?}", bundledata.allocation);
60        if bundledata.allocation != Allocation::none() {
61            bundledata.allocation
62        } else {
63            trace!(" -> spillset {:?}", bundledata.spillset);
64            trace!(
65                " -> spill slot {:?}",
66                self.spillsets[bundledata.spillset].slot
67            );
68            self.spillslots[self.spillsets[bundledata.spillset].slot.index()].alloc
69        }
70    }

72    pub fn apply_allocations_and_insert_moves(&mut self) -> InsertedMoves {
73        trace!("apply_allocations_and_insert_moves");
74        trace!("blockparam_ins: {:?}", self.blockparam_ins);
75        trace!("blockparam_outs: {:?}", self.blockparam_outs);
76
77        let mut inserted_moves = InsertedMoves::default();
78
79        // Now that all splits are done, we can pay the cost once to
80        // sort VReg range lists and update with the final ranges.
81        for vreg in &mut self.vregs {
82            for entry in &mut vreg.ranges {
83                entry.range = self.ranges[entry.index].range;
84            }
85            vreg.ranges.sort_unstable_by_key(|entry| entry.range.from);
86        }

88        /// Buffered information about the previous liverange that was processed.
89        struct PrevBuffer {
90            prev: Option<LiveRangeListEntry>,
91            prev_ins_idx: usize,
92            buffered: Option<LiveRangeListEntry>,
93            buffered_ins_idx: usize,
94        }
95
96        impl PrevBuffer {
97            fn new(prev_ins_idx: usize) -> Self {
98                Self {
99                    prev: None,
100                    prev_ins_idx,
101                    buffered: None,
102                    buffered_ins_idx: prev_ins_idx,
103                }
104            }
105
106            /// Returns the previous `LiveRangeListEntry` when it's present.
107            #[inline(always)]
108            fn is_valid(&self) -> Option<LiveRangeListEntry> {
109                self.prev
110            }
111
112            /// Fetch the current index into the `Env::blockparam_ins` vector.
113            #[inline(always)]
114            fn blockparam_ins_idx(&self) -> usize {
115                self.prev_ins_idx
116            }
117
118            /// Record this index as the next index to use when the previous liverange buffer
119            /// anvances.
120            #[inline(always)]
121            fn update_blockparam_ins_idx(&mut self, idx: usize) {
122                self.buffered_ins_idx = idx;
123            }
124
125            /// As overlapping liveranges might start at the same program point, we buffer the
126            /// previous liverange used when determining where to take the last value from for
127            /// intra-block moves. The liveranges we process are buffered until we encounter one
128            /// that starts at a later program point, indicating that it's now safe to advance the
129            /// previous LR buffer. We accumulate the longest-lived liverange in the buffer as a
130            /// heuristic for finding the most stable source of a value.
131            ///
132            /// We also buffer the index into the `Env::blockparam_ins` vector, as we may see
133            /// multiple uses of a blockparam within a single instruction, and as such may need to
134            /// generate multiple blockparam move destinations by re-traversing that section of the
135            /// vector.
136            #[inline(always)]
137            fn advance(&mut self, current: LiveRangeListEntry) {
138                // Advance the `prev` pointer to the `next` pointer, as long as the `next` pointer
139                // does not start at the same time as the current LR we're processing.
140                if self
141                    .buffered
142                    .map(|entry| entry.range.from < current.range.from)
143                    .unwrap_or(false)
144                {
145                    self.prev = self.buffered;
146                    self.prev_ins_idx = self.buffered_ins_idx;
147                }
148
149                // Advance the `next` pointer to the currently processed LR, as long as it ends
150                // later than the current `next`.
151                if self
152                    .buffered
153                    .map(|entry| entry.range.to < current.range.to)
154                    .unwrap_or(true)
155                {
156                    self.buffered = Some(current);
157                }
158            }
159        }
160
161        // Determine the ProgPoint where moves on this (from, to)
162        // edge should go:
163        // - If there is more than one in-edge to `to`, then
164        //   `from` must have only one out-edge; moves go at tail of
165        //   `from` just before last Branch/Ret.
166        // - Otherwise, there must be at most one in-edge to `to`,
167        //   and moves go at start of `to`.
168        #[inline(always)]
169        fn choose_move_location<'a, F: Function>(
170            env: &Env<'a, F>,
171            from: Block,
172            to: Block,
173        ) -> (ProgPoint, InsertMovePrio) {
174            let from_last_insn = env.func.block_insns(from).last();
175            let to_first_insn = env.func.block_insns(to).first();
176            let from_is_ret = env.func.is_ret(from_last_insn);
177            let to_is_entry = env.func.entry_block() == to;
178            let from_outs = env.func.block_succs(from).len() + if from_is_ret { 1 } else { 0 };
179            let to_ins = env.func.block_preds(to).len() + if to_is_entry { 1 } else { 0 };
180
181            if to_ins > 1 && from_outs <= 1 {
182                (
183                    // N.B.: though semantically the edge moves happen
184                    // after the branch, we must insert them before
185                    // the branch because otherwise, of course, they
186                    // would never execute. This is correct even in
187                    // the presence of branches that read register
188                    // inputs (e.g. conditional branches on some RISCs
189                    // that branch on reg zero/not-zero, or any
190                    // indirect branch), but for a very subtle reason:
191                    // all cases of such branches will (or should)
192                    // have multiple successors, and thus due to
193                    // critical-edge splitting, their successors will
194                    // have only the single predecessor, and we prefer
195                    // to insert at the head of the successor in that
196                    // case (rather than here). We make this a
197                    // requirement, in fact: the user of this library
198                    // shall not read registers in a branch
199                    // instruction of there is only one successor per
200                    // the given CFG information.
201                    ProgPoint::before(from_last_insn),
202                    InsertMovePrio::OutEdgeMoves,
203                )
204            } else if to_ins <= 1 {
205                (
206                    ProgPoint::before(to_first_insn),
207                    InsertMovePrio::InEdgeMoves,
208                )
209            } else {
210                panic!(
211                    "Critical edge: can't insert moves between blocks {:?} and {:?}",
212                    from, to
213                );
214            }
215        }

218        struct InterBlockDest {
219            to: Block,
220            from: Block,
221            alloc: Allocation,
222        }

224        impl InterBlockDest {
225            fn key(&self) -> u64 {
226                u64_key(self.from.raw_u32(), self.to.raw_u32())

228        }

230        let mut inter_block_sources: FxHashMap<Block, Allocation> = FxHashMap::default();
231        let mut inter_block_dests = Vec::with_capacity(self.func.num_blocks());

236        }
237
238        impl BlockparamSourceKey {
239            fn new(from_block: Block, to_vreg: VRegIndex) -> Self {

246        struct BlockparamDest {
247            from_block: Block,
248            to_block: Block,
249            to_vreg: VRegIndex,
250            alloc: Allocation,
251        }
252
253        impl BlockparamDest {
254            fn key(&self) -> u64 {
255                u64_key(self.to_block.raw_u32(), self.from_block.raw_u32())
256            }
257
258            fn source(&self) -> BlockparamSourceKey {
259                BlockparamSourceKey::new(self.from_block, self.to_vreg)
260            }
261        }
262
263        let mut block_param_sources =
264            FxHashMap::<BlockparamSourceKey, Allocation>::with_capacity_and_hasher(
265                3 * self.func.num_insts(),
266                Default::default(),
267            );
268        let mut block_param_dests = Vec::with_capacity(3 * self.func.num_insts());
269
270        let debug_labels = self.func.debug_value_labels();
271
272        let mut reuse_input_insts = Vec::with_capacity(self.func.num_insts() / 2);
273
274        let mut blockparam_in_idx = 0;
275        let mut blockparam_out_idx = 0;
276        for vreg in 0..self.vregs.len() {
277            let vreg = VRegIndex::new(vreg);
278            if !self.is_vreg_used(vreg) {
279                continue;
280            }
281
282            inter_block_sources.clear();
283
284            // For each range in each vreg, insert moves or
285            // half-moves.  We also scan over `blockparam_ins` and
286            // `blockparam_outs`, which are sorted by (block, vreg),
287            // to fill in allocations.
288            let mut prev = PrevBuffer::new(blockparam_in_idx);
289            for range_idx in 0..self.vregs[vreg].ranges.len() {
290                let entry = self.vregs[vreg].ranges[range_idx];
291                let alloc = self.get_alloc_for_range(entry.index);
292                let range = entry.range;
293                trace!(
294                    "apply_allocations: vreg {:?} LR {:?} with range {:?} has alloc {:?}",
295                    vreg,
296                    entry.index,
297                    range,
298                    alloc,
299                );
300                debug_assert!(alloc != Allocation::none());
301
302                if self.annotations_enabled {
303                    self.annotate(
304                        range.from,
305                        format!(
306                            " <<< start v{} in {} (range{}) (bundle{})",
307                            vreg.index(),
308                            alloc,
309                            entry.index.index(),
310                            self.ranges[entry.index].bundle.raw_u32(),
311                        ),
312                    );
313                    self.annotate(
314                        range.to,
315                        format!(
316                            "     end   v{} in {} (range{}) (bundle{}) >>>",
317                            vreg.index(),
318                            alloc,
319                            entry.index.index(),
320                            self.ranges[entry.index].bundle.raw_u32(),
321                        ),
322                    );
323                }
324
325                prev.advance(entry);
326
327                // Does this range follow immediately after a prior
328                // range in the same block? If so, insert a move (if
329                // the allocs differ). We do this directly rather than
330                // with half-moves because we eagerly know both sides
331                // already (and also, half-moves are specific to
332                // inter-block transfers).
333                //
334                // Note that we do *not* do this if there is also a
335                // def as the first use in the new range: it's
336                // possible that an old liverange covers the Before
337                // pos of an inst, a new liverange covers the After
338                // pos, and the def also happens at After. In this
339                // case we don't want to an insert a move after the
340                // instruction copying the old liverange.
341                //
342                // Note also that we assert that the new range has to
343                // start at the Before-point of an instruction; we
344                // can't insert a move that logically happens just
345                // before After (i.e. in the middle of a single
346                // instruction).
347                if let Some(prev) = prev.is_valid() {
348                    let prev_alloc = self.get_alloc_for_range(prev.index);
349                    debug_assert!(prev_alloc != Allocation::none());
350
351                    if prev.range.to >= range.from
352                        && (prev.range.to > range.from || !self.is_start_of_block(range.from))
353                        && !self.ranges[entry.index].has_flag(LiveRangeFlag::StartsAtDef)
354                    {
355                        trace!(
356                            "prev LR {} abuts LR {} in same block; moving {} -> {} for v{}",
357                            prev.index.index(),
358                            entry.index.index(),
359                            prev_alloc,
360                            alloc,
361                            vreg.index()
362                        );
363                        debug_assert_eq!(range.from.pos(), InstPosition::Before);
364                        inserted_moves.push(
365                            range.from,
366                            InsertMovePrio::Regular,
367                            prev_alloc,
368                            alloc,
369                            self.vreg(vreg),
370                        );
371                    }
372                }
373
374                // Scan over blocks whose ends are covered by this
375                // range. For each, for each successor that is not
376                // already in this range (hence guaranteed to have the
377                // same allocation) and if the vreg is live, add a
378                // Source half-move.
379                let mut block = self.cfginfo.insn_block[range.from.inst().index()];
380                while block.is_valid() && block.index() < self.func.num_blocks() {
381                    if range.to < self.cfginfo.block_exit[block.index()].next() {
382                        break;
383                    }
384                    trace!("examining block with end in range: block{}", block.index());
385
386                    match inter_block_sources.entry(block) {
387                        // If the entry is already present in the map, we'll try to prefer a
388                        // register allocation.
389                        Entry::Occupied(mut entry) => {
390                            if !entry.get().is_reg() {
391                                entry.insert(alloc);
392                            }
393                        }
394                        Entry::Vacant(entry) => {
395                            entry.insert(alloc);
396                        }
397                    }
398
399                    // Scan forward in `blockparam_outs`, adding all
400                    // half-moves for outgoing values to blockparams
401                    // in succs.
402                    trace!(
403                        "scanning blockparam_outs for v{} block{}: blockparam_out_idx = {}",
404                        vreg.index(),
405                        block.index(),
406                        blockparam_out_idx,
407                    );
408                    while blockparam_out_idx < self.blockparam_outs.len() {
409                        let BlockparamOut {
410                            from_vreg,
411                            from_block,
412                            to_block,
413                            to_vreg,
414                        } = self.blockparam_outs[blockparam_out_idx];
415                        if (from_vreg, from_block) > (vreg, block) {
416                            break;
417                        }
418                        if (from_vreg, from_block) == (vreg, block) {
419                            trace!(
420                                " -> found: from v{} block{} to v{} block{}",
421                                from_vreg.index(),
422                                from_block.index(),
423                                to_vreg.index(),
424                                to_vreg.index()
425                            );
426
427                            let key = BlockparamSourceKey::new(from_block, to_vreg);
428                            match block_param_sources.entry(key) {
429                                // As with inter-block moves, if the entry is already present we'll
430                                // try to prefer a register allocation.
431                                Entry::Occupied(mut entry) => {
432                                    if !entry.get().is_reg() {
433                                        entry.insert(alloc);
434                                    }
435                                }
436                                Entry::Vacant(entry) => {
437                                    entry.insert(alloc);
438                                }
439                            }
440
441                            if self.annotations_enabled {
442                                self.annotate(
443                                    self.cfginfo.block_exit[block.index()],
444                                    format!(
445                                        "blockparam-out: block{} to block{}: v{} to v{} in {}",
446                                        from_block.index(),
447                                        to_block.index(),
448                                        from_vreg.index(),
449                                        to_vreg.index(),
450                                        alloc
451                                    ),
452                                );
453                            }
454                        }
455
456                        blockparam_out_idx += 1;
457                    }
458
459                    block = block.next();
460                }
461
462                // Scan over blocks whose beginnings are covered by
463                // this range and for which the vreg is live at the
464                // start of the block. For each, for each predecessor,
465                // add a Dest half-move.
466                let mut block = self.cfginfo.insn_block[range.from.inst().index()];
467                if self.cfginfo.block_entry[block.index()] < range.from {
468                    block = block.next();
469                }
470                while block.is_valid() && block.index() < self.func.num_blocks() {
471                    if self.cfginfo.block_entry[block.index()] >= range.to {
472                        break;
473                    }
474
475                    // Add half-moves for blockparam inputs.
476                    trace!(
477                        "scanning blockparam_ins at vreg {} block {}: blockparam_in_idx = {}",
478                        vreg.index(),
479                        block.index(),
480                        prev.prev_ins_idx,
481                    );
482                    let mut idx = prev.blockparam_ins_idx();
483                    while idx < self.blockparam_ins.len() {
484                        let BlockparamIn {
485                            from_block,
486                            to_block,
487                            to_vreg,
488                        } = self.blockparam_ins[idx];
489                        if (to_vreg, to_block) > (vreg, block) {
490                            break;
491                        }
492                        if (to_vreg, to_block) == (vreg, block) {
493                            block_param_dests.push(BlockparamDest {
494                                from_block,
495                                to_block,
496                                to_vreg,
497                                alloc,
498                            });
499                            trace!(
500                                "match: blockparam_in: v{} in block{} from block{} into {}",
501                                to_vreg.index(),
502                                to_block.index(),
503                                from_block.index(),
504                                alloc,
505                            );
506                            #[cfg(debug_assertions)]
507                            if self.annotations_enabled {
508                                self.annotate(
509                                    self.cfginfo.block_entry[block.index()],
510                                    format!(
511                                        "blockparam-in: block{} to block{}:into v{} in {}",
512                                        from_block.index(),
513                                        to_block.index(),
514                                        to_vreg.index(),
515                                        alloc
516                                    ),
517                                );
518                            }
519                        }
520                        idx += 1;
521                    }
522
523                    prev.update_blockparam_ins_idx(idx);
524
525                    if !self.is_live_in(block, vreg) {
526                        block = block.next();
527                        continue;
528                    }
529
530                    trace!(
531                        "scanning preds at vreg {} block {} for ends outside the range",
532                        vreg.index(),
533                        block.index()
534                    );
535
536                    // Now find any preds whose ends are not in the
537                    // same range, and insert appropriate moves.
538                    for &pred in self.func.block_preds(block) {
539                        trace!(
540                            "pred block {} has exit {:?}",
541                            pred.index(),
542                            self.cfginfo.block_exit[pred.index()]
543                        );
544                        if range.contains_point(self.cfginfo.block_exit[pred.index()]) {
545                            continue;
546                        }
547
548                        inter_block_dests.push(InterBlockDest {
549                            from: pred,
550                            to: block,
551                            alloc,
552                        })
553                    }
554
555                    block = block.next();
556                }
557
558                // Scan over def/uses and apply allocations.
559                for use_idx in 0..self.ranges[entry.index].uses.len() {
560                    let usedata = self.ranges[entry.index].uses[use_idx];
561                    trace!("applying to use: {:?}", usedata);
562                    debug_assert!(range.contains_point(usedata.pos));
563                    let inst = usedata.pos.inst();
564                    let slot = usedata.slot;
565                    let operand = usedata.operand;
566                    self.set_alloc(inst, slot as usize, alloc);
567                    if let OperandConstraint::Reuse(_) = operand.constraint() {
568                        reuse_input_insts.push(inst);
569                    }
570                }
571
572                // Scan debug-labels on this vreg that overlap with
573                // this range, producing a debug-info output record
574                // giving the allocation location for each label.
575                if !debug_labels.is_empty() {
576                    // Do a binary search to find the start of any
577                    // labels for this vreg. Recall that we require
578                    // debug-label requests to be sorted by vreg as a
579                    // precondition (which we verified above).
580                    let start = debug_labels
581                        .binary_search_by(|&(label_vreg, _label_from, _label_to, _label)| {
582                            // Search for the point just before the first
583                            // tuple that could be for `vreg` overlapping
584                            // with `range`. Never return
585                            // `Ordering::Equal`; `binary_search_by` in
586                            // this case returns the index of the first
587                            // entry that is greater as an `Err`.
588                            if label_vreg.vreg() < vreg.index() {
589                                core::cmp::Ordering::Less
590                            } else {
591                                core::cmp::Ordering::Greater
592                            }
593                        })
594                        .unwrap_err();
595
596                    for &(label_vreg, label_from, label_to, label) in &debug_labels[start..] {
597                        let label_from = ProgPoint::before(label_from);
598                        let label_to = ProgPoint::before(label_to);
599                        let label_range = CodeRange {
600                            from: label_from,
601                            to: label_to,
602                        };
603                        if label_vreg.vreg() != vreg.index() {
604                            break;
605                        }
606                        if !range.overlaps(&label_range) {
607                            continue;
608                        }
609
610                        let from = core::cmp::max(label_from, range.from);
611                        let to = core::cmp::min(label_to, range.to);
612
613                        self.debug_locations.push((label, from, to, alloc));
614                    }
615                }
616            }
617
618            if !inter_block_dests.is_empty() {
619                self.stats.halfmoves_count += inter_block_dests.len() * 2;
620
621                inter_block_dests.sort_unstable_by_key(InterBlockDest::key);
622
623                let vreg = self.vreg(vreg);
624                trace!("processing inter-block moves for {}", vreg);
625                for dest in inter_block_dests.drain(..) {
626                    let src = inter_block_sources[&dest.from];
627
628                    trace!(
629                        " -> moving from {} to {} between {:?} and {:?}",
630                        src,
631                        dest.alloc,
632                        dest.from,
633                        dest.to
634                    );
635
636                    let (pos, prio) = choose_move_location(self, dest.from, dest.to);
637                    inserted_moves.push(pos, prio, src, dest.alloc, vreg);
638                }
639            }
640
641            blockparam_in_idx = prev.blockparam_ins_idx();
642        }
643
644        if !block_param_dests.is_empty() {
645            self.stats.halfmoves_count += block_param_sources.len();
646            self.stats.halfmoves_count += block_param_dests.len();
647
648            trace!("processing block-param moves");
649            for dest in block_param_dests {
650                let src = dest.source();
651                let src_alloc = block_param_sources.get(&src).unwrap();
652                let (pos, prio) = choose_move_location(self, dest.from_block, dest.to_block);
653                inserted_moves.push(pos, prio, *src_alloc, dest.alloc, self.vreg(dest.to_vreg));
654            }
655        }
656
657        // Handle multi-fixed-reg constraints by copying.
658        for fixup in core::mem::replace(&mut self.multi_fixed_reg_fixups, vec![]) {
659            let from_alloc = self.get_alloc(fixup.pos.inst(), fixup.from_slot as usize);
660            let to_alloc = Allocation::reg(PReg::from_index(fixup.to_preg.index()));
661            trace!(
662                "multi-fixed-move constraint at {:?} from {} to {} for v{}",
663                fixup.pos,
664                from_alloc,
665                to_alloc,
666                fixup.vreg.index(),
667            );
668            let prio = match fixup.level {
669                FixedRegFixupLevel::Initial => InsertMovePrio::MultiFixedRegInitial,
670                FixedRegFixupLevel::Secondary => InsertMovePrio::MultiFixedRegSecondary,
671            };
672            inserted_moves.push(fixup.pos, prio, from_alloc, to_alloc, self.vreg(fixup.vreg));
673            self.set_alloc(
674                fixup.pos.inst(),
675                fixup.to_slot as usize,
676                Allocation::reg(PReg::from_index(fixup.to_preg.index())),
677            );
678        }
679
680        // Handle outputs that reuse inputs: copy beforehand, then set
681        // input's alloc to output's.
682        //
683        // Note that the output's allocation may not *actually* be
684        // valid until InstPosition::After, but the reused input may
685        // occur at InstPosition::Before. This may appear incorrect,
686        // but we make it work by ensuring that all *other* inputs are
687        // extended to InstPosition::After so that the def will not
688        // interfere. (The liveness computation code does this -- we
689        // do not require the user to do so.)
690        //
691        // One might ask: why not insist that input-reusing defs occur
692        // at InstPosition::Before? this would be correct, but would
693        // mean that the reused input and the reusing output
694        // interfere, *guaranteeing* that every such case would
695        // require a move. This is really bad on ISAs (like x86) where
696        // reused inputs are ubiquitous.
697        //
698        // Another approach might be to put the def at Before, and
699        // trim the reused input's liverange back to the previous
700        // instruction's After. This is kind of OK until (i) a block
701        // boundary occurs between the prior inst and this one, or
702        // (ii) any moves/spills/reloads occur between the two
703        // instructions. We really do need the input to be live at
704        // this inst's Before.
705        //
706        // In principle what we really need is a "BeforeBefore"
707        // program point, but we don't want to introduce that
708        // everywhere and pay the cost of twice as many ProgPoints
709        // throughout the allocator.
710        //
711        // Or we could introduce a separate move instruction -- this
712        // is the approach that regalloc.rs takes with "mod" operands
713        // -- but that is also costly.
714        //
715        // So we take this approach (invented by IonMonkey -- somewhat
716        // hard to discern, though see [0] for a comment that makes
717        // this slightly less unclear) to avoid interference between
718        // the actual reused input and reusing output, ensure
719        // interference (hence no incorrectness) between other inputs
720        // and the reusing output, and not require a separate explicit
721        // move instruction.
722        //
723        // [0] https://searchfox.org/mozilla-central/rev/3a798ef9252896fb389679f06dd3203169565af0/js/src/jit/shared/Lowering-shared-inl.h#108-110
724        for inst in reuse_input_insts {
725            let mut input_reused: SmallVec<[usize; 4]> = smallvec![];
726            for output_idx in 0..self.func.inst_operands(inst).len() {
727                let operand = self.func.inst_operands(inst)[output_idx];
728                if let OperandConstraint::Reuse(input_idx) = operand.constraint() {
729                    debug_assert!(!input_reused.contains(&input_idx));
730                    debug_assert_eq!(operand.pos(), OperandPos::Late);
731                    input_reused.push(input_idx);
732                    let input_alloc = self.get_alloc(inst, input_idx);
733                    let output_alloc = self.get_alloc(inst, output_idx);
734                    trace!(
735                        "reuse-input inst {:?}: output {} has alloc {:?}, input {} has alloc {:?}",
736                        inst,
737                        output_idx,
738                        output_alloc,
739                        input_idx,
740                        input_alloc
741                    );
742                    if input_alloc != output_alloc {
743                        #[cfg(debug_assertions)]
744                        if self.annotations_enabled {
745                            self.annotate(
746                                ProgPoint::before(inst),
747                                format!(" reuse-input-copy: {} -> {}", input_alloc, output_alloc),
748                            );
749                        }
750                        let input_operand = self.func.inst_operands(inst)[input_idx];
751                        inserted_moves.push(
752                            ProgPoint::before(inst),
753                            InsertMovePrio::ReusedInput,
754                            input_alloc,
755                            output_alloc,
756                            input_operand.vreg(),
757                        );
758                        self.set_alloc(inst, input_idx, output_alloc);
759                    }
760                }
761            }
762        }
763
764        // Sort the debug-locations vector; we provide this
765        // invariant to the client.
766        self.debug_locations.sort_unstable();
767
768        inserted_moves
769    }
770
771    pub fn resolve_inserted_moves(&mut self, mut inserted_moves: InsertedMoves) -> Edits {
772        // For each program point, gather all moves together. Then
773        // resolve (see cases below).
774        let mut i = 0;
775        inserted_moves
776            .moves
777            .sort_unstable_by_key(|m| m.pos_prio.key());
778
779        // Redundant-move elimination state tracker.
780        let mut redundant_moves = RedundantMoveEliminator::default();
781
782        fn redundant_move_process_side_effects<'a, F: Function>(
783            this: &Env<'a, F>,
784            redundant_moves: &mut RedundantMoveEliminator,
785            from: ProgPoint,
786            to: ProgPoint,
787        ) {
788            // If we cross a block boundary, clear and return.
789            if this.cfginfo.insn_block[from.inst().index()]
790                != this.cfginfo.insn_block[to.inst().index()]
791            {
792                redundant_moves.clear();
793                return;
794            }
795
796            let start_inst = if from.pos() == InstPosition::Before {
797                from.inst()
798            } else {
799                from.inst().next()
800            };
801            let end_inst = if to.pos() == InstPosition::Before {
802                to.inst()
803            } else {
804                to.inst().next()
805            };
806            for inst in start_inst.index()..end_inst.index() {
807                let inst = Inst::new(inst);
808                for (i, op) in this.func.inst_operands(inst).iter().enumerate() {
809                    match op.kind() {
810                        OperandKind::Def => {
811                            let alloc = this.get_alloc(inst, i);
812                            redundant_moves.clear_alloc(alloc);
813                        }
814                        _ => {}
815                    }
816                }
817                for reg in this.func.inst_clobbers(inst) {
818                    redundant_moves.clear_alloc(Allocation::reg(reg));
819                }
820                // The dedicated scratch registers may be clobbered by any
821                // instruction.
822                for reg in this.env.scratch_by_class {
823                    if let Some(reg) = reg {
824                        redundant_moves.clear_alloc(Allocation::reg(reg));
825                    }
826                }
827            }
828        }
829
830        let mut last_pos = ProgPoint::before(Inst::new(0));
831        let mut edits = Edits::with_capacity(self.func.num_insts());
832
833        while i < inserted_moves.moves.len() {
834            let start = i;
835            let pos_prio = inserted_moves.moves[i].pos_prio;
836            while i < inserted_moves.moves.len() && inserted_moves.moves[i].pos_prio == pos_prio {
837                i += 1;
838            }
839            let moves = &inserted_moves.moves[start..i];
840
841            redundant_move_process_side_effects(self, &mut redundant_moves, last_pos, pos_prio.pos);
842            last_pos = pos_prio.pos;
843
844            // Gather all the moves in each RegClass separately.
845            // These cannot interact, so it is safe to have separate
846            // ParallelMove instances. They need to be separate because
847            // moves between the classes are impossible. (We could
848            // enhance ParallelMoves to understand register classes, but
849            // this seems simpler.)
850            let mut int_moves: SmallVec<[InsertedMove; 8]> = smallvec![];
851            let mut float_moves: SmallVec<[InsertedMove; 8]> = smallvec![];
852            let mut vec_moves: SmallVec<[InsertedMove; 8]> = smallvec![];
853
854            for m in moves {
855                match m.to_vreg.class() {
856                    RegClass::Int => {
857                        int_moves.push(m.clone());
858                    }
859                    RegClass::Float => {
860                        float_moves.push(m.clone());
861                    }
862                    RegClass::Vector => {
863                        vec_moves.push(m.clone());
864                    }
865                }
866            }
867
868            for &(regclass, moves) in &[
869                (RegClass::Int, &int_moves),
870                (RegClass::Float, &float_moves),
871                (RegClass::Vector, &vec_moves),
872            ] {
873                // All moves in `moves` semantically happen in
874                // parallel. Let's resolve these to a sequence of moves
875                // that can be done one at a time.
876                let mut parallel_moves = ParallelMoves::new();
877                trace!(
878                    "parallel moves at pos {:?} prio {:?}",
879                    pos_prio.pos,
880                    pos_prio.prio
881                );
882                for m in moves {
883                    trace!(" {} -> {}", m.from_alloc, m.to_alloc);
884                    parallel_moves.add(m.from_alloc, m.to_alloc, Some(m.to_vreg));
885                }
886
887                let resolved = parallel_moves.resolve();
888                let mut scratch_iter = RegTraversalIter::new(
889                    self.env,
890                    regclass,
891                    PReg::invalid(),
892                    PReg::invalid(),
893                    0,
894                    None,
895                );
896                let mut dedicated_scratch = self.env.scratch_by_class[regclass as usize];
897                let key = LiveRangeKey::from_range(&CodeRange {
898                    from: pos_prio.pos,
899                    to: pos_prio.pos.next(),
900                });
901                let find_free_reg = || {
902                    // Use the dedicated scratch register first if it is
903                    // available.
904                    if let Some(reg) = dedicated_scratch.take() {
905                        return Some(Allocation::reg(reg));
906                    }
907                    while let Some(preg) = scratch_iter.next() {
908                        if !self.pregs[preg.index()]
909                            .allocations
910                            .btree
911                            .contains_key(&key)
912                        {
913                            let alloc = Allocation::reg(preg);
914                            if moves
915                                .iter()
916                                .any(|m| m.from_alloc == alloc || m.to_alloc == alloc)
917                            {
918                                // Skip pregs used by moves in this
919                                // parallel move set, even if not
920                                // marked used at progpoint: edge move
921                                // liveranges meet but don't overlap
922                                // so otherwise we may incorrectly
923                                // overwrite a source reg.
924                                continue;
925                            }
926                            return Some(alloc);
927                        }
928                    }
929                    None
930                };
931                let mut stackslot_idx = 0;
932                let get_stackslot = || {
933                    let idx = stackslot_idx;
934                    stackslot_idx += 1;
935                    // We can't borrow `self` as mutable, so we create
936                    // these placeholders then allocate the actual
937                    // slots if needed with `self.allocate_spillslot`
938                    // below.
939                    Allocation::stack(SpillSlot::new(SpillSlot::MAX - idx))
940                };
941                let is_stack_alloc = |alloc: Allocation| {
942                    if let Some(preg) = alloc.as_reg() {
943                        self.pregs[preg.index()].is_stack
944                    } else {
945                        alloc.is_stack()
946                    }
947                };
948                let preferred_victim = self.preferred_victim_by_class[regclass as usize];
949
950                let scratch_resolver = MoveAndScratchResolver {
951                    find_free_reg,
952                    get_stackslot,
953                    is_stack_alloc,
954                    borrowed_scratch_reg: preferred_victim,
955                };
956
957                let resolved = scratch_resolver.compute(resolved);
958
959                let mut rewrites = FxHashMap::default();
960                for i in 0..stackslot_idx {
961                    if i >= self.extra_spillslots_by_class[regclass as usize].len() {
962                        let slot =
963                            self.allocate_spillslot(self.func.spillslot_size(regclass) as u32);
964                        self.extra_spillslots_by_class[regclass as usize].push(slot);
965                    }
966                    rewrites.insert(
967                        Allocation::stack(SpillSlot::new(SpillSlot::MAX - i)),
968                        self.extra_spillslots_by_class[regclass as usize][i],
969                    );
970                }
971
972                for (src, dst, to_vreg) in resolved {
973                    let src = rewrites.get(&src).cloned().unwrap_or(src);
974                    let dst = rewrites.get(&dst).cloned().unwrap_or(dst);
975                    trace!("  resolved: {} -> {} ({:?})", src, dst, to_vreg);
976                    let action = redundant_moves.process_move(src, dst, to_vreg);
977                    if !action.elide {
978                        edits.add(pos_prio, src, dst);
979                    } else {
980                        trace!("    -> redundant move elided");
981                    }
982                }
983            }
984        }
985
986        // Ensure edits are in sorted ProgPoint order. N.B.: this must
987        // be a stable sort! We have to keep the order produced by the
988        // parallel-move resolver for all moves within a single sort
989        // key.
990        edits.sort();
991        self.stats.edits_count = edits.len();
992
993        // Add debug annotations.
994        if self.annotations_enabled {
995            for &(pos_prio, ref edit) in edits.iter() {
996                match edit {
997                    &Edit::Move { from, to } => {
998                        self.annotate(pos_prio.pos, format!("move {} -> {}", from, to));
999                    }
1000                }
1001            }
1002        }
1003
1004        edits
1005    }
