6//! Lightweight CFG analyses.
7
8use crate::{domtree, postorder, Block, Function, Inst, ProgPoint, RegAllocError};
9use alloc::vec;
10use alloc::vec::Vec;
11use smallvec::{smallvec, SmallVec};
12
13#[derive(Clone, Debug)]
14pub struct CFGInfo {
15    /// Postorder traversal of blocks.
16    pub postorder: Vec<Block>,
17    /// Domtree parents, indexed by block.
18    pub domtree: Vec<Block>,
19    /// For each instruction, the block it belongs to.
20    pub insn_block: Vec<Block>,
21    /// For each block, the first instruction.
22    pub block_entry: Vec<ProgPoint>,
23    /// For each block, the last instruction.
24    pub block_exit: Vec<ProgPoint>,
25    /// For each block, what is the approximate loop depth?
26    ///
27    /// This measure is fully precise iff the input CFG is reducible
28    /// and blocks are in RPO, so that loop backedges are precisely
29    /// those whose block target indices are less than their source
30    /// indices. Otherwise, it will be approximate, but should still
31    /// be usable for heuristic purposes.
32    pub approx_loop_depth: Vec<u32>,
33}
34
35impl CFGInfo {
36    pub fn new<F: Function>(f: &F) -> Result<CFGInfo, RegAllocError> {
37        let postorder = postorder::calculate(f.num_blocks(), f.entry_block(), |block| {
38            f.block_succs(block)
39        });
40        let domtree = domtree::calculate(
41            f.num_blocks(),
42            |block| f.block_preds(block),
43            &postorder[..],
44            f.entry_block(),
45        );
46        let mut insn_block = vec![Block::invalid(); f.num_insts()];
47        let mut block_entry = vec![ProgPoint::before(Inst::invalid()); f.num_blocks()];
48        let mut block_exit = vec![ProgPoint::before(Inst::invalid()); f.num_blocks()];
49        let mut backedge_in = vec![0; f.num_blocks()];
50        let mut backedge_out = vec![0; f.num_blocks()];
51
52        for block in 0..f.num_blocks() {
53            let block = Block::new(block);
54            for inst in f.block_insns(block).iter() {
55                insn_block[inst.index()] = block;
56            }
57            block_entry[block.index()] = ProgPoint::before(f.block_insns(block).first());
58            block_exit[block.index()] = ProgPoint::after(f.block_insns(block).last());
59
60            // Check critical edge condition: if there is more than
61            // one predecessor, each must have only one successor
62            // (this block).
63            let preds = f.block_preds(block).len() + if block == f.entry_block() { 1 } else { 0 };
64            if preds > 1 {
65                for &pred in f.block_preds(block) {
66                    let succs = f.block_succs(pred).len();
67                    if succs > 1 {
68                        return Err(RegAllocError::CritEdge(pred, block));
69                    }
70                }
71            }
72
73            // Check branch-arg condition: if any successors have more
74            // than one predecessor (given above, there will only be
75            // one such successor), then the last instruction of this
76            // block (the branch) cannot have any args other than the
77            // blockparams.
78            let mut require_no_branch_args = false;
79            for &succ in f.block_succs(block) {
80                let preds = f.block_preds(succ).len() + if succ == f.entry_block() { 1 } else { 0 };
81                if preds > 1 {
82                    require_no_branch_args = true;
83                    break;
84                }
85            }
86            if require_no_branch_args {
87                let last = f.block_insns(block).last();
88                if !f.inst_operands(last).is_empty() {
89                    return Err(RegAllocError::DisallowedBranchArg(last));
90                }
91            }
92
93            for &succ in f.block_succs(block) {
94                if succ.index() <= block.index() {
95                    backedge_in[succ.index()] += 1;
96                    backedge_out[block.index()] += 1;
97                }
98            }
99        }
100
101        let mut approx_loop_depth = vec![];
102        let mut backedge_stack: SmallVec<[usize; 4]> = smallvec![];
103        let mut cur_depth = 0;
104        for block in 0..f.num_blocks() {
105            if backedge_in[block] > 0 {
106                cur_depth += 1;
107                backedge_stack.push(backedge_in[block]);
108            }
109
110            approx_loop_depth.push(cur_depth);
111
112            while backedge_stack.len() > 0 && backedge_out[block] > 0 {
113                backedge_out[block] -= 1;
114                *backedge_stack.last_mut().unwrap() -= 1;
115                if *backedge_stack.last().unwrap() == 0 {
116                    cur_depth -= 1;
117                    backedge_stack.pop();
118                }
119            }
120        }
121
122        Ok(CFGInfo {
123            postorder,
124            domtree,
125            insn_block,
126            block_entry,
127            block_exit,
128            approx_loop_depth,
129        })
130    }
131
132    pub fn dominates(&self, a: Block, b: Block) -> bool {
133        domtree::dominates(&self.domtree[..], a, b)
134    }
135}
