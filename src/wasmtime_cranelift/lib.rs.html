38/// Creates a new cranelift `Signature` with no wasm params/results for the
39/// given calling convention.
40///
41/// This will add the default vmctx/etc parameters to the signature returned.
42fn blank_sig(isa: &dyn TargetIsa, call_conv: CallConv) -> ir::Signature {
43    let pointer_type = isa.pointer_type();
44    let mut sig = ir::Signature::new(call_conv);
45    // Add the caller/callee `vmctx` parameters.
46    sig.params.push(ir::AbiParam::special(
47        pointer_type,
48        ir::ArgumentPurpose::VMContext,
49    ));
50    sig.params.push(ir::AbiParam::new(pointer_type));
51    return sig;
52}

54/// Emit code for the following unbarriered memory write of the given type:
55///
56/// ```ignore
57/// *(base + offset) = value
58/// ```
59///
60/// This is intended to be used with things like `ValRaw` and the array calling
61/// convention.
62fn unbarriered_store_type_at_offset(
63    pos: &mut FuncCursor,
64    flags: ir::MemFlags,
65    base: ir::Value,
66    offset: i32,
67    value: ir::Value,
68) {
69    pos.ins().store(flags, value, base, offset);
70}
71
72/// Emit code to do the following unbarriered memory read of the given type and
73/// with the given flags:
74///
75/// ```ignore
76/// result = *(base + offset)
77/// ```
78///
79/// This is intended to be used with things like `ValRaw` and the array calling
80/// convention.
81fn unbarriered_load_type_at_offset(
82    isa: &dyn TargetIsa,
83    pos: &mut FuncCursor,
84    ty: WasmValType,
85    flags: ir::MemFlags,
86    base: ir::Value,
87    offset: i32,
88) -> ir::Value {
89    let ir_ty = value_type(isa, ty);
90    pos.ins().load(ir_ty, flags, base, offset)
91}

93/// Returns the corresponding cranelift type for the provided wasm type.
94fn value_type(isa: &dyn TargetIsa, ty: WasmValType) -> ir::types::Type {
95    match ty {
96        WasmValType::I32 => ir::types::I32,
97        WasmValType::I64 => ir::types::I64,
98        WasmValType::F32 => ir::types::F32,
99        WasmValType::F64 => ir::types::F64,
100        WasmValType::V128 => ir::types::I8X16,
101        WasmValType::Ref(rt) => reference_type(rt.heap_type, isa.pointer_type()),
102    }
103}

105/// Get the Cranelift signature for all array-call functions, that is:
106///
107/// ```ignore
108/// unsafe extern "C" fn(
109///     callee_vmctx: *mut VMOpaqueContext,
110///     caller_vmctx: *mut VMOpaqueContext,
111///     values_ptr: *mut ValRaw,
112///     values_len: usize,
113/// )
114/// ```
115///
116/// This signature uses the target's default calling convention.
117///
118/// Note that regardless of the Wasm function type, the array-call calling
119/// convention always uses that same signature.
120fn array_call_signature(isa: &dyn TargetIsa) -> ir::Signature {
121    let mut sig = blank_sig(isa, CallConv::triple_default(isa.triple()));
122    // The array-call signature has an added parameter for the `values_vec`
123    // input/output buffer in addition to the size of the buffer, in units
124    // of `ValRaw`.
125    sig.params.push(ir::AbiParam::new(isa.pointer_type()));
126    sig.params.push(ir::AbiParam::new(isa.pointer_type()));
127    sig
128}

130/// Get the internal Wasm calling convention signature for the given type.
131fn wasm_call_signature(
132    isa: &dyn TargetIsa,
133    wasm_func_ty: &WasmFuncType,
134    tunables: &Tunables,
135) -> ir::Signature {
136    // The default calling convention is `CallConv::Tail` to enable the use of
137    // tail calls in modules when needed. Note that this is used even if the
138    // tail call proposal is disabled in wasm. This is not interacted with on
139    // the host so it's purely an internal detail of wasm itself.
140    //
141    // The Winch calling convention is used instead when generating trampolines
142    // which call Winch-generated functions. The winch calling convention is
143    // only implemented for x64 and aarch64, so assert that here and panic on
144    // other architectures.
145    let call_conv = if tunables.winch_callable {
146        assert!(
147            matches!(
148                isa.triple().architecture,
149                Architecture::X86_64 | Architecture::Aarch64(_)
150            ),
151            "The Winch calling convention is only implemented for x86_64 and aarch64"
152        );
153        CallConv::Winch
154    } else {
155        CallConv::Tail
156    };
157    let mut sig = blank_sig(isa, call_conv);
158    let cvt = |ty: &WasmValType| ir::AbiParam::new(value_type(isa, *ty));
159    sig.params.extend(wasm_func_ty.params().iter().map(&cvt));
160    sig.returns.extend(wasm_func_ty.returns().iter().map(&cvt));
161    sig
162}

164/// Returns the reference type to use for the provided wasm type.
165fn reference_type(wasm_ht: WasmHeapType, pointer_type: ir::Type) -> ir::Type {
166    match wasm_ht.top() {
167        WasmHeapTopType::Func => pointer_type,
168        WasmHeapTopType::Any | WasmHeapTopType::Extern => ir::types::I32,
169    }
170}

196/// Converts cranelift_codegen settings to the wasmtime_environ equivalent.
197pub fn clif_flags_to_wasmtime(
198    flags: impl IntoIterator<Item = settings::Value>,
199) -> Vec<(&'static str, FlagValue<'static>)> {
200    flags
201        .into_iter()
202        .map(|val| (val.name, to_flag_value(&val)))
203        .collect()
204}
205
206fn to_flag_value(v: &settings::Value) -> FlagValue<'static> {
207    match v.kind() {
208        settings::SettingKind::Enum => FlagValue::Enum(v.as_enum().unwrap()),
209        settings::SettingKind::Num => FlagValue::Num(v.as_num().unwrap()),
210        settings::SettingKind::Bool => FlagValue::Bool(v.as_bool().unwrap()),
211        settings::SettingKind::Preset => unreachable!(),
212    }
213}

226/// Converts machine traps to trap information.
227pub fn mach_trap_to_trap(trap: &MachTrap) -> Option<TrapInformation> {
228    let &MachTrap { offset, code } = trap;
229    Some(TrapInformation {
230        code_offset: offset,
231        trap_code: match code {
232            ir::TrapCode::StackOverflow => Trap::StackOverflow,
233            ir::TrapCode::HeapOutOfBounds => Trap::MemoryOutOfBounds,
234            ir::TrapCode::HeapMisaligned => Trap::HeapMisaligned,
235            ir::TrapCode::TableOutOfBounds => Trap::TableOutOfBounds,
236            ir::TrapCode::IndirectCallToNull => Trap::IndirectCallToNull,
237            ir::TrapCode::BadSignature => Trap::BadSignature,
238            ir::TrapCode::IntegerOverflow => Trap::IntegerOverflow,
239            ir::TrapCode::IntegerDivisionByZero => Trap::IntegerDivisionByZero,
240            ir::TrapCode::BadConversionToInteger => Trap::BadConversionToInteger,
241            ir::TrapCode::UnreachableCodeReached => Trap::UnreachableCodeReached,
242            ir::TrapCode::Interrupt => Trap::Interrupt,
243            ir::TrapCode::User(ALWAYS_TRAP_CODE) => Trap::AlwaysTrapAdapter,
244            ir::TrapCode::User(CANNOT_ENTER_CODE) => Trap::CannotEnterComponent,
245            ir::TrapCode::NullReference => Trap::NullReference,
246            ir::TrapCode::NullI31Ref => Trap::NullI31Ref,
247
248            // These do not get converted to wasmtime traps, since they
249            // shouldn't ever be hit in theory. Instead of catching and handling
250            // these, we let the signal crash the process.
251            ir::TrapCode::User(DEBUG_ASSERT_TRAP_CODE) => return None,
252
253            // these should never be emitted by wasmtime-cranelift
254            ir::TrapCode::User(_) => unreachable!(),
255        },
256    })
257}

259/// Converts machine relocations to relocation information
260/// to perform.
261fn mach_reloc_to_reloc(
262    reloc: &FinalizedMachReloc,
263    name_map: &PrimaryMap<ir::UserExternalNameRef, ir::UserExternalName>,
264) -> Relocation {
265    let &FinalizedMachReloc {
266        offset,
267        kind,
268        ref target,
269        addend,
270    } = reloc;
271    let reloc_target = match *target {
272        FinalizedRelocTarget::ExternalName(ExternalName::User(user_func_ref)) => {
273            let name = &name_map[user_func_ref];
274            match name.namespace {
275                NS_WASM_FUNC => RelocationTarget::Wasm(FuncIndex::from_u32(name.index)),
276                NS_WASMTIME_BUILTIN => {
277                    RelocationTarget::Builtin(BuiltinFunctionIndex::from_u32(name.index))
278                }
279                _ => panic!("unknown namespace {}", name.namespace),
280            }
281        }
282        FinalizedRelocTarget::ExternalName(ExternalName::LibCall(libcall)) => {
283            let libcall = libcall_cranelift_to_wasmtime(libcall);
284            RelocationTarget::HostLibcall(libcall)
285        }
286        _ => panic!("unrecognized external name"),
287    };
288    Relocation {
289        reloc: kind,
290        reloc_target,
291        offset,
292        addend,
293    }
294}

296fn libcall_cranelift_to_wasmtime(call: ir::LibCall) -> wasmtime_environ::obj::LibCall {
297    use wasmtime_environ::obj::LibCall as LC;
298    match call {
299        ir::LibCall::FloorF32 => LC::FloorF32,
300        ir::LibCall::FloorF64 => LC::FloorF64,
301        ir::LibCall::NearestF32 => LC::NearestF32,
302        ir::LibCall::NearestF64 => LC::NearestF64,
303        ir::LibCall::CeilF32 => LC::CeilF32,
304        ir::LibCall::CeilF64 => LC::CeilF64,
305        ir::LibCall::TruncF32 => LC::TruncF32,
306        ir::LibCall::TruncF64 => LC::TruncF64,
307        ir::LibCall::FmaF32 => LC::FmaF32,
308        ir::LibCall::FmaF64 => LC::FmaF64,
309        ir::LibCall::X86Pshufb => LC::X86Pshufb,
310        _ => panic!("cranelift emitted a libcall wasmtime does not support: {call:?}"),
311    }
312}

314/// Helper structure for creating a `Signature` for all builtins.
315struct BuiltinFunctionSignatures {
316    pointer_type: ir::Type,
317
318    #[cfg(feature = "gc")]
319    reference_type: ir::Type,
320
321    call_conv: CallConv,
322}
323
324impl BuiltinFunctionSignatures {
325    fn new(isa: &dyn TargetIsa) -> Self {
326        Self {
327            pointer_type: isa.pointer_type(),
328            call_conv: CallConv::triple_default(isa.triple()),
329
330            #[cfg(feature = "gc")]
331            reference_type: ir::types::I32,
332        }
333    }
334
335    fn vmctx(&self) -> AbiParam {
336        AbiParam::special(self.pointer_type, ArgumentPurpose::VMContext)
337    }
338
339    #[cfg(feature = "gc")]
340    fn reference(&self) -> AbiParam {
341        AbiParam::new(self.reference_type)
342    }
343
344    fn pointer(&self) -> AbiParam {
345        AbiParam::new(self.pointer_type)
346    }
347
348    fn i32(&self) -> AbiParam {
349        // Some platform ABIs require i32 values to be zero- or sign-
350        // extended to the full register width.  We need to indicate
351        // this here by using the appropriate .uext or .sext attribute.
352        // The attribute can be added unconditionally; platforms whose
353        // ABI does not require such extensions will simply ignore it.
354        // Note that currently all i32 arguments or return values used
355        // by builtin functions are unsigned, so we always use .uext.
356        // If that ever changes, we will have to add a second type
357        // marker here.
358        AbiParam::new(ir::types::I32).uext()
359    }
360
361    fn i64(&self) -> AbiParam {
362        AbiParam::new(ir::types::I64)
363    }
364
365    fn signature(&self, builtin: BuiltinFunctionIndex) -> Signature {
366        let mut _cur = 0;
367        macro_rules! iter {
368            (
369                $(
370                    $( #[$attr:meta] )*
371                    $name:ident( $( $pname:ident: $param:ident ),* ) $( -> $result:ident )?;
372                )*
373            ) => {
374                $(
375                    $( #[$attr] )*
376                    if _cur == builtin.index() {
377                        return Signature {
378                            params: vec![ $( self.$param() ),* ],
379                            returns: vec![ $( self.$result() )? ],
380                            call_conv: self.call_conv,
381                        };
382                    }
383                    _cur += 1;
384                )*
385            };
386        }
387
388        wasmtime_environ::foreach_builtin_function!(iter);
389
390        unreachable!();
391    }
392}
