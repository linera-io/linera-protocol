1//! Serde mappings from GitHub's Webhook payloads to structs.
2//!
3//! Github Applications can be written as simple servers (using e.g. Axum), and then
4//! deserialize the webhook payloads it receives thanks to the models included in
5//! this module.
6//!
7//! The main entry point is to read the "event type" from the HTTP request header sent by github (at
8//! time of writing, the header is named
9//! [`X-GitHub-Event`](https://docs.github.com/en/webhooks-and-events/webhooks/webhook-events-and-payloads#delivery-headers))
10//! and then pass the value, along with the payload, to [`WebhookEvent::try_from_header_and_body`]
11//! which will validate the payload and return a valid [`WebhookEvent`](WebhookEvent).
12//!
13//! ```
14//! use octocrab::models::{AppId, webhook_events::{WebhookEvent, WebhookEventPayload, WebhookEventType}};
15//!
16//! let json = r#"{
17//!   "zen": "Design for failure.",
18//!   "hook_id": 423885699,
19//!   "hook": {
20//!     "type": "App",
21//!     "id": 423885699,
22//!     "name": "web",
23//!     "active": true,
24//!     "events": [
25//!       "issues",
26//!       "issue_comment",
27//!       "meta",
28//!       "pull_request",
29//!       "pull_request_review",
30//!       "pull_request_review_comment",
31//!       "pull_request_review_thread",
32//!       "repository"
33//!     ],
34//!     "config": {
35//!       "content_type": "json",
36//!       "insecure_ssl": "0",
37//!       "secret": "********",
38//!       "url": "https://smee.io/R"
39//!     },
40//!     "updated_at": "2023-07-13T09:30:45Z",
41//!     "created_at": "2023-07-13T09:30:45Z",
42//!     "app_id": 360617,
43//!     "deliveries_url": "https://api.github.com/app/hook/deliveries"
44//!   }
45//! }"#;
46//!
47//! // Value picked from the `X-GitHub-Event` header
48//! let event_name = "ping";
49//!
50//! let event = WebhookEvent::try_from_header_and_body(event_name, json)?;
51//! assert_eq!(event.kind, WebhookEventType::Ping);
52//! let WebhookEventPayload::Ping(ping_event) = event.specific else { panic!("checked for event.kind type before unwrapping") };
53//! assert_eq!(ping_event.hook.unwrap().app_id.unwrap(), AppId(360617));
54//! # Ok::<(), serde_json::Error>(())
55//! ```
56//!
57//! ### Stability
58//!
59//! The GitHub API for webhooks is not stable, nor specified [as far as we
60//! know](https://github.com/orgs/community/discussions/61453). Therefore, any contribution to
61//! provide more precise structures for payloads, with some actual example values to test, is very
62//! welcome.
63//!
64//! This also means you should consider octocrab's support for webhooks in beta state: it is usable,
65//! but the API needs to change according to actual usage in order to refine the different payloads
66//! received.

75/// A GitHub webhook event.
76///
77/// The structure is separated in common fields and specific fields, so you can
78/// always access the common values without needing to match the exact variant.
79#[derive(Debug, Clone, PartialEq, Serialize)]
80#[non_exhaustive]
81pub struct WebhookEvent {
82    pub sender: Option<Author>,
83    pub repository: Option<Repository>,
84    pub organization: Option<Organization>,
85    pub installation: Option<EventInstallation>,
86    #[serde(skip)]
87    pub kind: WebhookEventType,
88    #[serde(flatten)]
89    pub specific: WebhookEventPayload,
90}
91
92impl WebhookEvent {
93    /// Deserialize the body of a webhook event according to the category in the header of the request.
94    pub fn try_from_header_and_body<B>(header: &str, body: &B) -> Result<Self, serde_json::Error>
95    where
96        B: AsRef<[u8]> + ?Sized,
97    {
98        // NOTE: this is inefficient code to simply reuse the code from "derived" serde::Deserialize instead
99        // of writing specific deserialization code for the enum.
100        let kind = if header.starts_with('"') {
101            serde_json::from_str::<WebhookEventType>(header)?
102        } else {
103            serde_json::from_str::<WebhookEventType>(&format!("\"{header}\""))?
104        };
105
106        // Intermediate structure allows to separate the common fields from
107        // the event specific one.
108        #[derive(Deserialize)]
109        struct Intermediate {
110            sender: Option<Author>,
111            repository: Option<Repository>,
112            organization: Option<Organization>,
113            installation: Option<EventInstallation>,
114            #[serde(flatten)]
115            specific: serde_json::Value,
116        }
117
118        let Intermediate {
119            sender,
120            repository,
121            organization,
122            installation,
123            specific,
124        } = serde_json::from_slice::<Intermediate>(body.as_ref())?;
125
126        let specific = kind.parse_specific_payload(specific)?;
127
128        Ok(Self {
129            sender,
130            repository,
131            organization,
132            installation,
133            kind,
134            specific,
135        })
136    }
137}
