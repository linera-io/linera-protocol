74/// The payload in an event type.
75///
76/// Different event types have different payloads. Any event type not specifically part
77/// of this enum will be captured in the variant `UnknownEvent` with a value of
78/// [`serde_json::Value`](serde_json::Value).
79#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
80#[non_exhaustive]
81#[serde(untagged)]
82pub enum EventPayload {
83    PushEvent(Box<PushEventPayload>),
84    CreateEvent(Box<CreateEventPayload>),
85    DeleteEvent(Box<DeleteEventPayload>),
86    InstallationEvent(Box<InstallationEventPayload>),
87    InstallationRepositoriesEvent(Box<InstallationRepositoriesEventPayload>),
88    InstallationTargetEvent(Box<InstallationTargetEventPayload>),
89    IssuesEvent(Box<IssuesEventPayload>),
90    IssueCommentEvent(Box<IssueCommentEventPayload>),
91    CommitCommentEvent(Box<CommitCommentEventPayload>),
92    ForkEvent(Box<ForkEventPayload>),
93    GollumEvent(Box<GollumEventPayload>),
94    MemberEvent(Box<MemberEventPayload>),
95    PublicEvent(Box<PublicEventPayload>),
96    PullRequestEvent(Box<PullRequestEventPayload>),
97    PullRequestReviewEvent(Box<PullRequestReviewEventPayload>),
98    PullRequestReviewCommentEvent(Box<PullRequestReviewCommentEventPayload>),
99    ReleaseEvent(Box<ReleaseEventPayload>),
100    WatchEvent(Box<WatchEventPayload>),
101    WorkflowRunEvent(Box<WorkflowRunEventPayload>),
102    UnknownEvent(Box<serde_json::Value>),
103}

127#[cfg(test)]
128mod tests {
129    use super::*;
130
131    #[test]
132    fn should_deserialize_installation_event() {
133        // The payload has been extracted as the `payload` key from a webhook installation event.
134        let json = include_str!("../../../tests/resources/installation_event.json");
135        let event: WrappedEventPayload = serde_json::from_str(json).unwrap();
136
137        let installation = event.installation.unwrap();
138        let specific = event.specific.unwrap();
139
140        match installation {
141            EventInstallation::Full(install) => {
142                assert_eq!(install.id, 7777777.into());
143                assert_eq!(install.repository_selection.unwrap(), "all");
144            }
145            EventInstallation::Minimal(_) => {
146                panic!("expected a Full installation payload for the event.")
147            }
148        };
149
150        match specific {
151            EventPayload::InstallationEvent(install) => {
152                let repos = install.repositories;
153                assert_eq!(repos.len(), 3);
154                assert!(
155                    repos.iter().any(|repo| repo.name == "ViscoElRebound"),
156                    "ViscoElRebound should be in the list of repositories"
157                );
158                assert!(
159                    repos.iter().any(|repo| repo.name == "OSSU"),
160                    "OSSU should be in the list of repositories"
161                );
162                assert!(
163                    repos.iter().any(|repo| repo.name == "octocrab"),
164                    "octocrab should be in the list of repositories"
165                );
166            }
167            EventPayload::PushEvent(_)
168            | EventPayload::CreateEvent(_)
169            | EventPayload::DeleteEvent(_)
170            | EventPayload::InstallationRepositoriesEvent(_)
171            | EventPayload::InstallationTargetEvent(_)
172            | EventPayload::IssuesEvent(_)
173            | EventPayload::IssueCommentEvent(_)
174            | EventPayload::CommitCommentEvent(_)
175            | EventPayload::ForkEvent(_)
176            | EventPayload::GollumEvent(_)
177            | EventPayload::MemberEvent(_)
178            | EventPayload::PublicEvent(_)
179            | EventPayload::PullRequestEvent(_)
180            | EventPayload::PullRequestReviewEvent(_)
181            | EventPayload::PullRequestReviewCommentEvent(_)
182            | EventPayload::ReleaseEvent(_)
183            | EventPayload::WorkflowRunEvent(_)
184            | EventPayload::WatchEvent(_)
185            | EventPayload::UnknownEvent(_) => {
186                panic!("Expected an installation event, got {:?}", specific)
187            }
188        }
189    }
