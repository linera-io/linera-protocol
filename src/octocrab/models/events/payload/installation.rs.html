1//! This event occurs when there is activity relating to a GitHub App
2//! installation. All GitHub Apps receive this event by default. You cannot
3//! manually subscribe to this event.
4
5use serde::{Deserialize, Serialize};
6
7use super::InstallationEventRepository;
8use crate::models::Author;
9
10/// The payload in a webhook installation event type.
11#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
12#[non_exhaustive]
13pub struct InstallationEventPayload {
14    /// The action this event represents.
15    pub action: InstallationEventAction,
16    /// An enterprise on GitHub
17    pub enterprise: Option<serde_json::Value>,
18    /// An array of repositories that the installation can access
19    pub repositories: Vec<InstallationEventRepository>,
20    /// The initiator of the request, mainly for the [`created`](InstallationAction::Created) action
21    pub requester: Option<Author>,
22}
23
24/// The action on an installation this event corresponds to.
25#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
26#[serde(rename_all = "snake_case")]
27#[non_exhaustive]
28pub enum InstallationEventAction {
29    /// Someone installed a GitHub App on a user or organization account.
30    Created,
31    /// Someone uninstalled a GitHub App on a user or organization account.
32    Deleted,
33    /// Someone granted new permissions to a GitHub App.
34    NewPermissionsAccepted,
35    /// Someone blocked access by a GitHub App to their user or organization account.
36    Suspend,
37    /// A GitHub App that was blocked from accessing a user or organization account was given access the account again.
38    Unsuspend,
39}
