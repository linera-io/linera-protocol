1//! Serialization and Deserialization of timestamps in Github API
2//!
3//! GitHub API can give (from past experience) either:
4//! - a seconds timestamp relative to Epoch, or
5//! - a string containing the timestamp in [RFC 3339](https://datatracker.ietf.org/doc/html/rfc3339#section-5.6) format.
6//!
7//! This module handles transparently both formats to deserialize to [`DateTime<Utc>`](chrono::DateTime). It mostly
8//! redo things existing in [chrono::serde], because it is otherwise impossible to combine existing `serde_with` modules.
9
10use core::fmt;
11
12use chrono::{DateTime, LocalResult, TimeZone, Utc};
13use serde::{de, Deserialize};
14
15pub(crate) fn deserialize<'de, D>(deserializer: D) -> Result<DateTime<Utc>, D::Error>
16where
17    D: de::Deserializer<'de>,
18{
19    deserializer.deserialize_any(GithubTimestampVisitor)
20}
21
22/// Helper struct to tell serde the deserializer to use when working with Option<DateTime<Utc>>
23#[derive(Debug, Deserialize)]
24struct WrappedGithubTimestamp(#[serde(deserialize_with = "deserialize")] DateTime<Utc>);
25
26pub(crate) fn deserialize_opt<'de, D>(deserializer: D) -> Result<Option<DateTime<Utc>>, D::Error>
27where
28    D: de::Deserializer<'de>,
29{
30    Option::<WrappedGithubTimestamp>::deserialize(deserializer)
31        .map(|opt_wrapped| opt_wrapped.map(|wrapped| wrapped.0))
32}
33
34struct GithubTimestampVisitor;
35
36impl<'de> de::Visitor<'de> for GithubTimestampVisitor {
37    type Value = DateTime<Utc>;
38
39    fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
40        formatter
41            .write_str("a RFC3339 date and time _string_ or a unix timestamp _integer_ in seconds")
42    }
43
44    fn visit_i64<E>(self, v: i64) -> Result<Self::Value, E>
45    where
46        E: de::Error,
47    {
48        serde_from(Utc.timestamp_opt(v, 0), &v)
49    }
50
51    fn visit_u64<E>(self, v: u64) -> Result<Self::Value, E>
52    where
53        E: de::Error,
54    {
55        serde_from(Utc.timestamp_opt(v as i64, 0), &v)
56    }
57
58    fn visit_str<E>(self, v: &str) -> Result<Self::Value, E>
59    where
60        E: de::Error,
61    {
62        v.parse().map_err(E::custom)
63    }
64}
65
66// Convert from chrono local result to something usable by serde
67//
68// Ref: https://github.com/chronotope/chrono/blob/bc410bb054932518822eb393147aad939862c7a5/src/naive/datetime/serde.rs#L1051
69pub(crate) fn serde_from<T, E, V>(me: LocalResult<T>, ts: &V) -> Result<T, E>
70where
71    E: de::Error,
72    V: fmt::Display,
73    T: fmt::Display,
74{
75    match me {
76        LocalResult::None => Err(E::custom(format!("value is not a legal timestamp: {ts}"))),
77        LocalResult::Ambiguous(min, max) => Err(E::custom(format!(
78            "value is an ambiguous timestamp: {ts}. It could be {min} or {max}"
79        ))),
80        LocalResult::Single(val) => Ok(val),
81    }
82}
