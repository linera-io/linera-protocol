32macro_rules! event_type {
33    ( $( ($name:ident, $payload:ident)),+ $(,)? ) => {
34        /// The type of an event.
35        #[derive(Debug, Clone, PartialEq, Deserialize)]
36        #[non_exhaustive]
37        pub enum EventType {
38            $($name),+,
39            UnknownEvent(String),
40        }
41
42        impl Serialize for EventType {
43            fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
44            where
45                S: serde::Serializer,
46            {
47                match self {
48                    $(EventType::$name => serializer.serialize_str(stringify!($name))),+,
49                    EventType::UnknownEvent(typ) => serializer.serialize_str(typ),
50                }
51            }
52        }
53
54        fn deserialize_event_type(event_type: &str) -> EventType {
55            match event_type {
56                $(stringify!($name) => EventType::$name),+,
57                unknown => EventType::UnknownEvent(unknown.to_owned()),
58            }
59        }
60
61        fn deserialize_payload(
62            event_type: &EventType,
63            data: serde_json::Value,
64        ) -> Result<Option<EventPayload>, serde_json::Error> {
65            let maybe_payload = match event_type {
66                $(EventType::$name=> {
67                    serde_json::from_value::<Box<$payload>>(data).map(EventPayload::$name)?
68                }),+,
69                _ => EventPayload::UnknownEvent(Box::new(data)),
70            };
71            Ok(Some(maybe_payload))
72        }
73    };
74}

127impl<'de> Deserialize<'de> for Event {
128    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
129    where
130        D: serde::Deserializer<'de>,
131    {
132        #[derive(Deserialize)]
133        struct Intermediate {
134            id: String,
135            #[serde(rename = "type")]
136            typ: String,
137            actor: Actor,
138            repo: Repository,
139            public: bool,
140            created_at: DateTime<Utc>,
141            org: Option<Org>,
142            payload: Option<IntermediatePayload>,
143        }
144        #[derive(Deserialize)]
145        struct IntermediatePayload {
146            installation: Option<EventInstallation>,
147            organization: Option<crate::models::orgs::Organization>,
148            repository: Option<crate::models::Repository>,
149            sender: Option<crate::models::Author>,
150            #[serde(flatten)]
151            specific: Option<serde_json::Value>,
152        }
153        let intermediate = Intermediate::deserialize(deserializer)?;
154        let event_type = deserialize_event_type(intermediate.typ.as_ref());
155        let payload = intermediate.payload.map_or(Ok(None), |data| {
156            let specific = deserialize_payload(
157                &event_type,
158                data.specific.unwrap_or(serde_json::Value::Null),
159            )
160            .map_err(|e| Error::custom(e.to_string()))?;
161            Ok(Some(WrappedEventPayload {
162                installation: data.installation,
163                organization: data.organization,
164                repository: data.repository,
165                sender: data.sender,
166                specific,
167            }))
168        })?;
169        let event = Event {
170            id: intermediate.id,
171            r#type: event_type,
172            actor: intermediate.actor,
173            repo: intermediate.repo,
174            public: intermediate.public,
175            created_at: intermediate.created_at,
176            org: intermediate.org,
177            payload,
178        };
179        Ok(event)
180    }
181}

402    fn serialize_and_deserialize(event_type: &str, json: &str) {
403        let event: Event = serde_json::from_str(json).unwrap();
404        let serialized = serde_json::to_string(&event).unwrap();
405        // do it again so we can compare Events, otherwise we are comparing strings which may have
406        // different whitespace characteristics.
407        let deserialized = serde_json::from_str::<Event>(&serialized);
408        assert!(
409            deserialized.is_ok(),
410            "expected deserialized result for {} to be ok, got error instead {:?}",
411            event_type,
412            deserialized
413        );
414        let deserialized = deserialized.unwrap();
415        assert_eq!(
416            deserialized, event,
417            "unexpected event deserialized for {event_type}"
418        );
419    }
