1use std::sync::Arc;
2
3use http::{header::AUTHORIZATION, request::Request, HeaderValue, Uri};
4use tower::{Layer, Service};
5
6#[derive(Clone)]
7/// Layer that adds the authentication header to github-bound requests
8pub struct AuthHeaderLayer {
9    pub(crate) auth_header: Arc<Option<HeaderValue>>,
10    base_uri: Uri,
11    upload_uri: Uri,
12}
13
14impl AuthHeaderLayer {
15    pub fn new(auth_header: Option<HeaderValue>, base_uri: Uri, upload_uri: Uri) -> Self {
16        AuthHeaderLayer {
17            auth_header: Arc::new(auth_header),
18            base_uri,
19            upload_uri,
20        }
21    }
22}
23
24impl<S> Layer<S> for AuthHeaderLayer {
25    type Service = AuthHeader<S>;
26
27    fn layer(&self, inner: S) -> Self::Service {
28        AuthHeader {
29            inner,
30            auth_header: self.auth_header.clone(),
31            base_uri: self.base_uri.clone(),
32            upload_uri: self.upload_uri.clone(),
33        }
34    }
35}
36
37#[derive(Clone)]
38/// Service that adds a static set of extra headers to each request
39pub struct AuthHeader<S> {
40    inner: S,
41    pub(crate) auth_header: Arc<Option<HeaderValue>>,
42    base_uri: Uri,
43    upload_uri: Uri,
44}
45
46impl<S, ReqBody> Service<Request<ReqBody>> for AuthHeader<S>
47where
48    S: Service<Request<ReqBody>>,
49{
50    type Error = S::Error;
51    type Future = S::Future;
52    type Response = S::Response;
53
54    fn poll_ready(
55        &mut self,
56        cx: &mut std::task::Context<'_>,
57    ) -> std::task::Poll<Result<(), Self::Error>> {
58        self.inner.poll_ready(cx)
59    }
60
61    fn call(&mut self, mut req: Request<ReqBody>) -> Self::Future {
62        // Only set the auth_header if the authority (host) is destined for
63        // GitHub. Otherwise, leave it off as we could have been redirected
64        // away from GitHub (via follow_location_to_data()), and we don't
65        // want to give our credentials to third-party services.
66        let authority = req.uri().authority();
67        let allowed_authorities = [self.base_uri.authority(), self.upload_uri.authority()];
68        if authority.is_none() || allowed_authorities.contains(&authority) {
69            if let Some(auth_header) = &*self.auth_header {
70                req.headers_mut().append(AUTHORIZATION, auth_header.clone());
71            }
72        }
73        self.inner.call(req)
74    }
75}
