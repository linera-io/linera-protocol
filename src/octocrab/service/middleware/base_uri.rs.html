1//! Set base URI of requests.
2use http::{uri, Request, Uri};
3use tower::{Layer, Service};

5/// Layer that applies [`BaseUri`] which makes all requests relative to the URI.
6///
7/// Path in the base URI is preseved.
8#[derive(Debug, Clone)]
9pub struct BaseUriLayer {
10    base_uri: http::Uri,
11}
12
13impl BaseUriLayer {
14    /// Set base URI of requests.
15    pub fn new(base_uri: http::Uri) -> Self {
16        Self { base_uri }
17    }
18}
19
20impl<S> Layer<S> for BaseUriLayer {
21    type Service = BaseUri<S>;
22
23    fn layer(&self, inner: S) -> Self::Service {
24        BaseUri {
25            base_uri: self.base_uri.clone(),
26            inner,
27        }
28    }
29}

31/// Middleware that sets base URI so that all requests are relative to it.
32#[derive(Debug, Clone)]
33pub struct BaseUri<S> {
34    base_uri: http::Uri,
35    inner: S,
36}
37
38impl<S> BaseUri<S> {
39    pub fn set_base_uri(&mut self, base_uri: http::Uri) {
40        self.base_uri = base_uri;
41    }
42}
43
44impl<S, ReqBody> Service<Request<ReqBody>> for BaseUri<S>
45where
46    S: Service<Request<ReqBody>>,
47{
48    type Error = S::Error;
49    type Future = S::Future;
50    type Response = S::Response;
51
52    fn poll_ready(
53        &mut self,
54        cx: &mut std::task::Context<'_>,
55    ) -> std::task::Poll<Result<(), Self::Error>> {
56        self.inner.poll_ready(cx)
57    }
58
59    fn call(&mut self, req: Request<ReqBody>) -> Self::Future {
60        let (mut parts, body) = req.into_parts();
61        parts.uri = overwrite_base_uri(&self.base_uri, parts.uri);
62        self.inner.call(Request::from_parts(parts, body))
63    }
64}

66// Join base URI and Path+Query, preserving any path in the base.
67fn overwrite_base_uri(base_uri: &http::Uri, current_uri: Uri) -> http::Uri {
68    let req_pandq = current_uri.path_and_query();
69    let mut builder = uri::Builder::new();
70    match current_uri.scheme() {
71        Some(scheme) => builder = builder.scheme(scheme.as_str()),
72        None => {
73            if let Some(scheme) = base_uri.scheme() {
74                builder = builder.scheme(scheme.as_str());
75            }
76        }
77    }
78    match current_uri.authority() {
79        Some(authority) => builder = builder.authority(authority.as_str()),
80        None => {
81            if let Some(authority) = base_uri.authority() {
82                builder = builder.authority(authority.as_str());
83            }
84        }
85    }
86
87    if let Some(pandq) = base_uri.path_and_query() {
88        builder = if let Some(req_pandq) = req_pandq {
89            // Remove any trailing slashes and join.
90            // `PathAndQuery` always starts with a slash.
91            let base_path = pandq.path().trim_end_matches('/');
92            if !req_pandq.as_str().starts_with(base_path) {
93                builder.path_and_query(format!("{base_path}{req_pandq}"))
94            } else {
95                builder.path_and_query(req_pandq.as_str())
96            }
97        } else {
98            builder.path_and_query(pandq.as_str())
99        };
100    } else if let Some(req_pandq) = req_pandq {
101        builder = builder.path_and_query(req_pandq.as_str());
102    }
103
104    // Joining a valid Uri and valid PathAndQuery should result in a valid Uri.
105    builder.build().expect("Valid Uri")
106}

108#[cfg(test)]
109mod tests {
110    #[test]
111    fn normal_host() {
112        let base_uri = http::Uri::from_static("https://192.168.1.65:8443");
113        let apipath = http::Uri::from_static("/api/v1/nodes?hi=yes");
114        assert_eq!(
115            super::overwrite_base_uri(&base_uri, apipath),
116            "https://192.168.1.65:8443/api/v1/nodes?hi=yes"
117        );
118    }
119
120    #[test]
121    fn rancher_host() {
122        // in rancher, kubernetes server names are not hostnames, but a host with a path:
123        let base_uri = http::Uri::from_static("https://example.com/foo/bar");
124        let api_path = http::Uri::from_static("/api/v1/nodes?hi=yes");
125        assert_eq!(
126            super::overwrite_base_uri(&base_uri, api_path),
127            "https://example.com/foo/bar/api/v1/nodes?hi=yes"
128        );
129    }
