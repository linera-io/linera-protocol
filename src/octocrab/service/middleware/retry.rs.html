1use futures_util::future;
2use http::{Request, Response};
3use hyper_util::client::legacy::Error;
4use tower::retry::Policy;
5
6use crate::body::OctoBody;
7
8#[derive(Clone)]
9pub enum RetryConfig {
10    None,
11    Simple(usize),
12}
13
14impl<B> Policy<Request<OctoBody>, Response<B>, Error> for RetryConfig {
15    type Future = futures_util::future::Ready<()>;
16
17    fn retry(
18        &mut self,
19        _req: &mut Request<OctoBody>,
20        result: &mut Result<Response<B>, Error>,
21    ) -> Option<Self::Future> {
22        match self {
23            RetryConfig::None => None,
24            RetryConfig::Simple(count) => match result {
25                Ok(response) => {
26                    if response.status().is_server_error() || response.status() == 429 {
27                        if *count > 0 {
28                            *count -= 1;
29                            Some(future::ready(()))
30                        } else {
31                            None
32                        }
33                    } else {
34                        None
35                    }
36                }
37                Err(_) => {
38                    if *count > 0 {
39                        *count -= 1;
40                        Some(future::ready(()))
41                    } else {
42                        None
43                    }
44                }
45            },
46        }
47    }
48
49    fn clone_request(&mut self, req: &Request<OctoBody>) -> Option<Request<OctoBody>> {
50        match self {
51            RetryConfig::None => None,
52            _ => {
53                // `Request` can't be cloned
54                let mut new_req = Request::builder()
55                    .uri(req.uri())
56                    .method(req.method())
57                    .version(req.version());
58                for (name, value) in req.headers() {
59                    new_req = new_req.header(name, value);
60                }
61
62                let body = req.body().clone();
63                let new_req = new_req.body(body).expect(
64                    "This should never panic, as we are cloning a components from existing request",
65                );
66                Some(new_req)
67            }
68        }
69    }
70}
