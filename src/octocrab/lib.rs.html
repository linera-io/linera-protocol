1//! # Octocrab: A modern, extensible GitHub API client.
2//! Octocrab is an third party GitHub API client, allowing you to easily build
3//! your own GitHub integrations or bots. `octocrab` comes with two primary
4//! set of APIs for communicating with GitHub, a high level strongly typed
5//! semantic API, and a lower level HTTP API for extending behaviour.
6//!
7//! ## Semantic API
8//! The semantic API provides strong typing around GitHub's API, as well as a
9//! set of [`models`] that maps to GitHub's types. Currently the following
10//! modules are available.
11//!
12//! - [`activity`] GitHub Activity
13//! - [`actions`] GitHub Actions
14//! - [`apps`] GitHub Apps
15//! - [`current`] Information about the current user.
16//! - [`gitignore`] Gitignore templates
17//! - [`Octocrab::graphql`] GraphQL.
18//! - [`issues`] Issues and related items, e.g. comments, labels, etc.
19//! - [`licenses`] License Metadata.
20//! - [`markdown`] Rendering Markdown with GitHub
21//! - [`orgs`] GitHub Organisations
22//! - [`projects`] GitHub Projects
23//! - [`pulls`] Pull Requests
24//! - [`repos`] Repositories
25//! - [`repos::forks`] Repositories
26//! - [`repos::releases`] Repositories
27//! - [`search`] Using GitHub's search.
28//! - [`teams`] Teams
29//! - [`users`] Users
30//!
31//! #### Getting a Pull Request
32//! ```no_run
33//! # async fn run() -> octocrab::Result<()> {
34//! // Get pull request #404 from `octocrab/repo`.
35//! let pr = octocrab::instance().pulls("octocrab", "repo").get(404).await?;
36//! # Ok(())
37//! # }
38//! ```
39//!
40//! All methods with multiple optional parameters are built as `Builder`
41//! structs, allowing you to easily specify parameters.
42//!
43//! #### Listing issues
44//! ```no_run
45//! # async fn run() -> octocrab::Result<()> {
46//! use octocrab::{models, params};
47//!
48//! let octocrab = octocrab::instance();
49//! // Returns the first page of all issues.
50//! let mut page = octocrab.issues("octocrab", "repo")
51//!     .list()
52//!     // Optional Parameters
53//!     .creator("octocrab")
54//!     .state(params::State::All)
55//!     .per_page(50)
56//!     .send()
57//!     .await?;
58//!
59//! // Go through every page of issues. Warning: There's no rate limiting so
60//! // be careful.
61//! let results = octocrab.all_pages::<models::issues::Issue>(page).await?;
62//!
63//! # Ok(())
64//! # }
65//! ```
66//!
67//! ## HTTP API
68//! The typed API currently doesn't cover all of GitHub's API at this time, and
69//! even if it did GitHub is in active development and this library will
70//! likely always be somewhat behind GitHub at some points in time. However that
71//! shouldn't mean that in order to use those features that you have to now fork
72//! or replace `octocrab` with your own solution.
73//!
74//! Instead `octocrab` exposes a suite of HTTP methods allowing you to easily
75//! extend `Octocrab`'s existing behaviour. Using these HTTP methods allows you
76//! to keep using the same authentication and configuration, while having
77//! control over the request and response. There is a method for each HTTP
78//! method `get`, `post`, `patch`, `put`, `delete`, all of which accept a
79//! relative route and a optional body.
80//!
81//! ```no_run
82//! # async fn run() -> octocrab::Result<()> {
83//! let user: octocrab::models::Author = octocrab::instance()
84//!     .get("/user", None::<&()>)
85//!     .await?;
86//! # Ok(())
87//! # }
88//! ```
89//!
90//! Each of the HTTP methods expects a body, formats the URL with the base
91//! URL, and errors if GitHub doesn't return a successful status, but this isn't
92//! always desired when working with GitHub's API, sometimes you need to check
93//! the response status or headers. As such there are companion methods `_get`,
94//! `_post`, etc. that perform no additional pre or post-processing to
95//! the request.
96//!
97//! ```no_run
98//! # use http::Uri;
99//! # async fn run() -> octocrab::Result<()> {
100//! let octocrab = octocrab::instance();
101//! let response = octocrab
102//!     ._get("https://api.github.com/organizations")
103//!     .await?;
104//!
105//! // You can also use `Uri::builder().authority("<my custom base>").path_and_query("<my custom path>")` if you want to customize the base uri and path.
106//! let response =  octocrab
107//!     ._get(Uri::builder().path_and_query("/organizations").build().expect("valid uri"))
108//!     .await?;
109//! # Ok(())
110//! # }
111//! ```
112//!
113//! You can use the those HTTP methods to easily create your own extensions to
114//! `Octocrab`'s typed API. (Requires `async_trait`).
115//! ```
116//! use octocrab::{Octocrab, Page, Result, models};
117//!
118//! #[async_trait::async_trait]
119//! trait OrganisationExt {
120//!   async fn list_every_organisation(&self) -> Result<Page<models::orgs::Organization>>;
121//! }
122//!
123//! #[async_trait::async_trait]
124//! impl OrganisationExt for Octocrab {
125//!   async fn list_every_organisation(&self) -> Result<Page<models::orgs::Organization>> {
126//!     self.get("organizations", None::<&()>).await
127//!   }
128//! }
129//! ```
130//!
131//! You can also easily access new properties that aren't available in the
132//! current models using `serde`.
133//!
134//! ## Static API
135//! `octocrab` also provides a statically reference count version of its API,
136//! allowing you to easily plug it into existing systems without worrying
137//! about having to integrate and pass around the client.
138//!
139//! ```
140//! // Initialises the static instance with your configuration and returns an
141//! // instance of the client.
142//! # use octocrab::Octocrab;
143//! tokio_test::block_on(async {
144//! octocrab::initialise(Octocrab::default());
145//! // Gets a instance of `Octocrab` from the static API. If you call this
146//! // without first calling `octocrab::initialise` a default client will be
147//! // initialised and returned instead.
148//! octocrab::instance();
149//! # })
150//! ```
151//!
152//! ## GitHub webhook application support
153//!
154//! `octocrab` provides [deserializable datatypes](crate::models::webhook_events)
155//! for the payloads received by a GitHub application [responding to
156//! webhooks](https://docs.github.com/en/apps/creating-github-apps/writing-code-for-a-github-app/building-a-github-app-that-responds-to-webhook-events).
157//! This allows you to write a typesafe application using Rust with
158//! pattern-matching/enum-dispatch to respond to events.
159//!
160//! **Note**: Webhook support in `octocrab` is still beta, not all known webhook events are
161//! strongly typed.
162//!
163//! ```no_run
164//! # use http::request::Request;
165//! # use tracing::{warn, info};
166//! # use octocrab::models::webhook_events::*;
167//! # let request_from_github = Request::post("https://my-webhook-url.com").body(vec![0_u8]).unwrap();
168//! // request_from_github is the HTTP request your webhook handler received
169//! let (parts, body) = request_from_github.into_parts();
170//! let header = parts.headers.get("X-GitHub-Event").unwrap().to_str().unwrap();
171//!
172//! let event = WebhookEvent::try_from_header_and_body(header, &body).unwrap();
173//! // Now you can match on event type and call any specific handling logic
174//! match event.kind {
175//!     WebhookEventType::Ping => info!("Received a ping"),
176//!     WebhookEventType::PullRequest => info!("Received a pull request event"),
177//!     // ...
178//!     _ => warn!("Ignored event"),
179//! };
180//! ```

278/// Formats a GitHub preview from it's name into the full value for the

285}

287/// Formats a media type from it's name into the full value for the
288/// `Accept` header.
289/// ```
290/// assert_eq!(octocrab::format_media_type("html"), "application/vnd.github.v3.html+json");
291/// assert_eq!(octocrab::format_media_type("json"), "application/vnd.github.v3.json");
292/// assert_eq!(octocrab::format_media_type("patch"), "application/vnd.github.v3.patch");
293/// ```
294pub fn format_media_type(media_type: impl AsRef<str>) -> String {
295    let media_type = media_type.as_ref();
296    let json_suffix = match media_type {
297        "raw" | "text" | "html" | "full" => "+json",
298        _ => "",
299    };
300
301    format!("application/vnd.github.v3.{media_type}{json_suffix}")
302}

311/// Maps a GitHub error response into and `Err()` variant if the status is
312/// not a success.
313pub async fn map_github_error(
314    response: http::Response<BoxBody<Bytes, crate::Error>>,
315) -> Result<http::Response<BoxBody<Bytes, crate::Error>>> {
316    if response.status().is_success() {
317        Ok(response)
318    } else {
319        let (parts, body) = response.into_parts();
320        let GitHubErrorBody {
321            documentation_url,
322            errors,
323            message,
324        } = serde_json::from_slice(body.collect().await?.to_bytes().as_ref())
325            .context(error::SerdeSnafu)?;
326
327        Err(error::Error::GitHub {
328            source: GitHubError {
329                status_code: parts.status,
330                documentation_url,
331                errors,
332                message,
333            },
334            backtrace: Backtrace::capture(),
335        })
336    }
337}

506impl<Svc, B, LayerState> OctocrabBuilder<Svc, NoConfig, AuthState, LayerState>
507where
508    Svc: Service<Request<OctoBody>, Response = Response<B>> + Send + 'static,
509    Svc::Future: Send + 'static,
510    Svc::Error: Into<BoxError>,
511    B: http_body::Body<Data = bytes::Bytes> + Send + Sync + 'static,
512    B::Error: Into<BoxError>,
513{
514    /// Build a [`Client`] instance with the current [`Service`] stack.
515    pub fn build(self) -> Result<Octocrab, Infallible> {
516        // Transform response body to `BoxBody<Bytes, crate::Error>` and use type erased error to avoid type parameters.
517        let service = MapResponseBodyLayer::new(|b: B| {
518            b.map_err(|e| ServiceSnafu.into_error(e.into())).boxed()
519        })
520        .layer(self.service)
521        .map_err(|e| e.into());
522
523        if let Some(executor) = self.executor {
524            return Ok(Octocrab::new_with_executor(service, self.auth, executor));
525        }
526
527        Ok(Octocrab::new(service, self.auth))
528    }
529}

670    /// Build a [`Client`] instance with the current [`Service`] stack.
671    #[cfg(feature = "default-client")]
672    #[cfg_attr(docsrs, doc(cfg(feature = "default-client")))]
673    pub fn build(self) -> Result<Octocrab> {
674        let client: hyper_util::client::legacy::Client<_, OctoBody> = {
675            #[cfg(all(not(feature = "opentls"), not(feature = "rustls")))]
676            let mut connector = hyper::client::conn::http1::HttpConnector::new();
677
678            #[cfg(all(feature = "rustls", not(feature = "opentls")))]
679            let connector = {
680                let builder = HttpsConnectorBuilder::new();
681                #[cfg(feature = "rustls-webpki-tokio")]
682                let builder = builder.with_webpki_roots();
683                #[cfg(not(feature = "rustls-webpki-tokio"))]
684                let builder = builder
685                    .with_native_roots()
686                    .map_err(Into::into)
687                    .context(error::OtherSnafu)?; // enabled the `rustls-native-certs` feature in hyper-rustls
688
689                builder
690                    .https_or_http() //  Disable .https_only() during tests until: https://github.com/LukeMathWalker/wiremock-rs/issues/58 is resolved. Alternatively we can use conditional compilation to only enable this feature in tests, but it becomes rather ugly with integration tests.
691                    .enable_http1()
692                    .build()
693            };
694
695            #[cfg(all(feature = "opentls", not(feature = "rustls")))]
696            let connector = HttpsConnector::new();
697
698            #[cfg(feature = "timeout")]
699            let connector = self.set_connect_timeout_service(connector);
700
701            hyper_util::client::legacy::Client::builder(hyper_util::rt::TokioExecutor::new())
702                .build(connector)
703        };
704
705        #[cfg(feature = "retry")]
706        let client = self.set_connector_retry_service(client);
707
708        #[cfg(feature = "tracing")]
709        let client = TraceLayer::new_for_http()
710            .make_span_with(|req: &Request<OctoBody>| {
711                tracing::debug_span!(
712                    "HTTP",
713                     http.method = %req.method(),
714                     http.url = %req.uri(),
715                     http.status_code = tracing::field::Empty,
716                     otel.name = req.extensions().get::<&'static str>().unwrap_or(&"HTTP"),
717                     otel.kind = "client",
718                     otel.status_code = tracing::field::Empty,
719                )
720            })
721            .on_request(|_req: &Request<OctoBody>, _span: &Span| {
722                tracing::debug!("requesting");
723            })
724            .on_response(
725                |res: &Response<hyper::body::Incoming>, _latency: Duration, span: &Span| {
726                    let status = res.status();
727                    span.record("http.status_code", status.as_u16());
728                    if status.is_client_error() || status.is_server_error() {
729                        span.record("otel.status_code", "ERROR");
730                    }
731                },
732            )
733            // Explicitly disable `on_body_chunk`. The default does nothing.
734            .on_body_chunk(())
735            .on_eos(|_: Option<&HeaderMap>, _duration: Duration, _span: &Span| {
736                tracing::debug!("stream closed");
737            })
738            .on_failure(
739                |ec: ServerErrorsFailureClass, _latency: Duration, span: &Span| {
740                    // Called when
741                    // - Calling the inner service errored
742                    // - Polling `Body` errored
743                    // - the response was classified as failure (5xx)
744                    // - End of stream was classified as failure
745                    span.record("otel.status_code", "ERROR");
746                    match ec {
747                        ServerErrorsFailureClass::StatusCode(status) => {
748                            span.record("http.status_code", status.as_u16());
749                            tracing::error!("failed with status {}", status)
750                        }
751                        ServerErrorsFailureClass::Error(err) => {
752                            tracing::error!("failed with error {}", err)
753                        }
754                    }
755                },
756            )
757            .layer(client);
758
759        #[cfg(feature = "follow-redirect")]
760        let client = tower_http::follow_redirect::FollowRedirectLayer::new().layer(client);
761
762        let mut hmap: Vec<(HeaderName, HeaderValue)> = vec![];
763
764        // Add the user agent header required by GitHub
765        hmap.push((USER_AGENT, HeaderValue::from_str("octocrab").unwrap()));
766
767        for preview in &self.config.previews {
768            hmap.push((
769                http::header::ACCEPT,
770                HeaderValue::from_str(crate::format_preview(preview).as_str()).unwrap(),
771            ));
772        }
773
774        let (auth_header, auth_state): (Option<HeaderValue>, _) = match self.config.auth {
775            Auth::None => (None, AuthState::None),
776            Auth::Basic { username, password } => {
777                (None, AuthState::BasicAuth { username, password })
778            }
779            Auth::PersonalToken(token) => (
780                Some(format!("Bearer {}", token.expose_secret()).parse().unwrap()),
781                AuthState::None,
782            ),
783            Auth::UserAccessToken(token) => (
784                Some(format!("Bearer {}", token.expose_secret()).parse().unwrap()),
785                AuthState::None,
786            ),
787            Auth::App(app_auth) => (None, AuthState::App(app_auth)),
788            Auth::OAuth(device) => (
789                Some(
790                    format!(
791                        "{} {}",
792                        device.token_type,
793                        &device.access_token.expose_secret()
794                    )
795                    .parse()
796                    .unwrap(),
797                ),
798                AuthState::None,
799            ),
800        };
801
802        for (key, value) in self.config.extra_headers.iter() {
803            hmap.push((
804                key.clone(),
805                HeaderValue::from_str(value.as_str())
806                    .map_err(http::Error::from)
807                    .context(HttpSnafu)?,
808            ));
809        }
810
811        let client = ExtraHeadersLayer::new(Arc::new(hmap)).layer(client);
812
813        let client = MapResponseBodyLayer::new(|body| {
814            BodyExt::map_err(body, |e| HyperSnafu.into_error(e)).boxed()
815        })
816        .layer(client);
817
818        let base_uri = self
819            .config
820            .base_uri
821            .clone()
822            .unwrap_or_else(|| Uri::from_str(GITHUB_BASE_URI).unwrap());
823
824        let upload_uri = self
825            .config
826            .upload_uri
827            .clone()
828            .unwrap_or_else(|| Uri::from_str(GITHUB_BASE_UPLOAD_URI).unwrap());
829
830        let client = BaseUriLayer::new(base_uri.clone()).layer(client);
831
832        let client = AuthHeaderLayer::new(auth_header, base_uri, upload_uri).layer(client);
833
834        if let Some(executor) = self.executor {
835            return Ok(Octocrab::new_with_executor(client, auth_state, executor));
836        }
837
838        Ok(Octocrab::new(client, auth_state))
839    }
840}

885struct CachedTokenInner {
886    expiration: Option<DateTime<Utc>>,
887    secret: SecretString,
888}
889
890impl CachedTokenInner {
891    fn new(secret: SecretString, expiration: Option<DateTime<Utc>>) -> Self {
892        Self { secret, expiration }
893    }
894
895    fn expose_secret(&self) -> &str {
896        self.secret.expose_secret()
897    }
898}

900/// A cached API access token (which may be None)
901pub struct CachedToken(RwLock<Option<CachedTokenInner>>);
902
903impl CachedToken {
904    fn clear(&self) {
905        *self.0.write().unwrap() = None;
906    }
907
908    /// Returns a valid token if it exists and is not expired or if there is no expiration date.
909    fn valid_token_with_buffer(&self, buffer: chrono::Duration) -> Option<SecretString> {
910        let inner = self.0.read().unwrap();
911
912        if let Some(token) = inner.as_ref() {
913            if let Some(exp) = token.expiration {
914                if exp - Utc::now() > buffer {
915                    return Some(token.secret.clone());
916                }
917            } else {
918                return Some(token.secret.clone());
919            }
920        }
921
922        None
923    }
924
925    fn valid_token(&self) -> Option<SecretString> {
926        self.valid_token_with_buffer(chrono::Duration::seconds(30))
927    }
928
929    fn set<S: Into<SecretString>>(&self, token: S, expiration: Option<DateTime<Utc>>) {
930        *self.0.write().unwrap() = Some(CachedTokenInner::new(token.into(), expiration));
931    }
932}

934impl fmt::Debug for CachedToken {
935    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
936        self.0.read().unwrap().fmt(f)
937    }
938}
939
940impl fmt::Display for CachedToken {
941    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
942        let option = self.0.read().unwrap();
943        option
944            .as_ref()
945            .map(|s| s.expose_secret().fmt(f))
946            .unwrap_or_else(|| write!(f, "<none>"))
947    }
948}

952        CachedToken(RwLock::new(self.0.read().unwrap().clone()))
953    }
954}

958        CachedToken(RwLock::new(None))

962/// State used for authenticate to Github

966    /// an Authorization HTTP header to be set to provide authentication.

985    },

1028    /// Creates a new `Octocrab`.
1029    fn new<S>(service: S, auth_state: AuthState) -> Self
1030    where
1031        S: Service<Request<OctoBody>, Response = Response<BoxBody<Bytes, crate::Error>>>
1032            + Send
1033            + 'static,
1034        S::Future: Send + 'static,
1035        S::Error: Into<BoxError>,
1036    {
1037        let service = Buffer::new(BoxService::new(service.map_err(Into::into)), 1024);
1038
1039        Self {
1040            client: service,
1041            auth_state,
1042        }
1043    }

1045    /// Creates a new `Octocrab` with a custom executor
1046    fn new_with_executor<S>(
1047        service: S,
1048        auth_state: AuthState,
1049        executor: Box<dyn Fn(Pin<Box<dyn Future<Output = ()>>>) -> ()>,
1050    ) -> Self
1051    where
1052        S: Service<Request<OctoBody>, Response = Response<BoxBody<Bytes, crate::Error>>>
1053            + Send
1054            + 'static,
1055        S::Future: Send + 'static,
1056        S::Error: Into<BoxError>,
1057    {
1058        // Use Buffer pair to return the background worker
1059        let (service, worker) = Buffer::pair(BoxService::new(service.map_err(Into::into)), 1024);
1060
1061        // Execute the background worker with the custom executor
1062        executor(Box::pin(worker));
1063
1064        Self {
1065            client: service,
1066            auth_state,
1067        }
1068    }

1070    /// Returns a new `Octocrab` based on the current builder but
1071    /// authorizing via a specific installation ID.
1072    /// Typically you will first construct an `Octocrab` using
1073    /// `OctocrabBuilder::app` to authenticate as your Github App,
1074    /// then obtain an installation ID, and then pass that here to
1075    /// obtain a new `Octocrab` with which you can make API calls
1076    /// with the permissions of that installation.
1077    pub fn installation(&self, id: InstallationId) -> Result<Octocrab> {
1078        let app_auth = if let AuthState::App(ref app_auth) = self.auth_state {
1079            app_auth.clone()
1080        } else {
1081            return Err(Error::Installation {
1082                backtrace: Backtrace::capture(),
1083            });
1084        };
1085        Ok(Octocrab {
1086            client: self.client.clone(),
1087            auth_state: AuthState::Installation {
1088                app: app_auth,
1089                installation: id,
1090                token: CachedToken::default(),
1091            },
1092        })
1093    }
1094
1095    /// Similar to `installation`, but also eagerly caches the installation
1096    /// token and returns the token. The returned token can be used to make
1097    /// https git requests to e.g. clone repositories that the installation
1098    /// has access to.
1099    ///
1100    /// See also https://docs.github.com/en/developers/apps/building-github-apps/authenticating-with-github-apps#http-based-git-access-by-an-installation
1101    pub async fn installation_and_token(
1102        &self,
1103        id: InstallationId,
1104    ) -> Result<(Octocrab, SecretString)> {
1105        let crab = self.installation(id)?;
1106        let token = crab.request_installation_auth_token().await?;
1107        Ok((crab, token))
1108    }
1109}

1314/// # HTTP Methods
1315/// A collection of different of HTTP methods to use with Octocrab's
1316/// configuration (Authenication, etc.). All of the HTTP methods (`get`, `post`,
1317/// etc.) perform some amount of pre-processing such as making relative urls
1318/// absolute, and post processing such as mapping any potential GitHub errors
1319/// into `Err()` variants, and deserializing the response body.
1320///
1321/// This isn't always ideal when working with GitHub's API and as such there are
1322/// additional methods available prefixed with `_` (e.g.  `_get`, `_post`,
1323/// etc.) that perform no pre or post processing and directly return the
1324/// `http::Response` struct.

1326    /// Send a `POST` request to `route` with an optional body, returning the body
1327    /// of the response.
1328    pub async fn post<P: Serialize + ?Sized, R: FromResponse>(
1329        &self,
1330        route: impl AsRef<str>,
1331        body: Option<&P>,
1332    ) -> Result<R> {

1334            ._post(self.parameterized_uri(route, None::<&()>)?, body)
1335            .await?;

1337    }

1373    /// Convenience method to accept any &str, and attempt to convert it to a Uri.
1374    /// the method also attempts to serialize any parameters into a query string, and append it to the uri.
1375    fn parameterized_uri<A, P>(&self, uri: A, parameters: Option<&P>) -> Result<Uri>
1376    where
1377        A: AsRef<str>,
1378        P: Serialize + ?Sized,
1379    {
1380        let mut uri = uri.as_ref().to_string();
1381        if let Some(parameters) = parameters {
1382            if uri.contains('?') {
1383                uri = format!("{uri}&");
1384            } else {
1385                uri = format!("{uri}?");
1386            }
1387            uri = format!(
1388                "{}{}",
1389                uri,
1390                serde_urlencoded::to_string(parameters)
1391                    .context(SerdeUrlEncodedSnafu)?
1392                    .as_str()
1393            );
1394        }
1395        let uri = Uri::from_str(uri.as_str()).context(UriSnafu);
1396        uri
1397    }

1504    pub fn build_request<B: Serialize + ?Sized>(
1505        &self,
1506        mut builder: Builder,
1507        body: Option<&B>,
1508    ) -> Result<http::Request<OctoBody>> {
1509        // Since Octocrab doesn't require streamable bodies(aka, file upload) because it is serde::Serialize),
1510        // we can just use String body, since it is both http_body::Body(required by Hyper::Client), and Clone(required by BoxService).
1511
1512        // In case octocrab needs to support cases where body is strictly streamable, it should use something like reqwest::Body,
1513        // since it differentiates between retryable bodies, and streams(aka, it implements try_clone(), which is needed for middlewares like retry).
1514
1515        if let Some(body) = body {
1516            builder = builder.header(http::header::CONTENT_TYPE, "application/json");
1517            let serialized = serde_json::to_string(body).context(SerdeSnafu)?;
1518            let body: OctoBody = serialized.into();
1519            let request = builder.body(body).context(HttpSnafu)?;
1520            Ok(request)
1521        } else {
1522            Ok(builder
1523                .header(http::header::CONTENT_LENGTH, "0")
1524                .body(OctoBody::empty())
1525                .context(HttpSnafu)?)
1526        }
1527    }

1558    /// Requests a fresh installation auth token and caches it. Returns the token.
1559    async fn request_installation_auth_token(&self) -> Result<SecretString> {
1560        let (app, installation, token) = if let AuthState::Installation {
1561            ref app,
1562            installation,
1563            ref token,
1564        } = self.auth_state
1565        {
1566            (app, installation, token)
1567        } else {
1568            return Err(Error::Installation {
1569                backtrace: Backtrace::capture(),
1570            });
1571        };
1572        let mut request = Builder::new();
1573        let mut sensitive_value =
1574            HeaderValue::from_str(format!("Bearer {}", app.generate_bearer_token()?).as_str())
1575                .map_err(http::Error::from)
1576                .context(HttpSnafu)?;
1577
1578        let uri = http::Uri::builder()
1579            .path_and_query(format!("/app/installations/{installation}/access_tokens"))
1580            .build()
1581            .context(HttpSnafu)?;
1582
1583        sensitive_value.set_sensitive(true);
1584        request = request
1585            .header(http::header::AUTHORIZATION, sensitive_value)
1586            .method(http::Method::POST)
1587            .uri(uri);
1588        let response = self
1589            .send(request.body("{}".into()).context(HttpSnafu)?)
1590            .await?;
1591        let _status = response.status();
1592
1593        let token_object =
1594            InstallationToken::from_response(crate::map_github_error(response).await?).await?;
1595
1596        let expiration = token_object
1597            .expires_at
1598            .map(|time| {
1599                DateTime::<Utc>::from_str(&time).map_err(|e| error::Error::Other {
1600                    source: Box::new(e),
1601                    backtrace: snafu::Backtrace::capture(),
1602                })
1603            })
1604            .transpose()?;
1605
1606        #[cfg(feature = "tracing")]
1607        tracing::debug!("Token expires at: {:?}", expiration);
1608
1609        token.set(token_object.token.clone(), expiration);
1610
1611        Ok(SecretString::from(token_object.token))
1612    }

1640    /// Execute the given `request` using octocrab's Client.
1641    pub async fn execute(
1642        &self,
1643        request: http::Request<impl Into<OctoBody>>,
1644    ) -> Result<http::Response<BoxBody<Bytes, crate::Error>>> {
1645        let (mut parts, body) = request.into_parts();
1646        let body: OctoBody = body.into();
1647        // Saved request that we can retry later if necessary
1648        let auth_header: Option<HeaderValue> = match self.auth_state {
1649            AuthState::None => None,
1650            AuthState::App(ref app) => Some(
1651                HeaderValue::from_str(format!("Bearer {}", app.generate_bearer_token()?).as_str())
1652                    .map_err(http::Error::from)
1653                    .context(HttpSnafu)?,
1654            ),
1655            AuthState::BasicAuth {
1656                ref username,
1657                ref password,
1658            } => {
1659                // Equivalent implementation of: https://github.com/seanmonstar/reqwest/blob/df2b3baadc1eade54b1c22415792b778442673a4/src/util.rs#L3-L23
1660                use base64::prelude::BASE64_STANDARD;
1661                use base64::write::EncoderWriter;
1662
1663                let mut buf = b"Basic ".to_vec();
1664                {
1665                    let mut encoder = EncoderWriter::new(&mut buf, &BASE64_STANDARD);
1666                    write!(encoder, "{}:{}", username, password)
1667                        .expect("writing to a Vec never fails");
1668                }
1669                Some(HeaderValue::from_bytes(&buf).expect("base64 is always valid HeaderValue"))
1670            }
1671            AuthState::Installation { ref token, .. } => {
1672                let token = if let Some(token) = token.valid_token() {
1673                    token
1674                } else {
1675                    self.request_installation_auth_token().await?
1676                };
1677
1678                Some(
1679                    HeaderValue::from_str(format!("Bearer {}", token.expose_secret()).as_str())
1680                        .map_err(http::Error::from)
1681                        .context(HttpSnafu)?,
1682                )
1683            }
1684        };
1685
1686        if let Some(mut auth_header) = auth_header {
1687            // Only set the auth_header if the authority (host) is empty (destined for
1688            // GitHub). Otherwise, leave it off as we could have been redirected
1689            // away from GitHub (via follow_location_to_data()), and we don't
1690            // want to give our credentials to third-party services.
1691            if parts.uri.authority().is_none() {
1692                auth_header.set_sensitive(true);
1693                parts
1694                    .headers
1695                    .insert(http::header::AUTHORIZATION, auth_header);
1696            }
1697        }
1698
1699        let request = http::Request::from_parts(parts, body);
1700
1701        let response = self.send(request).await?;
1702
1703        let status = response.status();
1704        if StatusCode::UNAUTHORIZED == status {
1705            if let AuthState::Installation { ref token, .. } = self.auth_state {
1706                token.clear();
1707            }
1708        }
1709        Ok(response)
1710    }

1713        &self,
1714        response: http::Response<BoxBody<Bytes, Error>>,
1715    ) -> crate::Result<http::Response<BoxBody<Bytes, crate::Error>>> {
1716        if let Some(redirect) = response.headers().get(http::header::LOCATION) {
1717            let location = redirect.to_str().expect("Location URL not valid str");
1718
1719            self._get(location).await
1720        } else {
1721            Ok(response)
1722        }
1723    }
