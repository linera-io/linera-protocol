18/// A client to GitHub's organization API.
19///
20/// Created with [`Octocrab::orgs`].

31    /// Add or update organization membership
32    ///
33    /// **Note**
34    /// - Only authenticated organization owners can add a member to the
35    ///   organization or update the member's role.
36    /// - If the authenticated user is adding a member to the organization, the
37    ///   invited user will receive an email inviting them to the organization.
38    ///   The user's membership status will be pending until they accept
39    ///   the invitation.
40    /// - Authenticated users can update a user's membership by passing the role
41    ///   parameter. If the authenticated user changes a member's role to admin,
42    ///   the affected user will receive an email notifying them that they've
43    ///   been made an organization owner. If the authenticated user changes an
44    ///   owner's role to member, no email will be sent.
45    /// ```no_run
46    /// # async fn run() -> octocrab::Result<()> {
47    /// # let octocrab = octocrab::Octocrab::default();
48    /// let invitation = octocrab.orgs("owner").add_or_update_membership("ferris", None).await?;
49    /// # Ok(())
50    /// # }
51    /// ```
52    pub async fn add_or_update_membership(
53        &self,
54        username: impl AsRef<str>,
55        role: Option<crate::params::orgs::Role>,
56    ) -> crate::Result<crate::models::orgs::MembershipInvitation> {
57        let route = format!(
58            "/orgs/{org}/memberships/{username}",
59            org = self.owner,
60            username = username.as_ref(),
61        );
62
63        let body = role.map(|role| serde_json::json!({ "role": role }));
64
65        self.crab.post(route, body.as_ref()).await
66    }

68    /// Check if a user is, publicly or privately, a member of the organization.
69    ///
70    /// ```no_run
71    /// # async fn run() -> octocrab::Result<()> {
72    /// # let octocrab = octocrab::Octocrab::default();
73    /// assert!(octocrab.orgs("owner").check_membership("ferris").await?);
74    /// # Ok(())
75    /// # }
76    /// ```
77    pub async fn check_membership(&self, username: impl AsRef<str>) -> crate::Result<bool> {
78        let route = format!(
79            "/orgs/{org}/members/{username}",
80            org = self.owner,
81            username = username.as_ref(),
82        );
83
84        let uri = Uri::builder()
85            .path_and_query(route)
86            .build()
87            .context(HttpSnafu)?;
88
89        let response = self.crab._get(uri).await?;
90
91        match response.status() {
92            StatusCode::NO_CONTENT => Ok(true),
93            StatusCode::NOT_FOUND => Ok(false),
94            _ => Err(crate::map_github_error(response).await.unwrap_err()),
95        }
96    }

98    /// Get an organization
99    ///
100    /// To see many of the organization response values, you need to be an
101    /// authenticated organization owner with the `admin:org` scope. When the
102    /// value of `two_factor_requirement_enabled` is true, the organization
103    /// requires all members, billing managers, and outside collaborators to
104    /// enable two-factor authentication.
105    /// ```no_run
106    /// # async fn run() -> octocrab::Result<()> {
107    /// # let octocrab = octocrab::Octocrab::default();
108    /// let org = octocrab.orgs("owner").get().await?;
109    /// # Ok(())
110    /// # }
111    /// ```
112    pub async fn get(&self) -> crate::Result<crate::models::orgs::Organization> {
113        let route = format!("/orgs/{org}", org = self.owner);
114
115        self.crab.get(route, None::<&()>).await
116    }

175    /// Creates a new webhook for the specified organization.
176    ///
177    /// # Notes
178    /// Only authorized users or apps can modify organization webhooks.
179    ///
180    /// # Examples
181    /// ```no_run
182    /// # async fn run() -> octocrab::Result<()> {
183    /// # let octocrab = octocrab::Octocrab::default();
184    /// use octocrab::models::hooks::{Hook, Config as HookConfig, ContentType as HookContentType};
185    ///
186    /// let config = HookConfig {
187    ///   url: "https://example.com".to_string(),
188    ///   content_type: Some(HookContentType::Json),
189    ///   insecure_ssl: None,
190    ///   secret: None
191    /// };
192    ///
193    /// let hook = Hook {
194    ///   name: "web".to_string(),
195    ///   config,
196    ///   ..Hook::default()
197    /// };
198    ///
199    /// let hook = octocrab.orgs("owner").create_hook(hook).await?;
200    /// # Ok(())
201    /// # }
202    /// ```
203    pub async fn create_hook(
204        &self,
205        hook: crate::models::hooks::Hook,
206    ) -> crate::Result<crate::models::hooks::Hook> {
207        let route = format!("/orgs/{org}/hooks", org = self.owner);
208        let res = self.crab.post(route, Some(&hook)).await?;
209
210        Ok(res)
211    }
