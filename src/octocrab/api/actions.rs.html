30impl<'octo> ListWorkflowRunArtifacts<'octo> {
31    pub(crate) fn new(crab: &'octo Octocrab, owner: String, repo: String, run_id: RunId) -> Self {
32        Self {
33            crab,
34            owner,
35            repo,
36            run_id,
37            per_page: None,
38            page: None,
39            etag: None,
40        }
41    }
42
43    /// Etag for this request.
44    pub fn etag(mut self, etag: Option<EntityTag>) -> Self {
45        self.etag = etag;
46        self
47    }
48
49    /// Results per page (max 100).
50    pub fn per_page(mut self, per_page: impl Into<u8>) -> Self {
51        self.per_page = Some(per_page.into());
52        self
53    }
54
55    /// Page number of the results to fetch.
56    pub fn page(mut self, page: impl Into<u32>) -> Self {
57        self.page = Some(page.into());
58        self
59    }
60
61    pub async fn send(self) -> crate::Result<Etagged<Page<WorkflowListArtifact>>> {
62        let path = format!(
63            "/repos/{owner}/{repo}/actions/runs/{run_id}/artifacts",
64            owner = self.owner,
65            repo = self.repo,
66            run_id = self.run_id
67        );
68        let uri = Uri::builder()
69            .path_and_query(path)
70            .build()
71            .context(HttpSnafu)?;
72        let mut headers = HeaderMap::new();
73        if let Some(etag) = self.etag {
74            EntityTag::insert_if_none_match_header(&mut headers, etag)?;
75        }
76
77        let request = self
78            .crab
79            .build_request(Builder::new().method(Method::GET).uri(uri), None::<&()>)?;
80        let response = self.crab.execute(request).await?;
81        let etag = EntityTag::extract_from_response(&response);
82        if response.status() == StatusCode::NOT_MODIFIED {
83            Ok(Etagged { etag, value: None })
84        } else {
85            <Page<WorkflowListArtifact>>::from_response(crate::map_github_error(response).await?)
86                .await
87                .map(|page| Etagged {
88                    etag,
89                    value: Some(page),
90                })
91        }
92    }
93}

95pub struct WorkflowDispatchBuilder<'octo> {
96    crab: &'octo Octocrab,
97    owner: String,
98    repo: String,
99    workflow_id: String,
100    data: WorkflowDispatch,
101}
102
103impl<'octo> WorkflowDispatchBuilder<'octo> {
104    pub(crate) fn new(
105        crab: &'octo Octocrab,
106        owner: String,
107        repo: String,
108        workflow_id: String,
109        r#ref: String,
110    ) -> Self {
111        let mut this = Self {
112            crab,
113            owner,
114            repo,
115            workflow_id,
116            data: Default::default(),
117        };
118        this.data.r#ref = r#ref;
119        this
120    }
121
122    /// Input keys and values configured in the workflow file. The maximum number of properties is 10.
123    /// Any default properties configured in the workflow file will be used when inputs are omitted.
124    ///
125    /// # Panics
126    /// If `inputs` is not `Value::Object`.
127    pub fn inputs(mut self, inputs: serde_json::Value) -> Self {
128        assert!(inputs.is_object(), "Inputs should be a JSON object");
129        self.data.inputs = inputs;
130        self
131    }
132
133    pub async fn send(self) -> crate::Result<()> {
134        let route = format!(
135            "/repos/{owner}/{repo}/actions/workflows/{workflow_id}/dispatches",
136            owner = self.owner,
137            repo = self.repo,
138            workflow_id = self.workflow_id
139        );
140
141        let uri = Uri::builder()
142            .path_and_query(route)
143            .build()
144            .context(HttpSnafu)?;
145
146        let response = self.crab._post(uri, Some(&self.data)).await?;
147        if !response.status().is_success() {
148            return Err(crate::map_github_error(response).await.unwrap_err());
149        }
150
151        Ok(())
152    }
153}
