7/// Handler for GitHub's code scanning API.
8///
9/// Created with [`Octocrab::code_scannings`].
10pub struct CodeScanningHandler<'octo> {
11    crab: &'octo Octocrab,
12    owner: String,
13    repo: Option<String>,
14}
15
16impl<'octo> CodeScanningHandler<'octo> {
17    pub(crate) fn new(crab: &'octo Octocrab, owner: String, repo: Option<String>) -> Self {
18        Self { crab, owner, repo }
19    }
20
21    /// Gets an code scanning from the repository.
22    /// ```no_run
23    /// # async fn run() -> octocrab::Result<()> {
24    /// # let octocrab = octocrab::Octocrab::default();
25    /// let code_scanning = octocrab.code_scannings("owner", "repo").get(3).await?;
26    /// # Ok(())
27    /// # }
28    /// ```
29    pub async fn get(&mut self, number: u64) -> Result<models::code_scannings::CodeScanningAlert> {
30        let route = format!(
31            "/repos/{owner}/{repo}/code-scanning/alerts/{number}",
32            owner = self.owner,
33            repo = self.repo.as_mut().expect("Repository must be specified"),
34            number = number,
35        );
36
37        self.crab.get(route, None::<&()>).await
38    }
39
40    /// List code scannings in the repository.
41    pub fn list(&self) -> list::ListCodeScanningsBuilder<'_, '_> {
42        list::ListCodeScanningsBuilder::new(self)
43    }
44
45    /// Update a code scanning alert
46    /// ```no_run
47    /// # use octocrab::params;
48    ///
49    ///  async fn run() -> octocrab::Result<()> {
50    /// # let octocrab = octocrab::Octocrab::default();
51    /// use octocrab::models;
52    ///
53    /// let issue = octocrab.code_scannings("owner", "repo")
54    ///     .update(1234u64)
55    ///     .state(params::AlertState::Dismissed)
56    ///     // Send the request
57    ///     .send()
58    ///     .await?;
59    /// # Ok(())
60    /// # }
61    /// ```
62    pub fn update(&self, number: u64) -> update::UpdateCodeScanningBuilder<'_, '_> {
63        update::UpdateCodeScanningBuilder::new(self, number)
64    }
65}
