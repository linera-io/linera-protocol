28    ///Retrieves a pull request review by its ID.
29    ///see https://docs.github.com/en/rest/pulls/reviews?apiVersion=2022-11-28#get-a-review-for-a-pull-request
30    /// ```no_run
31    /// # async fn run() -> octocrab::Result<()> {
32    /// # let octocrab = octocrab::Octocrab::default();
33    /// use octocrab::params;
34    ///
35    /// let _ = octocrab.pulls("owner", "repo")
36    /// .pull_number(42)
37    /// .reviews()
38    /// .review(42)
39    /// .get()
40    /// .await?;
41    /// # Ok(())
42    /// # }
43    /// ```
44    pub async fn get(&self) -> crate::Result<crate::models::pulls::Review> {
45        let route = format!(
46            "/repos/{owner}/{repo}/pulls/{pull_number}/reviews/{review_id}",
47            owner = self.handler.owner,
48            repo = self.handler.repo,
49            pull_number = self.pr_number,
50            review_id = self.review_id
51        );
52        self.handler.crab.get(route, Some(&self)).await
53    }

112    ///Submits a pending review for a pull request.
113    ///see https://docs.github.com/en/rest/pulls/reviews?apiVersion=2022-11-28#submit-a-review-for-a-pull-request
114    ///```no_run
115    /// # use octocrab::models::pulls::ReviewAction;
116    ///  async fn run() -> octocrab::Result<()> {
117    /// # let octocrab = octocrab::Octocrab::default();
118    /// use octocrab::params;
119    ///
120    /// let _ = octocrab.pulls("owner", "repo")
121    /// .pull_number(42)
122    /// .reviews()
123    /// .review(42)
124    /// .submit(ReviewAction::RequestChanges, "comment body")
125    /// .await?;
126    /// # Ok(())
127    /// # }
128    /// ```
129    pub async fn submit(
130        &self,
131        action: ReviewAction,
132        body: impl Into<String>,
133    ) -> crate::Result<Review> {
134        let route = format!(
135            "/repos/{owner}/{repo}/pulls/{pull_number}/reviews/{review_id}/events",
136            owner = self.handler.owner,
137            repo = self.handler.repo,
138            pull_number = self.pr_number,
139            review_id = self.review_id
140        );
141        self.handler
142            .crab
143            .post(
144                route,
145                Some(&serde_json::json!({ "body": body.into(), "event": action })),
146            )
147            .await
148    }

152    /// ```no_run

166    pub async fn dismiss(&self, message: impl Into<String>) -> crate::Result<Review> {
167        let route = format!(
168            "/repos/{owner}/{repo}/pulls/{pull_number}/reviews/{review_id}/dismissals",
169            owner = self.handler.owner,
170            repo = self.handler.repo,
171            pull_number = self.pr_number,
172            review_id = self.review_id
173        );
174        self.handler
175            .crab
176            .put(
177                route,
178                Some(&serde_json::json!({ "message": message.into(), "event": "DISMISS" })),
179            )
180            .await
181    }
