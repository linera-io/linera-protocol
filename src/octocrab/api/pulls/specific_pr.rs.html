12/// A builder pattern struct for working with a specific pull request data,
13/// e.g. reviews, commits, comments, etc.
14///
15/// created by [`PullRequestHandler::pull_number`]
16///
17/// [`PullRequestHandler::pull_number`]: ./struct.PullRequestHandler.html#method.pull_number
18#[derive(serde::Serialize)]
19pub struct SpecificPullRequestBuilder<'octo, 'b> {
20    #[serde(skip)]
21    handler: &'b PullRequestHandler<'octo>,
22    pr_number: u64,
23    #[serde(skip_serializing_if = "Option::is_none")]
24    per_page: Option<u8>,
25
26    #[serde(skip_serializing_if = "Option::is_none")]
27    page: Option<u32>,
28}
29
30impl<'octo, 'b> SpecificPullRequestBuilder<'octo, 'b> {
31    pub(crate) fn new(handler: &'b PullRequestHandler<'octo>, pr_number: u64) -> Self {
32        Self {
33            handler,
34            pr_number,
35            per_page: None,
36            page: None,
37        }
38    }
39
40    /// Results per page (max: 100, default: 30).
41    pub fn per_page(mut self, per_page: impl Into<u8>) -> Self {
42        self.per_page = Some(per_page.into());
43        self
44    }
45
46    /// Page number of the results to fetch. (default: 1)
47    pub fn page(mut self, page: impl Into<u32>) -> Self {
48        self.page = Some(page.into());
49        self
50    }
51
52    ///Lists a maximum of 250 commits for a pull request.
53    /// To receive a complete commit list for pull requests with more than 250 commits,
54    /// use the [List commits](https://docs.github.com/rest/commits/commits#list-commits) endpoint.
55    pub async fn commits(&self) -> crate::Result<Page<RepoCommit>> {
56        let route = format!(
57            "/repos/{owner}/{repo}/pulls/{pr_number}/commits",
58            owner = self.handler.owner,
59            repo = self.handler.repo,
60            pr_number = self.pr_number
61        );
62        self.handler.crab.get(route, Some(&self)).await
63    }
64
65    /// Creates a new `ReviewsBuilder`
66    /// ```no_run
67    /// # use octocrab::models::CommentId;
68    ///  async fn run() -> octocrab::Result<()> {
69    /// # let octocrab = octocrab::Octocrab::default();
70    /// use octocrab::params;
71    ///
72    /// let _ = octocrab.pulls("owner", "repo")
73    ///     .pull_number(42)
74    ///     .reviews()
75    ///     .review(42)
76    ///     .get()
77    ///     .await;
78    ///     Ok(())
79    /// }
80    /// ```
81    pub fn reviews(&self) -> ReviewsBuilder<'octo, '_> {
82        ReviewsBuilder::new(self.handler, self.pr_number)
83    }
84
85    /// Creates a new `SpecificPullRequestCommentBuilder`
86    /// ```no_run
87    /// # use octocrab::models::CommentId;
88    ///  async fn run() -> octocrab::Result<()> {
89    /// # let octocrab = octocrab::Octocrab::default();
90    /// use octocrab::params;
91    ///
92    /// let _ = octocrab.pulls("owner", "repo")
93    ///     .pull_number(42)
94    ///     .comment(CommentId(42))
95    ///     .reply("new comment")
96    ///     .await;
97    ///     Ok(())
98    ///   }
99    /// ```
100    pub fn comment(&self, comment_id: CommentId) -> SpecificPullRequestCommentBuilder {
101        SpecificPullRequestCommentBuilder::new(self.handler, self.pr_number, comment_id)
102    }
103}
