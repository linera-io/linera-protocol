9/// A builder pattern struct for listing self-hosted runners.
10///
11/// Created by [`ActionsHandler::list_org_self_hosted_runners`] or
12/// [`ActionsHandler::list_repo_self_hosted_runners`].
13///
14/// [`ActionsHandler::list_org_self_hosted_runners`]: ../struct.ActionsHandler.html#method.list_org_self_hosted_runners
15/// [`ActionsHandler::list_repo_self_hosted_runners`]: ../struct.ActionsHandler.html#method.list_repo_self_hosted_runners
16#[derive(Serialize)]
17pub struct ListSelfHostedRunnersBuilder<'octo, 'r> {
18    #[serde(skip)]
19    handler: &'r ActionsHandler<'octo>,
20    #[serde(skip)]
21    scope: RunnerScope,
22    #[serde(skip_serializing_if = "Option::is_none")]
23    name: Option<String>,
24    #[serde(skip_serializing_if = "Option::is_none")]
25    per_page: Option<u8>,
26    #[serde(skip_serializing_if = "Option::is_none")]
27    page: Option<u32>,
28}
29
30impl<'octo, 'r> ListSelfHostedRunnersBuilder<'octo, 'r> {
31    pub(crate) fn new_org(handler: &'r ActionsHandler<'octo>, org: String) -> Self {
32        Self {
33            handler,
34            scope: RunnerScope::Org(org),
35            name: None,
36            per_page: None,
37            page: None,
38        }
39    }
40
41    pub(crate) fn new_repo(
42        handler: &'r ActionsHandler<'octo>,
43        owner: String,
44        repo: String,
45    ) -> Self {
46        Self {
47            handler,
48            scope: RunnerScope::Repo { owner, repo },
49            name: None,
50            per_page: None,
51            page: None,
52        }
53    }
54
55    /// Filter the list by runners matching the given name.
56    pub fn name(mut self, name: impl Into<String>) -> Self {
57        self.name = Some(name.into());
58        self
59    }
60
61    /// Results per page (max 100).
62    pub fn per_page(mut self, per_page: impl Into<u8>) -> Self {
63        self.per_page = Some(per_page.into());
64        self
65    }
66
67    /// Page number of the results to fetch.
68    pub fn page(mut self, page: impl Into<u32>) -> Self {
69        self.page = Some(page.into());
70        self
71    }
72
73    /// Sends the actual request.
74    pub async fn send(
75        self,
76    ) -> crate::Result<crate::Page<crate::models::actions::SelfHostedRunner>> {
77        let route = match &self.scope {
78            RunnerScope::Org(org) => format!("/orgs/{org}/actions/runners"),
79            RunnerScope::Repo { owner, repo } => format!("/repos/{owner}/{repo}/actions/runners"),
80        };
81
82        self.handler.crab.get(route, Some(&self)).await
83    }
84}
