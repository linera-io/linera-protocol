3/// A client to GitHub's repository dependabot API.
4///
5/// Created with [`Octocrab::repos`].
6pub struct RepoDependabotAlertsHandler<'octo> {

54    /// Lists all Dependabot Alerts available in a repository.
55    /// You must authenticate using an access token with the `repo` or `security_events` scope to use this endpoint.
56    /// ```no_run
57    /// # async fn run() -> octocrab::Result<()> {
58    /// # let octocrab = octocrab::Octocrab::default();
59    /// let all_secrets = octocrab.repos("owner", "repo")
60    ///     .dependabot()
61    ///     .direction("asc")
62    ///     .get_alerts()
63    ///     .await?;
64    /// # Ok(())
65    /// # }
66    pub async fn get_alerts(
67        &self,
68    ) -> crate::Result<crate::Page<crate::models::repos::dependabot::DependabotAlert>> {
69        let route = format!("/{}/dependabot/alerts", self.handler.repo);
70        self.handler.crab.get(route, Some(&self.params)).await
71    }
72

133    /// Lists single Dependabot Alert for a repository.
134    /// You must authenticate using an access token with the `repo` or `security_events` scope to use this endpoint.

146        alert_number: u32,

148        let route = format!("/{}/dependabot/alerts/{}", self.handler.repo, alert_number);

152    /// Updates a dependabot alert.
153    /// You must authenticate using an access token with the `security_events ` scope to use this endpoint.
154    /// ```no_run
155    /// # async fn run() -> octocrab::Result<()> {
156    /// # let octocrab = octocrab::Octocrab::default();
157    /// use octocrab::models::repos::dependabot::UpdateDependabotAlert;
158    ///
159    /// let result = octocrab.repos("owner", "repo")
160    ///     .dependabot()
161    ///     .update_alert(
162    ///         5,
163    ///         Some(&UpdateDependabotAlert {
164    ///             state: "dismissed",
165    ///             dismissed_reason: Some("no_bandwidth"),
166    ///             dismissed_comment: Some("I don't have time to fix this right now"),
167    ///         })
168    ///     )
169    ///     .await?;
170    /// # Ok(())
171    /// # }
172    pub async fn update_alert(
173        &self,
174        alert_number: u32,
175        alert_update: Option<&crate::models::repos::dependabot::UpdateDependabotAlert<'_>>,
176    ) -> crate::Result<crate::models::repos::dependabot::DependabotAlert> {
177        let route = format!("/{}/dependabot/alerts/{}", self.handler.repo, alert_number);
178        self.handler.crab.patch(route, alert_update).await
179    }
