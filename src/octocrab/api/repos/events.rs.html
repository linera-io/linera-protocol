1//! GitHub Repository Events
2use crate::{
3    etag::{EntityTag, Etagged},
4    models::events,
5    repos::RepoHandler,
6    FromResponse, Page,
7};
8use http::request::Builder;
9use http::{header::HeaderMap, Method, StatusCode};
10
11pub struct ListRepoEventsBuilder<'octo, 'handler> {
12    handler: &'handler RepoHandler<'octo>,
13    headers: Headers,
14    params: Params,
15}
16
17struct Headers {
18    etag: Option<EntityTag>,
19}
20
21#[derive(serde::Serialize)]
22struct Params {
23    #[serde(skip_serializing_if = "Option::is_none")]
24    per_page: Option<u8>,
25    #[serde(skip_serializing_if = "Option::is_none")]
26    page: Option<u32>,
27}
28
29impl<'octo, 'handler> ListRepoEventsBuilder<'octo, 'handler> {
30    pub(crate) fn new(handler: &'handler RepoHandler<'octo>) -> Self {
31        Self {
32            handler,
33            headers: Headers { etag: None },
34            params: Params {
35                per_page: None,
36                page: None,
37            },
38        }
39    }
40
41    /// Etag for this request.
42    pub fn etag(mut self, etag: Option<EntityTag>) -> Self {
43        self.headers.etag = etag;
44        self
45    }
46
47    /// Results per page (max 100).
48    pub fn per_page(mut self, per_page: impl Into<u8>) -> Self {
49        self.params.per_page = Some(per_page.into());
50        self
51    }
52
53    /// Page number of the results to fetch.
54    pub fn page(mut self, page: impl Into<u32>) -> Self {
55        self.params.page = Some(page.into());
56        self
57    }
58
59    /// Sends the actual request.
60    pub async fn send(self) -> crate::Result<Etagged<Page<events::Event>>> {
61        let route = format!("/{}/events", self.handler.repo);
62
63        let uri = self
64            .handler
65            .crab
66            .parameterized_uri(route, Some(&self.params))?;
67
68        let mut headers = HeaderMap::new();
69        if let Some(etag) = self.headers.etag {
70            EntityTag::insert_if_none_match_header(&mut headers, etag)?;
71        }
72
73        let mut request = Builder::new().uri(uri).method(Method::GET);
74        for (key, value) in headers.iter() {
75            request = request.header(key, value);
76        }
77
78        let request = self.handler.crab.build_request(request, None::<&()>)?;
79        let response = self.handler.crab.execute(request).await?;
80        let etag = EntityTag::extract_from_response(&response);
81        if response.status() == StatusCode::NOT_MODIFIED {
82            Ok(Etagged { etag, value: None })
83        } else {
84            <Page<events::Event>>::from_response(crate::map_github_error(response).await?)
85                .await
86                .map(|page| Etagged {
87                    etag,
88                    value: Some(page),
89                })
90        }
91    }
92}
