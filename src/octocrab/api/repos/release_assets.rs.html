3/// Handler for GitHub's releases API.
4///
5/// Created with [`RepoHandler::release_assets`].
6pub struct ReleaseAssetsHandler<'octo, 'r> {
7    handler: &'r RepoHandler<'octo>,
8}
9
10impl<'octo, 'r> ReleaseAssetsHandler<'octo, 'r> {
11    pub(crate) fn new(parent: &'r RepoHandler<'octo>) -> Self {
12        Self { handler: parent }
13    }
14    /// Gets the release asset using its id.
15    /// ```no_run
16    /// # async fn run() -> octocrab::Result<()> {
17    /// let release = octocrab::instance()
18    ///     .repos("owner", "repo")
19    ///     .release_assets()
20    ///     .get(3)
21    ///     .await?;
22    /// # Ok(())
23    /// # }
24    /// ```
25    pub async fn get(&self, id: u64) -> Result<models::repos::Asset> {
26        let route = format!("/{}/releases/assets/{id}", self.handler.repo, id = id,);
27
28        self.handler.crab.get(route, None::<&()>).await
29    }
30    /// Creates a new [`UpdateReleaseAssetBuilder`] with `asset_id`.
31    /// ```no_run
32    /// # async fn run() -> octocrab::Result<()> {
33    /// # let octocrab = octocrab::Octocrab::default();
34    /// let release = octocrab.repos("owner", "repo")
35    ///     .release_assets()
36    ///     .update(1)
37    ///     // Optional Parameters
38    ///     .name("asset_1.tar.gz")
39    ///     .label("Asset 1")
40    ///     .state(octocrab::params::repos::release_assets::State::Uploaded)
41    ///     // Send the request
42    ///     .send()
43    ///     .await?;
44    /// # Ok(())
45    /// # }
46    /// ```
47    pub fn update(&self, asset_id: u64) -> UpdateReleaseAssetBuilder<'_, '_, '_, '_, '_> {
48        UpdateReleaseAssetBuilder::new(self, asset_id)
49    }
50
51    /// Delete a release asset using its id.
52    /// ```no_run
53    /// # async fn run() -> octocrab::Result<()> {
54    /// let release = octocrab::instance()
55    ///     .repos("owner", "repo")
56    ///     .release_assets()
57    ///     .delete(3)
58    ///     .await?;
59    /// # Ok(())
60    /// # }
61    /// ```
62    pub async fn delete(&self, id: u64) -> Result<()> {
63        let route = format!("/{}/releases/assets/{id}", self.handler.repo, id = id,);
64
65        self.handler.crab._delete(route, None::<&()>).await?;
66        Ok(())
67    }
68
69    /// Streams the binary contents of an asset.
70    /// ```no_run
71    /// # async fn run() -> octocrab::Result<()> {
72    /// use futures_util::StreamExt;
73    ///
74    /// let mut stream = octocrab::instance()
75    ///     .repos("owner", "repo")
76    ///     .release_assets()
77    ///     .stream(42u64)
78    ///     .await?;
79    ///
80    /// while let Some(chunk) = stream.next().await {
81    ///     println!("{:?}", chunk);
82    /// }
83    /// # Ok(())
84    /// # }
85    /// ```
86    #[cfg(feature = "stream")]
87    #[cfg_attr(docsrs, doc(cfg(feature = "stream")))]
88    pub async fn stream(
89        &self,
90        id: u64,
91    ) -> crate::Result<impl futures_core::Stream<Item = crate::Result<bytes::Bytes>>> {
92        use futures_util::TryStreamExt;
93        //use snafu::GenerateImplicitData;
94
95        let route = format!("/{}/releases/assets/{id}", self.handler.repo, id = id,);
96
97        let uri = Uri::builder()
98            .path_and_query(route)
99            .build()
100            .context(HttpSnafu)?;
101        let builder = Builder::new()
102            .method(http::Method::GET)
103            .uri(uri)
104            .header(http::header::ACCEPT, "application/octet-stream");
105        let request = self.handler.crab.build_request(builder, None::<&()>)?;
106        let response = self.handler.crab.execute(request).await?;
107        let response = self.handler.crab.follow_location_to_data(response).await?;
108        Ok(http_body_util::BodyStream::new(response.into_body())
109            .try_filter_map(|frame| futures_util::future::ok(frame.into_data().ok())))
110    }
111}

113/// A builder pattern struct for updating release assets.
114///
115/// created by [`ReleaseAssetsHandler::update`].
116#[derive(serde::Serialize)]
117pub struct UpdateReleaseAssetBuilder<'octo, 'repos, 'handler, 'name, 'label> {
118    #[serde(skip)]
119    handler: &'handler ReleaseAssetsHandler<'octo, 'repos>,
120    #[serde(skip)]
121    asset_id: u64,
122    #[serde(skip_serializing_if = "Option::is_none")]
123    name: Option<&'name str>,
124    #[serde(skip_serializing_if = "Option::is_none")]
125    label: Option<&'label str>,
126    #[serde(skip_serializing_if = "Option::is_none")]
127    state: Option<crate::params::repos::release_assets::State>,
128}
129
130impl<'octo, 'repos, 'handler, 'name, 'label, 'state>
131    UpdateReleaseAssetBuilder<'octo, 'repos, 'handler, 'name, 'label>
132{
133    pub(crate) fn new(
134        handler: &'handler ReleaseAssetsHandler<'octo, 'repos>,
135        asset_id: u64,
136    ) -> Self {
137        Self {
138            handler,
139            asset_id,
140            name: None,
141            label: None,
142            state: None,
143        }
144    }
145
146    /// The name of the release asset.
147    pub fn name(mut self, name: &'name (impl AsRef<str> + ?Sized)) -> Self {
148        self.name = Some(name.as_ref());
149        self
150    }
151
152    /// The label of the release asset.
153    pub fn label(mut self, label: &'label (impl AsRef<str> + ?Sized)) -> Self {
154        self.label = Some(label.as_ref());
155        self
156    }
157
158    /// The state of the release asset.
159    pub fn state<A: Into<crate::params::repos::release_assets::State>>(
160        mut self,
161        state: impl Into<Option<A>>,
162    ) -> Self {
163        self.state = state.into().map(A::into);
164        self
165    }
166
167    /// Sends the actual request.
168    pub async fn send(self) -> crate::Result<crate::models::repos::Asset> {
169        let route = format!(
170            "/{repo}/releases/assets/{asset_id}",
171            repo = self.handler.handler.repo,
172            asset_id = self.asset_id,
173        );
174        self.handler.handler.crab.patch(route, Some(&self)).await
175    }
