1//! Metadata about popular open source licenses and information about a
2//! project's license file.
3
4use crate::{models, Octocrab};
5
6/// Handler for GitHub's license API.
7///
8/// Created with [`Octocrab::licenses`].
9pub struct LicenseHandler<'octo> {
10    crab: &'octo Octocrab,
11}
12
13impl<'octo> LicenseHandler<'octo> {
14    pub(crate) fn new(crab: &'octo Octocrab) -> Self {
15        Self { crab }
16    }
17
18    /// List commonly used licenses.
19    /// ```no_run
20    /// # async fn run() -> octocrab::Result<()> {
21    /// let licenses = octocrab::instance().licenses().list_commonly_used().await?;
22    /// # Ok(())
23    /// # }
24    /// ```
25    pub async fn list_commonly_used(&self) -> crate::Result<Vec<models::License>> {
26        self.crab.get("/licenses", None::<&()>).await
27    }
28
29    /// Get an individual license.
30    /// ```no_run
31    /// # async fn run() -> octocrab::Result<()> {
32    /// let license = octocrab::instance().licenses().get("mit").await?;
33    /// # Ok(())
34    /// # }
35    /// ```
36    pub async fn get(&self, key: impl AsRef<str>) -> crate::Result<models::License> {
37        self.crab
38            .get(format!("/licenses/{}", key.as_ref()), None::<&()>)
39            .await
40    }
41}
