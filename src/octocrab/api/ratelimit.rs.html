5/// Handler for GitHub's rate_limit API.
6///
7/// Created with [`Octocrab::ratelimit`].
8pub struct RateLimitHandler<'octo> {
9    crab: &'octo Octocrab,
10}
11
12impl<'octo> RateLimitHandler<'octo> {
13    pub(crate) fn new(crab: &'octo Octocrab) -> Self {
14        Self { crab }
15    }
16
17    /// Get the rate limit.
18    /// ```no_run
19    /// # async fn run() -> octocrab::Result<()> {
20    /// let ratelimit = octocrab::instance()
21    ///     .ratelimit()
22    ///     .get()
23    ///     .await?;
24    /// # Ok(())
25    /// # }
26    /// ```
27    pub async fn get(&self) -> Result<models::RateLimit> {
28        self.crab.get("/rate_limit", None::<&()>).await
29    }
30}
