118/// A handler for handling search queries to GitHub.
119#[derive(Clone, Debug, serde::Serialize)]
120pub struct QueryHandler<'octo, 'query, T> {
121    #[serde(skip)]
122    return_type: std::marker::PhantomData<T>,
123    #[serde(skip)]
124    crab: &'octo Octocrab,
125    #[serde(skip)]
126    route: &'static str,
127    #[serde(rename = "q")]
128    query: &'query str,
129    per_page: Option<u8>,
130    page: Option<u32>,
131    #[serde(skip_serializing_if = "Option::is_none")]
132    sort: Option<String>,
133    #[serde(skip_serializing_if = "Option::is_none")]
134    order: Option<String>,
135}
136
137impl<'octo, 'query, T> QueryHandler<'octo, 'query, T> {
138    pub(crate) fn new(crab: &'octo Octocrab, route: &'static str, query: &'query str) -> Self {
139        Self {
140            crab,
141            order: None,
142            page: None,
143            per_page: None,
144            query,
145            return_type: std::marker::PhantomData,
146            route,
147            sort: None,
148        }
149    }
150
151    /// Sets the `sort` parameter for the query. The exact parameters for this
152    /// method will vary based on what is being searched.
153    pub fn sort<S: Into<String>>(mut self, sort: impl Into<Option<S>>) -> Self {
154        self.sort = sort.into().map(S::into);
155        self
156    }
157
158    /// Sets the `order` parameter for the query. The exact parameters for this
159    /// method will vary based on what is being searched.
160    pub fn order<S: Into<String>>(mut self, order: impl Into<Option<S>>) -> Self {
161        self.order = order.into().map(S::into);
162        self
163    }
164
165    /// Results per page (max 100).
166    pub fn per_page(mut self, per_page: impl Into<u8>) -> Self {
167        self.per_page = Some(per_page.into());
168        self
169    }
170
171    /// Page number of the results to fetch.
172    pub fn page(mut self, page: impl Into<u32>) -> Self {
173        self.page = Some(page.into());
174        self
175    }
176}
177
178impl<'octo, 'query, T: serde::de::DeserializeOwned> QueryHandler<'octo, 'query, T> {
179    /// Send the actual request.
180    pub async fn send(self) -> crate::Result<crate::Page<T>> {
181        self.crab
182            .get(&format!("/search/{}", self.route), Some(&self))
183            .await
184    }
185}
