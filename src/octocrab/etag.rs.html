10/// Represents resources identified by etags.
11#[derive(Debug, PartialEq)]
12pub struct Etagged<T> {
13    /// A possible etag.
14    ///
15    /// It is possible, although unlikely, that a response which should contain an etag header does
16    /// not, or that etag header is invalid. In such cases this field will be `None`.
17    pub etag: Option<EntityTag>,
18    /// The value identified by this etag.
19    ///
20    /// This can be `None` if we have already received the data which this etag identifies.
21    pub value: Option<T>,
22}

28/// An entity tag, defined in [RFC7232](https://tools.ietf.org/html/rfc7232#section-2.3)
29///
30/// An entity tag consists of a string enclosed by two literal double quotes.
31/// Preceding the first double quote is an optional weakness indicator,
32/// which always looks like `W/`. Examples for valid tags are `"xyzzy"` and `W/"xyzzy"`.
33///
34/// # ABNF
35///
36/// ```text
37/// entity-tag = [ weak ] opaque-tag
38/// weak       = %x57.2F ; "W/", case-sensitive
39/// opaque-tag = DQUOTE *etagc DQUOTE
40/// etagc      = %x21 / %x23-7E / obs-text
41///            ; VCHAR except double quotes, plus obs-text
42/// ```
43///
44/// # Comparison
45/// To check if two entity tags are equivalent in an application always use the `strong_eq` or
46/// `weak_eq` methods based on the context of the Tag. Only use `==` to check if two tags are
47/// identical.
48///
49/// The example below shows the results for a set of entity-tag pairs and
50/// both the weak and strong comparison function results:
51///
52/// | ETag 1  | ETag 2  | Strong Comparison | Weak Comparison |
53/// |---------|---------|-------------------|-----------------|
54/// | `W/"1"` | `W/"1"` | no match          | match           |
55/// | `W/"1"` | `W/"2"` | no match          | no match        |
56/// | `W/"1"` | `"1"`   | no match          | match           |
57/// | `"1"`   | `"1"`   | match             | match           |
58#[derive(Clone, Debug, Eq, PartialEq)]
59pub struct EntityTag {
60    /// Weakness indicator for the tag
61    pub weak: bool,
62    /// The opaque string in between the DQUOTEs
63    tag: String,
64}
65
66impl EntityTag {
67    pub fn extract_from_response<B>(response: &http::Response<B>) -> Option<EntityTag> {
68        response
69            .headers()
70            .get("ETag")
71            .and_then(|it| it.to_str().ok())
72            .and_then(|it| EntityTag::from_str(it).ok())
73    }
74
75    pub fn insert_if_none_match_header(
76        headers: &mut HeaderMap,
77        etag: EntityTag,
78    ) -> Result<(), crate::Error> {
79        headers.insert(
80            "If-None-Match",
81            etag.to_string()
82                .parse()
83                .map_err(|err: InvalidHeaderValue| crate::Error::InvalidHeaderValue {
84                    source: err,
85                    backtrace: snafu::Backtrace::capture(),
86                })?,
87        );
88        Ok(())
89    }
90
91    /// Constructs a new EntityTag.
92    /// # Panics
93    /// If the tag contains invalid characters.
94    pub fn new(weak: bool, tag: String) -> EntityTag {
95        assert!(check_slice_validity(&tag), "Invalid tag: {:?}", tag);
96        EntityTag { weak, tag }
97    }
98
99    /// Constructs a new weak EntityTag.
100    /// # Panics
101    /// If the tag contains invalid characters.
102    pub fn weak(tag: String) -> EntityTag {
103        EntityTag::new(true, tag)
104    }
105
106    /// Constructs a new strong EntityTag.
107    /// # Panics
108    /// If the tag contains invalid characters.
109    pub fn strong(tag: String) -> EntityTag {
110        EntityTag::new(false, tag)
111    }
112
113    /// Get the tag.
114    pub fn tag(&self) -> &str {
115        self.tag.as_ref()
116    }
117
118    /// Set the tag.
119    /// # Panics
120    /// If the tag contains invalid characters.
121    pub fn set_tag(&mut self, tag: String) {
122        assert!(check_slice_validity(&tag), "Invalid tag: {:?}", tag);
123        self.tag = tag
124    }
125
126    /// For strong comparison two entity-tags are equivalent if both are not weak and their
127    /// opaque-tags match character-by-character.
128    pub fn strong_eq(&self, other: &EntityTag) -> bool {
129        !self.weak && !other.weak && self.tag == other.tag
130    }
131
132    /// For weak comparison two entity-tags are equivalent if their
133    /// opaque-tags match character-by-character, regardless of either or
134    /// both being tagged as "weak".
135    pub fn weak_eq(&self, other: &EntityTag) -> bool {
136        self.tag == other.tag
137    }
138
139    /// The inverse of `EntityTag.strong_eq()`.
140    pub fn strong_ne(&self, other: &EntityTag) -> bool {
141        !self.strong_eq(other)
142    }
143
144    /// The inverse of `EntityTag.weak_eq()`.
145    pub fn weak_ne(&self, other: &EntityTag) -> bool {
146        !self.weak_eq(other)
147    }
148}

150impl Display for EntityTag {
151    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
152        if self.weak {
153            write!(f, "W/\"{}\"", self.tag)
154        } else {
155            write!(f, "\"{}\"", self.tag)
156        }
157    }
158}

160/// check that each char in the slice is either:
161/// 1. `%x21`, or
162/// 2. in the range `%x23` to `%x7E`, or
163/// 3. above `%x80`
164fn check_slice_validity(slice: &str) -> bool {
165    slice
166        .bytes()
167        .all(|c| c == b'\x21' || (b'\x23'..=b'\x7e').contains(&c) | (c >= b'\x80'))
168}
169
170impl FromStr for EntityTag {
171    type Err = String;
172    fn from_str(s: &str) -> Result<EntityTag, Self::Err> {
173        let length: usize = s.len();
174        let slice = s;
175        // Early exits if it doesn't terminate in a DQUOTE.
176        if !slice.ends_with('"') || slice.len() < 2 {
177            return Err("Does not end with double quote character".to_string());
178        }
179        // The etag is weak if its first char is not a DQUOTE.
180        if slice.len() >= 2 && slice.starts_with('"') && check_slice_validity(&slice[1..length - 1])
181        {
182            // No need to check if the last char is a DQUOTE,
183            // we already did that above.
184            return Ok(EntityTag {
185                weak: false,
186                tag: slice[1..length - 1].to_owned(),
187            });
188        } else if slice.len() >= 4
189            && slice.starts_with("W/\"")
190            && check_slice_validity(&slice[3..length - 1])
191        {
192            return Ok(EntityTag {
193                weak: true,
194                tag: slice[3..length - 1].to_owned(),
195            });
196        }
197        Err("Could not parse EntityTag".to_string())
198    }
199}

205    #[test]
206    fn test_etag_parse_success() {
207        // Expected success
208        assert_eq!(
209            "\"foobar\"".parse::<EntityTag>().unwrap(),
210            EntityTag::strong("foobar".to_owned())
211        );
212        assert_eq!(
213            "\"\"".parse::<EntityTag>().unwrap(),
214            EntityTag::strong("".to_owned())
215        );
216        assert_eq!(
217            "W/\"weaktag\"".parse::<EntityTag>().unwrap(),
218            EntityTag::weak("weaktag".to_owned())
219        );
220        assert_eq!(
221            "W/\"\x65\x62\"".parse::<EntityTag>().unwrap(),
222            EntityTag::weak("\x65\x62".to_owned())
223        );
224        assert_eq!(
225            "W/\"\"".parse::<EntityTag>().unwrap(),
226            EntityTag::weak("".to_owned())
227        );
228    }

230    #[test]
231    fn test_etag_parse_failures() {
232        // Expected failures
233        assert!("no-dquotes".parse::<EntityTag>().is_err());

237        assert!("".parse::<EntityTag>().is_err());

239        assert!("unmatched-dquotes2\"".parse::<EntityTag>().is_err());
240        assert!("matched-\"dquotes\"".parse::<EntityTag>().is_err());

261    #[test]
262    fn test_cmp() {
263        // | ETag 1  | ETag 2  | Strong Comparison | Weak Comparison |
264        // |---------|---------|-------------------|-----------------|
265        // | `W/"1"` | `W/"1"` | no match          | match           |
266        // | `W/"1"` | `W/"2"` | no match          | no match        |
267        // | `W/"1"` | `"1"`   | no match          | match           |
268        // | `"1"`   | `"1"`   | match             | match           |
269        let mut etag1 = EntityTag::weak("1".to_owned());
270        let mut etag2 = EntityTag::weak("1".to_owned());
271        assert!(!etag1.strong_eq(&etag2));
272        assert!(etag1.weak_eq(&etag2));
273        assert!(etag1.strong_ne(&etag2));
274        assert!(!etag1.weak_ne(&etag2));
275
276        etag1 = EntityTag::weak("1".to_owned());
277        etag2 = EntityTag::weak("2".to_owned());
278        assert!(!etag1.strong_eq(&etag2));
279        assert!(!etag1.weak_eq(&etag2));
280        assert!(etag1.strong_ne(&etag2));
281        assert!(etag1.weak_ne(&etag2));
282
283        etag1 = EntityTag::weak("1".to_owned());
284        etag2 = EntityTag::strong("1".to_owned());
285        assert!(!etag1.strong_eq(&etag2));
286        assert!(etag1.weak_eq(&etag2));
287        assert!(etag1.strong_ne(&etag2));
288        assert!(!etag1.weak_ne(&etag2));
289
290        etag1 = EntityTag::strong("1".to_owned());
291        etag2 = EntityTag::strong("1".to_owned());
292        assert!(etag1.strong_eq(&etag2));
293        assert!(etag1.weak_eq(&etag2));
294        assert!(!etag1.strong_ne(&etag2));
295        assert!(!etag1.weak_ne(&etag2));
296    }
