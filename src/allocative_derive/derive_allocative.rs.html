32const fn hash(s: &str) -> u64 {
33    let mut hash = 0xcbf29ce484222325;
34    let mut i = 0;
35    while i < s.as_bytes().len() {
36        let b = s.as_bytes()[i];
37        hash ^= b as u64;
38        hash = hash.wrapping_mul(0x100000001b3);
39        i += 1;
40    }
41    hash
42}

44pub(crate) fn derive_allocative(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
45    match derive_allocative_impl(input.into()) {
46        Ok(tokens) => tokens.into(),
47        Err(err) => err.to_compile_error().into(),
48    }
49}

51fn impl_generics(
52    generics: &Generics,
53    attrs: &AllocativeAttrs,
54) -> syn::Result<proc_macro2::TokenStream> {
55    if let Some(bound) = &attrs.bound {
56        if !bound.is_empty() {
57            let span = attrs.span.unwrap_or_else(Span::call_site);
58            let bound = bound.parse::<proc_macro2::TokenStream>()?;
59            return Ok(quote_spanned! { span => < #bound > });
60        }
61    }
62
63    let mut impl_generics = Vec::new();
64    for p in &generics.params {
65        impl_generics.push(match p {
66            GenericParam::Type(tp) => {
67                let mut tp = tp.clone();
68                if attrs.bound.is_none() && !attrs.skip {
69                    tp.bounds.push(syn::parse2(quote_spanned! { tp.span() =>
70                        allocative::Allocative
71                    })?);
72                }
73                tp.default = None;
74                tp.to_token_stream()
75            }
76            GenericParam::Lifetime(l) => l.to_token_stream(),
77            GenericParam::Const(c) => c.to_token_stream(),
78        });
79    }
80    if impl_generics.is_empty() {
81        Ok(quote_spanned! { generics.span() => })
82    } else {
83        Ok(quote_spanned! { generics.span() => <#(#impl_generics),*> })
84    }
85}

87fn derive_allocative_impl(
88    input: proc_macro2::TokenStream,
89) -> syn::Result<proc_macro2::TokenStream> {
90    let input: DeriveInput = syn::parse2(input)?;
91    let name = &input.ident;
92    let (_impl_generics, type_generics, where_clause) = input.generics.split_for_impl();
93
94    let attrs = extract_attrs(&input.attrs)?;
95    let impl_generics = impl_generics(&input.generics, &attrs)?;
96
97    let body = if attrs.skip {
98        quote_spanned! { input.span() =>
99        }
100    } else {
101        gen_visit_body(&input)?
102    };
103
104    Ok(quote_spanned! {input.span()=>
105        impl #impl_generics allocative::Allocative for #name #type_generics #where_clause {
106            #[allow(unused, warnings)]
107            fn visit<'allocative_a, 'allocative_b: 'allocative_a>(
108                &self,
109                visitor: &'allocative_a mut allocative::Visitor<'allocative_b>,
110            ) {
111                let mut visitor = visitor.enter_self::<Self>(self);
112                #body
113                visitor.exit();
114            }
115        }
116    })
117}

119fn gen_visit_body(input: &DeriveInput) -> syn::Result<proc_macro2::TokenStream> {
120    match &input.data {
121        Data::Struct(s) => gen_visit_struct(s),
122        Data::Union(_) => Err(syn::Error::new_spanned(input, "unions are not supported")),
123        Data::Enum(e) => gen_visit_enum(e),
124    }
125}

127fn gen_visit_enum(input: &DataEnum) -> syn::Result<proc_macro2::TokenStream> {
128    let cases = input
129        .variants
130        .iter()
131        .map(gen_visit_enum_variant)
132        .collect::<syn::Result<proc_macro2::TokenStream>>()?;
133    if cases.is_empty() {
134        // Rust does not understand
135        // ```
136        // match self {
137        // }
138        // ```
139        // for enums with no variants where `self` is `&Self`.
140        Ok(quote_spanned! {input.variants.span()=>
141        })
142    } else {
143        Ok(quote_spanned! {input.variants.span()=>
144        match self {
145                #cases
146            }
147        })
148    }
149}

151fn allocative_key(s: &str) -> proc_macro2::TokenStream {
152    // Compile hash at proc macro time, otherwise it will have to be computed by MIRI.
153    let hash = hash(s);
154    quote_spanned! {proc_macro2::Span::call_site()=>
155        allocative::Key::new_unchecked(#hash, #s)
156    }
157}

159fn gen_visit_enum_variant(input: &Variant) -> syn::Result<proc_macro2::TokenStream> {
160    let name = &input.ident;
161    let name_str = name.to_string();
162
163    let variant_attrs = extract_attrs(&input.attrs)?;
164
165    let variant_key = allocative_key(&name_str);
166
167    // TODO: enter variant.
168    match &input.fields {
169        Fields::Unit => Ok(quote_spanned! {input.span()=>
170            Self::#name => {},
171        }),
172        Fields::Unnamed(unnamed) => {
173            let field_names = fields_unnamed_names(unnamed)?;
174            let visit_fields = unnamed
175                .unnamed
176                .iter()
177                .enumerate()
178                .zip(field_names.iter())
179                .map(|((i, f), n)| {
180                    if variant_attrs.skip {
181                        Ok(quote_spanned! {f.span()=>
182                        })
183                    } else {
184                        gen_visit_field(&i.to_string(), n, f)
185                    }
186                })
187                .collect::<syn::Result<proc_macro2::TokenStream>>()?;
188            Ok(quote_spanned! {input.span()=>
189                Self::#name(#(#field_names),*) => {
190                    let mut visitor = visitor.enter(#variant_key, std::mem::size_of::<Self>());
191                    #visit_fields
192                    visitor.exit();
193                },
194            })
195        }
196        Fields::Named(named) => {
197            let field_names = fields_named_names(named)?;
198            let visit_fields = field_names
199                .iter()
200                .zip(named.named.iter())
201                .map(|(ident, f)| {
202                    if variant_attrs.skip {
203                        Ok(quote_spanned! {f.span()=>
204                        })
205                    } else {
206                        gen_visit_field(&ident.to_string(), ident, f)
207                    }
208                })
209                .collect::<syn::Result<proc_macro2::TokenStream>>()?;
210            Ok(quote_spanned! {input.span()=>
211                Self::#name { #(#field_names),* } => {
212                    let mut visitor = visitor.enter(#variant_key, std::mem::size_of::<Self>());
213                    #visit_fields
214                    visitor.exit();
215                },
216            })
217        }
218    }
219}

221fn fields_named_names(fields: &FieldsNamed) -> syn::Result<Vec<&Ident>> {
222    fields
223        .named
224        .iter()
225        .map(|f| {
226            f.ident
227                .as_ref()
228                .ok_or_else(|| syn::Error::new_spanned(f, "unnamed field in named"))
229        })
230        .collect()
231}
232
233fn fields_unnamed_names(fields: &FieldsUnnamed) -> syn::Result<Vec<Ident>> {
234    fields
235        .unnamed
236        .iter()
237        .enumerate()
238        .map(|(i, f)| Ok(Ident::new(&format!("f{}", i), f.span())))
239        .collect()
240}

242fn gen_visit_struct(input: &DataStruct) -> syn::Result<proc_macro2::TokenStream> {
243    match &input.fields {
244        Fields::Named(named) => {
245            let names = fields_named_names(named)?;
246            let visit_fields = names
247                .iter()
248                .zip(named.named.iter())
249                .map(|(ident, f)| gen_visit_field(&ident.to_string(), ident, f))
250                .collect::<syn::Result<proc_macro2::TokenStream>>()?;
251            Ok(quote_spanned! {input.fields.span()=>
252                let Self { #(#names),* } = self;
253                #visit_fields
254            })
255        }
256        Fields::Unnamed(unnamed) => {
257            let names = fields_unnamed_names(unnamed)?;
258            let visit_fields = names
259                .iter()
260                .enumerate()
261                .zip(unnamed.unnamed.iter())
262                .map(|((i, ident), field)| gen_visit_field(&i.to_string(), ident, field))
263                .collect::<syn::Result<proc_macro2::TokenStream>>()?;
264            Ok(quote_spanned! {input.fields.span()=>
265                let Self(#(#names),*) = self;
266                #visit_fields
267            })
268        }
269        Fields::Unit => Ok(quote_spanned! {input.fields.span()=>}),
270    }
271}

273fn gen_visit_field(
274    label: &str,
275    ident: &Ident,
276    field: &Field,
277) -> syn::Result<proc_macro2::TokenStream> {
278    let attrs = extract_attrs(&field.attrs)?;
279    let field_key = allocative_key(label);
280    if attrs.skip {
281        Ok(quote_spanned! {field.span()=>})
282    } else if let Some(visit) = attrs.visit {
283        let ty = &field.ty;
284        Ok(quote_spanned! {field.span()=>
285            // TODO(nga): figure out how to put this snippet in a member function of the visitor.
286            {
287                let mut visitor = visitor.enter(#field_key, std::mem::size_of::<#ty>());
288                #visit(#ident, &mut visitor);
289                visitor.exit();
290            }
291        })
292    } else {
293        // Specify type parameter explicitly to prevent implicit conversion.
294        let ty = &field.ty;
295        Ok(quote_spanned! {ident.span()=>
296            visitor.visit_field::<#ty>(#field_key, #ident);
297        })
298    }
299}

301#[derive(Default)]
302struct AllocativeAttrs {
303    span: Option<Span>,
304    skip: bool,
305    bound: Option<String>,
306    visit: Option<Path>,
307}
308
309/// Parse an `#[allocative(...)]` annotation.
310fn extract_attrs(attrs: &[Attribute]) -> syn::Result<AllocativeAttrs> {
311    syn::custom_keyword!(skip);
312    syn::custom_keyword!(bound);
313    syn::custom_keyword!(visit);
314
315    let mut opts = AllocativeAttrs::default();
316
317    for attr in attrs.iter() {
318        if !attr.path().is_ident("allocative") {
319            continue;
320        }
321
322        opts.span = Some(attr.span());
323
324        attr.parse_args_with(|input: ParseStream| {
325            loop {
326                if input.parse::<skip>().is_ok() {
327                    if opts.skip {
328                        return Err(input.error("`skip` was set twice"));
329                    }
330                    opts.skip = true;
331                } else if input.parse::<bound>().is_ok() {
332                    input.parse::<Token![=]>()?;
333                    let bound = input.parse::<LitStr>()?;
334                    if opts.bound.is_some() {
335                        return Err(input.error("`bound` was set twice"));
336                    }
337                    opts.bound = Some(bound.value());
338                } else if input.parse::<visit>().is_ok() {
339                    input.parse::<Token![=]>()?;
340                    let visit = input.parse::<Path>()?;
341                    if opts.visit.is_some() {
342                        return Err(input.error("`visit` was set twice"));
343                    }
344                    opts.visit = Some(visit);
345                }
346
347                if input.is_empty() {
348                    return Ok(());
349                }
350
351                input.parse::<Token![,]>()?;
352            }
353        })?;
354    }
355
356    Ok(opts)
