4macro_rules! impl_to_primitive {
5    ($ty:ty, $to:ident) => {
6        #[inline]
7        fn $to(&self) -> Option<$ty> {
8            if self.im.is_zero() {
9                self.re.$to()
10            } else {
11                None
12            }
13        }
14    };
15} // impl_to_primitive

35macro_rules! impl_from_primitive {
