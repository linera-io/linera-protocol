10#[inline]
11pub fn get_memory_input_and_out_ranges(
12    interpreter: &mut Interpreter<impl InterpreterTypes>,
13) -> Option<(Range<usize>, Range<usize>)> {
14    popn!([in_offset, in_len, out_offset, out_len], interpreter, None);
15
16    let mut in_range = resize_memory(interpreter, in_offset, in_len)?;
17
18    if !in_range.is_empty() {
19        let offset = interpreter.memory.local_memory_offset();
20        in_range = in_range.start.saturating_add(offset)..in_range.end.saturating_add(offset);
21    }
22
23    let ret_range = resize_memory(interpreter, out_offset, out_len)?;
24    Some((in_range, ret_range))
25}

27/// Resize memory and return range of memory.
28/// If `len` is 0 dont touch memory and return `usize::MAX` as offset and 0 as length.
29#[inline]
30pub fn resize_memory(
31    interpreter: &mut Interpreter<impl InterpreterTypes>,
32    offset: U256,
33    len: U256,
34) -> Option<Range<usize>> {
35    let len = as_usize_or_fail_ret!(interpreter, len, None);
36    let offset = if len != 0 {
37        let offset = as_usize_or_fail_ret!(interpreter, offset, None);
38        resize_memory!(interpreter, offset, len, None);
39        offset
40    } else {
41        usize::MAX //unrealistic value so we are sure it is not used
42    };
43    Some(offset..offset + len)
44}

46#[inline]
47pub fn calc_call_gas(
48    interpreter: &mut Interpreter<impl InterpreterTypes>,
49    account_load: StateLoad<AccountLoad>,
50    has_transfer: bool,
51    local_gas_limit: u64,
52) -> Option<u64> {
53    let call_cost = gas::call_cost(
54        interpreter.runtime_flag.spec_id(),
55        has_transfer,
56        account_load,
57    );
58    gas!(interpreter, call_cost, None);
59
60    // EIP-150: Gas cost changes for IO-heavy operations
61    let gas_limit = if interpreter.runtime_flag.spec_id().is_enabled_in(TANGERINE) {
62        // Take l64 part of gas_limit
63        min(
64            interpreter.control.gas().remaining_63_of_64_parts(),
65            local_gas_limit,
66        )
67    } else {
68        local_gas_limit
69    };
70
71    Some(gas_limit)
