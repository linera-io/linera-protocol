3#[derive(Debug, Clone)]
4struct Entry<T> {
5    item: Option<T>,
6    /// The next item in the list (possibly itself).
7    /// When `item` is None, `next` points ot the next free token in the slab.
8    next: Token,
9    /// The previous item in the list (possibly itself).
10    /// Unused if `item` is None.
11    prev: Token,
12}
13
14/// A slab like structure that also maintains circular lists with its items.
15#[derive(Debug, Clone)]
16pub struct LinkedSlab<T> {
17    entries: Vec<Entry<T>>,
18    next_free: Token,
19}

21impl<T> LinkedSlab<T> {

39    #[cfg(any(fuzzing, test))]
40    pub fn validate(&self) {
41        let mut freelist = std::collections::HashSet::new();
42        let mut next_free = self.next_free;
43        while next_free.get() as usize - 1 != self.entries.len() {
44            freelist.insert(next_free.get() as usize);
45            let e = &self.entries[next_free.get() as usize - 1];
46            assert!(e.item.is_none(), "{next_free} is in freelist but has item");
47            next_free = e.next;
48        }
49        for (i, e) in self.entries.iter().enumerate() {
50            if e.item.is_some() {
51                assert!(!freelist.contains(&(i + 1)));
52                assert!(!freelist.contains(&(e.prev.get() as usize)));
53                assert!(!freelist.contains(&(e.next.get() as usize)));
54            }
55        }
56    }

58    /// Inserts a new entry in the list.
59    /// Initially, the item will belong to a list only containing itself.
60    ///
61    /// # Panics
62    /// Panics if number of items exceed `u32::MAX - 1`.
63    pub fn insert(&mut self, item: T) -> Token {
64        let token = self.next_free;
65        // eprintln!("linkedslab::insert token {token} head {head:?}");
66        let idx = (token.get() - 1) as usize;
67        if idx < self.entries.len() {
68            let entry = &mut self.entries[idx];
69            self.next_free = entry.next;
70            (entry.prev, entry.next) = (token, token);
71            debug_assert!(entry.item.is_none());
72            entry.item = Some(item);
73        } else {
74            debug_assert_eq!(idx, self.entries.len());
75            self.next_free = Token::new(token.get().wrapping_add(1)).expect("Capacity overflow");
76            self.entries.push(Entry {
77                next: token,
78                prev: token,
79                item: Some(item),
80            });
81        }
82        token
83    }

85    /// Gets an entry and a token to the next entry.
86    #[inline]
87    pub fn get(&self, index: Token) -> Option<(&T, Token)> {
88        if let Some(entry) = self.entries.get((index.get() - 1) as usize) {
89            if let Some(v) = &entry.item {
90                return Some((v, entry.next));
91            }
92        }
93        None
94    }

123    /// Links an entry before `target_head`. Returns the item next to the linked item,
124    /// which is either the item itself or `target_head`.
125    ///
126    /// # Panics
127    /// Panics on out of bounds access.
128    /// Panics (in debug mode) if linking an absent entry.
129    pub fn link(&mut self, idx: Token, target_head: Option<Token>) -> Token {
130        // eprintln!("linkedslab::link {idx} head {target_head:?}");
131        let (prev, next) = if let Some(target_head) = target_head {
132            let head = &mut self.entries[(target_head.get() - 1) as usize];
133            debug_assert!(head.item.is_some());
134            if head.prev == target_head {
135                // existing list has a single item linking to itself
136                head.prev = idx;
137                head.next = idx;
138                (target_head, target_head)
139            } else {
140                let before_head_idx = head.prev;
141                head.prev = idx;
142                let before_head = &mut self.entries[(before_head_idx.get() - 1) as usize];
143                debug_assert!(before_head.item.is_some());
144                debug_assert_eq!(before_head.next, target_head);
145                before_head.next = idx;
146                (before_head_idx, target_head)
147            }
148        } else {
149            (idx, idx)
150        };
151
152        let entry = &mut self.entries[(idx.get() - 1) as usize];
153        debug_assert!(entry.item.is_some());
154        debug_assert_eq!(entry.next, idx);
155        debug_assert_eq!(entry.prev, idx);
156        (entry.prev, entry.next) = (prev, next);
157        next
158    }

160    /// Unlinks an entry without removing it from the ring
161    /// Returns next item in the list, if not self.
162    ///
163    /// # Panics
164    /// Panics on out of bounds access.
165    /// Panics (in debug mode) if unlinking an absent entry.
166    pub fn unlink(&mut self, idx: Token) -> Option<Token> {
167        // eprintln!("linkedslab::unlink {idx}");
168        let entry = &mut self.entries[(idx.get() - 1) as usize];
169        debug_assert!(entry.item.is_some());
170        let (prev_idx, next_idx) = (entry.prev, entry.next);
171        if next_idx == idx {
172            debug_assert_eq!(prev_idx, idx);
173            // single item list, nothing to do
174            None
175        } else {
176            (entry.prev, entry.next) = (idx, idx);
177            let next = &mut self.entries[(next_idx.get() - 1) as usize];
178            debug_assert!(next.item.is_some());
179            debug_assert_eq!(next.prev, idx);
180            next.prev = prev_idx;
181            let prev = &mut self.entries[(prev_idx.get() - 1) as usize];
182            debug_assert!(prev.item.is_some());
183            debug_assert_eq!(prev.next, idx);
184            prev.next = next_idx;
185            Some(next_idx)
186        }
187    }

189    /// Unlinks and removes the entry from the ring.
190    /// Returns next item in the list, if not self.
191    ///
192    /// # Panics
193    /// Panics on out of bounds access.
194    pub fn remove(&mut self, idx: Token) -> Option<(T, Option<Token>)> {
195        // eprintln!("linkedslab::remove {idx}");
196        let next = self.unlink(idx);
197        let entry = &mut self.entries[(idx.get() - 1) as usize];
198        let old_item = entry.item.take()?;
199        entry.next = self.next_free;
200        self.next_free = idx;
201        Some((old_item, next))
202    }

204    /// The Token that will be returned by the next call to `insert()`
205    pub fn next_free(&self) -> Token {
206        self.next_free
207    }

209    /// Drains all items from the slab.
210    ///
211    /// The slab will be emptied even if the returned iterator isn't fully consumed.
212    pub fn drain(&mut self) -> impl Iterator<Item = T> + '_ {
213        self.next_free = Token::new(1).unwrap();
214        self.entries.drain(..).flat_map(|i| i.item)
215    }

217    /// Iterator for the items in the slab
218    pub fn iter(&self) -> impl Iterator<Item = &'_ T> + '_ {
219        self.entries.iter().flat_map(|i| i.item.as_ref())
220    }

222    /// Iterator for the items in the slab, starting after a given token.
223    pub fn iter_from(
224        &self,
225        continuation: Option<Token>,
226    ) -> impl Iterator<Item = (Token, &'_ T)> + '_ {
227        let skip = continuation.map_or(0, |c| c.get() as usize);
228        self.entries
229            .iter()
230            .skip(skip)
231            .enumerate()
232            .filter_map(move |(i, e)| {
233                if let Some(item) = &e.item {
234                    Some((Token::new((skip + i + 1) as u32)?, item))
235                } else {
236                    None
237                }
238            })
239    }
