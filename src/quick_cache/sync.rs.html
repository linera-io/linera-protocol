19/// A concurrent cache
20///
21/// The concurrent cache is internally composed of equally sized shards, each of which is independently
22/// synchronized. This allows for low contention when multiple threads are accessing the cache but limits the
23/// maximum weight capacity of each shard.
24///
25/// # Value
26/// Cache values are cloned when fetched. Users should wrap their values with `Arc<_>`
27/// if necessary to avoid expensive clone operations. If interior mutability is required
28/// `Arc<Mutex<_>>` or `Arc<RwLock<_>>` can also be used.
29///
30/// # Thread Safety and Concurrency
31/// The cache instance can wrapped with an `Arc` (or equivalent) and shared between threads.
32/// All methods are accessible via non-mut references so no further synchronization (e.g. Mutex) is needed.
33pub struct Cache<
34    Key,
35    Val,
36    We = UnitWeighter,
37    B = DefaultHashBuilder,
38    L = DefaultLifecycle<Key, Val>,
39> {
40    hash_builder: B,
41    shards: Box<[RwLock<CacheShard<Key, Val, We, B, L, SharedPlaceholder<Val>>>]>,
42    shards_mask: u64,
43    lifecycle: L,
44}

83    /// Creates a new cache that can hold up to `weight_capacity` in weight.
84    /// `estimated_items_capacity` is the estimated number of items the cache is expected to hold,
85    /// roughly equivalent to `weight_capacity / average item weight`.
86    pub fn with(
87        estimated_items_capacity: usize,
88        weight_capacity: u64,
89        weighter: We,
90        hash_builder: B,
91        lifecycle: L,
92    ) -> Self {
93        Self::with_options(
94            OptionsBuilder::new()
95                .estimated_items_capacity(estimated_items_capacity)
96                .weight_capacity(weight_capacity)
97                .build()
98                .unwrap(),
99            weighter,
100            hash_builder,
101            lifecycle,
102        )
103    }

105    /// Constructs a cache based on [OptionsBuilder].
106    ///
107    /// # Example
108    ///
109    /// ```rust
110    /// use quick_cache::{sync::{Cache, DefaultLifecycle}, OptionsBuilder, UnitWeighter, DefaultHashBuilder};
111    ///
112    /// Cache::<(String, u64), String>::with_options(
113    ///   OptionsBuilder::new()
114    ///     .estimated_items_capacity(10000)
115    ///     .weight_capacity(10000)
116    ///     .build()
117    ///     .unwrap(),
118    ///     UnitWeighter,
119    ///     DefaultHashBuilder::default(),
120    ///     DefaultLifecycle::default(),
121    /// );
122    /// ```
123    pub fn with_options(options: Options, weighter: We, hash_builder: B, lifecycle: L) -> Self {
124        let mut num_shards = options.shards.next_power_of_two() as u64;
125        let estimated_items_capacity = options.estimated_items_capacity as u64;
126        let weight_capacity = options.weight_capacity;
127        let mut shard_items_cap =
128            estimated_items_capacity.saturating_add(num_shards - 1) / num_shards;
129        let mut shard_weight_cap =
130            options.weight_capacity.saturating_add(num_shards - 1) / num_shards;
131        // try to make each shard hold at least 32 items
132        while shard_items_cap < 32 && num_shards > 1 {
133            num_shards /= 2;
134            shard_items_cap = estimated_items_capacity.saturating_add(num_shards - 1) / num_shards;
135            shard_weight_cap = weight_capacity.saturating_add(num_shards - 1) / num_shards;
136        }
137        let shards = (0..num_shards)
138            .map(|_| {
139                RwLock::new(CacheShard::new(
140                    options.hot_allocation,
141                    options.ghost_allocation,
142                    shard_items_cap as usize,
143                    shard_weight_cap,
144                    weighter.clone(),
145                    hash_builder.clone(),
146                    lifecycle.clone(),
147                ))
148            })
149            .collect::<Vec<_>>();
150        Self {
151            shards: shards.into_boxed_slice(),
152            hash_builder,
153            shards_mask: num_shards - 1,
154            lifecycle,
155        }
156    }

209    #[inline]
210    fn shard_for<Q>(
211        &self,
212        key: &Q,
213    ) -> Option<(
214        &RwLock<CacheShard<Key, Val, We, B, L, SharedPlaceholder<Val>>>,
215        u64,
216    )>
217    where
218        Q: Hash + Equivalent<Key> + ?Sized,
219    {
220        let hash = self.hash_builder.hash_one(key);
221        // When choosing the shard, rotate the hash bits usize::BITS / 2 so that we
222        // give preference to the bits in the middle of the hash.
223        // Internally hashbrown uses the lower bits for start of probing + the 7 highest,
224        // so by picking something else we improve the real entropy available to each hashbrown shard.
225        let shard_idx = (hash.rotate_right(usize::BITS / 2) & self.shards_mask) as usize;
226        self.shards.get(shard_idx).map(|s| (s, hash))
227    }

229    /// Reserver additional space for `additional` entries.
230    /// Note that this is counted in entries, and is not weighted.
231    pub fn reserve(&self, additional: usize) {
232        let additional_per_shard =
233            additional.saturating_add(self.shards.len() - 1) / self.shards.len();
234        for s in &*self.shards {
235            s.write().reserve(additional_per_shard);
236        }
237    }

289    /// Inserts an item in the cache, but _only_ if an entry with key `key` already exists.
290    /// If `soft` is set, the replace operation won't affect the "hotness" of the entry,
291    /// even if the value is replaced.

293    /// Returns `Ok` if the entry was admitted and `Err(_)` if it wasn't.
294    pub fn replace_with_lifecycle(
295        &self,
296        key: Key,
297        value: Val,
298        soft: bool,
299    ) -> Result<L::RequestState, (Key, Val)> {
300        let mut lcs = self.lifecycle.begin_request();
301        let (shard, hash) = self.shard_for(&key).unwrap();
302        shard
303            .write()
304            .insert(&mut lcs, hash, key, value, InsertStrategy::Replace { soft })?;
305        Ok(lcs)

329        let (shard, hash) = self.shard_for(&key).unwrap();
330        let result = shard

379    /// Gets an item from the cache with key `key` .
380    ///
381    /// If the corresponding value isn't present in the cache, this functions returns a guard
382    /// that can be used to insert the value once it's computed.
383    /// While the returned guard is alive, other calls with the same key using the
384    /// `get_value_guard` or `get_or_insert` family of functions will wait until the guard
385    /// is dropped or the value is inserted.
386    ///
387    /// A `None` `timeout` means waiting forever.
388    /// A `Some(<zero>)` timeout will return a Timeout error immediately if the value is not present
389    /// and a guard is alive elsewhere.
390    pub fn get_value_or_guard<Q>(
391        &self,
392        key: &Q,
393        timeout: Option<Duration>,
394    ) -> GuardResult<'_, Key, Val, We, B, L>
395    where
396        Q: Hash + Equivalent<Key> + ToOwned<Owned = Key> + ?Sized,
397    {
398        let (shard, hash) = self.shard_for(key).unwrap();
399        if let Some(v) = shard.read().get(hash, key) {
400            return GuardResult::Value(v.clone());
401        }
402        PlaceholderGuard::join(&self.lifecycle, shard, hash, key, timeout)
403    }

405    /// Gets or inserts an item in the cache with key `key`.
406    ///
407    /// See also `get_value_or_guard` and `get_value_or_guard_async`.
408    pub fn get_or_insert_with<Q, E>(
409        &self,
410        key: &Q,
411        with: impl FnOnce() -> Result<Val, E>,
412    ) -> Result<Val, E>
413    where
414        Q: Hash + Equivalent<Key> + ToOwned<Owned = Key> + ?Sized,
415    {
416        match self.get_value_or_guard(key, None) {
417            GuardResult::Value(v) => Ok(v),
418            GuardResult::Guard(g) => {
419                let v = with()?;
420                let _ = g.insert(v.clone());
421                Ok(v)
422            }
423            GuardResult::Timeout => unsafe { unreachable_unchecked() },
424        }
425    }

427    /// Gets an item from the cache with key `key`.
428    ///
429    /// If the corresponding value isn't present in the cache, this functions returns a guard
430    /// that can be used to insert the value once it's computed.
431    /// While the returned guard is alive, other calls with the same key using the
432    /// `get_value_guard` or `get_or_insert` family of functions will wait until the guard
433    /// is dropped or the value is inserted.
434    pub async fn get_value_or_guard_async<'a, Q>(
435        &'a self,
436        key: &Q,
437    ) -> Result<Val, PlaceholderGuard<'a, Key, Val, We, B, L>>
438    where
439        Q: Hash + Equivalent<Key> + ToOwned<Owned = Key> + ?Sized,
440    {
441        let (shard, hash) = self.shard_for(key).unwrap();
442        if let Some(v) = shard.read().get(hash, key) {
443            return Ok(v.clone());
444        }
445        JoinFuture::new(&self.lifecycle, shard, hash, key).await
446    }

448    /// Gets or inserts an item in the cache with key `key`.
449    pub async fn get_or_insert_async<Q, E>(
450        &self,
451        key: &Q,
452        with: impl Future<Output = Result<Val, E>>,
453    ) -> Result<Val, E>
454    where
455        Q: Hash + Equivalent<Key> + ToOwned<Owned = Key> + ?Sized,
456    {
457        match self.get_value_or_guard_async(key).await {
458            Ok(v) => Ok(v),
459            Err(g) => {
460                let v = with.await?;
461                let _ = g.insert(v.clone());
462                Ok(v)
463            }
464        }
465    }

474/// Iterator over the items in the cache.
475///
476/// See [`Cache::iter`] for more details.
477pub struct Iter<'a, Key, Val, We, B, L> {
478    shards: &'a [RwLock<CacheShard<Key, Val, We, B, L, SharedPlaceholder<Val>>>],
479    current_shard: usize,
480    last: Option<Token>,
481}
482
483impl<Key, Val, We, B, L> Iterator for Iter<'_, Key, Val, We, B, L>
484where
485    Key: Clone,
486    Val: Clone,
487{
488    type Item = (Key, Val);
489
490    fn next(&mut self) -> Option<Self::Item> {
491        while self.current_shard < self.shards.len() {
492            let shard = &self.shards[self.current_shard];
493            let lock = shard.read();
494            if let Some((new_last, key, val)) = lock.iter_from(self.last).next() {
495                self.last = Some(new_last);
496                return Some((key.clone(), val.clone()));
497            }
498            self.last = None;
499            self.current_shard += 1;
500        }
501        None
502    }
503}

506    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
507        f.debug_struct("Iter").finish_non_exhaustive()
508    }

511/// Draining iterator for the items in the cache.
512///
513/// See [`Cache::drain`] for more details.
514pub struct Drain<'a, Key, Val, We, B, L> {
515    shards: &'a [RwLock<CacheShard<Key, Val, We, B, L, SharedPlaceholder<Val>>>],
516    current_shard: usize,
517    last: Option<Token>,
518}
519
520impl<Key, Val, We, B, L> Iterator for Drain<'_, Key, Val, We, B, L>
521where
522    Key: Hash + Eq,
523    We: Weighter<Key, Val>,
524    B: BuildHasher,
525    L: Lifecycle<Key, Val>,
526{
527    type Item = (Key, Val);
528
529    fn next(&mut self) -> Option<Self::Item> {
530        while self.current_shard < self.shards.len() {
531            let shard = &self.shards[self.current_shard];
532            let mut lock = shard.write();
533            if let Some((new_last, key, value)) = lock.remove_next(self.last) {
534                self.last = Some(new_last);
535                return Some((key, value));
536            }
537            self.last = None;
538            self.current_shard += 1;
539        }
540        None
541    }
542}

574impl<Key, Val> Lifecycle<Key, Val> for DefaultLifecycle<Key, Val> {
575    // Why two items?
576    // Because assuming the cache has roughly similarly weighted items,
577    // we can expect that at one or two items will be evicted per request
578    // in most cases. And we want to avoid introducing any extra
579    // overhead (e.g. a vector) for this default lifecycle.
580    type RequestState = [Option<(Key, Val)>; 2];
581
582    #[inline]
583    fn begin_request(&self) -> Self::RequestState {
584        [None, None]
585    }
586
587    #[inline]
588    fn on_evict(&self, state: &mut Self::RequestState, key: Key, val: Val) {
589        if std::mem::needs_drop::<(Key, Val)>() {
590            if state[0].is_none() {
591                state[0] = Some((key, val));
592            } else if state[1].is_none() {
593                state[1] = Some((key, val));
594            }
595        }
596    }
597}
