157    /// Convert GraphQL system operation metadata to a SystemOperation object.
158    fn convert_system_operation(
159        system_op: block::BlockBlockBlockBodyTransactionMetadataOperationSystemOperation,
160    ) -> Result<SystemOperation, ConversionError> {
161        match system_op.system_operation_type.as_str() {
162            "Transfer" => {
163                let transfer = system_op.transfer.ok_or_else(|| {
164                    ConversionError::UnexpectedCertificateType(
165                        "Missing transfer metadata for Transfer operation".to_string(),
166                    )
167                })?;
168                Ok(SystemOperation::Transfer {
169                    owner: transfer.owner,
170                    recipient: Account {
171                        chain_id: transfer.recipient.chain_id,
172                        owner: transfer.recipient.owner,
173                    },
174                    amount: transfer.amount,
175                })
176            }
177            "Claim" => {
178                let claim = system_op.claim.ok_or_else(|| {
179                    ConversionError::UnexpectedCertificateType(
180                        "Missing claim metadata for Claim operation".to_string(),
181                    )
182                })?;
183                Ok(SystemOperation::Claim {
184                    owner: claim.owner,
185                    target_id: claim.target_id,
186                    recipient: Account {
187                        chain_id: claim.recipient.chain_id,
188                        owner: claim.recipient.owner,
189                    },
190                    amount: claim.amount,
191                })
192            }
193            "OpenChain" => {
194                let open_chain = system_op.open_chain.ok_or_else(|| {
195                    ConversionError::UnexpectedCertificateType(
196                        "Missing open_chain metadata for OpenChain operation".to_string(),
197                    )
198                })?;
199
200                let ownership: ChainOwnership =
201                    serde_json::from_str(&open_chain.ownership.ownership_json)
202                        .map_err(ConversionError::Serde)?;
203
204                let application_permissions: ApplicationPermissions =
205                    serde_json::from_str(&open_chain.application_permissions.permissions_json)
206                        .map_err(ConversionError::Serde)?;
207
208                Ok(SystemOperation::OpenChain(OpenChainConfig {
209                    ownership,
210                    balance: open_chain.balance,
211                    application_permissions,
212                }))
213            }
214            "CloseChain" => Ok(SystemOperation::CloseChain),
215            "ChangeOwnership" => {
216                let change_ownership = system_op.change_ownership.ok_or_else(|| {
217                    ConversionError::UnexpectedCertificateType(
218                        "Missing change_ownership metadata for ChangeOwnership operation"
219                            .to_string(),
220                    )
221                })?;
222
223                let timeout_config = TimeoutConfig {
224                    fast_round_duration: change_ownership
225                        .timeout_config
226                        .fast_round_ms
227                        .as_ref()
228                        .map(|s| {
229                            s.parse::<u64>().map_err(|_| {
230                                ConversionError::UnexpectedCertificateType(
231                                    "Invalid fast_round_ms value".to_string(),
232                                )
233                            })
234                        })
235                        .transpose()?
236                        .map(|ms| TimeDelta::from_micros(ms * 1000)),
237                    base_timeout: TimeDelta::from_micros(
238                        change_ownership
239                            .timeout_config
240                            .base_timeout_ms
241                            .parse::<u64>()
242                            .map_err(|_| {
243                                ConversionError::UnexpectedCertificateType(
244                                    "Invalid base_timeout_ms value".to_string(),
245                                )
246                            })?
247                            * 1000,
248                    ),
249                    timeout_increment: TimeDelta::from_micros(
250                        change_ownership
251                            .timeout_config
252                            .timeout_increment_ms
253                            .parse::<u64>()
254                            .map_err(|_| {
255                                ConversionError::UnexpectedCertificateType(
256                                    "Invalid timeout_increment_ms value".to_string(),
257                                )
258                            })?
259                            * 1000,
260                    ),
261                    fallback_duration: TimeDelta::from_micros(
262                        change_ownership
263                            .timeout_config
264                            .fallback_duration_ms
265                            .parse::<u64>()
266                            .map_err(|_| {
267                                ConversionError::UnexpectedCertificateType(
268                                    "Invalid fallback_duration_ms value".to_string(),
269                                )
270                            })?
271                            * 1000,
272                    ),
273                };
274
275                Ok(SystemOperation::ChangeOwnership {
276                    super_owners: change_ownership.super_owners,
277                    owners: change_ownership
278                        .owners
279                        .into_iter()
280                        .map(|ow| {
281                            let weight = ow.weight.parse::<u64>().map_err(|_| {
282                                ConversionError::UnexpectedCertificateType(
283                                    "Invalid owner weight value".to_string(),
284                                )
285                            })?;
286                            Ok((ow.owner, weight))
287                        })
288                        .collect::<Result<Vec<_>, ConversionError>>()?,
289                    first_leader: change_ownership.first_leader,
290                    multi_leader_rounds: change_ownership.multi_leader_rounds as u32,
291                    open_multi_leader_rounds: change_ownership.open_multi_leader_rounds,
292                    timeout_config,
293                })
294            }
295            "ChangeApplicationPermissions" => {
296                let change_permissions =
297                    system_op.change_application_permissions.ok_or_else(|| {
298                        ConversionError::UnexpectedCertificateType(
299                            "Missing change_application_permissions metadata".to_string(),
300                        )
301                    })?;
302
303                let permissions: ApplicationPermissions =
304                    serde_json::from_str(&change_permissions.permissions.permissions_json)
305                        .map_err(ConversionError::Serde)?;
306
307                Ok(SystemOperation::ChangeApplicationPermissions(permissions))
308            }
309            "PublishModule" => {
310                let publish_module = system_op.publish_module.ok_or_else(|| {
311                    ConversionError::UnexpectedCertificateType(
312                        "Missing publish_module metadata for PublishModule operation".to_string(),
313                    )
314                })?;
315
316                let module_id: ModuleId = publish_module.module_id.parse().map_err(|_| {
317                    ConversionError::UnexpectedCertificateType(
318                        "Invalid module_id format".to_string(),
319                    )
320                })?;
321
322                Ok(SystemOperation::PublishModule { module_id })
323            }
324            "PublishDataBlob" => {
325                let publish_data_blob = system_op.publish_data_blob.ok_or_else(|| {
326                    ConversionError::UnexpectedCertificateType(
327                        "Missing publish_data_blob metadata".to_string(),
328                    )
329                })?;
330                Ok(SystemOperation::PublishDataBlob {
331                    blob_hash: publish_data_blob.blob_hash,
332                })
333            }
334            "VerifyBlob" => {
335                let verify_blob = system_op.verify_blob.ok_or_else(|| {
336                    ConversionError::UnexpectedCertificateType(
337                        "Missing verify_blob metadata for VerifyBlob operation".to_string(),
338                    )
339                })?;
340                let blob_id: BlobId = verify_blob.blob_id.parse().map_err(|_| {
341                    ConversionError::UnexpectedCertificateType("Invalid blob_id format".to_string())
342                })?;
343                Ok(SystemOperation::VerifyBlob { blob_id })
344            }
345            "CreateApplication" => {
346                let create_application = system_op.create_application.ok_or_else(|| {
347                    ConversionError::UnexpectedCertificateType(
348                        "Missing create_application metadata".to_string(),
349                    )
350                })?;
351
352                let module_id: ModuleId = create_application.module_id.parse().map_err(|_| {
353                    ConversionError::UnexpectedCertificateType(
354                        "Invalid module_id format".to_string(),
355                    )
356                })?;
357
358                let parameters = hex::decode(create_application.parameters_hex).map_err(|_| {
359                    ConversionError::UnexpectedCertificateType(
360                        "Invalid hex in parameters_hex".to_string(),
361                    )
362                })?;
363
364                let instantiation_argument =
365                    hex::decode(create_application.instantiation_argument_hex).map_err(|_| {
366                        ConversionError::UnexpectedCertificateType(
367                            "Invalid hex in instantiation_argument_hex".to_string(),
368                        )
369                    })?;
370
371                let required_application_ids = create_application
372                    .required_application_ids
373                    .into_iter()
374                    .map(|id| {
375                        id.parse::<RealApplicationId>().map_err(|_| {
376                            ConversionError::UnexpectedCertificateType(
377                                "Invalid required_application_id format".to_string(),
378                            )
379                        })
380                    })
381                    .collect::<Result<Vec<_>, _>>()?;
382
383                Ok(SystemOperation::CreateApplication {
384                    module_id,
385                    parameters,
386                    instantiation_argument,
387                    required_application_ids,
388                })
389            }
390            "Admin" => {
391                let admin = system_op.admin.ok_or_else(|| {
392                    ConversionError::UnexpectedCertificateType(
393                        "Missing admin metadata for Admin operation".to_string(),
394                    )
395                })?;
396
397                let admin_op = match admin.admin_operation_type.as_str() {
398                    "PublishCommitteeBlob" => {
399                        let blob_hash = admin.blob_hash.ok_or_else(|| {
400                            ConversionError::UnexpectedCertificateType(
401                                "Missing blob_hash for PublishCommitteeBlob".to_string(),
402                            )
403                        })?;
404                        AdminOperation::PublishCommitteeBlob { blob_hash }
405                    }
406                    "CreateCommittee" => {
407                        let epoch_val = admin.epoch.ok_or_else(|| {
408                            ConversionError::UnexpectedCertificateType(
409                                "Missing epoch for CreateCommittee".to_string(),
410                            )
411                        })?;
412                        let epoch = Epoch(epoch_val as u32);
413                        let blob_hash = admin.blob_hash.ok_or_else(|| {
414                            ConversionError::UnexpectedCertificateType(
415                                "Missing blob_hash for CreateCommittee".to_string(),
416                            )
417                        })?;
418                        AdminOperation::CreateCommittee { epoch, blob_hash }
419                    }
420                    "RemoveCommittee" => {
421                        let epoch_val = admin.epoch.ok_or_else(|| {
422                            ConversionError::UnexpectedCertificateType(
423                                "Missing epoch for RemoveCommittee".to_string(),
424                            )
425                        })?;
426                        let epoch = Epoch(epoch_val as u32);
427                        AdminOperation::RemoveCommittee { epoch }
428                    }
429                    _ => {
430                        return Err(ConversionError::UnexpectedCertificateType(format!(
431                            "Unknown admin operation type: {}",
432                            admin.admin_operation_type
433                        )));
434                    }
435                };
436
437                Ok(SystemOperation::Admin(admin_op))
438            }
439            "ProcessNewEpoch" => {
440                let epoch_val = system_op.epoch.ok_or_else(|| {
441                    ConversionError::UnexpectedCertificateType(
442                        "Missing epoch for ProcessNewEpoch operation".to_string(),
443                    )
444                })?;
445                Ok(SystemOperation::ProcessNewEpoch(Epoch(epoch_val as u32)))
446            }
447            "ProcessRemovedEpoch" => {
448                let epoch_val = system_op.epoch.ok_or_else(|| {
449                    ConversionError::UnexpectedCertificateType(
450                        "Missing epoch for ProcessRemovedEpoch operation".to_string(),
451                    )
452                })?;
453                Ok(SystemOperation::ProcessRemovedEpoch(Epoch(
454                    epoch_val as u32,
455                )))
456            }
457            "UpdateStreams" => {
458                let update_streams = system_op.update_streams.ok_or_else(|| {
459                    ConversionError::UnexpectedCertificateType(
460                        "Missing update_streams metadata for UpdateStreams operation".to_string(),
461                    )
462                })?;
463
464                let streams = update_streams
465                    .into_iter()
466                    .map(|stream| {
467                        let stream_id_parsed: StreamId =
468                            stream.stream_id.parse().map_err(|_| {
469                                ConversionError::UnexpectedCertificateType(
470                                    "Invalid stream_id format".to_string(),
471                                )
472                            })?;
473                        Ok((stream.chain_id, stream_id_parsed, stream.next_index as u32))
474                    })
475                    .collect::<Result<Vec<_>, ConversionError>>()?;
476
477                Ok(SystemOperation::UpdateStreams(streams))
478            }
479            _ => Err(ConversionError::UnexpectedCertificateType(format!(
480                "Unknown system operation type: {}",
481                system_op.system_operation_type
482            ))),
483        }
484    }

486    /// Convert GraphQL transaction metadata to a Transaction object.
487    fn convert_transaction_metadata(
488        metadata: block::BlockBlockBlockBodyTransactionMetadata,
489    ) -> Result<Transaction, ConversionError> {
490        match metadata.transaction_type.as_str() {
491            "ReceiveMessages" => {
492                let incoming_bundle = metadata.incoming_bundle.ok_or_else(|| {
493                    ConversionError::UnexpectedCertificateType(
494                        "Missing incoming_bundle for ReceiveMessages transaction".to_string(),
495                    )
496                })?;
497
498                let bundle = IncomingBundle {
499                    origin: incoming_bundle.origin,
500                    bundle: MessageBundle {
501                        height: incoming_bundle.bundle.height,
502                        timestamp: incoming_bundle.bundle.timestamp,
503                        certificate_hash: incoming_bundle.bundle.certificate_hash,
504                        transaction_index: incoming_bundle.bundle.transaction_index as u32,
505                        messages: incoming_bundle
506                            .bundle
507                            .messages
508                            .into_iter()
509                            .map(|msg| linera_chain::data_types::PostedMessage {
510                                authenticated_owner: msg.authenticated_owner,
511                                grant: msg.grant,
512                                refund_grant_to: msg.refund_grant_to.map(|rgt| Account {
513                                    chain_id: rgt.chain_id,
514                                    owner: rgt.owner,
515                                }),
516                                kind: msg.kind,
517                                index: msg.index as u32,
518                                message: msg.message,
519                            })
520                            .collect(),
521                    },
522                    action: incoming_bundle.action,
523                };
524
525                Ok(Transaction::ReceiveMessages(bundle))
526            }
527            "ExecuteOperation" => {
528                let graphql_operation = metadata.operation.ok_or_else(|| {
529                    ConversionError::UnexpectedCertificateType(
530                        "Missing operation for ExecuteOperation transaction".to_string(),
531                    )
532                })?;
533
534                let operation = match graphql_operation.operation_type.as_str() {
535                    "System" => {
536                        let system_op = graphql_operation.system_operation.ok_or_else(|| {
537                            ConversionError::UnexpectedCertificateType(
538                                "Missing system_operation for System operation".to_string(),
539                            )
540                        })?;
541
542                        let sys_op = convert_system_operation(system_op)?;
543                        Operation::System(Box::new(sys_op))
544                    }
545                    "User" => {
546                        let application_id = graphql_operation.application_id.ok_or_else(|| {
547                            ConversionError::UnexpectedCertificateType(
548                                "Missing application_id for User operation".to_string(),
549                            )
550                        })?;
551
552                        let bytes_hex = graphql_operation.user_bytes_hex.ok_or_else(|| {
553                            ConversionError::UnexpectedCertificateType(
554                                "Missing user_bytes_hex for User operation".to_string(),
555                            )
556                        })?;
557
558                        // Convert hex string to bytes
559                        let bytes = hex::decode(bytes_hex).map_err(|_| {
560                            ConversionError::UnexpectedCertificateType(
561                                "Invalid hex in user_bytes_hex".to_string(),
562                            )
563                        })?;
564
565                        Operation::User {
566                            application_id: application_id.parse().map_err(|_| {
567                                ConversionError::UnexpectedCertificateType(
568                                    "Invalid application_id format".to_string(),
569                                )
570                            })?,
571                            bytes,
572                        }
573                    }
574                    _ => {
575                        return Err(ConversionError::UnexpectedCertificateType(format!(
576                            "Unknown operation type: {}",
577                            graphql_operation.operation_type
578                        )));
579                    }
580                };
581
582                Ok(Transaction::ExecuteOperation(operation))
583            }
584            _ => Err(ConversionError::UnexpectedCertificateType(format!(
585                "Unknown transaction type: {}",
586                metadata.transaction_type
587            ))),
588        }
589    }

615    impl TryFrom<block::BlockBlockBlock> for Block {
616        type Error = ConversionError;
617
618        fn try_from(val: block::BlockBlockBlock) -> Result<Self, Self::Error> {
619            let block::BlockBlockBlock { header, body } = val;
620            let block::BlockBlockBlockHeader {
621                chain_id,
622                epoch,
623                height,
624                timestamp,
625                authenticated_owner,
626                previous_block_hash,
627                state_hash,
628                transactions_hash,
629                messages_hash,
630                previous_message_blocks_hash,
631                previous_event_blocks_hash,
632                oracle_responses_hash,
633                events_hash,
634                blobs_hash,
635                operation_results_hash,
636            } = header;
637            let block::BlockBlockBlockBody {
638                messages,
639                previous_message_blocks,
640                previous_event_blocks,
641                oracle_responses,
642                events,
643                blobs,
644                operation_results,
645                transaction_metadata,
646            } = body;
647
648            let block_header = BlockHeader {
649                chain_id,
650                epoch,
651                height,
652                timestamp,
653                authenticated_owner,
654                previous_block_hash,
655                state_hash,
656                transactions_hash,
657                messages_hash,
658                previous_message_blocks_hash,
659                previous_event_blocks_hash,
660                oracle_responses_hash,
661                events_hash,
662                blobs_hash,
663                operation_results_hash,
664            };
665
666            // Convert GraphQL transaction metadata to Transaction objects
667            let transactions = transaction_metadata
668                .into_iter()
669                .map(convert_transaction_metadata)
670                .collect::<Result<Vec<_>, _>>()?;
671
672            let block_body = BlockBody {
673                transactions,
674                messages: messages
675                    .into_iter()
676                    .map(|messages| messages.into_iter().map(Into::into).collect())
677                    .collect::<Vec<Vec<_>>>(),
678                previous_message_blocks: serde_json::from_value(previous_message_blocks)
679                    .map_err(ConversionError::Serde)?,
680                previous_event_blocks: serde_json::from_value(previous_event_blocks)
681                    .map_err(ConversionError::Serde)?,
682                oracle_responses: oracle_responses.into_iter().collect(),
683                events: events
684                    .into_iter()
685                    .map(|events| events.into_iter().map(Into::into).collect())
686                    .collect(),
687                blobs: blobs
688                    .into_iter()
689                    .map(|blobs| blobs.into_iter().collect())
690                    .collect(),
691                operation_results,
692            };
693
694            Ok(Block {
695                header: block_header,
696                body: block_body,
697            })
698        }
699    }

720    impl TryFrom<block::BlockBlock> for ConfirmedBlock {
721        type Error = ConversionError;
722
723        fn try_from(val: block::BlockBlock) -> Result<Self, Self::Error> {
724            match (val.status.as_str(), val.block) {
725                ("confirmed", block) => Ok(ConfirmedBlock::new(block.try_into()?)),
726                _ => Err(ConversionError::UnexpectedCertificateType(val.status)),
727            }
728        }
