157    /// Convert GraphQL system operation metadata to a SystemOperation object.
158    fn convert_system_operation(
159        system_op: block::BlockBlockBlockBodyTransactionMetadataOperationSystemOperation,
160    ) -> Result<SystemOperation, ConversionError> {
161        match system_op.system_operation_type.as_str() {
162            "Transfer" => {
163                let transfer = system_op.transfer.ok_or_else(|| {
164                    ConversionError::UnexpectedCertificateType(
165                        "Missing transfer metadata for Transfer operation".to_string(),
166                    )
167                })?;
168                Ok(SystemOperation::Transfer {
169                    owner: transfer.owner,
170                    recipient: Account {
171                        chain_id: transfer.recipient.chain_id,
172                        owner: transfer.recipient.owner,
173                    },
174                    amount: transfer.amount,
175                })
176            }
177            "Claim" => {
178                let claim = system_op.claim.ok_or_else(|| {
179                    ConversionError::UnexpectedCertificateType(
180                        "Missing claim metadata for Claim operation".to_string(),
181                    )
182                })?;
183                Ok(SystemOperation::Claim {
184                    owner: claim.owner,
185                    target_id: claim.target_id,
186                    recipient: Account {
187                        chain_id: claim.recipient.chain_id,
188                        owner: claim.recipient.owner,
189                    },
190                    amount: claim.amount,
191                })
192            }
193            "OpenChain" => {
194                let open_chain = system_op.open_chain.ok_or_else(|| {
195                    ConversionError::UnexpectedCertificateType(
196                        "Missing open_chain metadata for OpenChain operation".to_string(),
197                    )
198                })?;
199
200                let ownership: ChainOwnership =
201                    serde_json::from_str(&open_chain.ownership.ownership_json)
202                        .map_err(ConversionError::Serde)?;
203
204                let application_permissions: ApplicationPermissions =
205                    serde_json::from_str(&open_chain.application_permissions.permissions_json)
206                        .map_err(ConversionError::Serde)?;
207
208                Ok(SystemOperation::OpenChain(OpenChainConfig {
209                    ownership,
210                    balance: open_chain.balance,
211                    application_permissions,
212                }))
213            }
214            "CloseChain" => Ok(SystemOperation::CloseChain),
215            "ChangeOwnership" => {
216                let change_ownership = system_op.change_ownership.ok_or_else(|| {
217                    ConversionError::UnexpectedCertificateType(
218                        "Missing change_ownership metadata for ChangeOwnership operation"
219                            .to_string(),
220                    )
221                })?;
222
223                let timeout_config = TimeoutConfig {
224                    fast_round_duration: change_ownership
225                        .timeout_config
226                        .fast_round_ms
227                        .as_ref()
228                        .map(|s| {
229                            s.parse::<u64>().map_err(|_| {
230                                ConversionError::UnexpectedCertificateType(
231                                    "Invalid fast_round_ms value".to_string(),
232                                )
233                            })
234                        })
235                        .transpose()?
236                        .map(|ms| TimeDelta::from_micros(ms * 1000)),
237                    base_timeout: TimeDelta::from_micros(
238                        change_ownership
239                            .timeout_config
240                            .base_timeout_ms
241                            .parse::<u64>()
242                            .map_err(|_| {
243                                ConversionError::UnexpectedCertificateType(
244                                    "Invalid base_timeout_ms value".to_string(),
245                                )
246                            })?
247                            * 1000,
248                    ),
249                    timeout_increment: TimeDelta::from_micros(
250                        change_ownership
251                            .timeout_config
252                            .timeout_increment_ms
253                            .parse::<u64>()
254                            .map_err(|_| {
255                                ConversionError::UnexpectedCertificateType(
256                                    "Invalid timeout_increment_ms value".to_string(),
257                                )
258                            })?
259                            * 1000,
260                    ),
261                    fallback_duration: TimeDelta::from_micros(
262                        change_ownership
263                            .timeout_config
264                            .fallback_duration_ms
265                            .parse::<u64>()
266                            .map_err(|_| {
267                                ConversionError::UnexpectedCertificateType(
268                                    "Invalid fallback_duration_ms value".to_string(),
269                                )
270                            })?
271                            * 1000,
272                    ),
273                };
274
275                Ok(SystemOperation::ChangeOwnership {
276                    super_owners: change_ownership.super_owners,
277                    owners: change_ownership
278                        .owners
279                        .into_iter()
280                        .map(|ow| {
281                            let weight = ow.weight.parse::<u64>().map_err(|_| {
282                                ConversionError::UnexpectedCertificateType(
283                                    "Invalid owner weight value".to_string(),
284                                )
285                            })?;
286                            Ok((ow.owner, weight))
287                        })
288                        .collect::<Result<Vec<_>, ConversionError>>()?,
289                    multi_leader_rounds: change_ownership.multi_leader_rounds as u32,
290                    open_multi_leader_rounds: change_ownership.open_multi_leader_rounds,
291                    timeout_config,
292                })
293            }
294            "ChangeApplicationPermissions" => {
295                let change_permissions =
296                    system_op.change_application_permissions.ok_or_else(|| {
297                        ConversionError::UnexpectedCertificateType(
298                            "Missing change_application_permissions metadata".to_string(),
299                        )
300                    })?;
301
302                let permissions: ApplicationPermissions =
303                    serde_json::from_str(&change_permissions.permissions.permissions_json)
304                        .map_err(ConversionError::Serde)?;
305
306                Ok(SystemOperation::ChangeApplicationPermissions(permissions))
307            }
308            "PublishModule" => {
309                let publish_module = system_op.publish_module.ok_or_else(|| {
310                    ConversionError::UnexpectedCertificateType(
311                        "Missing publish_module metadata for PublishModule operation".to_string(),
312                    )
313                })?;
314
315                let module_id: ModuleId = publish_module.module_id.parse().map_err(|_| {
316                    ConversionError::UnexpectedCertificateType(
317                        "Invalid module_id format".to_string(),
318                    )
319                })?;
320
321                Ok(SystemOperation::PublishModule { module_id })
322            }
323            "PublishDataBlob" => {
324                let publish_data_blob = system_op.publish_data_blob.ok_or_else(|| {
325                    ConversionError::UnexpectedCertificateType(
326                        "Missing publish_data_blob metadata".to_string(),
327                    )
328                })?;
329                Ok(SystemOperation::PublishDataBlob {
330                    blob_hash: publish_data_blob.blob_hash,
331                })
332            }
333            "VerifyBlob" => {
334                let verify_blob = system_op.verify_blob.ok_or_else(|| {
335                    ConversionError::UnexpectedCertificateType(
336                        "Missing verify_blob metadata for VerifyBlob operation".to_string(),
337                    )
338                })?;
339                let blob_id: BlobId = verify_blob.blob_id.parse().map_err(|_| {
340                    ConversionError::UnexpectedCertificateType("Invalid blob_id format".to_string())
341                })?;
342                Ok(SystemOperation::VerifyBlob { blob_id })
343            }
344            "CreateApplication" => {
345                let create_application = system_op.create_application.ok_or_else(|| {
346                    ConversionError::UnexpectedCertificateType(
347                        "Missing create_application metadata".to_string(),
348                    )
349                })?;
350
351                let module_id: ModuleId = create_application.module_id.parse().map_err(|_| {
352                    ConversionError::UnexpectedCertificateType(
353                        "Invalid module_id format".to_string(),
354                    )
355                })?;
356
357                let parameters = hex::decode(create_application.parameters_hex).map_err(|_| {
358                    ConversionError::UnexpectedCertificateType(
359                        "Invalid hex in parameters_hex".to_string(),
360                    )
361                })?;
362
363                let instantiation_argument =
364                    hex::decode(create_application.instantiation_argument_hex).map_err(|_| {
365                        ConversionError::UnexpectedCertificateType(
366                            "Invalid hex in instantiation_argument_hex".to_string(),
367                        )
368                    })?;
369
370                let required_application_ids = create_application
371                    .required_application_ids
372                    .into_iter()
373                    .map(|id| {
374                        id.parse::<RealApplicationId>().map_err(|_| {
375                            ConversionError::UnexpectedCertificateType(
376                                "Invalid required_application_id format".to_string(),
377                            )
378                        })
379                    })
380                    .collect::<Result<Vec<_>, _>>()?;
381
382                Ok(SystemOperation::CreateApplication {
383                    module_id,
384                    parameters,
385                    instantiation_argument,
386                    required_application_ids,
387                })
388            }
389            "Admin" => {
390                let admin = system_op.admin.ok_or_else(|| {
391                    ConversionError::UnexpectedCertificateType(
392                        "Missing admin metadata for Admin operation".to_string(),
393                    )
394                })?;
395
396                let admin_op = match admin.admin_operation_type.as_str() {
397                    "PublishCommitteeBlob" => {
398                        let blob_hash = admin.blob_hash.ok_or_else(|| {
399                            ConversionError::UnexpectedCertificateType(
400                                "Missing blob_hash for PublishCommitteeBlob".to_string(),
401                            )
402                        })?;
403                        AdminOperation::PublishCommitteeBlob { blob_hash }
404                    }
405                    "CreateCommittee" => {
406                        let epoch_val = admin.epoch.ok_or_else(|| {
407                            ConversionError::UnexpectedCertificateType(
408                                "Missing epoch for CreateCommittee".to_string(),
409                            )
410                        })?;
411                        let epoch = Epoch(epoch_val as u32);
412                        let blob_hash = admin.blob_hash.ok_or_else(|| {
413                            ConversionError::UnexpectedCertificateType(
414                                "Missing blob_hash for CreateCommittee".to_string(),
415                            )
416                        })?;
417                        AdminOperation::CreateCommittee { epoch, blob_hash }
418                    }
419                    "RemoveCommittee" => {
420                        let epoch_val = admin.epoch.ok_or_else(|| {
421                            ConversionError::UnexpectedCertificateType(
422                                "Missing epoch for RemoveCommittee".to_string(),
423                            )
424                        })?;
425                        let epoch = Epoch(epoch_val as u32);
426                        AdminOperation::RemoveCommittee { epoch }
427                    }
428                    _ => {
429                        return Err(ConversionError::UnexpectedCertificateType(format!(
430                            "Unknown admin operation type: {}",
431                            admin.admin_operation_type
432                        )));
433                    }
434                };
435
436                Ok(SystemOperation::Admin(admin_op))
437            }
438            "ProcessNewEpoch" => {
439                let epoch_val = system_op.epoch.ok_or_else(|| {
440                    ConversionError::UnexpectedCertificateType(
441                        "Missing epoch for ProcessNewEpoch operation".to_string(),
442                    )
443                })?;
444                Ok(SystemOperation::ProcessNewEpoch(Epoch(epoch_val as u32)))
445            }
446            "ProcessRemovedEpoch" => {
447                let epoch_val = system_op.epoch.ok_or_else(|| {
448                    ConversionError::UnexpectedCertificateType(
449                        "Missing epoch for ProcessRemovedEpoch operation".to_string(),
450                    )
451                })?;
452                Ok(SystemOperation::ProcessRemovedEpoch(Epoch(
453                    epoch_val as u32,
454                )))
455            }
456            "UpdateStreams" => {
457                let update_streams = system_op.update_streams.ok_or_else(|| {
458                    ConversionError::UnexpectedCertificateType(
459                        "Missing update_streams metadata for UpdateStreams operation".to_string(),
460                    )
461                })?;
462
463                let streams = update_streams
464                    .into_iter()
465                    .map(|stream| {
466                        let stream_id_parsed: StreamId =
467                            stream.stream_id.parse().map_err(|_| {
468                                ConversionError::UnexpectedCertificateType(
469                                    "Invalid stream_id format".to_string(),
470                                )
471                            })?;
472                        Ok((stream.chain_id, stream_id_parsed, stream.next_index as u32))
473                    })
474                    .collect::<Result<Vec<_>, ConversionError>>()?;
475
476                Ok(SystemOperation::UpdateStreams(streams))
477            }
478            _ => Err(ConversionError::UnexpectedCertificateType(format!(
479                "Unknown system operation type: {}",
480                system_op.system_operation_type
481            ))),
482        }
483    }

485    /// Convert GraphQL transaction metadata to a Transaction object.
486    fn convert_transaction_metadata(
487        metadata: block::BlockBlockBlockBodyTransactionMetadata,
488    ) -> Result<Transaction, ConversionError> {
489        match metadata.transaction_type.as_str() {
490            "ReceiveMessages" => {
491                let incoming_bundle = metadata.incoming_bundle.ok_or_else(|| {
492                    ConversionError::UnexpectedCertificateType(
493                        "Missing incoming_bundle for ReceiveMessages transaction".to_string(),
494                    )
495                })?;
496
497                let bundle = IncomingBundle {
498                    origin: incoming_bundle.origin,
499                    bundle: MessageBundle {
500                        height: incoming_bundle.bundle.height,
501                        timestamp: incoming_bundle.bundle.timestamp,
502                        certificate_hash: incoming_bundle.bundle.certificate_hash,
503                        transaction_index: incoming_bundle.bundle.transaction_index as u32,
504                        messages: incoming_bundle
505                            .bundle
506                            .messages
507                            .into_iter()
508                            .map(|msg| linera_chain::data_types::PostedMessage {
509                                authenticated_owner: msg.authenticated_owner,
510                                grant: msg.grant,
511                                refund_grant_to: msg.refund_grant_to.map(|rgt| Account {
512                                    chain_id: rgt.chain_id,
513                                    owner: rgt.owner,
514                                }),
515                                kind: msg.kind,
516                                index: msg.index as u32,
517                                message: msg.message,
518                            })
519                            .collect(),
520                    },
521                    action: incoming_bundle.action,
522                };
523
524                Ok(Transaction::ReceiveMessages(bundle))
525            }
526            "ExecuteOperation" => {
527                let graphql_operation = metadata.operation.ok_or_else(|| {
528                    ConversionError::UnexpectedCertificateType(
529                        "Missing operation for ExecuteOperation transaction".to_string(),
530                    )
531                })?;
532
533                let operation = match graphql_operation.operation_type.as_str() {
534                    "System" => {
535                        let system_op = graphql_operation.system_operation.ok_or_else(|| {
536                            ConversionError::UnexpectedCertificateType(
537                                "Missing system_operation for System operation".to_string(),
538                            )
539                        })?;
540
541                        let sys_op = convert_system_operation(system_op)?;
542                        Operation::System(Box::new(sys_op))
543                    }
544                    "User" => {
545                        let application_id = graphql_operation.application_id.ok_or_else(|| {
546                            ConversionError::UnexpectedCertificateType(
547                                "Missing application_id for User operation".to_string(),
548                            )
549                        })?;
550
551                        let bytes_hex = graphql_operation.user_bytes_hex.ok_or_else(|| {
552                            ConversionError::UnexpectedCertificateType(
553                                "Missing user_bytes_hex for User operation".to_string(),
554                            )
555                        })?;
556
557                        // Convert hex string to bytes
558                        let bytes = hex::decode(bytes_hex).map_err(|_| {
559                            ConversionError::UnexpectedCertificateType(
560                                "Invalid hex in user_bytes_hex".to_string(),
561                            )
562                        })?;
563
564                        Operation::User {
565                            application_id: application_id.parse().map_err(|_| {
566                                ConversionError::UnexpectedCertificateType(
567                                    "Invalid application_id format".to_string(),
568                                )
569                            })?,
570                            bytes,
571                        }
572                    }
573                    _ => {
574                        return Err(ConversionError::UnexpectedCertificateType(format!(
575                            "Unknown operation type: {}",
576                            graphql_operation.operation_type
577                        )));
578                    }
579                };
580
581                Ok(Transaction::ExecuteOperation(operation))
582            }
583            _ => Err(ConversionError::UnexpectedCertificateType(format!(
584                "Unknown transaction type: {}",
585                metadata.transaction_type
586            ))),
587        }
588    }

614    impl TryFrom<block::BlockBlockBlock> for Block {
615        type Error = ConversionError;
616
617        fn try_from(val: block::BlockBlockBlock) -> Result<Self, Self::Error> {
618            let block::BlockBlockBlock { header, body } = val;
619            let block::BlockBlockBlockHeader {
620                chain_id,
621                epoch,
622                height,
623                timestamp,
624                authenticated_owner,
625                previous_block_hash,
626                state_hash,
627                transactions_hash,
628                messages_hash,
629                previous_message_blocks_hash,
630                previous_event_blocks_hash,
631                oracle_responses_hash,
632                events_hash,
633                blobs_hash,
634                operation_results_hash,
635            } = header;
636            let block::BlockBlockBlockBody {
637                messages,
638                previous_message_blocks,
639                previous_event_blocks,
640                oracle_responses,
641                events,
642                blobs,
643                operation_results,
644                transaction_metadata,
645            } = body;
646
647            let block_header = BlockHeader {
648                chain_id,
649                epoch,
650                height,
651                timestamp,
652                authenticated_owner,
653                previous_block_hash,
654                state_hash,
655                transactions_hash,
656                messages_hash,
657                previous_message_blocks_hash,
658                previous_event_blocks_hash,
659                oracle_responses_hash,
660                events_hash,
661                blobs_hash,
662                operation_results_hash,
663            };
664
665            // Convert GraphQL transaction metadata to Transaction objects
666            let transactions = transaction_metadata
667                .into_iter()
668                .map(convert_transaction_metadata)
669                .collect::<Result<Vec<_>, _>>()?;
670
671            let block_body = BlockBody {
672                transactions,
673                messages: messages
674                    .into_iter()
675                    .map(|messages| messages.into_iter().map(Into::into).collect())
676                    .collect::<Vec<Vec<_>>>(),
677                previous_message_blocks: serde_json::from_value(previous_message_blocks)
678                    .map_err(ConversionError::Serde)?,
679                previous_event_blocks: serde_json::from_value(previous_event_blocks)
680                    .map_err(ConversionError::Serde)?,
681                oracle_responses: oracle_responses.into_iter().collect(),
682                events: events
683                    .into_iter()
684                    .map(|events| events.into_iter().map(Into::into).collect())
685                    .collect(),
686                blobs: blobs
687                    .into_iter()
688                    .map(|blobs| blobs.into_iter().collect())
689                    .collect(),
690                operation_results,
691            };
692
693            Ok(Block {
694                header: block_header,
695                body: block_body,
696            })
697        }
698    }

719    impl TryFrom<block::BlockBlock> for ConfirmedBlock {
720        type Error = ConversionError;
721
722        fn try_from(val: block::BlockBlock) -> Result<Self, Self::Error> {
723            match (val.status.as_str(), val.block) {
724                ("confirmed", block) => Ok(ConfirmedBlock::new(block.try_into()?)),
725                _ => Err(ConversionError::UnexpectedCertificateType(val.status)),
726            }
727        }
