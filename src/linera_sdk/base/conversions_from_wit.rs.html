4//! Conversions from types generated by [`wit-bindgen`] to types declared in [`linera-sdk`].
5
6use linera_base::{
7    crypto::CryptoHash,
8    data_types::{
9        Amount, ApplicationDescription, ApplicationPermissions, BlockHeight, TimeDelta, Timestamp,
10    },
11    http,
12    identifiers::{AccountOwner, ApplicationId, ChainId, DataBlobHash, ModuleId},
13    ownership::{ChainOwnership, TimeoutConfig},
14    vm::VmRuntime,
15};
16
17use crate::{
18    contract::wit::base_runtime_api as base_contract_api,
19    service::wit::base_runtime_api as base_service_api,
20};
21
22macro_rules! impl_from_wit {
23    ($wit_base_api:ident) => {
24        impl From<$wit_base_api::CryptoHash> for CryptoHash {
25            fn from(hash_value: $wit_base_api::CryptoHash) -> Self {
26                CryptoHash::from([
27                    hash_value.part1,
28                    hash_value.part2,
29                    hash_value.part3,
30                    hash_value.part4,
31                ])
32            }
33        }
34
35        impl From<$wit_base_api::DataBlobHash> for DataBlobHash {
36            fn from(hash_value: $wit_base_api::DataBlobHash) -> Self {
37                DataBlobHash(hash_value.inner0.into())
38            }
39        }
40
41        impl From<$wit_base_api::Array20> for [u8; 20] {
42            fn from(ethereum_address: $wit_base_api::Array20) -> Self {
43                let mut bytes = [0u8; 20];
44                bytes[0..8].copy_from_slice(&ethereum_address.part1.to_be_bytes());
45                bytes[8..16].copy_from_slice(&ethereum_address.part2.to_be_bytes());
46                bytes[16..20].copy_from_slice(&ethereum_address.part3.to_be_bytes()[0..4]);
47                bytes
48            }
49        }
50
51        impl From<$wit_base_api::AccountOwner> for AccountOwner {
52            fn from(account_owner: $wit_base_api::AccountOwner) -> Self {
53                match account_owner {
54                    $wit_base_api::AccountOwner::Reserved(value) => AccountOwner::Reserved(value),
55                    $wit_base_api::AccountOwner::Address32(value) => {
56                        AccountOwner::Address32(value.into())
57                    }
58                    $wit_base_api::AccountOwner::Address20(value) => {
59                        AccountOwner::Address20(value.into())
60                    }
61                }
62            }
63        }
64
65        impl From<$wit_base_api::Amount> for Amount {
66            fn from(balance: $wit_base_api::Amount) -> Self {
67                let (lower_half, upper_half) = balance.inner0;
68                let value = ((upper_half as u128) << 64) | (lower_half as u128);
69                Amount::from_attos(value)
70            }
71        }
72
73        impl From<$wit_base_api::BlockHeight> for BlockHeight {
74            fn from(block_height: $wit_base_api::BlockHeight) -> Self {
75                BlockHeight(block_height.inner0)
76            }
77        }
78
79        impl From<$wit_base_api::ChainId> for ChainId {
80            fn from(chain_id: $wit_base_api::ChainId) -> Self {
81                ChainId(chain_id.inner0.into())
82            }
83        }
84
85        impl From<$wit_base_api::ApplicationId> for ApplicationId {
86            fn from(application_id: $wit_base_api::ApplicationId) -> Self {
87                ApplicationId::new(application_id.application_description_hash.into())
88            }
89        }
90
91        impl From<$wit_base_api::Timestamp> for Timestamp {
92            fn from(timestamp: $wit_base_api::Timestamp) -> Self {
93                Timestamp::from(timestamp.inner0)
94            }
95        }
96
97        impl From<$wit_base_api::TimeDelta> for TimeDelta {
98            fn from(guest: $wit_base_api::TimeDelta) -> Self {
99                TimeDelta::from_micros(guest.inner0)
100            }
101        }
102
103        impl From<$wit_base_api::TimeoutConfig> for TimeoutConfig {
104            fn from(guest: $wit_base_api::TimeoutConfig) -> TimeoutConfig {
105                let $wit_base_api::TimeoutConfig {
106                    fast_round_duration,
107                    base_timeout,
108                    timeout_increment,
109                    fallback_duration,
110                } = guest;
111                TimeoutConfig {
112                    fast_round_duration: fast_round_duration.map(TimeDelta::from),
113                    base_timeout: base_timeout.into(),
114                    timeout_increment: timeout_increment.into(),
115                    fallback_duration: fallback_duration.into(),
116                }
117            }
118        }
119
120        impl From<$wit_base_api::ChainOwnership> for ChainOwnership {
121            fn from(guest: $wit_base_api::ChainOwnership) -> ChainOwnership {
122                let $wit_base_api::ChainOwnership {
123                    super_owners,
124                    owners,
125                    first_leader,
126                    multi_leader_rounds,
127                    open_multi_leader_rounds,
128                    timeout_config,
129                } = guest;
130                ChainOwnership {
131                    super_owners: super_owners.into_iter().map(Into::into).collect(),
132                    owners: owners
133                        .into_iter()
134                        .map(|(owner, weight)| (owner.into(), weight))
135                        .collect(),
136                    first_leader: first_leader.map(Into::into),
137                    multi_leader_rounds,
138                    open_multi_leader_rounds,
139                    timeout_config: timeout_config.into(),
140                }
141            }
142        }
143
144        impl From<$wit_base_api::ApplicationPermissions> for ApplicationPermissions {
145            fn from(guest: $wit_base_api::ApplicationPermissions) -> ApplicationPermissions {
146                let $wit_base_api::ApplicationPermissions {
147                    execute_operations,
148                    mandatory_applications,
149                    manage_chain,
150                    call_service_as_oracle,
151                    make_http_requests,
152                } = guest;
153                ApplicationPermissions {
154                    execute_operations: execute_operations
155                        .map(|apps| apps.into_iter().map(Into::into).collect()),
156                    mandatory_applications: mandatory_applications
157                        .into_iter()
158                        .map(Into::into)
159                        .collect(),
160                    manage_chain: manage_chain.into_iter().map(Into::into).collect(),
161                    call_service_as_oracle: call_service_as_oracle
162                        .map(|apps| apps.into_iter().map(Into::into).collect()),
163                    make_http_requests: make_http_requests
164                        .map(|apps| apps.into_iter().map(Into::into).collect()),
165                }
166            }
167        }
168
169        impl From<$wit_base_api::HttpResponse> for http::Response {
170            fn from(response: $wit_base_api::HttpResponse) -> http::Response {
171                http::Response {
172                    status: response.status,
173                    headers: response
174                        .headers
175                        .into_iter()
176                        .map(http::Header::from)
177                        .collect(),
178                    body: response.body,
179                }
180            }
181        }
182
183        impl From<$wit_base_api::HttpHeader> for http::Header {
184            fn from(header: $wit_base_api::HttpHeader) -> http::Header {
185                http::Header::new(header.name, header.value)
186            }
187        }
188
189        impl From<$wit_base_api::VmRuntime> for VmRuntime {
190            fn from(vm_runtime: $wit_base_api::VmRuntime) -> Self {
191                match vm_runtime {
192                    $wit_base_api::VmRuntime::Wasm => VmRuntime::Wasm,
193                    $wit_base_api::VmRuntime::Evm => VmRuntime::Evm,
194                }
195            }
196        }
197
198        impl From<$wit_base_api::ModuleId> for ModuleId {
199            fn from(module_id: $wit_base_api::ModuleId) -> Self {
200                ModuleId::new(
201                    module_id.contract_blob_hash.into(),
202                    module_id.service_blob_hash.into(),
203                    module_id.vm_runtime.into(),
204                )
205            }
206        }
207
208        impl From<$wit_base_api::ApplicationDescription> for ApplicationDescription {
209            fn from(description: $wit_base_api::ApplicationDescription) -> Self {
210                ApplicationDescription {
211                    module_id: description.module_id.into(),
212                    creator_chain_id: description.creator_chain_id.into(),
213                    block_height: description.block_height.into(),
214                    application_index: description.application_index,
215                    parameters: description.parameters,
216                    required_application_ids: description
217                        .required_application_ids
218                        .into_iter()
219                        .map(Into::into)
220                        .collect(),
221                }
222            }
223        }
224    };
225}
