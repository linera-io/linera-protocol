46/// Helper type to keep track of created promises by one of the functions.
47#[derive(Default)]
48struct PromiseRegistry<T> {
49    promises: Mutex<BTreeMap<u32, T>>,
50    id_counter: AtomicU32,
51}
52
53impl<T> PromiseRegistry<T> {
54    /// Creates a new promise tracking the internal `value`.
55    pub fn register(&self, value: T) -> u32 {
56        let id = self.id_counter.fetch_add(1, Ordering::AcqRel);
57        self.promises
58            .try_lock()
59            .expect("Unit-tests should run in a single-thread")
60            .insert(id, value);
61        id
62    }
63
64    /// Retrieves a tracked promise by its ID.
65    pub fn take(&self, id: u32) -> T {
66        self.promises
67            .try_lock()
68            .expect("Unit-tests should run in a single-thread")
69            .remove(&id)
70            .expect("Use of an invalid promise ID")
71    }
72}

74impl MockKeyValueStore {
75    /// Checks if `key` is present in the storage, returning a promise to retrieve the final
76    /// value.
77    pub(crate) fn contains_key_new(&self, key: &[u8]) -> u32 {
78        self.contains_key_promises.register(
79            self.store
80                .contains_key(key)
81                .now_or_never()
82                .expect("Memory store should never wait for anything")
83                .expect("Memory store should never fail"),
84        )
85    }
86
87    /// Returns if the key used in the respective call to [`contains_key_new`] is present.
88    pub(crate) fn contains_key_wait(&self, promise: u32) -> bool {
89        self.contains_key_promises.take(promise)
90    }
