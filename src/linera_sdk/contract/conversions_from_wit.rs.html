4//! Conversions from types generated by [`wit-bindgen`] to types declared in [`linera-sdk`].
5

14    vm::VmRuntime,
15};

18    contract_runtime_api as wit_contract_api,

22impl From<wit_contract_api::CryptoHash> for CryptoHash {
23    fn from(crypto_hash: wit_contract_api::CryptoHash) -> Self {
24        CryptoHash::from([
25            crypto_hash.part1,
26            crypto_hash.part2,
27            crypto_hash.part3,
28            crypto_hash.part4,
29        ])
30    }
31}
32
33impl From<wit_contract_api::DataBlobHash> for DataBlobHash {
34    fn from(hash_value: wit_contract_api::DataBlobHash) -> Self {
35        DataBlobHash(hash_value.inner0.into())
36    }
37}
38
39impl From<wit_contract_api::Array20> for [u8; 20] {
40    fn from(ethereum_address: wit_contract_api::Array20) -> Self {
41        let mut bytes = [0u8; 20];
42        bytes[0..8].copy_from_slice(&ethereum_address.part1.to_be_bytes());
43        bytes[8..16].copy_from_slice(&ethereum_address.part2.to_be_bytes());
44        bytes[16..20].copy_from_slice(&ethereum_address.part3.to_be_bytes()[0..4]);
45        bytes
46    }
47}

49impl From<wit_contract_api::AccountOwner> for AccountOwner {
50    fn from(account_owner: wit_contract_api::AccountOwner) -> Self {
51        match account_owner {
52            wit_contract_api::AccountOwner::Reserved(value) => AccountOwner::Reserved(value),
53            wit_contract_api::AccountOwner::Address32(value) => {
54                AccountOwner::Address32(value.into())
55            }
56            wit_contract_api::AccountOwner::Address20(value) => {
57                AccountOwner::Address20(value.into())
58            }
59        }
60    }
61}

63impl From<wit_contract_api::ModuleId> for ModuleId {
64    fn from(module_id: wit_contract_api::ModuleId) -> Self {
65        ModuleId::new(
66            module_id.contract_blob_hash.into(),
67            module_id.service_blob_hash.into(),
68            module_id.vm_runtime.into(),
69        )
70    }
71}

73impl From<wit_contract_api::VmRuntime> for VmRuntime {
74    fn from(vm_runtime: wit_contract_api::VmRuntime) -> Self {
75        match vm_runtime {
76            wit_contract_api::VmRuntime::Wasm => VmRuntime::Wasm,
77            wit_contract_api::VmRuntime::Evm => VmRuntime::Evm,
78        }
79    }
80}

94impl From<wit_contract_api::Amount> for Amount {
95    fn from(balance: wit_contract_api::Amount) -> Self {
96        let (lower_half, upper_half) = balance.inner0;
97        let value = ((upper_half as u128) << 64) | (lower_half as u128);
98        Amount::from_attos(value)
99    }
100}

102impl From<wit_contract_api::CloseChainError> for CloseChainError {
103    fn from(guest: wit_contract_api::CloseChainError) -> Self {
104        match guest {

106        }
107    }

139impl From<wit_entrypoints::GenericApplicationId> for GenericApplicationId {
140    fn from(generic_application_id: wit_entrypoints::GenericApplicationId) -> Self {
141        match generic_application_id {
142            wit_entrypoints::GenericApplicationId::System => GenericApplicationId::System,
143            wit_entrypoints::GenericApplicationId::User(application_id) => {
144                GenericApplicationId::User(application_id.into())
145            }
146        }
147    }
148}
