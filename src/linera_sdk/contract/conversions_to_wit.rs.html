21impl From<CryptoHash> for wit_contract_api::CryptoHash {
22    fn from(crypto_hash: CryptoHash) -> Self {
23        let parts = <[u64; 4]>::from(crypto_hash);
24
25        wit_contract_api::CryptoHash {
26            part1: parts[0],
27            part2: parts[1],
28            part3: parts[2],
29            part4: parts[3],
30        }
31    }
32}
33
34impl From<DataBlobHash> for wit_contract_api::DataBlobHash {
35    fn from(hash_value: DataBlobHash) -> Self {
36        wit_contract_api::DataBlobHash {
37            inner0: hash_value.0.into(),
38        }
39    }
40}
41
42impl From<ChainId> for wit_contract_api::CryptoHash {
43    fn from(chain_id: ChainId) -> Self {
44        chain_id.0.into()
45    }
46}
47
48impl From<[u8; 20]> for wit_contract_api::Array20 {
49    fn from(bytes: [u8; 20]) -> Self {
50        wit_contract_api::Array20 {
51            part1: u64::from_be_bytes(bytes[0..8].try_into().unwrap()),
52            part2: u64::from_be_bytes(bytes[8..16].try_into().unwrap()),
53            part3: (u32::from_be_bytes(bytes[16..20].try_into().unwrap()) as u64) << 32,
54        }
55    }
56}

58impl From<Amount> for wit_contract_api::Amount {
59    fn from(host: Amount) -> Self {
60        wit_contract_api::Amount {
61            inner0: (host.lower_half(), host.upper_half()),
62        }
63    }
64}

66impl From<Account> for wit_contract_api::Account {

75impl From<AccountOwner> for wit_contract_api::AccountOwner {
76    fn from(account_owner: AccountOwner) -> Self {
77        match account_owner {
78            AccountOwner::Reserved(value) => wit_contract_api::AccountOwner::Reserved(value),
79            AccountOwner::Address32(owner) => {
80                wit_contract_api::AccountOwner::Address32(owner.into())
81            }
82            AccountOwner::Address20(owner) => {
83                wit_contract_api::AccountOwner::Address20(owner.into())
84            }
85        }
86    }
87}

175impl From<TimeoutConfig> for wit_contract_api::TimeoutConfig {
176    fn from(config: TimeoutConfig) -> Self {
177        let TimeoutConfig {
178            fast_round_duration,
179            base_timeout,
180            timeout_increment,
181            fallback_duration,
182        } = config;
183        Self {
184            fast_round_duration: fast_round_duration.map(Into::into),
185            base_timeout: base_timeout.into(),
186            timeout_increment: timeout_increment.into(),
187            fallback_duration: fallback_duration.into(),
188        }
189    }
190}

192impl From<ApplicationPermissions> for wit_contract_api::ApplicationPermissions {
193    fn from(permissions: ApplicationPermissions) -> Self {
194        let ApplicationPermissions {
195            execute_operations,
196            mandatory_applications,
197            close_chain,
198            change_application_permissions,
199            call_service_as_oracle,
200            make_http_requests,
201        } = permissions;
202        Self {
203            execute_operations: execute_operations
204                .map(|app_ids| app_ids.into_iter().map(Into::into).collect()),
205            mandatory_applications: mandatory_applications.into_iter().map(Into::into).collect(),
206            close_chain: close_chain.into_iter().map(Into::into).collect(),
207            change_application_permissions: change_application_permissions
208                .into_iter()
209                .map(Into::into)
210                .collect(),
211            call_service_as_oracle: call_service_as_oracle
212                .map(|app_ids| app_ids.into_iter().map(Into::into).collect()),
213            make_http_requests: make_http_requests
214                .map(|app_ids| app_ids.into_iter().map(Into::into).collect()),
215        }
216    }
217}

219impl From<ChainOwnership> for wit_contract_api::ChainOwnership {
220    fn from(ownership: ChainOwnership) -> Self {
221        let ChainOwnership {
222            super_owners,
223            owners,
224            multi_leader_rounds,
225            open_multi_leader_rounds,
226            timeout_config,
227        } = ownership;
228        Self {
229            super_owners: super_owners.into_iter().map(Into::into).collect(),
230            owners: owners
231                .into_iter()
232                .map(|(owner, weight)| (owner.into(), weight))
233                .collect(),
234            multi_leader_rounds,
235            open_multi_leader_rounds,
236            timeout_config: timeout_config.into(),
237        }
238    }
239}

241impl From<WriteOperation> for wit_contract_api::WriteOperation {
242    fn from(write_operation: WriteOperation) -> Self {
243        match write_operation {
244            WriteOperation::Delete { key } => wit_contract_api::WriteOperation::Delete(key),
245            WriteOperation::DeletePrefix { key_prefix } => {
246                wit_contract_api::WriteOperation::DeletePrefix(key_prefix)
247            }
248            WriteOperation::Put { key, value } => {
249                wit_contract_api::WriteOperation::Put((key, value))
250            }
251        }
252    }
253}
