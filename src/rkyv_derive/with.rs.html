3#[inline]
4pub fn with<B, F: FnMut(B, &Type) -> B>(field: &Field, init: B, f: F) -> Result<B, Error> {
5    let fields = field
6        .attrs
7        .iter()
8        .filter_map(|attr| {
9            if attr.path.is_ident("with") {
10                Some(attr.parse_args_with(Punctuated::<Type, Comma>::parse_separated_nonempty))
11            } else {
12                None
13            }
14        })
15        .collect::<Result<Vec<_>, _>>()?;
16    Ok(fields.iter().flatten().rev().fold(init, f))
17}
