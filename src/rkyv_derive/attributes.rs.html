22fn try_set_attribute<T: ToTokens>(
23    attribute: &mut Option<T>,
24    value: T,
25    name: &'static str,
26) -> Result<(), Error> {
27    if attribute.is_none() {
28        *attribute = Some(value);
29        Ok(())
30    } else {
31        Err(Error::new_spanned(
32            value,
33            format!("{} already specified", name),
34        ))
35    }
36}

38fn parse_archive_attributes(attributes: &mut Attributes, meta: &Meta) -> Result<(), Error> {
39    match meta {
40        Meta::Path(path) => {
41            if path.is_ident("check_bytes") {
42                try_set_attribute(&mut attributes.check_bytes, path.clone(), "check_bytes")
43            } else if path.is_ident("copy_safe") {
44                try_set_attribute(&mut attributes.copy_safe, path.clone(), "copy_safe")
45            } else {
46                Err(Error::new_spanned(meta, "unrecognized archive argument"))
47            }
48        }
49        Meta::List(list) => {
50            if list.path.is_ident("compare") {
51                if attributes.compares.is_none() {
52                    let mut compares = Vec::new();
53                    for compare in list.nested.iter() {
54                        if let NestedMeta::Meta(Meta::Path(path)) = compare {
55                            compares.push(path.clone());
56                        } else {
57                            return Err(Error::new_spanned(
58                                compare,
59                                "compare arguments must be compare traits to derive",
60                            ));
61                        }
62                    }
63                    attributes.compares = Some((list.path.clone(), compares));
64                    Ok(())
65                } else {
66                    Err(Error::new_spanned(list, "compares already specified"))
67                }
68            } else if list.path.is_ident("bound") {
69                for bound in list.nested.iter() {
70                    if let NestedMeta::Meta(Meta::NameValue(name_value)) = bound {
71                        if let Lit::Str(ref lit_str) = name_value.lit {
72                            if name_value.path.is_ident("archive") {
73                                try_set_attribute(
74                                    &mut attributes.archive_bound,
75                                    lit_str.clone(),
76                                    "archive bound",
77                                )?;
78                            } else if name_value.path.is_ident("serialize") {
79                                try_set_attribute(
80                                    &mut attributes.serialize_bound,
81                                    lit_str.clone(),
82                                    "serialize bound",
83                                )?;
84                            } else if name_value.path.is_ident("deserialize") {
85                                try_set_attribute(
86                                    &mut attributes.deserialize_bound,
87                                    lit_str.clone(),
88                                    "deserialize bound",
89                                )?;
90                            } else {
91                                return Err(Error::new_spanned(
92                                    bound,
93                                    "bound must be either serialize or deserialize",
94                                ));
95                            }
96                        } else {
97                            return Err(Error::new_spanned(
98                                bound,
99                                "bound arguments must be a string",
100                            ));
101                        }
102                    } else {
103                        return Err(Error::new_spanned(
104                            bound,
105                            "bound arguments must be serialize or deserialize bounds to apply",
106                        ));
107                    }
108                }
109                Ok(())
110            } else if list.path.is_ident("repr") {
111                // TODO: remove `archive(repr(...))` syntax
112                attributes.archived_repr.parse_args(list.nested.iter())
113            } else {
114                Err(Error::new_spanned(
115                    &list.path,
116                    "unrecognized archive argument",
117                ))
118            }
119        }
120        Meta::NameValue(meta) => {
121            if meta.path.is_ident("archived") {
122                if let Lit::Str(ref lit_str) = meta.lit {
123                    try_set_attribute(
124                        &mut attributes.archived,
125                        Ident::new(&lit_str.value(), lit_str.span()),
126                        "archived",
127                    )
128                } else {
129                    Err(Error::new_spanned(meta, "archived must be a string"))
130                }
131            } else if meta.path.is_ident("resolver") {
132                if let Lit::Str(ref lit_str) = meta.lit {
133                    try_set_attribute(
134                        &mut attributes.resolver,
135                        Ident::new(&lit_str.value(), lit_str.span()),
136                        "resolver",
137                    )
138                } else {
139                    Err(Error::new_spanned(meta, "resolver must be a string"))
140                }
141            } else if meta.path.is_ident("as") {
142                if let Lit::Str(ref lit_str) = meta.lit {
143                    try_set_attribute(&mut attributes.archive_as, lit_str.clone(), "archive as")
144                } else {
145                    Err(Error::new_spanned(meta, "archive as must be a string"))
146                }
147            } else if meta.path.is_ident("crate") {
148                if let Lit::Str(ref lit_str) = meta.lit {
149                    let stream = syn::parse_str(&lit_str.value())?;
150                    let tokens = crate::serde::respan::respan(stream, lit_str.span());
151                    let path = syn::parse2(tokens)?;
152                    try_set_attribute(&mut attributes.rkyv_path, path, "crate")?;
153                    attributes.rkyv_path_str = Some(lit_str.clone());
154                    Ok(())
155                } else {
156                    Err(Error::new_spanned(meta, "crate must be a string"))
157                }
158            } else {
159                Err(Error::new_spanned(meta, "unrecognized archive argument"))
160            }
161        }
162    }
163}

165pub fn parse_attributes(input: &DeriveInput) -> Result<Attributes, Error> {
166    let mut result = Attributes::default();
167    for attr in input.attrs.iter() {
168        if let AttrStyle::Outer = attr.style {
169            if attr.path.is_ident("archive") || attr.path.is_ident("archive_attr") {
170                if let Meta::List(list) = attr.parse_meta()? {
171                    if list.path.is_ident("archive") {
172                        for nested in list.nested.iter() {
173                            if let NestedMeta::Meta(meta) = nested {
174                                parse_archive_attributes(&mut result, meta)?;
175                            } else {
176                                return Err(Error::new_spanned(
177                                    nested,
178                                    "archive arguments must be metas",
179                                ));
180                            }
181                        }
182                    } else if list.path.is_ident("archive_attr") {
183                        for nested in list.nested.iter() {
184                            if let NestedMeta::Meta(meta) = nested {
185                                if let Meta::List(list) = meta {
186                                    if list.path.is_ident("repr") {
187                                        result.archived_repr.parse_args(list.nested.iter())?;
188                                    } else {
189                                        result.attrs.push(meta.clone());
190                                    }
191                                } else {
192                                    result.attrs.push(meta.clone());
193                                }
194                            } else {
195                                return Err(Error::new_spanned(
196                                    nested,
197                                    "archive_attr arguments must be metas",
198                                ));
199                            }
200                        }
201                    }
202                } else {
203                    return Err(Error::new_spanned(
204                        attr,
205                        "archive and archive_attr may only be structured list attributes",
206                    ));
207                }
208            }
209        }
210    }
211
212    Ok(result)
