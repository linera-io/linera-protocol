155/// Handles batched forwarding of notifications to proxy and exporters.
156struct BatchForwarder {
157    nickname: String,
158    client: NotifierServiceClient<Channel>,
159    exporter_clients: Vec<NotifierServiceClient<Channel>>,
160    pending_notifications: Vec<Notification>,
161    futures: FuturesUnordered<BoxFuture<'static, ()>>,
162    batch_limit: usize,
163    max_tasks: usize,
164}
165
166impl BatchForwarder {
167    /// Spawns batch send tasks up to max_tasks limit.
168    fn spawn_batches(&mut self) {
169        while !self.pending_notifications.is_empty() && self.futures.len() < self.max_tasks {
170            let chunk_size = std::cmp::min(self.batch_limit, self.pending_notifications.len());
171            let batch: Vec<Notification> = self.pending_notifications.drain(..chunk_size).collect();
172
173            #[cfg(with_metrics)]
174            metrics::NOTIFICATION_BATCH_SIZE
175                .with_label_values(&[])
176                .observe(batch.len() as f64);
177
178            let client = self.client.clone();
179            let exporter_clients = self.exporter_clients.clone();
180            let nickname = self.nickname.clone();
181
182            self.futures.push(
183                async move {
184                    Self::send_batch(nickname, client, exporter_clients, batch).await;
185                }
186                .boxed(),
187            );
188        }
189    }
190
191    /// Returns true if there are no pending notifications and no in-flight tasks.
192    fn is_fully_drained(&self) -> bool {
193        self.pending_notifications.is_empty() && self.futures.is_empty()
194    }
195
196    /// Sends a batch of notifications to the proxy and exporters.
197    async fn send_batch(
198        nickname: String,
199        mut client: NotifierServiceClient<Channel>,
200        mut exporter_clients: Vec<NotifierServiceClient<Channel>>,
201        batch: Vec<Notification>,
202    ) {
203        // Convert to proto notifications, logging any deserialization errors
204        let mut proto_notifications = Vec::with_capacity(batch.len());
205        for notification in &batch {
206            match notification.clone().try_into() {
207                Ok(proto) => proto_notifications.push(proto),
208                Err(error) => {
209                    warn!(
210                        %error,
211                        nickname,
212                        ?notification.chain_id,
213                        ?notification.reason,
214                        "could not deserialize notification"
215                    );
216                }
217            }
218        }
219
220        // Collect chain_ids for error logging
221        let chain_ids: Vec<_> = batch.iter().map(|n| n.chain_id).collect();
222
223        // Send batch to proxy
224        let request = Request::new(api::NotificationBatch {
225            notifications: proto_notifications.clone(),
226        });
227        let result = client.notify_batch(request).await;
228
229        #[cfg(with_metrics)]
230        {
231            let status = if result.is_ok() { "success" } else { "error" };
232            metrics::NOTIFICATION_BATCHES_SENT
233                .with_label_values(&[status])
234                .inc();
235        }
236
237        if let Err(error) = result {
238            error!(
239                %error,
240                nickname,
241                batch_size = proto_notifications.len(),
242                ?chain_ids,
243                "proxy: could not send notification batch",
244            );
245        }
246
247        // Send NewBlock notifications to exporters
248        let new_block_notifications: Vec<_> = batch
249            .iter()
250            .filter(|n| matches!(n.reason, Reason::NewBlock { .. }))
251            .collect();
252
253        let exporter_notifications: Vec<api::Notification> = new_block_notifications
254            .iter()
255            .filter_map(|n| (*n).clone().try_into().ok())
256            .collect();
257
258        if !exporter_notifications.is_empty() {
259            let exporter_chain_ids: Vec<_> =
260                new_block_notifications.iter().map(|n| n.chain_id).collect();
261
262            for exporter_client in &mut exporter_clients {
263                let request = Request::new(api::NotificationBatch {
264                    notifications: exporter_notifications.clone(),
265                });
266                if let Err(error) = exporter_client.notify_batch(request).await {
267                    error!(
268                        %error,
269                        nickname,
270                        batch_size = exporter_notifications.len(),
271                        ?exporter_chain_ids,
272                        "block exporter: could not send notification batch",
273                    );
274                }
275            }
276        }
277    }
278}

318impl<S, B> Service<http::Request<B>> for GrpcPrometheusMetricsMiddlewareService<S>
319where
320    S::Future: Send + 'static,
321    S: Service<http::Request<B>> + std::marker::Send,
322    B: Send + 'static,
323{
324    type Response = S::Response;
325    type Error = S::Error;
326    type Future = BoxFuture<'static, Result<S::Response, S::Error>>;
327
328    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
329        self.service.poll_ready(cx)
330    }
331
332    fn call(&mut self, request: http::Request<B>) -> Self::Future {
333        #[cfg(with_metrics)]
334        let start = Instant::now();
335
336        // Extract traffic type from request extensions (set by OtelContextLayer).
337        // When opentelemetry is enabled but no baggage is set, defaults to "organic".
338        // When opentelemetry is disabled, defaults to "unknown".
339        #[cfg(all(with_metrics, feature = "opentelemetry"))]
340        let traffic_type: &'static str = get_traffic_type_from_request(&request);
341        #[cfg(all(with_metrics, not(feature = "opentelemetry")))]
342        let traffic_type: &'static str = "unknown";
343
344        let future = self.service.call(request);
345        async move {
346            let response = future.await?;
347            #[cfg(with_metrics)]
348            {
349                metrics::SERVER_REQUEST_LATENCY
350                    .with_label_values(&[traffic_type])
351                    .observe(start.elapsed().as_secs_f64() * 1000.0);
352                metrics::SERVER_REQUEST_COUNT
353                    .with_label_values(&[traffic_type])
354                    .inc();
355            }
356            Ok(response)
357        }
358        .boxed()
359    }
360}

366    #[expect(clippy::too_many_arguments)]
367    pub fn spawn(
368        host: String,
369        port: u16,
370        state: WorkerState<S>,
371        shard_id: ShardId,
372        internal_network: ValidatorInternalNetworkConfig,
373        cross_chain_config: CrossChainConfig,
374        notification_config: NotificationConfig,
375        shutdown_signal: CancellationToken,
376        join_set: &mut JoinSet,
377    ) -> GrpcServerHandle {
378        info!(
379            "spawning gRPC server on {}:{} for shard {}",
380            host, port, shard_id
381        );
382
383        let (cross_chain_sender, cross_chain_receiver) =
384            mpsc::channel(cross_chain_config.queue_size);
385
386        let (notification_sender, _) =
387            tokio::sync::broadcast::channel(notification_config.notification_queue_size);
388
389        join_set.spawn_task({
390            info!(
391                nickname = state.nickname(),
392                "spawning cross-chain queries thread on {} for shard {}", host, shard_id
393            );
394            Self::forward_cross_chain_queries(
395                state.nickname().to_string(),
396                internal_network.clone(),
397                cross_chain_config.max_retries,
398                Duration::from_millis(cross_chain_config.retry_delay_ms),
399                Duration::from_millis(cross_chain_config.max_backoff_ms),
400                Duration::from_millis(cross_chain_config.sender_delay_ms),
401                cross_chain_config.sender_failure_rate,
402                shard_id,
403                cross_chain_receiver,
404            )
405        });
406
407        let mut exporter_forwarded = false;
408        for proxy in &internal_network.proxies {
409            let receiver = notification_sender.subscribe();
410            join_set.spawn_task({
411                info!(
412                    nickname = state.nickname(),
413                    "spawning notifications thread on {} for shard {}", host, shard_id
414                );
415                let exporter_addresses = if exporter_forwarded {
416                    vec![]
417                } else {
418                    exporter_forwarded = true;
419                    internal_network.exporter_addresses()
420                };
421                Self::forward_notifications(
422                    state.nickname().to_string(),
423                    proxy.internal_address(&internal_network.protocol),
424                    exporter_addresses,
425                    receiver,
426                    notification_config.clone(),
427                )
428            });
429        }
430
431        let (health_reporter, health_service) = tonic_health::server::health_reporter();
432
433        let grpc_server = GrpcServer {
434            state,
435            shard_id,
436            network: internal_network,
437            cross_chain_sender,
438            notification_sender,
439        };
440
441        let worker_node = ValidatorWorkerServer::new(grpc_server)
442            .max_encoding_message_size(GRPC_MAX_MESSAGE_SIZE)
443            .max_decoding_message_size(GRPC_MAX_MESSAGE_SIZE);
444
445        let handle = join_set.spawn_task(async move {
446            let server_address = SocketAddr::from((IpAddr::from_str(&host)?, port));
447
448            let reflection_service = tonic_reflection::server::Builder::configure()
449                .register_encoded_file_descriptor_set(crate::FILE_DESCRIPTOR_SET)
450                .build_v1()?;
451
452            health_reporter
453                .set_serving::<ValidatorWorkerServer<Self>>()
454                .await;
455
456            #[cfg(feature = "opentelemetry")]
457            let mut server = tonic::transport::Server::builder().layer(
458                ServiceBuilder::new()
459                    .layer(crate::propagation::OtelContextLayer)
460                    .layer(GrpcPrometheusMetricsMiddlewareLayer)
461                    .into_inner(),
462            );
463            #[cfg(not(feature = "opentelemetry"))]
464            let mut server = tonic::transport::Server::builder().layer(
465                ServiceBuilder::new()
466                    .layer(GrpcPrometheusMetricsMiddlewareLayer)
467                    .into_inner(),
468            );
469            server
470                .add_service(health_service)
471                .add_service(reflection_service)
472                .add_service(worker_node)
473                .serve_with_shutdown(server_address, shutdown_signal.cancelled_owned())
474                .await?;
475
476            Ok(())
477        });
478
479        GrpcServerHandle { handle }
480    }

482    /// Continuously waits for receiver to receive notifications and sends them to
483    /// the proxy in batches for improved throughput.
484    #[instrument(skip(receiver, config))]
485    async fn forward_notifications(
486        nickname: String,
487        proxy_address: String,
488        exporter_addresses: Vec<String>,
489        mut receiver: tokio::sync::broadcast::Receiver<Notification>,
490        config: NotificationConfig,
491    ) {
492        let channel = tonic::transport::Channel::from_shared(proxy_address.clone())
493            .expect("Proxy URI should be valid")
494            .connect_lazy();
495        let client = NotifierServiceClient::new(channel)
496            .max_encoding_message_size(GRPC_MAX_MESSAGE_SIZE)
497            .max_decoding_message_size(GRPC_MAX_MESSAGE_SIZE);
498
499        let exporter_clients: Vec<NotifierServiceClient<Channel>> = exporter_addresses
500            .iter()
501            .map(|address| {
502                let channel = tonic::transport::Channel::from_shared(address.clone())
503                    .expect("Exporter URI should be valid")
504                    .connect_lazy();
505                NotifierServiceClient::new(channel)
506                    .max_encoding_message_size(GRPC_MAX_MESSAGE_SIZE)
507                    .max_decoding_message_size(GRPC_MAX_MESSAGE_SIZE)
508            })
509            .collect::<Vec<_>>();
510
511        let mut forwarder = BatchForwarder {
512            nickname: nickname.clone(),
513            client,
514            exporter_clients,
515            pending_notifications: Vec::new(),
516            futures: FuturesUnordered::new(),
517            batch_limit: config.notification_batch_size,
518            max_tasks: config.notification_max_in_flight,
519        };
520
521        loop {
522            tokio::select! {
523                biased;
524
525                result = receiver.recv() => {
526                    match result {
527                        Ok(notification) => {
528                            forwarder.pending_notifications.push(notification);
529
530                            if forwarder.futures.is_empty()
531                               || (forwarder.pending_notifications.len() >= forwarder.batch_limit
532                                   && forwarder.futures.len() < forwarder.max_tasks) {
533                                forwarder.spawn_batches();
534                            }
535                        }
536                        Err(RecvError::Lagged(skipped_count)) => {
537                            warn!(
538                                nickname,
539                                skipped_count, "notification receiver lagged, messages were skipped"
540                            );
541                            #[cfg(with_metrics)]
542                            metrics::NOTIFICATIONS_SKIPPED_RECEIVER_LAG
543                                .with_label_values(&[])
544                                .inc_by(skipped_count);
545                        }
546                        Err(RecvError::Closed) => {
547                            warn!(
548                                nickname,
549                                "notification channel closed, draining pending notifications"
550                            );
551                            // Drain all pending notifications before exiting
552                            loop {
553                                forwarder.spawn_batches();
554                                if forwarder.is_fully_drained() {
555                                    break;
556                                }
557                                forwarder.futures.next().await;
558                            }
559                            break;
560                        }
561                    }
562                }
563
564                Some(()) = forwarder.futures.next() => {
565                    forwarder.spawn_batches();
566                }
567            }
568        }
569    }

571    fn handle_network_actions(&self, actions: NetworkActions) {
572        let mut cross_chain_sender = self.cross_chain_sender.clone();
573        let notification_sender = self.notification_sender.clone();
574
575        for request in actions.cross_chain_requests {
576            let shard_id = self.network.get_shard_id(request.target_chain_id());
577            trace!(
578                source_shard_id = self.shard_id,
579                target_shard_id = shard_id,
580                "Scheduling cross-chain query",
581            );
582
583            if let Err(error) = cross_chain_sender.try_send((request, shard_id)) {
584                error!(%error, "dropping cross-chain request");
585                #[cfg(with_metrics)]
586                if error.is_full() {
587                    metrics::CROSS_CHAIN_MESSAGE_CHANNEL_FULL
588                        .with_label_values(&[])
589                        .inc();
590                }
591            }
592        }
593
594        for notification in actions.notifications {
595            trace!("Scheduling notification query");
596            if let Err(error) = notification_sender.send(notification) {
597                error!(%error, "dropping notification");
598                #[cfg(with_metrics)]
599                metrics::NOTIFICATIONS_DROPPED_NO_RECEIVER
600                    .with_label_values(&[])
601                    .inc();
602            }
603        }
604    }

606    #[instrument(skip_all, fields(nickname, %this_shard))]
607    #[expect(clippy::too_many_arguments)]
608    async fn forward_cross_chain_queries(
609        nickname: String,
610        network: ValidatorInternalNetworkConfig,
611        cross_chain_max_retries: u32,
612        cross_chain_retry_delay: Duration,
613        cross_chain_max_backoff: Duration,
614        cross_chain_sender_delay: Duration,
615        cross_chain_sender_failure_rate: f32,
616        this_shard: ShardId,
617        receiver: mpsc::Receiver<(linera_core::data_types::CrossChainRequest, ShardId)>,
618    ) {
619        let pool = GrpcConnectionPool::default();
620        let handle_request =
621            move |shard_id: ShardId, request: linera_core::data_types::CrossChainRequest| {
622                let channel_result = pool.channel(network.shard(shard_id).http_address());
623                async move {
624                    let mut client = ValidatorWorkerClient::new(channel_result?)
625                        .max_encoding_message_size(GRPC_MAX_MESSAGE_SIZE)
626                        .max_decoding_message_size(GRPC_MAX_MESSAGE_SIZE);
627                    client
628                        .handle_cross_chain_request(Request::new(request.try_into()?))
629                        .await?;
630                    anyhow::Result::<_, anyhow::Error>::Ok(())
631                }
632            };
633        cross_chain_message_queue::forward_cross_chain_queries(
634            nickname,
635            cross_chain_max_retries,
636            cross_chain_retry_delay,
637            cross_chain_max_backoff,
638            cross_chain_sender_delay,
639            cross_chain_sender_failure_rate,
640            this_shard,
641            receiver,
642            handle_request,
643        )
644        .await;
645    }

683    fn log_error(&self, error: &linera_core::worker::WorkerError, context: &str) {
684        let nickname = self.state.nickname();
685        if error.is_local() {
686            error!(nickname, %error, "{}", context);
