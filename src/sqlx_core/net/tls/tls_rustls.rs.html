25pub struct RustlsSocket<S: Socket> {
26    inner: StdSocket<S>,
27    state: ClientConnection,
28    close_notify_sent: bool,
29}

31impl<S: Socket> RustlsSocket<S> {
32    fn poll_complete_io(&mut self, cx: &mut Context<'_>) -> Poll<io::Result<()>> {
33        loop {
34            match self.state.complete_io(&mut self.inner) {
35                Err(e) if e.kind() == io::ErrorKind::WouldBlock => {
36                    futures_util::ready!(self.inner.poll_ready(cx))?;
37                }
38                ready => return Poll::Ready(ready.map(|_| ())),
39            }
40        }
41    }
42
43    async fn complete_io(&mut self) -> io::Result<()> {
44        future::poll_fn(|cx| self.poll_complete_io(cx)).await
45    }
46}
47
48impl<S: Socket> Socket for RustlsSocket<S> {
49    fn try_read(&mut self, buf: &mut dyn ReadBuf) -> io::Result<usize> {
50        self.state.reader().read(buf.init_mut())
51    }
52
53    fn try_write(&mut self, buf: &[u8]) -> io::Result<usize> {
54        match self.state.writer().write(buf) {
55            // Returns a zero-length write when the buffer is full.
56            Ok(0) => Err(io::ErrorKind::WouldBlock.into()),
57            other => other,
58        }
59    }
60
61    fn poll_read_ready(&mut self, cx: &mut Context<'_>) -> Poll<io::Result<()>> {
62        self.poll_complete_io(cx)
63    }
64
65    fn poll_write_ready(&mut self, cx: &mut Context<'_>) -> Poll<io::Result<()>> {
66        self.poll_complete_io(cx)
67    }
68
69    fn poll_flush(&mut self, cx: &mut Context<'_>) -> Poll<io::Result<()>> {
70        self.poll_complete_io(cx)
71    }
72
73    fn poll_shutdown(&mut self, cx: &mut Context<'_>) -> Poll<io::Result<()>> {
74        if !self.close_notify_sent {
75            self.state.send_close_notify();
76            self.close_notify_sent = true;
77        }
78
79        futures_util::ready!(self.poll_complete_io(cx))?;
80
81        // Server can close socket as soon as it receives the connection shutdown request.
82        // We shouldn't expect it to stick around for the TLS session to close cleanly.
83        // https://security.stackexchange.com/a/82034
84        let _ = futures_util::ready!(self.inner.socket.poll_shutdown(cx));
85
86        Poll::Ready(Ok(()))
87    }
88}

90pub async fn handshake<S>(socket: S, tls_config: TlsConfig<'_>) -> Result<RustlsSocket<S>, Error>
91where
92    S: Socket,
93{
94    #[cfg(all(
95        feature = "_tls-rustls-aws-lc-rs",
96        not(feature = "_tls-rustls-ring-webpki"),
97        not(feature = "_tls-rustls-ring-native-roots")
98    ))]
99    let provider = Arc::new(rustls::crypto::aws_lc_rs::default_provider());
100    #[cfg(any(
101        feature = "_tls-rustls-ring-webpki",
102        feature = "_tls-rustls-ring-native-roots"
103    ))]
104    let provider = Arc::new(rustls::crypto::ring::default_provider());
105
106    // Unwrapping is safe here because we use a default provider.
107    let config = ClientConfig::builder_with_provider(provider.clone())
108        .with_safe_default_protocol_versions()
109        .unwrap();
110
111    // authentication using user's key and its associated certificate
112    let user_auth = match (tls_config.client_cert_path, tls_config.client_key_path) {
113        (Some(cert_path), Some(key_path)) => {
114            let cert_chain = certs_from_pem(cert_path.data().await?)?;
115            let key_der = private_key_from_pem(key_path.data().await?)?;
116            Some((cert_chain, key_der))
117        }
118        (None, None) => None,
119        (_, _) => {
120            return Err(Error::Configuration(
121                "user auth key and certs must be given together".into(),
122            ))
123        }
124    };
125
126    let config = if tls_config.accept_invalid_certs {
127        if let Some(user_auth) = user_auth {
128            config
129                .dangerous()
130                .with_custom_certificate_verifier(Arc::new(DummyTlsVerifier { provider }))
131                .with_client_auth_cert(user_auth.0, user_auth.1)
132                .map_err(Error::tls)?
133        } else {
134            config
135                .dangerous()
136                .with_custom_certificate_verifier(Arc::new(DummyTlsVerifier { provider }))
137                .with_no_client_auth()
138        }
139    } else {
140        #[cfg(any(feature = "_tls-rustls-aws-lc-rs", feature = "_tls-rustls-ring-webpki"))]
141        let mut cert_store = certs_from_webpki();
142        #[cfg(feature = "_tls-rustls-ring-native-roots")]
143        let mut cert_store = certs_from_native_store();
144
145        if let Some(ca) = tls_config.root_cert_path {
146            let data = ca.data().await?;
147
148            for result in CertificateDer::pem_slice_iter(&data) {
149                let Ok(cert) = result else {
150                    return Err(Error::Tls(format!("Invalid certificate {ca}").into()));
151                };
152
153                cert_store.add(cert).map_err(|err| Error::Tls(err.into()))?;
154            }
155        }
156
157        if tls_config.accept_invalid_hostnames {
158            let verifier = WebPkiServerVerifier::builder(Arc::new(cert_store))
159                .build()
160                .map_err(|err| Error::Tls(err.into()))?;
161
162            if let Some(user_auth) = user_auth {
163                config
164                    .dangerous()
165                    .with_custom_certificate_verifier(Arc::new(NoHostnameTlsVerifier { verifier }))
166                    .with_client_auth_cert(user_auth.0, user_auth.1)
167                    .map_err(Error::tls)?
168            } else {
169                config
170                    .dangerous()
171                    .with_custom_certificate_verifier(Arc::new(NoHostnameTlsVerifier { verifier }))
172                    .with_no_client_auth()
173            }
174        } else if let Some(user_auth) = user_auth {
175            config
176                .with_root_certificates(cert_store)
177                .with_client_auth_cert(user_auth.0, user_auth.1)
178                .map_err(Error::tls)?
179        } else {
180            config
181                .with_root_certificates(cert_store)
182                .with_no_client_auth()
183        }
184    };
185
186    let host = ServerName::try_from(tls_config.hostname.to_owned()).map_err(Error::tls)?;
187
188    let mut socket = RustlsSocket {
189        inner: StdSocket::new(socket),
190        state: ClientConnection::new(Arc::new(config), host).map_err(Error::tls)?,
191        close_notify_sent: false,
192    };
193
194    // Performs the TLS handshake or bails
195    socket.complete_io().await?;
196
197    Ok(socket)
198}

200fn certs_from_pem(pem: Vec<u8>) -> Result<Vec<CertificateDer<'static>>, Error> {
201    CertificateDer::pem_slice_iter(&pem)
202        .map(|result| result.map_err(|err| Error::Tls(err.into())))
203        .collect()
204}
205
206fn private_key_from_pem(pem: Vec<u8>) -> Result<PrivateKeyDer<'static>, Error> {
207    match PrivateKeyDer::from_pem_slice(&pem) {
208        Ok(key) => Ok(key),
209        Err(pem::Error::NoItemsFound) => Err(Error::Configuration("no keys found pem file".into())),
210        Err(e) => Err(Error::Configuration(e.to_string().into())),
211    }
212}

214#[cfg(any(feature = "_tls-rustls-aws-lc-rs", feature = "_tls-rustls-ring-webpki"))]
215fn certs_from_webpki() -> RootCertStore {
216    RootCertStore::from_iter(webpki_roots::TLS_SERVER_ROOTS.iter().cloned())
217}
218
219#[cfg(feature = "_tls-rustls-ring-native-roots")]
220fn certs_from_native_store() -> RootCertStore {
221    let mut root_cert_store = RootCertStore::empty();
222
223    let load_results = rustls_native_certs::load_native_certs();
224    for e in load_results.errors {
225        log::warn!("Error loading native certificates: {e:?}");
226    }
227    for cert in load_results.certs {
228        if let Err(e) = root_cert_store.add(cert.into()) {
229            log::warn!("rustls failed to parse native certificate: {e:?}");
230        }
231    }
232
233    root_cert_store
234}

241impl ServerCertVerifier for DummyTlsVerifier {

243        &self,

245        _intermediates: &[CertificateDer<'_>],
246        _server_name: &ServerName<'_>,

265    }

273        verify_tls13_signature(

275            cert,
276            dss,
277            &self.provider.signature_verification_algorithms,
278        )
279    }

288#[derive(Debug)]
289pub struct NoHostnameTlsVerifier {
290    verifier: Arc<WebPkiServerVerifier>,
291}
292
293impl ServerCertVerifier for NoHostnameTlsVerifier {
294    fn verify_server_cert(
295        &self,
296        end_entity: &CertificateDer<'_>,
297        intermediates: &[CertificateDer<'_>],
298        server_name: &ServerName<'_>,
299        ocsp_response: &[u8],
300        now: UnixTime,
301    ) -> Result<ServerCertVerified, TlsError> {
302        match self.verifier.verify_server_cert(
303            end_entity,
304            intermediates,
305            server_name,
306            ocsp_response,
307            now,
308        ) {
309            Err(TlsError::InvalidCertificate(CertificateError::NotValidForName)) => {
310                Ok(ServerCertVerified::assertion())
311            }
312            res => res,
313        }
314    }
315
316    fn verify_tls12_signature(
317        &self,
318        message: &[u8],
319        cert: &CertificateDer<'_>,
320        dss: &rustls::DigitallySignedStruct,
321    ) -> Result<rustls::client::danger::HandshakeSignatureValid, TlsError> {
322        self.verifier.verify_tls12_signature(message, cert, dss)
323    }
324
325    fn verify_tls13_signature(
326        &self,
327        message: &[u8],
328        cert: &CertificateDer<'_>,
329        dss: &rustls::DigitallySignedStruct,
330    ) -> Result<rustls::client::danger::HandshakeSignatureValid, TlsError> {
331        self.verifier.verify_tls13_signature(message, cert, dss)
332    }
333
334    fn supported_verify_schemes(&self) -> Vec<rustls::SignatureScheme> {
335        self.verifier.supported_verify_schemes()
336    }
