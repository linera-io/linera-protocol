40    pub(super) fn new_arc(
41        options: PoolOptions<DB>,
42        connect_options: <DB::Connection as Connection>::Options,
43    ) -> Arc<Self> {
44        let capacity = options.max_connections as usize;
45
46        let semaphore_capacity = if let Some(parent) = &options.parent_pool {
47            assert!(options.max_connections <= parent.options().max_connections);
48            assert_eq!(options.fair, parent.options().fair);
49            // The child pool must steal permits from the parent
50            0
51        } else {
52            capacity
53        };
54
55        let pool = Self {
56            connect_options: RwLock::new(Arc::new(connect_options)),
57            idle_conns: ArrayQueue::new(capacity),
58            semaphore: AsyncSemaphore::new(options.fair, semaphore_capacity),
59            size: AtomicU32::new(0),
60            num_idle: AtomicUsize::new(0),
61            is_closed: AtomicBool::new(false),
62            on_closed: event_listener::Event::new(),
63            acquire_time_level: private_level_filter_to_trace_level(options.acquire_time_level),
64            acquire_slow_level: private_level_filter_to_trace_level(options.acquire_slow_level),
65            options,
66        };
67
68        let pool = Arc::new(pool);
69
70        spawn_maintenance_tasks(&pool);
71
72        pool
73    }

75    pub(super) fn size(&self) -> u32 {
76        self.size.load(Ordering::Acquire)

79    pub(super) fn num_idle(&self) -> usize {
80        // We don't use `self.idle_conns.len()` as it waits for the internal
81        // head and tail pointers to stop changing for a moment before calculating the length,
82        // which may take a long time at high levels of churn.
83        //
84        // By maintaining our own atomic count, we avoid that issue entirely.
85        self.num_idle.load(Ordering::Acquire)
86    }

97    pub(super) fn close<'a>(self: &'a Arc<Self>) -> impl Future<Output = ()> + 'a {
98        self.mark_closed();
99
100        async move {
101            for permits in 1..=self.options.max_connections {
102                // Close any currently idle connections in the pool.
103                while let Some(idle) = self.idle_conns.pop() {
104                    let _ = idle.live.float((*self).clone()).close().await;
105                }
106
107                if self.size() == 0 {
108                    break;
109                }
110
111                // Wait for all permits to be released.
112                let _permits = self.semaphore.acquire(permits).await;
113            }
114        }
115    }

117    pub(crate) fn close_event(&self) -> CloseEvent {
118        CloseEvent {
119            listener: (!self.is_closed()).then(|| self.on_closed.listen()),
120        }

123    /// Attempt to pull a permit from `self.semaphore` or steal one from the parent.
124    ///
125    /// If we steal a permit from the parent but *don't* open a connection,
126    /// it should be returned to the parent.
127    async fn acquire_permit<'a>(self: &'a Arc<Self>) -> Result<AsyncSemaphoreReleaser<'a>, Error> {
128        let parent = self
129            .parent()
130            // If we're already at the max size, we shouldn't try to steal from the parent.
131            // This is just going to cause unnecessary churn in `acquire()`.
132            .filter(|_| self.size() < self.options.max_connections);
133
134        let mut acquire_self = pin!(self.semaphore.acquire(1).fuse());
135        let mut close_event = pin!(self.close_event());
136
137        if let Some(parent) = parent {
138            let mut acquire_parent = pin!(parent.0.semaphore.acquire(1));
139            let mut parent_close_event = pin!(parent.0.close_event());
140
141            let mut poll_parent = false;
142
143            future::poll_fn(|cx| {
144                if close_event.as_mut().poll(cx).is_ready() {
145                    return Poll::Ready(Err(Error::PoolClosed));
146                }
147
148                if parent_close_event.as_mut().poll(cx).is_ready() {
149                    // Propagate the parent's close event to the child.
150                    self.mark_closed();
151                    return Poll::Ready(Err(Error::PoolClosed));
152                }
153
154                if let Poll::Ready(permit) = acquire_self.as_mut().poll(cx) {
155                    return Poll::Ready(Ok(permit));
156                }
157
158                // Don't try the parent right away.
159                if poll_parent {
160                    acquire_parent.as_mut().poll(cx).map(Ok)
161                } else {
162                    poll_parent = true;
163                    cx.waker().wake_by_ref();
164                    Poll::Pending
165                }
166            })
167            .await
168        } else {
169            close_event.do_until(acquire_self).await
170        }
171    }

177    #[inline]
178    pub(super) fn try_acquire(self: &Arc<Self>) -> Option<Floating<DB, Idle<DB>>> {
179        if self.is_closed() {
180            return None;
181        }
182
183        let permit = self.semaphore.try_acquire(1)?;
184
185        self.pop_idle(permit).ok()
186    }
187
188    fn pop_idle<'a>(
189        self: &'a Arc<Self>,
190        permit: AsyncSemaphoreReleaser<'a>,
191    ) -> Result<Floating<DB, Idle<DB>>, AsyncSemaphoreReleaser<'a>> {
192        if let Some(idle) = self.idle_conns.pop() {
193            self.num_idle.fetch_sub(1, Ordering::AcqRel);
194            Ok(Floating::from_idle(idle, (*self).clone(), permit))
195        } else {
196            Err(permit)
197        }
198    }

200    pub(super) fn release(&self, floating: Floating<DB, Live<DB>>) {
201        // `options.after_release` and other checks are in `PoolConnection::return_to_pool()`.
202
203        let Floating { inner: idle, guard } = floating.into_idle();
204
205        if self.idle_conns.push(idle).is_err() {
206            panic!("BUG: connection queue overflow in release()");
207        }
208
209        // NOTE: we need to make sure we drop the permit *after* we push to the idle queue
210        // don't decrease the size
211        guard.release_permit();
212
213        self.num_idle.fetch_add(1, Ordering::AcqRel);
214    }

216    /// Try to atomically increment the pool size for a new connection.
217    ///
218    /// Returns `Err` if the pool is at max capacity already or is closed.
219    pub(super) fn try_increment_size<'a>(
220        self: &'a Arc<Self>,
221        permit: AsyncSemaphoreReleaser<'a>,
222    ) -> Result<DecrementSizeGuard<DB>, AsyncSemaphoreReleaser<'a>> {
223        let result = self
224            .size
225            .fetch_update(Ordering::AcqRel, Ordering::Acquire, |size| {
226                if self.is_closed() {
227                    return None;
228                }
229
230                size.checked_add(1)
231                    .filter(|size| size <= &self.options.max_connections)
232            });
233
234        match result {
235            // we successfully incremented the size
236            Ok(_) => Ok(DecrementSizeGuard::from_permit((*self).clone(), permit)),
237            // the pool is at max capacity or is closed
238            Err(_) => Err(permit),
239        }
240    }

242    pub(super) async fn acquire(self: &Arc<Self>) -> Result<Floating<DB, Live<DB>>, Error> {
243        if self.is_closed() {
244            return Err(Error::PoolClosed);
245        }
246
247        let acquire_started_at = Instant::now();
248        let deadline = acquire_started_at + self.options.acquire_timeout;
249
250        let acquired = crate::rt::timeout(
251            self.options.acquire_timeout,
252            async {
253                loop {
254                    // Handles the close-event internally
255                    let permit = self.acquire_permit().await?;
256
257
258                    // First attempt to pop a connection from the idle queue.
259                    let guard = match self.pop_idle(permit) {
260
261                        // Then, check that we can use it...
262                        Ok(conn) => match check_idle_conn(conn, &self.options).await {
263
264                            // All good!
265                            Ok(live) => return Ok(live),
266
267                            // if the connection isn't usable for one reason or another,
268                            // we get the `DecrementSizeGuard` back to open a new one
269                            Err(guard) => guard,
270                        },
271                        Err(permit) => if let Ok(guard) = self.try_increment_size(permit) {
272                            // we can open a new connection
273                            guard
274                        } else {
275                            // This can happen for a child pool that's at its connection limit,
276                            // or if the pool was closed between `acquire_permit()` and
277                            // `try_increment_size()`.
278                            tracing::debug!("woke but was unable to acquire idle connection or open new one; retrying");
279                            // If so, we're likely in the current-thread runtime if it's Tokio,
280                            // and so we should yield to let any spawned return_to_pool() tasks
281                            // execute.
282                            crate::rt::yield_now().await;
283                            continue;
284                        }
285                    };
286
287                    // Attempt to connect...
288                    return self.connect(deadline, guard).await;
289                }
290            }
291        )
292            .await
293            .map_err(|_| Error::PoolTimedOut)??;
294
295        let acquired_after = acquire_started_at.elapsed();
296
297        let acquire_slow_level = self
298            .acquire_slow_level
299            .filter(|_| acquired_after > self.options.acquire_slow_threshold);
300
301        if let Some(level) = acquire_slow_level {
302            private_tracing_dynamic_event!(
303                target: "sqlx::pool::acquire",
304                level,
305                aquired_after_secs = acquired_after.as_secs_f64(),
306                slow_acquire_threshold_secs = self.options.acquire_slow_threshold.as_secs_f64(),
307                "acquired connection, but time to acquire exceeded slow threshold"
308            );
309        } else if let Some(level) = self.acquire_time_level {
310            private_tracing_dynamic_event!(
311                target: "sqlx::pool::acquire",
312                level,
313                aquired_after_secs = acquired_after.as_secs_f64(),
314                "acquired connection"
315            );
316        }
317
318        Ok(acquired)
319    }

321    pub(super) async fn connect(
322        self: &Arc<Self>,
323        deadline: Instant,
324        guard: DecrementSizeGuard<DB>,
325    ) -> Result<Floating<DB, Live<DB>>, Error> {
326        if self.is_closed() {
327            return Err(Error::PoolClosed);
328        }
329
330        let mut backoff = Duration::from_millis(10);
331        let max_backoff = deadline_as_timeout(deadline)? / 5;
332
333        loop {
334            let timeout = deadline_as_timeout(deadline)?;
335
336            // clone the connect options arc so it can be used without holding the RwLockReadGuard
337            // across an async await point
338            let connect_options = self
339                .connect_options
340                .read()
341                .expect("write-lock holder panicked")
342                .clone();
343
344            // result here is `Result<Result<C, Error>, TimeoutError>`
345            // if this block does not return, sleep for the backoff timeout and try again
346            match crate::rt::timeout(timeout, connect_options.connect()).await {
347                // successfully established connection
348                Ok(Ok(mut raw)) => {
349                    // See comment on `PoolOptions::after_connect`
350                    let meta = PoolConnectionMetadata {
351                        age: Duration::ZERO,
352                        idle_for: Duration::ZERO,
353                    };
354
355                    let res = if let Some(callback) = &self.options.after_connect {
356                        callback(&mut raw, meta).await
357                    } else {
358                        Ok(())
359                    };
360
361                    match res {
362                        Ok(()) => return Ok(Floating::new_live(raw, guard)),
363                        Err(error) => {
364                            tracing::error!(%error, "error returned from after_connect");
365                            // The connection is broken, don't try to close nicely.
366                            let _ = raw.close_hard().await;
367
368                            // Fall through to the backoff.
369                        }
370                    }
371                }
372
373                // an IO error while connecting is assumed to be the system starting up
374                Ok(Err(Error::Io(e))) if e.kind() == std::io::ErrorKind::ConnectionRefused => (),
375
376                // We got a transient database error, retry.
377                Ok(Err(Error::Database(error))) if error.is_transient_in_connect_phase() => (),
378
379                // Any other error while connection should immediately
380                // terminate and bubble the error up
381                Ok(Err(e)) => return Err(e),
382
383                // timed out
384                Err(_) => return Err(Error::PoolTimedOut),
385            }
386
387            // If the connection is refused, wait in exponentially
388            // increasing steps for the server to come up,
389            // capped by a factor of the remaining time until the deadline
390            crate::rt::sleep(backoff).await;
391            backoff = cmp::min(backoff * 2, max_backoff);
392        }
393    }

395    /// Try to maintain `min_connections`, returning any errors (including `PoolTimedOut`).
396    pub async fn try_min_connections(self: &Arc<Self>, deadline: Instant) -> Result<(), Error> {
397        while self.size() < self.options.min_connections {
398            // Don't wait for a semaphore permit.
399            //
400            // If no extra permits are available then we shouldn't be trying to spin up
401            // connections anyway.
402            let Some(permit) = self.semaphore.try_acquire(1) else {
403                return Ok(());
404            };
405
406            // We must always obey `max_connections`.
407            let Some(guard) = self.try_increment_size(permit).ok() else {
408                return Ok(());
409            };
410
411            // We skip `after_release` since the connection was never provided to user code
412            // besides `after_connect`, if they set it.
413            self.release(self.connect(deadline, guard).await?);
414        }
415
416        Ok(())
417    }
418
419    /// Attempt to maintain `min_connections`, logging if unable.
420    pub async fn min_connections_maintenance(self: &Arc<Self>, deadline: Option<Instant>) {
421        let deadline = deadline.unwrap_or_else(|| {
422            // Arbitrary default deadline if the caller doesn't care.
423            Instant::now() + Duration::from_secs(300)
424        });
425
426        match self.try_min_connections(deadline).await {
427            Ok(()) => (),
428            Err(Error::PoolClosed) => (),
429            Err(Error::PoolTimedOut) => {
430                tracing::debug!("unable to complete `min_connections` maintenance before deadline")
431            }
432            Err(error) => tracing::debug!(%error, "error while maintaining min_connections"),
433        }
434    }

437impl<DB: Database> Drop for PoolInner<DB> {
438    fn drop(&mut self) {
439        self.mark_closed();
440
441        if let Some(parent) = &self.options.parent_pool {
442            // Release the stolen permits.
443            parent.0.semaphore.release(self.semaphore.permits());
444        }
445    }

448/// Returns `true` if the connection has exceeded `options.max_lifetime` if set, `false` otherwise.
449pub(super) fn is_beyond_max_lifetime<DB: Database>(
450    live: &Live<DB>,
451    options: &PoolOptions<DB>,
452) -> bool {
453    options
454        .max_lifetime
455        .map_or(false, |max| live.created_at.elapsed() > max)
456}
457
458/// Returns `true` if the connection has exceeded `options.idle_timeout` if set, `false` otherwise.
459fn is_beyond_idle_timeout<DB: Database>(idle: &Idle<DB>, options: &PoolOptions<DB>) -> bool {
460    options
461        .idle_timeout
462        .map_or(false, |timeout| idle.idle_since.elapsed() > timeout)
463}

465async fn check_idle_conn<DB: Database>(
466    mut conn: Floating<DB, Idle<DB>>,
467    options: &PoolOptions<DB>,
468) -> Result<Floating<DB, Live<DB>>, DecrementSizeGuard<DB>> {
469    if options.test_before_acquire {
470        // Check that the connection is still live
471        if let Err(error) = conn.ping().await {
472            // an error here means the other end has hung up or we lost connectivity
473            // either way we're fine to just discard the connection
474            // the error itself here isn't necessarily unexpected so WARN is too strong
475            tracing::info!(%error, "ping on idle connection returned error");
476            // connection is broken so don't try to close nicely
477            return Err(conn.close_hard().await);
478        }
479    }
480
481    if let Some(test) = &options.before_acquire {
482        let meta = conn.metadata();
483        match test(&mut conn.live.raw, meta).await {
484            Ok(false) => {
485                // connection was rejected by user-defined hook, close nicely
486                return Err(conn.close().await);
487            }
488
489            Err(error) => {
490                tracing::warn!(%error, "error from `before_acquire`");
491                // connection is broken so don't try to close nicely
492                return Err(conn.close_hard().await);
493            }
494
495            Ok(true) => {}
496        }
497    }
498
499    // No need to re-connect; connection is alive or we don't care
500    Ok(conn.into_live())
501}

503fn spawn_maintenance_tasks<DB: Database>(pool: &Arc<PoolInner<DB>>) {
504    // NOTE: use `pool_weak` for the maintenance tasks
505    // so they don't keep `PoolInner` from being dropped.
506    let pool_weak = Arc::downgrade(pool);
507
508    let period = match (pool.options.max_lifetime, pool.options.idle_timeout) {
509        (Some(it), None) | (None, Some(it)) => it,
510
511        (Some(a), Some(b)) => cmp::min(a, b),
512
513        (None, None) => {
514            if pool.options.min_connections > 0 {
515                crate::rt::spawn(async move {
516                    if let Some(pool) = pool_weak.upgrade() {
517                        pool.min_connections_maintenance(None).await;
518                    }
519                });
520            }
521
522            return;
523        }
524    };
525
526    // Immediately cancel this task if the pool is closed.
527    let mut close_event = pool.close_event();
528
529    crate::rt::spawn(async move {
530        let _ = close_event
531            .do_until(async {
532                // If the last handle to the pool was dropped while we were sleeping
533                while let Some(pool) = pool_weak.upgrade() {
534                    if pool.is_closed() {
535                        return;
536                    }
537
538                    let next_run = Instant::now() + period;
539
540                    // Go over all idle connections, check for idleness and lifetime,
541                    // and if we have fewer than min_connections after reaping a connection,
542                    // open a new one immediately. Note that other connections may be popped from
543                    // the queue in the meantime - that's fine, there is no harm in checking more
544                    for _ in 0..pool.num_idle() {
545                        if let Some(conn) = pool.try_acquire() {
546                            if is_beyond_idle_timeout(&conn, &pool.options)
547                                || is_beyond_max_lifetime(&conn, &pool.options)
548                            {
549                                let _ = conn.close().await;
550                                pool.min_connections_maintenance(Some(next_run)).await;
551                            } else {
552                                pool.release(conn.into_live());
553                            }
554                        }
555                    }
556
557                    // Don't hold a reference to the pool while sleeping.
558                    drop(pool);
559
560                    if let Some(duration) = next_run.checked_duration_since(Instant::now()) {
561                        // `async-std` doesn't have a `sleep_until()`
562                        crate::rt::sleep(duration).await;
563                    } else {
564                        // `next_run` is in the past, just yield.
565                        crate::rt::yield_now().await;
566                    }
567                }
568            })
569            .await;
570    });
571}

573/// RAII guard returned by `Pool::try_increment_size()` and others.
574///
575/// Will decrement the pool size if dropped, to avoid semantically "leaking" connections
576/// (where the pool thinks it has more connections than it does).
577pub(in crate::pool) struct DecrementSizeGuard<DB: Database> {
578    pub(crate) pool: Arc<PoolInner<DB>>,
579    cancelled: bool,
580}
581
582impl<DB: Database> DecrementSizeGuard<DB> {
583    /// Create a new guard that will release a semaphore permit on-drop.
584    pub fn new_permit(pool: Arc<PoolInner<DB>>) -> Self {
585        Self {
586            pool,
587            cancelled: false,
588        }
589    }
590
591    pub fn from_permit(pool: Arc<PoolInner<DB>>, permit: AsyncSemaphoreReleaser<'_>) -> Self {
592        // here we effectively take ownership of the permit
593        permit.disarm();
594        Self::new_permit(pool)
595    }
596
597    /// Release the semaphore permit without decreasing the pool size.
598    ///
599    /// If the permit was stolen from the pool's parent, it will be returned to the child's semaphore.
600    fn release_permit(self) {
601        self.pool.semaphore.release(1);
602        self.cancel();
603    }
604
605    pub fn cancel(mut self) {
606        self.cancelled = true;
607    }
608}
609
610impl<DB: Database> Drop for DecrementSizeGuard<DB> {
611    fn drop(&mut self) {
612        if !self.cancelled {
613            self.pool.size.fetch_sub(1, Ordering::AcqRel);
614
615            // and here we release the permit we got on construction
616            self.pool.semaphore.release(1);
617        }
618    }
