1//! Conversions between Rust and SQL types.
2//!
3//! To see how each SQL type maps to a Rust type, see the corresponding `types` module for each
4//! database:
5//!
6//!  * [PostgreSQL](crate::postgres::types)
7//!  * [MySQL](crate::mysql::types)
8//!  * [SQLite](crate::sqlite::types)
9//!  * [MSSQL](crate::mssql::types)
10//!
11//! Any external types that have had [`Type`] implemented for, are re-exported in this module
12//! for convenience as downstream users need to use a compatible version of the external crate
13//! to take advantage of the implementation.
14//!
15//! # Nullable
16//!
17//! To represent nullable SQL types, `Option<T>` is supported where `T` implements `Type`.
18//! An `Option<T>` represents a potentially `NULL` value from SQL.

98/// Indicates that a SQL type is supported for a database.
99///
100/// ## Compile-time verification
101///
102/// Type definitions are *not* verified against the database at compile-time.
103/// The [`query!()`](macro.query.html) macros have no implicit knowledge of user-defined types.
104///
105/// When using custom types in query parameters or output columns with `query!()`,
106/// the use of [type overrides](macro.query.html#type-overrides-bind-parameters-postgres-only) is required.
107///
108/// ```rust,ignore
109/// struct MyUser { id: UserId, name: String }
110///
111/// // fetch all properties from user and override the type in Rust for `id`
112/// let user = query_as!(MyUser, r#"SELECT users.*, id as "id: UserId" FROM users"#)
113///     .fetch_one(&pool).await?;
114/// ```
115///
116/// ## Derivable
117///
118/// This trait can be derived by SQLx to support Rust-only wrapper types, enumerations, and (for
119/// postgres) structured records. Additionally, an implementation of [`Encode`](crate::encode::Encode) and [`Decode`](crate::decode::Decode) is
120/// generated.
121///
122/// ### Transparent
123///
124/// Rust-only domain wrappers around SQL types. The generated implementations directly delegate
125/// to the implementation of the inner type.
126///
127/// ```rust,ignore
128/// #[derive(sqlx::Type)]
129/// #[sqlx(transparent)]
130/// struct UserId(i64);
131/// ```
132///
133/// ##### Note: `PgHasArrayType`
134/// If you have the `postgres` feature enabled, this derive also generates a `PgHasArrayType` impl
135/// so that you may use it with `Vec` and other types that decode from an array in Postgres:
136///
137/// ```rust,ignore
138/// let user_ids: Vec<UserId> = sqlx::query_scalar("select '{ 123, 456 }'::int8[]")
139///    .fetch(&mut pg_connection)
140///    .await?;
141/// ```
142///
143/// However, if you are wrapping a type that does not implement `PgHasArrayType`
144/// (e.g. `Vec` itself, because we don't currently support multidimensional arrays),
145/// you may receive an error:
146///
147/// ```rust,ignore
148/// #[derive(sqlx::Type)] // ERROR: `Vec<i64>` does not implement `PgHasArrayType`
149/// #[sqlx(transparent)]
150/// struct UserIds(Vec<i64>);
151/// ```
152///
153/// To remedy this, add `#[sqlx(no_pg_array)]`, which disables the generation
154/// of the `PgHasArrayType` impl:
155///
156/// ```rust,ignore
157/// #[derive(sqlx::Type)]
158/// #[sqlx(transparent, no_pg_array)]
159/// struct UserIds(Vec<i64>);
160/// ```
161///
162/// ##### Attributes
163///
164/// * `#[sqlx(type_name = "<SQL type name>")]` on struct definition: instead of inferring the SQL
165///   type name from the inner field (in the above case, `BIGINT`), explicitly set it to
166///   `<SQL type name>` instead. May trigger errors or unexpected behavior if the encoding of the
167///   given type is different than that of the inferred type (e.g. if you rename the above to
168///   `VARCHAR`). Affects Postgres only.
169/// * `#[sqlx(rename_all = "<strategy>")]` on struct definition: See [`derive docs in FromRow`](crate::from_row::FromRow#rename_all)
170/// * `#[sqlx(no_pg_array)]`: do not emit a `PgHasArrayType` impl (see above).
171///
172/// ### Enumeration
173///
174/// Enumerations may be defined in Rust and can match SQL by
175/// integer discriminant or variant name.
176///
177/// With `#[repr(_)]` the integer representation is used when converting from/to SQL and expects
178/// that SQL type (e.g., `INT`). Without, the names of the variants are used instead and
179/// expects a textual SQL type (e.g., `VARCHAR`, `TEXT`).
180///
181/// ```rust,ignore
182/// #[derive(sqlx::Type)]
183/// #[repr(i32)]
184/// enum Color { Red = 1, Green = 2, Blue = 3 }
185/// ```
186///
187/// ```rust,ignore
188/// #[derive(sqlx::Type)]
189/// #[sqlx(type_name = "color")] // only for PostgreSQL to match a type definition
190/// #[sqlx(rename_all = "lowercase")]
191/// enum Color { Red, Green, Blue }
192/// ```
193///
194/// ### Records
195///
196/// User-defined composite types are supported through deriving a `struct`.
197///
198/// This is only supported for PostgreSQL.
199///
200/// ```rust,ignore
201/// #[derive(sqlx::Type)]
202/// #[sqlx(type_name = "interface_type")]
203/// struct InterfaceType {
204///     name: String,
205///     supplier_id: i32,
206///     price: f64
207/// }
208/// ```
209pub trait Type<DB: Database> {
210    /// Returns the canonical SQL type for this Rust type.
211    ///
212    /// When binding arguments, this is used to tell the database what is about to be sent; which,
213    /// the database then uses to guide query plans. This can be overridden by `Encode::produces`.
214    ///
215    /// A map of SQL types to Rust types is populated with this and used
216    /// to determine the type that is returned from the anonymous struct type from `query!`.
217    fn type_info() -> DB::TypeInfo;
218
219    /// Determines if this Rust type is compatible with the given SQL type.
220    ///
221    /// When decoding values from a row, this method is checked to determine if we should continue
222    /// or raise a runtime type mismatch error.
223    ///
224    /// When binding arguments with `query!` or `query_as!`, this method is consulted to determine
225    /// if the Rust type is acceptable.
226    ///
227    /// Defaults to checking [`TypeInfo::type_compatible()`].
228    fn compatible(ty: &DB::TypeInfo) -> bool {
229        Self::type_info().type_compatible(ty)
230    }
231}
232
233// for references, the underlying SQL type is identical
234impl<T: ?Sized + Type<DB>, DB: Database> Type<DB> for &'_ T {
235    fn type_info() -> DB::TypeInfo {
236        <T as Type<DB>>::type_info()
237    }
238
239    fn compatible(ty: &DB::TypeInfo) -> bool {
240        <T as Type<DB>>::compatible(ty)
241    }
242}
243
244// for optionals, the underlying SQL type is identical
245impl<T: Type<DB>, DB: Database> Type<DB> for Option<T> {
246    fn type_info() -> DB::TypeInfo {
247        <T as Type<DB>>::type_info()
248    }
249
250    fn compatible(ty: &DB::TypeInfo) -> bool {
251        ty.is_null() || <T as Type<DB>>::compatible(ty)
252    }
253}
