32/// Flattens all the `#[doc = "..."]` attributes into a single string.
33pub fn docs_str(attrs: &[Attribute]) -> String {
34    let mut doc = String::new();
35    for attr in docs(attrs) {
36        let syn::Meta::NameValue(syn::MetaNameValue {
37            value: syn::Expr::Lit(syn::ExprLit { lit: syn::Lit::Str(s), .. }),
38            ..
39        }) = &attr.meta
40        else {
41            continue;
42        };
43
44        let value = s.value();
45        if !value.is_empty() {
46            if !doc.is_empty() {
47                doc.push('\n');
48            }
49            doc.push_str(&value);
50        }
51    }
52    doc
53}

119impl SolAttrs {
120    /// Parse the `#[sol(...)]` attributes from a list of attributes.
121    pub fn parse(attrs: &[Attribute]) -> Result<(Self, Vec<Attribute>)> {
122        let mut this = Self::default();
123        let mut others = Vec::with_capacity(attrs.len());
124        for attr in attrs {
125            if !attr.path().is_ident("sol") {
126                others.push(attr.clone());
127                continue;
128            }
129
130            attr.meta.require_list()?.parse_nested_meta(|meta| {
131                let path = meta.path.get_ident().ok_or_else(|| meta.error("expected ident"))?;
132                let s = path.to_string();
133
134                macro_rules! match_ {
135                    ($($l:ident => $e:expr),* $(,)?) => {
136                        match s.as_str() {
137                            $(
138                                stringify!($l) => if this.$l.is_some() {
139                                    return Err(meta.error(DUPLICATE_ERROR))
140                                } else {
141                                    this.$l = Some($e);
142                                },
143                            )*
144                            _ => return Err(meta.error(UNKNOWN_ERROR)),
145                        }
146                    };
147                }
148
149                // `path` => true, `path = <bool>` => <bool>
150                let bool = || {
151                    if let Ok(input) = meta.value() {
152                        input.parse::<LitBool>().map(|lit| lit.value)
153                    } else {
154                        Ok(true)
155                    }
156                };
157
158                // `path = <path>`
159                let path = || meta.value()?.parse::<Path>();
160
161                // `path = "<str>"`
162                let lit = || {
163                    let value = meta.value()?;
164                    let span = value.span();
165                    let macro_string::MacroString(value) =
166                        value.parse::<macro_string::MacroString>()?;
167                    Ok::<_, syn::Error>(LitStr::new(&value, span))
168                };
169
170                // `path = "0x<hex>"`
171                let bytes = || {
172                    let lit = lit()?;
173                    if let Err(e) = hex::check(lit.value()) {
174                        let msg = format!("invalid hex value: {e}");
175                        return Err(Error::new(lit.span(), msg));
176                    }
177                    Ok(lit)
178                };
179
180                // `path(comma, separated, list)`
181                fn list<T>(
182                    meta: &ParseNestedMeta<'_>,
183                    parser: fn(syn::parse::ParseStream<'_>) -> Result<T>,
184                ) -> Result<Vec<T>> {
185                    let content;
186                    syn::parenthesized!(content in meta.input);
187                    Ok(content.parse_terminated(parser, Token![,])?.into_iter().collect())
188                }
189
190                match_! {
191                    rpc => bool()?,
192                    abi => bool()?,
193                    all_derives => bool()?,
194                    extra_derives => list(&meta, Path::parse)?,
195                    extra_methods => bool()?,
196                    docs => bool()?,
197
198                    alloy_sol_types => path()?,
199                    alloy_contract => path()?,
200
201                    rename => lit()?,
202                    rename_all => CasingStyle::from_lit(&lit()?)?,
203
204                    bytecode => bytes()?,
205                    deployed_bytecode => bytes()?,
206
207                    type_check => lit()?,
208                    ignore_unlinked => bool()?,
209                };
210                Ok(())
211            })?;
212        }
213        Ok((this, others))
214    }
215}

277/// Defines the casing for the attributes long representation.
278#[derive(Clone, Copy, Debug, PartialEq, Eq)]
279pub enum CasingStyle {
280    /// Indicate word boundaries with uppercase letter, excluding the first
281    /// word.
282    Camel,
283    /// Keep all letters lowercase and indicate word boundaries with hyphens.
284    Kebab,
285    /// Indicate word boundaries with uppercase letter, including the first
286    /// word.
287    Pascal,
288    /// Keep all letters uppercase and indicate word boundaries with
289    /// underscores.
290    ScreamingSnake,
291    /// Keep all letters lowercase and indicate word boundaries with
292    /// underscores.
293    Snake,
294    /// Keep all letters lowercase and remove word boundaries.
295    Lower,
296    /// Keep all letters uppercase and remove word boundaries.
297    Upper,
298    /// Use the original attribute name defined in the code.
299    Verbatim,
300}
301
302impl CasingStyle {
303    fn from_lit(name: &LitStr) -> Result<Self> {
304        let normalized = name.value().to_upper_camel_case().to_lowercase();
305        let s = match normalized.as_ref() {
306            "camel" | "camelcase" => Self::Camel,
307            "kebab" | "kebabcase" => Self::Kebab,
308            "pascal" | "pascalcase" => Self::Pascal,
309            "screamingsnake" | "screamingsnakecase" => Self::ScreamingSnake,
310            "snake" | "snakecase" => Self::Snake,
311            "lower" | "lowercase" => Self::Lower,
312            "upper" | "uppercase" => Self::Upper,
313            "verbatim" | "verbatimcase" => Self::Verbatim,
314            s => return Err(Error::new(name.span(), format!("unsupported casing: {s}"))),
315        };
316        Ok(s)
317    }
318
319    /// Apply the casing style to the given string.
320    #[allow(dead_code)]
321    pub fn apply(self, s: &str) -> String {
322        match self {
323            Self::Pascal => s.to_upper_camel_case(),
324            Self::Kebab => s.to_kebab_case(),
325            Self::Camel => s.to_lower_camel_case(),
326            Self::ScreamingSnake => s.to_shouty_snake_case(),
327            Self::Snake => s.to_snake_case(),
328            Self::Lower => s.to_snake_case().replace('_', ""),
329            Self::Upper => s.to_shouty_snake_case().replace('_', ""),
330            Self::Verbatim => s.to_owned(),
331        }
332    }
333}
