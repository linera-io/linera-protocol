8use std::error;
9use std::error::Error as ErrorT;
10use std::fmt;
11
12use ascii_utils;
13
14#[derive(Debug, PartialEq, Eq)]
15pub enum ParseError {
16    NoLocalPart,
17    NoDomainPart,
18    NoSignAt,
19
20    TooAt,
21    LocalTooLong,
22    DomainTooLong,
23    LabelTooLong,
24
25    LocalStartPeriod,
26    LocalEndPeriod,
27    DomainStartPeriod,
28    DomainEndPeriod,
29    ConsecutivePeriod,
30    NoPeriodDomain,
31
32    Ascii(ascii_utils::AsciiError),
33    WrongCharLocal(char),
34    WrongCharDomain(char),
35    WrongStartLabel(char),
36    WrongEndLabel(char),
37}
38
39impl From<ascii_utils::AsciiError> for ParseError {
40    fn from(err: ascii_utils::AsciiError) -> ParseError {
41        ParseError::Ascii(err)
42    }
43}
44
45impl error::Error for ParseError {
46    fn cause(&self) -> Option<&error::Error> {
47        match *self {
48            ParseError::Ascii(ref err) => Some(err),
49            _ => None,
50        }
51    }
52    fn description(&self) -> &str {
53        match *self {
54            ParseError::NoLocalPart => "no local part",
55            ParseError::NoDomainPart => "no domain part",
56            ParseError::NoSignAt => "no at sign (@)",
57
58            ParseError::TooAt => "wrong number of at sign (@)",
59            ParseError::LocalTooLong => "the local part has more than 64 characters",
60            ParseError::DomainTooLong => "the domain part has more than 255 characters",
61            ParseError::LabelTooLong => "a domain label has more than 63 characters",
62
63            ParseError::LocalStartPeriod => "the local part starts with a period",
64            ParseError::LocalEndPeriod => "the local part ends with a period",
65            ParseError::DomainStartPeriod => "the domain part starts with a period",
66            ParseError::DomainEndPeriod => "the domain part ends with a period",
67            ParseError::ConsecutivePeriod => "appear two or more consecutive periods",
68            ParseError::NoPeriodDomain => "no period at domain part",
69
70            ParseError::Ascii(ref err) => err.description(),
71            ParseError::WrongCharLocal(_) => "character not valid in local part",
72            ParseError::WrongCharDomain(_) => "character not valid in domain part",
73            ParseError::WrongStartLabel(_) => "character not valid at start of domain label",
74            ParseError::WrongEndLabel(_) => "character not valid at end of domain label",
75        }
76    }
77}
78
79const MSG_ERR: &'static str = "invalid email address";
80impl fmt::Display for ParseError {
81    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
82        match *self {
83            ParseError::WrongCharLocal(ch) => {
84                write!(f, "{}: {} ({})", MSG_ERR, self.description(), ch)
85            }
86            ParseError::WrongCharDomain(ch) => {
87                write!(f, "{}: {} ({})", MSG_ERR, self.description(), ch)
88            }
89            ParseError::WrongStartLabel(ch) => {
90                write!(f, "{}: {} ({})", MSG_ERR, self.description(), ch)
91            }
92            ParseError::WrongEndLabel(ch) => {
93                write!(f, "{}: {} ({})", MSG_ERR, self.description(), ch)
94            }
95
96            _ => write!(f, "{}: {}", MSG_ERR, self.description()),
97        }
98    }
99}
