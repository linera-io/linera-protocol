8use ascii_utils;
9
10use errors::*;
11
12// https://tools.ietf.org/html/rfc5321#section-4.5.3.1
13//
14// The maximum total length of a user name or other local-part is 64 octets.
15// The maximum total length of a domain name or number is 255 octets.
16const MAX_LOCAL_PART: usize = 64;
17const MAX_DOMAIN_PART: usize = 255;
18const MAX_LABEL: usize = 63;
19
20/// `is_valid_email` checks wheter an email address is valid.
21pub fn is_valid_email(address: &str) -> bool {
22    parse_email(address).is_ok()
23}
24
25/// `parse_email` scans an email address to check wheter it is correct.
26pub fn parse_email(address: &str) -> Result<(), ParseError> {
27    if address.starts_with('@') {
28        return Err(ParseError::NoLocalPart);
29    }
30    if address.ends_with('@') {
31        return Err(ParseError::NoDomainPart);
32    }
33
34    // https://tools.ietf.org/html/rfc5321#section-4.1.2
35    //
36    // Systems MUST NOT define mailboxes in such a way as to require the use in
37    // SMTP of non-ASCII characters (octets with the high order bit set to one)
38    // or ASCII "control characters" (decimal value 0-31 and 127).
39    ascii_utils::check_ascii_printable(address)?;
40
41    let mut address_iter = address.split('@');
42    let local = address_iter.next().unwrap();
43    let domain = address_iter.next().ok_or(ParseError::NoSignAt)?;
44    if address_iter.next().is_some() {
45        return Err(ParseError::TooAt);
46    }
47
48    // == Local part
49
50    // https://tools.ietf.org/html/rfc3696#section-3
51    //
52    // Period (".") may also appear, but may not be used to start or end
53    // the local part, nor may two or more consecutive periods appear.
54
55    if local.len() > MAX_LOCAL_PART {
56        return Err(ParseError::LocalTooLong);
57    }
58    if local.starts_with('.') {
59        return Err(ParseError::LocalStartPeriod);
60    }
61    if local.ends_with('.') {
62        return Err(ParseError::LocalEndPeriod);
63    }
64
65    let mut last_period: bool = false;
66    for ch in local.chars() {
67        if ascii_utils::Check::is_letter(ch) || ascii_utils::Check::is_digit(ch) {
68            if last_period {
69                last_period = false;
70            }
71            continue;
72        }
73
74        match ch {
75            // atom :: https://tools.ietf.org/html/rfc5322#section-3.2.3
76            '!' | '#' | '$' | '%' | '&' | '\'' | '*' | '+' | '-' | '/' | '=' | '?' | '^' |
77            '_' | '`' | '{' | '|' | '}' | '~' => {
78                if last_period {
79                    last_period = false;
80                }
81            }
82            '.' => {
83                if last_period {
84                    return Err(ParseError::ConsecutivePeriod);
85                }
86                last_period = true;
87            }
88            _ => return Err(ParseError::WrongCharLocal(ch)),
89        }
90    }
91
92    // == Domain part
93
94    // https://tools.ietf.org/html/rfc5321#section-4.1.2
95    //
96    // characters outside the set of alphabetic characters, digits, and hyphen
97    // MUST NOT appear in domain name labels for SMTP clients or servers.  In
98    // particular, the underscore character is not permitted.
99
100    // https://tools.ietf.org/html/rfc1034#section-3.5
101    //
102    // The labels must follow the rules for ARPANET host names.  They must start
103    // with a letter, end with a letter or digit, and have as interior
104    // characters only letters, digits, and hyphen.  There are also some
105    // restrictions on the length.  Labels must be 63 characters or less.
106
107    // https://tools.ietf.org/html/rfc3696#section-2
108    //
109    // It is likely that the better strategy has now become to make the "at
110    // least one period" test, to verify LDH conformance (including verification
111    // that the apparent TLD name is not all-numeric), and then to use the DNS
112    // to determine domain name validity, rather than trying to maintain a local
113    // list of valid TLD names.
114
115    if domain.len() > MAX_DOMAIN_PART {
116        return Err(ParseError::DomainTooLong);
117    }
118    if domain.starts_with('.') {
119        return Err(ParseError::DomainStartPeriod);
120    }
121    if domain.ends_with('.') {
122        return Err(ParseError::DomainEndPeriod);
123    }
124
125    let labels: Vec<&str> = domain.split('.').collect();
126    if labels.len() == 1 {
127        return Err(ParseError::NoPeriodDomain);
128    }
129
130    // label = let-dig [ [ ldh-str ] let-dig ]
131	// limited to a length of 63 characters by RFC 1034 section 3.5
132	//
133	// <let-dig> ::= <letter> | <digit>
134	// <ldh-str> ::= <let-dig-hyp> | <let-dig-hyp> <ldh-str>
135	// <let-dig-hyp> ::= <let-dig> | "-"
136    for label in labels {
137        if label.is_empty() {
138            return Err(ParseError::ConsecutivePeriod);
139        }
140        if label.len() > MAX_LABEL {
141            return Err(ParseError::LabelTooLong);
142        }
143
144        if let Some(ch) = label.chars().find(|&x| {
145            !ascii_utils::Check::is_letter(x) && !ascii_utils::Check::is_digit(x) && x != '-'
146        }) {
147            return Err(ParseError::WrongCharDomain(ch));
148        }
149
150        let label_bytes = label.as_bytes();
151
152        let first_char = label_bytes[0];
153        if !ascii_utils::Check::is_letter(first_char) && !ascii_utils::Check::is_digit(first_char) {
154            return Err(ParseError::WrongStartLabel(label_bytes[0] as char));
155        }
156        let last_char = label_bytes[label_bytes.len() - 1];
157        if !ascii_utils::Check::is_letter(last_char) && !ascii_utils::Check::is_digit(last_char) {
158            return Err(ParseError::WrongEndLabel(last_char as char));
159        }
160    }
161
162    Ok(())
163}

165// == Tests

168#[test]
169fn test_length() {
170    let local_part = "a".repeat(MAX_LOCAL_PART);
171
172    let label = format!("{}.", "x".repeat(MAX_LABEL));
173    let all_labels = format!("{}", label.repeat(3));
174    let last_label = "y".repeat(MAX_DOMAIN_PART - all_labels.len());
175
176    let input_ok = format!("{}@{}{}", local_part, all_labels, last_label);
177    assert_eq!(parse_email(&input_ok), Ok(()));
178
179    // == Errors
180
181    let mut input_err = format!("a{}@{}{}", local_part, all_labels, last_label);
182    assert_eq!(parse_email(&input_err), Err(ParseError::LocalTooLong));
183
184    input_err = format!("{}@{}{}z", local_part, all_labels, last_label);
185    assert_eq!(parse_email(&input_err), Err(ParseError::DomainTooLong));
186
187    input_err = format!("{}@{}x{}", local_part, label, last_label);
188    assert_eq!(parse_email(&input_err), Err(ParseError::LabelTooLong));
189}
