66struct ServerContext {
67    server_config: ValidatorServerConfig,
68    cross_chain_config: CrossChainConfig,
69    notification_config: NotificationConfig,
70    shard: Option<usize>,
71    block_time_grace_period: Duration,
72    chain_worker_ttl: Duration,
73    block_cache_size: usize,
74    execution_state_cache_size: usize,
75    chain_info_max_received_log_entries: usize,
76}

78impl ServerContext {
79    fn make_shard_state<S>(
80        &self,
81        local_ip_addr: &str,
82        shard_id: ShardId,
83        storage: S,
84    ) -> (WorkerState<S>, ShardId, ShardConfig)
85    where
86        S: Storage + Clone + Send + Sync + 'static,
87    {
88        let shard = self.server_config.internal_network.shard(shard_id);
89        info!("Shard booted on {}", shard.host);
90        info!(
91            "Public key: {}",
92            self.server_config.validator_secret.public()
93        );
94        let state = WorkerState::new(
95            format!("Shard {} @ {}:{}", shard_id, local_ip_addr, shard.port),
96            Some(self.server_config.validator_secret.copy()),
97            storage,
98            self.block_cache_size,
99            self.execution_state_cache_size,
100        )
101        .with_allow_inactive_chains(false)
102        .with_allow_messages_from_deprecated_epochs(false)
103        .with_block_time_grace_period(self.block_time_grace_period)
104        .with_chain_worker_ttl(self.chain_worker_ttl)
105        .with_chain_info_max_received_log_entries(self.chain_info_max_received_log_entries);
106        (state, shard_id, shard.clone())
107    }
108
109    fn spawn_simple<S>(
110        &self,
111        listen_address: &str,
112        states: Vec<(WorkerState<S>, ShardId, ShardConfig)>,
113        protocol: simple::TransportProtocol,
114        shutdown_signal: CancellationToken,
115    ) -> JoinSet<()>
116    where
117        S: Storage + Clone + Send + Sync + 'static,
118    {
119        let mut join_set = JoinSet::new();
120        let handles = FuturesUnordered::new();
121
122        let internal_network = self
123            .server_config
124            .internal_network
125            .clone_with_protocol(protocol);
126
127        for (state, shard_id, shard) in states {
128            let internal_network = internal_network.clone();
129            let cross_chain_config = self.cross_chain_config.clone();
130            let listen_address = listen_address.to_owned();
131
132            #[cfg(with_metrics)]
133            if let Some(port) = shard.metrics_port {
134                monitoring_server::start_metrics(
135                    (listen_address.clone(), port),
136                    shutdown_signal.clone(),
137                );
138            }
139
140            let server_handle = simple::Server::new(
141                internal_network,
142                listen_address,
143                shard.port,
144                state,
145                shard_id,
146                cross_chain_config,
147            )
148            .spawn(shutdown_signal.clone(), &mut join_set);
149
150            handles.push(
151                server_handle
152                    .join()
153                    .inspect_err(move |error| {
154                        error!("Error running server for shard {shard_id}: {error:?}")
155                    })
156                    .map(|_| ()),
157            );
158        }
159
160        join_set.spawn_task(handles.collect::<()>());
161
162        join_set
163    }
164
165    fn spawn_grpc<S>(
166        &self,
167        listen_address: &str,
168        states: Vec<(WorkerState<S>, ShardId, ShardConfig)>,
169        shutdown_signal: CancellationToken,
170    ) -> JoinSet<()>
171    where
172        S: Storage + Clone + Send + Sync + 'static,
173    {
174        let mut join_set = JoinSet::new();
175        let handles = FuturesUnordered::new();
176
177        for (state, shard_id, shard) in states {
178            #[cfg(with_metrics)]
179            if let Some(port) = shard.metrics_port {
180                monitoring_server::start_metrics(
181                    (listen_address.to_string(), port),
182                    shutdown_signal.clone(),
183                );
184            }
185
186            let server_handle = grpc::GrpcServer::spawn(
187                listen_address.to_string(),
188                shard.port,
189                state,
190                shard_id,
191                self.server_config.internal_network.clone(),
192                self.cross_chain_config.clone(),
193                self.notification_config.clone(),
194                shutdown_signal.clone(),
195                &mut join_set,
196            );
197
198            handles.push(
199                server_handle
200                    .join()
201                    .inspect_err(move |error| {
202                        error!("Error running server for shard {shard_id}: {error:?}")
203                    })
204                    .map(|_| ()),
205            );
206        }
207
208        join_set.spawn_task(handles.collect::<()>());
209
210        join_set
211    }
212
213    fn get_listen_address() -> String {
214        // Allow local IP address to be different from the public one.
215        "0.0.0.0".to_string()
216    }
217}

219#[async_trait]
220impl Runnable for ServerContext {
221    type Output = anyhow::Result<()>;
222
223    async fn run<S>(self, storage: S) -> anyhow::Result<()>
224    where
225        S: Storage + Clone + Send + Sync + 'static,
226    {
227        let shutdown_notifier = CancellationToken::new();
228        let listen_address = Self::get_listen_address();
229
230        tokio::spawn(listen_for_shutdown_signals(shutdown_notifier.clone()));
231
232        // Run the server
233        let states = match self.shard {
234            Some(shard) => {
235                info!("Running shard number {}", shard);
236                vec![self.make_shard_state(&listen_address, shard, storage)]
237            }
238            None => {
239                info!("Running all shards");
240                let num_shards = self.server_config.internal_network.shards.len();
241                (0..num_shards)
242                    .map(|shard| self.make_shard_state(&listen_address, shard, storage.clone()))
243                    .collect()
244            }
245        };
246
247        let mut join_set = match self.server_config.internal_network.protocol {
248            NetworkProtocol::Simple(protocol) => {
249                self.spawn_simple(&listen_address, states, protocol, shutdown_notifier)
250            }
251            NetworkProtocol::Grpc(tls_config) => match tls_config {
252                TlsConfig::ClearText => self.spawn_grpc(&listen_address, states, shutdown_notifier),
253                TlsConfig::Tls => bail!("TLS not supported between proxy and shards."),
254            },
255        };
256
257        join_set.await_all_tasks().await;
258
259        Ok(())
260    }
261}

323fn make_server_config<R: CryptoRng>(
324    path: &Path,
325    rng: &mut R,
326    options: ValidatorOptions,
327) -> anyhow::Result<persistent::File<ValidatorServerConfig>> {
328    let validator_keypair = ValidatorKeypair::generate_from(rng);
329    let account_secret = AccountSecretKey::Ed25519(Ed25519SecretKey::generate_from(rng));
330    let public_key = validator_keypair.public_key;
331    let network = ValidatorPublicNetworkConfig {
332        protocol: options.external_protocol,
333        host: options.host,
334        port: options.port,
335    };
336    let internal_network = ValidatorInternalNetworkConfig {
337        public_key,
338        protocol: options.internal_protocol,
339        shards: options.shards,
340        block_exporters: options.block_exporters,
341        proxies: options.proxies,
342    };
343    let validator = ValidatorConfig {
344        network,
345        public_key,
346        account_key: account_secret.public(),
347    };
348    Ok(persistent::File::new(
349        path,
350        ValidatorServerConfig {
351            validator,
352            validator_secret: validator_keypair.secret_key,
353            internal_network,
354        },
355    )?)
356}

465fn main() {
466    let options = <ServerOptions as clap::Parser>::parse();
467
468    let mut runtime = if options.tokio_threads == Some(1) {
469        tokio::runtime::Builder::new_current_thread()
470    } else {
471        let mut builder = tokio::runtime::Builder::new_multi_thread();
472
473        if let Some(threads) = options.tokio_threads {
474            builder.worker_threads(threads);
475        }
476
477        builder
478    };
479
480    if let Some(blocking_threads) = options.tokio_blocking_threads {
481        runtime.max_blocking_threads(blocking_threads);
482    }
483
484    runtime
485        .enable_all()
486        .build()
487        .expect("Failed to create Tokio runtime")
488        .block_on(run(options))
489}

491/// Returns the log file name to use based on the [`ServerCommand`] that will run.
492fn otlp_exporter_endpoint_for(command: &ServerCommand) -> Option<&str> {
493    match command {
494        ServerCommand::Run {
495            otlp_exporter_endpoint,
496            ..
497        } => otlp_exporter_endpoint.as_deref(),
498        ServerCommand::Generate { .. } | ServerCommand::EditShards { .. } => None,
499    }

502fn log_file_name_for(command: &ServerCommand) -> Cow<'static, str> {
503    match command {
504        ServerCommand::Run {
505            shard,
506            server_config_path,
507            ..
508        } => {
509            let server_config: ValidatorServerConfig =
510                util::read_json(server_config_path).expect("Failed to read server config");
511            let public_key = &server_config.validator.public_key;
512
513            if let Some(shard) = shard {
514                format!("validator-{public_key}-shard-{shard}")
515            } else {
516                format!("validator-{public_key}")
517            }
518            .into()
519        }
520        ServerCommand::Generate { .. } | ServerCommand::EditShards { .. } => "server".into(),
521    }
522}

524async fn run(options: ServerOptions) {
525    linera_service::tracing::opentelemetry::init(
526        &log_file_name_for(&options.command),
527        otlp_exporter_endpoint_for(&options.command),
528    );
529
530    match options.command {
531        ServerCommand::Run {
532            server_config_path,
533            storage_config,
534            common_storage_options,
535            cross_chain_config,
536            notification_config,
537            shard,
538            block_time_grace_period,
539            wasm_runtime,
540            chain_worker_ttl,
541            chain_info_max_received_log_entries,
542            otlp_exporter_endpoint: _,
543        } => {
544            linera_version::VERSION_INFO.log();
545
546            let server_config: ValidatorServerConfig =
547                util::read_json(&server_config_path).expect("Failed to read server config");
548
549            let job = ServerContext {
550                server_config,
551                cross_chain_config,
552                notification_config,
553                shard,
554                block_time_grace_period,
555                chain_worker_ttl,
556                block_cache_size: options.block_cache_size,
557                execution_state_cache_size: options.execution_state_cache_size,
558                chain_info_max_received_log_entries,
559            };
560            let wasm_runtime = wasm_runtime.with_wasm_default();
561            let store_config = storage_config
562                .add_common_storage_options(&common_storage_options)
563                .unwrap();
564            // Validators should not output contract logs.
565            let allow_application_logs = false;
566            store_config
567                .run_with_storage(wasm_runtime, allow_application_logs, job)
568                .boxed()
569                .await
570                .unwrap()
571                .unwrap();
572        }
573
574        ServerCommand::Generate {
575            validators,
576            committee,
577            testing_prng_seed,
578        } => {
579            let mut config_validators = Vec::new();
580            let mut rng = Box::<dyn CryptoRng>::from(testing_prng_seed);
581            for options_path in validators {
582                let options_string = fs_err::tokio::read_to_string(options_path)
583                    .await
584                    .expect("Unable to read validator options file");
585                let options: ValidatorOptions =
586                    toml::from_str(&options_string).unwrap_or_else(|_| {
587                        panic!("Invalid options file format: \n {}", options_string)
588                    });
589                let path = options.server_config_path.clone();
590                let mut server = make_server_config(&path, &mut rng, options)
591                    .expect("Unable to open server config file");
592                Persist::persist(&mut server)
593                    .await
594                    .expect("Unable to write server config file");
595                info!("Wrote server config {}", path.to_str().unwrap());
596                println!(
597                    "{},{}",
598                    server.validator.public_key, server.validator.account_key
599                );
600                config_validators.push(Persist::into_value(server).validator);
601            }
602            if let Some(committee) = committee {
603                let mut config = persistent::File::new(
604                    &committee,
605                    CommitteeConfig {
606                        validators: config_validators,
607                    },
608                )
609                .expect("Unable to open committee configuration");
610                Persist::persist(&mut config)
611                    .await
612                    .expect("Unable to write committee description");
613                info!("Wrote committee config {}", committee.to_str().unwrap());
614            }
615        }
616
617        ServerCommand::EditShards {
618            server_config_path,
619            num_shards,
620            host,
621            port,
622            metrics_port,
623        } => {
624            let mut server_config =
625                persistent::File::<ValidatorServerConfig>::read(&server_config_path)
626                    .expect("Failed to read server config");
627            let shards = generate_shard_configs(num_shards, host, port, metrics_port)
628                .expect("Failed to generate shard configs");
629            server_config.internal_network.shards = shards;
630            Persist::persist(&mut server_config)
631                .await
632                .expect("Failed to write updated server config");
633        }
634    }
635}

637fn generate_shard_configs(
638    num_shards: String,
639    host: String,
640    port: String,
641    metrics_port: Option<String>,
642) -> anyhow::Result<Vec<ShardConfig>> {
643    let mut shards = Vec::new();
644    let len = num_shards.len();
645    let num_shards = num_shards
646        .parse::<NonZeroU16>()
647        .context("Failed to parse the number of shards")?;
648    let pattern = "%".repeat(len);
649
650    for i in 1u16..=num_shards.into() {
651        let index = format!("{i:0len$}");
652        let host = host.replacen(&pattern, &index, 1);
653        let port = port
654            .replacen(&pattern, &index, 1)
655            .parse()
656            .context("Failed to decode port into an integers")?;
657        let metrics_port = metrics_port
658            .as_ref()
659            .map(|port| {
660                port.replacen(&pattern, &index, 1)
661                    .parse()
662                    .context("Failed to decode metrics port into an integers")
663            })
664            .transpose()?;
665        let shard = ShardConfig {
666            host,
667            port,
668            metrics_port,
669        };
670        shards.push(shard);
671    }
672    Ok(shards)
673}
