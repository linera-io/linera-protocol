79impl ServerContext {
80    fn make_shard_state<S>(
81        &self,
82        local_ip_addr: &str,
83        shard_id: ShardId,
84        storage: S,
85    ) -> (WorkerState<S>, ShardId, ShardConfig)
86    where
87        S: Storage + Clone + Send + Sync + 'static,
88    {
89        let shard = self.server_config.internal_network.shard(shard_id);
90        info!("Shard booted on {}", shard.host);
91        info!(
92            "Public key: {}",
93            self.server_config.validator_secret.public()
94        );
95        let config = ChainWorkerConfig {
96            nickname: format!("Shard {} @ {}:{}", shard_id, local_ip_addr, shard.port),
97            key_pair: Some(Arc::new(self.server_config.validator_secret.copy())),
98            allow_inactive_chains: false,
99            allow_messages_from_deprecated_epochs: false,
100            block_time_grace_period: self.block_time_grace_period,
101            ttl: util::non_zero_duration(self.chain_worker_ttl),
102            chain_info_max_received_log_entries: self.chain_info_max_received_log_entries,
103            cross_chain_message_chunk_limit: self.cross_chain_message_chunk_limit,
104            block_cache_size: self.block_cache_size,
105            execution_state_cache_size: self.execution_state_cache_size,
106            allow_revert_confirm: self.allow_revert_confirm,
107            reset_on_corrupted_chain_state: self
108                .reset_on_corrupted_chain_state_mins
109                .map(|m| Duration::from_secs(m * 60)),
110            recovery_whitelist: self.recovery_whitelist.clone(),
111            ..ChainWorkerConfig::default()
112        };
113        let state = WorkerState::new(storage, config, None);
114        (state, shard_id, shard.clone())
115    }
116
117    #[cfg_attr(not(with_metrics), allow(unused_variables))]
118    fn spawn_simple<S>(
119        &self,
120        listen_address: &str,
121        states: Vec<(WorkerState<S>, ShardId, ShardConfig)>,
122        protocol: simple::TransportProtocol,
123        shutdown_signal: &CancellationToken,
124        enable_memory_profiling: bool,
125    ) -> JoinSet<()>
126    where
127        S: Storage + Clone + Send + Sync + 'static,
128    {
129        let mut join_set = JoinSet::new();
130        let handles = FuturesUnordered::new();
131
132        let internal_network = self
133            .server_config
134            .internal_network
135            .clone_with_protocol(protocol);
136
137        for (state, shard_id, shard) in states {
138            let internal_network = internal_network.clone();
139            let cross_chain_config = self.cross_chain_config.clone();
140            let listen_address = listen_address.to_owned();
141
142            #[cfg(with_metrics)]
143            if let Some(port) = shard.metrics_port {
144                monitoring_server::start_metrics(
145                    (listen_address.clone(), port),
146                    shutdown_signal.clone(),
147                    monitoring_server::MemoryProfiling::from(enable_memory_profiling),
148                );
149            }
150
151            let server_handle = simple::Server::new(
152                internal_network,
153                listen_address,
154                shard.port,
155                state,
156                shard_id,
157                cross_chain_config,
158            )
159            .spawn(shutdown_signal.clone(), &mut join_set);
160
161            handles.push(
162                server_handle
163                    .join()
164                    .inspect_err(move |error| {
165                        error!("Error running server for shard {shard_id}: {error:?}")
166                    })
167                    .map(|_| ()),
168            );
169        }
170
171        join_set.spawn_task(handles.collect::<()>());
172
173        join_set
174    }
175
176    #[cfg_attr(not(with_metrics), allow(unused_variables))]
177    fn spawn_grpc<S>(
178        &self,
179        listen_address: &str,
180        states: Vec<(WorkerState<S>, ShardId, ShardConfig)>,
181        shutdown_signal: &CancellationToken,
182        enable_memory_profiling: bool,
183    ) -> JoinSet<()>
184    where
185        S: Storage + Clone + Send + Sync + 'static,
186    {
187        let mut join_set = JoinSet::new();
188        let handles = FuturesUnordered::new();
189
190        for (state, shard_id, shard) in states {
191            #[cfg(with_metrics)]
192            if let Some(port) = shard.metrics_port {
193                monitoring_server::start_metrics(
194                    (listen_address.to_string(), port),
195                    shutdown_signal.clone(),
196                    monitoring_server::MemoryProfiling::from(enable_memory_profiling),
197                );
198            }
199
200            let server_handle = grpc::GrpcServer::spawn(
201                listen_address.to_string(),
202                shard.port,
203                state,
204                shard_id,
205                self.server_config.internal_network.clone(),
206                &self.cross_chain_config,
207                &self.notification_config,
208                shutdown_signal.clone(),
209                &mut join_set,
210            );
211
212            handles.push(
213                server_handle
214                    .join()
215                    .inspect_err(move |error| {
216                        error!("Error running server for shard {shard_id}: {error:?}")
217                    })
218                    .map(|_| ()),
219            );
220        }
221
222        join_set.spawn_task(handles.collect::<()>());
223
224        join_set
225    }
226
227    fn get_listen_address() -> String {
228        // Allow local IP address to be different from the public one.
229        "0.0.0.0".to_string()
230    }
231}

233#[async_trait]
234impl Runnable for ServerContext {
235    type Output = anyhow::Result<()>;
236
237    async fn run<S>(self, storage: S) -> anyhow::Result<()>
238    where
239        S: Storage + Clone + Send + Sync + 'static,
240    {
241        let shutdown_notifier = CancellationToken::new();
242        let listen_address = Self::get_listen_address();
243
244        tokio::spawn(listen_for_shutdown_signals(shutdown_notifier.clone()));
245
246        // Activate memory profiling once before per-shard start_metrics calls.
247        #[cfg(with_metrics)]
248        let enable_memory_profiling = {
249            let memory_profiling =
250                monitoring_server::MemoryProfiling::try_activate(self.enable_memory_profiling)
251                    .await;
252            memory_profiling == monitoring_server::MemoryProfiling::Enabled
253        };
254        #[cfg(not(with_metrics))]
255        let enable_memory_profiling = false;
256
257        // Run the server
258        let states = match self.shard {
259            Some(shard) => {
260                info!("Running shard number {}", shard);
261                vec![self.make_shard_state(&listen_address, shard, storage)]
262            }
263            None => {
264                info!("Running all shards");
265                let num_shards = self.server_config.internal_network.shards.len();
266                (0..num_shards)
267                    .map(|shard| self.make_shard_state(&listen_address, shard, storage.clone()))
268                    .collect()
269            }
270        };
271
272        let mut join_set = match self.server_config.internal_network.protocol {
273            NetworkProtocol::Simple(protocol) => self.spawn_simple(
274                &listen_address,
275                states,
276                protocol,
277                &shutdown_notifier,
278                enable_memory_profiling,
279            ),
280            NetworkProtocol::Grpc(tls_config) => match tls_config {
281                TlsConfig::ClearText => self.spawn_grpc(
282                    &listen_address,
283                    states,
284                    &shutdown_notifier,
285                    enable_memory_profiling,
286                ),
287                TlsConfig::Tls => bail!("TLS not supported between proxy and shards."),
288            },
289        };
290
291        join_set.await_all_tasks().await;
292
293        Ok(())
294    }
295}

376fn make_server_config<R: CryptoRng>(
377    path: &Path,
378    rng: &mut R,
379    options: ValidatorOptions,
380) -> anyhow::Result<persistent::File<ValidatorServerConfig>> {
381    let validator_keypair = ValidatorKeypair::generate_from(rng);
382    let account_secret = AccountSecretKey::Ed25519(Ed25519SecretKey::generate_from(rng));
383    let public_key = validator_keypair.public_key;
384    let network = ValidatorPublicNetworkConfig {
385        protocol: options.external_protocol,
386        host: options.host,
387        port: options.port,
388    };
389    let internal_network = ValidatorInternalNetworkConfig {
390        public_key,
391        protocol: options.internal_protocol,
392        shards: options.shards,
393        block_exporters: options.block_exporters,
394        proxies: options.proxies,
395    };
396    let validator = ValidatorConfig {
397        network,
398        public_key,
399        account_key: account_secret.public(),
400    };
401    Ok(persistent::File::new(
402        path,
403        ValidatorServerConfig {
404            validator,
405            validator_secret: validator_keypair.secret_key,
406            internal_network,
407        },
408    )?)
409}

547fn main() {
548    let options = <ServerOptions as clap::Parser>::parse();
549
550    let mut runtime = if options.tokio_threads == Some(1) {
551        tokio::runtime::Builder::new_current_thread()
552    } else {
553        let mut builder = tokio::runtime::Builder::new_multi_thread();
554
555        if let Some(threads) = options.tokio_threads {
556            builder.worker_threads(threads);
557        }
558
559        builder
560    };
561
562    if let Some(blocking_threads) = options.tokio_blocking_threads {
563        runtime.max_blocking_threads(blocking_threads);
564    }
565
566    runtime
567        .enable_all()
568        .build()
569        .expect("Failed to create Tokio runtime")
570        .block_on(run(options))
571}

573/// Returns the log file name to use based on the [`ServerCommand`] that will run.
574fn otlp_exporter_endpoint_for(command: &ServerCommand) -> Option<&str> {
575    match command {
576        ServerCommand::Run {
577            otlp_exporter_endpoint,
578            ..
579        } => otlp_exporter_endpoint.as_deref(),
580        ServerCommand::Generate { .. } | ServerCommand::EditShards { .. } => None,
581    }
582}

584fn log_file_name_for(command: &ServerCommand) -> Cow<'static, str> {
585    match command {
586        ServerCommand::Run {
587            shard,
588            server_config_path,
589            ..
590        } => {
591            let server_config: ValidatorServerConfig =
592                util::read_json(server_config_path).expect("Failed to read server config");
593            let public_key = &server_config.validator.public_key;
594
595            if let Some(shard) = shard {
596                format!("validator-{public_key}-shard-{shard}")
597            } else {
598                format!("validator-{public_key}")
599            }
600            .into()
601        }
602        ServerCommand::Generate { .. } | ServerCommand::EditShards { .. } => "server".into(),
603    }
604}

606async fn run(options: ServerOptions) {
607    linera_service::tracing::opentelemetry::init(
608        &log_file_name_for(&options.command),
609        otlp_exporter_endpoint_for(&options.command),
610    );
611
612    #[cfg(with_metrics)]
613    let enable_memory_profiling = options.enable_memory_profiling();
614    match options.command {
615        ServerCommand::Run {
616            server_config_path,
617            storage_config,
618            common_storage_options,
619            cross_chain_config,
620            notification_config,
621            shard,
622            block_time_grace_period,
623            wasm_runtime,
624            chain_worker_ttl,
625            chain_info_max_received_log_entries,
626            cross_chain_message_chunk_limit,
627            allow_revert_confirm,
628            reset_on_corrupted_chain_state_mins,
629            recovery_whitelist,
630            otlp_exporter_endpoint: _,
631        } => {
632            linera_version::VERSION_INFO.log();
633
634            let server_config: ValidatorServerConfig =
635                util::read_json(&server_config_path).expect("Failed to read server config");
636
637            let job = ServerContext {
638                server_config,
639                cross_chain_config,
640                notification_config,
641                shard,
642                block_time_grace_period,
643                chain_worker_ttl,
644                block_cache_size: options.block_cache_size,
645                execution_state_cache_size: options.execution_state_cache_size,
646                chain_info_max_received_log_entries,
647                cross_chain_message_chunk_limit,
648                allow_revert_confirm,
649                reset_on_corrupted_chain_state_mins,
650                recovery_whitelist: recovery_whitelist.map(HashSet::from_iter),
651                #[cfg(with_metrics)]
652                enable_memory_profiling,
653            };
654            let wasm_runtime = wasm_runtime.with_wasm_default();
655            let store_config = storage_config
656                .add_common_storage_options(&common_storage_options)
657                .unwrap();
658            // Validators should not output contract logs.
659            let allow_application_logs = false;
660            let cache_sizes = common_storage_options.storage_cache_config();
661            store_config
662                .run_with_storage(wasm_runtime, allow_application_logs, cache_sizes, job)
663                .boxed()
664                .await
665                .unwrap()
666                .unwrap();
667        }
668
669        ServerCommand::Generate {
670            validators,
671            committee,
672            testing_prng_seed,
673        } => {
674            let mut config_validators = Vec::new();
675            let mut rng = Box::<dyn CryptoRng>::from(testing_prng_seed);
676            for options_path in validators {
677                let options_string = fs_err::tokio::read_to_string(options_path)
678                    .await
679                    .expect("Unable to read validator options file");
680                let options: ValidatorOptions =
681                    toml::from_str(&options_string).unwrap_or_else(|_| {
682                        panic!("Invalid options file format: \n {}", options_string)
683                    });
684                let path = options.server_config_path.clone();
685                let mut server = make_server_config(&path, &mut rng, options)
686                    .expect("Unable to open server config file");
687                Persist::persist(&mut server)
688                    .await
689                    .expect("Unable to write server config file");
690                info!("Wrote server config {}", path.to_str().unwrap());
691                println!(
692                    "{},{}",
693                    server.validator.public_key, server.validator.account_key
694                );
695                config_validators.push(Persist::into_value(server).validator);
696            }
697            if let Some(committee) = committee {
698                let mut config = persistent::File::new(
699                    &committee,
700                    CommitteeConfig {
701                        validators: config_validators,
702                    },
703                )
704                .expect("Unable to open committee configuration");
705                Persist::persist(&mut config)
706                    .await
707                    .expect("Unable to write committee description");
708                info!("Wrote committee config {}", committee.to_str().unwrap());
709            }
710        }
711
712        ServerCommand::EditShards {
713            server_config_path,
714            num_shards,
715            host,
716            port,
717            metrics_port,
718        } => {
719            let mut server_config =
720                persistent::File::<ValidatorServerConfig>::read(&server_config_path)
721                    .expect("Failed to read server config");
722            let shards = generate_shard_configs(&num_shards, &host, &port, &metrics_port)
723                .expect("Failed to generate shard configs");
724            server_config.internal_network.shards = shards;
725            Persist::persist(&mut server_config)
726                .await
727                .expect("Failed to write updated server config");
728        }
729    }
730}
731
732fn generate_shard_configs(
733    num_shards: &str,
734    host: &str,
735    port: &str,
736    metrics_port: &Option<String>,
737) -> anyhow::Result<Vec<ShardConfig>> {
738    let mut shards = Vec::new();
739    let len = num_shards.len();
740    let num_shards = num_shards
741        .parse::<NonZeroU16>()
742        .context("Failed to parse the number of shards")?;
743    let pattern = "%".repeat(len);
744
745    for i in 1u16..=num_shards.into() {
746        let index = format!("{i:0len$}");
747        let host = host.replacen(&pattern, &index, 1);
748        let port = port
749            .replacen(&pattern, &index, 1)
750            .parse()
751            .context("Failed to decode port into an integers")?;
752        let metrics_port = metrics_port
753            .as_ref()
754            .map(|port| {
755                port.replacen(&pattern, &index, 1)
756                    .parse()
757                    .context("Failed to decode metrics port into an integers")
758            })
759            .transpose()?;
760        let shard = ShardConfig {
761            host,
762            port,
763            metrics_port,
764        };
765        shards.push(shard);
766    }
767    Ok(shards)
768}
