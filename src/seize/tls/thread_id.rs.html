11use std::cmp::Reverse;
12use std::collections::BinaryHeap;

15/// Thread ID manager which allocates thread IDs. It attempts to aggressively
16/// reuse thread IDs where possible to avoid cases where a `ThreadLocal` grows
17/// indefinitely when it is used by many short-lived threads.
18#[derive(Default)]
19struct ThreadIdManager {
20    free_from: usize,
21    free_list: BinaryHeap<Reverse<usize>>,
22}
23
24impl ThreadIdManager {
25    /// Allocate a new thread ID.
26    fn alloc(&mut self) -> usize {
27        if let Some(id) = self.free_list.pop() {
28            id.0
29        } else {
30            let id = self.free_from;
31            self.free_from = self
32                .free_from
33                .checked_add(1)
34                .expect("Ran out of thread IDs");
35            id
36        }
37    }
38
39    /// Free a thread ID for reuse.
40    fn free(&mut self, id: usize) {
41        self.free_list.push(Reverse(id));
42    }
43}
44
45fn thread_id_manager() -> &'static Mutex<ThreadIdManager> {
46    static THREAD_ID_MANAGER: OnceLock<Mutex<ThreadIdManager>> = OnceLock::new();
47    THREAD_ID_MANAGER.get_or_init(Default::default)
48}

50/// Data which is unique to the current thread while it is running.
51/// A thread ID may be reused after a thread exits.
52#[derive(Clone, Copy)]
53pub struct Thread {
54    pub(crate) id: usize,
55    pub(crate) bucket: usize,
56    pub(crate) index: usize,
57}
58
59impl Thread {
60    pub(crate) fn new(id: usize) -> Thread {
61        let skipped = id.checked_add(SKIP).expect("exceeded maximum length");
62        let bucket = usize::BITS - skipped.leading_zeros();
63        let bucket = (bucket as usize) - (SKIP_BUCKET + 1);
64        let bucket_size = Thread::bucket_size(bucket);
65        let index = skipped ^ bucket_size;
66
67        Thread { id, bucket, index }
68    }
69
70    pub fn bucket_size(bucket: usize) -> usize {
71        1 << (bucket + SKIP_BUCKET)
72    }
73
74    /// Get the current thread.
75    #[inline]
76    pub fn current() -> Thread {
77        THREAD.with(|thread| {
78            if let Some(thread) = thread.get() {
79                thread
80            } else {
81                Thread::init_slow(thread)
82            }
83        })
84    }
85
86    /// Slow path for allocating a thread ID.
87    #[cold]
88    #[inline(never)]
89    fn init_slow(thread: &Cell<Option<Thread>>) -> Thread {
90        let new = Thread::create();
91        thread.set(Some(new));
92        THREAD_GUARD.with(|guard| guard.id.set(new.id));
93        new
94    }
95
96    /// Create a new thread.
97    pub fn create() -> Thread {
98        Thread::new(thread_id_manager().lock().unwrap().alloc())
99    }
100
101    /// Free the given thread.
102    ///
103    /// # Safety
104    ///
105    /// This function must only be called once on a given thread.
106    pub unsafe fn free(id: usize) {
107        thread_id_manager().lock().unwrap().free(id);
108    }
109}

111// This is split into 2 thread-local variables so that we can check whether the
112// thread is initialized without having to register a thread-local destructor.
113//
114// This makes the fast path smaller.
115thread_local! { static THREAD: Cell<Option<Thread>> = const { Cell::new(None) }; }
116thread_local! { static THREAD_GUARD: ThreadGuard = const { ThreadGuard { id: Cell::new(0) } }; }
117
118// Guard to ensure the thread ID is released on thread exit.
119struct ThreadGuard {
120    // We keep a copy of the thread ID in the `ThreadGuard`: we can't
121    // reliably access `THREAD` in our `Drop` impl due to the unpredictable
122    // order of TLS destructors.
123    id: Cell<usize>,
124}
125
126impl Drop for ThreadGuard {
127    fn drop(&mut self) {
128        // Release the thread ID. Any further accesses to the thread ID
129        // will go through get_slow which will either panic or
130        // initialize a new ThreadGuard.
131        let _ = THREAD.try_with(|thread| thread.set(None));
132
133        // Safety: We are in `drop` and the current thread uniquely owns this ID.
134        unsafe { Thread::free(self.id.get()) };
135    }
136}

138#[test]
139fn test_thread() {
140    use crate::tls::BUCKETS;
141
142    assert_eq!(Thread::bucket_size(0), 32);
143    for i in 0..32 {
144        let loc = Thread::new(i);
145        assert_eq!(loc.bucket, 0);
146        assert_eq!(loc.index, i);
147    }
148
149    assert_eq!(Thread::bucket_size(1), 64);
150    for i in 33..96 {
151        let loc = Thread::new(i);
152        assert_eq!(loc.bucket, 1);
153        assert_eq!(loc.index, i - 32);
154    }
155
156    assert_eq!(Thread::bucket_size(2), 128);
157    for i in 96..224 {
158        let loc = Thread::new(i);
159        assert_eq!(loc.bucket, 2);
160        assert_eq!(loc.index, i - 96);
161    }
162
163    let max = Thread::new(usize::MAX - SKIP);
164    assert_eq!(max.bucket, BUCKETS - 1);
165    assert_eq!(max.index, (1 << 63) - 1);
166}
