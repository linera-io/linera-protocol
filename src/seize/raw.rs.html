12/// Fast, lock-free, and robust concurrent memory reclamation.
13///
14/// The core algorithm is described [in this paper](https://arxiv.org/pdf/2108.02763.pdf).
15pub struct Collector {
16    /// The global epoch value.
17    ///
18    /// The global epoch is used to detect stalled threads. The current epoch
19    /// is recorded on each allocated object as well as every time a thread
20    /// attempts to access a object. This allows retirers to skip threads that
21    /// are lagging behind and could not have accessed a given object.
22    pub(crate) epoch: AtomicU64,
23
24    /// Per-thread batches of retired nodes.
25    ///
26    /// Retired values are added to thread-local batches before starting
27    /// the reclamation process, to amortize the cost of retirement.
28    batches: ThreadLocal<CachePadded<UnsafeCell<LocalBatch>>>,
29
30    /// Per-thread reservations lists.
31    ///
32    /// A reservation list is a list of batches that have been retired
33    /// while the current thread was active. The thread must decrement
34    /// the reference count and potentially free the batch of any
35    /// reservations before exiting.
36    reservations: ThreadLocal<CachePadded<Reservation>>,
37
38    /// The number of object allocations before advancing the global epoch
39    /// on a given thread.
40    ///
41    /// If this is `None`, epoch tracking is disabled.
42    pub(crate) epoch_frequency: Option<NonZeroU64>,
43
44    /// The minimum number of nodes required in a batch before freeing.
45    pub(crate) batch_size: usize,
46}
47
48impl Collector {
49    /// Create a collector with the provided configuration.
50    pub fn new(threads: usize, epoch_frequency: NonZeroU64, batch_size: usize) -> Self {
51        Self {
52            // Note the global epoch must start at 1 because thread-local epochs
53            // start at 0. All threads must be forced to synchronize when protecting
54            // objects created in the first epoch.
55            epoch: AtomicU64::new(1),
56            reservations: ThreadLocal::with_capacity(threads),
57            batches: ThreadLocal::with_capacity(threads),
58            epoch_frequency: Some(epoch_frequency),
59            batch_size: batch_size.next_power_of_two(),
60        }
61    }

63    /// Returns the birth epoch for a new node, without incrementing
64    /// allocation counters.
65    pub fn birth_epoch(&self) -> u64 {
66        if self.epoch_frequency.is_some() {
67            self.epoch.load(Ordering::Relaxed)
68        } else {
69            0
70        }
71    }

73    /// Create a new node.
74    ///
75    /// # Safety
76    ///
77    /// This method is not safe to call concurrently on the same thread.
78    #[inline]
79    pub fn node(&self, reservation: &Reservation) -> Node {
80        // Record the current epoch value.
81        //
82        // Note that it's fine if we see an older epoch, in which case more threads
83        // may be counted as active.
84        let birth_epoch = match self.epoch_frequency {
85            Some(ref frequency) => {
86                // Safety: Node counts are only accessed by the current thread.
87                let count = reservation.node_count.get();
88                reservation.node_count.set(count + 1);
89
90                // Advance the global epoch if we reached the epoch frequency.
91                //
92                // Relaxed: Synchronization only occurs when an epoch is recorded by a
93                // given thread to protect a pointer, not when incrementing the epoch.
94                if count >= frequency.get() {
95                    reservation.node_count.set(0);
96                    self.epoch.fetch_add(1, Ordering::Relaxed) + 1
97                } else {
98                    self.epoch.load(Ordering::Relaxed)
99                }
100            }
101
102            // We aren't tracking epochs.
103            None => 0,
104        };
105
106        Node { birth_epoch }
107    }

120    /// Mark the current thread as active.
121    ///
122    /// `enter` and `leave` calls maintain a reference count to allow
123    /// reentrancy. If the current thread is already marked as active, this
124    /// method simply increases the reference count.
125    ///
126    /// # Safety
127    ///
128    /// This method is not safe to call concurrently on the same thread. This
129    /// method must only be called if the current thread is inactive.
130    #[inline]
131    pub unsafe fn enter(&self, reservation: &Reservation) {
132        // Mark the current thread as active.
133        reservation
134            .head
135            .store(ptr::null_mut(), membarrier::light_store());
136
137        // This barrier, combined with the light store above, synchronizes with the heavy
138        // barrier in `retire`:
139        // - If our store comes first, the thread retiring will see that we are active.
140        // - If the fence comes first, we will see the new values of any objects being
141        //   retired by that thread
142        //
143        // Note that all pointer loads perform a light barrier to participate in the total order.
144        membarrier::light_store_barrier();
145    }

147    /// Load an atomic pointer.
148    ///
149    /// # Safety
150    ///
151    /// This method must only be called with the reservation of the current thread.
152    #[inline]
153    pub unsafe fn protect_local<T>(&self, ptr: &AtomicPtr<T>, reservation: &Reservation) -> *mut T {
154        if self.epoch_frequency.is_none() {
155            // Epoch tracking is disabled.
156            //
157            // We use `SeqCst` unconditionally because it is equivalent to `Acquire` on
158            // most platforms, and the minimum ordering depends on the membarrier strategy,
159            // so checking is not worth it.
160            let value = ptr.load(Ordering::SeqCst);
161
162            // The light barrier ensures that this load participates in the total order.
163            // See `enter` for details.
164            membarrier::light_load_barrier();
165
166            return value;
167        }
168
169        // Load the last epoch we recorded on this thread.
170        //
171        // Relaxed: The reservation is only modified by the current thread.
172        let mut prev_epoch = reservation.epoch.load(Ordering::Relaxed);
173
174        loop {
175            // Acquire: We need to record at least the birth epoch to let other threads
176            // know we are accessing this pointer. Note that this requires the pointer to
177            // have been stored with `Release` ordering, which is technically undocumented.
178            // However, any `Relaxed` stores would be unsound to access anyways.
179            let ptr = ptr.load(Ordering::SeqCst);
180
181            // The light barrier ensures that this load participates in the total order.
182            // See `enter` for details.
183            membarrier::light_load_barrier();
184
185            // Relaxed: We acquired at least the pointer's birth epoch above, which is the
186            // only epoch we care about.
187            let current_epoch = self.epoch.load(Ordering::Relaxed);
188
189            // We are marked as active in the birth epoch of the pointer we are accessing.
190            // Any threads performing retirement will see that we have access to the pointer
191            // and add to our reservation list.
192            if prev_epoch == current_epoch {
193                return ptr;
194            }
195
196            // Our epoch is out of date, record the new one and try again.
197            reservation
198                .epoch
199                .store(current_epoch, membarrier::light_store());
200
201            // This barrier, combined with the light store above, synchronizes with the heavy
202            // barrier in `retire`:
203            // - If our store comes first, the thread retiring will see that we are active
204            //   in the current epoch.
205            // - If the fence comes first, we will see the new values of any objects being
206            //   retired by that thread.
207            membarrier::light_store_barrier();
208
209            prev_epoch = current_epoch;
210        }
211    }

213    /// Load an atomic pointer.
214    ///
215    /// This method is safe to call concurrently from multiple threads with the
216    /// same `thread` object.
217    #[inline]
218    pub fn protect<T>(&self, ptr: &AtomicPtr<T>, reservation: &Reservation) -> *mut T {
219        if self.epoch_frequency.is_none() {
220            // Epoch tracking is disabled.
221            let value = ptr.load(Ordering::SeqCst);
222
223            // The light barrier ensures that this load participates in the total order.
224            // See `enter` for details.
225            membarrier::light_load_barrier();
226
227            return value;
228        }
229
230        // Load the last epoch we recorded for this reservation.
231        let mut prev_epoch = reservation.epoch.load(membarrier::light_load());
232
233        // This epoch may be modified concurrently by other threads. If a different
234        // thread recorded an epoch, we must force this thread to also participate
235        // in the total order and load the new values of any objects that may have
236        // been retired, hence the light barrier.
237        membarrier::light_load_barrier();
238
239        loop {
240            // Acquire: acquire the birth epoch of this pointer. See `protect_local`
241            // for details.
242            let ptr = ptr.load(Ordering::SeqCst);
243
244            // The light barrier ensures that this load participates in the total order.
245            // See `enter` for details.
246            membarrier::light_load_barrier();
247
248            // Relaxed: We acquired at least the pointer's birth epoch above, which is the
249            // only epoch we care about.
250            let current_epoch = self.epoch.load(Ordering::Relaxed);
251
252            // We are marked as active in the birth epoch of the pointer we are accessing.
253            // Any threads performing retirement will see that we have access to the pointer
254            // and add to our reservation list.
255            if prev_epoch == current_epoch {
256                return ptr;
257            }
258
259            // Our epoch is out of date, record the new one and try again.
260            //
261            // Note that this may be called concurrently, so the `fetch_max` ensures we
262            // never overwrite a newer epoch.
263            prev_epoch = reservation
264                .epoch
265                .fetch_max(current_epoch, membarrier::light_store())
266                .max(current_epoch);
267
268            // This barrier, combined with the light store above, synchronizes with the heavy
269            // barrier in `retire`:
270            // - If our store comes first, the thread retiring will see that we are active
271            //   in the current epoch.
272            // - If the fence comes first, we will see the new values of any objects being
273            //   retired by that thread.
274            //
275            // If a different thread beats us and writes a newer epoch, this implies a light load
276            // barrier that ensures that we still participate in the total order.
277            membarrier::light_store_barrier();
278        }
279    }

281    /// Mark the current thread as inactive.
282    ///
283    /// # Safety
284    ///
285    /// Any previously protected pointers may be invalidated after calling
286    /// `leave`. Additionally, method is not safe to call concurrently with
287    /// the same reservation.
288    #[inline]
289    pub unsafe fn leave(&self, reservation: &Reservation) {
290        // Release: Exit the critical section, ensuring that any pointer accesses
291        // happen-before we are marked as inactive.
292        let head = reservation.head.swap(Entry::INACTIVE, Ordering::Release);
293
294        if head != Entry::INACTIVE {
295            // Acquire any new entries in the reservation list, as well as the new values
296            // of any objects that were retired while we were active.
297            atomic::fence(Ordering::Acquire);
298
299            // Decrement the reference counts of any batches that were retired.
300            unsafe { Collector::traverse(head) }
301        }
302    }

304    /// Decrement any reference counts, keeping the thread marked as active.
305    ///
306    /// # Safety
307    ///
308    /// Any previously protected pointers may be invalidated after calling
309    /// `leave`. Additionally, method is not safe to call concurrently with
310    /// the same reservation.
311    #[inline]
312    pub unsafe fn refresh(&self, reservation: &Reservation) {
313        // This stores acts as a combined call to `leave` and `enter`.
314        //
315        // SeqCst: Establish the same ordering as `enter` and `leave`.
316        let head = reservation.head.swap(ptr::null_mut(), Ordering::SeqCst);
317
318        if head != Entry::INACTIVE {
319            // Decrement the reference counts of any batches that were retired.
320            unsafe { Collector::traverse(head) }
321        }
322    }

324    /// Add a node to the retirement batch, retiring the batch if `batch_size`
325    /// is reached.
326    ///
327    /// # Safety
328    ///
329    /// `ptr` must be a valid pointer that is no longer accessible to any
330    /// inactive threads. Additionally, this method is not safe to call
331    /// concurrently with the same `thread`.
332    #[inline]
333    pub unsafe fn add<T>(&self, ptr: *mut T, reclaim: unsafe fn(*mut Link), thread: Thread)
334    where
335        T: AsLink,
336    {
337        let local_batch = self.batches.load(thread).get();
338
339        // Safety: Local batches are only accessed by the current thread.
340        let batch = unsafe { (*local_batch).get_or_init(self.batch_size) };
341
342        // If we are in a recursive call during `drop` or `reclaim_all`, reclaim the
343        // object immediately.
344        if batch == LocalBatch::DROP {
345            unsafe { reclaim(ptr.cast::<Link>()) }
346            return;
347        }
348
349        // `ptr` is guaranteed to be a valid pointer that can be cast to a node
350        // (because of `T: AsLink`).
351        //
352        // Any other thread with a reference to the pointer only has a shared reference
353        // to the `UnsafeCell<Node>`, which is allowed to alias. The caller guarantees
354        // that the same pointer is not retired twice, so we can safely write to the
355        // node through the shared pointer.
356        let node = UnsafeCell::raw_get(ptr.cast::<UnsafeCell<Node>>());
357
358        // Safety: The node and batch are both valid for mutable access.
359        let len = unsafe {
360            // Keep track of the oldest node in the batch.
361            let birth_epoch = (*node).birth_epoch;
362            (*batch).min_epoch = (*batch).min_epoch.min(birth_epoch);
363
364            // Create an entry for this node.
365            (*batch).entries.push(Entry {
366                node,
367                reclaim,
368                batch,
369            });
370
371            (*batch).entries.len()
372        };
373
374        // Attempt to retire the batch if we have enough entries.
375        if len >= self.batch_size {
376            // Safety: The caller guarantees that this method is not called concurrently
377            // with the same `thread` and we are not holding on to any mutable references.
378            unsafe { self.try_retire(local_batch, thread) }
379        }
380    }

382    // Retire a batch of nodes.
383    //
384    // # Safety
385    //
386    // The batch must no longer accessible to any inactive threads.
387    /// Additionally, this method is not safe to call concurrently with the same
388    /// `thread`.
389    #[inline]
390    pub unsafe fn add_batch(
391        &self,
392        deferred: &mut Deferred,
393        reclaim: unsafe fn(*mut Link),
394        thread: Thread,
395    ) {
396        let local_batch = self.batches.load(thread).get();
397
398        // Safety: Local batches are only accessed by the current thread.
399        let batch = unsafe { (*local_batch).get_or_init(self.batch_size) };
400
401        // If we are in a recursive call during `drop` or `reclaim_all`, reclaim the
402        // batch immediately.
403        if batch == LocalBatch::DROP {
404            deferred.for_each(|ptr| unsafe { reclaim(ptr.cast::<Link>()) });
405            return;
406        }
407
408        let min_epoch = *deferred.min_epoch.get_mut();
409
410        // Safety: The deferred and local batch are both valid for mutable access.
411        let len = unsafe {
412            // Keep track of the oldest node in the batch.
413            (*batch).min_epoch = (*batch).min_epoch.min(min_epoch);
414
415            deferred.for_each(|node| {
416                // Create an entry for this node.
417                (*batch).entries.push(Entry {
418                    node,
419                    reclaim,
420                    batch,
421                });
422            });
423
424            (*batch).entries.len()
425        };
426
427        // Attempt to retire the batch if we have enough entries.
428        if len >= self.batch_size {
429            // Safety: The caller guarantees that this method is not called concurrently
430            // with the same `thread` and we are not holding on to any mutable references.
431            unsafe { self.try_retire(local_batch, thread) }
432        }
433    }

449    /// Attempt to retire objects in this batch.
450    ///
451    /// Note that if a guard on the current thread is active, the batch will
452    /// also be added to the current reservation list for deferred reclamation.
453    ///
454    /// # Safety
455    ///
456    /// This method is not safe to call concurrently with the same `thread`.
457    /// Additionally, the caller should not be holding on to any mutable
458    /// references the the local batch, as they may be invalidated by
459    /// recursive calls to `try_retire`.
460    #[inline]
461    pub unsafe fn try_retire(&self, local_batch: *mut LocalBatch, thread: Thread) {
462        // Establish a total order between the retirement of nodes in this batch and
463        // light stores marking a thread as active, or active in an epoch:
464        // - If the store comes first, we will see that the thread is active.
465        // - If this barrier comes first, the thread will see the new values of any
466        //   objects in this batch.
467        //
468        // This barrier also establishes synchronizes with the light store executed when a
469        // thread is created:
470        // - If our barrier comes first, they will see the new values of any objects in
471        //   this batch.
472        // - If their store comes first, we will see the new thread.
473        membarrier::heavy();
474
475        // Safety: Local batches are only accessed by the current thread.
476        let batch = unsafe { (*local_batch).batch };
477
478        // There is nothing to retire.
479        if batch.is_null() || batch == LocalBatch::DROP {
480            return;
481        }
482
483        // Safety: The batch is non-null.
484        let min_epoch = unsafe { (*batch).min_epoch };
485        let batch_entries = unsafe { (*batch).entries.as_mut_ptr() };
486
487        let mut marked = 0;
488        let current_reservation = self.reservations.load(thread);
489
490        // Record all active threads, including the current thread.
491        //
492        // We need to do this in a separate step before actually retiring the batch to
493        // ensure we have enough entries for reservation lists, as the number of threads
494        // can grow dynamically.
495        for reservation in self.reservations.iter() {
496            // If we don't have enough entries to insert into the reservation lists
497            // of all active threads, try again later.
498            //
499            // Safety: Local batch pointers are valid until relamation.
500            let Some(entry) = unsafe { &(*batch).entries }.get(marked) else {
501                return;
502            };
503
504            // If this thread is inactive, we can skip it. The heavy barrier above
505            // ensurse that the next time it becomes active, it will see the new
506            // values of any objects in this batch.
507            //
508            // Relaxed: See the Acquire fence below.
509            if reservation.head.load(Ordering::Relaxed) == Entry::INACTIVE {
510                continue;
511            }
512
513            // If this thread's epoch is behind the earliest birth epoch in this batch
514            // we can skip it, as there is no way it could have accessed any of the objects
515            // in this batch.  The heavy barrier above ensures that the next time it attempts
516            // to access an object in this batch in `protect`, it will see it's new value.
517            //
518            // We make sure never to skip the current thread even if it's epoch is behind
519            // because it may still have access to the pointer. The current thread is only
520            // skipped if there is no active guard.
521            //
522            // Relaxed: We already ensured that we will see the relevant epoch through the
523            // heavy barrier above. If the epoch is behind there is nothing to synchronize
524            // with.
525            //
526            // If epoch tracking is disabled this is always false (0 < 0).
527            if !ptr::eq(reservation, current_reservation)
528                && reservation.epoch.load(Ordering::Relaxed) < min_epoch
529            {
530                continue;
531            }
532
533            // Temporarily store this reservation list in the batch.
534            //
535            // Safety: All nodes in a batch are valid and this batch has not yet been shared
536            // to other threads.
537            unsafe { (*entry.node).head = &reservation.head }
538            marked += 1;
539        }
540
541        // For any inactive threads we skipped above, synchronize with `leave` to ensure
542        // any accesses happen-before we retire. We ensured with the heavy barrier above
543        // that the thread will see the new values of any objects in this batch the next
544        // time it becomes active.
545        atomic::fence(Ordering::Acquire);
546
547        // Add the batch to the reservation lists of any active threads.
548        let mut active = 0;
549        'retire: for i in 0..marked {
550            // Safety: Local batch pointers are valid until reclamation.
551            let curr = unsafe { batch_entries.add(i) };
552
553            // Safety: All nodes in the batch are valid and we just initialized `head` for
554            // all `marked` nodes in the loop above.
555            let head = unsafe { &*(*(*curr).node).head };
556
557            // Relaxed: All writes to the `head` use RMW instructions, so the previous node
558            // in the list is synchronized through the release sequence on `head`.
559            let mut prev = head.load(Ordering::Relaxed);
560
561            loop {
562                // The thread became inactive, skip it.
563                //
564                // As long as the thread became inactive at some point after the heavy barrier,
565                // it can no longer access any objects in this batch. The next time it becomes
566                // active it will load the new object values due to the heavy barrier above.
567                if prev == Entry::INACTIVE {
568                    // Acquire: Synchronize with `leave` to ensure any accesses happen-before we
569                    // retire.
570                    atomic::fence(Ordering::Acquire);
571                    continue 'retire;
572                }
573
574                // Link this node to the reservation list.
575                unsafe { (*(*curr).node).next = prev }
576
577                // Release the node, as well as the new values of any objects in this batch for
578                // when this thread calls `leave`.
579                match head.compare_exchange_weak(prev, curr, Ordering::Release, Ordering::Relaxed) {
580                    Ok(_) => break,
581                    // Lost the race to another thread, retry.
582                    Err(found) => prev = found,
583                }
584            }
585
586            active += 1;
587        }
588
589        // Release: If we don't free the list, release any access of the batch to the
590        // thread that eventually will.
591        //
592        // Safety: Local batch pointers are valid until relamation.
593        if unsafe { &*batch }
594            .active
595            .fetch_add(active, Ordering::Release)
596            .wrapping_add(active)
597            == 0
598        {
599            // Acquire: Ensure any access of objects in the batch happen-before we free it.
600            atomic::fence(Ordering::Acquire);
601
602            // Reset the batch.
603            unsafe { *local_batch = LocalBatch::default() };
604
605            // Safety: The reference count is zero, meaning that either no threads were
606            // active or they have all already decremented the reference count.
607            //
608            // Additionally, the local batch has been reset and we are not holding on to any
609            // mutable references, so any recursive calls to retire during
610            // reclamation are valid.
611            unsafe { Collector::free_batch(batch) }
612            return;
613        }
614
615        // Reset the batch.
616        unsafe { *local_batch = LocalBatch::default() };
617    }

619    /// Traverse the reservation list, decrementing the reference count of each
620    /// batch.
621    ///
622    /// # Safety
623    ///
624    /// `list` must be a valid reservation list.
625    #[cold]
626    #[inline(never)]
627    unsafe fn traverse(mut list: *mut Entry) {
628        while !list.is_null() {
629            let curr = list;
630
631            // Advance the cursor.
632            // Safety: `curr` is a valid, non-null node in the list.
633            list = unsafe { (*(*curr).node).next };
634            let batch = unsafe { (*curr).batch };
635
636            // Safety: Batch pointers are valid for reads until they are freed.
637            unsafe {
638                // Release: If we don't free the list, release any access of objects in the
639                // batch to the thread that will.
640                if (*batch).active.fetch_sub(1, Ordering::Release) == 1 {
641                    // Ensure any access of objects in the batch happen-before we free it.
642                    atomic::fence(Ordering::Acquire);
643
644                    // Safety: We have the last reference to the batch and it has been removed
645                    // from our reservation list.
646                    Collector::free_batch(batch)
647                }
648            }
649        }
650    }

652    /// Reclaim all values in the collector, including recursive calls to
653    /// retire.
654    ///
655    /// # Safety
656    ///
657    /// No threads may be accessing the collector or any values that have been
658    /// retired.
659    #[inline]
660    pub unsafe fn reclaim_all(&self) {
661        for local_batch in self.batches.iter() {
662            let local_batch = local_batch.value.get();
663
664            unsafe {
665                let batch = (*local_batch).batch;
666
667                // There is nothing to reclaim.
668                if batch.is_null() {
669                    continue;
670                }
671
672                // Tell any recursive calls to `retire` to reclaim immediately.
673                (*local_batch).batch = LocalBatch::DROP;
674
675                // Safety: we have `&mut self` and the batch is non-null.
676                Collector::free_batch(batch);
677
678                // Reset the batch.
679                (*local_batch).batch = ptr::null_mut();
680            }
681        }
682    }
683
684    /// Free a batch of objects.
685    ///
686    /// # Safety
687    ///
688    /// The batch reference count must be zero. Additionally, the current thread
689    /// must not be holding on to any mutable references to thread-locals,
690    /// as recursive calls to retire may still access the local batch. The
691    /// batch being retired must be unreachable through any recursive calls.
692    #[inline]
693    unsafe fn free_batch(batch: *mut Batch) {
694        // Safety: We have the last reference to the batch.
695        for entry in unsafe { (*batch).entries.iter_mut() } {
696            unsafe { (entry.reclaim)(entry.node.cast::<Link>()) };
697        }
698
699        unsafe { LocalBatch::free(batch) };
700    }
701}
702
703impl Drop for Collector {
704    #[inline]
705    fn drop(&mut self) {
706        // Safety: We have `&mut self`.
707        unsafe { self.reclaim_all() };
708    }
709}

711/// A node attached to every allocated object.
712///
713/// Nodes keep track of their birth epoch and are also used
714/// as links in thread-local reservation lists.
715#[repr(C)]
716pub union Node {
717    // Before retiring: The epoch this node was created in.
718    pub birth_epoch: u64,
719    // While retiring: Temporary location for an active reservation list.
720    head: *const AtomicPtr<Entry>,
721    // After retiring: next node in the thread's reservation list
722    next: *mut Entry,
723    // In deferred batch: next node in the batch
724    pub next_batch: *mut Node,
725}
726
727/// A per-thread reservation list.
728///
729/// Reservation lists are lists of retired entries, where each entry represents
730/// a batch.
731#[repr(C)]
732pub struct Reservation {
733    /// The head of the list
734    head: AtomicPtr<Entry>,
735    /// The epoch this thread last accessed a pointer in.
736    epoch: AtomicU64,
737    /// The number of active guards for this thread.
738    pub guards: Cell<u64>,
739    /// A lock used for owned guards to prevent concurrent operations.
740    pub lock: Mutex<()>,
741    /// The number of nodes allocated by this thread.
742    node_count: Cell<u64>,
743}
744
745impl Default for Reservation {
746    fn default() -> Self {
747        Reservation {
748            head: AtomicPtr::new(Entry::INACTIVE),
749            epoch: AtomicU64::new(0),
750            guards: Cell::new(0),
751            lock: Mutex::new(()),
752            node_count: Cell::new(0),
753        }
754    }
755}
756
757/// A batch of nodes waiting to be retired.
758struct Batch {
759    /// Nodes in this batch.
760    ///
761    /// TODO: This allocation could be flattened.
762    entries: Vec<Entry>,
763    /// The minimum epoch of all nodes in this batch.
764    min_epoch: u64,
765    /// The reference count for any active threads.
766    active: AtomicUsize,
767}
768
769impl Batch {
770    /// Create a new batch with the specified capacity.
771    #[inline]
772    fn new(capacity: usize) -> Batch {
773        Batch {
774            entries: Vec::with_capacity(capacity),
775            min_epoch: 0,
776            active: AtomicUsize::new(0),
777        }
778    }
779}

781/// A retired object.
782struct Entry {
783    /// Object metadata.
784    node: *mut Node,
785    /// The function used to reclaim this object.
786    reclaim: unsafe fn(*mut Link),
787    /// The batch this node is a part of.
788    batch: *mut Batch,
789}
790
791impl Entry {
792    /// Represents an inactive thread.
793    ///
794    /// While null indicates an empty list, INACTIVE indicates the thread has no
795    /// active guards and is not accessing any objects.
796    pub const INACTIVE: *mut Entry = usize::MAX as _;
797}

799/// A pointer to a batch, unique to the current thread.
800pub struct LocalBatch {
801    batch: *mut Batch,
802}
803
804impl Default for LocalBatch {
805    fn default() -> Self {
806        LocalBatch {
807            batch: ptr::null_mut(),
808        }
809    }
810}
811
812impl LocalBatch {
813    /// This is set during a call to reclaim_all, signalling recursive calls to
814    /// retire to reclaim immediately.
815    const DROP: *mut Batch = usize::MAX as _;
816
817    /// Return a pointer to the batch, initializing it if the batch was null.
818    #[inline]
819    fn get_or_init(&mut self, capacity: usize) -> *mut Batch {
820        if self.batch.is_null() {
821            self.batch = Box::into_raw(Box::new(Batch::new(capacity)));
822        }
823
824        self.batch
825    }
826
827    /// Free the batch.
828    #[inline]
829    unsafe fn free(ptr: *mut Batch) {
830        unsafe { drop(Box::from_raw(ptr)) }
831    }
832}
833
834/// Local batches are only accessed by the current thread.
835unsafe impl Send for LocalBatch {}
836unsafe impl Sync for LocalBatch {}
