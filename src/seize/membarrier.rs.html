3//! # Semantics
4//!
5//! There is a total order over all memory barriers provided by this module:
6//! - Light store barriers, created by a pair of [`light_store`] and [`light_store_barrier`].
7//! - Light load barriers, created by a pair of [`light_load`] and [`light_load_barrier`].
8//! - Sequentially consistent barriers, or cumulative light barriers.
9//! - Heavy barriers, created by [`heavy`].
10//!
11//! If thread A issues barrier X and thread B issues barrier Y and X occurs before Y in the
12//! total order, X is ordered before Y with respect to coherence only if either X or Y is
13//! a heavy barrier. In other words, there is no way to establish an ordering between light
14//! barriers without the presence of a heavy barrier.
15

40    ///

46    }

54    /// The ordering for a load operation that synchronizes with heavy barriers.
55    #[inline]
56    pub fn light_load() -> Ordering {
57        // Participate in the total order established by light and heavy `SeqCst` barriers.

59    }

62    #[inline]
63    pub fn light_load_barrier() {
64        // This is a no-op due to strong loads and stores.
65    }

67    /// Issues a heavy memory barrier for slow path that synchronizes with light stores.
68    #[inline]
69    pub fn heavy() {
70        // Synchronize with `SeqCst` light stores.

75#[cfg(all(target_os = "linux", feature = "fast-barrier", not(miri)))]
76mod linux {
77    use std::sync::atomic::{self, AtomicU8, Ordering};
78
79    /// The ordering for a store operation that synchronizes with heavy barriers.
80    ///
81    /// Must be followed by a light barrier.
82    #[inline]
83    pub fn light_store() -> Ordering {
84        match STRATEGY.load(Ordering::Relaxed) {
85            FALLBACK => Ordering::SeqCst,
86            _ => Ordering::Relaxed,
87        }
88    }
89
90    /// Issues a light memory barrier for a preceding store operation.
91    #[inline]
92    pub fn light_store_barrier() {
93        atomic::compiler_fence(atomic::Ordering::SeqCst)
94    }
95
96    /// The ordering for a load operation that synchronizes with heavy barriers.
97    #[inline]
98    pub fn light_load() -> Ordering {
99        // There is no difference between `Acquire` and `SeqCst` loads on most platforms, so
100        // checking the strategy is not worth it.
101        Ordering::SeqCst
102    }
103
104    /// Issues a light memory barrier for a preceding load operation.
105    #[inline]
106    pub fn light_load_barrier() {
107        atomic::compiler_fence(atomic::Ordering::SeqCst)
108    }
109
110    /// Issues a heavy memory barrier for slow path.
111    #[inline]
112    pub fn heavy() {
113        // Issue a private expedited membarrier using the `sys_membarrier()` system call, if
114        // supported; otherwise, fall back to `mprotect()`-based process-wide memory barrier.
115        match STRATEGY.load(Ordering::Relaxed) {
116            MEMBARRIER => membarrier::barrier(),
117            MPROTECT => mprotect::barrier(),
118            _ => atomic::fence(atomic::Ordering::SeqCst),
119        }
120    }
121
122    /// Use the `membarrier` system call.
123    const MEMBARRIER: u8 = 0;
124
125    /// Use the `mprotect`-based trick.
126    const MPROTECT: u8 = 1;
127
128    /// Use `SeqCst` fences.
129    const FALLBACK: u8 = 2;
130
131    /// The right strategy to use on the current machine.
132    static STRATEGY: AtomicU8 = AtomicU8::new(FALLBACK);
133
134    /// Perform runtime detection for a membarrier strategy.
135    pub fn detect() {
136        if membarrier::is_supported() {
137            STRATEGY.store(MEMBARRIER, Ordering::Relaxed);
138        } else if mprotect::is_supported() {
139            STRATEGY.store(MPROTECT, Ordering::Relaxed);
140        }
141    }

144        ($cond:expr) => {
145            if !$cond {
146                #[allow(unused_unsafe)]
147                unsafe {
148                    libc::abort();
149                }
150            }
151        };

154    mod membarrier {
155        /// Commands for the membarrier system call.
156        ///
157        /// # Caveat
158        ///
159        /// We're defining it here because, unfortunately, the `libc` crate currently doesn't
160        /// expose `membarrier_cmd` for us. You can find the numbers in the [Linux source
161        /// code](https://github.com/torvalds/linux/blob/master/include/uapi/linux/membarrier.h).
162        ///
163        /// This enum should really be `#[repr(libc::c_int)]`, but Rust currently doesn't allow it.
164        #[repr(i32)]
165        #[allow(dead_code, non_camel_case_types)]
166        enum membarrier_cmd {
167            MEMBARRIER_CMD_QUERY = 0,
168            MEMBARRIER_CMD_GLOBAL = (1 << 0),
169            MEMBARRIER_CMD_GLOBAL_EXPEDITED = (1 << 1),
170            MEMBARRIER_CMD_REGISTER_GLOBAL_EXPEDITED = (1 << 2),
171            MEMBARRIER_CMD_PRIVATE_EXPEDITED = (1 << 3),
172            MEMBARRIER_CMD_REGISTER_PRIVATE_EXPEDITED = (1 << 4),
173            MEMBARRIER_CMD_PRIVATE_EXPEDITED_SYNC_CORE = (1 << 5),
174            MEMBARRIER_CMD_REGISTER_PRIVATE_EXPEDITED_SYNC_CORE = (1 << 6),
175        }
176
177        /// Call the `sys_membarrier` system call.
178        #[inline]
179        fn sys_membarrier(cmd: membarrier_cmd) -> libc::c_long {
180            unsafe { libc::syscall(libc::SYS_membarrier, cmd as libc::c_int, 0 as libc::c_int) }
181        }
182
183        /// Returns `true` if the `sys_membarrier` call is available.
184        pub fn is_supported() -> bool {
185            // Queries which membarrier commands are supported. Checks if private expedited
186            // membarrier is supported.
187            let ret = sys_membarrier(membarrier_cmd::MEMBARRIER_CMD_QUERY);
188            if ret < 0
189                || ret & membarrier_cmd::MEMBARRIER_CMD_PRIVATE_EXPEDITED as libc::c_long == 0
190                || ret & membarrier_cmd::MEMBARRIER_CMD_REGISTER_PRIVATE_EXPEDITED as libc::c_long
191                    == 0
192            {
193                return false;
194            }
195
196            // Registers the current process as a user of private expedited membarrier.
197            if sys_membarrier(membarrier_cmd::MEMBARRIER_CMD_REGISTER_PRIVATE_EXPEDITED) < 0 {
198                return false;
199            }
200
201            true
202        }
203
204        /// Executes a heavy `sys_membarrier`-based barrier.
205        #[inline]
206        pub fn barrier() {
207            fatal_assert!(sys_membarrier(membarrier_cmd::MEMBARRIER_CMD_PRIVATE_EXPEDITED) >= 0);
208        }
209    }

211    mod mprotect {
212        use std::cell::UnsafeCell;
213        use std::mem::MaybeUninit;
214        use std::ptr;
215        use std::sync::{atomic, OnceLock};
216
217        struct Barrier {
218            lock: UnsafeCell<libc::pthread_mutex_t>,
219            page: u64,
220            page_size: libc::size_t,
221        }
222
223        unsafe impl Sync for Barrier {}
224
225        impl Barrier {
226            /// Issues a process-wide barrier by changing access protections of a single mmap-ed
227            /// page. This method is not as fast as the `sys_membarrier()` call, but works very
228            /// similarly.
229            #[inline]
230            fn barrier(&self) {
231                let page = self.page as *mut libc::c_void;
232
233                unsafe {
234                    // Lock the mutex.
235                    fatal_assert!(libc::pthread_mutex_lock(self.lock.get()) == 0);
236
237                    // Set the page access protections to read + write.
238                    fatal_assert!(
239                        libc::mprotect(page, self.page_size, libc::PROT_READ | libc::PROT_WRITE,)
240                            == 0
241                    );
242
243                    // Ensure that the page is dirty before we change the protection so that we
244                    // prevent the OS from skipping the global TLB flush.
245                    let atomic_usize = &*(page as *const atomic::AtomicUsize);
246                    atomic_usize.fetch_add(1, atomic::Ordering::SeqCst);
247
248                    // Set the page access protections to none.
249                    //
250                    // Changing a page protection from read + write to none causes the OS to issue
251                    // an interrupt to flush TLBs on all processors. This also results in flushing
252                    // the processor buffers.
253                    fatal_assert!(libc::mprotect(page, self.page_size, libc::PROT_NONE) == 0);
254
255                    // Unlock the mutex.
256                    fatal_assert!(libc::pthread_mutex_unlock(self.lock.get()) == 0);
257                }
258            }
259        }
260
261        /// An alternative solution to `sys_membarrier` that works on older Linux kernels and
262        /// x86/x86-64 systems.
263        static BARRIER: OnceLock<Barrier> = OnceLock::new();
264
265        /// Returns `true` if the `mprotect`-based trick is supported.
266        pub fn is_supported() -> bool {
267            cfg!(target_arch = "x86") || cfg!(target_arch = "x86_64")
268        }
269
270        /// Executes a heavy `mprotect`-based barrier.
271        #[inline]
272        pub fn barrier() {
273            let barrier = BARRIER.get_or_init(|| {
274                unsafe {
275                    // Find out the page size on the current system.
276                    let page_size = libc::sysconf(libc::_SC_PAGESIZE);
277                    fatal_assert!(page_size > 0);
278                    let page_size = page_size as libc::size_t;
279
280                    // Create a dummy page.
281                    let page = libc::mmap(
282                        ptr::null_mut(),
283                        page_size,
284                        libc::PROT_NONE,
285                        libc::MAP_PRIVATE | libc::MAP_ANONYMOUS,
286                        -1 as libc::c_int,
287                        0 as libc::off_t,
288                    );
289                    fatal_assert!(page != libc::MAP_FAILED);
290                    fatal_assert!(page as libc::size_t % page_size == 0);
291
292                    // Locking the page ensures that it stays in memory during the two mprotect
293                    // calls in `Barrier::barrier()`. If the page was unmapped between those calls,
294                    // they would not have the expected effect of generating IPI.
295                    libc::mlock(page, page_size as libc::size_t);
296
297                    // Initialize the mutex.
298                    let lock = UnsafeCell::new(libc::PTHREAD_MUTEX_INITIALIZER);
299                    let mut attr = MaybeUninit::<libc::pthread_mutexattr_t>::uninit();
300                    fatal_assert!(libc::pthread_mutexattr_init(attr.as_mut_ptr()) == 0);
301                    let mut attr = attr.assume_init();
302                    fatal_assert!(
303                        libc::pthread_mutexattr_settype(&mut attr, libc::PTHREAD_MUTEX_NORMAL) == 0
304                    );
305                    fatal_assert!(libc::pthread_mutex_init(lock.get(), &attr) == 0);
306                    fatal_assert!(libc::pthread_mutexattr_destroy(&mut attr) == 0);
307
308                    let page = page as u64;
309
310                    Barrier {
311                        lock,
312                        page,
313                        page_size,
314                    }
315                }
316            });
317
318            barrier.barrier();
319        }
320    }
321}
