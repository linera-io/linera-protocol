1//! [`ItemError`] expansion.
2
3use super::{ExpCtxt, expand_fields, expand_from_into_tuples, expand_tokenize};
4use alloy_sol_macro_input::{ContainsSolAttrs, mk_doc};
5use ast::ItemError;
6use proc_macro2::TokenStream;
7use quote::quote;
8use syn::Result;
9
10/// Expands an [`ItemError`]:
11///
12/// ```ignore (pseudo-code)
13/// pub struct #name {
14///     #(pub #parameter_name: #parameter_type,)*
15/// }
16///
17/// impl SolError for #name {
18///     ...
19/// }
20/// ```
21pub(super) fn expand(cx: &ExpCtxt<'_>, error: &ItemError) -> Result<TokenStream> {
22    let ItemError { parameters: params, .. } = error;
23    cx.assert_resolved(params)?;
24
25    let (sol_attrs, mut attrs) = error.split_attrs()?;
26    cx.derives(&mut attrs, params, true);
27    let docs = sol_attrs.docs.or(cx.attrs.docs).unwrap_or(true);
28    let abi = sol_attrs.abi.or(cx.attrs.abi).unwrap_or(false);
29
30    let tokenize_impl = expand_tokenize(params, cx, super::FieldKind::Deconstruct);
31
32    let name = cx.overloaded_name(error.into());
33    let signature = cx.error_signature(error);
34    let selector = crate::utils::selector(&signature);
35
36    let alloy_sol_types = &cx.crates.sol_types;
37
38    let converts = expand_from_into_tuples(&name.0, params, cx, super::FieldKind::Deconstruct);
39
40    let doc = docs.then(|| {
41        let selector = hex::encode_prefixed(selector.array.as_slice());
42        mk_doc(format!(
43            "Custom error with signature `{signature}` and selector `{selector}`.\n\
44             ```solidity\n{error}\n```"
45        ))
46    });
47    let abi: Option<TokenStream> = abi.then(|| {
48        if_json! {
49            let error = super::to_abi::generate(error, cx);
50            quote! {
51                #[automatically_derived]
52                impl alloy_sol_types::JsonAbiExt for #name {
53                    type Abi = alloy_sol_types::private::alloy_json_abi::Error;
54
55                    #[inline]
56                    fn abi() -> Self::Abi {
57                        #error
58                    }
59                }
60            }
61        }
62    });
63
64    let err_struct = match params.len() {
65        0 => {
66            // Expanded as a unit struct.
67            quote! {
68                pub struct #name;
69            }
70        }
71        1 if params[0].name.is_none() => {
72            let ty = cx.expand_rust_type(&params[0].ty);
73            // Expanded as tuple struct if only one _unnamed_ parameter.
74            quote! {
75                pub struct #name(pub #ty);
76            }
77        }
78        _ => {
79            let fields = expand_fields(params, cx);
80            quote! {
81                pub struct #name {
82                    #(#fields),*
83                }
84            }
85        }
86    };
87
88    let tokens = quote! {
89        #(#attrs)*
90        #doc
91        #[allow(non_camel_case_types, non_snake_case, clippy::pub_underscore_fields)]
92        #[derive(Clone)]
93        #err_struct
94
95        #[allow(non_camel_case_types, non_snake_case, clippy::pub_underscore_fields, clippy::style)]
96        const _: () = {
97            use #alloy_sol_types as alloy_sol_types;
98
99            #converts
100
101            #[automatically_derived]
102            impl alloy_sol_types::SolError for #name {
103                type Parameters<'a> = UnderlyingSolTuple<'a>;
104                type Token<'a> = <Self::Parameters<'a> as alloy_sol_types::SolType>::Token<'a>;
105
106                const SIGNATURE: &'static str = #signature;
107                const SELECTOR: [u8; 4] = #selector;
108
109                #[inline]
110                fn new<'a>(tuple: <Self::Parameters<'a> as alloy_sol_types::SolType>::RustType) -> Self {
111                    tuple.into()
112                }
113
114                #[inline]
115                fn tokenize(&self) -> Self::Token<'_> {
116                    #tokenize_impl
117                }
118
119                #[inline]
120                fn abi_decode_raw_validate(data: &[u8]) -> alloy_sol_types::Result<Self> {
121                    <Self::Parameters<'_> as alloy_sol_types::SolType>::abi_decode_sequence_validate(data).map(Self::new)
122                }
123            }
124
125            #abi
126        };
127    };
128    Ok(tokens)
129}
