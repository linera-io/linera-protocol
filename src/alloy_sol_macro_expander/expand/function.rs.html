13/// Expands an [`ItemFunction`]:
14///
15/// ```ignore (pseudo-code)
16/// pub struct #{name}Call {
17///     #(pub #argument_name: #argument_type,)*
18/// }
19///
20/// pub struct #{name}Return {
21///     #(pub #return_name: #return_type,)*
22/// }
23///
24/// impl SolCall for #{name}Call {
25///     type Return = #{name}Return;
26///     ...
27/// }
28/// ```
29pub(super) fn expand(cx: &ExpCtxt<'_>, function: &ItemFunction) -> Result<TokenStream> {
30    let ItemFunction { parameters, returns, name, kind, .. } = function;
31
32    if matches!(kind, FunctionKind::Constructor(_)) {
33        return expand_constructor(cx, function);
34    }
35
36    if name.is_none() {
37        // ignore functions without names (modifiers...)
38        return Ok(quote!());
39    };
40
41    let returns = returns.as_ref().map(|r| &r.returns).unwrap_or_default();
42
43    cx.assert_resolved(parameters)?;
44    if !returns.is_empty() {
45        cx.assert_resolved(returns)?;
46    }
47
48    let (sol_attrs, mut call_attrs) = function.split_attrs()?;
49    let mut return_attrs = call_attrs.clone();
50    cx.derives(&mut call_attrs, parameters, true);
51    if !returns.is_empty() {
52        cx.derives(&mut return_attrs, returns, true);
53    }
54    let docs = sol_attrs.docs.or(cx.attrs.docs).unwrap_or(true);
55    let abi = sol_attrs.abi.or(cx.attrs.abi).unwrap_or(false);
56
57    let call_name = cx.call_name(function);
58    let return_name = cx.return_name(function);
59
60    let call_fields = expand_fields(parameters, cx);
61    let return_fields = expand_fields(returns, cx);
62
63    let call_tuple = expand_tuple_types(parameters.types(), cx).0;
64    let return_tuple = expand_tuple_types(returns.types(), cx).0;
65
66    let converts = expand_from_into_tuples(&call_name, parameters, cx, FieldKind::Deconstruct);
67    let return_converts = expand_from_into_tuples(&return_name, returns, cx, FieldKind::Original);
68
69    let signature = cx.function_signature(function);
70    let selector = crate::utils::selector(&signature);
71    let tokenize_impl = expand_tokenize(parameters, cx, FieldKind::Deconstruct);
72
73    let call_doc = docs.then(|| {
74        let selector = hex::encode_prefixed(selector.array.as_slice());
75        mk_doc(format!(
76            "Function with signature `{signature}` and selector `{selector}`.\n\
77            ```solidity\n{function}\n```"
78        ))
79    });
80    let return_doc = docs.then(|| {
81        mk_doc(format!(
82            "Container type for the return parameters of the [`{signature}`]({call_name}) function."
83        ))
84    });
85
86    let abi: Option<TokenStream> = abi.then(|| {
87        if_json! {
88            let function = super::to_abi::generate(function, cx);
89            quote! {
90                #[automatically_derived]
91                impl alloy_sol_types::JsonAbiExt for #call_name {
92                    type Abi = alloy_sol_types::private::alloy_json_abi::Function;
93
94                    #[inline]
95                    fn abi() -> Self::Abi {
96                        #function
97                    }
98                }
99            }
100        }
101    });
102
103    let call_struct = if parameters.is_empty() {
104        quote! {
105            pub struct #call_name;
106        }
107    } else if parameters.len() == 1 && parameters[0].name.is_none() {
108        let ty = cx.expand_rust_type(&parameters[0].ty);
109        quote! {
110            pub struct #call_name(pub #ty);
111        }
112    } else {
113        quote! {
114            pub struct #call_name {
115                #(#call_fields),*
116            }
117        }
118    };
119
120    let alloy_sol_types = &cx.crates.sol_types;
121
122    let decode_sequence =
123        quote!(<Self::ReturnTuple<'_> as alloy_sol_types::SolType>::abi_decode_sequence(data));
124
125    // Determine whether the return type should directly yield result or the <name>Return struct.
126    let is_single_return = returns.len() == 1;
127    let return_type =
128        if is_single_return { cx.expand_rust_type(&returns[0].ty) } else { quote!(#return_name) };
129    let tokenize_returns_impl = if is_single_return {
130        quote!()
131    } else {
132        let imp = expand_tokenize(returns, cx, FieldKind::Original);
133        quote! {
134            impl #return_name {
135                fn _tokenize(&self) -> <#call_name as alloy_sol_types::SolCall>::ReturnToken<'_> {
136                    #imp
137                }
138            }
139        }
140    };
141    let tokenize_returns = if is_single_return {
142        let ty = cx.expand_type(&returns[0].ty);
143        quote! { (<#ty as alloy_sol_types::SolType>::tokenize(ret),) }
144    } else {
145        quote! { #return_name::_tokenize(ret) }
146    };
147    let decode_returns = if is_single_return {
148        let name = anon_name((0, returns[0].name.as_ref()));
149        quote! {
150            #decode_sequence.map(|r| {
151                let r: #return_name = r.into();
152                r.#name
153            })
154        }
155    } else {
156        quote!(#decode_sequence.map(Into::into))
157    };
158
159    let decode_sequence_validate = quote!(
160        <Self::ReturnTuple<'_> as alloy_sol_types::SolType>::abi_decode_sequence_validate(data)
161    );
162    let decode_returns_validate = if is_single_return {
163        let name = anon_name((0, returns[0].name.as_ref()));
164        quote! {
165            #decode_sequence_validate.map(|r| {
166                let r: #return_name = r.into();
167                r.#name
168            })
169        }
170    } else {
171        quote!(#decode_sequence_validate.map(Into::into))
172    };
173
174    let tokens = quote! {
175        #(#call_attrs)*
176        #call_doc
177        #[allow(non_camel_case_types, non_snake_case, clippy::pub_underscore_fields)]
178        #[derive(Clone)]
179        #call_struct
180
181        #(#return_attrs)*
182        #return_doc
183        #[allow(non_camel_case_types, non_snake_case, clippy::pub_underscore_fields)]
184        #[derive(Clone)]
185        pub struct #return_name {
186            #(#return_fields),*
187        }
188
189        #[allow(non_camel_case_types, non_snake_case, clippy::pub_underscore_fields, clippy::style)]
190        const _: () = {
191            use #alloy_sol_types as alloy_sol_types;
192
193            { #converts }
194            { #return_converts }
195
196            #tokenize_returns_impl
197
198            #[automatically_derived]
199            impl alloy_sol_types::SolCall for #call_name {
200                type Parameters<'a> = #call_tuple;
201                type Token<'a> = <Self::Parameters<'a> as alloy_sol_types::SolType>::Token<'a>;
202
203                type Return = #return_type;
204
205                type ReturnTuple<'a> = #return_tuple;
206                type ReturnToken<'a> = <Self::ReturnTuple<'a> as alloy_sol_types::SolType>::Token<'a>;
207
208                const SIGNATURE: &'static str = #signature;
209                const SELECTOR: [u8; 4] = #selector;
210
211                #[inline]
212                fn new<'a>(tuple: <Self::Parameters<'a> as alloy_sol_types::SolType>::RustType) -> Self {
213                    tuple.into()
214                }
215
216                #[inline]
217                fn tokenize(&self) -> Self::Token<'_> {
218                    #tokenize_impl
219                }
220
221                #[inline]
222                fn tokenize_returns(ret: &Self::Return) -> Self::ReturnToken<'_> {
223                    #tokenize_returns
224                }
225
226                #[inline]
227                fn abi_decode_returns(data: &[u8]) -> alloy_sol_types::Result<Self::Return> {
228                    #decode_returns
229                }
230
231                #[inline]
232                fn abi_decode_returns_validate(data: &[u8]) -> alloy_sol_types::Result<Self::Return> {
233                    #decode_returns_validate
234                }
235            }
236
237            #abi
238        };
239    };
240    Ok(tokens)
241}

243fn expand_constructor(cx: &ExpCtxt<'_>, constructor: &ItemFunction) -> Result<TokenStream> {
244    let ItemFunction { parameters, .. } = constructor;
245
246    let (sol_attrs, call_attrs) = constructor.split_attrs()?;
247    let docs = sol_attrs.docs.or(cx.attrs.docs).unwrap_or(true);
248
249    let alloy_sol_types = &cx.crates.sol_types;
250
251    let call_name = format_ident!("constructorCall").with_span(constructor.kind.span());
252    let call_fields = expand_fields(parameters, cx);
253    let call_tuple = expand_tuple_types(parameters.types(), cx).0;
254    let converts = expand_from_into_tuples(&call_name, parameters, cx, FieldKind::Original);
255    let tokenize_impl = expand_tokenize(parameters, cx, FieldKind::Original);
256
257    let call_doc = docs.then(|| {
258        mk_doc(format!(
259            "Constructor`.\n\
260            ```solidity\n{constructor}\n```"
261        ))
262    });
263
264    let tokens = quote! {
265        #(#call_attrs)*
266        #call_doc
267        #[allow(non_camel_case_types, non_snake_case, clippy::pub_underscore_fields)]
268        #[derive(Clone)]
269        pub struct #call_name {
270            #(#call_fields),*
271        }
272
273        const _: () = {
274            use #alloy_sol_types as alloy_sol_types;
275
276            { #converts }
277
278            #[automatically_derived]
279            impl alloy_sol_types::SolConstructor for #call_name {
280                type Parameters<'a> = #call_tuple;
281                type Token<'a> = <Self::Parameters<'a> as alloy_sol_types::SolType>::Token<'a>;
282
283                #[inline]
284                fn new<'a>(tuple: <Self::Parameters<'a> as alloy_sol_types::SolType>::RustType) -> Self {
285                    tuple.into()
286                }
287
288                #[inline]
289                fn tokenize(&self) -> Self::Token<'_> {
290                    #tokenize_impl
291                }
292            }
293        };
294    };
295    Ok(tokens)
