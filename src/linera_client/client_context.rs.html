66/// Results from querying a validator about version, network description, and chain info.
67pub struct ValidatorQueryResults {
68    /// The validator's version information.
69    pub version_info: Result<VersionInfo, Error>,
70    /// The validator's genesis config hash.
71    pub genesis_config_hash: Result<CryptoHash, Error>,
72    /// The validator's chain info (if valid and signature check passed).
73    pub chain_info: Result<ChainInfo, Error>,
74}

76impl ValidatorQueryResults {
77    /// Returns a vector of references to all errors in the query results.
78    pub fn errors(&self) -> Vec<&Error> {
79        let mut errors = Vec::new();
80        if let Err(e) = &self.version_info {
81            errors.push(e);
82        }
83        if let Err(e) = &self.genesis_config_hash {
84            errors.push(e);
85        }
86        if let Err(e) = &self.chain_info {
87            errors.push(e);
88        }
89        errors
90    }
91
92    /// Prints validator information to stdout.
93    ///
94    /// Prints public key, address, and optionally weight, version info, and chain info.
95    /// If `reference` is provided, only prints fields that differ from the reference.
96    pub fn print(
97        &self,
98        public_key: Option<&ValidatorPublicKey>,
99        address: Option<&str>,
100        weight: Option<u64>,
101        reference: Option<&ValidatorQueryResults>,
102    ) {
103        if let Some(key) = public_key {
104            println!("Public key: {}", key);
105        }
106        if let Some(address) = address {
107            println!("Address: {}", address);
108        }
109        if let Some(w) = weight {
110            println!("Weight: {}", w);
111        }
112
113        let ref_version = reference.and_then(|ref_results| ref_results.version_info.as_ref().ok());
114        match &self.version_info {
115            Ok(version_info) => {
116                if ref_version.is_none_or(|ref_v| ref_v.crate_version != version_info.crate_version)
117                {
118                    println!("Linera protocol: v{}", version_info.crate_version);
119                }
120                if ref_version.is_none_or(|ref_v| ref_v.rpc_hash != version_info.rpc_hash) {
121                    println!("RPC API hash: {}", version_info.rpc_hash);
122                }
123                if ref_version.is_none_or(|ref_v| ref_v.graphql_hash != version_info.graphql_hash) {
124                    println!("GraphQL API hash: {}", version_info.graphql_hash);
125                }
126                if ref_version.is_none_or(|ref_v| ref_v.wit_hash != version_info.wit_hash) {
127                    println!("WIT API hash: v{}", version_info.wit_hash);
128                }
129                if ref_version.is_none_or(|ref_v| {
130                    (&ref_v.git_commit, ref_v.git_dirty)
131                        != (&version_info.git_commit, version_info.git_dirty)
132                }) {
133                    println!(
134                        "Source code: {}/tree/{}{}",
135                        env!("CARGO_PKG_REPOSITORY"),
136                        version_info.git_commit,
137                        if version_info.git_dirty {
138                            " (dirty)"
139                        } else {
140                            ""
141                        }
142                    );
143                }
144            }
145            Err(err) => println!("Error getting version info: {err}"),
146        }
147
148        let ref_genesis_hash =
149            reference.and_then(|ref_results| ref_results.genesis_config_hash.as_ref().ok());
150        match &self.genesis_config_hash {
151            Ok(hash) if ref_genesis_hash.is_some_and(|ref_hash| ref_hash == hash) => {}
152            Ok(hash) => println!("Genesis config hash: {hash}"),
153            Err(err) => println!("Error getting genesis config: {err}"),
154        }
155
156        let ref_info = reference.and_then(|ref_results| ref_results.chain_info.as_ref().ok());
157        match &self.chain_info {
158            Ok(info) => {
159                if ref_info.is_none_or(|ref_info| info.block_hash != ref_info.block_hash) {
160                    if let Some(hash) = info.block_hash {
161                        println!("Block hash: {}", hash);
162                    } else {
163                        println!("Block hash: None");
164                    }
165                }
166                if ref_info
167                    .is_none_or(|ref_info| info.next_block_height != ref_info.next_block_height)
168                {
169                    println!("Next height: {}", info.next_block_height);
170                }
171                if ref_info.is_none_or(|ref_info| info.timestamp != ref_info.timestamp) {
172                    println!("Timestamp: {}", info.timestamp);
173                }
174                if ref_info.is_none_or(|ref_info| info.epoch != ref_info.epoch) {
175                    println!("Epoch: {}", info.epoch);
176                }
177                if ref_info.is_none_or(|ref_info| {
178                    info.manager.current_round != ref_info.manager.current_round
179                }) {
180                    println!("Round: {}", info.manager.current_round);
181                }
182                if let Some(locking) = &info.manager.requested_locking {
183                    match &**locking {
184                        LockingBlock::Fast(proposal) => {
185                            println!(
186                                "Locking fast block from {}",
187                                proposal.content.block.timestamp
188                            );
189                        }
190                        LockingBlock::Regular(validated) => {
191                            println!(
192                                "Locking block {} in {} from {}",
193                                validated.hash(),
194                                validated.round,
195                                validated.block().header.timestamp
196                            );
197                        }
198                    }
199                }
200            }
201            Err(err) => println!("Error getting chain info: {err}"),
202        }
203    }

493    pub async fn process_inbox(
494        &mut self,
495        chain_client: &ChainClient<Env>,
496    ) -> Result<Vec<ConfirmedBlockCertificate>, Error> {
497        let mut certificates = Vec::new();
498        // Try processing the inbox optimistically without waiting for validator notifications.
499        let (new_certificates, maybe_timeout) = {
500            chain_client.synchronize_from_validators().await?;
501            let result = chain_client.process_inbox_without_prepare().await;
502            self.update_wallet_from_client(chain_client).await?;
503            if result.is_err() {
504                self.save_wallet().await?;
505            }
506            result?
507        };
508        certificates.extend(new_certificates);
509        if maybe_timeout.is_none() {
510            self.save_wallet().await?;
511            return Ok(certificates);
512        }
513
514        // Start listening for notifications, so we learn about new rounds and blocks.
515        let (listener, _listen_handle, mut notification_stream) =
516            chain_client.listen(ListeningMode::FullChain).await?;
517        self.chain_listeners.spawn_task(listener);
518
519        loop {
520            let (new_certificates, maybe_timeout) = {
521                let result = chain_client.process_inbox().await;
522                self.update_wallet_from_client(chain_client).await?;
523                if result.is_err() {
524                    self.save_wallet().await?;
525                }
526                result?
527            };
528            certificates.extend(new_certificates);
529            if let Some(timestamp) = maybe_timeout {
530                util::wait_for_next_round(&mut notification_stream, timestamp).await
531            } else {
532                self.save_wallet().await?;
533                return Ok(certificates);
534            }
535        }
536    }

538    pub async fn assign_new_chain_to_key(
539        &mut self,
540        chain_id: ChainId,
541        owner: AccountOwner,
542    ) -> Result<(), Error> {
543        self.client.track_chain(chain_id);
544        let client = self.make_chain_client(chain_id);
545        let chain_description = client.get_chain_description().await?;
546        let config = chain_description.config();
547
548        if !config.ownership.verify_owner(&owner) {
549            tracing::error!(
550                "The chain with the ID returned by the faucet is not owned by you. \
551                Please make sure you are connecting to a genuine faucet."
552            );
553            return Err(error::Inner::ChainOwnership.into());
554        }
555
556        self.wallet_mut()
557            .mutate(|w| {
558                w.assign_new_chain_to_owner(
559                    owner,
560                    chain_id,
561                    chain_description.timestamp(),
562                    chain_description.config().epoch,
563                )
564            })
565            .await
566            .map_err(|e| error::Inner::Persistence(Box::new(e)))?
567            .context("assigning new chain")?;
568        Ok(())
569    }

571    /// Applies the given function to the chain client.
572    ///
573    /// Updates the wallet regardless of the outcome. As long as the function returns a round
574    /// timeout, it will wait and retry.
575    pub async fn apply_client_command<E, F, Fut, T>(
576        &mut self,
577        client: &ChainClient<Env>,
578        mut f: F,
579    ) -> Result<T, Error>
580    where
581        F: FnMut(&ChainClient<Env>) -> Fut,
582        Fut: Future<Output = Result<ClientOutcome<T>, E>>,
583        Error: From<E>,
584    {
585        client.prepare_chain().await?;
586        // Try applying f optimistically without validator notifications. Return if committed.
587        let result = f(client).await;
588        self.update_wallet_from_client(client).await?;
589        if let ClientOutcome::Committed(t) = result? {
590            return Ok(t);
591        }
592
593        // Start listening for notifications, so we learn about new rounds and blocks.
594        let (listener, _listen_handle, mut notification_stream) =
595            client.listen(ListeningMode::FullChain).await?;
596        self.chain_listeners.spawn_task(listener);
597
598        loop {
599            // Try applying f. Return if committed.
600            let result = f(client).await;
601            self.update_wallet_from_client(client).await?;
602            let timeout = match result? {
603                ClientOutcome::Committed(t) => return Ok(t),
604                ClientOutcome::WaitForTimeout(timeout) => timeout,
605            };
606            // Otherwise wait and try again in the next round.
607            util::wait_for_next_round(&mut notification_stream, timeout).await;
608        }
609    }

618    pub async fn change_ownership(
619        &mut self,
620        chain_id: Option<ChainId>,
621        ownership_config: ChainOwnershipConfig,
622    ) -> Result<(), Error> {
623        let chain_id = chain_id.unwrap_or_else(|| self.default_chain());
624        let chain_client = self.make_chain_client(chain_id);
625        info!(
626            ?ownership_config, %chain_id, preferred_owner=?chain_client.preferred_owner(),
627            "Changing ownership of a chain"
628        );
629        let time_start = Instant::now();
630        let ownership = ChainOwnership::try_from(ownership_config)?;
631
632        let certificate = self
633            .apply_client_command(&chain_client, |chain_client| {
634                let ownership = ownership.clone();
635                let chain_client = chain_client.clone();
636                async move {
637                    chain_client
638                        .change_ownership(ownership)
639                        .await
640                        .map_err(Error::from)
641                        .context("Failed to change ownership")
642                }
643            })
644            .await?;
645        let time_total = time_start.elapsed();
646        info!("Operation confirmed after {} ms", time_total.as_millis());
647        debug!("{:?}", certificate);
648        Ok(())
649    }

666    pub async fn check_compatible_version_info(
667        &self,
668        address: &str,
669        node: &impl ValidatorNode,
670    ) -> Result<VersionInfo, Error> {
671        match node.get_version_info().await {
672            Ok(version_info) if version_info.is_compatible_with(&linera_version::VERSION_INFO) => {
673                debug!(
674                    "Version information for validator {address}: {}",
675                    version_info
676                );
677                Ok(version_info)
678            }
679            Ok(version_info) => Err(error::Inner::UnexpectedVersionInfo {
680                remote: Box::new(version_info),
681                local: Box::new(linera_version::VERSION_INFO.clone()),
682            }
683            .into()),
684            Err(error) => Err(error::Inner::UnavailableVersionInfo {

692    pub async fn check_matching_network_description(
693        &self,
694        address: &str,
695        node: &impl ValidatorNode,
696    ) -> Result<CryptoHash, Error> {
697        let network_description = self.wallet().genesis_config().network_description();
698        match node.get_network_description().await {
699            Ok(description) => {
700                if description == network_description {
701                    Ok(description.genesis_config_hash)
702                } else {
703                    Err(error::Inner::UnexpectedNetworkDescription {
704                        remote: Box::new(description),
705                        local: Box::new(network_description),
706                    }
707                    .into())
708                }
709            }
710            Err(error) => Err(error::Inner::UnavailableNetworkDescription {
711                address: address.to_string(),
712                error: Box::new(error),
713            }
714            .into()),
715        }
716    }

718    pub async fn check_validator_chain_info_response(
719        &self,
720        public_key: Option<&ValidatorPublicKey>,
721        address: &str,
722        node: &impl ValidatorNode,
723        chain_id: ChainId,
724    ) -> Result<ChainInfo, Error> {
725        let query = ChainInfoQuery::new(chain_id).with_manager_values();
726        match node.handle_chain_info_query(query).await {
727            Ok(response) => {
728                debug!(
729                    "Validator {address} sees chain {chain_id} at block height {} and epoch {:?}",
730                    response.info.next_block_height, response.info.epoch,
731                );
732                if let Some(public_key) = public_key {
733                    if response.check(*public_key).is_ok() {
734                        debug!("Signature for public key {public_key} is OK.");
735                    } else {
736                        return Err(error::Inner::InvalidSignature {
737                            public_key: *public_key,
738                        }
739                        .into());
740                    }
741                } else {
742                    warn!("Not checking signature as public key was not given");
743                }
744                Ok(*response.info)
745            }
746            Err(error) => Err(error::Inner::UnavailableChainInfo {
747                address: address.to_string(),
748                chain_id,
749                error: Box::new(error),
750            }
751            .into()),
752        }
753    }

829                    chain_client

833                }
834            })
835            .await?;
836
837        info!("{}", "Module published successfully!");
838
839        info!("Synchronizing client and processing inbox");
840        self.process_inbox(chain_client).await?;

900    pub async fn prepare_for_benchmark(
901        &mut self,
902        num_chains: usize,
903        tokens_per_chain: Amount,
904        fungible_application_id: Option<ApplicationId>,
905        pub_keys: Vec<AccountPublicKey>,
906        chains_config_path: Option<&Path>,
907    ) -> Result<(Vec<ChainClient<Env>>, Vec<ChainId>), Error> {
908        let start = Instant::now();
909        // Below all block proposals are supposed to succeed without retries, we
910        // must make sure that all incoming payments have been accepted on-chain
911        // and that no validator is missing user certificates.
912        self.process_inboxes_and_force_validator_updates().await;
913        info!(
914            "Processed inboxes and forced validator updates in {} ms",
915            start.elapsed().as_millis()
916        );
917
918        let start = Instant::now();
919        let (benchmark_chains, chain_clients) = self
920            .make_benchmark_chains(
921                num_chains,
922                tokens_per_chain,
923                pub_keys,
924                chains_config_path.is_some(),
925            )
926            .await?;
927        info!(
928            "Got {} chains in {} ms",
929            num_chains,
930            start.elapsed().as_millis()
931        );
932
933        if let Some(id) = fungible_application_id {
934            let start = Instant::now();
935            self.supply_fungible_tokens(&benchmark_chains, id).await?;
936            info!(
937                "Supplied fungible tokens in {} ms",
938                start.elapsed().as_millis()
939            );
940            // Need to process inboxes to make sure the chains receive the supplied tokens.
941            let start = Instant::now();
942            for chain_client in &chain_clients {
943                chain_client.process_inbox().await?;
944            }
945            info!(
946                "Processed inboxes after supplying fungible tokens in {} ms",
947                start.elapsed().as_millis()
948            );
949        }
950
951        let all_chains = Benchmark::<Env>::get_all_chains(chains_config_path, &benchmark_chains)?;
952        let known_chain_ids: HashSet<_> = benchmark_chains.iter().map(|(id, _)| *id).collect();
953        let unknown_chain_ids: Vec<_> = all_chains
954            .iter()
955            .filter(|id| !known_chain_ids.contains(id))
956            .copied()
957            .collect();
958        if !unknown_chain_ids.is_empty() {
959            // The current client won't have the blobs for the chains in the other wallets. Even
960            // though it will eventually get those blobs, we're getting a head start here and
961            // fetching those blobs in advance.
962            for chain_id in &unknown_chain_ids {
963                self.client.get_chain_description(*chain_id).await?;
964            }
965        }
966
967        Ok((chain_clients, all_chains))
968    }

970    pub async fn wrap_up_benchmark(
971        &mut self,
972        chain_clients: Vec<ChainClient<Env>>,
973        close_chains: bool,
974        wrap_up_max_in_flight: usize,
975    ) -> Result<(), Error> {
976        if close_chains {
977            info!("Closing chains...");
978            let stream = stream::iter(chain_clients)
979                .map(|chain_client| async move {
980                    Benchmark::<Env>::close_benchmark_chain(&chain_client).await?;
981                    info!("Closed chain {:?}", chain_client.chain_id());
982                    Ok::<(), BenchmarkError>(())
983                })
984                .buffer_unordered(wrap_up_max_in_flight);
985            stream.try_collect::<Vec<_>>().await?;
986        } else {
987            info!("Processing inbox for all chains...");
988            let stream = stream::iter(chain_clients.clone())
989                .map(|chain_client| async move {
990                    chain_client.process_inbox().await?;
991                    info!("Processed inbox for chain {:?}", chain_client.chain_id());
992                    Ok::<(), chain_client::Error>(())
993                })
994                .buffer_unordered(wrap_up_max_in_flight);
995            stream.try_collect::<Vec<_>>().await?;
996
997            info!("Updating wallet from chain clients...");
998            for chain_client in chain_clients {
999                let info = chain_client.chain_info().await?;
1000                let client_owner = chain_client.preferred_owner();
1001                let pending_proposal = chain_client.pending_proposal().clone();
1002                self.wallet
1003                    .as_mut()
1004                    .update_from_info(pending_proposal, client_owner, &info);
1005            }
1006            self.save_wallet().await?;
1007        }
1008
1009        Ok(())
1010    }

1012    async fn process_inboxes_and_force_validator_updates(&mut self) {
1013        let mut chain_clients = vec![];
1014        for chain_id in &self.wallet.owned_chain_ids() {
1015            chain_clients.push(self.make_chain_client(*chain_id));
1016        }
1017
1018        let mut join_set = task::JoinSet::new();
1019        for chain_client in chain_clients {
1020            join_set.spawn(async move {
1021                Self::process_inbox_without_updating_wallet(&chain_client)
1022                    .await
1023                    .expect("Processing inbox should not fail!");
1024                chain_client
1025            });
1026        }
1027
1028        let chain_clients = join_set.join_all().await;
1029        for chain_client in &chain_clients {
1030            self.update_wallet_from_client(chain_client).await.unwrap();
1031        }
1032    }

1034    async fn process_inbox_without_updating_wallet(
1035        chain_client: &ChainClient<Env>,
1036    ) -> Result<Vec<ConfirmedBlockCertificate>, Error> {
1037        // Try processing the inbox optimistically without waiting for validator notifications.
1038        chain_client.synchronize_from_validators().await?;
1039        let (certificates, maybe_timeout) = chain_client.process_inbox_without_prepare().await?;
1040        assert!(
1041            maybe_timeout.is_none(),
1042            "Should not timeout within benchmark!"
1043        );
1044
1045        Ok(certificates)

1048    /// Creates chains if necessary, and returns a map of exactly `num_chains` chain IDs
1049    /// with key pairs, as well as a map of the chain clients.
1050    async fn make_benchmark_chains(
1051        &mut self,
1052        num_chains: usize,
1053        balance: Amount,
1054        pub_keys: Vec<AccountPublicKey>,
1055        wallet_only: bool,
1056    ) -> Result<(Vec<(ChainId, AccountOwner)>, Vec<ChainClient<Env>>), Error> {
1057        let mut chains_found_in_wallet = 0;
1058        let mut benchmark_chains = Vec::with_capacity(num_chains);
1059        let mut chain_clients = Vec::with_capacity(num_chains);
1060        let start = Instant::now();
1061        for chain_id in self.wallet.owned_chain_ids() {
1062            if chains_found_in_wallet == num_chains {
1063                break;
1064            }
1065            let chain_client = self.make_chain_client(chain_id);
1066            let ownership = chain_client.chain_info().await?.manager.ownership;
1067            if !ownership.owners.is_empty() || ownership.super_owners.len() != 1 {
1068                continue;
1069            }
1070            chain_client.process_inbox().await?;
1071            benchmark_chains.push((
1072                chain_id,
1073                *ownership
1074                    .super_owners
1075                    .first()
1076                    .expect("should have a super owner"),
1077            ));
1078            chain_clients.push(chain_client);
1079            chains_found_in_wallet += 1;
1080        }
1081        info!(
1082            "Got {} chains from the wallet in {} ms",
1083            benchmark_chains.len(),
1084            start.elapsed().as_millis()
1085        );
1086
1087        let num_chains_to_create = num_chains - chains_found_in_wallet;
1088
1089        let default_chain_id = self
1090            .wallet
1091            .default_chain()
1092            .expect("should have default chain");
1093        let default_chain_client = self.make_chain_client(default_chain_id);
1094
1095        if num_chains_to_create > 0 {
1096            if wallet_only {
1097                return Err(
1098                    error::Inner::Benchmark(BenchmarkError::NotEnoughChainsInWallet(
1099                        num_chains,
1100                        chains_found_in_wallet,
1101                    ))
1102                    .into(),
1103                );
1104            }
1105            let mut pub_keys_iter = pub_keys.into_iter().take(num_chains_to_create);
1106            let operations_per_block = 900; // Over this we seem to hit the block size limits.
1107            for i in (0..num_chains_to_create).step_by(operations_per_block) {
1108                let num_new_chains = operations_per_block.min(num_chains_to_create - i);
1109                let pub_key = pub_keys_iter.next().unwrap();
1110                let owner = pub_key.into();
1111
1112                let certificate = Self::execute_open_chains_operations(
1113                    num_new_chains,
1114                    &default_chain_client,
1115                    balance,
1116                    owner,
1117                )
1118                .await?;
1119                info!("Block executed successfully");
1120
1121                let block = certificate.block();
1122                for i in 0..num_new_chains {
1123                    let chain_id = block.body.blobs[i]
1124                        .iter()
1125                        .find(|blob| blob.id().blob_type == BlobType::ChainDescription)
1126                        .map(|blob| ChainId(blob.id().hash))
1127                        .expect("failed to create a new chain");
1128                    self.client.track_chain(chain_id);
1129
1130                    let mut chain_client = self.client.create_chain_client(
1131                        chain_id,
1132                        None,
1133                        BlockHeight::ZERO,
1134                        None,
1135                        Some(owner),
1136                        self.timing_sender(),
1137                    );
1138                    chain_client.set_preferred_owner(owner);
1139                    chain_client.process_inbox().await?;
1140                    benchmark_chains.push((chain_id, owner));
1141                    chain_clients.push(chain_client);
1142                }
1143            }
1144
1145            info!(
1146                "Created {} chains in {} ms",
1147                num_chains_to_create,
1148                start.elapsed().as_millis()
1149            );
1150        }
1151
1152        info!("Updating wallet from client");
1153        self.update_wallet_from_client(&default_chain_client)
1154            .await?;
1155        info!("Retrying pending outgoing messages");
1156        default_chain_client
1157            .retry_pending_outgoing_messages()
1158            .await
1159            .context("outgoing messages to create the new chains should be delivered")?;
1160        info!("Processing default chain inbox");
1161        default_chain_client.process_inbox().await?;
1162
1163        assert_eq!(
1164            benchmark_chains.len(),
1165            chain_clients.len(),
1166            "benchmark_chains and chain_clients must have the same size"
1167        );
1168
1169        Ok((benchmark_chains, chain_clients))
1170    }

1172    async fn execute_open_chains_operations(

1174        chain_client: &ChainClient<Env>,
1175        balance: Amount,
1176        owner: AccountOwner,
1177    ) -> Result<ConfirmedBlockCertificate, Error> {
1178        let config = OpenChainConfig {
1179            ownership: ChainOwnership::single_super(owner),
1180            balance,
1181            application_permissions: Default::default(),
1182        };
1183        let operations = iter::repeat_n(
1184            Operation::system(SystemOperation::OpenChain(config)),
1185            num_new_chains,
1186        )
1187        .collect();
1188        info!("Executing {} OpenChain operations", num_new_chains);
1189        Ok(chain_client
1190            .execute_operations(operations, vec![])
1191            .await?
1192            .expect("should execute block with OpenChain operations"))
1193    }

1198        key_pairs: &[(ChainId, AccountOwner)],
1199        application_id: ApplicationId,
1200    ) -> Result<(), Error> {

1203            .default_chain()
1204            .expect("should have default chain");

1206        // This should be enough to run the benchmark at 1M TPS for an hour.
1207        let amount = Amount::from_nanos(4);
1208        let operations: Vec<Operation> = key_pairs
1209            .iter()
1210            .map(|(chain_id, owner)| {
1211                Benchmark::<Env>::fungible_transfer(
1212                    application_id,
1213                    *chain_id,
1214                    default_key,
1215                    *owner,
1216                    amount,
1217                )
1218            })
1219            .collect();
1220        let chain_client = self.make_chain_client(default_chain_id);
1221        // Put at most 1000 fungible token operations in each block.
1222        for operation_chunk in operations.chunks(1000) {
1223            chain_client
1224                .execute_operations(operation_chunk.to_vec(), vec![])
1225                .await?
1226                .expect("should execute block with Transfer operations");
1227        }
1228        self.update_wallet_from_client(&chain_client).await?;
1229
1230        Ok(())
