67#[inline(always)]
68pub fn i16x8_packus(a: V128, b: V128) -> V128 {
69    let a: [i16; 8] = unsafe { t(a) };
70    let b: [i16; 8] = unsafe { t(b) };
71    let sat_u8 = |x: i16| {
72        if x < 0 {
73            0
74        } else if x > 255 {
75            255
76        } else {
77            x as u8
78        }
79    };
80    let mut c: [u8; 16] = [0; 16];
81    for i in 0..8 {
82        c[i] = sat_u8(a[i]);
83        c[i + 8] = sat_u8(b[i]);
84    }
85    V128::from_bytes(c)
