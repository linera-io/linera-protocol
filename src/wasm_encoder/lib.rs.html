1//! A WebAssembly encoder.
2//!
3//! The main builder is the [`Module`]. You can build a section with a
4//! section-specific builder, like [`TypeSection`] or [`ImportSection`], and
5//! then add it to the module with [`Module::section`]. When you are finished
6//! building the module, call either [`Module::as_slice`] or [`Module::finish`]
7//! to get the encoded bytes. The former gives a shared reference to the
8//! underlying bytes as a slice, while the latter gives you ownership of them as
9//! a vector.
10//!
11//! # Example
12//!
13//! If we wanted to build this module:
14//!
15//! ```wasm
16//! (module
17//!   (type (func (param i32 i32) (result i32)))
18//!   (func (type 0)
19//!     local.get 0
20//!     local.get 1
21//!     i32.add)
22//!   (export "f" (func 0)))
23//! ```
24//!
25//! then we would do this:
26//!
27//! ```
28//! use wasm_encoder::{
29//!     CodeSection, ExportKind, ExportSection, Function, FunctionSection, Instruction,
30//!     Module, TypeSection, ValType,
31//! };
32//!
33//! let mut module = Module::new();
34//!
35//! // Encode the type section.
36//! let mut types = TypeSection::new();
37//! let params = vec![ValType::I32, ValType::I32];
38//! let results = vec![ValType::I32];
39//! types.function(params, results);
40//! module.section(&types);
41//!
42//! // Encode the function section.
43//! let mut functions = FunctionSection::new();
44//! let type_index = 0;
45//! functions.function(type_index);
46//! module.section(&functions);
47//!
48//! // Encode the export section.
49//! let mut exports = ExportSection::new();
50//! exports.export("f", ExportKind::Func, 0);
51//! module.section(&exports);
52//!
53//! // Encode the code section.
54//! let mut codes = CodeSection::new();
55//! let locals = vec![];
56//! let mut f = Function::new(locals);
57//! f.instruction(&Instruction::LocalGet(0));
58//! f.instruction(&Instruction::LocalGet(1));
59//! f.instruction(&Instruction::I32Add);
60//! f.instruction(&Instruction::End);
61//! codes.function(&f);
62//! module.section(&codes);
63//!
64//! // Extract the encoded Wasm bytes for this module.
65//! let wasm_bytes = module.finish();
66//!
67//! // We generated a valid Wasm module!
68//! assert!(wasmparser::validate(&wasm_bytes).is_ok());
69//! ```

84/// Implemented by types that can be encoded into a byte sink.
85pub trait Encode {
86    /// Encode the type into the given byte sink.
87    fn encode(&self, sink: &mut Vec<u8>);
88}
89
90impl<T: Encode + ?Sized> Encode for &'_ T {
91    fn encode(&self, sink: &mut Vec<u8>) {
92        T::encode(self, sink)
93    }
94}
95
96impl<T: Encode> Encode for [T] {
97    fn encode(&self, sink: &mut Vec<u8>) {
98        self.len().encode(sink);
99        for item in self {
100            item.encode(sink);
101        }
102    }
103}
104
105impl Encode for [u8] {
106    fn encode(&self, sink: &mut Vec<u8>) {
107        self.len().encode(sink);
108        sink.extend(self);
109    }
110}
111
112impl Encode for str {
113    fn encode(&self, sink: &mut Vec<u8>) {
114        self.len().encode(sink);
115        sink.extend_from_slice(self.as_bytes());
116    }
117}
118
119impl Encode for usize {
120    fn encode(&self, sink: &mut Vec<u8>) {
121        assert!(*self <= u32::max_value() as usize);
122        (*self as u32).encode(sink)
123    }
124}
125
126impl Encode for u32 {
127    fn encode(&self, sink: &mut Vec<u8>) {
128        leb128::write::unsigned(sink, (*self).into()).unwrap();
129    }
130}
131
132impl Encode for i32 {
133    fn encode(&self, sink: &mut Vec<u8>) {
134        leb128::write::signed(sink, (*self).into()).unwrap();
135    }
136}
137
138impl Encode for u64 {
139    fn encode(&self, sink: &mut Vec<u8>) {
140        leb128::write::unsigned(sink, *self).unwrap();
141    }
142}
143
144impl Encode for i64 {
145    fn encode(&self, sink: &mut Vec<u8>) {
146        leb128::write::signed(sink, *self).unwrap();
147    }
148}
149
150impl Encode for f32 {
151    fn encode(&self, sink: &mut Vec<u8>) {
152        let bits = self.to_bits();
153        sink.extend(bits.to_le_bytes())
154    }
155}
156
157impl Encode for f64 {
158    fn encode(&self, sink: &mut Vec<u8>) {
159        let bits = self.to_bits();
160        sink.extend(bits.to_le_bytes())
161    }
162}

164fn encode_vec<T, V>(elements: V, sink: &mut Vec<u8>)
165where
166    T: Encode,
167    V: IntoIterator<Item = T>,
168    V::IntoIter: ExactSizeIterator,
169{
170    let elements = elements.into_iter();
171    u32::try_from(elements.len()).unwrap().encode(sink);
172    for x in elements {
173        x.encode(sink);
174    }
175}
176
177impl<T> Encode for Option<T>
178where
179    T: Encode,
180{
181    fn encode(&self, sink: &mut Vec<u8>) {
182        match self {
183            Some(v) => {
184                sink.push(0x01);
185                v.encode(sink);
186            }
187            None => sink.push(0x00),
188        }
189    }
190}
191
192fn encoding_size(n: u32) -> usize {
193    let mut buf = [0u8; 5];
194    leb128::write::unsigned(&mut &mut buf[..], n.into()).unwrap()
195}
196
197fn encode_section(sink: &mut Vec<u8>, count: u32, bytes: &[u8]) {
198    (encoding_size(count) + bytes.len()).encode(sink);
199    count.encode(sink);
200    sink.extend(bytes);
201}
