4/**
5Costs that aim to be roughly compatible with Wasmtime's defaults.
6
7<div class="warning">
8In some cases these will not yield the same results as Wasmtime,
9as Wasmtime will charge for the implicit return at the end of a function,
10while this does not.
11</div>
12*/
13pub struct Costs;
14impl super::Costs for Costs {
15    fn instruction(&self, instruction: &walrus::ir::Instr) -> i32 {
16        use walrus::ir::Instr::*;
17        match instruction {
18            Drop(_) | Block(_) | Loop(_) | Unreachable(_) => 0,
19            _ => 1,
20        }
21    }
22}
23
24#[cfg(test)]
25mod test {
26    use test_case::test_case;
27    use wasm_instrument::parity_wasm;
28
29    impl wasm_instrument::gas_metering::Rules for super::Costs {
30        fn instruction_cost(
31            &self,
32            instruction: &parity_wasm::elements::Instruction,
33        ) -> Option<u32> {
34            use parity_wasm::elements::Instruction::*;
35
36            Some(match instruction {
37                Nop | Drop | Block(_) | Loop(_) | Unreachable | Else | End => 0,
38                _ => 1,
39            })
40        }
41
42        fn memory_grow_cost(&self) -> wasm_instrument::gas_metering::MemoryGrowCost {
43            wasm_instrument::gas_metering::MemoryGrowCost::Free
44        }
45
46        fn call_per_local_cost(&self) -> u32 {
47            0
48        }
49    }
50
51    fn parity_instrument(module: &[u8]) -> anyhow::Result<Vec<u8>> {
52        Ok(wasm_instrument::gas_metering::inject(
53            parity_wasm::deserialize_buffer(module)?,
54            wasm_instrument::gas_metering::host_function::Injector::new("host", "spend"),
55            &super::Costs,
56        )
57        .map_err(|_| anyhow::anyhow!("injection failed"))?
58        .into_bytes()?)
59    }
60
61    #[test_case(include_str!("fixtures/branch.wat"); "branch")]
62    #[test_case(include_str!("fixtures/call.wat"); "call")]
63    #[test_case(include_str!("fixtures/ifs.wat"); "ifs")]
64    #[test_case(include_str!("fixtures/loop.wat"); "loop_")]
65    #[test_case(include_str!("fixtures/simple.wat"); "simple")]
66    #[test_case(include_str!("fixtures/merge.wat"); "merge")]
67    #[test_case(include_str!("fixtures/nesting.wat"); "nesting")]
68    fn wasm_instrument_equivalent(code: &str) -> wasmtime::Result<()> {
69        use wasmtime::ToWasmtimeResult as _;
70
71        let engine = wasmtime::Engine::default();
72        let mut store = wasmtime::Store::new(&engine, 0i64);
73        let mut linker = wasmtime::Linker::new(&engine);
74        linker.func_wrap(
75            "host",
76            "spend",
77            |mut caller: wasmtime::Caller<'_, i64>, cost: i64| {
78                *caller.data_mut() += cost;
79            },
80        )?;
81
82        let module = wat::parse_str(code)?;
83
84        let parity_instrument_cost = {
85            *store.data_mut() = 0;
86            linker.instantiate(
87                &mut store,
88                &wasmtime::Module::from_binary(
89                    &engine,
90                    &parity_instrument(&module).to_wasmtime_result()?,
91                )?,
92            )?;
93            *store.data() as u64
94        };
95
96        let walrus_instrument_cost = {
97            *store.data_mut() = 0;
98            linker.instantiate(
99                &mut store,
100                &wasmtime::Module::from_binary(
101                    &engine,
102                    &crate::instrument(&module, super::Costs, ("host", "spend"))
103                        .to_wasmtime_result()?,
104                )?,
105            )?;
106            *store.data() as u64
107        };
108
109        assert_eq!(
110            walrus_instrument_cost, parity_instrument_cost,
111            "\n\
112             \twalrus-meter cost:    {walrus_instrument_cost}\n\
113             \twasm-instrument cost: {parity_instrument_cost}",
114        );
115
116        Ok(())
117    }
118}
