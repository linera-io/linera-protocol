52/// `Collapse` trait as well. Implementing this trait gives you parallelism
53/// for free as long as you adhere to the requirements described in the

60    /// Process any header lines that precede the main body of samples.
61    ///
62    /// Some formats, such as `dtrace`, contain a header or other non-stack
63    /// information at the beginning of their input files. If header information
64    /// is present, this method **must** consume it (i.e. advance the provided
65    /// reader past it).
66    ///
67    /// This method also provides an opportunity to do processing of actual
68    /// stack data on the main thread before worker threads are spun up. For
69    /// example, `perf` requires reading the first stack in order to know how to
70    /// process the rest; so this method is used for that "upfront" processing.
71    ///
72    /// If the format you are working with does not contain header information
73    /// or does not need any special, up-front processing, just have this method
74    /// return `Ok(())` immediately.
75    fn pre_process<R>(&mut self, reader: &mut R, occurrences: &mut Occurrences) -> io::Result<()>
76    where
77        R: io::BufRead;
78
79    /// Process all samples in a chunk of input (the primary method).
80    ///
81    /// This method receives a reader whose header has already been consumed (see above),
82    /// as well as a mutable reference to an `Occurences` instance (just a hashamp that
83    /// works across multiple threads). Implementers should parse the stack data
84    /// contained in the reader and write output to the provided `Occurrences` map.
85    ///
86    /// This method may be called multiple times to process batches of incoming samples.
87    /// Therefore, make sure that when end-of-file is reached, the collapser now considers
88    /// itself back at the top-level context (e.g., not in the middle of a stack). This
89    /// means that some internal state, e.g. stack buffers, must be reset by the time this
90    /// method returns. Other internal state, e.g. caches, however, may be kept.
91    fn collapse_single_threaded<R>(
92        &mut self,
93        reader: R,
94        occurrences: &mut Occurrences,
95    ) -> io::Result<()>
96    where
97        R: io::BufRead;
98
99    /// Determine the end of a stack.
100    ///
101    /// Worker threads **must** receive full stacks (as opposed to partial stacks); so this method
102    /// determines, for your specific format, when the end of a stack has been reached.
103    ///
104    /// This method should return `true` if the provided line represents the end of a stack;
105    /// `false` otherwise.
106    ///
107    /// If your format requires more information than merely a line of the input data in order
108    /// to determine whether or not you are at the end of a stack, you can retrieve/store
109    /// information on the `self` instance, which is also available to you in this method. This
110    /// method will be called for every line of input data (excluding those consumed by the
111    /// `pre_process` method).
112    fn would_end_stack(&mut self, line: &[u8]) -> bool;
113
114    /// Creates a copy and prepares it to be sent to a different thread.
115    ///
116    /// This method creates a copy of `self` in order to send it to a different thread.
117    /// As such, it should clone all the internal fields of `self` **except** those that
118    /// should be reset because the collapser will now operate in a different stack context.
119    /// For example, any options should be cloned, but any stack buffers or similar "stack state"
120    /// should be reset to, for example, an empty vector before this method returns.
121    fn clone_and_reset_stack_context(&self) -> Self;
122
123    /// Determine if this format corresponds to the input data.
124    ///
125    /// This method, used by the `guess` collapser, should return whether or not the
126    /// implementation corresponds with the given input string, i.e. if the input data
127    /// matches the collapser.
128    ///
129    /// - `None` means "not sure -- need more input"
130    /// - `Some(true)` means "yes, this implementation should work with this string"
131    /// - `Some(false)` means "no, this implementation definitely won't work"
132    #[allow(clippy::wrong_self_convention)]
133    fn is_applicable(&mut self, input: &str) -> Option<bool>;
134
135    /// Returns the number of stacks per job to send to the threadpool.
136    fn nstacks_per_job(&self) -> usize;
137
138    /// Sets the number of stacks per job to send to the threadpool.
139    fn set_nstacks_per_job(&mut self, n: usize);
140
141    /// Returns the number of threads to use.
142    fn nthreads(&self) -> usize;
143
144    /// Sets the number of threads to use.
145    fn set_nthreads(&mut self, n: usize);

151    fn collapse<R, W>(&mut self, mut reader: R, writer: W) -> io::Result<()>
152    where
153        R: io::BufRead,
154        W: io::Write,
155    {
156        let mut occurrences = Occurrences::new(self.nthreads());
157
158        // Consume the header, if any, and do any other pre-processing
159        // that needs to occur.
160        self.pre_process(&mut reader, &mut occurrences)?;
161
162        // Do collapsing.
163        if occurrences.is_concurrent() {
164            self.collapse_multi_threaded(reader, &mut occurrences)?;
165        } else {
166            self.collapse_single_threaded(reader, &mut occurrences)?;
167        }
168
169        // Write results.
170        occurrences.write_and_clear(writer)
171    }

173    #[cfg(not(feature = "multithreaded"))]
174    fn collapse_multi_threaded<R>(&mut self, _: R, _: &mut Occurrences) -> io::Result<()>
175    where
176        R: io::BufRead,
177    {
178        unimplemented!();
179    }

181    #[cfg(feature = "multithreaded")]
182    fn collapse_multi_threaded<R>(
183        &mut self,
184        mut reader: R,
185        occurrences: &mut Occurrences,
186    ) -> io::Result<()>
187    where
188        R: io::BufRead,
189    {
190        let nstacks_per_job = self.nstacks_per_job();
191        let nthreads = self.nthreads();
192
193        assert_ne!(nstacks_per_job, 0);
194        assert!(nthreads > 1);
195        assert!(occurrences.is_concurrent());
196
197        crossbeam_utils::thread::scope(|scope| {
198            // Channel for sending an error from the worker threads to the main thread
199            // in the event a worker has failed.
200            let (tx_error, rx_error) = crossbeam_channel::bounded::<io::Error>(1);
201
202            // Channel for sending input data from the main thread to the worker threads.
203            // We choose `2 * nthreads` as the channel size here in order to limit memory
204            // usage in the case of particularly large input files.
205            let (tx_input, rx_input) = crossbeam_channel::bounded::<Vec<u8>>(2 * nthreads);
206
207            // Channel for worker threads that have errored to signal to all the other
208            // worker threads that they should stop work immediately and return.
209            let (tx_stop, rx_stop) = crossbeam_channel::bounded::<()>(nthreads - 1);
210
211            let mut handles = Vec::with_capacity(nthreads);
212            for _ in 0..nthreads {
213                let tx_error = tx_error.clone();
214                let rx_input = rx_input.clone();
215                let (tx_stop, rx_stop) = (tx_stop.clone(), rx_stop.clone());
216
217                let mut folder = self.clone_and_reset_stack_context();
218                let mut occurrences = occurrences.clone();
219
220                // Launch the worker thread...
221                let handle = scope.spawn(move |_| loop {
222                    crossbeam_channel::select! {
223                        recv(rx_input) -> input => {
224                            // Receive input from the main thread.
225                            let data = match input {
226                                Ok(data) => data,
227                                // The main threads drops it's handle to the input sender once it's
228                                // finished sending data; so if we get an error here, it means
229                                // there is no more data to be sent and we should exit.
230                                Err(_) => return,
231                            };
232                            // If there is input data, process it.
233                            if let Err(e) = folder.collapse_single_threaded(&data[..], &mut occurrences) {
234                                // In the event of an error...
235                                //
236                                // We notify all the threads about it here, rather than wait for the main input
237                                // loop to see the error, so that we can also stop the input loop from iterating
238                                // through the rest of the file.
239                                //
240                                // If the channel is full, it means another thread has also errored
241                                // and already sent a stop signal to the other threads; so there is
242                                // no need to wait or to check for a `SendError` here.
243                                for _ in 0..(nthreads - 1) {
244                                    let _ = tx_stop.try_send(());
245                                }
246
247                                // Then, send the error produced to the main thread for
248                                // propagation. If the channel is full, it means another thread
249                                // has also errored and already sent its error back to the
250                                // main thread; so there is no need to wait or to check for a
251                                // `SendError` here.
252                                let _ = tx_error.try_send(e);
253
254                                // Finally, return.
255                                return;
256                            }
257                            // If successful, return to the top of the loop and continue to poll
258                            // the input and stop channels.
259                        },
260                        recv(rx_stop) -> _ => {
261                            // Received a signal from another worker thread that it has errored;
262                            // so should cease work immediately and return.
263                            return;
264                        },
265                    }
266                });
267                handles.push(handle);
268            }
269
270            // On the main thread, we're about to start sending data to the worker threads,
271            // but we only want to send data to the worker threads **if** they're still alive!
272            // (if one of them produces an error, all of them will exit early). To ensure we don't try
273            // to send data to dead worker threads, drop the main thread's handle to the input receiver
274            // here. This way, if all the workers die, every handle to the input receiver will have
275            // been dropped and we'll get an error when trying to send data on the input sender,
276            // which will tell us (the main thread) to stop trying to send data and, instead,
277            // skip to trying to pull an error off the error channel.
278            drop(rx_input);
279
280            // Now that we've dropped the main thread's handle to the input sender, start
281            // trying to send data to the worker threads...
282
283            let buf_capacity = usize::next_power_of_two(NBYTES_PER_STACK_GUESS * nstacks_per_job);
284            let mut buf = Vec::with_capacity(buf_capacity);
285            let (mut index, mut nstacks) = (0, 0);
286
287            loop {
288                let n = reader.read_until(b'\n', &mut buf)?;
289                if n == 0 {
290                    // If we've reached the end of the data, send the final chunk to the worker
291                    // threads and break from the loop, The worker threads may or may not still
292                    // be alive (depending on if one errored in between the sending of the last
293                    // chunk and the sending of this one), but either way we should break the loop;
294                    // so there's no need to check for a `SendError` here.
295                    let _ = tx_input.send(buf);
296                    break;
297                }
298                let line = &buf[index..index + n];
299                index += n;
300                if self.would_end_stack(line) {
301                    // If we've reached the end of a stack, count it.
302                    nstacks += 1;
303                    if nstacks == nstacks_per_job {
304                        // If we've accumulated enough stacks to make up a chunk to send to the
305                        // worker threads, try to send it.
306                        let buf_capacity = usize::next_power_of_two(buf.capacity());
307                        let chunk = mem::replace(&mut buf, Vec::with_capacity(buf_capacity));
308                        if tx_input.send(chunk).is_err() {
309                            // If sending the chunk produces a `SendError`, this means that one
310                            // of the worker threads has errored, sent a signal to all the other
311                            // worker threads to shut down, and they have all shutdown, in which
312                            // case we know there will be an error waiting for us on the error
313                            // channel; so we should stop parsing input data (i.e. break).
314                            break;
315                        }
316                        index = 0;
317                        nstacks = 0;
318                    }
319                    continue
320                }
321            }
322
323            // The main thread needs to drop its handle to the input sender here because
324            // that's how we signal to the worker threads that there is no more data coming
325            // on the input channel, in which case they should exit.
326            drop(tx_input);
327
328            // The main thread needs to drop its handle to the error sender here because we
329            // are about to poll the error receiver for errors, which will block until all
330            // the error senders have been dropped (including ours).
331            drop(tx_error);
332
333            // Now we poll the error channel, which will block until either:
334            // * all work has been completely successfully,
335            //   in which case the expression below will evaluate to `None`, or
336            // * an error has occurred on one of the worker theads,
337            //   in which case the expression below will evaluate to `Some(<io::Error>)`.
338            if let Some(e) = rx_error.iter().next() {
339                return Err(e);
340            }
341
342            for handle in handles {
343                handle.join().unwrap();
344            }
345
346            Ok(())
347        })
348        .unwrap()
349    }
350}

352/// Occurrences is a HashMap, which uses:
353/// * AHashMap if single-threaded
354/// * DashMap if multi-threaded
355///
356/// This is public because it is part of the sealed `CollapsePrivate` trait's API, but it
357/// is in a crate-private module so is not nameable by downstream library users.
358#[derive(Clone, Debug)]
359pub enum Occurrences {
360    SingleThreaded(AHashMap<String, usize>),
361    #[cfg(feature = "multithreaded")]
362    MultiThreaded(Arc<DashMap<String, usize, ahash::RandomState>>),
363}
364
365impl Occurrences {
366    #[cfg(feature = "multithreaded")]
367    pub(crate) fn new(nthreads: usize) -> Self {
368        assert_ne!(nthreads, 0);
369        if nthreads == 1 {
370            Self::new_single_threaded()
371        } else {
372            Self::new_multi_threaded()
373        }
374    }
375
376    #[cfg(not(feature = "multithreaded"))]
377    pub(crate) fn new(nthreads: usize) -> Self {
378        assert_ne!(nthreads, 0);
379        Self::new_single_threaded()
380    }
381
382    fn new_single_threaded() -> Self {
383        let map =
384            AHashMap::with_capacity_and_hasher(CAPACITY_HASHMAP, ahash::RandomState::default());
385        Occurrences::SingleThreaded(map)
386    }
387
388    #[cfg(feature = "multithreaded")]
389    fn new_multi_threaded() -> Self {
390        let map =
391            DashMap::with_capacity_and_hasher(CAPACITY_HASHMAP, ahash::RandomState::default());
392        Occurrences::MultiThreaded(Arc::new(map))
393    }
394
395    /// Inserts a key-count pair into the map. If the map did not have this key
396    /// present, `None` is returned. If the map did have this key present, the
397    /// value is updated, and the old value is returned.
398    pub(crate) fn insert(&mut self, key: String, count: usize) -> Option<usize> {
399        use self::Occurrences::*;
400        match self {
401            SingleThreaded(map) => map.insert(key, count),
402            #[cfg(feature = "multithreaded")]
403            MultiThreaded(arc) => arc.insert(key, count),
404        }
405    }
406
407    /// Inserts a key-count pair into the map if the key does not already exist.
408    /// If the key does already exist, adds count to the current value of the
409    /// existing key.
410    pub(crate) fn insert_or_add(&mut self, key: String, count: usize) {
411        use self::Occurrences::*;
412        match self {
413            SingleThreaded(map) => *map.entry(key).or_insert(0) += count,
414            #[cfg(feature = "multithreaded")]
415            MultiThreaded(arc) => *arc.entry(key).or_insert(0) += count,
416        }
417    }
418
419    pub(crate) fn is_concurrent(&self) -> bool {
420        use self::Occurrences::*;
421        match self {
422            SingleThreaded(_) => false,
423            #[cfg(feature = "multithreaded")]
424            MultiThreaded(_) => true,
425        }
426    }
427
428    pub(crate) fn write_and_clear<W>(&mut self, mut writer: W) -> io::Result<()>
429    where
430        W: io::Write,
431    {
432        use self::Occurrences::*;
433        match self {
434            SingleThreaded(ref mut map) => {
435                let mut contents: Vec<_> = map.drain().collect();
436                contents.sort();
437                for (key, value) in contents {
438                    writeln!(writer, "{} {}", key, value)?;
439                }
440            }
441            #[cfg(feature = "multithreaded")]
442            MultiThreaded(ref mut arc) => {
443                let map = match Arc::get_mut(arc) {
444                    Some(map) => map,
445                    None => panic!(
446                        "Attempting to drain the contents of a concurrent HashMap \
447                         when more than one thread has access to it, which is \
448                         not allowed."
449                    ),
450                };
451                let map = mem::replace(
452                    map,
453                    DashMap::with_capacity_and_hasher(
454                        CAPACITY_HASHMAP,
455                        ahash::RandomState::default(),
456                    ),
457                );
458                let contents = map.iter().collect::<Vec<_>>();
459                let mut pairs = contents.iter().map(|pair| pair.pair()).collect::<Vec<_>>();
460                pairs.sort();
461                for (key, value) in pairs {
462                    writeln!(writer, "{} {}", key, value)?;
463                }
464            }
465        }
466        writer.flush()?;
467        Ok(())
468    }

471/// Demangles partially demangled Rust symbols that were demangled incorrectly by profilers like
472/// `sample` and `DTrace`.
473///
474/// For example:
475///     `_$LT$grep_searcher..searcher..glue..ReadByLine$LT$$u27$s$C$$u20$M$C$$u20$R$C$$u20$S$GT$$GT$::run::h30ecedc997ad7e32`
476/// becomes
477///     `<grep_searcher::searcher::glue::ReadByLine<'s, M, R, S>>::run`
478///
479/// Non-Rust symobols, or Rust symbols that are already demangled, will be returned unchanged.
480///
481/// Based on code in https://github.com/alexcrichton/rustc-demangle/blob/master/src/legacy.rs
482#[allow(clippy::cognitive_complexity)]
483pub(crate) fn fix_partially_demangled_rust_symbol(symbol: &str) -> Cow<str> {
484    // Rust hashes are hex digits with an `h` prepended.
485    let is_rust_hash =
486        |s: &str| s.starts_with('h') && s[1..].chars().all(|c| c.is_ascii_hexdigit());
487
488    // If there's no trailing Rust hash just return the symbol as is.
489    if symbol.len() < RUST_HASH_LENGTH || !is_rust_hash(&symbol[symbol.len() - RUST_HASH_LENGTH..])
490    {
491        return Cow::Borrowed(symbol);
492    }
493
494    // Strip off trailing hash.
495    let mut rest = &symbol[..symbol.len() - RUST_HASH_LENGTH];
496
497    if rest.ends_with("::") {
498        rest = &rest[..rest.len() - 2];
499    }
500
501    if rest.starts_with("_$") {
502        rest = &rest[1..];
503    }
504
505    let mut demangled = String::new();
506
507    while !rest.is_empty() {
508        if rest.starts_with('.') {
509            if let Some('.') = rest[1..].chars().next() {
510                demangled.push_str("::");
511                rest = &rest[2..];
512            } else {
513                demangled.push('.');
514                rest = &rest[1..];
515            }
516        } else if rest.starts_with('$') {
517            macro_rules! demangle {
518                ($($pat:expr => $demangled:expr,)*) => ({
519                    $(if rest.starts_with($pat) {
520                        demangled.push_str($demangled);
521                        rest = &rest[$pat.len()..];
522                        } else)*
523                    {
524                        demangled.push_str(rest);
525                        break;
526                    }
527
528                })
529            }
530
531            demangle! {
532                "$SP$" => "@",
533                "$BP$" => "*",
534                "$RF$" => "&",
535                "$LT$" => "<",
536                "$GT$" => ">",
537                "$LP$" => "(",
538                "$RP$" => ")",
539                "$C$" => ",",
540                "$u7e$" => "~",
541                "$u20$" => " ",
542                "$u27$" => "'",
543                "$u3d$" => "=",
544                "$u5b$" => "[",
545                "$u5d$" => "]",
546                "$u7b$" => "{",
547                "$u7d$" => "}",
548                "$u3b$" => ";",
549                "$u2b$" => "+",
550                "$u21$" => "!",
551                "$u22$" => "\"",
552            }
553        } else {
554            let idx = match rest.char_indices().find(|&(_, c)| c == '$' || c == '.') {
555                None => rest.len(),
556                Some((i, _)) => i,
557            };
558            demangled.push_str(&rest[..idx]);
559            rest = &rest[idx..];
560        }
561    }
562
563    Cow::Owned(demangled)
564}

618                let mut writer = Vec::new();
619                <C as Collapse>::collapse(folder, &bytes[..], &mut writer)?;
620                let actual = std::str::from_utf8(&writer[..]).unwrap();
621
622                assert_eq!(
623                    actual,
624                    expected,
625                    "Collapsing with {} threads does not produce the same output as collapsing with 1 thread for {}",
626                    n,
627                    path.display()
628                );
629            }
630        }
631
632        Ok(())
633    }

635    pub(crate) fn bench_nstacks<C, P>(folder: &mut C, inputs: &[P]) -> io::Result<()>
636    where
637        C: CollapsePrivate,
638        P: AsRef<Path>,
639    {
640        const MIN_LINES: usize = 2000;
641        const NSAMPLES: usize = 100;
642        const WARMUP_SECS: usize = 3;
643
644        let _stdout = io::stdout();
645        let _stderr = io::stdout();
646
647        let mut stdout = _stdout.lock();
648        let _stderr = _stderr.lock();
649
650        struct Foo<'a> {
651            default: usize,
652            nlines: usize,
653            nstacks: usize,
654            path: &'a Path,
655            results: HashMap<usize, u64>,
656        }
657
658        impl<'a> Foo<'a> {
659            fn new<C>(
660                folder: &mut C,
661                path: &'a Path,
662                bytes: &[u8],
663                stdout: &mut io::StdoutLock,
664            ) -> io::Result<Option<Self>>
665            where
666                C: CollapsePrivate,
667            {
668                let default = folder.nstacks_per_job();
669
670                let (nlines, nstacks) = count_lines_and_stacks(bytes);
671                if nlines < MIN_LINES {
672                    return Ok(None);
673                }
674
675                let mut results = HashMap::default();
676                let iter = vec![default]
677                    .into_iter()
678                    .chain(1..=10)
679                    .chain((20..=nstacks).step_by(10));
680                for nstacks_per_job in iter {
681                    folder.set_nstacks_per_job(nstacks_per_job);
682                    let mut durations = Vec::new();
683                    for _ in 0..NSAMPLES {
684                        let now = Instant::now();
685                        folder.collapse(bytes, io::sink())?;
686                        durations.push(now.elapsed().as_nanos());
687                    }
688                    let avg_duration =
689                        (durations.iter().sum::<u128>() as f64 / durations.len() as f64) as u64;
690                    results.insert(nstacks_per_job, avg_duration);
691                    stdout.write_all(&[b'.'])?;
692                    stdout.flush()?;
693                }
694                Ok(Some(Self {
695                    default,
696                    nlines,
697                    nstacks,
698                    path,
699                    results,
700                }))
701            }
702        }
703
704        impl<'a> fmt::Display for Foo<'a> {
705            fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
706                writeln!(
707                    f,
708                    "{} (nstacks: {}, lines: {})",
709                    self.path.display(),
710                    self.nstacks,
711                    self.nlines
712                )?;
713                let default_duration = self.results[&self.default];
714                let mut results = self.results.iter().collect::<Vec<_>>();
715                results.sort_by(|(_, d1), (_, d2)| (**d1).cmp(*d2));
716                for (nstacks_per_job, duration) in results.iter().take(10) {
717                    writeln!(
718                        f,
719                        "    nstacks_per_job: {:>4} (% of total: {:>3.0}%) | time: {:.0}% of default",
720                        nstacks_per_job,
721                        (**nstacks_per_job as f32 / self.nstacks as f32) * 100.0,
722                        **duration as f64 / default_duration as f64 * 100.0,
723                    )?;
724                }
725                writeln!(f)?;
726                Ok(())
727            }
728        }
729
730        fn count_lines_and_stacks(bytes: &[u8]) -> (usize, usize) {
731            let mut reader = io::BufReader::new(bytes);
732            let mut line = Vec::new();
733
734            let (mut nlines, mut nstacks) = (0, 0);
735            loop {
736                line.clear();
737                let n = reader.read_until(0x0A, &mut line).unwrap();
738                if n == 0 {
739                    nstacks += 1;
740                    break;
741                }
742                let l = String::from_utf8_lossy(&line);
743                nlines += 1;
744                if l.trim().is_empty() {
745                    nstacks += 1;
746                }
747            }
748            (nlines, nstacks)
749        }
750
751        let inputs = read_inputs(inputs)?;
752
753        // Warmup
754        let now = Instant::now();
755        stdout.write_fmt(format_args!(
756            "# Warming up for approximately {} seconds.\n",
757            WARMUP_SECS
758        ))?;
759        stdout.flush()?;
760        while now.elapsed() < std::time::Duration::from_secs(WARMUP_SECS as u64) {
761            for (_, bytes) in inputs.iter() {
762                folder.collapse(&bytes[..], io::sink())?;
763            }
764        }
765
766        // Time
767        let mut foos = Vec::new();
768        for (path, bytes) in &inputs {
769            stdout.write_fmt(format_args!("# {} ", path.display()))?;
770            stdout.flush()?;
771            if let Some(foo) = Foo::new(folder, path, bytes, &mut stdout)? {
772                foos.push(foo);
773            }
774            stdout.write_all(&[b'\n'])?;
775            stdout.flush()?;
776        }
777        stdout.write_all(&[b'\n'])?;
778        stdout.flush()?;
779        foos.sort_by(|a, b| b.nstacks.cmp(&a.nstacks));
780        for foo in foos {
781            stdout.write_fmt(format_args!("{}", foo))?;
782            stdout.flush()?;
783        }
784
785        Ok(())
786    }
