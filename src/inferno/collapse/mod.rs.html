88/// The abstract behavior of stack collapsing.
89///
90/// Implementors of this trait are providing a way to take the stack traces produced by a
91/// particular profiler's output (like `perf script`) and produce lines in the folded stack format
92/// expected by [`crate::flamegraph::from_lines`].
93///
94/// See also the [crate-level documentation] for details.
95///
96///   [crate-level documentation]: ../index.html
97// https://github.com/rust-lang/rust/issues/45040
98// #[doc(spotlight)]
99pub trait Collapse {
100    /// Collapses the contents of the provided `reader` and writes folded stack lines to the
101    /// provided `writer`.
102    fn collapse<R, W>(&mut self, reader: R, writer: W) -> io::Result<()>
103    where
104        R: io::BufRead,
105        W: io::Write;
106
107    /// Collapses the contents of the provided file (or of STDIN if `infile` is `None`) and
108    /// writes folded stack lines to provided `writer`.
109    fn collapse_file<P, W>(&mut self, infile: Option<P>, writer: W) -> io::Result<()>
110    where
111        P: AsRef<Path>,
112        W: io::Write,
113    {
114        match infile {
115            Some(ref path) => {
116                let file = File::open(path)?;
117                let reader = io::BufReader::with_capacity(CAPACITY_READER, file);
118                self.collapse(reader, writer)
119            }
120            None => {
121                let stdin = io::stdin();
122                let stdin_guard = stdin.lock();
123                let reader = io::BufReader::with_capacity(CAPACITY_READER, stdin_guard);
124                self.collapse(reader, writer)
125            }
126        }
127    }
128
129    /// Collapses the contents of the provided file (or of STDIN if `infile` is `None`) and
130    /// writes folded stack lines to STDOUT.
131    fn collapse_file_to_stdout<P>(&mut self, infile: Option<P>) -> io::Result<()>
132    where
133        P: AsRef<Path>,
134    {
135        if std::io::stdout().is_terminal() {
136            self.collapse_file(infile, io::stdout().lock())
137        } else {
138            self.collapse_file(infile, io::BufWriter::new(io::stdout().lock()))
139        }
140    }
141
142    /// Returns whether this implementation is appropriate for the given input.
143    ///
144    /// - `None` means "not sure -- need more input"
145    /// - `Some(true)` means "yes, this implementation should work with this string"
146    /// - `Some(false)` means "no, this implementation definitely won't work"
147    #[allow(clippy::wrong_self_convention)]
148    fn is_applicable(&mut self, input: &str) -> Option<bool>;
149}
150
151impl<T> Collapse for T
152where
153    T: CollapsePrivate,
154{
155    fn collapse<R, W>(&mut self, reader: R, writer: W) -> io::Result<()>
156    where
157        R: io::BufRead,
158        W: io::Write,
159    {
160        <Self as CollapsePrivate>::collapse(self, reader, writer)
161    }
162
163    fn is_applicable(&mut self, input: &str) -> Option<bool> {
164        <Self as CollapsePrivate>::is_applicable(self, input)
165    }
166}
