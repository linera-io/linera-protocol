8// The set of symbols to ignore for 'waiting' threads, for ease of use.
9// This will hide waiting threads from the view, making it easier to
10// see what is actually running in the sample.
11static IGNORE_SYMBOLS: &[&str] = &[

24// The call graph begins after this line.

41/// A stack collapser for the output of `sample` on macOS.
42///
43/// To construct one, either use `sample::Folder::default()` or create an [`Options`] and use
44/// `sample::Folder::from(options)`.
45#[derive(Clone, Default)]
46pub struct Folder {
47    /// Number of samples for the current stack frame.
48    current_samples: usize,
49
50    /// Function on the stack in this entry thus far.
51    stack: Vec<String>,
52
53    opt: Options,
54}

56impl Collapse for Folder {
57    fn collapse<R, W>(&mut self, mut reader: R, writer: W) -> io::Result<()>
58    where
59        R: io::BufRead,
60        W: io::Write,
61    {
62        // Consume the header...
63        let mut line = Vec::new();
64        loop {
65            line.clear();
66            if reader.read_until(0x0A, &mut line)? == 0 {
67                warn!("File ended before start of call graph");
68                return Ok(());
69            };
70            let l = String::from_utf8_lossy(&line);
71            if l.starts_with(START_LINE) {
72                break;
73            }
74        }
75
76        // Process the data...
77        let mut occurrences = Occurrences::new(1);
78        loop {
79            line.clear();
80            if reader.read_until(0x0A, &mut line)? == 0 {
81                return invalid_data_error!("File ended before end of call graph");
82            }
83            let l = String::from_utf8_lossy(&line);
84            let line = l.trim_end();
85            if line.is_empty() {
86                continue;
87            } else if line.starts_with("    ") {
88                self.on_line(line, &mut occurrences)?;
89            } else if line.starts_with(END_LINE) {
90                self.write_stack(&mut occurrences);
91                break;
92            } else {
93                return invalid_data_error!("Stack line doesn't start with 4 spaces:\n{}", line);
94            }
95        }
96
97        // Write the results...
98        occurrences.write_and_clear(writer)?;
99
100        // Reset the state...
101        self.current_samples = 0;
102        self.stack.clear();
103        Ok(())
104    }
105
106    /// Check for start and end lines of a call graph.
107    fn is_applicable(&mut self, input: &str) -> Option<bool> {
108        let mut found_start = false;
109        let mut input = input.as_bytes();
110        let mut line = String::new();
111        loop {
112            line.clear();
113            if let Ok(n) = input.read_line(&mut line) {
114                if n == 0 {
115                    break;
116                }
117            } else {
118                return Some(false);
119            }
120
121            if line.starts_with(START_LINE) {
122                found_start = true;
123                continue;
124            } else if line.starts_with(END_LINE) {
125                return Some(found_start);
126            }
127        }
128        None
129    }
130}

141impl Folder {
142    fn line_parts<'a>(&self, line: &'a str) -> Option<(&'a str, &'a str, &'a str)> {
143        let mut line = line.trim_start().splitn(2, ' ');
144        let time = line.next()?.trim_end();
145        let line = line.next()?;
146
147        let func = match line.find('(') {
148            Some(open) => &line[..open],
149            None => line,
150        }
151        .trim_end();
152
153        let mut module = "";
154        if !self.opt.no_modules {
155            // Modules are shown with "(in libfoo.dylib)" or "(in AppKit)".
156            let mut line = line.rsplitn(2, "(in ");
157            if let Some(line) = line.next() {
158                if let Some(close) = line.find(')') {
159                    module = &line[..close];
160                }
161
162                // Remove ".dylib", since it adds no value.
163                if module.ends_with(".dylib") {
164                    module = &module[..module.len() - 6]
165                }
166            }
167        }
168
169        Some((time, func, module))
170    }
171
172    fn is_indent_char(c: char) -> bool {
173        c == ' ' || c == '+' || c == '|' || c == ':' || c == '!'
174    }
175
176    // Handle call graph lines of the form:
177    //
178    // 5130 Thread_8749954
179    //    + 5130 start_wqthread  (in libsystem_pthread.dylib) ...
180    //    +   4282 _pthread_wqthread  (in libsystem_pthread.dylib) ...
181    //    +   ! 4282 __doworkq_kernreturn  (in libsystem_kernel.dylib) ...
182    //    +   848 _pthread_wqthread  (in libsystem_pthread.dylib) ...
183    //    +     848 __doworkq_kernreturn  (in libsystem_kernel.dylib) ...
184    fn on_line(&mut self, line: &str, occurrences: &mut Occurrences) -> io::Result<()> {
185        if let Some(indent_chars) = line[4..].find(|c| !Self::is_indent_char(c)) {
186            // Each indent is two characters
187            if indent_chars % 2 != 0 {
188                return invalid_data_error!(
189                    "Odd number of indentation characters for line:\n{}",
190                    line
191                );
192            }
193
194            let prev_depth = self.stack.len();
195            let depth = indent_chars / 2 + 1;
196
197            if depth <= prev_depth {
198                // Each sampled function will be a leaf node in the call tree.
199                // If the depth of this line is less than the previous one,
200                // it means the previous line was a leaf node and we should
201                // write out the stack and pop it back to one before the current depth.
202                self.write_stack(occurrences);
203                for _ in 0..=prev_depth - depth {
204                    self.stack.pop();
205                }
206            } else if depth > prev_depth + 1 {
207                return invalid_data_error!("Skipped indentation level at line:\n{}", line);
208            }
209
210            if let Some((samples, func, module)) = self.line_parts(&line[4 + indent_chars..]) {
211                if let Ok(samples) = samples.parse::<usize>() {
212                    // The sample counts of the direct children of a non-leaf entry will always
213                    // add up to that node's sample count so we only need to keep track of the
214                    // sample count at the top of the stack.
215                    self.current_samples = samples;
216                    // sample doesn't properly demangle Rust symbols, so fix those.
217                    let func = common::fix_partially_demangled_rust_symbol(func);
218                    if module.is_empty() {
219                        self.stack.push(func.to_string());
220                    } else {
221                        self.stack.push(format!("{}`{}", module, func));
222                    }
223                } else {
224                    return invalid_data_error!("Invalid samples field: {}", samples);
225                }
226            } else {
227                return invalid_data_error!("Unable to parse stack line:\n{}", line);
228            }
229        } else {
230            return invalid_data_error!("Found stack line with only indent characters:\n{}", line);
231        }
232
233        Ok(())
234    }
235
236    fn write_stack(&self, occurrences: &mut Occurrences) {
237        if let Some(func) = self.stack.last() {
238            for symbol in IGNORE_SYMBOLS {
239                if func.ends_with(symbol) {
240                    // Don't write out stacks with ignored symbols
241                    return;
242                }
243            }
244        }
245        occurrences.insert(self.stack.join(";"), self.current_samples);
246    }
