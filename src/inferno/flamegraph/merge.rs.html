27fn flow<'a, LI, TI>(
28    tmp: &mut HashMap<Frame<'a>, FrameTime>,
29    frames: &mut Vec<TimedFrame<'a>>,
30    last: LI,
31    this: TI,
32    time: usize,
33    delta: Option<isize>,
34) where
35    LI: IntoIterator<Item = &'a str>,
36    TI: IntoIterator<Item = &'a str>,
37{
38    let mut this = this.into_iter().peekable();
39    let mut last = last.into_iter().peekable();
40
41    // remove common prefix
42    let mut shared_depth = 0;
43    while last.peek() == this.peek() {
44        // they must both be None, so let's stop looping
45        if last.peek().is_none() {
46            break;
47        }
48
49        // move along prefix iterators
50        last.next();
51        this.next();
52        shared_depth += 1;
53    }
54
55    // TODO: document this..
56
57    for (i, func) in last.enumerate() {
58        let key = Frame {
59            function: func,
60            depth: shared_depth + i,
61        };
62
63        //eprintln!("at {} ending frame {:?}", time, key);
64        let frame_time = tmp.remove(&key).unwrap_or_else(|| {
65            unreachable!("did not have start time for {:?}", key);
66        });
67
68        let frame = TimedFrame {
69            location: key,
70            start_time: frame_time.start_time,
71            end_time: time,
72            delta: frame_time.delta,
73        };
74        frames.push(frame);
75    }
76
77    let mut i = 0;
78    while this.peek().is_some() {
79        let func = this.next().unwrap();
80        let key = Frame {
81            function: func,
82            depth: shared_depth + i,
83        };
84
85        let is_last = this.peek().is_none();
86        let delta = match delta {
87            Some(_) if !is_last => Some(0),
88            d => d,
89        };
90        let frame_time = FrameTime {
91            start_time: time,
92            // For some reason the Perl version does a `+=` for `delta`, but I can't figure out why.
93            // See https://github.com/brendangregg/FlameGraph/blob/1b1c6deede9c33c5134c920bdb7a44cc5528e9a7/flamegraph.pl#L588
94            delta,
95        };
96
97        //eprintln!("stored tmp for time {}: {:?}", time, key);
98        if let Some(frame_time) = tmp.insert(key, frame_time) {
99            unreachable!(
100                "start time {} already registered for frame",
101                frame_time.start_time
102            );
103        }
104
105        i += 1;
106    }
107}

109pub(super) fn frames<'a, I>(
110    lines: I,
111    suppress_sort_check: bool,
112) -> io::Result<(Vec<TimedFrame<'a>>, usize, usize, usize)>
113where
114    I: IntoIterator<Item = &'a str>,
115{
116    let mut time = 0;
117    let mut ignored = 0;
118    let mut last = "";
119    let mut tmp = Default::default();
120    let mut frames = Default::default();
121    let mut delta = None;
122    let mut delta_max = 1;
123    let mut stripped_fractional_samples = false;
124    let mut prev_line = None;
125    for line in lines {
126        let mut line = line.trim();
127
128        if !suppress_sort_check {
129            if let Some(prev_line) = prev_line {
130                if prev_line > line {
131                    return Err(io::Error::new(
132                        io::ErrorKind::InvalidData,
133                        "unsorted input lines detected",
134                    ));
135                }
136            }
137        }
138
139        // Parse the number of samples for the purpose of computing overall time passed.
140        // Usually there will only be one samples column at the end of a line,
141        // but for differentials there will be two. When there are two we compute the
142        // delta between them and use the second one.
143        let nsamples =
144            if let Some(samples) = parse_nsamples(&mut line, &mut stripped_fractional_samples) {
145                // See if there's also a differential column present
146                if let Some(original_samples) =
147                    parse_nsamples(&mut line, &mut stripped_fractional_samples)
148                {
149                    delta = Some(samples as isize - original_samples as isize);
150                    delta_max = std::cmp::max(delta.unwrap().unsigned_abs(), delta_max);
151                }
152                samples
153            } else {
154                ignored += 1;
155                continue;
156            };
157
158        if line.is_empty() {
159            ignored += 1;
160            continue;
161        }
162        let stack = line;
163
164        // inject empty first-level stack frame to capture "all"
165        let this = iter::once("").chain(stack.split(';'));
166        if last.is_empty() {
167            // need to special-case this, because otherwise iter("") + "".split(';') == ["", ""]
168            //eprintln!("flow(_, {}, {})", stack, time);
169            flow(&mut tmp, &mut frames, None, this, time, delta);
170        } else {
171            //eprintln!("flow({}, {}, {})", last, stack, time);
172            flow(
173                &mut tmp,
174                &mut frames,
175                iter::once("").chain(last.split(';')),
176                this,
177                time,
178                delta,
179            );
180        }
181
182        last = stack;
183        time += nsamples;
184        prev_line = Some(line);
185    }
186
187    if !last.is_empty() {
188        //eprintln!("flow({}, _, {})", last, time);
189        flow(
190            &mut tmp,
191            &mut frames,
192            iter::once("").chain(last.split(';')),
193            None,
194            time,
195            delta,
196        );
197    }
198
199    Ok((frames, time, ignored, delta_max))
200}

202// Parse and remove the number of samples from the end of a line.
203fn parse_nsamples(line: &mut &str, stripped_fractional_samples: &mut bool) -> Option<usize> {
204    if let Some((samplesi, doti)) = rfind_samples(line) {
205        let mut samples = &line[samplesi..];
206        // Strip fractional part (if any);
207        // foobar 1.klwdjlakdj
208        //
209        // The Perl version keeps the fractional part but this can be problematic
210        // because of cumulative floating point errors. Instead we recommend to
211        // use the --factor option. See https://github.com/brendangregg/FlameGraph/pull/18
212        //
213        // Warn if we're stripping a non-zero fractional part, but only the first time.
214        if !*stripped_fractional_samples
215            && doti < samples.len() - 1
216            && !samples[doti + 1..].chars().all(|c| c == '0')
217        {
218            *stripped_fractional_samples = true;
219            warn!(
220                "The input data has fractional sample counts that will be truncated to integers. \
221                 If you need to retain the extra precision you can scale up the sample data and \
222                 use the --factor option to scale it back down."
223            );
224        }
225        samples = &samples[..doti];
226        let nsamples = samples.parse::<usize>().ok()?;
227        // remove nsamples part we just parsed from line
228        *line = line[..samplesi].trim_end();
229        Some(nsamples)
230    } else {
231        None
232    }
233}

235// Tries to find a sample count at the end of a line.
236//
237// On success, the first value of the returned tuple will be the index to the sample count.
238// If the sample count is fractional, the second value will be the offset of the dot within
239// the sample count.
240// If the sample count is not fractional, the second value returned is the offset
241// to the last digit in the sample count.
242//
243// If no sample count is found, `None` will be returned.
244pub(super) fn rfind_samples(line: &str) -> Option<(usize, usize)> {
245    let samplesi = line.rfind(' ')? + 1;
246    let samples = &line[samplesi..];
247    if let Some(doti) = samples.find('.') {
248        if samples[..doti]
249            .chars()
250            .chain(samples[doti + 1..].chars())
251            .all(|c| c.is_ascii_digit())
252        {
253            Some((samplesi, doti))
254        } else {
255            None
256        }
257    } else if !samples.chars().all(|c| c.is_ascii_digit()) {
258        None
259    } else {
260        Some((samplesi, line.len() - samplesi))
261    }
