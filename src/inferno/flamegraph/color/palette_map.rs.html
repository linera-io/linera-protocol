11/// Mapping of the association between a function name and the color used when drawing information
12/// from this function.
13#[derive(Clone, Debug, Default, Eq, PartialEq)]
14pub struct PaletteMap(HashMap<String, Color>);
15
16impl PaletteMap {
17    /// Returns the color value corresponding to the given function name.
18    pub fn get(&self, func: &str) -> Option<Color> {
19        self.0.get(func).cloned()
20    }
21
22    /// Inserts a function name/color pair in the map.
23    pub fn insert<S: ToString>(&mut self, func: S, color: Color) -> Option<Color> {
24        self.0.insert(func.to_string(), color)
25    }
26
27    /// Provides an iterator over the elements of the map.
28    pub fn iter(&self) -> impl Iterator<Item = (&str, Color)> {
29        self.0.iter().map(|(func, color)| (func.as_str(), *color))
30    }
31
32    /// Builds a mapping based on the inputs given by the reader.
33    ///
34    /// The reader should provide name/color pairs as text input, each pair separated by a line
35    /// separator.
36    ///
37    /// Each line should follow the format: NAME->rgb(RED, GREEN, BLUE)
38    /// where NAME is the function name, and RED, GREEN, BLUE integer values between 0 and 255
39    /// included.
40    /// Any line which does not follow the previous format will be ignored.
41    ///
42    /// This function will propagate any [`std::io::Error`] returned by the given reader.
43    pub fn from_reader(reader: &mut dyn io::BufRead) -> io::Result<Self> {
44        let mut map = HashMap::default();
45        let mut ignored = 0;
46
47        for line in reader.lines() {
48            let line = line?;
49            if let Ok((name, color)) = parse_line(&line) {
50                map.insert(name.to_string(), color);
51            } else {
52                ignored += 1;
53            }
54        }
55
56        if ignored != 0 {
57            warn!("Ignored {} lines with invalid format", ignored);
58        }
59
60        Ok(PaletteMap(map))
61    }
62
63    /// Writes the palette map using the given writer.
64    ///
65    /// The output content will follow the same format described in
66    /// [`from_reader`](Self::from_reader).
67    ///
68    /// The name/color pairs will be sorted by name in lexicographic order.
69    pub fn to_writer(&self, writer: &mut dyn io::Write) -> io::Result<()> {
70        let mut entries = self.0.iter().collect::<Vec<_>>();
71        // We sort the palette because the Perl implementation does.
72        entries.sort_unstable();
73
74        for (name, color) in entries {
75            writer.write_all(
76                format!("{}->rgb({},{},{})\n", name, color.r, color.g, color.b).as_bytes(),
77            )?
78        }
79
80        Ok(())
81    }
82
83    /// Utility function to load a palette map from a file.
84    ///
85    /// The file content should follow the format described in [`from_reader`](Self::from_reader).
86    ///
87    /// If the file does not exist, an empty palette map is returned.
88    pub fn load_from_file_or_empty(path: &dyn AsRef<Path>) -> io::Result<Self> {
89        // If the file does not exist, it is probably the first call to flamegraph with a consistent
90        // palette: there is nothing to load.
91        if path.as_ref().exists() {
92            let file = File::open(path)?;
93            let mut reader = BufReader::new(file);
94            PaletteMap::from_reader(&mut reader)
95        } else {
96            Ok(PaletteMap::default())
97        }
98    }
99
100    /// Utility function to save a palette map to a file.
101    ///
102    /// The file content will follow the format described in [`from_reader`](Self::from_reader).
103    pub fn save_to_file(&self, path: &dyn AsRef<Path>) -> io::Result<()> {
104        let mut file = OpenOptions::new()
105            .write(true)
106            .create(true)
107            .truncate(true)
108            .open(path)?;
109        self.to_writer(&mut file)
110    }
111
112    /// Returns the color value corresponding to the given function name if it is present.
113    /// Otherwise compute the color, and insert the new function name/color in the map.
114    pub(crate) fn find_color_for<F: FnMut(&str) -> Color>(
115        &mut self,
116        name: &str,
117        mut compute_color: F,
118    ) -> Color {
119        match self.get(name) {
120            Some(color) => color,
121            None => {
122                let color = compute_color(name);
123                self.insert(name, color);
124                color
125            }
126        }
127    }
128}

130fn parse_line(line: &str) -> io::Result<(&str, Color)> {
131    // A line is formatted like this: NAME -> rbg(RED, GREEN, BLUE)
132    let mut words = line.split("->");
133
134    let name = match words.next() {
135        Some(name) => name,
136        None => return Err(io::Error::from(io::ErrorKind::InvalidInput)),
137    };
138
139    let color = match words.next() {
140        Some(name) => name,
141        None => return Err(io::Error::from(io::ErrorKind::InvalidInput)),
142    };
143
144    if words.next().is_some() {
145        return Err(io::Error::from(io::ErrorKind::InvalidInput));
146    }
147
148    let rgb_color =
149        parse_rgb_string(color).ok_or_else(|| io::Error::from(io::ErrorKind::InvalidData))?;
150
151    Ok((name, rgb_color))
152}
153
154fn parse_rgb_string(s: &str) -> Option<Color> {
155    let s = s.trim();
156
157    if !s.starts_with("rgb(") || !s.ends_with(')') {
158        return None;
159    }
160
161    let s = &s["rgb(".len()..s.len() - 1];
162    let r_end_index = s.find(',')?;
163    let r_str = s[..r_end_index].trim();
164    let r = u8::from_str(r_str).ok()?;
165
166    let s = &s[r_end_index + 1..];
167    let g_end_index = s.find(',')?;
168    let g_str = s[..g_end_index].trim();
169    let g = u8::from_str(g_str).ok()?;
170
171    let b_str = &s[g_end_index + 1..].trim();
172    let b = u8::from_str(b_str).ok()?;
173
174    Some(Color { r, g, b })
175}

194    #[test]
195    fn palette_map_test() {
196        let mut palette = PaletteMap::default();
197
198        assert_eq!(palette.insert("foo", color!(0, 50, 255)), None);
199        assert_eq!(palette.insert("bar", color!(50, 0, 60)), None);
200        assert_eq!(
201            palette.insert("foo", color!(80, 20, 63)),
202            Some(color!(0, 50, 255))
203        );
204        assert_eq!(
205            palette.insert("foo", color!(128, 128, 128)),
206            Some(color!(80, 20, 63))
207        );
208        assert_eq!(palette.insert("baz", color!(255, 0, 255)), None);
209
210        assert_eq!(palette.get("func"), None);
211        assert_eq!(palette.get("bar"), Some(color!(50, 0, 60)));
212        assert_eq!(palette.get("foo"), Some(color!(128, 128, 128)));
213        assert_eq!(palette.get("baz"), Some(color!(255, 0, 255)));
214
215        let mut vec = palette.iter().collect::<Vec<_>>();
216        vec.sort_unstable();
217        let mut iter = vec.iter();
218
219        assert_eq!(iter.next(), Some(&("bar", color!(50, 0, 60))));
220        assert_eq!(iter.next(), Some(&("baz", color!(255, 0, 255))));
221        assert_eq!(iter.next(), Some(&("foo", color!(128, 128, 128))));
222        assert_eq!(iter.next(), None);
223
224        let mut buf = Cursor::new(Vec::new());
225
226        palette.to_writer(&mut buf).unwrap();
227        buf.set_position(0);
228        let palette = PaletteMap::from_reader(&mut buf).unwrap();
229
230        let mut vec = palette.iter().collect::<Vec<_>>();
231        vec.sort_unstable();
232        let mut iter = vec.iter();
233
234        assert_eq!(iter.next(), Some(&("bar", color!(50, 0, 60))));
235        assert_eq!(iter.next(), Some(&("baz", color!(255, 0, 255))));
236        assert_eq!(iter.next(), Some(&("foo", color!(128, 128, 128))));
237        assert_eq!(iter.next(), None);
238    }

240    #[test]
241    fn parse_line_test() {
242        assert_eq!(
243            parse_line("func->rgb(0, 0, 0)").unwrap(),
244            ("func", color!(0, 0, 0))
245        );
246        assert_eq!(
247            parse_line("->rgb(255, 255, 255)").unwrap(),
248            ("", color!(255, 255, 255))
249        );
250
251        assert!(parse_line("").is_err());
252        assert!(parse_line("func->(0, 0, 0)").is_err());
253        assert!(parse_line("func->").is_err());
254        assert!(parse_line("func->foo->rgb(0, 0, 0)").is_err());
255        assert!(parse_line("func->rgb(0, 0, 0)->foo").is_err());
256        assert!(parse_line("func->rgb(255, 255, 256)").is_err());
257        assert!(parse_line("func->rgb(-1, 255, 255)").is_err());
258    }
