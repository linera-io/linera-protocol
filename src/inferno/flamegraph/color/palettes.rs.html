1pub(super) mod java {
2    use crate::flamegraph::color::BasicPalette;
3
4    /// Handle both annotations (_[j], _[i], ...; which are
5    /// accurate), as well as input that lacks any annotations, as
6    /// best as possible. Without annotations, we get a little hacky
7    /// and match on java|org|com, etc.
8    pub(in super::super) fn resolve(name: &str) -> BasicPalette {
9        if name.ends_with(']') {
10            if let Some(ai) = name.rfind("_[") {
11                if name[ai..].len() == 4 {
12                    match &name[ai + 2..ai + 3] {
13                        // kernel annotation
14                        "k" => return BasicPalette::Orange,
15                        // inline annotation
16                        "i" => return BasicPalette::Aqua,
17                        // jit annotation
18                        "j" => return BasicPalette::Green,
19                        _ => {}
20                    }
21                }
22            }
23        }
24
25        let java_prefix = name.strip_prefix('L').unwrap_or(name);
26
27        if name.contains("::") || name.starts_with("-[") || name.starts_with("+[") {
28            // C++ or Objective C
29            BasicPalette::Yellow
30        } else if java_prefix.contains('/')
31            || (java_prefix.contains('.') && !java_prefix.starts_with('['))
32            || match java_prefix.chars().next() {
33                Some(c) => c.is_ascii_uppercase(),
34                _ => false,
35            }
36        {
37            // Java
38            BasicPalette::Green
39        } else {
40            // system
41            BasicPalette::Red
42        }
43    }
44}

62pub(super) mod python {
63    use crate::flamegraph::color::BasicPalette;
64
65    fn split_any_path(path: &str) -> impl Iterator<Item = &str> {
66        path.split(['/', '\\'])
67    }
68
69    pub(in super::super) fn resolve(name: &str) -> BasicPalette {
70        if split_any_path(name).any(|part| part == "site-packages") {
71            BasicPalette::Aqua
72        } else if split_any_path(name).any(|part| {
73            part.strip_prefix("python")
74                .or_else(|| part.strip_prefix("Python"))
75                .is_some_and(|version| version.chars().all(|c| c.is_ascii_digit() || c == '.'))
76        }) || name.starts_with("<built-in")
77            || name.starts_with("<method")
78            || name.starts_with("<frozen")
79        {
80            // stdlib
81            BasicPalette::Yellow
82        } else {
83            BasicPalette::Red
84        }
85    }
86}

88pub(super) mod js {
89    use crate::flamegraph::color::BasicPalette;
90
91    pub(in super::super) fn resolve(name: &str) -> BasicPalette {
92        if !name.is_empty() && name.trim().is_empty() {
93            return BasicPalette::Green;
94        } else if name.ends_with("_[k]") {
95            return BasicPalette::Orange;
96        } else if name.ends_with("_[j]") {
97            if name.contains('/') {
98                return BasicPalette::Green;
99            } else {
100                return BasicPalette::Aqua;
101            }
102        } else if name.contains("::") {
103            return BasicPalette::Yellow;
104        } else if name.contains(':') {
105            return BasicPalette::Aqua;
106        } else if let Some(ai) = name.find('/') {
107            if name[ai..].contains("node_modules/") {
108                return BasicPalette::Purple;
109            } else if name[ai..].contains(".js") {
110                return BasicPalette::Green;
111            }
112        }
113
114        BasicPalette::Red
115    }
116}

126pub(super) mod rust {
127    use crate::flamegraph::color::BasicPalette;
128
129    pub(in super::super) fn resolve(name: &str) -> BasicPalette {
130        let name = name.split_once('`').map(|(_, after)| after).unwrap_or(name);
131        if name.starts_with("core::")
132            || name.starts_with("std::")
133            || name.starts_with("alloc::")
134            || (name.starts_with("<core::")
135                // Rust user-defined async functions are desugared into
136                // GenFutures so we don't want to include those as Rust
137                // system functions
138                && !name.starts_with("<core::future::from_generator::GenFuture<T>"))
139            || name.starts_with("<std::")
140            || name.starts_with("<alloc::")
141        {
142            // Rust system functions
143            BasicPalette::Orange
144        } else if name.contains("::") {
145            // Rust user functions.
146            // Although this will generate false positives for e.g. C++ code
147            // used with Rust, the intention is to color code from user
148            // crates and dependencies differently than Rust system code.
149            BasicPalette::Aqua
150        } else {
151            // Non-Rust functions
152            BasicPalette::Yellow
153        }
154    }
155}
