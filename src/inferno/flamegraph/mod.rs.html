266impl Options<'_> {
267    /// Calculate pad top, including title and subtitle
268    pub(super) fn ypad1(&self) -> usize {
269        let subtitle_height = if self.subtitle.is_some() {
270            self.font_size * 2
271        } else {
272            0
273        };
274        if self.direction == Direction::Straight {
275            self.font_size * 3 + subtitle_height
276        } else {
277            // Inverted (icicle) mode, put the details on top. The +4 is to add
278            // a little bit more space between the title (or subtitle if there
279            // is one) and the details.
280            self.font_size * 4 + subtitle_height + 4
281        }
282    }
283
284    /// Calculate pad bottom, including labels
285    pub(super) fn ypad2(&self) -> usize {
286        if self.direction == Direction::Straight {
287            self.font_size * 2 + 10
288        } else {
289            // Inverted (icicle) mode, put the details on top, so don't need
290            // room at the bottom.
291            self.font_size + 10
292        }
293    }
294}

380/// Produce a flame graph from an iterator over folded stack lines.
381///
382/// This function expects each folded stack to contain the following whitespace-separated fields:
383///
384///  - A semicolon-separated list of frame names (e.g., `main;foo;bar;baz`).
385///  - A sample count for the given stack.
386///  - An optional second sample count.
387///
388/// If two sample counts are provided, a [differential flame graph] is produced. In this mode, the
389/// flame graph uses the difference between the two sample counts to show how the sample counts for
390/// each stack has changed between the first and second profiling.
391///
392/// The resulting flame graph will be written out to `writer` in SVG format.
393///
394/// [differential flame graph]: http://www.brendangregg.com/blog/2014-11-09/differential-flame-graphs.html
395#[allow(clippy::cognitive_complexity)]
396pub fn from_lines<'a, I, W>(opt: &mut Options<'_>, lines: I, writer: W) -> io::Result<()>
397where
398    I: IntoIterator<Item = &'a str>,
399    W: Write,
400{
401    let mut reversed = StrStack::new();
402    let lines = lines
403        .into_iter()
404        .map(|line| line.trim())
405        .filter(|line| !(line.is_empty() || line.starts_with("# ")));
406
407    let (mut frames, time, ignored, delta_max) = if opt.reverse_stack_order {
408        if opt.no_sort {
409            warn!(
410                "Input lines are always sorted when `reverse_stack_order` is `true`. \
411                 The `no_sort` option is being ignored."
412            );
413        }
414        // Reverse order of stacks and sort.
415        let mut stack = String::new();
416        for line in lines {
417            stack.clear();
418            let samples_idx = merge::rfind_samples(line)
419                .map(|(i, _)| i)
420                .unwrap_or_else(|| line.len());
421            let samples_idx = merge::rfind_samples(&line[..samples_idx - 1])
422                .map(|(i, _)| i)
423                .unwrap_or(samples_idx);
424            for (i, func) in line[..samples_idx].trim().split(';').rev().enumerate() {
425                if i != 0 {
426                    stack.push(';');
427                }
428                stack.push_str(func);
429            }
430            stack.push(' ');
431            stack.push_str(&line[samples_idx..]);
432            // Trim to handle the case where functions names internally contain `;`.
433            // This can happen, for example, with types like `[u8; 8]` in Rust.
434            // See https://github.com/jonhoo/inferno/pull/338.
435            let stack = stack.trim();
436            reversed.push(stack);
437        }
438        let mut reversed: Vec<&str> = reversed.iter().collect();
439        reversed.sort_unstable();
440        merge::frames(reversed, false)?
441    } else if opt.flame_chart {
442        // In flame chart mode, just reverse the data so time moves from left to right.
443        let mut lines: Vec<&str> = lines.into_iter().collect();
444        lines.reverse();
445        merge::frames(lines, true)?
446    } else if opt.no_sort {
447        // Lines don't need sorting.
448        merge::frames(lines, false)?
449    } else {
450        // Sort lines by default.
451        let mut lines: Vec<&str> = if opt.base.is_empty() {
452            lines.into_iter().collect()
453        } else {
454            lines
455                .into_iter()
456                .filter_map(|line| {
457                    let mut cursor = line.len();
458                    for symbol in line.rsplit(';') {
459                        cursor -= symbol.len();
460                        if opt.base.iter().any(|b| b == symbol) {
461                            break;
462                        }
463                        cursor = cursor.saturating_sub(1);
464                    }
465                    if cursor == 0 {
466                        None
467                    } else {
468                        Some(&line[cursor..])
469                    }
470                })
471                .collect()
472        };
473        lines.sort_unstable();
474        merge::frames(lines, false)?
475    };
476
477    if ignored != 0 {
478        warn!("Ignored {} lines with invalid format", ignored);
479    }
480
481    let mut buffer = StrStack::new();
482
483    // let's start writing the svg!
484    let mut svg = if opt.pretty_xml {
485        Writer::new_with_indent(writer, b' ', 4)
486    } else {
487        Writer::new(writer)
488    };
489
490    if time == 0 {
491        error!("No stack counts found");
492        // emit an error message SVG, for tools automating flamegraph use
493        let imageheight = opt.font_size * 5;
494        svg::write_header(&mut svg, imageheight, opt)?;
495        svg::write_str(
496            &mut svg,
497            &mut buffer,
498            svg::TextItem {
499                x: Dimension::Percent(50.0),
500                y: (opt.font_size * 2) as f64,
501                text: "ERROR: No valid input provided to flamegraph".into(),
502                extra: None,
503            },
504        )?;
505        svg.write_event(Event::End(BytesEnd::new("svg")))?;
506        svg.write_event(Event::Eof)?;
507        return Err(io::Error::new(
508            io::ErrorKind::InvalidData,
509            "No stack counts found",
510        ));
511    }
512
513    let image_width = opt.image_width.unwrap_or(DEFAULT_IMAGE_WIDTH) as f64;
514    let timemax = time;
515    let widthpertime_pct = 100.0 / timemax as f64;
516    let minwidth_time = opt.min_width / widthpertime_pct;
517
518    // prune blocks that are too narrow
519    let mut depthmax = 0;
520    frames.retain(|frame| {
521        if ((frame.end_time - frame.start_time) as f64) < minwidth_time {
522            false
523        } else {
524            depthmax = std::cmp::max(depthmax, frame.location.depth);
525            true
526        }
527    });
528
529    // draw canvas, and embed interactive JavaScript program
530    let imageheight = ((depthmax + 1) * opt.frame_height) + opt.ypad1() + opt.ypad2();
531    svg::write_header(&mut svg, imageheight, opt)?;
532
533    let (bgcolor1, bgcolor2) = color::bgcolor_for(opt.bgcolors, opt.colors);
534    let strokecolor = match opt.stroke_color {
535        StrokeColor::Color(c) => Some(c.to_string()),
536        StrokeColor::None => None,
537    };
538    let uicolor = opt.uicolor.to_string();
539    let style_options = StyleOptions {
540        imageheight,
541        bgcolor1,
542        bgcolor2,
543        uicolor,
544        strokecolor,
545    };
546
547    svg::write_prelude(&mut svg, &style_options, opt)?;
548
549    // Used when picking color parameters at random, when no option determines how to pick these
550    // parameters. We instantiate it here because it may be called once for each iteration in the
551    // frames loop.
552    let mut thread_rng = rand::thread_rng();
553
554    // structs to reuse across loops to avoid allocations
555    let mut cache_g = Event::Start(BytesStart::new("g"));
556    let mut cache_a = Event::Start(BytesStart::new("a"));
557    let mut cache_rect = Event::Empty(BytesStart::new("rect"));
558    let cache_g_end = Event::End(BytesEnd::new("g"));
559    let cache_a_end = Event::End(BytesEnd::new("a"));
560
561    // create frames container
562    let container_x = format!("{}", XPAD);
563    let container_width = format!("{}", image_width as usize - XPAD - XPAD);
564    svg.write_event(Event::Start(BytesStart::new("svg").with_attributes(vec![
565        ("id", "frames"),
566        ("x", &container_x),
567        ("width", &container_width),
568        ("total_samples", &format!("{}", timemax)),
569    ])))?;
570
571    // draw frames
572    let mut samples_txt_buffer = num_format::Buffer::default();
573    for frame in frames {
574        let x1_pct = frame.start_time as f64 * widthpertime_pct;
575        let x2_pct = frame.end_time as f64 * widthpertime_pct;
576
577        let (y1, y2) = match opt.direction {
578            Direction::Straight => {
579                let y1 = imageheight - opt.ypad2() - (frame.location.depth + 1) * opt.frame_height
580                    + FRAMEPAD;
581                let y2 = imageheight - opt.ypad2() - frame.location.depth * opt.frame_height;
582                (y1, y2)
583            }
584            Direction::Inverted => {
585                let y1 = opt.ypad1() + frame.location.depth * opt.frame_height;
586                let y2 = opt.ypad1() + (frame.location.depth + 1) * opt.frame_height - FRAMEPAD;
587                (y1, y2)
588            }
589        };
590
591        let rect = Rectangle {
592            x1_pct,
593            x1_samples: frame.start_time,
594            y1,
595            x2_pct,
596            x2_samples: frame.end_time,
597            y2,
598        };
599
600        // The rounding here can differ from the Perl version when the fractional part is `0.5`.
601        // The Perl version does `my $samples = sprintf "%.0f", ($etime - $stime) * $factor;`,
602        // but this can format in strange ways as shown in these examples:
603        //     `sprintf "%.0f", 1.5` produces "2"
604        //     `sprintf "%.0f", 2.5` produces "2"
605        //     `sprintf "%.0f", 3.5` produces "4"
606        let samples = ((frame.end_time - frame.start_time) as f64 * opt.factor).round() as usize;
607
608        // add thousands separators to `samples`
609        let _ = samples_txt_buffer.write_formatted(&samples, &Locale::en);
610        let samples_txt = samples_txt_buffer.as_str();
611
612        let info = if frame.location.function.is_empty() && frame.location.depth == 0 {
613            write!(buffer, "all ({} {}, 100%)", samples_txt, opt.count_name)
614        } else {
615            let pct = (100 * samples) as f64 / (timemax as f64 * opt.factor);
616            let function = deannotate(frame.location.function);
617            match frame.delta {
618                None => write!(
619                    buffer,
620                    "{} ({} {}, {:.2}%)",
621                    function, samples_txt, opt.count_name, pct
622                ),
623                // Special case delta == 0 so we don't format percentage with a + sign.
624                Some(0) => write!(
625                    buffer,
626                    "{} ({} {}, {:.2}%; 0.00%)",
627                    function, samples_txt, opt.count_name, pct,
628                ),
629                Some(mut delta) => {
630                    if opt.negate_differentials {
631                        delta = -delta;
632                    }
633                    let delta_pct = (100 * delta) as f64 / (timemax as f64 * opt.factor);
634                    write!(
635                        buffer,
636                        "{} ({} {}, {:.2}%; {:+.2}%)",
637                        function, samples_txt, opt.count_name, pct, delta_pct
638                    )
639                }
640            }
641        };
642
643        let (has_href, title) = write_container_start(
644            opt,
645            &mut svg,
646            &mut cache_a,
647            &mut cache_g,
648            &frame,
649            &buffer[info],
650        )?;
651
652        svg.write_event(Event::Start(BytesStart::new("title")))?;
653        svg.write_event(Event::Text(BytesText::new(title)))?;
654        svg.write_event(Event::End(BytesEnd::new("title")))?;
655
656        // select the color of the rectangle
657        let color = if frame.location.function == "--" {
658            color::VDGREY
659        } else if frame.location.function == "-" {
660            color::DGREY
661        } else if opt.color_diffusion {
662            // We want to visually highlight high priority regions for
663            // optimization: wider frames are redder. Typically when optimizing,
664            // a frame that is 50% of width is high priority, so it seems wrong
665            // to give it half the saturation of 100%. So we use sqrt to make
666            // the red dropoff less linear.
667            color::color_scale((((x2_pct - x1_pct) / 100.0).sqrt() * 2000.0) as isize, 2000)
668        } else if let Some(mut delta) = frame.delta {
669            if opt.negate_differentials {
670                delta = -delta;
671            }
672            color::color_scale(delta, delta_max)
673        } else if let Some(ref mut palette_map) = opt.palette_map {
674            let colors = opt.colors;
675            let hash = opt.hash;
676            let deterministic = opt.deterministic;
677            palette_map.find_color_for(frame.location.function, |name| {
678                color::color(colors, hash, deterministic, name, &mut thread_rng)
679            })
680        } else {
681            color::color(
682                opt.colors,
683                opt.hash,
684                opt.deterministic,
685                frame.location.function,
686                &mut thread_rng,
687            )
688        };
689        filled_rectangle(&mut svg, &mut buffer, &rect, color, &mut cache_rect)?;
690
691        let fitchars = (rect.width_pct()
692            / (100.0 * opt.font_size as f64 * opt.font_width / image_width))
693            .trunc() as usize;
694        let text: svg::TextArgument<'_> = if fitchars >= 3 {
695            // room for one char plus two dots
696            let f = deannotate(frame.location.function);
697
698            // TODO: use Unicode grapheme clusters instead
699            if f.len() < fitchars {
700                // no need to truncate
701                f.into()
702            } else {
703                // need to truncate :'(
704                use std::fmt::Write;
705                let mut w = buffer.writer();
706                for c in f.chars().take(fitchars - 2) {
707                    w.write_char(c).expect("writing to buffer shouldn't fail");
708                }
709                w.write_str("..").expect("writing to buffer shouldn't fail");
710                w.finish().into()
711            }
712        } else {
713            // don't show the function name
714            "".into()
715        };
716
717        // write the text
718        svg::write_str(
719            &mut svg,
720            &mut buffer,
721            svg::TextItem {
722                x: Dimension::Percent(rect.x1_pct + 100.0 * 3.0 / image_width),
723                y: 3.0 + (rect.y1 + rect.y2) as f64 / 2.0,
724                text,
725                extra: None,
726            },
727        )?;
728
729        buffer.clear();
730        if has_href {
731            svg.write_event(cache_a_end.borrow())?;
732        } else {
733            svg.write_event(cache_g_end.borrow())?;
734        }
735    }
736
737    svg.write_event(Event::End(BytesEnd::new("svg")))?;
738    svg.write_event(Event::End(BytesEnd::new("svg")))?;
739    svg.write_event(Event::Eof)?;
740
741    svg.into_inner().flush()?;
742    Ok(())
743}

745#[cfg(feature = "nameattr")]
746fn write_container_start<'a, W: Write>(
747    opt: &'a Options<'a>,
748    svg: &mut Writer<W>,
749    cache_a: &mut Event<'_>,
750    cache_g: &mut Event<'_>,
751    frame: &merge::TimedFrame<'_>,
752    mut title: &'a str,
753) -> io::Result<(bool, &'a str)> {
754    let frame_attributes = opt
755        .func_frameattrs
756        .frameattrs_for_func(frame.location.function);
757
758    let mut has_href = false;
759    if let Some(frame_attributes) = frame_attributes {
760        if frame_attributes.attrs.contains_key("xlink:href") {
761            write_container_attributes(cache_a, frame_attributes);
762            svg.write_event(cache_a.borrow())?;
763            has_href = true;
764        } else {
765            write_container_attributes(cache_g, frame_attributes);
766            svg.write_event(cache_g.borrow())?;
767        }
768        if let Some(ref t) = frame_attributes.title {
769            title = t.as_str();
770        }
771    } else if let Event::Start(ref mut c) = cache_g {
772        c.clear_attributes();
773        svg.write_event(cache_g.borrow())?;
774    }
775
776    Ok((has_href, title))
777}

796/// Writes attributes to the container, container could be g or a
797#[cfg(feature = "nameattr")]
798fn write_container_attributes(event: &mut Event<'_>, frame_attributes: &FrameAttrs) {
799    if let Event::Start(ref mut c) = event {
800        c.clear_attributes();
801        c.extend_attributes(
802            frame_attributes
803                .attrs
804                .iter()
805                .map(|(k, v)| (k.as_str(), v.as_str())),
806        );
807    } else {
808        unreachable!("cache wrapper was of wrong type: {:?}", event);
809    }
810}

812/// Produce a flame graph from a reader that contains a sequence of folded stack lines.
813///
814/// See [`from_lines`] for the expected format of each line.
815///
816/// The resulting flame graph will be written out to `writer` in SVG format.
817pub fn from_reader<R, W>(opt: &mut Options<'_>, reader: R, writer: W) -> io::Result<()>
818where
819    R: Read,
820    W: Write,
821{
822    from_readers(opt, iter::once(reader), writer)
823}

825/// Produce a flame graph from a set of readers that contain folded stack lines.
826///
827/// See [`from_lines`] for the expected format of each line.
828///
829/// The resulting flame graph will be written out to `writer` in SVG format.
830pub fn from_readers<R, W>(opt: &mut Options<'_>, readers: R, writer: W) -> io::Result<()>
831where
832    R: IntoIterator,
833    R::Item: Read,
834    W: Write,
835{
836    let mut input = String::new();
837    for mut reader in readers {
838        reader.read_to_string(&mut input)?;
839    }
840    from_lines(opt, input.lines(), writer)
841}
842
843/// Produce a flame graph from files that contain folded stack lines
844/// and write the result to provided `writer`.
845///
846/// If files is empty, STDIN will be used as input.
847pub fn from_files<W: Write>(opt: &mut Options<'_>, files: &[PathBuf], writer: W) -> io::Result<()> {
848    if files.is_empty() || files.len() == 1 && files[0].to_str() == Some("-") {
849        let stdin = io::stdin();
850        let r = BufReader::with_capacity(128 * 1024, stdin.lock());
851        from_reader(opt, r, writer)
852    } else if files.len() == 1 {
853        let r = File::open(&files[0])?;
854        from_reader(opt, r, writer)
855    } else {
856        let stdin = io::stdin();
857        let mut stdin_added = false;
858        let mut readers: Vec<Box<dyn Read>> = Vec::with_capacity(files.len());
859        for infile in files.iter() {
860            if infile.to_str() == Some("-") {
861                if !stdin_added {
862                    let r = BufReader::with_capacity(128 * 1024, stdin.lock());
863                    readers.push(Box::new(r));
864                    stdin_added = true;
865                }
866            } else {
867                let r = File::open(infile)?;
868                readers.push(Box::new(r));
869            }
870        }
871
872        from_readers(opt, readers, writer)
873    }
874}

876fn deannotate(f: &str) -> &str {
877    if f.ends_with(']') {
878        if let Some(ai) = f.rfind("_[") {
879            if f[ai..].len() == 4 && "kwij".contains(&f[ai + 2..ai + 3]) {
880                return &f[..ai];
881            }
882        }
883    }
884    f
885}

887fn filled_rectangle<W: Write>(
888    svg: &mut Writer<W>,
889    buffer: &mut StrStack,
890    rect: &Rectangle,
891    color: Color,
892    cache_rect: &mut Event<'_>,
893) -> io::Result<()> {
894    let x = write!(buffer, "{:.4}%", rect.x1_pct);
895    let y = write_usize(buffer, rect.y1);
896    let width = write!(buffer, "{:.4}%", rect.width_pct());
897    let height = write_usize(buffer, rect.height());
898    let color = write!(buffer, "rgb({},{},{})", color.r, color.g, color.b);
899    let x_samples = write_usize(buffer, rect.x1_samples);
900    let width_samples = write_usize(buffer, rect.x2_samples - rect.x1_samples);
901
902    if let Event::Empty(bytes_start) = cache_rect {
903        // clear the state
904        bytes_start.clear_attributes();
905        bytes_start.extend_attributes(args!(
906            "x" => &buffer[x],
907            "y" => &buffer[y],
908            "width" => &buffer[width],
909            "height" => &buffer[height],
910            "fill" => &buffer[color],
911            "fg:x" => &buffer[x_samples],
912            "fg:w" => &buffer[width_samples]
913        ));
914    } else {
915        unreachable!("cache wrapper was of wrong type: {:?}", cache_rect);
916    }
917    svg.write_event(cache_rect.borrow())
918}
919
920fn write_usize(buffer: &mut StrStack, value: usize) -> usize {
921    buffer.push(itoa::Buffer::new().format(value))
922}
