1//! Runtime library calls.
2//!
3//! Note that Wasm compilers may sometimes perform these inline rather than
4//! calling them, particularly when CPUs have special instructions which compute
5//! them directly.
6//!
7//! These functions are called by compiled Wasm code, and therefore must take
8//! certain care about some things:
9//!
10//! * They must only contain basic, raw i32/i64/f32/f64/pointer parameters that
11//!   are safe to pass across the system ABI.
12//!
13//! * If any nested function propagates an `Err(trap)` out to the library
14//!   function frame, we need to raise it. This involves some nasty and quite
15//!   unsafe code under the covers! Notably, after raising the trap, drops
16//!   **will not** be run for local variables! This can lead to things like
17//!   leaking `InstanceHandle`s which leads to never deallocating JIT code,
18//!   instances, and modules if we are not careful!
19//!
20//! * The libcall must be entered via a Wasm-to-libcall trampoline that saves
21//!   the last Wasm FP and PC for stack walking purposes. (For more details, see
22//!   `crates/wasmtime/src/runtime/vm/backtrace.rs`.)
23//!
24//! To make it easier to correctly handle all these things, **all** libcalls
25//! must be defined via the `libcall!` helper macro! See its doc comments below
26//! for an example, or just look at the rest of the file.
27//!
28//! ## Dealing with `externref`s
29//!
30//! When receiving a raw `*mut u8` that is actually a `VMExternRef` reference,
31//! convert it into a proper `VMExternRef` with `VMExternRef::clone_from_raw` as
32//! soon as apossible. Any GC before raw pointer is converted into a reference
33//! can potentially collect the referenced object, which could lead to use after
34//! free.
35//!
36//! Avoid this by eagerly converting into a proper `VMExternRef`! (Unfortunately
37//! there is no macro to help us automatically get this correct, so stay
38//! vigilant!)
39//!
40//! ```ignore
41//! pub unsafe extern "C" my_libcall_takes_ref(raw_extern_ref: *mut u8) {
42//!     // Before `clone_from_raw`, `raw_extern_ref` is potentially unrooted,
43//!     // and doing GC here could lead to use after free!
44//!
45//!     let my_extern_ref = if raw_extern_ref.is_null() {
46//!         None
47//!     } else {
48//!         Some(VMExternRef::clone_from_raw(raw_extern_ref))
49//!     };
50//!
51//!     // Now that we did `clone_from_raw`, it is safe to do a GC (or do
52//!     // anything else that might transitively GC, like call back into
53//!     // Wasm!)
54//! }
55//! ```

69/// Raw functions which are actually called from compiled code.
70///
71/// Invocation of a builtin currently looks like:
72///
73/// * A wasm function calls a cranelift-compiled trampoline that's generated
74///   once-per-builtin.
75/// * The cranelift-compiled trampoline performs any necessary actions to exit
76///   wasm, such as dealing with fp/pc/etc.
77/// * The cranelift-compiled trampoline loads a function pointer from an array
78///   stored in `VMContext` That function pointer is defined in this module.
79/// * This module runs, handling things like `catch_unwind` and `Result` and
80///   such.
81/// * This module delegates to the outer module (this file) which has the actual
82///   implementation.
83pub mod raw {
84    // Allow these things because of the macro and how we can't differentiate
85    // between doc comments and `cfg`s.
86    #![allow(unused_doc_comments, unused_attributes)]
87
88    use crate::runtime::vm::{Instance, TrapReason, VMContext};
89
90    macro_rules! libcall {
91        (
92            $(
93                $( #[cfg($attr:meta)] )?
94                $name:ident( vmctx: vmctx $(, $pname:ident: $param:ident )* ) $( -> $result:ident )?;
95            )*
96        ) => {
97            $(
98                // This is the direct entrypoint from the compiled module which
99                // still has the raw signature.
100                //
101                // This will delegate to the outer module to the actual
102                // implementation and automatically perform `catch_unwind` along
103                // with conversion of the return value in the face of traps.
104                #[allow(unused_variables, missing_docs)]
105                pub unsafe extern "C" fn $name(
106                    vmctx: *mut VMContext,
107                    $( $pname : libcall!(@ty $param), )*
108                ) $( -> libcall!(@ty $result))? {
109                    $(#[cfg($attr)])?
110                    {
111                        let ret = crate::runtime::vm::traphandlers::catch_unwind_and_longjmp(|| {
112                            Instance::from_vmctx(vmctx, |instance| {
113                                {
114                                    super::$name(instance, $($pname),*)
115                                }
116                            })
117                        });
118                        LibcallResult::convert(ret)
119                    }
120                    $(
121                        #[cfg(not($attr))]
122                        unreachable!();
123                    )?
124                }
125
126                // This works around a `rustc` bug where compiling with LTO
127                // will sometimes strip out some of these symbols resulting
128                // in a linking failure.
129                #[allow(non_upper_case_globals)]
130                const _: () = {
131                    #[used]
132                    static I_AM_USED: unsafe extern "C" fn(
133                        *mut VMContext,
134                        $( $pname : libcall!(@ty $param), )*
135                    ) $( -> libcall!(@ty $result))? = $name;
136                };
137            )*
138        };
139
140        (@ty i32) => (u32);
141        (@ty i64) => (u64);
142        (@ty reference) => (u32);
143        (@ty pointer) => (*mut u8);
144        (@ty vmctx) => (*mut VMContext);
145    }
146
147    wasmtime_environ::foreach_builtin_function!(libcall);
148
149    // Helper trait to convert results of libcalls below into the ABI of what
150    // the libcall expects.
151    //
152    // This basically entirely exists for the `Result` implementation which
153    // "unwraps" via a throwing of a trap.
154    trait LibcallResult {
155        type Abi;
156        unsafe fn convert(self) -> Self::Abi;
157    }
158
159    impl LibcallResult for () {
160        type Abi = ();
161        unsafe fn convert(self) {}
162    }
163
164    impl<T, E> LibcallResult for Result<T, E>
165    where
166        E: Into<TrapReason>,
167    {
168        type Abi = T;
169        unsafe fn convert(self) -> T {
170            match self {
171                Ok(t) => t,
172                Err(e) => crate::runtime::vm::traphandlers::raise_trap(e.into()),
173            }
174        }
175    }
176
177    impl LibcallResult for *mut u8 {
178        type Abi = *mut u8;
179        unsafe fn convert(self) -> *mut u8 {
180            self
181        }
182    }
183}

185fn memory32_grow(
186    instance: &mut Instance,
187    delta: u64,
188    memory_index: u32,
189) -> Result<*mut u8, TrapReason> {
190    let memory_index = MemoryIndex::from_u32(memory_index);
191    let result =
192        match instance
193            .memory_grow(memory_index, delta)
194            .map_err(|error| TrapReason::User {
195                error,
196                needs_backtrace: true,
197            })? {
198            Some(size_in_bytes) => size_in_bytes / instance.memory_page_size(memory_index),
199            None => usize::max_value(),
200        };
201    Ok(result as *mut _)
202}

204/// Implementation of `table.grow` for `funcref` tables.
205unsafe fn table_grow_func_ref(
206    instance: &mut Instance,
207    table_index: u32,
208    delta: u32,
209    init_value: *mut u8,
210) -> Result<u32> {
211    let table_index = TableIndex::from_u32(table_index);
212
213    let element = match instance.table_element_type(table_index) {
214        TableElementType::Func => (init_value as *mut VMFuncRef).into(),
215        TableElementType::GcRef => unreachable!(),
216    };
217
218    Ok(match instance.table_grow(table_index, delta, element)? {
219        Some(r) => r,
220        None => (-1_i32).unsigned(),
221    })
222}

226unsafe fn table_grow_gc_ref(
227    instance: &mut Instance,

229    delta: u32,

239    };
240

242        Some(r) => r,
243        None => (-1_i32).unsigned(),
244    })

247/// Implementation of `table.fill` for `funcref`s.
248unsafe fn table_fill_func_ref(
249    instance: &mut Instance,
250    table_index: u32,
251    dst: u32,
252    val: *mut u8,
253    len: u32,
254) -> Result<(), Trap> {
255    let table_index = TableIndex::from_u32(table_index);
256    let table = &mut *instance.get_table(table_index);
257    match table.element_type() {
258        TableElementType::Func => {
259            let val = val.cast::<VMFuncRef>();
260            table.fill((*instance.store()).gc_store(), dst, val.into(), len)
261        }
262        TableElementType::GcRef => unreachable!(),
263    }
264}

271    val: u32,

283        }

287// Implementation of `table.copy`.
288unsafe fn table_copy(
289    instance: &mut Instance,
290    dst_table_index: u32,
291    src_table_index: u32,
292    dst: u32,
293    src: u32,
294    len: u32,
295) -> Result<(), Trap> {
296    let dst_table_index = TableIndex::from_u32(dst_table_index);
297    let src_table_index = TableIndex::from_u32(src_table_index);
298    let dst_table = instance.get_table(dst_table_index);
299    // Lazy-initialize the whole range in the source table first.
300    let src_range = src..(src.checked_add(len).unwrap_or(u32::MAX));
301    let src_table = instance.get_table_with_lazy_init(src_table_index, src_range);
302    let gc_store = (*instance.store()).gc_store();
303    Table::copy(gc_store, dst_table, src_table, dst, src, len)
304}

380// Returns a table entry after lazily initializing it.
381unsafe fn table_get_lazy_init_func_ref(
382    instance: &mut Instance,
383    table_index: u32,
384    index: u32,
385) -> *mut u8 {

404/// Do a GC, keeping `gc_ref` rooted and returning the updated `gc_ref`
405/// reference.
406#[cfg(feature = "gc")]
407unsafe fn gc(instance: &mut Instance, gc_ref: u32) -> Result<u32> {
408    let gc_ref = VMGcRef::from_raw_u32(gc_ref);
409    let gc_ref = gc_ref.map(|r| (*instance.store()).gc_store().clone_gc_ref(&r));
410
411    if let Some(gc_ref) = &gc_ref {
412        // It is possible that we are GC'ing because the DRC's activation
413        // table's bump region is full, and we failed to insert `gc_ref` into
414        // the bump region. But it is an invariant for DRC collection that all
415        // GC references on the stack are in the DRC's activations table at the
416        // time of a GC. So make sure to "expose" this GC reference to Wasm (aka
417        // insert it into the DRC's activation table) before we do the actual
418        // GC.
419        let gc_store = (*instance.store()).gc_store();
420        let gc_ref = gc_store.clone_gc_ref(gc_ref);
421        gc_store.expose_gc_ref_to_wasm(gc_ref);
422    }
423
424    match (*instance.store()).gc(gc_ref)? {
425        None => Ok(0),
426        Some(r) => {
427            let raw = r.as_raw_u32();
428            (*instance.store()).gc_store().expose_gc_ref_to_wasm(r);
429            Ok(raw)
430        }
431    }
432}

434/// Perform a Wasm `global.get` for GC reference globals.
435#[cfg(feature = "gc")]
436unsafe fn gc_ref_global_get(instance: &mut Instance, index: u32) -> Result<u32> {
437    use core::num::NonZeroUsize;
438
439    let index = wasmtime_environ::GlobalIndex::from_u32(index);
440    let global = instance.defined_or_imported_global_ptr(index);
441    let gc_store = (*instance.store()).gc_store();
442
443    if gc_store
444        .gc_heap
445        .need_gc_before_entering_wasm(NonZeroUsize::new(1).unwrap())
446    {
447        (*instance.store()).gc(None)?;
448    }
449
450    match (*global).as_gc_ref() {
451        None => Ok(0),
452        Some(gc_ref) => {
453            let gc_ref = gc_store.clone_gc_ref(gc_ref);
454            let ret = gc_ref.as_raw_u32();
455            gc_store.expose_gc_ref_to_wasm(gc_ref);
456            Ok(ret)
457        }
458    }
459}

527// Hook for validating malloc using wmemcheck_state.
528#[cfg(feature = "wmemcheck")]
529unsafe fn check_malloc(instance: &mut Instance, addr: u32, len: u32) -> Result<u32> {
530    if let Some(wmemcheck_state) = &mut instance.wmemcheck_state {
531        let result = wmemcheck_state.malloc(addr as usize, len as usize);
532        wmemcheck_state.memcheck_on();
533        match result {
534            Ok(()) => {
535                return Ok(0);
536            }
537            Err(DoubleMalloc { addr, len }) => {
538                bail!("Double malloc at addr {:#x} of size {}", addr, len)
539            }
540            Err(OutOfBounds { addr, len }) => {
541                bail!("Malloc out of bounds at addr {:#x} of size {}", addr, len);
542            }
543            _ => {
544                panic!("unreachable")
545            }
546        }
547    }
548    Ok(0)
549}

655/// This module contains functions which are used for resolving relocations at

657///
658/// These functions are not used by default and currently the only platform
659/// they're used for is on x86_64 when SIMD is disabled and then SSE features

722    const TOINT_32: f32 = 1.0 / f32::EPSILON;
723    const TOINT_64: f64 = 1.0 / f64::EPSILON;
724
725    // NB: replace with `round_ties_even` from libstd when it's stable as
726    // tracked by rust-lang/rust#96710
727    pub extern "C" fn nearestf32(x: f32) -> f32 {
728        // Rust doesn't have a nearest function; there's nearbyint, but it's not
729        // stabilized, so do it manually.
730        // Nearest is either ceil or floor depending on which is nearest or even.
731        // This approach exploited round half to even default mode.
732        let i = x.to_bits();
733        let e = i >> 23 & 0xff;
734        if e >= 0x7f_u32 + 23 {
735            // Check for NaNs.
736            if e == 0xff {
737                // Read the 23-bits significand.
738                if i & 0x7fffff != 0 {
739                    // Ensure it's arithmetic by setting the significand's most
740                    // significant bit to 1; it also works for canonical NaNs.
741                    return f32::from_bits(i | (1 << 22));
742                }
743            }
744            x
745        } else {
746            let abs = float_function! {
747                std: f32::abs,
748                core: libm::fabsf,
749            };
750            let copysign = float_function! {
751                std: f32::copysign,
752                core: libm::copysignf,
753            };
754
755            copysign(abs(x) + TOINT_32 - TOINT_32, x)
756        }
757    }
758
759    pub extern "C" fn nearestf64(x: f64) -> f64 {
760        let i = x.to_bits();
761        let e = i >> 52 & 0x7ff;
762        if e >= 0x3ff_u64 + 52 {
763            // Check for NaNs.
764            if e == 0x7ff {
765                // Read the 52-bits significand.
766                if i & 0xfffffffffffff != 0 {
767                    // Ensure it's arithmetic by setting the significand's most
768                    // significant bit to 1; it also works for canonical NaNs.
769                    return f64::from_bits(i | (1 << 51));
770                }
771            }
772            x
773        } else {
774            let abs = float_function! {
775                std: f64::abs,
776                core: libm::fabs,
777            };
778            let copysign = float_function! {
779                std: f64::copysign,
780                core: libm::copysign,
781            };
782
783            copysign(abs(x) + TOINT_64 - TOINT_64, x)
784        }
785    }

803    // This intrinsic is only used on x86_64 platforms as an implementation of
804    // the `pshufb` instruction when SSSE3 is not available.
805    #[cfg(target_arch = "x86_64")]
806    use core::arch::x86_64::__m128i;
807    #[cfg(target_arch = "x86_64")]
808    #[allow(improper_ctypes_definitions)]
809    pub extern "C" fn x86_pshufb(a: __m128i, b: __m128i) -> __m128i {
810        union U {
811            reg: __m128i,
812            mem: [u8; 16],
813        }
814
815        unsafe {
816            let a = U { reg: a }.mem;
817            let b = U { reg: b }.mem;
818
819            let select = |arr: &[u8; 16], byte: u8| {
820                if byte & 0x80 != 0 {
821                    0x00
822                } else {
823                    arr[(byte & 0xf) as usize]
824                }
825            };
826
827            U {
828                mem: [
829                    select(&a, b[0]),
830                    select(&a, b[1]),
831                    select(&a, b[2]),
832                    select(&a, b[3]),
833                    select(&a, b[4]),
834                    select(&a, b[5]),
835                    select(&a, b[6]),
836                    select(&a, b[7]),
837                    select(&a, b[8]),
838                    select(&a, b[9]),
839                    select(&a, b[10]),
840                    select(&a, b[11]),
841                    select(&a, b[12]),
842                    select(&a, b[13]),
843                    select(&a, b[14]),
844                    select(&a, b[15]),
845                ],
846            }
847            .reg
848        }
849    }
