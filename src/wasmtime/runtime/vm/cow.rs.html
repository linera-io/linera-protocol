21/// Backing images for memories in a module.
22///

36/// One backing image for one memory.
37#[derive(Debug, PartialEq)]
38pub struct MemoryImage {
39    /// The platform-specific source of this image.
40    ///
41    /// This might be a mapped `*.cwasm` file or on Unix it could also be a
42    /// `Memfd` as an anonymous file in memory on Linux. In either case this is
43    /// used as the backing-source for the CoW image.
44    source: MemoryImageSource,
45
46    /// Length of image, in bytes.
47    ///
48    /// Note that initial memory size may be larger; leading and trailing zeroes
49    /// are truncated (handled by backing fd).
50    ///
51    /// Must be a multiple of the system page size.
52    len: usize,
53
54    /// Image starts this many bytes into `source`.
55    ///
56    /// This is 0 for anonymous-backed memfd files and is the offset of the
57    /// data section in a `*.cwasm` file for `*.cwasm`-backed images.
58    ///
59    /// Must be a multiple of the system page size.
60    source_offset: u64,
61
62    /// Image starts this many bytes into heap space.
63    ///
64    /// Must be a multiple of the system page size.
65    linear_memory_offset: usize,
66}
67
68impl MemoryImage {
69    fn new(
70        page_size: u32,
71        offset: u64,
72        data: &[u8],
73        mmap: Option<&MmapVec>,
74    ) -> Result<Option<MemoryImage>> {
75        // Sanity-check that various parameters are page-aligned.
76        let len = data.len();
77        assert_eq!(offset % u64::from(page_size), 0);
78        assert_eq!((len as u32) % page_size, 0);
79        let linear_memory_offset = match usize::try_from(offset) {
80            Ok(offset) => offset,
81            Err(_) => return Ok(None),
82        };
83
84        // If a backing `mmap` is present then `data` should be a sub-slice of
85        // the `mmap`. The sanity-checks here double-check that. Additionally
86        // compilation should have ensured that the `data` section is
87        // page-aligned within `mmap`, so that's also all double-checked here.
88        //
89        // Finally if the `mmap` itself comes from a backing file on disk, such
90        // as a `*.cwasm` file, then that's a valid source of data for the
91        // memory image so we simply return referencing that.
92        //
93        // Note that this path is platform-agnostic in the sense of all
94        // platforms we support support memory mapping copy-on-write data from
95        // files, but for now this is still a Linux-specific region of Wasmtime.
96        // Some work will be needed to get this file compiling for macOS and
97        // Windows.
98        if let Some(mmap) = mmap {
99            let start = mmap.as_ptr() as usize;
100            let end = start + mmap.len();
101            let data_start = data.as_ptr() as usize;
102            let data_end = data_start + data.len();
103            assert!(start <= data_start && data_end <= end);
104            assert_eq!((start as u32) % page_size, 0);
105            assert_eq!((data_start as u32) % page_size, 0);
106            assert_eq!((data_end as u32) % page_size, 0);
107            assert_eq!((mmap.original_offset() as u32) % page_size, 0);
108
109            #[cfg(feature = "std")]
110            if let Some(file) = mmap.original_file() {
111                if let Some(source) = MemoryImageSource::from_file(file) {
112                    return Ok(Some(MemoryImage {
113                        source,
114                        source_offset: u64::try_from(mmap.original_offset() + (data_start - start))
115                            .unwrap(),
116                        linear_memory_offset,
117                        len,
118                    }));
119                }
120            }
121        }
122
123        // If `mmap` doesn't come from a file then platform-specific mechanisms
124        // may be used to place the data in a form that's amenable to an mmap.
125        if let Some(source) = MemoryImageSource::from_data(data).err2anyhow()? {
126            return Ok(Some(MemoryImage {
127                source,
128                source_offset: 0,
129                linear_memory_offset,
130                len,
131            }));
132        }
133
134        Ok(None)
135    }
136
137    unsafe fn map_at(&self, base: *mut u8) -> Result<()> {
138        self.source
139            .map_at(
140                base.add(self.linear_memory_offset),
141                self.len,
142                self.source_offset,
143            )
144            .err2anyhow()?;
145        Ok(())
146    }
147
148    unsafe fn remap_as_zeros_at(&self, base: *mut u8) -> Result<()> {
149        self.source
150            .remap_as_zeros_at(base.add(self.linear_memory_offset), self.len)
151            .err2anyhow()?;
152        Ok(())
153    }
154}

156impl ModuleMemoryImages {
157    /// Create a new `ModuleMemoryImages` for the given module. This can be
158    /// passed in as part of a `InstanceAllocationRequest` to speed up
159    /// instantiation and execution by using copy-on-write-backed memories.
160    pub fn new(
161        module: &Module,
162        wasm_data: &[u8],
163        mmap: Option<&MmapVec>,
164    ) -> Result<Option<ModuleMemoryImages>> {
165        let map = match &module.memory_initialization {
166            MemoryInitialization::Static { map } => map,
167            _ => return Ok(None),
168        };
169        let mut memories = PrimaryMap::with_capacity(map.len());
170        let page_size = crate::runtime::vm::host_page_size() as u32;
171        for (memory_index, init) in map {
172            // mmap-based-initialization only works for defined memories with a
173            // known starting point of all zeros, so bail out if the mmeory is
174            // imported.
175            let defined_memory = match module.defined_memory_index(memory_index) {
176                Some(idx) => idx,
177                None => return Ok(None),
178            };
179
180            // If there's no initialization for this memory known then we don't
181            // need an image for the memory so push `None` and move on.
182            let init = match init {
183                Some(init) => init,
184                None => {
185                    memories.push(None);
186                    continue;
187                }
188            };
189
190            // Get the image for this wasm module  as a subslice of `wasm_data`,
191            // and then use that to try to create the `MemoryImage`. If this
192            // creation files then we fail creating `ModuleMemoryImages` since this
193            // memory couldn't be represented.
194            let data = &wasm_data[init.data.start as usize..init.data.end as usize];
195            if module.memory_plans[memory_index]
196                .memory
197                .minimum_byte_size()
198                .map_or(false, |mem_initial_len| {
199                    init.offset + u64::try_from(data.len()).unwrap() > mem_initial_len
200                })
201            {
202                // The image is rounded up to multiples of the host OS page
203                // size. But if Wasm is using a custom page size, the Wasm page
204                // size might be smaller than the host OS page size, and that
205                // rounding might have made the image larger than the Wasm
206                // memory's initial length. This is *probably* okay, since the
207                // rounding would have just introduced new runs of zeroes in the
208                // image, but out of an abundance of caution we don't generate
209                // CoW images in this scenario.
210                return Ok(None);
211            }
212            let image = match MemoryImage::new(page_size, init.offset, data, mmap)? {
213                Some(image) => image,
214                None => return Ok(None),
215            };
216
217            let idx = memories.push(Some(Arc::new(image)));
218            assert_eq!(idx, defined_memory);
219        }
220
221        Ok(Some(ModuleMemoryImages { memories }))
222    }

225/// Slot management of a copy-on-write image which can be reused for the pooling
226/// allocator.
227///
228/// This data structure manages a slot of linear memory, primarily in the
229/// pooling allocator, which optionally has a contiguous memory image in the
230/// middle of it. Pictorially this data structure manages a virtual memory
231/// region that looks like:
232///
233/// ```text
234///   +--------------------+-------------------+--------------+--------------+
235///   |   anonymous        |      optional     |   anonymous  |    PROT_NONE |
236///   |     zero           |       memory      |     zero     |     memory   |
237///   |    memory          |       image       |    memory    |              |
238///   +--------------------+-------------------+--------------+--------------+
239///   |                     <------+---------->
240///   |<-----+------------>         \
241///   |      \                   image.len
242///   |       \
243///   |  image.linear_memory_offset
244///   |
245///   \
246///  self.base is this virtual address
247///
248///    <------------------+------------------------------------------------>
249///                        \
250///                      static_size
251///
252///    <------------------+---------------------------------->
253///                        \
254///                      accessible
255/// ```
256///
257/// When a `MemoryImageSlot` is created it's told what the `static_size` and
258/// `accessible` limits are. Initially there is assumed to be no image in linear
259/// memory.
260///
261/// When `MemoryImageSlot::instantiate` is called then the method will perform
262/// a "synchronization" to take the image from its prior state to the new state
263/// for the image specified. The first instantiation for example will mmap the
264/// heap image into place. Upon reuse of a slot nothing happens except possibly
265/// shrinking `self.accessible`. When a new image is used then the old image is
266/// mapped to anonymous zero memory and then the new image is mapped in place.
267///
268/// A `MemoryImageSlot` is either `dirty` or it isn't. When a `MemoryImageSlot`
269/// is dirty then it is assumed that any memory beneath `self.accessible` could
270/// have any value. Instantiation cannot happen into a `dirty` slot, however, so
271/// the `MemoryImageSlot::clear_and_remain_ready` returns this memory back to
272/// its original state to mark `dirty = false`. This is done by resetting all
273/// anonymous memory back to zero and the image itself back to its initial
274/// contents.
275///
276/// On Linux this is achieved with the `madvise(MADV_DONTNEED)` syscall. This
277/// syscall will release the physical pages back to the OS but retain the
278/// original mappings, effectively resetting everything back to its initial
279/// state. Non-linux platforms will replace all memory below `self.accessible`
280/// with a fresh zero'd mmap, meaning that reuse is effectively not supported.
281#[derive(Debug)]
282pub struct MemoryImageSlot {
283    /// The base address in virtual memory of the actual heap memory.
284    ///
285    /// Bytes at this address are what is seen by the Wasm guest code.
286    base: SendSyncPtr<u8>,
287
288    /// The maximum static memory size which `self.accessible` can grow to.
289    static_size: usize,
290
291    /// An optional image that is currently being used in this linear memory.
292    ///
293    /// This can be `None` in which case memory is originally all zeros. When
294    /// `Some` the image describes where it's located within the image.
295    image: Option<Arc<MemoryImage>>,
296
297    /// The size of the heap that is readable and writable.
298    ///
299    /// Note that this may extend beyond the actual linear memory heap size in
300    /// the case of dynamic memories in use. Memory accesses to memory below
301    /// `self.accessible` may still page fault as pages are lazily brought in
302    /// but the faults will always be resolved by the kernel.
303    accessible: usize,
304
305    /// Whether this slot may have "dirty" pages (pages written by an
306    /// instantiation). Set by `instantiate()` and cleared by
307    /// `clear_and_remain_ready()`, and used in assertions to ensure
308    /// those methods are called properly.
309    ///
310    /// Invariant: if !dirty, then this memory slot contains a clean
311    /// CoW mapping of `image`, if `Some(..)`, and anonymous-zero
312    /// memory beyond the image up to `static_size`. The addresses
313    /// from offset 0 to `self.accessible` are R+W and set to zero or the
314    /// initial image content, as appropriate. Everything between
315    /// `self.accessible` and `self.static_size` is inaccessible.
316    dirty: bool,
317
318    /// Whether this MemoryImageSlot is responsible for mapping anonymous
319    /// memory (to hold the reservation while overwriting mappings
320    /// specific to this slot) in place when it is dropped. Default
321    /// on, unless the caller knows what they are doing.
322    clear_on_drop: bool,
323}

325impl MemoryImageSlot {
326    /// Create a new MemoryImageSlot. Assumes that there is an anonymous
327    /// mmap backing in the given range to start.
328    ///
329    /// The `accessible` parameter describes how much of linear memory is
330    /// already mapped as R/W with all zero-bytes. The `static_size` value is
331    /// the maximum size of this image which `accessible` cannot grow beyond,
332    /// and all memory from `accessible` from `static_size` should be mapped as
333    /// `PROT_NONE` backed by zero-bytes.
334    pub(crate) fn create(base_addr: *mut c_void, accessible: usize, static_size: usize) -> Self {
335        MemoryImageSlot {
336            base: NonNull::new(base_addr.cast()).unwrap().into(),
337            static_size,
338            accessible,
339            image: None,
340            dirty: false,
341            clear_on_drop: true,
342        }
343    }

345    #[cfg(feature = "pooling-allocator")]

347        MemoryImageSlot {

354            image: None,

361    /// Inform the MemoryImageSlot that it should *not* clear the underlying
362    /// address space when dropped. This should be used only when the

365    pub(crate) fn no_clear_on_drop(&mut self) {

369    pub(crate) fn set_heap_limit(&mut self, size_bytes: usize) -> Result<()> {
370        assert!(size_bytes <= self.static_size);
371
372        // If the heap limit already addresses accessible bytes then no syscalls
373        // are necessary since the data is already mapped into the process and
374        // waiting to go.
375        //
376        // This is used for "dynamic" memories where memory is not always
377        // decommitted during recycling (but it's still always reset).
378        if size_bytes <= self.accessible {
379            return Ok(());
380        }
381
382        // Otherwise use `mprotect` to make the new pages read/write.
383        self.set_protection(self.accessible..size_bytes, true)?;
384        self.accessible = size_bytes;
385
386        Ok(())
387    }
388
389    /// Prepares this slot for the instantiation of a new instance with the
390    /// provided linear memory image.
391    ///
392    /// The `initial_size_bytes` parameter indicates the required initial size
393    /// of the heap for the instance. The `maybe_image` is an optional initial
394    /// image for linear memory to contains. The `style` is the way compiled
395    /// code will be accessing this memory.
396    ///
397    /// The purpose of this method is to take a previously pristine slot
398    /// (`!self.dirty`) and transform its prior state into state necessary for
399    /// the given parameters. This could include, for example:
400    ///
401    /// * More memory may be made read/write if `initial_size_bytes` is larger
402    ///   than `self.accessible`.
403    /// * For `MemoryStyle::Static` linear memory may be made `PROT_NONE` if
404    ///   `self.accessible` is larger than `initial_size_bytes`.
405    /// * If no image was previously in place or if the wrong image was
406    ///   previously in place then `mmap` may be used to setup the initial
407    ///   image.
408    pub(crate) fn instantiate(
409        &mut self,
410        initial_size_bytes: usize,
411        maybe_image: Option<&Arc<MemoryImage>>,
412        plan: &MemoryPlan,
413    ) -> Result<()> {
414        assert!(!self.dirty);
415        assert!(initial_size_bytes <= self.static_size);
416
417        // First order of business is to blow away the previous linear memory
418        // image if it doesn't match the image specified here. If one is
419        // detected then it's reset with anonymous memory which means that all
420        // of memory up to `self.accessible` will now be read/write and zero.
421        //
422        // Note that this intentionally a "small mmap" which only covers the
423        // extent of the prior initialization image in order to preserve
424        // resident memory that might come before or after the image.
425        if self.image.as_ref() != maybe_image {
426            self.remove_image()?;
427        }
428
429        // The next order of business is to ensure that `self.accessible` is
430        // appropriate. First up is to grow the read/write portion of memory if
431        // it's not large enough to accommodate `initial_size_bytes`.
432        if self.accessible < initial_size_bytes {
433            self.set_protection(self.accessible..initial_size_bytes, true)?;
434            self.accessible = initial_size_bytes;
435        }
436
437        // If (1) the accessible region is not in its initial state, and (2) the
438        // memory relies on virtual memory at all (i.e. has offset guard pages
439        // and/or is static), then we need to reset memory protections. Put
440        // another way, the only time it is safe to not reset protections is
441        // when we are using dynamic memory without any guard pages.
442        if initial_size_bytes < self.accessible
443            && (plan.offset_guard_size > 0 || matches!(plan.style, MemoryStyle::Static { .. }))
444        {
445            self.set_protection(initial_size_bytes..self.accessible, false)?;
446            self.accessible = initial_size_bytes;
447        }
448
449        // Now that memory is sized appropriately the final operation is to
450        // place the new image into linear memory. Note that this operation is
451        // skipped if `self.image` matches `maybe_image`.
452        assert!(initial_size_bytes <= self.accessible);
453        if self.image.as_ref() != maybe_image {
454            if let Some(image) = maybe_image.as_ref() {
455                assert!(
456                    image.linear_memory_offset.checked_add(image.len).unwrap()
457                        <= initial_size_bytes
458                );
459                if image.len > 0 {
460                    unsafe {
461                        image.map_at(self.base.as_ptr())?;
462                    }
463                }
464            }
465            self.image = maybe_image.cloned();
466        }
467
468        // Flag ourselves as `dirty` which means that the next operation on this
469        // slot is required to be `clear_and_remain_ready`.
470        self.dirty = true;
471
472        Ok(())
473    }
474
475    pub(crate) fn remove_image(&mut self) -> Result<()> {
476        if let Some(image) = &self.image {
477            unsafe {
478                image.remap_as_zeros_at(self.base.as_ptr())?;
479            }
480            self.image = None;
481        }
482        Ok(())
483    }
484
485    /// Resets this linear memory slot back to a "pristine state".
486    ///
487    /// This will reset the memory back to its original contents on Linux or
488    /// reset the contents back to zero on other platforms. The `keep_resident`
489    /// argument is the maximum amount of memory to keep resident in this
490    /// process's memory on Linux. Up to that much memory will be `memset` to
491    /// zero where the rest of it will be reset or released with `madvise`.
492    #[allow(dead_code)] // ignore warnings as this is only used in some cfgs
493    pub(crate) fn clear_and_remain_ready(
494        &mut self,
495        keep_resident: usize,
496        decommit: impl FnMut(*mut u8, usize),
497    ) -> Result<()> {
498        assert!(self.dirty);
499
500        unsafe {
501            self.reset_all_memory_contents(keep_resident, decommit)?;
502        }
503
504        self.dirty = false;
505        Ok(())
506    }
507
508    #[allow(dead_code)] // ignore warnings as this is only used in some cfgs
509    unsafe fn reset_all_memory_contents(
510        &mut self,
511        keep_resident: usize,
512        decommit: impl FnMut(*mut u8, usize),
513    ) -> Result<()> {
514        match vm::decommit_behavior() {
515            DecommitBehavior::Zero => {
516                // If we're not on Linux then there's no generic platform way to
517                // reset memory back to its original state, so instead reset memory
518                // back to entirely zeros with an anonymous backing.
519                //
520                // Additionally the previous image, if any, is dropped here
521                // since it's no longer applicable to this mapping.
522                self.reset_with_anon_memory()
523            }
524            DecommitBehavior::RestoreOriginalMapping => {
525                self.reset_with_original_mapping(keep_resident, decommit);
526                Ok(())
527            }
528        }
529    }
530
531    #[allow(dead_code)] // ignore warnings as this is only used in some cfgs
532    unsafe fn reset_with_original_mapping(
533        &mut self,
534        keep_resident: usize,
535        mut decommit: impl FnMut(*mut u8, usize),
536    ) {
537        match &self.image {
538            Some(image) => {
539                assert!(self.accessible >= image.linear_memory_offset + image.len);
540                if image.linear_memory_offset < keep_resident {
541                    // If the image starts below the `keep_resident` then
542                    // memory looks something like this:
543                    //
544                    //               up to `keep_resident` bytes
545                    //                          |
546                    //          +--------------------------+  remaining_memset
547                    //          |                          | /
548                    //  <-------------->                <------->
549                    //
550                    //                              image_end
551                    // 0        linear_memory_offset   |             accessible
552                    // |                |              |                  |
553                    // +----------------+--------------+---------+--------+
554                    // |  dirty memory  |    image     |   dirty memory   |
555                    // +----------------+--------------+---------+--------+
556                    //
557                    //  <------+-------> <-----+----->  <---+---> <--+--->
558                    //         |               |            |        |
559                    //         |               |            |        |
560                    //   memset (1)            /            |   madvise (4)
561                    //                  mmadvise (2)       /
562                    //                                    /
563                    //                              memset (3)
564                    //
565                    //
566                    // In this situation there are two disjoint regions that are
567                    // `memset` manually to zero. Note that `memset (3)` may be
568                    // zero bytes large. Furthermore `madvise (4)` may also be
569                    // zero bytes large.
570
571                    let image_end = image.linear_memory_offset + image.len;
572                    let mem_after_image = self.accessible - image_end;
573                    let remaining_memset =
574                        (keep_resident - image.linear_memory_offset).min(mem_after_image);
575
576                    // This is memset (1)
577                    ptr::write_bytes(self.base.as_ptr(), 0u8, image.linear_memory_offset);
578
579                    // This is madvise (2)
580                    self.restore_original_mapping(
581                        image.linear_memory_offset,
582                        image.len,
583                        &mut decommit,
584                    );
585
586                    // This is memset (3)
587                    ptr::write_bytes(self.base.as_ptr().add(image_end), 0u8, remaining_memset);
588
589                    // This is madvise (4)
590                    self.restore_original_mapping(
591                        image_end + remaining_memset,
592                        mem_after_image - remaining_memset,
593                        &mut decommit,
594                    );
595                } else {
596                    // If the image starts after the `keep_resident` threshold
597                    // then we memset the start of linear memory and then use
598                    // madvise below for the rest of it, including the image.
599                    //
600                    // 0             keep_resident                   accessible
601                    // |                |                                 |
602                    // +----------------+---+----------+------------------+
603                    // |  dirty memory      |  image   |   dirty memory   |
604                    // +----------------+---+----------+------------------+
605                    //
606                    //  <------+-------> <-------------+----------------->
607                    //         |                       |
608                    //         |                       |
609                    //   memset (1)                 madvise (2)
610                    //
611                    // Here only a single memset is necessary since the image
612                    // started after the threshold which we're keeping resident.
613                    // Note that the memset may be zero bytes here.
614
615                    // This is memset (1)
616                    ptr::write_bytes(self.base.as_ptr(), 0u8, keep_resident);
617
618                    // This is madvise (2)
619                    self.restore_original_mapping(
620                        keep_resident,
621                        self.accessible - keep_resident,
622                        decommit,
623                    );
624                }
625            }
626
627            // If there's no memory image for this slot then memset the first
628            // bytes in the memory back to zero while using `madvise` to purge
629            // the rest.
630            None => {
631                let size_to_memset = keep_resident.min(self.accessible);
632                ptr::write_bytes(self.base.as_ptr(), 0u8, size_to_memset);
633                self.restore_original_mapping(
634                    size_to_memset,
635                    self.accessible - size_to_memset,
636                    decommit,
637                );
638            }
639        }
640    }
641
642    #[allow(dead_code)] // ignore warnings as this is only used in some cfgs
643    unsafe fn restore_original_mapping(
644        &self,
645        base: usize,
646        len: usize,
647        mut decommit: impl FnMut(*mut u8, usize),
648    ) {
649        assert!(base + len <= self.accessible);
650        if len == 0 {
651            return;
652        }
653
654        assert_eq!(
655            vm::decommit_behavior(),
656            DecommitBehavior::RestoreOriginalMapping
657        );
658        decommit(self.base.as_ptr().add(base), len);
659    }
660
661    fn set_protection(&self, range: Range<usize>, readwrite: bool) -> Result<()> {
662        assert!(range.start <= range.end);
663        assert!(range.end <= self.static_size);
664        if range.len() == 0 {
665            return Ok(());
666        }
667
668        unsafe {
669            let start = self.base.as_ptr().add(range.start);
670            if readwrite {
671                vm::expose_existing_mapping(start, range.len()).err2anyhow()?;
672            } else {
673                vm::hide_existing_mapping(start, range.len()).err2anyhow()?;
674            }
675        }
676
677        Ok(())
678    }
679
680    pub(crate) fn has_image(&self) -> bool {
681        self.image.is_some()
682    }
683
684    #[allow(dead_code)] // ignore warnings as this is only used in some cfgs
685    pub(crate) fn is_dirty(&self) -> bool {
686        self.dirty
687    }
688
689    /// Map anonymous zeroed memory across the whole slot,
690    /// inaccessible. Used both during instantiate and during drop.
691    fn reset_with_anon_memory(&mut self) -> Result<()> {
692        if self.static_size == 0 {
693            assert!(self.image.is_none());
694            assert_eq!(self.accessible, 0);
695            return Ok(());
696        }
697
698        unsafe {
699            vm::erase_existing_mapping(self.base.as_ptr(), self.static_size).err2anyhow()?;
700        }
701
702        self.image = None;
703        self.accessible = 0;
704
705        Ok(())
706    }
707}
708
709impl Drop for MemoryImageSlot {
710    fn drop(&mut self) {
711        // The MemoryImageSlot may be dropped if there is an error during
712        // instantiation: for example, if a memory-growth limiter
713        // disallows a guest from having a memory of a certain size,
714        // after we've already initialized the MemoryImageSlot.
715        //
716        // We need to return this region of the large pool mmap to a
717        // safe state (with no module-specific mappings). The
718        // MemoryImageSlot will not be returned to the MemoryPool, so a new
719        // MemoryImageSlot will be created and overwrite the mappings anyway
720        // on the slot's next use; but for safety and to avoid
721        // resource leaks it's better not to have stale mappings to a
722        // possibly-otherwise-dead module's image.
723        //
724        // To "wipe the slate clean", let's do a mmap of anonymous
725        // memory over the whole region, with PROT_NONE. Note that we
726        // *can't* simply munmap, because that leaves a hole in the
727        // middle of the pooling allocator's big memory area that some
728        // other random mmap may swoop in and take, to be trampled
729        // over by the next MemoryImageSlot later.
730        //
731        // Since we're in drop(), we can't sanely return an error if
732        // this mmap fails. Instead though the result is unwrapped here to
733        // trigger a panic if something goes wrong. Otherwise if this
734        // reset-the-mapping fails then on reuse it might be possible, depending
735        // on precisely where errors happened, that stale memory could get
736        // leaked through.
737        //
738        // The exception to all of this is if the `clear_on_drop` flag
739        // (which is set by default) is false. If so, the owner of
740        // this MemoryImageSlot has indicated that it will clean up in some
741        // other way.
742        if self.clear_on_drop {
743            self.reset_with_anon_memory().unwrap();
744        }
745    }
746}

938        let plan = dummy_memory_plan(MemoryStyle::Dynamic { reserve: 200 });

946        // Instantiate the image and test that memory remains accessible after
947        // it's cleared.
948        memfd.instantiate(initial, Some(&image), &plan).unwrap();
949        assert!(memfd.has_image());
950        let slice = unsafe { mmap.slice_mut(0..(64 << 10) + page_size) };
951        assert_eq!(&[1, 2, 3, 4], &slice[page_size..][..4]);
952        slice[page_size] = 5;
953        assert_eq!(&[5, 2, 3, 4], &slice[page_size..][..4]);
954        memfd
955            .clear_and_remain_ready(0, |ptr, len| unsafe { decommit_pages(ptr, len).unwrap() })
956            .unwrap();
957        assert_eq!(&[1, 2, 3, 4], &slice[page_size..][..4]);
958
959        // Re-instantiate make sure it preserves memory. Grow a bit and set data
960        // beyond the initial size.
961        memfd.instantiate(initial, Some(&image), &plan).unwrap();
962        assert_eq!(&[1, 2, 3, 4], &slice[page_size..][..4]);
963        memfd.set_heap_limit(initial * 2).unwrap();
964        assert_eq!(&[0, 0], &slice[initial..initial + 2]);
965        slice[initial] = 100;
966        assert_eq!(&[100, 0], &slice[initial..initial + 2]);
967        memfd
968            .clear_and_remain_ready(0, |ptr, len| unsafe { decommit_pages(ptr, len).unwrap() })
969            .unwrap();
970
971        // Test that memory is still accessible, but it's been reset
972        assert_eq!(&[0, 0], &slice[initial..initial + 2]);
973
974        // Instantiate again, and again memory beyond the initial size should
975        // still be accessible. Grow into it again and make sure it works.
976        memfd.instantiate(initial, Some(&image), &plan).unwrap();
977        assert_eq!(&[0, 0], &slice[initial..initial + 2]);
978        memfd.set_heap_limit(initial * 2).unwrap();
979        assert_eq!(&[0, 0], &slice[initial..initial + 2]);
980        slice[initial] = 100;
981        assert_eq!(&[100, 0], &slice[initial..initial + 2]);
982        memfd
983            .clear_and_remain_ready(0, |ptr, len| unsafe { decommit_pages(ptr, len).unwrap() })
984            .unwrap();

987        memfd.instantiate(64 << 10, None, &plan).unwrap();
988        assert!(!memfd.has_image());
989        assert_eq!(&[0, 0, 0, 0], &slice[page_size..][..4]);
