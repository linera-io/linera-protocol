9/// The common header for all objects allocated in a GC heap.
10///
11/// This header is shared across all collectors, although particular collectors
12/// may always add their own trailing fields to this header for all of their own
13/// GC objects.
14///
15/// This is a bit-packed structure that logically has the following fields:
16///
17/// ```ignore
18/// struct VMGcHeader {
19///     // Highest 2 bits.
20///     kind: VMGcKind,
21///
22///     // 30 bits available for the `GcRuntime` to make use of however it sees fit.
23///     reserved: u30,
24///
25///     // The `VMSharedTypeIndex` for this GC object, if it isn't an
26///     // `externref` (or an `externref` re-wrapped as an `anyref`). `None` is
27///     // represented with `VMSharedTypeIndex::reserved_value()`.
28///     ty: Option<VMSharedTypeIndex>,
29/// }
30/// ```
31#[repr(align(8))]
32pub struct VMGcHeader(u64);
33
34unsafe impl GcHeapObject for VMGcHeader {
35    #[inline]
36    fn is(_: &VMGcHeader) -> bool {
37        true
38    }
39}

41const _: () = {
42    assert!((VMGcHeader::HEADER_SIZE as usize) == core::mem::size_of::<VMGcHeader>());
43    assert!((VMGcHeader::HEADER_ALIGN as usize) == core::mem::align_of::<VMGcHeader>());
44};

46impl VMGcHeader {
47    /// The size of this type on all architectures.
48    pub const HEADER_SIZE: u32 = 8;
49
50    /// The alignment of this type on all architectures.
51    pub const HEADER_ALIGN: u32 = 8;
52
53    /// Create the header for an `externref`.
54    pub fn externref() -> Self {
55        Self::from_kind_and_index(VMGcKind::ExternRef, VMSharedTypeIndex::reserved_value())
56    }
57
58    /// Create the header for the given kind and type index.
59    pub fn from_kind_and_index(kind: VMGcKind, index: VMSharedTypeIndex) -> Self {
60        let upper = kind as u32;
61        let upper = u64::from(upper) << 32;
62        let lower = u64::from(index.bits());
63        Self(upper | lower)
64    }
65
66    /// Get the kind of GC object that this is.
67    pub fn kind(&self) -> VMGcKind {
68        let upper = u32::try_from(self.0 >> 32).unwrap();
69        VMGcKind::from_high_bits_of_u32(upper)
70    }
71
72    /// Get the reserved 26 bits in this header.
73    ///
74    /// These are bits are reserved for `GcRuntime` implementations to make use
75    /// of however they see fit.
76    pub fn reserved_u26(&self) -> u32 {
77        let upper = u32::try_from(self.0 >> 32).unwrap();
78        upper & VMGcKind::UNUSED_MASK
79    }
80
81    /// Set the 26-bit reserved value.
82    ///
83    /// # Panics
84    ///
85    /// Panics if the given `value` has any of the upper 6 bits set.
86    pub fn set_reserved_u26(&mut self, value: u32) {
87        assert_eq!(
88            value & VMGcKind::MASK,
89            0,
90            "VMGcHeader::set_reserved_u26 with value using more than 26 bits"
91        );
92        self.0 |= u64::from(value) << 32;
93    }
94
95    /// Set the 26-bit reserved value.
96    ///
97    /// # Safety
98    ///
99    /// The given `value` must only use the lower 26 bits; its upper 6 bits must
100    /// be unset.
101    pub unsafe fn unchecked_set_reserved_u26(&mut self, value: u32) {
102        debug_assert_eq!(value & VMGcKind::MASK, 0);
103        self.0 |= u64::from(value) << 32;
104    }
105
106    /// Get this object's specific concrete type.
107    pub fn ty(&self) -> Option<VMSharedTypeIndex> {
108        let lower_mask = u64::from(u32::MAX);
109        let lower = u32::try_from(self.0 & lower_mask).unwrap();
110        if lower == u32::MAX {
111            None
112        } else {
113            Some(VMSharedTypeIndex::new(lower))
114        }
115    }
116}

130/// A raw, unrooted GC reference.
131///
132/// A `VMGcRef` is either:
133///
134/// * A reference to some kind of object on the GC heap, but we don't know
135///   exactly which kind without further reflection. Furthermore, this is not
136///   actually a pointer, but a compact index into a Wasm GC heap.
137///
138/// * An `i31ref`: it doesn't actually reference an object in the GC heap, but
139///   is instead an inline, unboxed 31-bit integer.
140///
141/// ## `VMGcRef` and GC Barriers
142///
143/// Depending on the garbage collector in use, cloning, writing, and dropping a
144/// `VMGcRef` may require invoking GC barriers (little snippets of code provided
145/// by the collector to ensure it is correctly tracking all GC references).
146///
147/// Therefore, to encourage correct usage of GC barriers, this type does *NOT*
148/// implement `Clone` or `Copy`. Use `GcStore::clone_gc_ref`,
149/// `GcStore::write_gc_ref`, and `GcStore::drop_gc_ref` to clone, write, and
150/// drop `VMGcRef`s respectively.
151///
152/// As an escape hatch, if you really need to copy a `VMGcRef` without invoking
153/// GC barriers and you understand why that will not lead to GC bugs in this
154/// particular case, you can use the `unchecked_copy` method.
155#[derive(Debug, PartialEq, Eq, Hash)]
156#[repr(transparent)]
157pub struct VMGcRef(NonZeroU32);
158
159impl<T> From<TypedGcRef<T>> for VMGcRef {
160    #[inline]
161    fn from(value: TypedGcRef<T>) -> Self {
162        value.gc_ref
163    }
164}

184impl VMGcRef {
185    /// If this bit is set on a GC reference, then the GC reference is actually an
186    /// unboxed `i31`.
187    ///
188    /// Must be kept in sync with `wasmtime_cranelift::I31_REF_DISCRIMINANT`.
189    pub const I31_REF_DISCRIMINANT: u32 = 1;
190
191    /// Create a new `VMGcRef` from the given raw u32 value.
192    ///
193    /// Does not discriminate between indices into a GC heap and `i31ref`s.
194    ///
195    /// Returns `None` for zero values.
196    ///
197    /// The given index should point to a valid GC-managed object within this
198    /// reference's associated heap. Failure to uphold this will be memory safe,
199    /// but will lead to general failures such as panics or incorrect results.
200    pub fn from_raw_u32(raw: u32) -> Option<Self> {
201        Some(Self::from_raw_non_zero_u32(NonZeroU32::new(raw)?))
202    }
203
204    /// Create a new `VMGcRef` from the given index into a GC heap.
205    ///
206    /// The given index should point to a valid GC-managed object within this
207    /// reference's associated heap. Failure to uphold this will be memory safe,
208    /// but will lead to general failures such as panics or incorrect results.
209    ///
210    /// Returns `None` when the index is not 2-byte aligned and therefore
211    /// conflicts with the `i31ref` discriminant.
212    pub fn from_heap_index(index: NonZeroU32) -> Option<Self> {
213        if (index.get() & Self::I31_REF_DISCRIMINANT) == 0 {
214            Some(Self::from_raw_non_zero_u32(index))
215        } else {
216            None
217        }
218    }
219
220    /// Create a new `VMGcRef` from the given raw value.
221    ///
222    /// Does not discriminate between indices into a GC heap and `i31ref`s.
223    pub fn from_raw_non_zero_u32(raw: NonZeroU32) -> Self {
224        VMGcRef(raw)
225    }
226
227    /// Construct a new `VMGcRef` from an unboxed 31-bit integer.
228    #[inline]
229    pub fn from_i31(val: I31) -> Self {
230        let val = (val.get_u32() << 1) | Self::I31_REF_DISCRIMINANT;
231        debug_assert_ne!(val, 0);
232        let non_zero = unsafe { NonZeroU32::new_unchecked(val) };
233        VMGcRef::from_raw_non_zero_u32(non_zero)
234    }
235
236    /// Copy this `VMGcRef` without running the GC's clone barriers.
237    ///
238    /// Prefer calling `clone(&mut GcStore)` instead! This is mostly an internal
239    /// escape hatch for collector implementations.
240    ///
241    /// Failure to run GC barriers when they would otherwise be necessary can
242    /// lead to leaks, panics, and wrong results. It cannot lead to memory
243    /// unsafety, however.
244    pub fn unchecked_copy(&self) -> Self {
245        VMGcRef(self.0)
246    }
247
248    /// Get this GC reference as a u32 index into its GC heap.
249    ///
250    /// Returns `None` for `i31ref`s.
251    pub fn as_heap_index(&self) -> Option<NonZeroU32> {
252        if self.is_i31() {
253            None
254        } else {
255            Some(self.0)
256        }
257    }
258
259    /// Get this GC reference as a raw u32 value, regardless whether it is
260    /// actually a reference to a GC object or is an `i31ref`.
261    pub fn as_raw_u32(&self) -> u32 {
262        self.0.get()
263    }
264
265    /// Creates a typed GC reference from `self`, checking that `self` actually
266    /// is a `T`.
267    ///
268    /// If this is not a GC reference to a `T`, then `Err(self)` is returned.
269    pub fn into_typed<T>(self, gc_heap: &impl GcHeap) -> Result<TypedGcRef<T>, Self>
270    where
271        T: GcHeapObject,
272    {
273        if self.is_i31() {
274            return Err(self);
275        }
276        if T::is(gc_heap.header(&self)) {
277            Ok(TypedGcRef {
278                gc_ref: self,
279                _phantom: marker::PhantomData,
280            })
281        } else {
282            Err(self)
283        }
284    }
285
286    /// Creates a typed GC reference without actually checking that `self` is a
287    /// `T`.
288    ///
289    /// `self` should point to a `T` object. Failure to uphold this invariant is
290    /// memory safe, but will lead to general incorrectness such as panics or
291    /// wrong results.
292    pub fn into_typed_unchecked<T>(self) -> TypedGcRef<T>
293    where
294        T: GcHeapObject,
295    {
296        debug_assert!(!self.is_i31());
297        TypedGcRef {
298            gc_ref: self,
299            _phantom: marker::PhantomData,
300        }
301    }
302
303    /// Is this GC reference pointing to a `T`?
304    pub fn is_typed<T>(&self, gc_heap: &impl GcHeap) -> bool
305    where
306        T: GcHeapObject,
307    {
308        if self.is_i31() {
309            return false;
310        }
311        T::is(gc_heap.header(&self))
312    }
313
314    /// Borrow `self` as a typed GC reference, checking that `self` actually is
315    /// a `T`.
316    pub fn as_typed<T>(&self, gc_heap: &impl GcHeap) -> Option<&TypedGcRef<T>>
317    where
318        T: GcHeapObject,
319    {
320        if self.is_i31() {
321            return None;
322        }
323        if T::is(gc_heap.header(&self)) {
324            let ptr = self as *const VMGcRef;
325            let ret = unsafe { &*ptr.cast() };
326            assert!(matches!(
327                ret,
328                TypedGcRef {
329                    gc_ref: VMGcRef(_),
330                    _phantom
331                }
332            ));
333            Some(ret)
334        } else {
335            None
336        }
337    }
338
339    /// Creates a typed GC reference without actually checking that `self` is a
340    /// `T`.
341    ///
342    /// `self` should point to a `T` object. Failure to uphold this invariant is
343    /// memory safe, but will lead to general incorrectness such as panics or
344    /// wrong results.
345    pub fn as_typed_unchecked<T>(&self) -> &TypedGcRef<T>
346    where
347        T: GcHeapObject,
348    {
349        debug_assert!(!self.is_i31());
350        let ptr = self as *const VMGcRef;
351        let ret = unsafe { &*ptr.cast() };
352        assert!(matches!(
353            ret,
354            TypedGcRef {
355                gc_ref: VMGcRef(_),
356                _phantom
357            }
358        ));
359        ret
360    }
361
362    /// Get a reference to the GC header that this GC reference is pointing to.
363    ///
364    /// Returns `None` when this is an `i31ref` and doesn't actually point to a
365    /// GC header.
366    pub fn gc_header<'a>(&self, gc_heap: &'a (impl GcHeap + ?Sized)) -> Option<&'a VMGcHeader> {
367        if self.is_i31() {
368            None
369        } else {
370            Some(gc_heap.header(self))
371        }
372    }
373
374    /// Is this `VMGcRef` actually an unboxed 31-bit integer, and not actually a
375    /// GC reference?
376    #[inline]
377    pub fn is_i31(&self) -> bool {
378        let val = self.0.get();
379        (val & Self::I31_REF_DISCRIMINANT) != 0
380    }
381
382    /// Get the underlying `i31`, if any.
383    #[inline]
384    pub fn as_i31(&self) -> Option<I31> {
385        if self.is_i31() {
386            let val = self.0.get();
387            Some(I31::wrapping_u32(val >> 1))
388        } else {
389            None
390        }
391    }
392
393    /// Get the underlying `i31`, panicking if this is not an `i31`.
394    #[inline]
395    pub fn unwrap_i31(&self) -> I31 {
396        self.as_i31().unwrap()
397    }
398
399    /// Is this `VMGcRef` a `VMExternRef`?
400    #[inline]
401    pub fn is_extern_ref(&self, gc_heap: &(impl GcHeap + ?Sized)) -> bool {
402        self.gc_header(gc_heap)
403            .map_or(false, |h| h.kind().matches(VMGcKind::ExternRef))
404    }
405}

407/// A trait implemented by all objects allocated inside a GC heap.
408///
409/// # Safety
410///
411/// All implementations must:
412///
413/// * Be `repr(C)` or `repr(transparent)`
414///
415/// * Begin with a `VMGcHeader` as their first field
416///
417/// * Not have `Drop` implementations (aka, `std::mem::needs_drop::<Self>()`
418///   should return `false`).
419///
420/// * Be memory safe to transmute to from an arbitrary byte sequence (that is,
421///   it is okay if some bit patterns are invalid with regards to correctness,
422///   so long as these invalid bit patterns cannot lead to memory unsafety).
423pub unsafe trait GcHeapObject: Send + Sync {
424    /// Check whether the GC object with the given header is an instance of
425    /// `Self`.
426    fn is(header: &VMGcHeader) -> bool;
427}
428
429/// A GC reference to a heap object of concrete type `T`.
430///
431/// Create typed GC refs via `VMGcRef::into_typed` and `VMGcRef::as_typed`, as
432/// well as via their unchecked equivalents `VMGcRef::into_typed_unchecked` and
433/// `VMGcRef::as_typed_unchecked`.
434#[derive(Debug, PartialEq, Eq, Hash)]
435#[repr(transparent)]
436pub struct TypedGcRef<T> {
437    gc_ref: VMGcRef,
438    _phantom: marker::PhantomData<*mut T>,
439}
440
441impl<T> TypedGcRef<T>
442where
443    T: GcHeapObject,
444{
445    /// Clone this `VMGcRef`, running any GC barriers as necessary.
446    pub fn clone(&self, gc_store: &mut GcStore) -> Self {
447        Self {
448            gc_ref: gc_store.clone_gc_ref(&self.gc_ref),
449            _phantom: marker::PhantomData,
450        }
451    }
452
453    /// Explicitly drop this GC reference, running any GC barriers as necessary.
454    pub fn drop(self, gc_store: &mut GcStore) {
455        gc_store.drop_gc_ref(self.gc_ref);
456    }
457
458    /// Copy this GC reference without running the GC's clone barriers.
459    ///
460    /// Prefer calling `clone(&mut GcStore)` instead! This is mostly an internal
461    /// escape hatch for collector implementations.
462    ///
463    /// Failure to run GC barriers when they would otherwise be necessary can
464    /// lead to leaks, panics, and wrong results. It cannot lead to memory
465    /// unsafety, however.
466    pub fn unchecked_copy(&self) -> Self {
467        Self {
468            gc_ref: self.gc_ref.unchecked_copy(),
469            _phantom: marker::PhantomData,
470        }
471    }
472}
473
474impl<T> TypedGcRef<T> {
475    /// Get the untyped version of this GC reference.
476    pub fn as_untyped(&self) -> &VMGcRef {
477        &self.gc_ref
478    }
