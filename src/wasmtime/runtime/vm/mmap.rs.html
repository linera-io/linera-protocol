1//! Low-level abstraction for allocating and managing zero-filled pages
2//! of memory.
3
4use crate::runtime::vm::sys::mmap;
5use crate::{prelude::*, vm::usize_is_multiple_of_host_page_size};
6use core::ops::Range;
7#[cfg(feature = "std")]
8use std::{fs::File, path::Path, sync::Arc};
9
10/// A simple struct consisting of a page-aligned pointer to page-aligned
11/// and initially-zeroed memory and a length.
12#[derive(Debug)]
13pub struct Mmap {
14    sys: mmap::Mmap,
15    #[cfg(feature = "std")]
16    file: Option<Arc<File>>,
17}
18
19impl Mmap {
20    /// Create a new `Mmap` pointing to at least `size` bytes of page-aligned
21    /// accessible memory.
22    pub fn with_at_least(size: usize) -> Result<Self> {
23        let rounded_size = crate::runtime::vm::round_usize_up_to_host_pages(size)?;
24        Self::accessible_reserved(rounded_size, rounded_size)
25    }
26
27    /// Creates a new `Mmap` by opening the file located at `path` and mapping
28    /// it into memory.
29    ///
30    /// The memory is mapped in read-only mode for the entire file. If portions
31    /// of the file need to be modified then the `region` crate can be use to
32    /// alter permissions of each page.
33    ///
34    /// The memory mapping and the length of the file within the mapping are
35    /// returned.
36    #[cfg(feature = "std")]
37    pub fn from_file(path: &Path) -> Result<Self> {
38        let (sys, file) = mmap::Mmap::from_file(path)?;
39        Ok(Mmap {
40            sys,
41            file: Some(Arc::new(file)),
42        })
43    }
44
45    /// Create a new `Mmap` pointing to `accessible_size` bytes of page-aligned
46    /// accessible memory, within a reserved mapping of `mapping_size` bytes.
47    /// `accessible_size` and `mapping_size` must be native page-size multiples.
48    ///
49    /// # Panics
50    ///
51    /// This function will panic if `accessible_size` is greater than
52    /// `mapping_size` or if either of them are not page-aligned.
53    pub fn accessible_reserved(accessible_size: usize, mapping_size: usize) -> Result<Self> {
54        assert!(accessible_size <= mapping_size);
55        assert!(usize_is_multiple_of_host_page_size(mapping_size));
56        assert!(usize_is_multiple_of_host_page_size(accessible_size));
57
58        if mapping_size == 0 {
59            Ok(Mmap {
60                sys: mmap::Mmap::new_empty(),
61                #[cfg(feature = "std")]
62                file: None,
63            })
64        } else if accessible_size == mapping_size {
65            Ok(Mmap {
66                sys: mmap::Mmap::new(mapping_size)
67                    .context(format!("mmap failed to allocate {mapping_size:#x} bytes"))?,
68                #[cfg(feature = "std")]
69                file: None,
70            })
71        } else {
72            let mut result = Mmap {
73                sys: mmap::Mmap::reserve(mapping_size)
74                    .context(format!("mmap failed to reserve {mapping_size:#x} bytes"))?,
75                #[cfg(feature = "std")]
76                file: None,
77            };
78            if accessible_size > 0 {
79                result.make_accessible(0, accessible_size).context(format!(
80                    "mmap failed to allocate {accessible_size:#x} bytes"
81                ))?;
82            }
83            Ok(result)
84        }
85    }
86
87    /// Make the memory starting at `start` and extending for `len` bytes
88    /// accessible. `start` and `len` must be native page-size multiples and
89    /// describe a range within `self`'s reserved memory.
90    ///
91    /// # Panics
92    ///
93    /// This function will panic if `start` or `len` is not page aligned or if
94    /// either are outside the bounds of this mapping.
95    pub fn make_accessible(&mut self, start: usize, len: usize) -> Result<()> {
96        let page_size = crate::runtime::vm::host_page_size();
97        assert_eq!(start & (page_size - 1), 0);
98        assert_eq!(len & (page_size - 1), 0);
99        assert!(len <= self.len());
100        assert!(start <= self.len() - len);
101
102        self.sys.make_accessible(start, len)
103    }
104
105    /// Return the allocated memory as a slice of u8.
106    ///
107    /// # Safety
108    ///
109    /// The caller must ensure that the range of bytes is accessible to the
110    /// program and additionally has previously been initialized.
111    ///
112    /// # Panics
113    ///
114    /// Panics of the `range` provided is outside of the limits of this mmap.
115    #[inline]
116    pub unsafe fn slice(&self, range: Range<usize>) -> &[u8] {
117        assert!(range.start <= range.end);
118        assert!(range.end <= self.len());
119        core::slice::from_raw_parts(self.as_ptr().add(range.start), range.end - range.start)
120    }
121
122    /// Return the allocated memory as a mutable slice of u8.
123    ///
124    /// # Safety
125    ///
126    /// The caller must ensure that the range of bytes is accessible to the
127    /// program and additionally has previously been initialized.
128    ///
129    /// # Panics
130    ///
131    /// Panics of the `range` provided is outside of the limits of this mmap.
132    pub unsafe fn slice_mut(&mut self, range: Range<usize>) -> &mut [u8] {
133        assert!(range.start <= range.end);
134        assert!(range.end <= self.len());
135        core::slice::from_raw_parts_mut(self.as_mut_ptr().add(range.start), range.end - range.start)
136    }
137
138    /// Return the allocated memory as a pointer to u8.
139    #[inline]
140    pub fn as_ptr(&self) -> *const u8 {
141        self.sys.as_ptr()
142    }
143
144    /// Return the allocated memory as a mutable pointer to u8.
145    #[inline]
146    pub fn as_mut_ptr(&mut self) -> *mut u8 {
147        self.sys.as_mut_ptr()
148    }
149
150    /// Return the length of the allocated memory.
151    ///
152    /// This is the byte length of this entire mapping which includes both
153    /// addressable and non-addressable memory.
154    #[inline]
155    pub fn len(&self) -> usize {
156        self.sys.len()
157    }
158
159    /// Return whether any memory has been allocated or reserved.
160    pub fn is_empty(&self) -> bool {
161        self.len() == 0
162    }
163
164    /// Makes the specified `range` within this `Mmap` to be read/execute.
165    ///
166    /// # Unsafety
167    ///
168    /// This method is unsafe as it's generally not valid to simply make memory
169    /// executable, so it's up to the caller to ensure that everything is in
170    /// order and this doesn't overlap with other memory that should only be
171    /// read or only read/write.
172    ///
173    /// # Panics
174    ///
175    /// Panics of `range` is out-of-bounds or not page-aligned.
176    pub unsafe fn make_executable(
177        &self,
178        range: Range<usize>,
179        enable_branch_protection: bool,
180    ) -> Result<()> {
181        assert!(range.start <= self.len());
182        assert!(range.end <= self.len());
183        assert!(range.start <= range.end);
184        assert!(
185            range.start % crate::runtime::vm::host_page_size() == 0,
186            "changing of protections isn't page-aligned",
187        );
188        self.sys
189            .make_executable(range, enable_branch_protection)
190            .context("failed to make memory executable")
191    }
192
193    /// Makes the specified `range` within this `Mmap` to be readonly.
194    pub unsafe fn make_readonly(&self, range: Range<usize>) -> Result<()> {
195        assert!(range.start <= self.len());
196        assert!(range.end <= self.len());
197        assert!(range.start <= range.end);
198        assert!(
199            range.start % crate::runtime::vm::host_page_size() == 0,
200            "changing of protections isn't page-aligned",
201        );
202        self.sys
203            .make_readonly(range)
204            .context("failed to make memory readonly")
205    }
206
207    /// Returns the underlying file that this mmap is mapping, if present.
208    #[cfg(feature = "std")]
209    pub fn original_file(&self) -> Option<&Arc<File>> {
210        self.file.as_ref()
211    }
212}
