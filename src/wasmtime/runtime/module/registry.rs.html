15/// Used for registering modules with a store.
16///
17/// Note that the primary reason for this registry is to ensure that everything
18/// in `Module` is kept alive for the duration of a `Store`. At this time we
19/// need "basically everything" within a `Module` to stay alive once it's
20/// instantiated within a store. While there's some smaller portions that could
21/// theoretically be omitted as they're not needed by the store they're
22/// currently small enough to not worry much about.
23#[derive(Default)]
24pub struct ModuleRegistry {
25    // Keyed by the end address of a `CodeObject`.
26    //
27    // The value here is the start address and the information about what's
28    // loaded at that address.
29    loaded_code: BTreeMap<usize, (usize, LoadedCode)>,
30
31    // Preserved for keeping data segments alive or similar
32    modules_without_code: Vec<Module>,
33}
34
35struct LoadedCode {
36    /// Kept alive here in the store to have a strong reference to keep the
37    /// relevant code mapped while the store is alive.
38    _code: Arc<CodeObject>,
39
40    /// Modules found within `self.code`, keyed by start address here of the
41    /// address of the first function in the module.
42    modules: BTreeMap<usize, Module>,
43}
44
45/// An identifier of a module that has previously been inserted into a
46/// `ModuleRegistry`.
47#[derive(Clone, Copy)]
48pub enum RegisteredModuleId {
49    /// Index into `ModuleRegistry::modules_without_code`.
50    WithoutCode(usize),
51    /// Start address of the module's code so that we can get it again via
52    /// `ModuleRegistry::lookup_module`.
53    LoadedCode(usize),
54}
55
56impl ModuleRegistry {
57    /// Get a previously-registered module by id.
58    pub fn lookup_module_by_id(&self, id: RegisteredModuleId) -> Option<&Module> {
59        match id {
60            RegisteredModuleId::WithoutCode(idx) => self.modules_without_code.get(idx),
61            RegisteredModuleId::LoadedCode(pc) => {
62                let (module, _) = self.module_and_offset(pc)?;
63                Some(module)
64            }
65        }
66    }
67
68    /// Fetches information about a registered module given a program counter value.
69    pub fn lookup_module_info(&self, pc: usize) -> Option<&dyn crate::runtime::vm::ModuleInfo> {
70        let (module, _) = self.module_and_offset(pc)?;
71        Some(module.module_info())
72    }
73
74    fn code(&self, pc: usize) -> Option<(&LoadedCode, usize)> {
75        let (end, (start, code)) = self.loaded_code.range(pc..).next()?;
76        if pc < *start || *end < pc {
77            return None;
78        }
79        Some((code, pc - *start))
80    }
81
82    fn module_and_offset(&self, pc: usize) -> Option<(&Module, usize)> {
83        let (code, offset) = self.code(pc)?;
84        Some((code.module(pc)?, offset))
85    }
86
87    /// Gets an iterator over all modules in the registry.
88    pub fn all_modules(&self) -> impl Iterator<Item = &'_ Module> + '_ {
89        self.loaded_code
90            .values()
91            .flat_map(|(_, code)| code.modules.values())
92            .chain(self.modules_without_code.iter())
93    }
94
95    /// Registers a new module with the registry.
96    pub fn register_module(&mut self, module: &Module) -> RegisteredModuleId {
97        self.register(module.code_object(), Some(module)).unwrap()
98    }
99
100    #[cfg(feature = "component-model")]
101    pub fn register_component(&mut self, component: &Component) {
102        self.register(component.code_object(), None);
103    }
104
105    /// Registers a new module with the registry.
106    fn register(
107        &mut self,
108        code: &Arc<CodeObject>,
109        module: Option<&Module>,
110    ) -> Option<RegisteredModuleId> {
111        let text = code.code_memory().text();
112
113        // If there's not actually any functions in this module then we may
114        // still need to preserve it for its data segments. Instances of this
115        // module will hold a pointer to the data stored in the module itself,
116        // and for schemes that perform lazy initialization which could use the
117        // module in the future. For that reason we continue to register empty
118        // modules and retain them.
119        if text.is_empty() {
120            return module.map(|module| {
121                let id = RegisteredModuleId::WithoutCode(self.modules_without_code.len());
122                self.modules_without_code.push(module.clone());
123                id
124            });
125        }
126
127        // The module code range is exclusive for end, so make it inclusive as
128        // it may be a valid PC value
129        let start_addr = text.as_ptr() as usize;
130        let end_addr = start_addr + text.len() - 1;
131        let id = module.map(|_| RegisteredModuleId::LoadedCode(start_addr));
132
133        // If this module is already present in the registry then that means
134        // it's either an overlapping image, for example for two modules
135        // found within a component, or it's a second instantiation of the same
136        // module. Delegate to `push_module` to find out.
137        if let Some((other_start, prev)) = self.loaded_code.get_mut(&end_addr) {
138            assert_eq!(*other_start, start_addr);
139            if let Some(module) = module {
140                prev.push_module(module);
141            }
142            return id;
143        }
144
145        // Assert that this module's code doesn't collide with any other
146        // registered modules
147        if let Some((_, (prev_start, _))) = self.loaded_code.range(start_addr..).next() {
148            assert!(*prev_start > end_addr);
149        }
150        if let Some((prev_end, _)) = self.loaded_code.range(..=start_addr).next_back() {
151            assert!(*prev_end < start_addr);
152        }
153
154        let mut item = LoadedCode {
155            _code: code.clone(),
156            modules: Default::default(),
157        };
158        if let Some(module) = module {
159            item.push_module(module);
160        }
161        let prev = self.loaded_code.insert(end_addr, (start_addr, item));
162        assert!(prev.is_none());
163        id
164    }
165
166    /// Fetches frame information about a program counter in a backtrace.
167    ///
168    /// Returns an object if this `pc` is known to some previously registered
169    /// module, or returns `None` if no information can be found. The first
170    /// boolean returned indicates whether the original module has unparsed
171    /// debug information due to the compiler's configuration. The second
172    /// boolean indicates whether the engine used to compile this module is
173    /// using environment variables to control debuginfo parsing.
174    pub(crate) fn lookup_frame_info(&self, pc: usize) -> Option<(FrameInfo, &Module)> {
175        let (module, offset) = self.module_and_offset(pc)?;
176        let info = FrameInfo::new(module.clone(), offset)?;
177        Some((info, module))
178    }
179
180    pub fn wasm_to_array_trampoline(
181        &self,
182        sig: VMSharedTypeIndex,
183    ) -> Option<NonNull<VMWasmCallFunction>> {
184        // TODO: We are doing a linear search over each module. This is fine for
185        // now because we typically have very few modules per store (almost
186        // always one, in fact). If this linear search ever becomes a
187        // bottleneck, we could avoid it by incrementally and lazily building a
188        // `VMSharedSignatureIndex` to `SignatureIndex` map.
189        //
190        // See also the comment in `ModuleInner::wasm_to_native_trampoline`.
191        for (_, code) in self.loaded_code.values() {
192            for module in code.modules.values() {
193                if let Some(trampoline) = module.wasm_to_array_trampoline(sig) {
194                    return Some(trampoline);
195                }
196            }
197        }
198        None
199    }
200}

202impl LoadedCode {
203    fn push_module(&mut self, module: &Module) {
204        let func = match module.compiled_module().finished_functions().next() {
205            Some((_, func)) => func,
206            // There are no compiled functions in this module so there's no
207            // need to push onto `self.modules` which is only used for frame
208            // information lookup for a trap which only symbolicates defined
209            // functions.
210            None => return,
211        };
212        let start = func.as_ptr() as usize;
213
214        match self.modules.entry(start) {
215            // This module is already present, and it should be the same as
216            // `module`.
217            Entry::Occupied(m) => {
218                debug_assert!(Arc::ptr_eq(&module.inner, &m.get().inner));
219            }
220            // This module was not already present, so now it's time to insert.
221            Entry::Vacant(v) => {
222                v.insert(module.clone());
223            }
224        }
225    }
226
227    fn module(&self, pc: usize) -> Option<&Module> {
228        // The `modules` map is keyed on the start address of the first
229        // function in the module, so find the first module whose start address
230        // is less than the `pc`. That may be the wrong module but lookup
231        // within the module should fail in that case.
232        let (_start, module) = self.modules.range(..=pc).next_back()?;
233        Some(module)
234    }
235}

237// This is the global code registry that stores information for all loaded code
238// objects that are currently in use by any `Store` in the current process.
239//
240// The purpose of this map is to be called from signal handlers to determine
241// whether a program counter is a wasm trap or not. Specifically macOS has
242// no contextual information about the thread available, hence the necessity
243// for global state rather than using thread local state.
244//
245// This is similar to `ModuleRegistry` except that it has less information and
246// supports removal. Any time anything is registered with a `ModuleRegistry`
247// it is also automatically registered with the singleton global module
248// registry. When a `ModuleRegistry` is destroyed then all of its entries
249// are removed from the global registry.
250fn global_code() -> &'static RwLock<GlobalRegistry> {
251    static GLOBAL_CODE: OnceLock<RwLock<GlobalRegistry>> = OnceLock::new();
252    GLOBAL_CODE.get_or_init(Default::default)
253}
254
255type GlobalRegistry = BTreeMap<usize, (usize, Arc<CodeMemory>)>;
256
257/// Find which registered region of code contains the given program counter, and
258/// what offset that PC is within that module's code.
259pub fn lookup_code(pc: usize) -> Option<(Arc<CodeMemory>, usize)> {
260    let all_modules = global_code().read();
261    let (_end, (start, module)) = all_modules.range(pc..).next()?;
262    let text_offset = pc.checked_sub(*start)?;
263    Some((module.clone(), text_offset))
264}
265
266/// Registers a new region of code.
267///
268/// Must not have been previously registered and must be `unregister`'d to
269/// prevent leaking memory.
270///
271/// This is required to enable traps to work correctly since the signal handler
272/// will lookup in the `GLOBAL_CODE` list to determine which a particular pc
273/// is a trap or not.
274pub fn register_code(code: &Arc<CodeMemory>) {
275    let text = code.text();
276    if text.is_empty() {
277        return;
278    }
279    let start = text.as_ptr() as usize;
280    let end = start + text.len() - 1;
281    let prev = global_code().write().insert(end, (start, code.clone()));
282    assert!(prev.is_none());
283}
284
285/// Unregisters a code mmap from the global map.
286///
287/// Must have been previously registered with `register`.
288pub fn unregister_code(code: &Arc<CodeMemory>) {
289    let text = code.text();
290    if text.is_empty() {
291        return;
292    }
293    let end = (text.as_ptr() as usize) + text.len() - 1;
294    let code = global_code().write().remove(&end);
295    assert!(code.is_some());
296}

299#[cfg_attr(miri, ignore)]
300fn test_frame_info() -> Result<(), anyhow::Error> {
301    use crate::*;
302    let mut store = Store::<()>::default();
303    let module = Module::new(
304        store.engine(),
305        r#"
306            (module
307                (func (export "add") (param $x i32) (param $y i32) (result i32) (i32.add (local.get $x) (local.get $y)))
308                (func (export "sub") (param $x i32) (param $y i32) (result i32) (i32.sub (local.get $x) (local.get $y)))
309                (func (export "mul") (param $x i32) (param $y i32) (result i32) (i32.mul (local.get $x) (local.get $y)))
310                (func (export "div_s") (param $x i32) (param $y i32) (result i32) (i32.div_s (local.get $x) (local.get $y)))
311                (func (export "div_u") (param $x i32) (param $y i32) (result i32) (i32.div_u (local.get $x) (local.get $y)))
312                (func (export "rem_s") (param $x i32) (param $y i32) (result i32) (i32.rem_s (local.get $x) (local.get $y)))
313                (func (export "rem_u") (param $x i32) (param $y i32) (result i32) (i32.rem_u (local.get $x) (local.get $y)))
314            )
315         "#,
316    )?;
317    // Create an instance to ensure the frame information is registered.
318    Instance::new(&mut store, &module, &[])?;
319
320    for (i, alloc) in module.compiled_module().finished_functions() {
321        let (start, end) = {
322            let ptr = alloc.as_ptr();
323            let len = alloc.len();
324            (ptr as usize, ptr as usize + len)
325        };
326        for pc in start..end {
327            let (frame, _) = store
328                .as_context()
329                .0
330                .modules()
331                .lookup_frame_info(pc)
332                .unwrap();
333            assert!(
334                frame.func_index() == i.as_u32(),
335                "lookup of {:#x} returned {}, expected {}",
336                pc,
337                frame.func_index(),
338                i.as_u32()
339            );
340        }
341    }
342    Ok(())
