14/// The Ethereum [EIP-2718] Transaction Envelope, modified for OP Stack chains.
15///
16/// # Note:
17///
18/// This enum distinguishes between tagged and untagged legacy transactions, as
19/// the in-protocol merkle tree may commit to EITHER 0-prefixed or raw.
20/// Therefore we must ensure that encoding returns the precise byte-array that
21/// was decoded, preserving the presence or absence of the `TransactionType`
22/// flag.
23///
24/// [EIP-2718]: https://eips.ethereum.org/EIPS/eip-2718
25#[derive(Debug, Clone, TransactionEnvelope)]
26#[envelope(tx_type_name = OpTxType, typed = OpTypedTransaction, serde_cfg(feature = "serde"))]
27pub enum OpTxEnvelope {
28    /// An untagged [`TxLegacy`].
29    #[envelope(ty = 0)]
30    Legacy(Signed<TxLegacy>),
31    /// A [`TxEip2930`] tagged with type 1.
32    #[envelope(ty = 1)]
33    Eip2930(Signed<TxEip2930>),
34    /// A [`TxEip1559`] tagged with type 2.
35    #[envelope(ty = 2)]
36    Eip1559(Signed<TxEip1559>),
37    /// A [`TxEip7702`] tagged with type 4.
38    #[envelope(ty = 4)]
39    Eip7702(Signed<TxEip7702>),
40    /// A [`TxDeposit`] tagged with type 0x7E.
41    #[envelope(ty = 126)]
42    #[serde(serialize_with = "crate::serde_deposit_tx_rpc")]
43    Deposit(Sealed<TxDeposit>),
44}

123impl From<Signed<OpTypedTransaction>> for OpTxEnvelope {
124    fn from(value: Signed<OpTypedTransaction>) -> Self {
125        let (tx, sig, hash) = value.into_parts();
126        match tx {
127            OpTypedTransaction::Legacy(tx_legacy) => {
128                let tx = Signed::new_unchecked(tx_legacy, sig, hash);
129                Self::Legacy(tx)
130            }
131            OpTypedTransaction::Eip2930(tx_eip2930) => {
132                let tx = Signed::new_unchecked(tx_eip2930, sig, hash);
133                Self::Eip2930(tx)
134            }
135            OpTypedTransaction::Eip1559(tx_eip1559) => {
136                let tx = Signed::new_unchecked(tx_eip1559, sig, hash);
137                Self::Eip1559(tx)
138            }
139            OpTypedTransaction::Eip7702(tx_eip7702) => {
140                let tx = Signed::new_unchecked(tx_eip7702, sig, hash);
141                Self::Eip7702(tx)
142            }
143            OpTypedTransaction::Deposit(tx) => Self::Deposit(Sealed::new_unchecked(tx, hash)),
144        }
145    }
146}

330    /// Attempts to convert an ethereum [`TxEnvelope`] into the optimism variant.
331    ///
332    /// Returns the given envelope as error if [`OpTxEnvelope`] doesn't support the variant
333    /// (EIP-4844)
334    #[cfg(feature = "alloy-compat")]
335    pub fn try_from_any_envelope(
336        tx: alloy_network::AnyTxEnvelope,
337    ) -> Result<Self, alloy_network::AnyTxEnvelope> {
338        match tx.try_into_envelope() {
339            Ok(eth) => {
340                Self::try_from_eth_envelope(eth).map_err(alloy_network::AnyTxEnvelope::Ethereum)
341            }
342            Err(err) => match err.into_value() {
343                alloy_network::AnyTxEnvelope::Unknown(unknown) => {
344                    let Ok(deposit) = unknown.inner.clone().try_into() else {
345                        return Err(alloy_network::AnyTxEnvelope::Unknown(unknown));
346                    };
347                    Ok(Self::Deposit(Sealed::new_unchecked(deposit, unknown.hash)))
348                }
349                unsupported => Err(unsupported),
350            },
351        }
352    }
