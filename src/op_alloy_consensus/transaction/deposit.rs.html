14/// Deposit transactions, also known as deposits are initiated on L1, and executed on L2.
15#[derive(Debug, Clone, PartialEq, Eq, Hash, Default)]
16#[cfg_attr(feature = "arbitrary", derive(arbitrary::Arbitrary))]
17#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
18#[cfg_attr(feature = "serde", serde(rename_all = "camelCase"))]
19pub struct TxDeposit {
20    /// Hash that uniquely identifies the source of the deposit.
21    pub source_hash: B256,
22    /// The address of the sender account.
23    pub from: Address,
24    /// The address of the recipient account, or the null (zero-length) address if the deposited
25    /// transaction is a contract creation.
26    #[cfg_attr(feature = "serde", serde(default, skip_serializing_if = "TxKind::is_create"))]
27    pub to: TxKind,
28    /// The ETH value to mint on L2.
29    #[cfg_attr(feature = "serde", serde(default, with = "alloy_serde::quantity"))]
30    pub mint: u128,
31    ///  The ETH value to send to the recipient account.
32    pub value: U256,
33    /// The gas limit for the L2 transaction.
34    #[cfg_attr(feature = "serde", serde(with = "alloy_serde::quantity", rename = "gas"))]
35    pub gas_limit: u64,
36    /// Field indicating if this transaction is exempt from the L2 gas limit.
37    #[cfg_attr(
38        feature = "serde",
39        serde(
40            default,
41            with = "alloy_serde::quantity",
42            rename = "isSystemTx",
43            skip_serializing_if = "core::ops::Not::not"
44        )
45    )]
46    pub is_system_transaction: bool,
47    /// Input has two uses depending if transaction is Create or Call (if `to` field is None or
48    /// Some).
49    pub input: Bytes,
50}

52impl TxDeposit {
53    /// Decodes the inner [TxDeposit] fields from RLP bytes.
54    ///
55    /// NOTE: This assumes a RLP header has already been decoded, and _just_ decodes the following
56    /// RLP fields in the following order:
57    ///
58    /// - `source_hash`
59    /// - `from`
60    /// - `to`
61    /// - `mint`
62    /// - `value`
63    /// - `gas_limit`
64    /// - `is_system_transaction`
65    /// - `input`
66    pub fn rlp_decode_fields(buf: &mut &[u8]) -> alloy_rlp::Result<Self> {
67        Ok(Self {
68            source_hash: Decodable::decode(buf)?,
69            from: Decodable::decode(buf)?,
70            to: Decodable::decode(buf)?,
71            mint: Decodable::decode(buf)?,
72            value: Decodable::decode(buf)?,
73            gas_limit: Decodable::decode(buf)?,
74            is_system_transaction: Decodable::decode(buf)?,
75            input: Decodable::decode(buf)?,
76        })
77    }
78
79    /// Decodes the transaction from RLP bytes.
80    pub fn rlp_decode(buf: &mut &[u8]) -> alloy_rlp::Result<Self> {
81        let header = Header::decode(buf)?;
82        if !header.list {
83            return Err(alloy_rlp::Error::UnexpectedString);
84        }
85        let remaining = buf.len();
86
87        if header.payload_length > remaining {
88            return Err(alloy_rlp::Error::InputTooShort);
89        }
90
91        let this = Self::rlp_decode_fields(buf)?;
92
93        if buf.len() + header.payload_length != remaining {
94            return Err(alloy_rlp::Error::UnexpectedLength);
95        }
96
97        Ok(this)
98    }
99
100    /// Outputs the length of the transaction's fields, without a RLP header or length of the
101    /// eip155 fields.
102    pub(crate) fn rlp_encoded_fields_length(&self) -> usize {
103        self.source_hash.length()
104            + self.from.length()
105            + self.to.length()
106            + self.mint.length()
107            + self.value.length()
108            + self.gas_limit.length()
109            + self.is_system_transaction.length()
110            + self.input.0.length()
111    }
112
113    /// Encodes only the transaction's fields into the desired buffer, without a RLP header.
114    /// <https://github.com/ethereum-optimism/specs/blob/main/specs/protocol/deposits.md#the-deposited-transaction-type>
115    pub(crate) fn rlp_encode_fields(&self, out: &mut dyn alloy_rlp::BufMut) {
116        self.source_hash.encode(out);
117        self.from.encode(out);
118        self.to.encode(out);
119        self.mint.encode(out);
120        self.value.encode(out);
121        self.gas_limit.encode(out);
122        self.is_system_transaction.encode(out);
123        self.input.encode(out);
124    }
125
126    /// Calculates a heuristic for the in-memory size of the [TxDeposit] transaction.
127    #[inline]
128    pub fn size(&self) -> usize {
129        mem::size_of::<B256>() + // source_hash
130        mem::size_of::<Address>() + // from
131        self.to.size() + // to
132        mem::size_of::<u128>() + // mint
133        mem::size_of::<U256>() + // value
134        mem::size_of::<u128>() + // gas_limit
135        mem::size_of::<bool>() + // is_system_transaction
136        self.input.len() // input
137    }
138
139    /// Get the transaction type
140    pub(crate) const fn tx_type(&self) -> OpTxType {
141        OpTxType::Deposit
142    }
143
144    /// Create an rlp header for the transaction.
145    fn rlp_header(&self) -> Header {
146        Header { list: true, payload_length: self.rlp_encoded_fields_length() }
147    }
148
149    /// RLP encodes the transaction.
150    pub fn rlp_encode(&self, out: &mut dyn BufMut) {
151        self.rlp_header().encode(out);
152        self.rlp_encode_fields(out);
153    }
154
155    /// Get the length of the transaction when RLP encoded.
156    pub fn rlp_encoded_length(&self) -> usize {
157        self.rlp_header().length_with_payload()
158    }
159
160    /// Get the length of the transaction when EIP-2718 encoded. This is the
161    /// 1 byte type flag + the length of the RLP encoded transaction.
162    pub fn eip2718_encoded_length(&self) -> usize {
163        self.rlp_encoded_length() + 1
164    }
165
166    fn network_header(&self) -> Header {
167        Header { list: false, payload_length: self.eip2718_encoded_length() }
168    }
169
170    /// Get the length of the transaction when network encoded. This is the
171    /// EIP-2718 encoded length with an outer RLP header.
172    pub fn network_encoded_length(&self) -> usize {
173        self.network_header().length_with_payload()
174    }
175
176    /// Network encode the transaction with the given signature.
177    pub fn network_encode(&self, out: &mut dyn BufMut) {
178        self.network_header().encode(out);
179        self.encode_2718(out);
180    }
181
182    /// Calculate the transaction hash.
183    pub fn tx_hash(&self) -> TxHash {
184        let mut buf = Vec::with_capacity(self.eip2718_encoded_length());
185        self.encode_2718(&mut buf);
186        keccak256(&buf)
187    }
188
189    /// Returns the signature for the optimism deposit transactions, which don't include a
190    /// signature.
191    pub const fn signature() -> Signature {
192        Signature::new(U256::ZERO, U256::ZERO, false)
193    }
194}

293impl Decodable2718 for TxDeposit {
294    fn typed_decode(ty: u8, data: &mut &[u8]) -> Eip2718Result<Self> {
295        let ty: OpTxType = ty.try_into().map_err(|_| Eip2718Error::UnexpectedType(ty))?;
296        if ty != OpTxType::Deposit as u8 {
297            return Err(Eip2718Error::UnexpectedType(ty as u8));
298        }
299        let tx = Self::decode(data)?;
300        Ok(tx)
301    }
302
303    fn fallback_decode(data: &mut &[u8]) -> Eip2718Result<Self> {
304        let tx = Self::decode(data)?;
305        Ok(tx)
306    }
307}

358/// A trait representing a deposit transaction with specific attributes.
359pub trait DepositTransaction: Transaction {
360    /// Returns the hash that uniquely identifies the source of the deposit.
361    ///
362    /// # Returns
363    /// An `Option<B256>` containing the source hash if available.
364    fn source_hash(&self) -> Option<B256>;
365
366    /// Returns the optional mint value of the deposit transaction.
367    ///
368    /// # Returns
369    /// An `u128` representing the ETH value to mint on L2, if any.
370    fn mint(&self) -> u128;
371
372    /// Indicates whether the transaction is exempt from the L2 gas limit.
373    ///
374    /// # Returns
375    /// A `bool` indicating if the transaction is a system transaction.
376    fn is_system_transaction(&self) -> bool;
377}

396/// Deposit transactions don't have a signature, however, we include an empty signature in the
397/// response for better compatibility.
398///
399/// This function can be used as `serialize_with` serde attribute for the [`TxDeposit`] and will
400/// flatten [`TxDeposit::signature`] into response.
401#[cfg(feature = "serde")]
402pub fn serde_deposit_tx_rpc<T: serde::Serialize, S: serde::Serializer>(
403    value: &T,
404    serializer: S,
405) -> Result<S::Ok, S::Error> {
406    use serde::Serialize;
407
408    #[derive(Serialize)]
409    struct SerdeHelper<'a, T> {
410        #[serde(flatten)]
411        value: &'a T,
412        #[serde(flatten)]
413        signature: Signature,
414    }
415
416    SerdeHelper { value, signature: TxDeposit::signature() }.serialize(serializer)
417}

713            #[derive(Debug, PartialEq, Eq, Serialize, Deserialize)]

715                #[serde_as(as = "serde_bincode_compat::TxDeposit")]

721            let data = Data {

723                    .unwrap(),
724            };
