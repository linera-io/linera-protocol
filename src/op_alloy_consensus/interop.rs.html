11/// The safety level of a message.
12#[derive(Debug, Clone, Copy, PartialEq, Eq, Display)]
13#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
14#[cfg_attr(feature = "serde", serde(rename_all = "kebab-case"))]
15pub enum SafetyLevel {
16    /// The message is finalized.
17    Finalized,
18    /// The message is safe across chains.
19    #[cfg_attr(feature = "serde", serde(rename = "safe"))]
20    CrossSafe,
21    /// The message is safe locally.
22    LocalSafe,
23    /// The message is unsafe across chains.
24    CrossUnsafe,
25    /// The message is unsafe locally.
26    #[cfg_attr(feature = "serde", serde(rename = "unsafe"))]
27    LocalUnsafe,
28    /// The message is invalid.
29    Invalid,
30}
31
32impl FromStr for SafetyLevel {
33    type Err = SafetyLevelParseError;
34
35    fn from_str(s: &str) -> Result<Self, Self::Err> {
36        match s.to_lowercase().as_str() {
37            "finalized" => Ok(Self::Finalized),
38            "safe" => Ok(Self::CrossSafe),
39            "local-safe" | "localsafe" => Ok(Self::LocalSafe),
40            "cross-unsafe" | "crossunsafe" => Ok(Self::CrossUnsafe),
41            "unsafe" => Ok(Self::LocalUnsafe),
42            "invalid" => Ok(Self::Invalid),
43            _ => Err(SafetyLevelParseError(s.to_string())),
44        }
45    }
46}
47
48/// Error when parsing SafetyLevel from string.
49#[derive(thiserror::Error, Debug)]
50#[error("Invalid SafetyLevel, error: {0}")]
51pub struct SafetyLevelParseError(pub String);

53#[cfg(test)]
54mod tests {

56
57    #[test]
58    #[cfg(feature = "serde")]
59    fn test_safety_level_serde() {
60        let level = SafetyLevel::Finalized;
61        let json = serde_json::to_string(&level).unwrap();
62        assert_eq!(json, r#""finalized""#);
63
64        let level: SafetyLevel = serde_json::from_str(&json).unwrap();
65        assert_eq!(level, SafetyLevel::Finalized);
66    }
67
68    #[test]
69    #[cfg(feature = "serde")]
70    fn test_serde_safety_level_fails() {
71        let json = r#""failed""#;
72        let level: Result<SafetyLevel, _> = serde_json::from_str(json);
73        assert!(level.is_err());
74    }
75
76    #[test]
77    fn test_safety_level_from_str_valid() {
78        assert_eq!(SafetyLevel::from_str("finalized").unwrap(), SafetyLevel::Finalized);
79        assert_eq!(SafetyLevel::from_str("safe").unwrap(), SafetyLevel::CrossSafe);
80        assert_eq!(SafetyLevel::from_str("local-safe").unwrap(), SafetyLevel::LocalSafe);
81        assert_eq!(SafetyLevel::from_str("localsafe").unwrap(), SafetyLevel::LocalSafe);
82        assert_eq!(SafetyLevel::from_str("cross-unsafe").unwrap(), SafetyLevel::CrossUnsafe);
83        assert_eq!(SafetyLevel::from_str("crossunsafe").unwrap(), SafetyLevel::CrossUnsafe);
84        assert_eq!(SafetyLevel::from_str("unsafe").unwrap(), SafetyLevel::LocalUnsafe);
85        assert_eq!(SafetyLevel::from_str("invalid").unwrap(), SafetyLevel::Invalid);
86    }
