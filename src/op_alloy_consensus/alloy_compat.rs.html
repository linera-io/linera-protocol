3use crate::{DEPOSIT_TX_TYPE_ID, OpTxEnvelope, TxDeposit};

7use alloy_network::{AnyRpcTransaction, AnyTxEnvelope, UnknownTxEnvelope, UnknownTypedTransaction};
8use alloy_rpc_types_eth::{ConversionError, Transaction as AlloyRpcTransaction};
9use alloy_serde::WithOtherFields;

11impl TryFrom<UnknownTxEnvelope> for TxDeposit {
12    type Error = ConversionError;
13
14    fn try_from(value: UnknownTxEnvelope) -> Result<Self, Self::Error> {
15        value.inner.try_into()
16    }
17}
18
19impl TryFrom<UnknownTypedTransaction> for TxDeposit {
20    type Error = ConversionError;
21
22    fn try_from(value: UnknownTypedTransaction) -> Result<Self, Self::Error> {
23        if !value.is_type(DEPOSIT_TX_TYPE_ID) {
24            return Err(ConversionError::Custom("invalid transaction type".to_string()));
25        }
26        value
27            .fields
28            .deserialize_into()
29            .map_err(|_| ConversionError::Custom("invalid transaction data".to_string()))
30    }
31}
32
33impl TryFrom<AnyTxEnvelope> for OpTxEnvelope {
34    type Error = AnyTxEnvelope;
35
36    fn try_from(value: AnyTxEnvelope) -> Result<Self, Self::Error> {
37        Self::try_from_any_envelope(value)
38    }
39}
40
41impl TryFrom<AnyRpcTransaction> for OpTxEnvelope {
42    type Error = ConversionError;
43
44    fn try_from(tx: AnyRpcTransaction) -> Result<Self, Self::Error> {
45        let WithOtherFields { inner: AlloyRpcTransaction { inner, .. }, other: _ } = tx.0;
46
47        let from = inner.signer();
48        match inner.into_inner() {
49            AnyTxEnvelope::Ethereum(tx) => Self::try_from_eth_envelope(tx).map_err(|_| {
50                ConversionError::Custom("unable to convert from ethereum type".to_string())
51            }),
52            AnyTxEnvelope::Unknown(mut tx) => {
53                // Re-insert `from` field which was consumed by outer `Transaction`.
54                // Ref hack in op-alloy <https://github.com/alloy-rs/op-alloy/blob/7d50b698631dd73f8d20f9f60ee78cd0597dc278/crates/rpc-types/src/transaction.rs#L236-L237>
55                tx.inner
56                    .fields
57                    .insert_value("from".to_string(), from)
58                    .map_err(|err| ConversionError::Custom(err.to_string()))?;
59                Ok(Self::Deposit(Sealed::new(tx.try_into()?)))
60            }
61        }
62    }
63}
