6pub fn page_size() -> usize {

8}

10pub unsafe fn alloc(base: *const (), size: usize, protection: Protection) -> Result<*const ()> {
11  let mut native_prot = protection.to_native();
12
13  // This adjustment ensures that the behavior of memory allocation is
14  // orthogonal across all platforms by aligning NetBSD's protection flags and
15  // PaX behavior with those of other operating systems.
16  if cfg!(target_os = "netbsd") {
17    let max_protection = (PROT_READ | PROT_WRITE | PROT_EXEC) << 3;
18    native_prot |= max_protection;
19  }
20
21  let mut flags = MAP_PRIVATE | MAP_ANON;
22
23  if !base.is_null() {
24    flags |= MAP_FIXED;
25  }
26
27  #[cfg(all(target_vendor = "apple", target_arch = "aarch64"))]
28  if matches!(
29    protection,
30    Protection::WRITE_EXECUTE | Protection::READ_WRITE_EXECUTE
31  ) {
32    // On hardened context, MAP_JIT is necessary (on arm64) to allow W/X'ed regions.
33    flags |= libc::MAP_JIT;
34  }
35
36  match libc::mmap(base as *mut _, size, native_prot, flags, -1, 0) {
37    MAP_FAILED => Err(Error::SystemCall(io::Error::last_os_error())),
38    address => Ok(address as *const ()),
39  }
40}

42pub unsafe fn free(base: *const (), size: usize) -> Result<()> {
43  match libc::munmap(base as *mut _, size) {
44    0 => Ok(()),
45    _ => Err(Error::SystemCall(io::Error::last_os_error())),
46  }
47}

49pub unsafe fn protect(base: *const (), size: usize, protection: Protection) -> Result<()> {
50  match libc::mprotect(base as *mut _, size, protection.to_native()) {
51    0 => Ok(()),
52    _ => Err(Error::SystemCall(io::Error::last_os_error())),
53  }
54}

56pub fn lock(base: *const (), size: usize) -> Result<()> {

60  }

63pub fn unlock(base: *const (), size: usize) -> Result<()> {

70impl Protection {
71  fn to_native(self) -> libc::c_int {
72    // This is directly mapped to its native counterpart to allow users to
73    // include non-standard flags with `Protection::from_bits_unchecked`.
74    self.bits as libc::c_int
75  }
76}

78#[cfg(test)]
79mod tests {
80  use super::*;
81  use libc::PROT_NONE;
82
83  #[test]
84  fn protection_flags_match_unix_constants() {
85    assert_eq!(Protection::NONE.bits, PROT_NONE as usize);
86    assert_eq!(Protection::READ.bits, PROT_READ as usize);
87    assert_eq!(Protection::WRITE.bits, PROT_WRITE as usize);
88    assert_eq!(
89      Protection::READ_WRITE_EXECUTE,
90      Protection::from_bits_truncate((PROT_READ | PROT_WRITE | PROT_EXEC) as usize)
91    );
92  }
93
94  #[test]
95  fn protection_flags_are_mapped_to_native() {
96    let rwx = PROT_READ | PROT_WRITE | PROT_EXEC;
97
98    assert_eq!(Protection::NONE.to_native(), 0);
99    assert_eq!(Protection::READ.to_native(), PROT_READ);
100    assert_eq!(Protection::READ_WRITE.to_native(), PROT_READ | PROT_WRITE);
101    assert_eq!(Protection::READ_WRITE_EXECUTE.to_native(), rwx);
102  }
