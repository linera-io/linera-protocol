14/// The error type for the key-value stores.
15pub trait KeyValueStoreError:
16    std::error::Error + From<bcs::Error> + Debug + Send + Sync + 'static
17{
18    /// The name of the backend.
19    const BACKEND: &'static str;
20}
21
22impl<E: KeyValueStoreError> From<E> for ViewError {
23    fn from(error: E) -> Self {
24        Self::StoreError {
25            backend: E::BACKEND,
26            error: Box::new(error),
27        }
28    }
29}

31/// Define an associated [`KeyValueStoreError`].
32pub trait WithError {
33    /// The error type.
34    type Error: KeyValueStoreError;
35}

37/// Low-level, asynchronous read key-value operations. Useful for storage APIs not based on views.
38#[cfg_attr(not(web), trait_variant::make(Send + Sync))]
39pub trait ReadableKeyValueStore: WithError {
40    /// The maximal size of keys that can be stored.
41    const MAX_KEY_SIZE: usize;
42
43    /// Returns type for key search operations.
44    type Keys: KeyIterable<Self::Error>;
45
46    /// Returns type for key-value search operations.
47    type KeyValues: KeyValueIterable<Self::Error>;
48
49    /// Retrieve the number of stream queries.
50    fn max_stream_queries(&self) -> usize;
51
52    /// Retrieves a `Vec<u8>` from the database using the provided `key`.
53    async fn read_value_bytes(&self, key: &[u8]) -> Result<Option<Vec<u8>>, Self::Error>;
54
55    /// Tests whether a key exists in the database
56    async fn contains_key(&self, key: &[u8]) -> Result<bool, Self::Error>;
57
58    /// Tests whether a list of keys exist in the database
59    async fn contains_keys(&self, keys: Vec<Vec<u8>>) -> Result<Vec<bool>, Self::Error>;
60
61    /// Retrieves multiple `Vec<u8>` from the database using the provided `keys`.
62    async fn read_multi_values_bytes(
63        &self,
64        keys: Vec<Vec<u8>>,
65    ) -> Result<Vec<Option<Vec<u8>>>, Self::Error>;
66
67    /// Finds the `key` matching the prefix. The prefix is not included in the returned keys.
68    async fn find_keys_by_prefix(&self, key_prefix: &[u8]) -> Result<Self::Keys, Self::Error>;
69
70    /// Finds the `(key,value)` pairs matching the prefix. The prefix is not included in the returned keys.
71    async fn find_key_values_by_prefix(
72        &self,
73        key_prefix: &[u8],
74    ) -> Result<Self::KeyValues, Self::Error>;
75
76    // We can't use `async fn` here in the below implementations due to
77    // https://github.com/rust-lang/impl-trait-utils/issues/17, but once that bug is fixed
78    // we can revert them to `async fn` syntax, which is neater.
79
80    /// Reads a single `key` and deserializes the result if present.
81    fn read_value<V: DeserializeOwned>(
82        &self,
83        key: &[u8],
84    ) -> impl Future<Output = Result<Option<V>, Self::Error>> {
85        async { Ok(from_bytes_option(&self.read_value_bytes(key).await?)?) }
86    }
87
88    /// Reads multiple `keys` and deserializes the results if present.
89    fn read_multi_values<V: DeserializeOwned + Send + Sync>(
90        &self,
91        keys: Vec<Vec<u8>>,
92    ) -> impl Future<Output = Result<Vec<Option<V>>, Self::Error>> {
93        async {
94            let mut values = Vec::with_capacity(keys.len());
95            for entry in self.read_multi_values_bytes(keys).await? {
96                values.push(from_bytes_option(&entry)?);
97            }
98            Ok(values)
99        }
100    }
101}

103/// Low-level, asynchronous write key-value operations. Useful for storage APIs not based on views.
104#[cfg_attr(not(web), trait_variant::make(Send + Sync))]
105pub trait WritableKeyValueStore: WithError {
106    /// The maximal size of values that can be stored.
107    const MAX_VALUE_SIZE: usize;
108
109    /// Writes the `batch` in the database.
110    async fn write_batch(&self, batch: Batch) -> Result<(), Self::Error>;
111
112    /// Clears any journal entry that may remain.
113    /// The journal is located at the `root_key`.
114    async fn clear_journal(&self) -> Result<(), Self::Error>;
115}

117/// Low-level trait for the administration of stores and their namespaces.
118#[cfg_attr(not(web), trait_variant::make(Send + Sync))]
119pub trait AdminKeyValueStore: WithError + Sized {
120    /// The configuration needed to interact with a new store.
121    type Config: Send + Sync;
122    /// The name of this class of stores
123    fn get_name() -> String;
124
125    /// Connects to an existing namespace using the given configuration.
126    async fn connect(config: &Self::Config, namespace: &str) -> Result<Self, Self::Error>;
127
128    /// Opens the key partition starting at `root_key` and returns a clone of the
129    /// connection to work in this partition.
130    ///
131    /// IMPORTANT: It is assumed that the returned connection is the only user of the
132    /// partition (for both read and write) and will remain so until it is ended. Future
133    /// implementations of this method may fail if this is not the case.
134    fn open_exclusive(&self, root_key: &[u8]) -> Result<Self, Self::Error>;
135
136    /// Obtains the list of existing namespaces.
137    async fn list_all(config: &Self::Config) -> Result<Vec<String>, Self::Error>;
138
139    /// Lists the root keys of the namespace.
140    /// It is possible that some root keys have no keys.
141    async fn list_root_keys(
142        config: &Self::Config,
143        namespace: &str,
144    ) -> Result<Vec<Vec<u8>>, Self::Error>;
145
146    /// Deletes all the existing namespaces.
147    fn delete_all(config: &Self::Config) -> impl Future<Output = Result<(), Self::Error>> {
148        async {
149            let namespaces = Self::list_all(config).await?;
150            for namespace in namespaces {
151                Self::delete(config, &namespace).await?;
152            }
153            Ok(())
154        }
155    }
156
157    /// Tests if a given namespace exists.
158    async fn exists(config: &Self::Config, namespace: &str) -> Result<bool, Self::Error>;
159
160    /// Creates a namespace. Returns an error if the namespace exists.
161    async fn create(config: &Self::Config, namespace: &str) -> Result<(), Self::Error>;
162
163    /// Deletes the given namespace.
164    async fn delete(config: &Self::Config, namespace: &str) -> Result<(), Self::Error>;
165
166    /// Initializes a storage if missing and provides it.
167    fn maybe_create_and_connect(
168        config: &Self::Config,
169        namespace: &str,
170    ) -> impl Future<Output = Result<Self, Self::Error>> {
171        async {
172            if !Self::exists(config, namespace).await? {
173                Self::create(config, namespace).await?;
174            }
175            Self::connect(config, namespace).await
176        }
177    }
178
179    /// Creates a new storage. Overwrites it if this namespace already exists.
180    fn recreate_and_connect(
181        config: &Self::Config,
182        namespace: &str,
183    ) -> impl Future<Output = Result<Self, Self::Error>> {
184        async {
185            if Self::exists(config, namespace).await? {
186                Self::delete(config, namespace).await?;
187            }
188            Self::create(config, namespace).await?;
189            Self::connect(config, namespace).await
190        }
191    }
192}

225/// Iterates keys by reference in a vector of keys.
226/// Inspired by https://depth-first.com/articles/2020/06/22/returning-rust-iterators/
227pub struct SimpleKeyIterator<'a, E> {
228    iter: std::slice::Iter<'a, Vec<u8>>,
229    _error_type: std::marker::PhantomData<E>,
230}
231
232impl<'a, E> Iterator for SimpleKeyIterator<'a, E> {
233    type Item = Result<&'a [u8], E>;
234
235    fn next(&mut self) -> Option<Self::Item> {
236        self.iter.next().map(|key| Result::Ok(key.as_ref()))
237    }
238}
239
240impl<E> KeyIterable<E> for Vec<Vec<u8>> {
241    type Iterator<'a> = SimpleKeyIterator<'a, E>;
242
243    fn iterator(&self) -> Self::Iterator<'_> {
244        SimpleKeyIterator {
245            iter: self.iter(),
246            _error_type: std::marker::PhantomData,
247        }
248    }
249}

251#[doc(hidden)]
252/// Same as `SimpleKeyIterator` but for key-value pairs.
253pub struct SimpleKeyValueIterator<'a, E> {
254    iter: std::slice::Iter<'a, (Vec<u8>, Vec<u8>)>,
255    _error_type: std::marker::PhantomData<E>,
256}
257
258impl<'a, E> Iterator for SimpleKeyValueIterator<'a, E> {
259    type Item = Result<(&'a [u8], &'a [u8]), E>;
260
261    fn next(&mut self) -> Option<Self::Item> {
262        self.iter
263            .next()
264            .map(|entry| Ok((&entry.0[..], &entry.1[..])))
265    }
266}

268#[doc(hidden)]
269/// Same as `SimpleKeyValueIterator` but key-value pairs are passed by value.
270pub struct SimpleKeyValueIteratorOwned<E> {
271    iter: std::vec::IntoIter<(Vec<u8>, Vec<u8>)>,
272    _error_type: std::marker::PhantomData<E>,
273}
274
275impl<E> Iterator for SimpleKeyValueIteratorOwned<E> {
276    type Item = Result<(Vec<u8>, Vec<u8>), E>;
277
278    fn next(&mut self) -> Option<Self::Item> {
279        self.iter.next().map(Result::Ok)
280    }
281}
282
283impl<E> KeyValueIterable<E> for Vec<(Vec<u8>, Vec<u8>)> {
284    type Iterator<'a> = SimpleKeyValueIterator<'a, E>;
285    type IteratorOwned = SimpleKeyValueIteratorOwned<E>;
286
287    fn iterator(&self) -> Self::Iterator<'_> {
288        SimpleKeyValueIterator {
289            iter: self.iter(),
290            _error_type: std::marker::PhantomData,
291        }
292    }
293
294    fn into_iterator_owned(self) -> Self::IteratorOwned {
295        SimpleKeyValueIteratorOwned {
296            iter: self.into_iter(),
297            _error_type: std::marker::PhantomData,
298        }
299    }
300}

302/// How to iterate over the keys returned by a search query.
303pub trait KeyIterable<Error> {
304    /// The iterator returning keys by reference.
305    type Iterator<'a>: Iterator<Item = Result<&'a [u8], Error>>
306    where
307        Self: 'a;
308
309    /// Iterates keys by reference.
310    fn iterator(&self) -> Self::Iterator<'_>;
311}
312
313/// How to iterate over the key-value pairs returned by a search query.
314pub trait KeyValueIterable<Error> {
315    /// The iterator that returns key-value pairs by reference.
316    type Iterator<'a>: Iterator<Item = Result<(&'a [u8], &'a [u8]), Error>>
317    where
318        Self: 'a;
319
320    /// The iterator that returns key-value pairs by value.
321    type IteratorOwned: Iterator<Item = Result<(Vec<u8>, Vec<u8>), Error>>;
322
323    /// Iterates keys and values by reference.
324    fn iterator(&self) -> Self::Iterator<'_>;
325
326    /// Iterates keys and values by value.
327    fn into_iterator_owned(self) -> Self::IteratorOwned;
328}
