54/// A view gives exclusive access to read and write the data stored at an underlying
55/// address in storage.
56#[cfg_attr(not(web), trait_variant::make(Send + Sync))]
57pub trait View: Sized {
58    /// The number of keys used for the initialization
59    const NUM_INIT_KEYS: usize;
60
61    /// The type of context stored in this view
62    type Context: crate::context::Context;
63
64    /// Obtains a mutable reference to the internal context.
65    fn context(&self) -> &Self::Context;
66
67    /// Creates the keys needed for loading the view
68    fn pre_load(context: &Self::Context) -> Result<Vec<Vec<u8>>, ViewError>;
69
70    /// Loads a view from the values
71    fn post_load(context: Self::Context, values: &[Option<Vec<u8>>]) -> Result<Self, ViewError>;
72
73    /// Loads a view
74    fn load(context: Self::Context) -> impl Future<Output = Result<Self, ViewError>> {
75        async {
76            if Self::NUM_INIT_KEYS == 0 {
77                Self::post_load(context, &[])
78            } else {
79                use crate::{context::Context, store::ReadableKeyValueStore};
80                let keys = Self::pre_load(&context)?;
81                let values = context.store().read_multi_values_bytes(keys).await?;
82                Self::post_load(context, &values)
83            }
84        }
85    }
86
87    /// Discards all pending changes. After that `flush` should have no effect to storage.
88    fn rollback(&mut self);
89
90    /// Returns [`true`] if flushing this view would result in changes to the persistent storage.
91    async fn has_pending_changes(&self) -> bool;
92
93    /// Clears the view. That can be seen as resetting to default. If the clear is followed
94    /// by a flush then all the relevant data is removed on the storage.
95    fn clear(&mut self);
96
97    /// Persists changes to storage. This leaves the view still usable and is essentially neutral to the
98    /// program running. Crash-resistant storage implementations are expected to accumulate the desired
99    /// changes in the `batch` variable first. If the view is dropped without calling `flush`, staged
100    /// changes are simply lost.
101    /// The returned boolean indicates whether the operation removes the view or not.
102    fn flush(&mut self, batch: &mut Batch) -> Result<bool, ViewError>;
103
104    /// Builds a trivial view that is already deleted
105    fn new(context: Self::Context) -> Result<Self, ViewError> {
106        let values = vec![None; Self::NUM_INIT_KEYS];
107        let mut view = Self::post_load(context, &values)?;
108        view.clear();
109        Ok(view)
110    }
111}

113/// A view which can have its context replaced.
114pub trait ReplaceContext<C: crate::context::Context>: View {
115    /// The type returned after replacing the context.
116    type Target: View<Context = C>;
117
118    /// Returns a view with a replaced context.
119    async fn with_context(&mut self, ctx: impl FnOnce(&Self::Context) -> C + Clone)
120        -> Self::Target;
121}

123/// A view that supports hashing its values.
124#[cfg_attr(not(web), trait_variant::make(Send))]
125pub trait HashableView: View {
126    /// How to compute hashes.
127    type Hasher: Hasher;
128
129    /// Computes the hash of the values.
130    ///
131    /// Implementations do not need to include a type tag. However, the usual precautions
132    /// to enforce collision resistance must be applied (e.g. including the length of a
133    /// collection of values).
134    async fn hash(&self) -> Result<<Self::Hasher as Hasher>::Output, ViewError>;
135
136    /// Same as `hash` but guaranteed to be wait-free.
137    async fn hash_mut(&mut self) -> Result<<Self::Hasher as Hasher>::Output, ViewError>;
138}
139
140/// The requirement for the hasher type in [`HashableView`].
141pub trait Hasher: Default + Write + Send + Sync + 'static {
142    /// The output type.
143    type Output: Debug + Clone + Eq + AsRef<[u8]> + 'static;
144
145    /// Finishes the hashing process and returns its output.
146    fn finalize(self) -> Self::Output;
147
148    /// Serializes a value with BCS and includes it in the hash.
149    fn update_with_bcs_bytes(&mut self, value: &impl Serialize) -> Result<(), ViewError> {
150        bcs::serialize_into(self, value)?;
151        Ok(())
152    }
153
154    /// Includes bytes in the hash.
155    fn update_with_bytes(&mut self, value: &[u8]) -> Result<(), ViewError> {
156        self.write_all(value)?;
157        Ok(())
158    }
159}
160
161impl Hasher for sha3::Sha3_256 {
162    type Output = HasherOutput;
163
164    fn finalize(self) -> Self::Output {
165        <sha3::Sha3_256 as sha3::Digest>::finalize(self)
166    }
167}

176/// A [`View`] that also supports crypto hash
177#[cfg_attr(not(web), trait_variant::make(Send))]
178pub trait CryptoHashView: HashableView {
179    /// Computing the hash and attributing the type to it. May require locking.

182    /// Same as `crypto_hash` but guaranteed to be wait-free.
183    async fn crypto_hash_mut(&mut self) -> Result<CryptoHash, ViewError>;
184}

190/// A view that can be shared (unsafely) by cloning it.
191///
192/// Note: Calling `flush` on any of the shared views will break the other views. Therefore,
193/// cloning views is only safe if `flush` only ever happens after all the copies but one
194/// have been dropped.
195pub trait ClonableView: View {
196    /// Creates a clone of this view, sharing the underlying storage context but prone to
197    /// data races which can corrupt the view state.
198    fn clone_unchecked(&mut self) -> Result<Self, ViewError>;
199}
