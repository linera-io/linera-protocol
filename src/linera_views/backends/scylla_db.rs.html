57/// Fundamental constant in ScyllaDB: The maximum size of a multi keys query
58/// The limit is in reality 100. But we need one entry for the root key.
59const MAX_MULTI_KEYS: usize = 100 - 1;
60
61/// The maximal size of an operation on ScyllaDB seems to be 16 MiB
62/// https://www.scylladb.com/2019/03/27/best-practices-for-scylla-applications/
63/// "There is a hard limit at 16 MiB, and nothing bigger than that can arrive at once
64///  at the database at any particular time"
65/// So, we set up the maximal size of 16 MiB - 10 KiB for the values and 10 KiB for the keys
66/// We also arbitrarily decrease the size by 4000 bytes because an amount of size is
67/// taken internally by the database.
68const RAW_MAX_VALUE_SIZE: usize = 16 * 1024 * 1024 - 10 * 1024 - 4000;
69const MAX_KEY_SIZE: usize = 10 * 1024;
70const MAX_BATCH_TOTAL_SIZE: usize = RAW_MAX_VALUE_SIZE + MAX_KEY_SIZE;
71
72/// The `RAW_MAX_VALUE_SIZE` is the maximum size on the ScyllaDB storage.
73/// However, the value being written can also be the serialization of a `SimpleUnorderedBatch`
74/// Therefore the actual `MAX_VALUE_SIZE` is lower.
75/// At the maximum the key size is 1024 bytes (see below) and we pack just one entry.
76/// So if the key has 1024 bytes this gets us the inequality
77/// `1 + 1 + 1 + serialized_size(MAX_KEY_SIZE)? + serialized_size(x)? <= RAW_MAX_VALUE_SIZE`.
78/// and so this simplifies to `1 + 1 + 1 + (2 + 10240) + (4 + x) <= RAW_MAX_VALUE_SIZE`
79/// Note on the above formula:
80/// * We write 4 because `get_uleb128_size(RAW_MAX_VALUE_SIZE) = 4)`
81/// * We write `1 + 1 + 1`  because the `UnorderedBatch` has three entries.
82///
83/// This gets us to a maximal value of 16752727.
84const VISIBLE_MAX_VALUE_SIZE: usize = RAW_MAX_VALUE_SIZE
85    - MAX_KEY_SIZE
86    - get_uleb128_size(RAW_MAX_VALUE_SIZE)
87    - get_uleb128_size(MAX_KEY_SIZE)
88    - 3;
89
90/// The constant 14000 is an empirical constant that was found to be necessary
91/// to make the ScyllaDB system work. We have not been able to find this or
92/// a similar constant in the source code or the documentation.
93/// An experimental approach gets us that 14796 is the latest value that is
94/// correct.
95const MAX_BATCH_SIZE: usize = 5000;

98const KEYSPACE: &str = "kv";

100/// The client for ScyllaDB:
101/// * The session allows to pass queries
102/// * The namespace that is being assigned to the database
103/// * The prepared queries used for implementing the features of `KeyValueStore`.

122    async fn new(session: Session, namespace: &str) -> Result<Self, ScyllaDbStoreInternalError> {
123        let namespace = namespace.to_string();
124        let read_value = session
125            .prepare(format!(
126                "SELECT v FROM {}.\"{}\" WHERE root_key = ? AND k = ?",
127                KEYSPACE, namespace
128            ))
129            .await?;
130
131        let contains_key = session
132            .prepare(format!(
133                "SELECT root_key FROM {}.\"{}\" WHERE root_key = ? AND k = ?",
134                KEYSPACE, namespace
135            ))
136            .await?;
137
138        let write_batch_delete_prefix_unbounded = session
139            .prepare(format!(
140                "DELETE FROM {}.\"{}\" WHERE root_key = ? AND k >= ?",
141                KEYSPACE, namespace
142            ))
143            .await?;
144
145        let write_batch_delete_prefix_bounded = session
146            .prepare(format!(
147                "DELETE FROM {}.\"{}\" WHERE root_key = ? AND k >= ? AND k < ?",
148                KEYSPACE, namespace
149            ))
150            .await?;
151
152        let write_batch_deletion = session
153            .prepare(format!(
154                "DELETE FROM {}.\"{}\" WHERE root_key = ? AND k = ?",
155                KEYSPACE, namespace
156            ))
157            .await?;
158
159        let write_batch_insertion = session
160            .prepare(format!(
161                "INSERT INTO {}.\"{}\" (root_key, k, v) VALUES (?, ?, ?)",
162                KEYSPACE, namespace
163            ))
164            .await?;
165
166        let find_keys_by_prefix_unbounded = session
167            .prepare(format!(
168                "SELECT k FROM {}.\"{}\" WHERE root_key = ? AND k >= ?",
169                KEYSPACE, namespace
170            ))
171            .await?;
172
173        let find_keys_by_prefix_bounded = session
174            .prepare(format!(
175                "SELECT k FROM {}.\"{}\" WHERE root_key = ? AND k >= ? AND k < ?",
176                KEYSPACE, namespace
177            ))
178            .await?;
179
180        let find_key_values_by_prefix_unbounded = session
181            .prepare(format!(
182                "SELECT k,v FROM {}.\"{}\" WHERE root_key = ? AND k >= ?",
183                KEYSPACE, namespace
184            ))
185            .await?;
186
187        let find_key_values_by_prefix_bounded = session
188            .prepare(format!(
189                "SELECT k,v FROM {}.\"{}\" WHERE root_key = ? AND k >= ? AND k < ?",
190                KEYSPACE, namespace
191            ))
192            .await?;
193
194        Ok(Self {
195            session,
196            namespace,
197            read_value,
198            contains_key,
199            write_batch_delete_prefix_unbounded,
200            write_batch_delete_prefix_bounded,
201            write_batch_deletion,

204            find_keys_by_prefix_bounded,
205            find_key_values_by_prefix_unbounded,
206            find_key_values_by_prefix_bounded,
207            multi_key_values: papaya::HashMap::new(),
208            multi_keys: papaya::HashMap::new(),
209        })
210    }

241    async fn get_multi_key_values_statement(
242        &self,
243        num_markers: usize,
244    ) -> Result<PreparedStatement, ScyllaDbStoreInternalError> {
245        if let Some(prepared_statement) = self.multi_key_values.pin().get(&num_markers) {
246            return Ok(prepared_statement.clone());
247        }
248        let markers = std::iter::repeat_n("?", num_markers)
249            .collect::<Vec<_>>()
250            .join(",");
251        let prepared_statement = self
252            .session
253            .prepare(format!(
254                "SELECT k,v FROM {}.\"{}\" WHERE root_key = ? AND k IN ({})",
255                KEYSPACE, self.namespace, markers
256            ))
257            .await?;
258        self.multi_key_values
259            .pin()
260            .insert(num_markers, prepared_statement.clone());
261        Ok(prepared_statement)
262    }

311    async fn read_value_internal(
312        &self,
313        root_key: &[u8],
314        key: Vec<u8>,
315    ) -> Result<Option<Vec<u8>>, ScyllaDbStoreInternalError> {
316        Self::check_key_size(&key)?;
317        let session = &self.session;
318        // Read the value of a key
319        let values = (root_key.to_vec(), key);
320
321        let (result, _) = session
322            .execute_single_page(&self.read_value, &values, PagingState::start())
323            .await?;
324        let rows = result.into_rows_result()?;
325        let mut rows = rows.rows::<(Vec<u8>,)>()?;
326        Ok(match rows.next() {
327            Some(row) => Some(row?.0),
328            None => None,
329        })
330    }
331
332    fn get_occurrences_map(
333        keys: Vec<Vec<u8>>,
334    ) -> Result<HashMap<Vec<u8>, Vec<usize>>, ScyllaDbStoreInternalError> {
335        let mut map = HashMap::<Vec<u8>, Vec<usize>>::new();
336        for (i_key, key) in keys.into_iter().enumerate() {
337            Self::check_key_size(&key)?;
338            map.entry(key).or_default().push(i_key);
339        }
340        Ok(map)
341    }
342
343    async fn read_multi_values_internal(
344        &self,
345        root_key: &[u8],
346        keys: Vec<Vec<u8>>,
347    ) -> Result<Vec<Option<Vec<u8>>>, ScyllaDbStoreInternalError> {
348        let mut values = vec![None; keys.len()];
349        let map = Self::get_occurrences_map(keys)?;
350        let statement = self.get_multi_key_values_statement(map.len()).await?;
351        let mut inputs = vec![root_key.to_vec()];
352        inputs.extend(map.keys().cloned());
353        let mut rows = Box::pin(self.session.execute_iter(statement, &inputs))
354            .await?
355            .rows_stream::<(Vec<u8>, Vec<u8>)>()?;
356
357        while let Some(row) = rows.next().await {
358            let (key, value) = row?;
359            if let Some((&last, rest)) = map[&key].split_last() {
360                for position in rest {
361                    values[*position] = Some(value.clone());
362                }
363                values[last] = Some(value);
364            }
365        }
366        Ok(values)
367    }

369    async fn contains_keys_internal(
370        &self,
371        root_key: &[u8],
372        keys: Vec<Vec<u8>>,
373    ) -> Result<Vec<bool>, ScyllaDbStoreInternalError> {
374        let mut values = vec![false; keys.len()];
375        let map = Self::get_occurrences_map(keys)?;
376        let statement = self.get_multi_keys_statement(map.len()).await?;
377        let mut inputs = vec![root_key.to_vec()];
378        inputs.extend(map.keys().cloned());
379        let mut rows = Box::pin(self.session.execute_iter(statement, &inputs))
380            .await?
381            .rows_stream::<(Vec<u8>,)>()?;
382
383        while let Some(row) = rows.next().await {
384            let (key,) = row?;
385            for i_key in &map[&key] {
386                values[*i_key] = true;
387            }
388        }
389
390        Ok(values)
391    }

393    async fn contains_key_internal(
394        &self,
395        root_key: &[u8],
396        key: Vec<u8>,
397    ) -> Result<bool, ScyllaDbStoreInternalError> {
398        Self::check_key_size(&key)?;
399        let session = &self.session;
400        // Read the value of a key
401        let values = (root_key.to_vec(), key);
402
403        let (result, _) = session
404            .execute_single_page(&self.contains_key, &values, PagingState::start())
405            .await?;
406        let rows = result.into_rows_result()?;
407        let mut rows = rows.rows::<(Vec<u8>,)>()?;
408        Ok(rows.next().is_some())
409    }

411    async fn write_batch_internal(
412        &self,
413        root_key: &[u8],
414        batch: UnorderedBatch,
415    ) -> Result<(), ScyllaDbStoreInternalError> {
416        let session = &self.session;
417        let mut batch_query = scylla::statement::batch::Batch::new(BatchType::Unlogged);
418        let mut batch_values = Vec::new();
419        let query1 = &self.write_batch_delete_prefix_unbounded;
420        let query2 = &self.write_batch_delete_prefix_bounded;
421        Self::check_batch_len(&batch)?;
422        for key_prefix in batch.key_prefix_deletions {
423            Self::check_key_size(&key_prefix)?;
424            match get_upper_bound_option(&key_prefix) {
425                None => {
426                    let values = vec![root_key.to_vec(), key_prefix];
427                    batch_values.push(values);
428                    batch_query.append_statement(query1.clone());
429                }
430                Some(upper_bound) => {
431                    let values = vec![root_key.to_vec(), key_prefix, upper_bound];
432                    batch_values.push(values);
433                    batch_query.append_statement(query2.clone());
434                }
435            }
436        }
437        let query3 = &self.write_batch_deletion;
438        for key in batch.simple_unordered_batch.deletions {
439            Self::check_key_size(&key)?;
440            let values = vec![root_key.to_vec(), key];
441            batch_values.push(values);
442            batch_query.append_statement(query3.clone());
443        }
444        let query4 = &self.write_batch_insertion;
445        for (key, value) in batch.simple_unordered_batch.insertions {
446            Self::check_key_size(&key)?;
447            Self::check_value_size(&value)?;
448            let values = vec![root_key.to_vec(), key, value];
449            batch_values.push(values);
450            batch_query.append_statement(query4.clone());
451        }
452        session.batch(&batch_query, batch_values).await?;
453        Ok(())
454    }

456    async fn find_keys_by_prefix_internal(
457        &self,
458        root_key: &[u8],
459        key_prefix: Vec<u8>,
460    ) -> Result<Vec<Vec<u8>>, ScyllaDbStoreInternalError> {
461        Self::check_key_size(&key_prefix)?;
462        let session = &self.session;
463        // Read the value of a key
464        let len = key_prefix.len();
465        let query_unbounded = &self.find_keys_by_prefix_unbounded;
466        let query_bounded = &self.find_keys_by_prefix_bounded;
467        let rows = match get_upper_bound_option(&key_prefix) {
468            None => {
469                let values = (root_key.to_vec(), key_prefix.clone());
470                Box::pin(session.execute_iter(query_unbounded.clone(), values)).await?
471            }
472            Some(upper_bound) => {
473                let values = (root_key.to_vec(), key_prefix.clone(), upper_bound);
474                Box::pin(session.execute_iter(query_bounded.clone(), values)).await?
475            }
476        };
477        let mut rows = rows.rows_stream::<(Vec<u8>,)>()?;
478        let mut keys = Vec::new();
479        while let Some(row) = rows.next().await {
480            let (key,) = row?;
481            let short_key = key[len..].to_vec();
482            keys.push(short_key);
483        }
484        Ok(keys)
485    }

487    async fn find_key_values_by_prefix_internal(
488        &self,
489        root_key: &[u8],
490        key_prefix: Vec<u8>,
491    ) -> Result<Vec<(Vec<u8>, Vec<u8>)>, ScyllaDbStoreInternalError> {
492        Self::check_key_size(&key_prefix)?;
493        let session = &self.session;
494        // Read the value of a key
495        let len = key_prefix.len();
496        let query_unbounded = &self.find_key_values_by_prefix_unbounded;
497        let query_bounded = &self.find_key_values_by_prefix_bounded;
498        let rows = match get_upper_bound_option(&key_prefix) {
499            None => {
500                let values = (root_key.to_vec(), key_prefix.clone());
501                Box::pin(session.execute_iter(query_unbounded.clone(), values)).await?
502            }
503            Some(upper_bound) => {
504                let values = (root_key.to_vec(), key_prefix.clone(), upper_bound);
505                Box::pin(session.execute_iter(query_bounded.clone(), values)).await?
506            }
507        };
508        let mut rows = rows.rows_stream::<(Vec<u8>, Vec<u8>)>()?;
509        let mut key_values = Vec::new();
510        while let Some(row) = rows.next().await {
511            let (key, value) = row?;
512            let short_key = key[len..].to_vec();
513            key_values.push((short_key, value));
514        }
515        Ok(key_values)
516    }

636        keys: Vec<Vec<u8>>,
637    ) -> Result<Vec<bool>, ScyllaDbStoreInternalError> {
638        if keys.is_empty() {
639            return Ok(Vec::new());
640        }
641        let store = self.store.deref();
642        let _guard = self.acquire().await;
643        let handles = keys
644            .chunks(MAX_MULTI_KEYS)
645            .map(|keys| store.contains_keys_internal(&self.root_key, keys.to_vec()));
646        let results: Vec<_> = join_all(handles)
647            .await
648            .into_iter()
649            .collect::<Result<_, _>>()?;
650        Ok(results.into_iter().flatten().collect())
651    }
652
653    async fn read_multi_values_bytes(
654        &self,
655        keys: Vec<Vec<u8>>,
656    ) -> Result<Vec<Option<Vec<u8>>>, ScyllaDbStoreInternalError> {
657        if keys.is_empty() {
658            return Ok(Vec::new());
659        }
660        let store = self.store.deref();
661        let _guard = self.acquire().await;
662        let handles = keys
663            .chunks(MAX_MULTI_KEYS)
664            .map(|keys| store.read_multi_values_internal(&self.root_key, keys.to_vec()));
665        let results: Vec<_> = join_all(handles)
666            .await
667            .into_iter()
668            .collect::<Result<_, _>>()?;
669        Ok(results.into_iter().flatten().collect())
670    }
671
672    async fn find_keys_by_prefix(
673        &self,
674        key_prefix: &[u8],
675    ) -> Result<Vec<Vec<u8>>, ScyllaDbStoreInternalError> {
676        let store = self.store.deref();
677        let _guard = self.acquire().await;
678        Box::pin(store.find_keys_by_prefix_internal(&self.root_key, key_prefix.to_vec())).await
679    }
680
681    async fn find_key_values_by_prefix(
682        &self,
683        key_prefix: &[u8],
684    ) -> Result<Vec<(Vec<u8>, Vec<u8>)>, ScyllaDbStoreInternalError> {
685        let store = self.store.deref();
686        let _guard = self.acquire().await;
687        Box::pin(store.find_key_values_by_prefix_internal(&self.root_key, key_prefix.to_vec()))
688            .await
689    }

692impl DirectWritableKeyValueStore for ScyllaDbStoreInternal {
693    const MAX_BATCH_SIZE: usize = MAX_BATCH_SIZE;
694    const MAX_BATCH_TOTAL_SIZE: usize = MAX_BATCH_TOTAL_SIZE;
695    const MAX_VALUE_SIZE: usize = VISIBLE_MAX_VALUE_SIZE;
696
697    // ScyllaDB cannot take a `crate::batch::Batch` directly. Indeed, if a delete is
698    // followed by a write, then the delete takes priority. See the sentence "The first
699    // tie-breaking rule when two cells have the same write timestamp is that dead cells
700    // win over live cells" from
701    // https://github.com/scylladb/scylladb/blob/master/docs/dev/timestamp-conflict-resolution.md
702    type Batch = UnorderedBatch;
703
704    async fn write_batch(&self, batch: Self::Batch) -> Result<(), ScyllaDbStoreInternalError> {
705        let store = self.store.deref();
706        let _guard = self.acquire().await;
707        store.write_batch_internal(&self.root_key, batch).await
708    }
709}

711// ScyllaDB requires that the keys are non-empty.
712fn get_big_root_key(root_key: &[u8]) -> Vec<u8> {
713    let mut big_key = vec![0];
714    big_key.extend(root_key);
715    big_key
716}

722    pub uri: String,
723    /// Maximum number of concurrent database queries allowed for this client.
724    pub max_concurrent_queries: Option<usize>,
725    /// Preferred buffer size for async streams.
726    pub max_stream_queries: usize,

731impl KeyValueDatabase for ScyllaDbDatabaseInternal {
732    type Config = ScyllaDbStoreInternalConfig;
733    type Store = ScyllaDbStoreInternal;
734
735    fn get_name() -> String {
736        "scylladb internal".to_string()
737    }
738
739    async fn connect(
740        config: &Self::Config,
741        namespace: &str,
742    ) -> Result<Self, ScyllaDbStoreInternalError> {
743        Self::check_namespace(namespace)?;
744        let session = ScyllaDbClient::build_default_session(&config.uri).await?;
745        let store = ScyllaDbClient::new(session, namespace).await?;
746        let store = Arc::new(store);
747        let semaphore = config
748            .max_concurrent_queries
749            .map(|n| Arc::new(Semaphore::new(n)));
750        let max_stream_queries = config.max_stream_queries;
751        Ok(Self {
752            store,
753            semaphore,
754            max_stream_queries,
755        })
756    }
757
758    fn open_shared(&self, root_key: &[u8]) -> Result<Self::Store, ScyllaDbStoreInternalError> {
759        let store = self.store.clone();
760        let semaphore = self.semaphore.clone();
761        let max_stream_queries = self.max_stream_queries;
762        let root_key = get_big_root_key(root_key);
763        Ok(ScyllaDbStoreInternal {
764            store,
765            semaphore,
766            max_stream_queries,
767            root_key,
768        })
769    }
770
771    fn open_exclusive(&self, root_key: &[u8]) -> Result<Self::Store, ScyllaDbStoreInternalError> {
772        self.open_shared(root_key)
773    }
774
775    async fn list_all(config: &Self::Config) -> Result<Vec<String>, ScyllaDbStoreInternalError> {
776        let session = ScyllaDbClient::build_default_session(&config.uri).await?;
777        let statement = session
778            .prepare(format!("DESCRIBE KEYSPACE {}", KEYSPACE))
779            .await?;
780        let result = Box::pin(session.execute_iter(statement, &[])).await;
781        let miss_msg = format!("'{}' not found in keyspaces", KEYSPACE);
782        let result = match result {
783            Ok(result) => result,
784            Err(error) => {
785                let invalid_or_keyspace_not_found = match &error {
786                    PagerExecutionError::NextPageError(NextPageError::RequestFailure(
787                        RequestError::LastAttemptError(RequestAttemptError::DbError(db_error, msg)),
788                    )) => *db_error == DbError::Invalid && msg.as_str() == miss_msg,
789                    _ => false,
790                };
791                if invalid_or_keyspace_not_found {
792                    return Ok(Vec::new());
793                } else {
794                    return Err(ScyllaDbStoreInternalError::PagerExecutionError(error));
795                }
796            }
797        };
798        let mut namespaces = Vec::new();
799        let mut rows_stream = result.rows_stream::<(String, String, String, String)>()?;
800        while let Some(row) = rows_stream.next().await {
801            let (_, object_kind, name, _) = row?;
802            if object_kind == "table" {
803                namespaces.push(name);
804            }
805        }
806        Ok(namespaces)
807    }
808
809    async fn list_root_keys(&self) -> Result<Vec<Vec<u8>>, ScyllaDbStoreInternalError> {
810        let statement = self
811            .store
812            .session
813            .prepare(format!(
814                "SELECT root_key FROM {}.\"{}\" ALLOW FILTERING",
815                KEYSPACE, self.store.namespace
816            ))
817            .await?;
818
819        // Execute the query
820        let rows = Box::pin(self.store.session.execute_iter(statement, &[])).await?;
821        let mut rows = rows.rows_stream::<(Vec<u8>,)>()?;
822        let mut root_keys = BTreeSet::new();
823        while let Some(row) = rows.next().await {
824            let (root_key,) = row?;
825            let root_key = root_key[1..].to_vec();
826            root_keys.insert(root_key);
827        }
828        Ok(root_keys.into_iter().collect::<Vec<_>>())
829    }
830
831    async fn delete_all(store_config: &Self::Config) -> Result<(), ScyllaDbStoreInternalError> {
832        let session = ScyllaDbClient::build_default_session(&store_config.uri).await?;
833        let statement = session
834            .prepare(format!("DROP KEYSPACE IF EXISTS {}", KEYSPACE))
835            .await?;
836
837        session
838            .execute_single_page(&statement, &[], PagingState::start())
839            .await?;
840        Ok(())
841    }
842
843    async fn exists(
844        config: &Self::Config,
845        namespace: &str,
846    ) -> Result<bool, ScyllaDbStoreInternalError> {
847        Self::check_namespace(namespace)?;
848        let session = ScyllaDbClient::build_default_session(&config.uri).await?;
849
850        // We check the way the test can fail. It can fail in different ways.
851        let result = session
852            .prepare(format!(
853                "SELECT root_key FROM {}.\"{}\" LIMIT 1 ALLOW FILTERING",
854                KEYSPACE, namespace
855            ))
856            .await;
857
858        // The missing table translates into a very specific error that we matched
859        let miss_msg1 = format!("unconfigured table {}", namespace);
860        let miss_msg1 = miss_msg1.as_str();
861        let miss_msg2 = "Undefined name root_key in selection clause";
862        let miss_msg3 = format!("Keyspace {} does not exist", KEYSPACE);
863        let Err(error) = result else {
864            // If OK, then the table exists
865            return Ok(true);
866        };
867        let missing_table = match &error {
868            PrepareError::AllAttemptsFailed {
869                first_attempt: RequestAttemptError::DbError(db_error, msg),
870            } => {
871                if *db_error != DbError::Invalid {
872                    false
873                } else {
874                    msg.as_str() == miss_msg1
875                        || msg.as_str() == miss_msg2
876                        || msg.as_str() == miss_msg3
877                }
878            }
879            _ => false,
880        };
881        if missing_table {
882            Ok(false)
883        } else {
884            Err(ScyllaDbStoreInternalError::PrepareError(error))
885        }
886    }
887
888    async fn create(
889        config: &Self::Config,
890        namespace: &str,
891    ) -> Result<(), ScyllaDbStoreInternalError> {
892        Self::check_namespace(namespace)?;
893        let session = ScyllaDbClient::build_default_session(&config.uri).await?;
894
895        // Create a keyspace if it doesn't exist
896        let statement = session
897            .prepare(format!(
898                "CREATE KEYSPACE IF NOT EXISTS {} WITH REPLICATION = {{ \
899                    'class' : 'NetworkTopologyStrategy', \
900                    'replication_factor' : {} \
901                }}",
902                KEYSPACE, config.replication_factor
903            ))
904            .await?;
905        session
906            .execute_single_page(&statement, &[], PagingState::start())
907            .await?;
908
909        // This explicitly sets a lot of default parameters for clarity and for making future
910        // changes easier.
911        let statement = session
912            .prepare(format!(
913                "CREATE TABLE {}.\"{}\" (\
914                    root_key blob, \
915                    k blob, \
916                    v blob, \
917                    PRIMARY KEY (root_key, k) \
918                ) \
919                WITH compaction = {{ \
920                    'class'            : 'SizeTieredCompactionStrategy', \
921                    'min_sstable_size' : 52428800, \
922                    'bucket_low'       : 0.5, \
923                    'bucket_high'      : 1.5, \
924                    'min_threshold'    : 4, \
925                    'max_threshold'    : 32 \
926                }} \
927                AND compression = {{ \
928                    'sstable_compression': 'LZ4Compressor', \
929                    'chunk_length_in_kb':'4' \
930                }} \
931                AND caching = {{ \
932                    'enabled': 'true' \
933                }}",
934                KEYSPACE, namespace
935            ))
936            .await?;
937        session
938            .execute_single_page(&statement, &[], PagingState::start())
939            .await?;
940        Ok(())
941    }
942
943    async fn delete(
944        config: &Self::Config,
945        namespace: &str,
946    ) -> Result<(), ScyllaDbStoreInternalError> {
947        Self::check_namespace(namespace)?;
948        let session = ScyllaDbClient::build_default_session(&config.uri).await?;
949        let statement = session
950            .prepare(format!(
951                "DROP TABLE IF EXISTS {}.\"{}\";",
952                KEYSPACE, namespace
953            ))
954            .await?;
955        session
956            .execute_single_page(&statement, &[], PagingState::start())
957            .await?;
958        Ok(())
959    }

962impl ScyllaDbStoreInternal {
963    /// Obtains the semaphore lock on the database if needed.
964    async fn acquire(&self) -> Option<SemaphoreGuard<'_>> {
965        match &self.semaphore {
966            None => None,
967            Some(count) => Some(count.acquire().await),
968        }
969    }
970}

972impl ScyllaDbDatabaseInternal {
973    fn check_namespace(namespace: &str) -> Result<(), ScyllaDbStoreInternalError> {
974        if !namespace.is_empty()
975            && namespace.len() <= 48
976            && namespace
977                .chars()
978                .all(|c| c.is_ascii_alphanumeric() || c == '_')
979        {
980            return Ok(());
981        }
982        Err(ScyllaDbStoreInternalError::InvalidNamespace)
983    }
984}
