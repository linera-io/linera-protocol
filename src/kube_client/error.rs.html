6/// Possible errors when working with [`kube`][crate]
7#[cfg_attr(docsrs, doc(cfg(any(feature = "config", feature = "client"))))]
8#[derive(Error, Debug)]
9pub enum Error {
10    /// ApiError for when things fail
11    ///
12    /// This can be parsed into as an error handling fallback.
13    /// It's also used in `WatchEvent` from watch calls.
14    ///
15    /// It's quite common to get a `410 Gone` when the `resourceVersion` is too old.
16    #[error("ApiError: {0} ({0:?})")]
17    Api(#[source] ErrorResponse),
18
19    /// Hyper error
20    #[cfg(feature = "client")]
21    #[error("HyperError: {0}")]
22    HyperError(#[source] hyper::Error),
23    /// Service error
24    #[cfg(feature = "client")]
25    #[error("ServiceError: {0}")]
26    Service(#[source] tower::BoxError),
27
28    /// UTF-8 Error
29    #[error("UTF-8 Error: {0}")]
30    FromUtf8(#[source] std::string::FromUtf8Error),
31
32    /// Returned when failed to find a newline character within max length.
33    /// Only returned by `Client::request_events` and this should never happen as
34    /// the max is `usize::MAX`.
35    #[error("Error finding newline character")]
36    LinesCodecMaxLineLengthExceeded,
37
38    /// Returned on `std::io::Error` when reading event stream.
39    #[error("Error reading events stream: {0}")]
40    ReadEvents(#[source] std::io::Error),
41
42    /// Http based error
43    #[error("HttpError: {0}")]
44    HttpError(#[source] http::Error),
45
46    /// Common error case when requesting parsing into own structs
47    #[error("Error deserializing response")]
48    SerdeError(#[source] serde_json::Error),
49
50    /// Failed to build request
51    #[error("Failed to build request: {0}")]
52    BuildRequest(#[source] kube_core::request::Error),
53
54    /// Failed to infer config
55    #[error("Failed to infer configuration: {0}")]
56    InferConfig(#[source] crate::config::InferConfigError),
57
58    /// Discovery errors
59    #[error("Error from discovery: {0}")]
60    Discovery(#[source] DiscoveryError),
61
62    /// Errors from OpenSSL TLS
63    #[cfg(feature = "openssl-tls")]
64    #[cfg_attr(docsrs, doc(cfg(feature = "openssl-tls")))]
65    #[error("openssl tls error: {0}")]
66    OpensslTls(#[source] crate::client::OpensslTlsError),
67
68    /// Errors from Rustls TLS
69    #[cfg(feature = "rustls-tls")]
70    #[cfg_attr(docsrs, doc(cfg(feature = "rustls-tls")))]
71    #[error("rustls tls error: {0}")]
72    RustlsTls(#[source] crate::client::RustlsTlsError),
73
74    /// Missing TLS stacks when TLS is required
75    #[error("TLS required but no TLS stack selected")]
76    TlsRequired,
77
78    /// Failed to upgrade to a WebSocket connection
79    #[cfg(feature = "ws")]
80    #[cfg_attr(docsrs, doc(cfg(feature = "ws")))]
81    #[error("failed to upgrade to a WebSocket connection: {0}")]
82    UpgradeConnection(#[source] crate::client::UpgradeConnectionError),
83
84    /// Errors related to client auth
85    #[cfg(feature = "client")]
86    #[cfg_attr(docsrs, doc(cfg(feature = "client")))]
87    #[error("auth error: {0}")]
88    Auth(#[source] crate::client::AuthError),
89}
