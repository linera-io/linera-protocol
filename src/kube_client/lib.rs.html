1//! Crate for interacting with the Kubernetes API
2//!
3//! This crate includes the tools for manipulating Kubernetes resources as
4//! well as keeping track of those resources as they change over time
5//!
6//! # Example
7//!
8//! The following example will create a [`Pod`](k8s_openapi::api::core::v1::Pod)
9//! and then watch for it to become available using a manual [`Api::watch`] call.
10//!
11//! ```rust,no_run
12//! use futures::{StreamExt, TryStreamExt};
13//! use kube_client::api::{Api, ResourceExt, ListParams, PatchParams, Patch};
14//! use kube_client::Client;
15//! use k8s_openapi::api::core::v1::Pod;
16//!
17//! #[tokio::main]
18//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
19//!     // Read the environment to find config for kube client.
20//!     // Note that this tries an in-cluster configuration first,
21//!     // then falls back on a kubeconfig file.
22//!     let client = Client::try_default().await?;
23//!
24//!     // Interact with pods in the configured namespace with the typed interface from k8s-openapi
25//!     let pods: Api<Pod> = Api::default_namespaced(client);
26//!
27//!     // Create a Pod (cheating here with json, but it has to validate against the type):
28//!     let patch: Pod = serde_json::from_value(serde_json::json!({
29//!         "apiVersion": "v1",
30//!         "kind": "Pod",
31//!         "metadata": {
32//!             "name": "my-pod"
33//!         },
34//!         "spec": {
35//!             "containers": [
36//!                 {
37//!                     "name": "my-container",
38//!                     "image": "myregistry.azurecr.io/hello-world:v1",
39//!                 },
40//!             ],
41//!         }
42//!     }))?;
43//!
44//!     // Apply the Pod via server-side apply
45//!     let params = PatchParams::apply("myapp");
46//!     let result = pods.patch("my-pod", &params, &Patch::Apply(&patch)).await?;
47//!
48//!     // List pods in the configured namespace
49//!     for p in pods.list(&ListParams::default()).await? {
50//!         println!("found pod {}", p.name_any());
51//!     }
52//!
53//!     Ok(())
54//! }
55//! ```
56//!
57//! For more details, see:
58//!
59//! - [`Client`](crate::client) for the extensible Kubernetes client
60//! - [`Config`](crate::config) for the Kubernetes config abstraction
61//! - [`Api`](crate::Api) for the generic api methods available on Kubernetes resources
62//! - [k8s-openapi](https://docs.rs/k8s-openapi/*/k8s_openapi/) for how to create typed kubernetes objects directly

195    #[ignore = "needs cluster (will create and edit a pod)"]
196    async fn pod_can_use_core_apis() -> Result<(), Box<dyn std::error::Error>> {
197        use kube::api::{DeleteParams, ListParams, Patch, PatchParams, PostParams, WatchEvent};
198
199        let client = Client::try_default().await?;
200        let pods: Api<Pod> = Api::default_namespaced(client);
201
202        // create busybox pod that's alive for at most 30s
203        let p: Pod = serde_json::from_value(json!({
204            "apiVersion": "v1",
205            "kind": "Pod",
206            "metadata": {
207                "name": "busybox-kube1",
208                "labels": { "app": "kube-rs-test" },
209            },
210            "spec": {
211                "terminationGracePeriodSeconds": 1,
212                "restartPolicy": "Never",
213                "containers": [{
214                  "name": "busybox",
215                  "image": "busybox:1.34.1",
216                  "command": ["sh", "-c", "sleep 30"],
217                }],
218            }
219        }))?;
220
221        let pp = PostParams::default();
222        match pods.create(&pp, &p).await {
223            Ok(o) => assert_eq!(p.name_unchecked(), o.name_unchecked()),
224            Err(crate::Error::Api(ae)) => assert_eq!(ae.code, 409), // if we failed to clean-up
225            Err(e) => return Err(e.into()),                         // any other case if a failure
226        }
227
228        // Manual watch-api for it to become ready
229        // NB: don't do this; using conditions (see pod_api example) is easier and less error prone
230        let wp = WatchParams::default()
231            .fields(&format!("metadata.name={}", "busybox-kube1"))
232            .timeout(15);
233        let mut stream = pods.watch(&wp, "0").await?.boxed();
234        while let Some(ev) = stream.try_next().await? {
235            // can debug format watch event
236            let _ = format!("we: {ev:?}");
237            match ev {
238                WatchEvent::Modified(o) => {
239                    let s = o.status.as_ref().expect("status exists on pod");
240                    let phase = s.phase.clone().unwrap_or_default();
241                    if phase == "Running" {
242                        break;
243                    }
244                }
245                WatchEvent::Error(e) => panic!("watch error: {e}"),
246                _ => {}
247            }
248        }
249
250        // Verify we can get it
251        let mut pod = pods.get("busybox-kube1").await?;
252        assert_eq!(p.spec.as_ref().unwrap().containers[0].name, "busybox");
253
254        // verify replace with explicit resource version
255        // NB: don't do this; use server side apply
256        {
257            assert!(pod.resource_version().is_some());
258            pod.spec.as_mut().unwrap().active_deadline_seconds = Some(5);
259
260            let pp = PostParams::default();
261            let patched_pod = pods.replace("busybox-kube1", &pp, &pod).await?;
262            assert_eq!(patched_pod.spec.unwrap().active_deadline_seconds, Some(5));
263        }
264
265        // Delete it
266        let dp = DeleteParams::default();
267        pods.delete("busybox-kube1", &dp).await?.map_left(|pdel| {
268            assert_eq!(pdel.name_unchecked(), "busybox-kube1");
269        });
270
271        Ok(())
272    }

277    async fn pod_can_exec_and_write_to_stdin() -> Result<(), Box<dyn std::error::Error>> {
278        use crate::api::{DeleteParams, ListParams, Patch, PatchParams, WatchEvent};
279
280        let client = Client::try_default().await?;
281        let pods: Api<Pod> = Api::default_namespaced(client);
282
283        // create busybox pod that's alive for at most 30s
284        let p: Pod = serde_json::from_value(json!({
285            "apiVersion": "v1",
286            "kind": "Pod",
287            "metadata": {
288                "name": "busybox-kube2",
289                "labels": { "app": "kube-rs-test" },
290            },
291            "spec": {
292                "terminationGracePeriodSeconds": 1,
293                "restartPolicy": "Never",
294                "containers": [{
295                  "name": "busybox",
296                  "image": "busybox:1.34.1",
297                  "command": ["sh", "-c", "sleep 30"],
298                }],
299            }
300        }))?;
301
302        match pods.create(&Default::default(), &p).await {
303            Ok(o) => assert_eq!(p.name_unchecked(), o.name_unchecked()),
304            Err(crate::Error::Api(ae)) => assert_eq!(ae.code, 409), // if we failed to clean-up
305            Err(e) => return Err(e.into()),                         // any other case if a failure
306        }
307
308        // Manual watch-api for it to become ready
309        // NB: don't do this; using conditions (see pod_api example) is easier and less error prone
310        let wp = WatchParams::default()
311            .fields(&format!("metadata.name={}", "busybox-kube2"))
312            .timeout(15);
313        let mut stream = pods.watch(&wp, "0").await?.boxed();
314        while let Some(ev) = stream.try_next().await? {
315            match ev {
316                WatchEvent::Modified(o) => {
317                    let s = o.status.as_ref().expect("status exists on pod");
318                    let phase = s.phase.clone().unwrap_or_default();
319                    if phase == "Running" {
320                        break;
321                    }
322                }
323                WatchEvent::Error(e) => panic!("watch error: {e}"),
324                _ => {}
325            }
326        }
327
328        // Verify exec works and we can get the output
329        {
330            let mut attached = pods
331                .exec(
332                    "busybox-kube2",
333                    vec!["sh", "-c", "for i in $(seq 1 3); do echo $i; done"],
334                    &AttachParams::default().stderr(false),
335                )
336                .await?;
337            let stdout = tokio_util::io::ReaderStream::new(attached.stdout().unwrap());
338            let out = stdout
339                .filter_map(|r| async { r.ok().and_then(|v| String::from_utf8(v.to_vec()).ok()) })
340                .collect::<Vec<_>>()
341                .await
342                .join("");
343            attached.join().await.unwrap();
344            assert_eq!(out.lines().count(), 3);
345            assert_eq!(out, "1\n2\n3\n");
346        }
347
348        // Verify we can write to Stdin
349        {
350            use tokio::io::AsyncWriteExt;
351            let mut attached = pods
352                .exec(
353                    "busybox-kube2",
354                    vec!["sh"],
355                    &AttachParams::default().stdin(true).stderr(false),
356                )
357                .await?;
358            let mut stdin_writer = attached.stdin().unwrap();
359            let mut stdout_stream = tokio_util::io::ReaderStream::new(attached.stdout().unwrap());
360            let next_stdout = stdout_stream.next();
361            stdin_writer.write_all(b"echo test string 1\n").await?;
362            let stdout = String::from_utf8(next_stdout.await.unwrap().unwrap().to_vec()).unwrap();
363            println!("{stdout}");
364            assert_eq!(stdout, "test string 1\n");
365
366            // AttachedProcess resolves with status object.
367            // Send `exit 1` to get a failure status.
368            stdin_writer.write_all(b"exit 1\n").await?;
369            let status = attached.take_status().unwrap();
370            if let Some(status) = status.await {
371                println!("{status:?}");
372                assert_eq!(status.status, Some("Failure".to_owned()));
373                assert_eq!(status.reason, Some("NonZeroExitCode".to_owned()));
374            }
375        }
376
377        // Delete it
378        let dp = DeleteParams::default();
379        pods.delete("busybox-kube2", &dp).await?.map_left(|pdel| {
380            assert_eq!(pdel.name_unchecked(), "busybox-kube2");
381        });
382
383        Ok(())
384    }

431                    let s = o.status.as_ref().expect("status exists on pod");
432                    let phase = s.phase.clone().unwrap_or_default();
433                    if phase == "Running" {
434                        break;
435                    }
436                }
437                WatchEvent::Error(e) => panic!("watch error: {e}"),
438                _ => {}
439            }
440        }
441
442        // Get current list of logs
443        let lp = LogParams {
444            follow: true,
445            ..LogParams::default()
446        };
447        let mut logs_stream = pods.log_stream("busybox-kube3", &lp).await?.lines();

462        // evict the pod

464        let eres = pods.evict("busybox-kube3", &ep).await?;
465        assert_eq!(eres.code, 201); // created
466        assert!(eres.is_success());
467
468        Ok(())

614    #[tokio::test]
615    #[ignore = "needs cluster for ephemeral containers operations"]
616    async fn can_operate_on_ephemeral_containers() -> Result<(), Box<dyn std::error::Error>> {
617        let client = Client::try_default().await?;
618
619        // Ephemeral containers were stabilized in Kubernetes v1.25.
620        // This test therefore exits early if the current cluster version is older than v1.25.
621        let api_version = client.apiserver_version().await?;
622        if api_version.major.parse::<i32>()? < 1 || api_version.minor.parse::<i32>()? < 25 {
623            return Ok(());
624        }
625
626        let pod: Pod = serde_json::from_value(serde_json::json!({
627            "apiVersion": "v1",
628            "kind": "Pod",
629            "metadata": {
630                "name": "ephemeral-container-test",
631                "labels": { "app": "kube-rs-test" },
632            },
633            "spec": {
634                "restartPolicy": "Never",
635                "containers": [{
636                  "name": "busybox",
637                  "image": "busybox:1.34.1",
638                  "command": ["sh", "-c", "sleep 2"],
639                }],
640            }
641        }))?;
642
643        let pod_name = pod.name_any();
644        let pods = Api::<Pod>::default_namespaced(client);
645
646        // If cleanup failed and a pod already exists, we attempt to remove it
647        // before proceeding. This is important as ephemeral containers can't
648        // be removed from a Pod's spec. Therefore this test must start with a fresh
649        // Pod every time.
650        let _ = pods
651            .delete(&pod.name_any(), &DeleteParams::default())
652            .await
653            .map(|v| v.map_left(|pdel| assert_eq!(pdel.name_any(), pod.name_any())));
654
655        // Ephemeral containes can only be applied to a running pod, so one must
656        // be created before any operations are tested.
657        match pods.create(&Default::default(), &pod).await {
658            Ok(o) => assert_eq!(pod.name_unchecked(), o.name_unchecked()),
659            Err(e) => return Err(e.into()), // any other case if a failure
660        }
661
662        let current_ephemeral_containers = pods
663            .get_ephemeral_containers(&pod.name_any())
664            .await?
665            .spec
666            .unwrap()
667            .ephemeral_containers;
668
669        // We expect no ephemeral containers initially, get_ephemeral_containers should
670        // reflect that.
671        assert_eq!(current_ephemeral_containers, None);
672
673        let mut busybox_eph: EphemeralContainer = serde_json::from_value(json!(
674            {
675                "name": "myephemeralcontainer1",
676                "image": "busybox:1.34.1",
677                "command": ["sh", "-c", "sleep 2"],
678            }
679        ))?;
680
681        // Attempt to replace ephemeral containers.
682
683        let patch: Pod = serde_json::from_value(json!({
684            "metadata": { "name": pod_name },
685            "spec":{ "ephemeralContainers": [ busybox_eph ] }
686        }))?;
687
688        let current_containers = pods
689            .replace_ephemeral_containers(&pod_name, &PostParams::default(), &patch)
690            .await?
691            .spec
692            .unwrap()
693            .ephemeral_containers
694            .expect("could find ephemeral container");
695
696        // Note that we can't compare the whole ephemeral containers object, as some fields
697        // are set by the cluster. We therefore compare the fields specified in the patch.
698        assert_eq!(current_containers.len(), 1);
699        assert_eq!(current_containers[0].name, busybox_eph.name);
700        assert_eq!(current_containers[0].image, busybox_eph.image);
701        assert_eq!(current_containers[0].command, busybox_eph.command);
702
703        // Attempt to patch ephemeral containers.
704
705        // The new ephemeral container will have different values from the
706        // first to ensure we can test for its presence.
707        busybox_eph = serde_json::from_value(json!(
708            {
709                "name": "myephemeralcontainer2",
710                "image": "busybox:1.35.0",
711                "command": ["sh", "-c", "sleep 1"],
712            }
713        ))?;
714
715        let patch: Pod =
716            serde_json::from_value(json!({ "spec": { "ephemeralContainers": [ busybox_eph ] }}))?;
717
718        let current_containers = pods
719            .patch_ephemeral_containers(&pod_name, &PatchParams::default(), &Patch::Strategic(patch))
720            .await?
721            .spec
722            .unwrap()
723            .ephemeral_containers
724            .expect("could find ephemeral container");
725
726        // There should only be 2 ephemeral containers at this point,
727        // one from each patch
728        assert_eq!(current_containers.len(), 2);
729
730        let new_container = current_containers
731            .iter()
732            .find(|c| c.name == busybox_eph.name)
733            .expect("could find myephemeralcontainer2");
734
735        // Note that we can't compare the whole ephemeral container object, as some fields
736        // get set in the cluster. We therefore compare the fields specified in the patch.
737        assert_eq!(new_container.image, busybox_eph.image);
738        assert_eq!(new_container.command, busybox_eph.command);
739
740        // Attempt to get ephemeral containers.
741
742        let expected_containers = current_containers;
743
744        let current_containers = pods
745            .get_ephemeral_containers(&pod.name_any())
746            .await?
747            .spec
748            .unwrap()
749            .ephemeral_containers
750            .unwrap();
751
752        assert_eq!(current_containers, expected_containers);
753
754        pods.delete(&pod.name_any(), &DeleteParams::default())
755            .await?
756            .map_left(|pdel| {
757                assert_eq!(pdel.name_any(), pod.name_any());
758            });
759
760        Ok(())
761    }
