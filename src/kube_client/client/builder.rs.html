19/// HTTP body of a dynamic backing type.
20///
21/// The suggested implementation type is [`hyper::Body`].
22pub type DynBody = dyn http_body::Body<Data = Bytes, Error = BoxError> + Send + Unpin;

24/// Builder for [`Client`] instances with customized [tower](`Service`) middleware.
25pub struct ClientBuilder<Svc> {
26    service: Svc,
27    default_ns: String,
28}
29
30impl<Svc> ClientBuilder<Svc> {
31    /// Construct a [`ClientBuilder`] from scratch with a fully custom [`Service`] stack.
32    ///
33    /// This method is only intended for advanced use cases, most users will want to use [`ClientBuilder::try_from`] instead,
34    /// which provides a default stack as a starting point.
35    pub fn new(service: Svc, default_namespace: impl Into<String>) -> Self
36    where
37        Svc: Service<Request<hyper::Body>>,
38    {
39        Self {
40            service,
41            default_ns: default_namespace.into(),
42        }
43    }
44
45    /// Add a [`Layer`] to the current [`Service`] stack.
46    pub fn with_layer<L: Layer<Svc>>(self, layer: &L) -> ClientBuilder<L::Service> {
47        let Self {
48            service: stack,
49            default_ns,
50        } = self;
51        ClientBuilder {
52            service: layer.layer(stack),
53            default_ns,
54        }
55    }
56
57    /// Build a [`Client`] instance with the current [`Service`] stack.
58    pub fn build<B>(self) -> Client
59    where
60        Svc: Service<Request<hyper::Body>, Response = Response<B>> + Send + 'static,
61        Svc::Future: Send + 'static,
62        Svc::Error: Into<BoxError>,
63        B: http_body::Body<Data = bytes::Bytes> + Send + 'static,
64        B::Error: Into<BoxError>,
65    {
66        Client::new(self.service, self.default_ns)
67    }
68}

70pub type GenericService = BoxService<Request<hyper::Body>, Response<Box<DynBody>>, BoxError>;
71
72impl TryFrom<Config> for ClientBuilder<GenericService> {
73    type Error = Error;
74
75    /// Builds a default [`ClientBuilder`] stack from a given configuration
76    fn try_from(config: Config) -> Result<Self> {
77        let mut connector = HttpConnector::new();
78        connector.enforce_http(false);
79
80        #[cfg(feature = "socks5")]
81        if let Some(proxy_addr) = config.proxy_url.clone() {
82            let connector = hyper_socks2::SocksConnector {
83                proxy_addr,
84                auth: None,
85                connector,
86            };
87
88            return make_generic_builder(connector, config);
89        }
90
91        make_generic_builder(connector, config)
92    }
93}
94
95/// Helper function for implementation of [`TryFrom<Config>`] for [`ClientBuilder`].
96/// Ignores [`Config::proxy_url`], which at this point is already handled.
97fn make_generic_builder<H>(connector: H, config: Config) -> Result<ClientBuilder<GenericService>, Error>
98where
99    H: 'static + Clone + Send + Sync + Service<http::Uri>,
100    H::Response: 'static + Connection + AsyncRead + AsyncWrite + Send + Unpin,
101    H::Future: 'static + Send,
102    H::Error: 'static + Send + Sync + std::error::Error,
103{
104    let default_ns = config.default_namespace.clone();
105    let auth_layer = config.auth_layer()?;
106
107    let client: hyper::Client<_, hyper::Body> = {
108        // Current TLS feature precedence when more than one are set:
109        // 1. rustls-tls
110        // 2. openssl-tls
111        // Create a custom client to use something else.
112        // If TLS features are not enabled, http connector will be used.
113        #[cfg(feature = "rustls-tls")]
114        let connector = config.rustls_https_connector_with_connector(connector)?;
115        #[cfg(all(not(feature = "rustls-tls"), feature = "openssl-tls"))]
116        let connector = config.openssl_https_connector_with_connector(connector)?;
117        #[cfg(all(not(feature = "rustls-tls"), not(feature = "openssl-tls")))]
118        if config.cluster_url.scheme() == Some(&http::uri::Scheme::HTTPS) {
119            // no tls stack situation only works with http scheme
120            return Err(Error::TlsRequired);
121        }
122
123        let mut connector = TimeoutConnector::new(connector);
124
125        // Set the timeouts for the client
126        connector.set_connect_timeout(config.connect_timeout);
127        connector.set_read_timeout(config.read_timeout);
128        connector.set_write_timeout(config.write_timeout);
129
130        hyper::Client::builder().build(connector)
131    };
132
133    let stack = ServiceBuilder::new().layer(config.base_uri_layer()).into_inner();
134    #[cfg(feature = "gzip")]
135    let stack = ServiceBuilder::new()
136        .layer(stack)
137        .layer(tower_http::decompression::DecompressionLayer::new())
138        .into_inner();
139
140    let service = ServiceBuilder::new()
141        .layer(stack)
142        .option_layer(auth_layer)
143        .layer(config.extra_headers_layer()?)
144        .layer(
145            // Attribute names follow [Semantic Conventions].
146            // [Semantic Conventions]: https://github.com/open-telemetry/opentelemetry-specification/blob/main/specification/trace/semantic_conventions/http.md
147            TraceLayer::new_for_http()
148                .make_span_with(|req: &Request<hyper::Body>| {
149                    tracing::debug_span!(
150                        "HTTP",
151                         http.method = %req.method(),
152                         http.url = %req.uri(),
153                         http.status_code = tracing::field::Empty,
154                         otel.name = req.extensions().get::<&'static str>().unwrap_or(&"HTTP"),
155                         otel.kind = "client",
156                         otel.status_code = tracing::field::Empty,
157                    )
158                })
159                .on_request(|_req: &Request<hyper::Body>, _span: &Span| {
160                    tracing::debug!("requesting");
161                })
162                .on_response(|res: &Response<hyper::Body>, _latency: Duration, span: &Span| {
163                    let status = res.status();
164                    span.record("http.status_code", status.as_u16());
165                    if status.is_client_error() || status.is_server_error() {
166                        span.record("otel.status_code", "ERROR");
167                    }
168                })
169                // Explicitly disable `on_body_chunk`. The default does nothing.
170                .on_body_chunk(())
171                .on_eos(|_: Option<&HeaderMap>, _duration: Duration, _span: &Span| {
172                    tracing::debug!("stream closed");
173                })
174                .on_failure(|ec: ServerErrorsFailureClass, _latency: Duration, span: &Span| {
175                    // Called when
176                    // - Calling the inner service errored
177                    // - Polling `Body` errored
178                    // - the response was classified as failure (5xx)
179                    // - End of stream was classified as failure
180                    span.record("otel.status_code", "ERROR");
181                    match ec {
182                        ServerErrorsFailureClass::StatusCode(status) => {
183                            span.record("http.status_code", status.as_u16());
184                            tracing::error!("failed with status {}", status)
185                        }
186                        ServerErrorsFailureClass::Error(err) => {
187                            tracing::error!("failed with error {}", err)
188                        }
189                    }
190                }),
191        )
192        .service(client);
193
194    Ok(ClientBuilder::new(
195        BoxService::new(
196            MapResponseBodyLayer::new(|body| {
197                Box::new(http_body::Body::map_err(body, BoxError::from)) as Box<DynBody>
198            })
199            .layer(service),
200        ),
201        default_ns,
202    ))
203}
