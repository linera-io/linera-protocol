117// Token file reference. Reloads at least once per minute.
118#[derive(Debug)]
119pub struct TokenFile {
120    path: PathBuf,
121    token: SecretString,
122    expires_at: DateTime<Utc>,
123}
124
125impl TokenFile {
126    fn new<P: AsRef<Path>>(path: P) -> Result<TokenFile, Error> {
127        let token = std::fs::read_to_string(&path)
128            .map_err(|source| Error::ReadTokenFile(source, path.as_ref().to_owned()))?;
129        Ok(Self {
130            path: path.as_ref().to_owned(),
131            token: SecretString::from(token),
132            // Try to reload at least once a minute
133            expires_at: Utc::now() + Duration::seconds(60),
134        })
135    }
136
137    fn is_expiring(&self) -> bool {
138        Utc::now() + Duration::seconds(10) > self.expires_at
139    }
140
141    /// Get the cached token. Returns `None` if it's expiring.
142    fn cached_token(&self) -> Option<&str> {
143        (!self.is_expiring()).then(|| self.token.expose_secret().as_ref())
144    }
145
146    /// Get a token. Reloads from file if the cached token is expiring.
147    fn token(&mut self) -> &str {
148        if self.is_expiring() {
149            // > If reload from file fails, the last-read token should be used to avoid breaking
150            // > clients that make token files available on process start and then remove them to
151            // > limit credential exposure.
152            // > https://github.com/kubernetes/kubernetes/issues/68164
153            if let Ok(token) = std::fs::read_to_string(&self.path) {
154                self.token = SecretString::from(token);
155            }
156            self.expires_at = Utc::now() + Duration::seconds(60);
157        }
158        self.token.expose_secret()
159    }
160}

162// See https://github.com/kubernetes/kubernetes/tree/master/staging/src/k8s.io/client-go/plugin/pkg/client/auth
163// for the list of auth-plugins supported by client-go.
164// We currently support the following:
165// - exec
166// - token-file refreshed at least once per minute
167// - gcp: command based token source (exec)
168// - gcp: application credential based token source (requires `oauth` feature)

170// Note that the visibility must be `pub` for `impl Layer for AuthLayer`, but this is not exported from the crate.
171// It's not accessible from outside and not shown on docs.
172#[derive(Debug, Clone)]
173pub enum RefreshableToken {
174    Exec(Arc<Mutex<(SecretString, DateTime<Utc>, AuthInfo)>>),
175    File(Arc<RwLock<TokenFile>>),
176    #[cfg(feature = "oauth")]
177    GcpOauth(Arc<Mutex<oauth::Gcp>>),
178    #[cfg(feature = "oidc")]
179    Oidc(Arc<Mutex<oidc::Oidc>>),
180}

182// For use with `AsyncFilterLayer` to add `Authorization` header with a refreshed token.
183impl<B> AsyncPredicate<Request<B>> for RefreshableToken
184where
185    B: http_body::Body + Send + 'static,
186{
187    type Future = BoxFuture<'static, Result<Request<B>, BoxError>>;
188    type Request = Request<B>;
189
190    fn check(&mut self, mut request: Self::Request) -> Self::Future {
191        let refreshable = self.clone();
192        Box::pin(async move {
193            refreshable.to_header().await.map_err(Into::into).map(|value| {
194                request.headers_mut().insert(AUTHORIZATION, value);
195                request
196            })
197        })
198    }
199}

201impl RefreshableToken {
202    async fn to_header(&self) -> Result<HeaderValue, Error> {
203        match self {
204            RefreshableToken::Exec(data) => {
205                let mut locked_data = data.lock().await;
206                // Add some wiggle room onto the current timestamp so we don't get any race
207                // conditions where the token expires while we are refreshing
208                if Utc::now() + Duration::seconds(60) >= locked_data.1 {
209                    // TODO Improve refreshing exec to avoid `Auth::try_from`
210                    match Auth::try_from(&locked_data.2)? {
211                        Auth::None | Auth::Basic(_, _) | Auth::Bearer(_) | Auth::Certificate(_, _) => {
212                            return Err(Error::UnrefreshableTokenResponse);
213                        }
214
215                        Auth::RefreshableToken(RefreshableToken::Exec(d)) => {
216                            let (new_token, new_expire, new_info) = Arc::try_unwrap(d)
217                                .expect("Unable to unwrap Arc, this is likely a programming error")
218                                .into_inner();
219                            locked_data.0 = new_token;
220                            locked_data.1 = new_expire;
221                            locked_data.2 = new_info;
222                        }
223
224                        // Unreachable because the token source does not change
225                        Auth::RefreshableToken(RefreshableToken::File(_)) => unreachable!(),
226                        #[cfg(feature = "oauth")]
227                        Auth::RefreshableToken(RefreshableToken::GcpOauth(_)) => unreachable!(),
228                        #[cfg(feature = "oidc")]
229                        Auth::RefreshableToken(RefreshableToken::Oidc(_)) => unreachable!(),
230                    }
231                }
232
233                bearer_header(locked_data.0.expose_secret())
234            }
235
236            RefreshableToken::File(token_file) => {
237                let guard = token_file.read().await;
238                if let Some(header) = guard.cached_token().map(bearer_header) {
239                    return header;
240                }
241                // Drop the read guard before a write lock attempt to prevent deadlock.
242                drop(guard);
243                // Note that `token()` only reloads if the cached token is expiring.
244                // A separate method to conditionally reload minimizes the need for an exclusive access.
245                bearer_header(token_file.write().await.token())
246            }
247
248            #[cfg(feature = "oauth")]
249            RefreshableToken::GcpOauth(data) => {
250                let gcp_oauth = data.lock().await;
251                let token = (*gcp_oauth).token().await.map_err(Error::OAuth)?;
252                bearer_header(&token.access_token)
253            }
254
255            #[cfg(feature = "oidc")]
256            RefreshableToken::Oidc(oidc) => {
257                let token = oidc.lock().await.id_token().await.map_err(Error::Oidc)?;
258                bearer_header(&token)
259            }
260        }
261    }
262}
263
264fn bearer_header(token: &str) -> Result<HeaderValue, Error> {
265    let mut value = HeaderValue::try_from(format!("Bearer {token}")).map_err(Error::InvalidBearerToken)?;
266    value.set_sensitive(true);
267    Ok(value)
268}

270impl TryFrom<&AuthInfo> for Auth {
271    type Error = Error;
272
273    /// Loads the authentication header from the credentials available in the kubeconfig. This supports
274    /// exec plugins as well as specified in
275    /// https://kubernetes.io/docs/reference/access-authn-authz/authentication/#client-go-credential-plugins
276    fn try_from(auth_info: &AuthInfo) -> Result<Self, Self::Error> {
277        if let Some(provider) = &auth_info.auth_provider {
278            match token_from_provider(provider)? {
279                #[cfg(feature = "oidc")]
280                ProviderToken::Oidc(oidc) => {
281                    return Ok(Self::RefreshableToken(RefreshableToken::Oidc(Arc::new(
282                        Mutex::new(oidc),
283                    ))));
284                }
285
286                #[cfg(not(feature = "oidc"))]
287                ProviderToken::Oidc(token) => {
288                    return Ok(Self::Bearer(SecretString::from(token)));
289                }
290
291                ProviderToken::GcpCommand(token, Some(expiry)) => {
292                    let mut info = auth_info.clone();
293                    let mut provider = provider.clone();
294                    provider.config.insert("access-token".into(), token.clone());
295                    provider.config.insert("expiry".into(), expiry.to_rfc3339());
296                    info.auth_provider = Some(provider);
297                    return Ok(Self::RefreshableToken(RefreshableToken::Exec(Arc::new(
298                        Mutex::new((SecretString::from(token), expiry, info)),
299                    ))));
300                }
301
302                ProviderToken::GcpCommand(token, None) => {
303                    return Ok(Self::Bearer(SecretString::from(token)));
304                }
305
306                #[cfg(feature = "oauth")]
307                ProviderToken::GcpOauth(gcp) => {
308                    return Ok(Self::RefreshableToken(RefreshableToken::GcpOauth(Arc::new(
309                        Mutex::new(gcp),
310                    ))));
311                }
312            }
313        }
314
315        if let (Some(u), Some(p)) = (&auth_info.username, &auth_info.password) {
316            return Ok(Self::Basic(u.to_owned(), p.to_owned()));
317        }
318
319        // Inline token. Has precedence over `token_file`.
320        if let Some(token) = &auth_info.token {
321            return Ok(Self::Bearer(token.clone()));
322        }
323
324        // Token file reference. Must be reloaded at least once a minute.
325        if let Some(file) = &auth_info.token_file {
326            return Ok(Self::RefreshableToken(RefreshableToken::File(Arc::new(
327                RwLock::new(TokenFile::new(file)?),
328            ))));
329        }
330
331        if let Some(exec) = &auth_info.exec {
332            let creds = auth_exec(exec)?;
333            let status = creds.status.ok_or(Error::ExecPluginFailed)?;
334            if let (Some(client_certificate_data), Some(client_key_data)) =
335                (status.client_certificate_data, status.client_key_data)
336            {
337                return Ok(Self::Certificate(client_certificate_data, client_key_data.into()));
338            }
339            let expiration = status
340                .expiration_timestamp
341                .map(|ts| ts.parse())
342                .transpose()
343                .map_err(Error::MalformedTokenExpirationDate)?;
344            match (status.token.map(SecretString::from), expiration) {
345                (Some(token), Some(expire)) => Ok(Self::RefreshableToken(RefreshableToken::Exec(Arc::new(
346                    Mutex::new((token, expire, auth_info.clone())),
347                )))),
348                (Some(token), None) => Ok(Self::Bearer(token)),
349                _ => Ok(Self::None),
350            }
351        } else {
352            Ok(Self::None)
353        }
354    }
355}

357// We need to differentiate providers because the keys/formats to store token expiration differs.
358enum ProviderToken {
359    #[cfg(feature = "oidc")]
360    Oidc(oidc::Oidc),
361    #[cfg(not(feature = "oidc"))]
362    Oidc(String),
363    // "access-token", "expiry" (RFC3339)
364    GcpCommand(String, Option<DateTime<Utc>>),
365    #[cfg(feature = "oauth")]
366    GcpOauth(oauth::Gcp),
367    // "access-token", "expires-on" (timestamp)
368    // Azure(String, Option<DateTime<Utc>>),
369}
370
371fn token_from_provider(provider: &AuthProviderConfig) -> Result<ProviderToken, Error> {
372    match provider.name.as_ref() {
373        "oidc" => token_from_oidc_provider(provider),
374        "gcp" => token_from_gcp_provider(provider),
375        "azure" => Err(Error::AuthExec(
376            "The azure auth plugin is not supported; use https://github.com/Azure/kubelogin instead".into(),
377        )),
378        _ => Err(Error::AuthExec(format!(
379            "Authentication with provider {:} not supported",
380            provider.name
381        ))),
382    }
383}

385#[cfg(feature = "oidc")]
386fn token_from_oidc_provider(provider: &AuthProviderConfig) -> Result<ProviderToken, Error> {
387    oidc::Oidc::from_config(&provider.config)
388        .map_err(Error::Oidc)
389        .map(ProviderToken::Oidc)

392#[cfg(not(feature = "oidc"))]
393fn token_from_oidc_provider(provider: &AuthProviderConfig) -> Result<ProviderToken, Error> {
394    match provider.config.get("id-token") {
395        Some(id_token) => Ok(ProviderToken::Oidc(id_token.clone())),
396        None => Err(Error::AuthExec(
397            "No id-token for oidc Authentication provider".into(),
398        )),
399    }

402fn token_from_gcp_provider(provider: &AuthProviderConfig) -> Result<ProviderToken, Error> {
403    if let Some(id_token) = provider.config.get("id-token") {
404        return Ok(ProviderToken::GcpCommand(id_token.clone(), None));
405    }
406
407    // Return cached access token if it's still valid
408    if let Some(access_token) = provider.config.get("access-token") {
409        if let Some(expiry) = provider.config.get("expiry") {
410            let expiry_date = expiry
411                .parse::<DateTime<Utc>>()
412                .map_err(Error::MalformedTokenExpirationDate)?;
413            if Utc::now() + Duration::seconds(60) < expiry_date {
414                return Ok(ProviderToken::GcpCommand(access_token.clone(), Some(expiry_date)));
415            }
416        }
417    }
418
419    // Command-based token source
420    if let Some(cmd) = provider.config.get("cmd-path") {
421        let params = provider.config.get("cmd-args").cloned().unwrap_or_default();
422        // NB: This property does currently not exist upstream in client-go
423        // See https://github.com/kube-rs/kube/issues/1060
424        let drop_env = provider.config.get("cmd-drop-env").cloned().unwrap_or_default();
425        // TODO splitting args by space is not safe
426        let mut command = Command::new(cmd);
427        // Do not pass the following env vars to the command
428        for env in drop_env.trim().split(' ') {
429            command.env_remove(env);
430        }
431        let output = command
432            .args(params.trim().split(' '))
433            .output()
434            .map_err(|e| Error::AuthExec(format!("Executing {cmd:} failed: {e:?}")))?;
435
436        if !output.status.success() {
437            return Err(Error::AuthExecRun {
438                cmd: format!("{cmd} {params}"),
439                status: output.status,
440                out: output,
441            });
442        }
443
444        if let Some(field) = provider.config.get("token-key") {
445            let json_output: serde_json::Value =
446                serde_json::from_slice(&output.stdout).map_err(Error::ParseTokenKey)?;
447            let token = extract_value(&json_output, "token-key", field)?;
448            if let Some(field) = provider.config.get("expiry-key") {
449                let expiry = extract_value(&json_output, "expiry-key", field)?;
450                let expiry = expiry
451                    .parse::<DateTime<Utc>>()
452                    .map_err(Error::MalformedTokenExpirationDate)?;
453                return Ok(ProviderToken::GcpCommand(token, Some(expiry)));
454            } else {
455                return Ok(ProviderToken::GcpCommand(token, None));
456            }
457        } else {
458            let token = std::str::from_utf8(&output.stdout)
459                .map_err(|e| Error::AuthExec(format!("Result is not a string {e:?} ")))?
460                .to_owned();
461            return Ok(ProviderToken::GcpCommand(token, None));
462        }
463    }
464
465    // Google Application Credentials-based token source
466    #[cfg(feature = "oauth")]
467    {
468        Ok(ProviderToken::GcpOauth(
469            oauth::Gcp::default_credentials_with_scopes(provider.config.get("scopes"))
470                .map_err(Error::OAuth)?,
471        ))
472    }
473    #[cfg(not(feature = "oauth"))]
474    {
475        Err(Error::AuthExec(
476            "Enable oauth feature to use Google Application Credentials-based token source".into(),
477        ))
478    }
479}

481fn extract_value(json: &serde_json::Value, context: &str, path: &str) -> Result<String, Error> {
482    let parsed_path = path
483        .trim_matches(|c| c == '"' || c == '{' || c == '}')
484        .parse::<JsonPathInst>()
485        .map_err(|err| {
486            Error::AuthExec(format!(
487                "Failed to parse {context:?} as a JsonPath: {path}\n
488                 Error: {err}"
489            ))
490        })?;
491
492    let res = parsed_path.find_slice(json);
493
494    let Some(res) = res.into_iter().next() else {
495        return Err(Error::AuthExec(format!(
496            "Target {context:?} value {path:?} not found"
497        )));
498    };
499
500    if let Some(val) = res.as_str() {
501        Ok(val.to_owned())
502    } else {
503        Err(Error::AuthExec(format!(
504            "Target {:?} value {:?} is not a string: {:?}",
505            context, path, *res
506        )))
507    }
508}

537    pub expiration_timestamp: Option<String>,
538    pub token: Option<String>,

545fn auth_exec(auth: &ExecConfig) -> Result<ExecCredential, Error> {
546    let mut cmd = match &auth.command {
547        Some(cmd) => Command::new(cmd),
548        None => return Err(Error::MissingCommand),
549    };
550
551    if let Some(args) = &auth.args {
552        cmd.args(args);
553    }
554    if let Some(env) = &auth.env {
555        let envs = env
556            .iter()
557            .flat_map(|env| match (env.get("name"), env.get("value")) {
558                (Some(name), Some(value)) => Some((name, value)),
559                _ => None,
560            });
561        cmd.envs(envs);
562    }
563
564    let interactive = auth.interactive_mode != Some(ExecInteractiveMode::Never);
565    if interactive {
566        cmd.stdin(std::process::Stdio::inherit());
567    } else {
568        cmd.stdin(std::process::Stdio::piped());
569    }
570
571    let mut exec_credential_spec = ExecCredentialSpec {
572        interactive: Some(interactive),
573        cluster: None,
574    };
575
576    if auth.provide_cluster_info {
577        exec_credential_spec.cluster = Some(auth.cluster.clone().ok_or(Error::ExecMissingClusterInfo)?);
578    }
579
580    // Provide exec info to child process
581    let exec_info = serde_json::to_string(&ExecCredential {
582        api_version: auth.api_version.clone(),
583        kind: "ExecCredential".to_string().into(),
584        spec: Some(exec_credential_spec),
585        status: None,
586    })
587    .map_err(Error::AuthExecSerialize)?;
588    cmd.env("KUBERNETES_EXEC_INFO", exec_info);
589
590    if let Some(envs) = &auth.drop_env {
591        for env in envs {
592            cmd.env_remove(env);
593        }
594    }
595
596    #[cfg(target_os = "windows")]
597    {
598        const CREATE_NO_WINDOW: u32 = 0x08000000;
599        cmd.creation_flags(CREATE_NO_WINDOW);
600    }
601
602    let out = cmd.output().map_err(Error::AuthExecStart)?;
603    if !out.status.success() {
604        return Err(Error::AuthExecRun {
605            cmd: format!("{cmd:?}"),
606            status: out.status,
607            out,
608        });
609    }
610    let creds = serde_json::from_slice(&out.stdout).map_err(Error::AuthExecParse)?;
611
612    Ok(creds)
613}

668    fn token_file() {

671        let mut token_file = TokenFile::new(file.path()).unwrap();
672        assert_eq!(token_file.cached_token().unwrap(), "token1");
673        assert!(!token_file.is_expiring());
674        assert_eq!(token_file.token(), "token1");
675        // Doesn't reload unless expiring
676        std::fs::write(file.path(), "token2").unwrap();
677        assert_eq!(token_file.token(), "token1");
678
679        token_file.expires_at = Utc::now();
680        assert!(token_file.is_expiring());
681        assert_eq!(token_file.cached_token(), None);
682        assert_eq!(token_file.token(), "token2");
683        assert!(!token_file.is_expiring());
684        assert_eq!(token_file.cached_token().unwrap(), "token2");
