14/// This struct implements a protection-swapping assembling buffer
15#[derive(Debug)]
16pub struct MemoryManager {
17    // buffer where the end result is copied into
18    execbuffer: Arc<RwLock<ExecutableBuffer>>,
19
20    // size of the allocated mmap (so we don't have to go through RwLock to get it)
21    execbuffer_size: usize,
22    // length of the allocated mmap that has been written into
23    asmoffset: usize,
24
25    // the address that the current execbuffer starts at
26    execbuffer_addr: usize
27}
28
29impl MemoryManager {
30    /// Create a new memory manager, with `initial_mmap_size` data allocated
31    pub fn new(initial_mmap_size: usize) -> io::Result<Self> {
32        let execbuffer = ExecutableBuffer::new(initial_mmap_size)?;
33        let execbuffer_addr = execbuffer.as_ptr() as usize;
34
35        Ok(MemoryManager {
36            execbuffer: Arc::new(RwLock::new(execbuffer)),
37            execbuffer_size: initial_mmap_size,
38            asmoffset: 0,
39            execbuffer_addr
40        })
41    }
42
43    /// Returns the amount of bytes already committed to the manager
44    pub fn committed(&self) -> usize {
45        self.asmoffset
46    }
47
48    /// Returns the current start address of the managed executable memory
49    pub fn execbuffer_addr(&self) -> usize {
50        self.execbuffer_addr
51    }
52
53    /// Commits the data from `new` into the managed memory, calling `f` when the buffer is moved to fix anything
54    /// that relies on the address of the buffer
55    pub fn commit<F>(&mut self, new: &mut Vec<u8>, f: F) where F: FnOnce(&mut [u8], usize, usize) {
56        let old_asmoffset = self.asmoffset;
57        let new_asmoffset = self.asmoffset + new.len();
58
59        if old_asmoffset >= new_asmoffset {
60            return;
61        }
62
63        // see if we need to request a new buffer
64        if new_asmoffset > self.execbuffer_size {
65            while self.execbuffer_size <= new_asmoffset {
66                self.execbuffer_size *= 2;
67            }
68
69            // create a larger writable buffer
70            let mut new_buffer = MutableBuffer::new(self.execbuffer_size).expect("Could not allocate a larger buffer");
71            new_buffer.set_len(new_asmoffset);
72
73            // copy over the data
74            new_buffer[.. old_asmoffset].copy_from_slice(&self.execbuffer.read().unwrap());
75            new_buffer[old_asmoffset..].copy_from_slice(&new);
76            let new_buffer_addr = new_buffer.as_ptr() as usize;
77
78            // allow modifications to be made
79            f(&mut new_buffer, self.execbuffer_addr, new_buffer_addr);
80
81            // swap the buffers
82            self.execbuffer_addr = new_buffer_addr;
83            *self.execbuffer.write().unwrap() = new_buffer.make_exec().expect("Could not swap buffer protection modes")
84
85        } else {
86
87            // temporarily change the buffer protection modes and copy in new data
88            let mut lock = self.write();
89            let buffer = mem::replace(&mut *lock, ExecutableBuffer::default());
90            let mut buffer = buffer.make_mut().expect("Could not swap buffer protection modes");
91
92            // update buffer and length
93            buffer.set_len(new_asmoffset);
94            buffer[old_asmoffset..].copy_from_slice(&new);
95
96            // repack the buffer
97            let buffer = buffer.make_exec().expect("Could not swap buffer protection modes");
98            *lock = buffer;
99        }
100
101        new.clear();
102        self.asmoffset = new_asmoffset;
103    }
104
105    /// Borrow the internal memory buffer mutably
106    pub fn write(&self) -> RwLockWriteGuard<ExecutableBuffer> {
107        self.execbuffer.write().unwrap()
108    }
109
110    /// finalizes the currently committed part of the buffer.
111    pub fn finalize(self) -> Result<ExecutableBuffer, Self> {
112        match Arc::try_unwrap(self.execbuffer) {
113            Ok(execbuffer) => Ok(execbuffer.into_inner().unwrap()),
114            Err(arc) => Err(Self {
115                execbuffer: arc,
116                ..self
117            })
118        }
119    }
120
121    /// Create an atomically refcounted reference to the internal executable buffer
122    pub fn reader(&self) -> Arc<RwLock<ExecutableBuffer>> {
123        self.execbuffer.clone()
124    }
125}

128/// A registry of labels. Contains all necessessities for keeping track of dynasm labels.
129/// This is useful when implementing your own assembler and can also be used to query
130/// assemblers for the offsets of labels.
131#[derive(Debug, Clone, Default)]
132pub struct LabelRegistry {
133    // mapping of global labels to offsets
134    global_labels: HashMap<&'static str, AssemblyOffset>,
135    // mapping of local labels to offsets
136    local_labels: HashMap<&'static str, AssemblyOffset>,
137    // mapping of dynamic label ids to offsets
138    dynamic_labels: Vec<Option<AssemblyOffset>>,
139}
140
141impl LabelRegistry {
142    /// Create a new, empty label registry
143    pub fn new() -> LabelRegistry {
144        LabelRegistry {
145            global_labels: HashMap::new(),
146            local_labels: HashMap::new(),
147            dynamic_labels: Vec::new(),
148        }
149    }
150
151    /// Create a new dynamic label id
152    pub fn new_dynamic_label(&mut self) -> DynamicLabel {
153        let id = self.dynamic_labels.len();
154        self.dynamic_labels.push(None);
155        DynamicLabel(id)
156    }
157
158    /// Define a the dynamic label `id` to be located at `offset`.
159    pub fn define_dynamic(&mut self, id: DynamicLabel, offset: AssemblyOffset) -> Result<(), DynasmError> {
160        let entry = &mut self.dynamic_labels[id.0];
161        if entry.is_some() {
162            return Err(DynasmError::DuplicateLabel(LabelKind::Dynamic(id)));
163        }
164
165        *entry = Some(offset);
166        Ok(())
167    }
168
169    /// Define a the global label `name` to be located at `offset`.
170    pub fn define_global(&mut self, name: &'static str, offset: AssemblyOffset) -> Result<(), DynasmError> {
171        match self.global_labels.entry(name) {
172            Entry::Occupied(_) => Err(DynasmError::DuplicateLabel(LabelKind::Global(name))),
173            Entry::Vacant(v) => {
174                v.insert(offset);
175                Ok(())
176            }
177        }
178    }
179
180    /// Define a the local label `name` to be located at `offset`.
181    pub fn define_local(&mut self, name: &'static str, offset: AssemblyOffset) {
182        self.local_labels.insert(name, offset);
183    }
184
185    /// Returns the offset at which the dynamic label `id` was defined, if one was defined.
186    pub fn resolve_dynamic(&self, id: DynamicLabel) -> Result<AssemblyOffset, DynasmError> {
187        self.dynamic_labels.get(id.0).and_then(|&e| e).ok_or_else(|| DynasmError::UnknownLabel(LabelKind::Dynamic(id)))
188    }
189
190    /// Returns the offset at which the global label `name` was defined, if one was defined.
191    pub fn resolve_global(&self, name: &'static str) -> Result<AssemblyOffset, DynasmError> {
192        self.global_labels.get(&name).cloned().ok_or_else(|| DynasmError::UnknownLabel(LabelKind::Global(name)))
193    }
194
195    /// Returns the offset at which the last local label named `id` was defined, if one was defined.
196    pub fn resolve_local(&self, name: &'static str) -> Result<AssemblyOffset, DynasmError> {
197        self.local_labels.get(&name).cloned().ok_or_else(|| DynasmError::UnknownLabel(LabelKind::Local(name)))
198    }
199}

202/// An abstraction of a relocation of type `R`, located at `location`.
203#[derive(Clone, Debug)]
204pub struct PatchLoc<R: Relocation> {
205    /// The AssemblyOffset at which this relocation was emitted
206    pub location: AssemblyOffset,
207    /// The offset, backwards, from location that the actual field to be modified starts at
208    pub field_offset: u8,
209    /// The offset, backwards, to be subtracted from location to get the address that the relocation should be calculated relative to.
210    pub ref_offset: u8,
211    /// The type of relocation to be emitted.
212    pub relocation: R,
213    /// A constant offset added to the destination address of this relocation when it is calculated.
214    pub target_offset: isize,
215}
216
217impl<R: Relocation> PatchLoc<R> {
218    /// create a new `PatchLoc`
219    pub fn new(location: AssemblyOffset, target_offset: isize, field_offset: u8, ref_offset: u8, relocation: R) -> PatchLoc<R> {
220        PatchLoc {
221            location,
222            field_offset,
223            ref_offset,
224            relocation,
225            target_offset
226        }
227    }
228
229    /// Returns a range that covers the entire relocation in its assembling buffer
230    /// `buf_offset` is a value that is subtracted from this range when the buffer you want to slice
231    /// with this range is only a part of a bigger buffer.
232    pub fn range(&self, buf_offset: usize) -> std::ops::Range<usize> {
233        let field_offset = self.location.0 - buf_offset -  self.field_offset as usize;
234        field_offset .. field_offset + self.relocation.size()
235    }
236
237    /// Returns the actual value that should be inserted at the relocation site.
238    pub fn value(&self, target: usize, buf_addr: usize) -> isize {
239        (match self.relocation.kind() {
240            RelocationKind::Relative => target.wrapping_sub(self.location.0 - self.ref_offset as usize),
241            RelocationKind::RelToAbs => target.wrapping_sub(self.location.0 - self.ref_offset as usize + buf_addr),
242            RelocationKind::AbsToRel => target + buf_addr
243        }) as isize + self.target_offset
244    }
245
246    /// Patch `buffer` so that this relocation patch will point to `target`.
247    /// `buf_addr` is the address that the assembling buffer will come to reside at when it is assembled.
248    /// `target` is the offset that this relocation will be targetting.
249    pub fn patch(&self, buffer: &mut [u8], buf_addr: usize, target: usize) -> Result<(), ImpossibleRelocation> {
250        let value = self.value(target, buf_addr);
251        self.relocation.write_value(buffer, value)
252    }
253
254    /// Patch `buffer` so that this relocation will still point to the right location due to a change in the address of the containing buffer.
255    /// `buffer` is a subsection of a larger buffer, located at offset `buf_offset` in this larger buffer.
256    /// `adjustment` is `new_buf_addr - old_buf_addr`.
257    pub fn adjust(&self, buffer: &mut [u8], adjustment: isize) -> Result<(), ImpossibleRelocation> {
258        let value = self.relocation.read_value(buffer);
259        let value = match self.relocation.kind() {
260            RelocationKind::Relative => value,
261            RelocationKind::RelToAbs => value.wrapping_sub(adjustment),
262            RelocationKind::AbsToRel => value.wrapping_add(adjustment),
263        };
264        self.relocation.write_value(buffer, value)
265    }
266
267    /// Returns if this patch requires adjustment when the address of the buffer it resides in is altered.
268    pub fn needs_adjustment(&self) -> bool {
269        match self.relocation.kind() {
270            RelocationKind::Relative => false,
271            RelocationKind::RelToAbs
272            | RelocationKind::AbsToRel => true,
273        }
274    }
275}

278/// A registry of relocations and the respective labels they point towards.
279#[derive(Debug, Default)]
280pub struct RelocRegistry<R: Relocation> {
281    global: Vec<(PatchLoc<R>, &'static str)>,
282    dynamic: Vec<(PatchLoc<R>, DynamicLabel)>,
283    local: HashMap<&'static str, Vec<PatchLoc<R>>>
284}
285
286impl<R: Relocation> RelocRegistry<R> {
287    /// Create a new, empty relocation registry.
288    pub fn new() -> RelocRegistry<R> {
289        RelocRegistry {
290            global: Vec::new(),
291            dynamic: Vec::new(),
292            local: HashMap::new()
293        }
294    }
295
296    /// Add a new patch targetting the global label `name`.
297    pub fn add_global(&mut self, name: &'static str, patchloc: PatchLoc<R>) {
298        self.global.push((patchloc, name));
299    }
300
301    /// Add a new patch targetting the dynamic label `id`.
302    pub fn add_dynamic(&mut self, id: DynamicLabel, patchloc: PatchLoc<R>) {
303        self.dynamic.push((patchloc, id))
304    }
305
306    /// Add a new patch targetting the next local label `name`.
307    /// As any relocation targetting a previous local label can be immediately resolved these should not be recorded.
308    pub fn add_local(&mut self, name: &'static str, patchloc: PatchLoc<R>) {
309        match self.local.entry(name) {
310            Entry::Occupied(mut o) => o.get_mut().push(patchloc),
311            Entry::Vacant(v) => {
312                v.insert(vec![patchloc]);
313            }
314        }
315    }
316
317    /// Return an iterator through all defined relocations targetting local label `name`.
318    /// These relocations are removed from the registry.
319    pub fn take_locals_named<'a>(&'a mut self, name: &'static str) -> impl Iterator<Item=PatchLoc<R>> + 'a {
320        self.local.get_mut(&name).into_iter().flat_map(|v| v.drain(..))
321    }
322
323    /// Return an iterator through all defined relocations targeting global labels and the labels they target.
324    /// These relocations are removed from the registry.
325    pub fn take_globals<'a>(&'a mut self) -> impl Iterator<Item=(PatchLoc<R>, &'static str)> + 'a {
326        self.global.drain(..)
327    }
328
329    /// Return an iterator through all defined relocations targeting dynamic labels and the labels they target.
330    /// These relocations are removed from the registry.
331    pub fn take_dynamics<'a>(&'a mut self) -> impl Iterator<Item=(PatchLoc<R>, DynamicLabel)> + 'a {
332        self.dynamic.drain(..)
333    }
334
335    /// Return an iterator through all defined relocations targeting local labels and the labels they target.
336    /// These relocations are removed from the registry.
337    pub fn take_locals<'a>(&'a mut self) -> impl Iterator<Item=(PatchLoc<R>, &'static str)> + 'a {
338        self.local.iter_mut().flat_map(|(&k, v)| v.drain(..).map(move |p| (p, k)))
339    }
340}

343/// A registry of relocations that have been encoded previously, but need to be adjusted when the address of the buffer they
344/// reside in changes.
345#[derive(Debug, Default)]
346pub struct ManagedRelocs<R: Relocation> {
347    managed: BTreeMap<usize, PatchLoc<R>>
348}
349
350impl<R: Relocation> ManagedRelocs<R> {
351    /// Create a new, empty managed relocation registry.
352    pub fn new() -> Self {
353        Self {
354            managed: BTreeMap::new()
355        }
356    }
357
358    /// Add a relocation to this registry.
359    pub fn add(&mut self, patchloc: PatchLoc<R>) {
360        self.managed.insert(patchloc.location.0 - patchloc.field_offset as usize, patchloc);
361    }
362
363    /// Take all items from another registry and add them to this registry
364    pub fn append(&mut self, other: &mut ManagedRelocs<R>) {
365        self.managed.append(&mut other.managed);
366    }
367
368    /// Remove all managed relocations whose byte fields start in the range start .. end.
369    /// This is useful when implementing an `Assembler::alter` API, as any managed relocations
370    /// that were overwritten should be removed from the registry, otherwise the replacement code
371    /// would be corrupted when managed relocations are updated.
372    pub fn remove_between(&mut self, start: usize, end: usize) {
373        if start == end {
374            return;
375        }
376
377        let keys: Vec<_> = self.managed.range(start .. end).map(|(&k, _)| k).collect();
378        for k in keys {
379            self.managed.remove(&k);
380        }
381    }
382
383    /// Iterate through all defined managed relocations.
384    pub fn iter<'a>(&'a self) -> impl Iterator<Item=&'a PatchLoc<R>> + 'a {
385        self.managed.values()
386    } 
387}

390#[derive(Clone, Debug)]
391enum LitPoolEntry {
392    U8(u8),
393    U16(u16),
394    U32(u32),
395    U64(u64),
396    Dynamic(RelocationSize, DynamicLabel),
397    Global(RelocationSize, &'static str),
398    Forward(RelocationSize, &'static str),
399    Backward(RelocationSize, &'static str),
400    Align(u8, usize),
401}
402
403/// Literal pool implementation. One can programmatically push items in this literal pool and retrieve offsets to them in the pool.
404/// Then later, the pool will be encoded into the instruction stream and items can be retrieved using the address of the literal pool.
405/// and the previously emitted offsets. Values are always at least aligned to their size.
406#[derive(Clone, Debug, Default)]
407pub struct LitPool {
408    offset: usize,
409    entries: Vec<LitPoolEntry>,
410}
411
412impl LitPool {
413    /// Create a new, empty literal pool
414    pub fn new() -> Self {
415        LitPool {
416            offset: 0,
417            entries: Vec::new(),
418        }
419    }
420
421    // align the pool to the specified size, record the offset, and bump the offset
422    fn bump_offset(&mut self, size: RelocationSize) -> isize {
423        // Correct for alignment
424        self.align(size as usize, 0);
425        let offset = self.offset;
426        self.offset += size as usize;
427        offset as isize
428    }
429
430    /// Add extra alignment for the next value in the literal pool
431    pub fn align(&mut self, size: usize, with: u8) {
432        let misalign = self.offset % (size as usize);
433        if misalign == 0 {
434            return;
435        }
436
437        self.entries.push(LitPoolEntry::Align(with, size));
438        self.offset += size as usize - misalign;
439    }
440
441    /// Encode `value` into the literal pool.
442    pub fn push_u8(&mut self, value: u8) -> isize {
443        let offset = self.bump_offset(RelocationSize::Byte);
444        self.entries.push(LitPoolEntry::U8(value));
445        offset
446    }
447
448    /// Encode `value` into the literal pool.
449    pub fn push_u16(&mut self, value: u16) -> isize {
450        let offset = self.bump_offset(RelocationSize::Word);
451        self.entries.push(LitPoolEntry::U16(value));
452        offset
453    }
454
455    /// Encode `value` into the literal pool.
456    pub fn push_u32(&mut self, value: u32) -> isize {
457        let offset = self.bump_offset(RelocationSize::DWord);
458        self.entries.push(LitPoolEntry::U32(value));
459        offset
460    }
461
462    /// Encode `value` into the literal pool.
463    pub fn push_u64(&mut self, value: u64) -> isize {
464        let offset = self.bump_offset(RelocationSize::QWord);
465        self.entries.push(LitPoolEntry::U64(value));
466        offset
467    }
468
469    /// Encode the relative address of a label into the literal pool (relative to the location in the pool)
470    pub fn push_dynamic(&mut self, id: DynamicLabel, size: RelocationSize) -> isize {
471        let offset = self.bump_offset(size);
472        self.entries.push(LitPoolEntry::Dynamic(size, id));
473        offset
474    }
475
476    /// Encode the relative address of a label into the literal pool (relative to the location in the pool)
477    pub fn push_global(&mut self, name: &'static str, size: RelocationSize) -> isize {
478        let offset = self.bump_offset(size);
479        self.entries.push(LitPoolEntry::Global(size, name));
480        offset
481    }
482
483    /// Encode the relative address of a label into the literal pool (relative to the location in the pool)
484    pub fn push_forward(&mut self, name: &'static str, size: RelocationSize) -> isize {
485        let offset = self.bump_offset(size);
486        self.entries.push(LitPoolEntry::Forward(size, name));
487        offset
488    }
489
490    /// Encode the relative address of a label into the literal pool (relative to the location in the pool)
491    pub fn push_backward(&mut self, name: &'static str, size: RelocationSize) -> isize {
492        let offset = self.bump_offset(size);
493        self.entries.push(LitPoolEntry::Backward(size, name));
494        offset
495    }
496
497    fn pad_sized<D: DynasmLabelApi>(size: RelocationSize, assembler: &mut D) {
498        match size {
499            RelocationSize::Byte => assembler.push(0),
500            RelocationSize::Word => assembler.push_u16(0),
501            RelocationSize::DWord => assembler.push_u32(0),
502            RelocationSize::QWord => assembler.push_u64(0),
503        }
504    }
505
506    /// Emit this literal pool into the specified assembler
507    pub fn emit<D: DynasmLabelApi>(self, assembler: &mut D) {
508        for entry in self.entries {
509            match entry {
510                LitPoolEntry::U8(value) => assembler.push(value),
511                LitPoolEntry::U16(value) => assembler.push_u16(value),
512                LitPoolEntry::U32(value) => assembler.push_u32(value),
513                LitPoolEntry::U64(value) => assembler.push_u64(value),
514                LitPoolEntry::Dynamic(size, id) => {
515                    Self::pad_sized(size, assembler);
516                    assembler.dynamic_relocation(id, 0, size as u8, size as u8, D::Relocation::from_size(size));
517                },
518                LitPoolEntry::Global(size, name) => {
519                    Self::pad_sized(size, assembler);
520                    assembler.global_relocation(name, 0, size as u8, size as u8, D::Relocation::from_size(size));
521                },
522                LitPoolEntry::Forward(size, name) => {
523                    Self::pad_sized(size, assembler);
524                    assembler.forward_relocation(name, 0, size as u8, size as u8, D::Relocation::from_size(size));
525                },
526                LitPoolEntry::Backward(size, name) => {
527                    Self::pad_sized(size, assembler);
528                    assembler.backward_relocation(name, 0, size as u8, size as u8, D::Relocation::from_size(size));
529                },
530                LitPoolEntry::Align(with, alignment) => assembler.align(alignment, with),
531            }
532        }
533    }
534}

562    fn test_litpool<R: Relocation + Debug>() {
563        let mut ops = Assembler::<R>::new().unwrap();
564        let dynamic1 = ops.new_dynamic_label();
565
566        let mut pool = components::LitPool::new();
567

569
570        assert_eq!(pool.push_u8(0x12), 0);
571        assert_eq!(pool.push_u8(0x34), 1);
572        assert_eq!(pool.push_u8(0x56), 2);
573
574        assert_eq!(pool.push_u16(0x789A), 4);
575
576        assert_eq!(pool.push_u32(0xBCDE_F012), 8);
577
578        assert_eq!(pool.push_u64(0x3456_789A_BCDE_F012), 16);
579
580        assert_eq!(pool.push_forward("forward1", RelocationSize::Byte), 24);
581
582        pool.align(4, 0xCC);
583
584        assert_eq!(pool.push_global("global1", RelocationSize::Word), 28);
585
586        assert_eq!(pool.push_dynamic(dynamic1, RelocationSize::DWord), 32);
587
588        assert_eq!(pool.push_backward("backward1", RelocationSize::QWord), 40);
589
590        pool.emit(&mut ops);
591
592        assert_eq!(ops.offset().0, 48);
593
594        ops.local_label("forward1");
595        ops.global_label("global1");
596        ops.dynamic_label(dynamic1);
597
598        assert_eq!(ops.commit(), Ok(()));
599        let buf = ops.finalize().unwrap();
600
601        assert_eq!(&*buf, &[
602            0x12, 0x34, 0x56, 0x00, 0x9A, 0x78, 0x00, 0x00,
603            0x12, 0xF0, 0xDE, 0xBC, 0x00, 0x00, 0x00, 0x00,
604            0x12, 0xF0, 0xDE, 0xBC, 0x9A, 0x78, 0x56, 0x34,
605            24  , 0xCC, 0xCC, 0xCC, 20  , 0   , 0x00, 0x00,
606            16  , 0   , 0   , 0   , 0x00, 0x00, 0x00, 0x00,
607            0xD8, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFF, 0xFFu8, 
608        ] as &[u8]);
609    }
