104    /// Handles a request and applies it to the chain state.
105    #[instrument(skip_all)]
106    pub async fn handle_request(&mut self, request: ChainWorkerRequest<StorageClient::Context>) {
107        tracing::trace!("Handling chain worker request: {request:?}");
108        // TODO(#2237): Spawn concurrent tasks for read-only operations
109        let responded = match request {
110            #[cfg(with_testing)]
111            ChainWorkerRequest::ReadCertificate { height, callback } => {
112                callback.send(self.read_certificate(height).await).is_ok()
113            }
114            ChainWorkerRequest::GetChainStateView { callback } => {
115                callback.send(self.chain_state_view().await).is_ok()
116            }
117            ChainWorkerRequest::QueryApplication { query, callback } => {
118                callback.send(self.query_application(query).await).is_ok()
119            }
120            ChainWorkerRequest::DescribeApplication {
121                application_id,
122                callback,
123            } => callback
124                .send(self.describe_application(application_id).await)
125                .is_ok(),
126            ChainWorkerRequest::StageBlockExecution {
127                block,
128                round,
129                published_blobs,
130                callback,
131            } => callback
132                .send(
133                    self.stage_block_execution(block, round, &published_blobs)
134                        .await,
135                )
136                .is_ok(),
137            ChainWorkerRequest::ProcessTimeout {
138                certificate,
139                callback,
140            } => callback
141                .send(self.process_timeout(certificate).await)
142                .is_ok(),
143            ChainWorkerRequest::HandleBlockProposal { proposal, callback } => callback
144                .send(self.handle_block_proposal(proposal).await)
145                .is_ok(),
146            ChainWorkerRequest::ProcessValidatedBlock {
147                certificate,
148                callback,
149            } => callback
150                .send(self.process_validated_block(certificate).await)
151                .is_ok(),
152            ChainWorkerRequest::ProcessConfirmedBlock {
153                certificate,
154                notify_when_messages_are_delivered,
155                callback,
156            } => callback
157                .send(
158                    self.process_confirmed_block(certificate, notify_when_messages_are_delivered)
159                        .await,
160                )
161                .is_ok(),
162            ChainWorkerRequest::ProcessCrossChainUpdate {
163                origin,
164                bundles,
165                callback,
166            } => callback
167                .send(self.process_cross_chain_update(origin, bundles).await)
168                .is_ok(),
169            ChainWorkerRequest::ConfirmUpdatedRecipient {
170                recipient,
171                latest_height,
172                callback,
173            } => callback
174                .send(
175                    self.confirm_updated_recipient(recipient, latest_height)
176                        .await,
177                )
178                .is_ok(),
179            ChainWorkerRequest::HandleChainInfoQuery { query, callback } => callback
180                .send(self.handle_chain_info_query(query).await)
181                .is_ok(),
182            ChainWorkerRequest::DownloadPendingBlob { blob_id, callback } => callback
183                .send(self.download_pending_blob(blob_id).await)
184                .is_ok(),
185            ChainWorkerRequest::HandlePendingBlob { blob, callback } => {
186                callback.send(self.handle_pending_blob(blob).await).is_ok()
187            }
188            ChainWorkerRequest::UpdateReceivedCertificateTrackers {
189                new_trackers,
190                callback,
191            } => callback
192                .send(
193                    self.update_received_certificate_trackers(new_trackers)
194                        .await,
195                )
196                .is_ok(),
197        };
198
199        if !responded {
200            warn!("Callback for `ChainWorkerActor` was dropped before a response was sent");
201        }
202    }

204    /// Returns a read-only view of the [`ChainStateView`].
205    ///
206    /// The returned view holds a lock on the chain state, which prevents the worker from changing
207    /// it.
208    pub(super) async fn chain_state_view(
209        &mut self,
210    ) -> Result<OwnedRwLockReadGuard<ChainStateView<StorageClient::Context>>, WorkerError> {
211        if self.shared_chain_view.is_none() {
212            self.shared_chain_view = Some(Arc::new(RwLock::new(self.chain.clone_unchecked()?)));
213        }
214
215        Ok(self
216            .shared_chain_view
217            .as_ref()
218            .expect("`shared_chain_view` should be initialized above")
219            .clone()
220            .read_owned()
221            .await)
222    }

283    /// Handles a proposal for the next block for this chain.
284    #[tracing::instrument(level = "debug", skip(self))]
285    pub(super) async fn handle_block_proposal(
286        &mut self,
287        proposal: BlockProposal,
288    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
289        self.ensure_is_active().await?;
290        if ChainWorkerStateWithTemporaryChanges::new(&mut *self)
291            .await
292            .check_proposed_block(&proposal)
293            .await?
294            == manager::Outcome::Skip
295        {
296            // Skipping: We already voted for this block.
297            let info = ChainInfoResponse::new(&self.chain, self.config.key_pair());
298            return Ok((info, NetworkActions::default()));
299        };
300        let published_blobs = ChainWorkerStateWithAttemptedChanges::new(&mut *self)
301            .await
302            .load_proposal_blobs(&proposal)
303            .await?;
304        let (outcome, local_time) = ChainWorkerStateWithTemporaryChanges::new(self)
305            .await
306            .validate_proposal_content(&proposal.content, &published_blobs)
307            .await?;
308
309        ChainWorkerStateWithAttemptedChanges::new(&mut *self)
310            .await
311            .vote_for_block_proposal(proposal, outcome, local_time)
312            .await?;
313        let actions = self.create_network_actions().await?;
314
315        let info = ChainInfoResponse::new(&self.chain, self.config.key_pair());
316        Ok((info, actions))
317    }

319    /// Clears the shared chain view, and acquires and drops its write lock.
320    ///
321    /// This is the only place a write lock is acquired, and read locks are acquired in
322    /// the `chain_state_view` method, which has a `&mut self` receiver like this one.
323    /// That means that when this function returns, no readers will be waiting to acquire
324    /// the lock and it is safe to write the chain state to storage without any readers
325    /// having a stale view of it.
326    pub(super) async fn clear_shared_chain_view(&mut self) {
327        if let Some(shared_chain_view) = self.shared_chain_view.take() {
328            let _: RwLockWriteGuard<_> = shared_chain_view.write().await;
329        }
330    }

382    /// Handles a [`ChainInfoQuery`], potentially voting on the next block.
383    #[tracing::instrument(level = "debug", skip(self))]
384    pub(super) async fn handle_chain_info_query(
385        &mut self,
386        query: ChainInfoQuery,
387    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
388        if let Some((height, round)) = query.request_leader_timeout {
389            ChainWorkerStateWithAttemptedChanges::new(&mut *self)
390                .await
391                .vote_for_leader_timeout(height, round)
392                .await?;
393        }
394        if query.request_fallback {
395            ChainWorkerStateWithAttemptedChanges::new(&mut *self)
396                .await
397                .vote_for_fallback()
398                .await?;
399        }
400        let response = ChainWorkerStateWithTemporaryChanges::new(self)
401            .await
402            .prepare_chain_info_response(query)
403            .await?;
404        // Trigger any outgoing cross-chain messages that haven't been confirmed yet.
405        let actions = self.create_network_actions().await?;
406        Ok((response, actions))
407    }

430    /// Ensures that the current chain is active, returning an error otherwise.
431    async fn ensure_is_active(&mut self) -> Result<(), WorkerError> {
432        if !self.knows_chain_is_active {
433            let local_time = self.storage.clock().current_time();
434            self.chain.ensure_is_active(local_time).await?;
435            self.clear_shared_chain_view().await;
436            self.chain.save().await?;
437            self.knows_chain_is_active = true;
438        }
439        Ok(())
440    }

442    /// Reads the blobs from the chain manager or from storage. Returns an error if any are
443    /// missing.
444    async fn get_required_blobs(
445        &self,
446        required_blob_ids: impl IntoIterator<Item = BlobId>,
447        created_blobs: &BTreeMap<BlobId, Blob>,
448    ) -> Result<BTreeMap<BlobId, Blob>, WorkerError> {
449        let maybe_blobs = self
450            .maybe_get_required_blobs(required_blob_ids, Some(created_blobs))
451            .await?;
452        let not_found_blob_ids = missing_blob_ids(&maybe_blobs);
453        ensure!(
454            not_found_blob_ids.is_empty(),
455            WorkerError::BlobsNotFound(not_found_blob_ids)
456        );
457        Ok(maybe_blobs
458            .into_iter()
459            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
460            .collect())
461    }

463    /// Tries to read the blobs from the chain manager or storage. Returns `None` if not found.
464    async fn maybe_get_required_blobs(
465        &self,
466        blob_ids: impl IntoIterator<Item = BlobId>,
467        created_blobs: Option<&BTreeMap<BlobId, Blob>>,
468    ) -> Result<BTreeMap<BlobId, Option<Blob>>, WorkerError> {
469        let mut maybe_blobs = BTreeMap::from_iter(blob_ids.into_iter().zip(iter::repeat(None)));
470
471        for (blob_id, maybe_blob) in &mut maybe_blobs {
472            if let Some(blob) = created_blobs.and_then(|blob_map| blob_map.get(blob_id)) {
473                *maybe_blob = Some(blob.clone());
474            } else if let Some(blob) = self.chain.manager.pending_blob(blob_id).await? {
475                *maybe_blob = Some(blob);
476            } else if let Some(blob) = self.chain.pending_validated_blobs.get(blob_id).await? {
477                *maybe_blob = Some(blob);
478            } else {
479                for (_, pending_blobs) in self
480                    .chain
481                    .pending_proposed_blobs
482                    .try_load_all_entries()
483                    .await?
484                {
485                    if let Some(blob) = pending_blobs.get(blob_id).await? {
486                        *maybe_blob = Some(blob);
487                        break;
488                    }
489                }
490            }
491        }
492        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
493        let blobs_from_storage = self.storage.read_blobs(&missing_blob_ids).await?;
494        for (blob_id, maybe_blob) in missing_blob_ids.into_iter().zip(blobs_from_storage) {
495            maybe_blobs.insert(blob_id, maybe_blob);
496        }
497        Ok(maybe_blobs)
498    }

500    /// Adds any newly created chains to the set of `tracked_chains`, if the parent chain is
501    /// also tracked.
502    ///
503    /// Chains that are not tracked are usually processed only because they sent some message
504    /// to one of the tracked chains. In most use cases, their children won't be of interest.
505    fn track_newly_created_chains(
506        &self,
507        proposed_block: &ProposedBlock,
508        outcome: &BlockExecutionOutcome,
509    ) {
510        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
511            if !tracked_chains
512                .read()
513                .expect("Panics should not happen while holding a lock to `tracked_chains`")
514                .contains(&proposed_block.chain_id)
515            {
516                return; // The parent chain is not tracked; don't track the child.
517            }
518            let new_chain_ids = outcome
519                .created_blobs_ids()
520                .into_iter()
521                .filter(|blob_id| blob_id.blob_type == BlobType::ChainDescription)
522                .map(|blob_id| ChainId(blob_id.hash));
523
524            tracked_chains
525                .write()
526                .expect("Panics should not happen while holding a lock to `tracked_chains`")
527                .extend(new_chain_ids);
528        }
529    }

531    /// Loads pending cross-chain requests.
532    async fn create_network_actions(&self) -> Result<NetworkActions, WorkerError> {
533        let mut heights_by_recipient = BTreeMap::<_, Vec<_>>::new();
534        let mut targets = self.chain.nonempty_outbox_chain_ids();
535        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
536            let tracked_chains = tracked_chains
537                .read()
538                .expect("Panics should not happen while holding a lock to `tracked_chains`");
539            targets.retain(|target| tracked_chains.contains(target));
540        }
541        let outboxes = self.chain.load_outboxes(&targets).await?;
542        for (target, outbox) in targets.into_iter().zip(outboxes) {
543            let heights = outbox.queue.elements().await?;
544            heights_by_recipient.insert(target, heights);
545        }
546        self.create_cross_chain_requests(heights_by_recipient).await
547    }
548
549    async fn create_cross_chain_requests(
550        &self,
551        heights_by_recipient: BTreeMap<ChainId, Vec<BlockHeight>>,
552    ) -> Result<NetworkActions, WorkerError> {
553        // Load all the certificates we will need, regardless of the medium.
554        let heights = BTreeSet::from_iter(heights_by_recipient.values().flatten().copied());
555        let next_block_height = self.chain.tip_state.get().next_block_height;
556        let log_heights = heights
557            .range(..next_block_height)
558            .copied()
559            .map(usize::try_from)
560            .collect::<Result<Vec<_>, _>>()?;
561        let mut hashes = self
562            .chain
563            .confirmed_log
564            .multi_get(log_heights)
565            .await?
566            .into_iter()
567            .zip(&heights)
568            .map(|(maybe_hash, height)| {
569                maybe_hash.ok_or_else(|| WorkerError::ConfirmedLogEntryNotFound {
570                    height: *height,
571                    chain_id: self.chain_id(),
572                })
573            })
574            .collect::<Result<Vec<_>, _>>()?;
575        for height in heights.range(next_block_height..) {
576            hashes.push(
577                self.chain
578                    .preprocessed_blocks
579                    .get(height)
580                    .await?
581                    .ok_or_else(|| WorkerError::PreprocessedBlocksEntryNotFound {
582                        height: *height,
583                        chain_id: self.chain_id(),
584                    })?,
585            );
586        }
587        let certificates = self.storage.read_certificates(hashes.clone()).await?;
588        let certificates = match ResultReadCertificates::new(certificates, hashes) {
589            ResultReadCertificates::Certificates(certificates) => certificates,
590            ResultReadCertificates::InvalidHashes(hashes) => {
591                return Err(WorkerError::ReadCertificatesError(hashes))
592            }
593        };
594        let certificates = heights
595            .into_iter()
596            .zip(certificates)
597            .collect::<HashMap<_, _>>();
598        // For each medium, select the relevant messages.
599        let mut actions = NetworkActions::default();
600        for (recipient, heights) in heights_by_recipient {
601            let mut bundles = Vec::new();
602            for height in heights {
603                let cert = certificates
604                    .get(&height)
605                    .ok_or_else(|| ChainError::InternalError("missing certificates".to_string()))?;
606                bundles.extend(cert.message_bundles_for(recipient));
607            }
608            let request = CrossChainRequest::UpdateRecipient {
609                sender: self.chain.chain_id(),
610                recipient,
611                bundles,
612            };
613            actions.cross_chain_requests.push(request);
614        }
615        Ok(actions)
616    }

618    /// Returns true if there are no more outgoing messages in flight up to the given
619    /// block height.
620    pub async fn all_messages_to_tracked_chains_delivered_up_to(
621        &self,
622        height: BlockHeight,
623    ) -> Result<bool, WorkerError> {
624        if self.chain.all_messages_delivered_up_to(height) {
625            return Ok(true);
626        }
627        let Some(tracked_chains) = self.tracked_chains.as_ref() else {
628            return Ok(false);
629        };
630        let mut targets = self.chain.nonempty_outbox_chain_ids();
631        {
632            let tracked_chains = tracked_chains.read().unwrap();
633            targets.retain(|target| tracked_chains.contains(target));
634        }
635        let outboxes = self.chain.load_outboxes(&targets).await?;
636        for outbox in outboxes {
637            let front = outbox.queue.front();
638            if front.is_some_and(|key| *key <= height) {
639                return Ok(false);
640            }
641        }
642        Ok(true)
643    }
644
645    /// Updates the received certificate trackers to at least the given values.
646    pub async fn update_received_certificate_trackers(
647        &mut self,
648        new_trackers: BTreeMap<ValidatorPublicKey, u64>,
649    ) -> Result<(), WorkerError> {
650        ChainWorkerStateWithAttemptedChanges::new(self)
651            .await
652            .update_received_certificate_trackers(new_trackers)
653            .await
654    }
655}

657/// Returns the keys whose value is `None`.
658fn missing_blob_ids(maybe_blobs: &BTreeMap<BlobId, Option<Blob>>) -> Vec<BlobId> {
659    maybe_blobs
660        .iter()
661        .filter(|(_, maybe_blob)| maybe_blob.is_none())
662        .map(|(blob_id, _)| *blob_id)
663        .collect()
664}

667fn check_block_epoch(
668    chain_epoch: Epoch,
669    block_chain: ChainId,
670    block_epoch: Epoch,
671) -> Result<(), WorkerError> {
672    ensure!(
673        block_epoch == chain_epoch,
674        WorkerError::InvalidEpoch {
675            chain_id: block_chain,
676            epoch: block_epoch,
677            chain_epoch
678        }
679    );
