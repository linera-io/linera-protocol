105    /// Handles a request and applies it to the chain state.
106    #[instrument(skip_all)]
107    pub async fn handle_request(&mut self, request: ChainWorkerRequest<StorageClient::Context>) {
108        tracing::trace!("Handling chain worker request: {request:?}");
109        // TODO(#2237): Spawn concurrent tasks for read-only operations
110        let responded = match request {
111            #[cfg(with_testing)]
112            ChainWorkerRequest::ReadCertificate { height, callback } => {
113                callback.send(self.read_certificate(height).await).is_ok()
114            }
115            ChainWorkerRequest::GetChainStateView { callback } => {
116                callback.send(self.chain_state_view().await).is_ok()
117            }
118            ChainWorkerRequest::QueryApplication { query, callback } => {
119                callback.send(self.query_application(query).await).is_ok()
120            }
121            ChainWorkerRequest::DescribeApplication {
122                application_id,
123                callback,
124            } => callback
125                .send(self.describe_application(application_id).await)
126                .is_ok(),
127            ChainWorkerRequest::StageBlockExecution {
128                block,
129                round,
130                published_blobs,
131                callback,
132            } => callback
133                .send(
134                    self.stage_block_execution(block, round, &published_blobs)
135                        .await,
136                )
137                .is_ok(),
138            ChainWorkerRequest::ProcessTimeout {
139                certificate,
140                callback,
141            } => callback
142                .send(self.process_timeout(certificate).await)
143                .is_ok(),
144            ChainWorkerRequest::HandleBlockProposal { proposal, callback } => callback
145                .send(self.handle_block_proposal(proposal).await)
146                .is_ok(),
147            ChainWorkerRequest::ProcessValidatedBlock {
148                certificate,
149                callback,
150            } => callback
151                .send(self.process_validated_block(certificate).await)
152                .is_ok(),
153            ChainWorkerRequest::ProcessConfirmedBlock {
154                certificate,
155                notify_when_messages_are_delivered,
156                callback,
157            } => callback
158                .send(
159                    self.process_confirmed_block(certificate, notify_when_messages_are_delivered)
160                        .await,
161                )
162                .is_ok(),
163            ChainWorkerRequest::ProcessCrossChainUpdate {
164                origin,
165                bundles,
166                callback,
167            } => callback
168                .send(self.process_cross_chain_update(origin, bundles).await)
169                .is_ok(),
170            ChainWorkerRequest::ConfirmUpdatedRecipient {
171                recipient,
172                latest_height,
173                callback,
174            } => callback
175                .send(
176                    self.confirm_updated_recipient(recipient, latest_height)
177                        .await,
178                )
179                .is_ok(),
180            ChainWorkerRequest::HandleChainInfoQuery { query, callback } => callback
181                .send(self.handle_chain_info_query(query).await)
182                .is_ok(),
183            ChainWorkerRequest::DownloadPendingBlob { blob_id, callback } => callback
184                .send(self.download_pending_blob(blob_id).await)
185                .is_ok(),
186            ChainWorkerRequest::HandlePendingBlob { blob, callback } => {
187                callback.send(self.handle_pending_blob(blob).await).is_ok()
188            }
189            ChainWorkerRequest::UpdateReceivedCertificateTrackers {
190                new_trackers,
191                callback,
192            } => callback
193                .send(
194                    self.update_received_certificate_trackers(new_trackers)
195                        .await,
196                )
197                .is_ok(),
198        };
199
200        if !responded {
201            warn!("Callback for `ChainWorkerActor` was dropped before a response was sent");
202        }
203
204        // Roll back any unsaved changes to the chain state: If there was an error while trying
205        // to handle the request, the chain state might contain unsaved and potentially invalid
206        // changes. The next request needs to be applied to the chain state as it is in storage.
207        self.chain.rollback();
208    }

210    /// Returns a read-only view of the [`ChainStateView`].
211    ///
212    /// The returned view holds a lock on the chain state, which prevents the worker from changing
213    /// it.
214    pub(super) async fn chain_state_view(
215        &mut self,
216    ) -> Result<OwnedRwLockReadGuard<ChainStateView<StorageClient::Context>>, WorkerError> {
217        if self.shared_chain_view.is_none() {
218            self.shared_chain_view = Some(Arc::new(RwLock::new(self.chain.clone_unchecked())));
219        }
220
221        Ok(self
222            .shared_chain_view
223            .as_ref()
224            .expect("`shared_chain_view` should be initialized above")
225            .clone()
226            .read_owned()
227            .await)
228    }

230    /// Clears the shared chain view, and acquires and drops its write lock.
231    ///
232    /// This is the only place a write lock is acquired, and read locks are acquired in
233    /// the `chain_state_view` method, which has a `&mut self` receiver like this one.
234    /// That means that when this function returns, no readers will be waiting to acquire
235    /// the lock and it is safe to write the chain state to storage without any readers
236    /// having a stale view of it.
237    pub(super) async fn clear_shared_chain_view(&mut self) {
238        if let Some(shared_chain_view) = self.shared_chain_view.take() {
239            let _: RwLockWriteGuard<_> = shared_chain_view.write().await;
240        }
241    }

243    /// Handles a [`ChainInfoQuery`], potentially voting on the next block.
244    #[tracing::instrument(level = "debug", skip(self))]
245    pub(super) async fn handle_chain_info_query(
246        &mut self,
247        query: ChainInfoQuery,
248    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
249        if let Some((height, round)) = query.request_leader_timeout {
250            self.vote_for_leader_timeout(height, round).await?;
251        }
252        if query.request_fallback {
253            self.vote_for_fallback().await?;
254        }
255        let response = self.prepare_chain_info_response(query).await?;
256        // Trigger any outgoing cross-chain messages that haven't been confirmed yet.
257        let actions = self.create_network_actions().await?;
258        Ok((response, actions))
259    }

261    /// Returns the requested blob, if it belongs to the current locking block or pending proposal.
262    pub(super) async fn download_pending_blob(&self, blob_id: BlobId) -> Result<Blob, WorkerError> {
263        if let Some(blob) = self.chain.manager.pending_blob(&blob_id).await? {
264            return Ok(blob);
265        }
266        let blob = self.storage.read_blob(blob_id).await?;
267        blob.ok_or(WorkerError::BlobsNotFound(vec![blob_id]))
268    }

270    /// Reads the blobs from the chain manager or from storage. Returns an error if any are
271    /// missing.
272    async fn get_required_blobs(
273        &self,
274        required_blob_ids: impl IntoIterator<Item = BlobId>,
275        created_blobs: &BTreeMap<BlobId, Blob>,
276    ) -> Result<BTreeMap<BlobId, Blob>, WorkerError> {
277        let maybe_blobs = self
278            .maybe_get_required_blobs(required_blob_ids, Some(created_blobs))
279            .await?;
280        let not_found_blob_ids = missing_blob_ids(&maybe_blobs);
281        ensure!(
282            not_found_blob_ids.is_empty(),
283            WorkerError::BlobsNotFound(not_found_blob_ids)
284        );
285        Ok(maybe_blobs
286            .into_iter()
287            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
288            .collect())
289    }

291    /// Tries to read the blobs from the chain manager or storage. Returns `None` if not found.
292    async fn maybe_get_required_blobs(
293        &self,
294        blob_ids: impl IntoIterator<Item = BlobId>,
295        created_blobs: Option<&BTreeMap<BlobId, Blob>>,
296    ) -> Result<BTreeMap<BlobId, Option<Blob>>, WorkerError> {
297        let mut maybe_blobs = BTreeMap::from_iter(blob_ids.into_iter().zip(iter::repeat(None)));
298
299        for (blob_id, maybe_blob) in &mut maybe_blobs {
300            if let Some(blob) = created_blobs.and_then(|blob_map| blob_map.get(blob_id)) {
301                *maybe_blob = Some(blob.clone());
302            } else if let Some(blob) = self.chain.manager.pending_blob(blob_id).await? {
303                *maybe_blob = Some(blob);
304            } else if let Some(blob) = self.chain.pending_validated_blobs.get(blob_id).await? {
305                *maybe_blob = Some(blob);
306            } else {
307                for (_, pending_blobs) in self
308                    .chain
309                    .pending_proposed_blobs
310                    .try_load_all_entries()
311                    .await?
312                {
313                    if let Some(blob) = pending_blobs.get(blob_id).await? {
314                        *maybe_blob = Some(blob);
315                        break;
316                    }
317                }
318            }
319        }
320        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
321        let blobs_from_storage = self.storage.read_blobs(&missing_blob_ids).await?;
322        for (blob_id, maybe_blob) in missing_blob_ids.into_iter().zip(blobs_from_storage) {
323            maybe_blobs.insert(blob_id, maybe_blob);
324        }
325        Ok(maybe_blobs)
326    }

328    /// Adds any newly created chains to the set of `tracked_chains`, if the parent chain is
329    /// also tracked.
330    ///
331    /// Chains that are not tracked are usually processed only because they sent some message
332    /// to one of the tracked chains. In most use cases, their children won't be of interest.
333    fn track_newly_created_chains(
334        &self,
335        proposed_block: &ProposedBlock,
336        outcome: &BlockExecutionOutcome,
337    ) {
338        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
339            if !tracked_chains
340                .read()
341                .expect("Panics should not happen while holding a lock to `tracked_chains`")
342                .contains(&proposed_block.chain_id)
343            {
344                return; // The parent chain is not tracked; don't track the child.
345            }
346            let new_chain_ids = outcome
347                .created_blobs_ids()
348                .into_iter()
349                .filter(|blob_id| blob_id.blob_type == BlobType::ChainDescription)
350                .map(|blob_id| ChainId(blob_id.hash));
351
352            tracked_chains
353                .write()
354                .expect("Panics should not happen while holding a lock to `tracked_chains`")
355                .extend(new_chain_ids);
356        }
357    }

359    /// Loads pending cross-chain requests.
360    async fn create_network_actions(&self) -> Result<NetworkActions, WorkerError> {
361        let mut heights_by_recipient = BTreeMap::<_, Vec<_>>::new();
362        let mut targets = self.chain.nonempty_outbox_chain_ids();
363        if let Some(tracked_chains) = self.tracked_chains.as_ref() {
364            let tracked_chains = tracked_chains
365                .read()
366                .expect("Panics should not happen while holding a lock to `tracked_chains`");
367            targets.retain(|target| tracked_chains.contains(target));
368        }
369        let outboxes = self.chain.load_outboxes(&targets).await?;
370        for (target, outbox) in targets.into_iter().zip(outboxes) {
371            let heights = outbox.queue.elements().await?;
372            heights_by_recipient.insert(target, heights);
373        }
374        self.create_cross_chain_requests(heights_by_recipient).await
375    }

377    async fn create_cross_chain_requests(
378        &self,
379        heights_by_recipient: BTreeMap<ChainId, Vec<BlockHeight>>,
380    ) -> Result<NetworkActions, WorkerError> {
381        // Load all the certificates we will need, regardless of the medium.
382        let heights = BTreeSet::from_iter(heights_by_recipient.values().flatten().copied());
383        let next_block_height = self.chain.tip_state.get().next_block_height;
384        let log_heights = heights
385            .range(..next_block_height)
386            .copied()
387            .map(usize::try_from)
388            .collect::<Result<Vec<_>, _>>()?;
389        let mut hashes = self
390            .chain
391            .confirmed_log
392            .multi_get(log_heights)
393            .await?
394            .into_iter()
395            .zip(&heights)
396            .map(|(maybe_hash, height)| {
397                maybe_hash.ok_or_else(|| WorkerError::ConfirmedLogEntryNotFound {
398                    height: *height,
399                    chain_id: self.chain_id(),
400                })
401            })
402            .collect::<Result<Vec<_>, _>>()?;
403        for height in heights.range(next_block_height..) {
404            hashes.push(
405                self.chain
406                    .preprocessed_blocks
407                    .get(height)
408                    .await?
409                    .ok_or_else(|| WorkerError::PreprocessedBlocksEntryNotFound {
410                        height: *height,
411                        chain_id: self.chain_id(),
412                    })?,
413            );
414        }
415        let certificates = self.storage.read_certificates(hashes.clone()).await?;
416        let certificates = match ResultReadCertificates::new(certificates, hashes) {
417            ResultReadCertificates::Certificates(certificates) => certificates,
418            ResultReadCertificates::InvalidHashes(hashes) => {
419                return Err(WorkerError::ReadCertificatesError(hashes))
420            }
421        };
422        let certificates = heights
423            .into_iter()
424            .zip(certificates)
425            .collect::<HashMap<_, _>>();
426        // For each medium, select the relevant messages.
427        let mut actions = NetworkActions::default();
428        for (recipient, heights) in heights_by_recipient {
429            let mut bundles = Vec::new();
430            for height in heights {
431                let cert = certificates
432                    .get(&height)
433                    .ok_or_else(|| ChainError::InternalError("missing certificates".to_string()))?;
434                bundles.extend(cert.message_bundles_for(recipient));
435            }
436            let request = CrossChainRequest::UpdateRecipient {
437                sender: self.chain.chain_id(),
438                recipient,
439                bundles,
440            };
441            actions.cross_chain_requests.push(request);
442        }
443        Ok(actions)
444    }

446    /// Returns true if there are no more outgoing messages in flight up to the given
447    /// block height.
448    pub async fn all_messages_to_tracked_chains_delivered_up_to(
449        &self,
450        height: BlockHeight,
451    ) -> Result<bool, WorkerError> {
452        if self.chain.all_messages_delivered_up_to(height) {
453            return Ok(true);
454        }
455        let Some(tracked_chains) = self.tracked_chains.as_ref() else {
456            return Ok(false);
457        };
458        let mut targets = self.chain.nonempty_outbox_chain_ids();
459        {
460            let tracked_chains = tracked_chains.read().unwrap();
461            targets.retain(|target| tracked_chains.contains(target));
462        }
463        let outboxes = self.chain.load_outboxes(&targets).await?;
464        for outbox in outboxes {
465            let front = outbox.queue.front();
466            if front.is_some_and(|key| *key <= height) {
467                return Ok(false);
468            }
469        }
470        Ok(true)
471    }

473    /// Processes a leader timeout issued for this multi-owner chain.
474    pub(super) async fn process_timeout(
475        &mut self,
476        certificate: TimeoutCertificate,
477    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
478        // Check that the chain is active and ready for this timeout.
479        // Verify the certificate. Returns a catch-all error to make client code more robust.
480        self.ensure_is_active().await?;
481        let (chain_epoch, committee) = self.chain.current_committee()?;
482        ensure!(
483            certificate.inner().epoch() == chain_epoch,
484            WorkerError::InvalidEpoch {
485                chain_id: certificate.inner().chain_id(),
486                chain_epoch,
487                epoch: certificate.inner().epoch()
488            }
489        );
490        certificate.check(committee)?;
491        let mut actions = NetworkActions::default();
492        if self
493            .chain
494            .tip_state
495            .get()
496            .already_validated_block(certificate.inner().height())?
497        {
498            return Ok((self.chain_info_response(), actions));
499        }
500        let old_round = self.chain.manager.current_round();
501        let timeout_chain_id = certificate.inner().chain_id();
502        let timeout_height = certificate.inner().height();
503        self.chain
504            .manager
505            .handle_timeout_certificate(certificate, self.storage.clock().current_time());
506        let round = self.chain.manager.current_round();
507        if round > old_round {
508            actions.notifications.push(Notification {
509                chain_id: timeout_chain_id,
510                reason: Reason::NewRound {
511                    height: timeout_height,
512                    round,
513                },
514            })
515        }
516        self.save().await?;
517        Ok((self.chain_info_response(), actions))
518    }

520    /// Tries to load all blobs published in this proposal.
521    ///
522    /// If they cannot be found, it creates an entry in `pending_proposed_blobs` so they can be
523    /// submitted one by one.
524    pub(super) async fn load_proposal_blobs(
525        &mut self,
526        proposal: &BlockProposal,
527    ) -> Result<Vec<Blob>, WorkerError> {
528        let owner = proposal.owner();
529        let BlockProposal {
530            content:
531                ProposalContent {
532                    block,
533                    round,
534                    outcome: _,
535                },
536            original_proposal,
537            signature: _,
538        } = proposal;
539
540        let mut maybe_blobs = self
541            .maybe_get_required_blobs(proposal.required_blob_ids(), None)
542            .await?;
543        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
544        if !missing_blob_ids.is_empty() {
545            let chain = &mut self.chain;
546            if chain.ownership().open_multi_leader_rounds {
547                // TODO(#3203): Allow multiple pending proposals on permissionless chains.
548                chain.pending_proposed_blobs.clear();
549            }
550            let validated = matches!(original_proposal, Some(OriginalProposal::Regular { .. }));
551            chain
552                .pending_proposed_blobs
553                .try_load_entry_mut(&owner)
554                .await?
555                .update(*round, validated, maybe_blobs)
556                .await?;
557            self.save().await?;
558            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
559        }
560        let published_blobs = block
561            .published_blob_ids()
562            .iter()
563            .filter_map(|blob_id| maybe_blobs.remove(blob_id).flatten())
564            .collect::<Vec<_>>();
565        Ok(published_blobs)
566    }

568    /// Processes a validated block issued for this multi-owner chain.
569    pub(super) async fn process_validated_block(
570        &mut self,
571        certificate: ValidatedBlockCertificate,
572    ) -> Result<(ChainInfoResponse, NetworkActions, bool), WorkerError> {
573        let block = certificate.block();
574
575        let header = &block.header;
576        let height = header.height;
577        // Check that the chain is active and ready for this validated block.
578        // Verify the certificate. Returns a catch-all error to make client code more robust.
579        self.ensure_is_active().await?;
580        let (epoch, committee) = self.chain.current_committee()?;
581        check_block_epoch(epoch, header.chain_id, header.epoch)?;
582        certificate.check(committee)?;
583        let mut actions = NetworkActions::default();
584        let already_committed_block = self.chain.tip_state.get().already_validated_block(height)?;
585        let should_skip_validated_block = || {
586            self.chain
587                .manager
588                .check_validated_block(&certificate)
589                .map(|outcome| outcome == manager::Outcome::Skip)
590        };
591        if already_committed_block || should_skip_validated_block()? {
592            // If we just processed the same pending block, return the chain info unchanged.
593            return Ok((self.chain_info_response(), actions, true));
594        }
595
596        self.block_values
597            .insert(Cow::Borrowed(certificate.inner().inner()));
598        let required_blob_ids = block.required_blob_ids();
599        let maybe_blobs = self
600            .maybe_get_required_blobs(required_blob_ids, Some(&block.created_blobs()))
601            .await?;
602        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
603        if !missing_blob_ids.is_empty() {
604            self.chain
605                .pending_validated_blobs
606                .update(certificate.round, true, maybe_blobs)
607                .await?;
608            self.save().await?;
609            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
610        }
611        let blobs = maybe_blobs
612            .into_iter()
613            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
614            .collect();
615        let old_round = self.chain.manager.current_round();
616        self.chain.manager.create_final_vote(
617            certificate,
618            self.config.key_pair(),
619            self.storage.clock().current_time(),
620            blobs,
621        )?;
622        self.save().await?;
623        let round = self.chain.manager.current_round();
624        if round > old_round {
625            actions.notifications.push(Notification {
626                chain_id: self.chain_id(),
627                reason: Reason::NewRound { height, round },
628            })
629        }
630        Ok((self.chain_info_response(), actions, false))
631    }

633    /// Processes a confirmed block (aka a commit).
634    pub(super) async fn process_confirmed_block(
635        &mut self,
636        certificate: ConfirmedBlockCertificate,
637        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
638    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
639        let block = certificate.block();
640        let height = block.header.height;
641        let chain_id = block.header.chain_id;
642
643        // Check that the chain is active and ready for this confirmation.
644        let tip = self.chain.tip_state.get().clone();
645        if tip.next_block_height > height {
646            // We already processed this block.
647            let actions = self.create_network_actions().await?;
648            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
649                .await;
650            return Ok((self.chain_info_response(), actions));
651        }
652
653        // We haven't processed the block - verify the certificate first
654        let epoch = block.header.epoch;
655        // Get the committee for the block's epoch from storage.
656        if let Some(committee) = self
657            .chain
658            .execution_state
659            .system
660            .committees
661            .get()
662            .get(&epoch)
663        {
664            certificate.check(committee)?;
665        } else {
666            let committees = self.storage.committees_for(epoch..=epoch).await?;
667            let Some(committee) = committees.get(&epoch) else {
668                let net_description = self
669                    .storage
670                    .read_network_description()
671                    .await?
672                    .ok_or_else(|| WorkerError::MissingNetworkDescription)?;
673                return Err(WorkerError::EventsNotFound(vec![EventId {
674                    chain_id: net_description.admin_chain_id,
675                    stream_id: StreamId::system(EPOCH_STREAM_NAME),
676                    index: epoch.0,
677                }]));
678            };
679            // This line is duplicated, but this avoids cloning and a lifetimes error.
680            certificate.check(committee)?;
681        }
682
683        // Certificate check passed - which means the blobs the block requires are legitimate and
684        // we can take note of it, so that if any are missing, we will accept them when the client
685        // sends them.
686        let required_blob_ids = block.required_blob_ids();
687        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
688        let blobs_result = self
689            .get_required_blobs(required_blob_ids.iter().copied(), &created_blobs)
690            .await
691            .map(|blobs| blobs.into_values().collect::<Vec<_>>());
692
693        if let Ok(blobs) = &blobs_result {
694            self.storage
695                .write_blobs_and_certificate(blobs, &certificate)
696                .await?;
697            let events = block
698                .body
699                .events
700                .iter()
701                .flatten()
702                .map(|event| (event.id(chain_id), event.value.clone()));
703            self.storage.write_events(events).await?;
704        }
705
706        // Update the blob state with last used certificate hash.
707        let blob_state = certificate.value().to_blob_state(blobs_result.is_ok());
708        let blob_ids = required_blob_ids.into_iter().collect::<Vec<_>>();
709        self.storage
710            .maybe_write_blob_states(&blob_ids, blob_state)
711            .await?;
712
713        let mut blobs = blobs_result?
714            .into_iter()
715            .map(|blob| (blob.id(), blob))
716            .collect::<BTreeMap<_, _>>();
717
718        // If this block is higher than the next expected block in this chain, we're going
719        // to have a gap: do not execute this block, only update the outboxes and return.
720        if tip.next_block_height < height {
721            // Update the outboxes.
722            self.chain.preprocess_block(certificate.value()).await?;
723            // Persist chain.
724            self.save().await?;
725            let actions = self.create_network_actions().await?;
726            trace!("Preprocessed confirmed block {height} on chain {chain_id:.8}");
727            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
728                .await;
729            return Ok((self.chain_info_response(), actions));
730        }
731
732        // This should always be true for valid certificates.
733        ensure!(
734            tip.block_hash == block.header.previous_block_hash,
735            WorkerError::InvalidBlockChaining
736        );
737
738        // If we got here, `height` is equal to `tip.next_block_height` and the block is
739        // properly chained. Verify that the chain is active and that the epoch we used for
740        // verifying the certificate is actually the active one on the chain.
741        self.ensure_is_active().await?;
742        let (epoch, _) = self.chain.current_committee()?;
743        check_block_epoch(epoch, chain_id, block.header.epoch)?;
744
745        let published_blobs = block
746            .published_blob_ids()
747            .iter()
748            .filter_map(|blob_id| blobs.remove(blob_id))
749            .collect::<Vec<_>>();
750
751        // If height is zero, we haven't initialized the chain state or verified the epoch before -
752        // do it now.
753        // This will fail if the chain description blob is still missing - but that's alright,
754        // because we already wrote the blob state above, so the client can now upload the
755        // blob, which will get accepted, and retry.
756        if height == BlockHeight::ZERO {
757            self.ensure_is_active().await?;
758            let (epoch, _) = self.chain.current_committee()?;
759            check_block_epoch(epoch, chain_id, block.header.epoch)?;
760        }
761
762        // Execute the block and update inboxes.
763        let local_time = self.storage.clock().current_time();
764        let chain = &mut self.chain;
765        chain
766            .remove_bundles_from_inboxes(block.header.timestamp, block.body.incoming_bundles())
767            .await?;
768        let oracle_responses = Some(block.body.oracle_responses.clone());
769        let (proposed_block, outcome) = block.clone().into_proposal();
770        let verified_outcome = if let Some(mut execution_state) =
771            self.execution_state_cache.remove(&outcome.state_hash)
772        {
773            chain.execution_state = execution_state
774                .with_context(|ctx| {
775                    chain
776                        .execution_state
777                        .context()
778                        .clone_with_base_key(ctx.base_key().bytes.clone())
779                })
780                .await;
781            outcome.clone()
782        } else {
783            chain
784                .execute_block(
785                    &proposed_block,
786                    local_time,
787                    None,
788                    &published_blobs,
789                    oracle_responses,
790                )
791                .await?
792        };
793        // We should always agree on the messages and state hash.
794        ensure!(
795            outcome == verified_outcome,
796            WorkerError::IncorrectOutcome {
797                submitted: Box::new(outcome),
798                computed: Box::new(verified_outcome),
799            }
800        );
801        // Update the rest of the chain state.
802        chain
803            .apply_confirmed_block(certificate.value(), local_time)
804            .await?;
805        self.track_newly_created_chains(&proposed_block, &outcome);
806        let mut actions = self.create_network_actions().await?;
807        trace!("Processed confirmed block {height} on chain {chain_id:.8}");
808        let hash = certificate.hash();
809        let event_streams = certificate
810            .value()
811            .block()
812            .body
813            .events
814            .iter()
815            .flatten()
816            .map(|event| event.stream_id.clone())
817            .collect();
818        actions.notifications.push(Notification {
819            chain_id,
820            reason: Reason::NewBlock {
821                height,
822                hash,
823                event_streams,
824            },
825        });
826        // Persist chain.
827        self.save().await?;
828
829        self.block_values
830            .insert(Cow::Owned(certificate.into_inner().into_inner()));
831
832        self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
833            .await;
834
835        Ok((self.chain_info_response(), actions))
836    }

838    /// Schedules a notification for when cross-chain messages are delivered up to the given
839    /// `height`.
840    #[instrument(level = "trace", skip(self, notify_when_messages_are_delivered))]
841    async fn register_delivery_notifier(
842        &mut self,
843        height: BlockHeight,
844        actions: &NetworkActions,
845        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
846    ) {
847        if let Some(notifier) = notify_when_messages_are_delivered {
848            if actions
849                .cross_chain_requests
850                .iter()
851                .any(|request| request.has_messages_lower_or_equal_than(height))
852            {
853                self.delivery_notifier.register(height, notifier);
854            } else {
855                // No need to wait. Also, cross-chain requests may not trigger the
856                // notifier later, even if we register it.
857                if let Err(()) = notifier.send(()) {
858                    warn!("Failed to notify message delivery to caller");
859                }
860            }
861        }
862    }

864    /// Updates the chain's inboxes, receiving messages from a cross-chain update.
865    #[instrument(level = "trace", skip(self, bundles))]
866    pub(super) async fn process_cross_chain_update(
867        &mut self,
868        origin: ChainId,
869        bundles: Vec<(Epoch, MessageBundle)>,
870    ) -> Result<Option<BlockHeight>, WorkerError> {
871        // Only process certificates with relevant heights and epochs.
872        let next_height_to_receive = self.chain.next_block_height_to_receive(&origin).await?;
873        let last_anticipated_block_height =
874            self.chain.last_anticipated_block_height(&origin).await?;
875        let helper = CrossChainUpdateHelper::new(&self.config, &self.chain);
876        let recipient = self.chain_id();
877        let bundles = helper.select_message_bundles(
878            &origin,
879            recipient,
880            next_height_to_receive,
881            last_anticipated_block_height,
882            bundles,
883        )?;
884        let Some(last_updated_height) = bundles.last().map(|bundle| bundle.height) else {
885            return Ok(None);
886        };
887        // Process the received messages in certificates.
888        let local_time = self.storage.clock().current_time();
889        let mut previous_height = None;
890        for bundle in bundles {
891            let add_to_received_log = previous_height != Some(bundle.height);
892            previous_height = Some(bundle.height);
893            // Update the staged chain state with the received block.
894            self.chain
895                .receive_message_bundle(&origin, bundle, local_time, add_to_received_log)
896                .await?;
897        }
898        if !self.config.allow_inactive_chains && !self.chain.is_active() {
899            // Refuse to create a chain state if the chain is still inactive by
900            // now. Accordingly, do not send a confirmation, so that the
901            // cross-chain update is retried later.
902            warn!(
903                "Refusing to deliver messages to {recipient:?} from {origin:?} \
904                at height {last_updated_height} because the recipient is still inactive",
905            );
906            return Ok(None);
907        }
908        // Save the chain.
909        self.save().await?;
910        Ok(Some(last_updated_height))
911    }

996    pub(super) async fn handle_pending_blob(
997        &mut self,
998        blob: Blob,
999    ) -> Result<ChainInfoResponse, WorkerError> {
1000        let mut was_expected = self
1001            .chain
1002            .pending_validated_blobs
1003            .maybe_insert(&blob)
1004            .await?;
1005        for (_, mut pending_blobs) in self
1006            .chain
1007            .pending_proposed_blobs
1008            .try_load_all_entries_mut()
1009            .await?
1010        {
1011            if !pending_blobs.validated.get() {
1012                let (_, committee) = self.chain.current_committee()?;
1013                let policy = committee.policy();
1014                policy
1015                    .check_blob_size(blob.content())
1016                    .with_execution_context(ChainExecutionContext::Block)?;
1017                ensure!(
1018                    u64::try_from(pending_blobs.pending_blobs.count().await?)
1019                        .is_ok_and(|count| count < policy.maximum_published_blobs),
1020                    WorkerError::TooManyPublishedBlobs(policy.maximum_published_blobs)
1021                );
1022            }
1023            was_expected = was_expected || pending_blobs.maybe_insert(&blob).await?;
1024        }
1025        ensure!(was_expected, WorkerError::UnexpectedBlob);
1026        self.save().await?;
1027        Ok(self.chain_info_response())
1028    }

1105    /// Validates and executes a block proposed to extend this chain.
1106    pub(super) async fn handle_block_proposal(
1107        &mut self,
1108        proposal: BlockProposal,
1109    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
1110        self.ensure_is_active().await?;
1111        proposal
1112            .check_invariants()
1113            .map_err(|msg| WorkerError::InvalidBlockProposal(msg.to_string()))?;
1114        proposal.check_signature()?;
1115        let owner = proposal.owner();
1116        let BlockProposal {
1117            content,
1118            original_proposal,
1119            signature: _,
1120        } = &proposal;
1121        let block = &content.block;
1122        let chain = &self.chain;
1123        // Check the epoch.
1124        let (epoch, committee) = chain.current_committee()?;
1125        check_block_epoch(epoch, block.chain_id, block.epoch)?;
1126        let policy = committee.policy().clone();
1127        block.check_proposal_size(policy.maximum_block_proposal_size)?;
1128        // Check the authentication of the block.
1129        ensure!(
1130            chain.manager.verify_owner(&owner, proposal.content.round)?,
1131            WorkerError::InvalidOwner
1132        );
1133        match original_proposal {
1134            None => {
1135                if let Some(signer) = block.authenticated_signer {
1136                    // Check the authentication of the operations in the new block.
1137                    ensure!(signer == owner, WorkerError::InvalidSigner(owner));
1138                }
1139            }
1140            Some(OriginalProposal::Regular { certificate }) => {
1141                // Verify that this block has been validated by a quorum before.
1142                certificate.check(committee)?;
1143            }
1144            Some(OriginalProposal::Fast(signature)) => {
1145                let original_proposal = BlockProposal {
1146                    content: ProposalContent {
1147                        block: content.block.clone(),
1148                        round: Round::Fast,
1149                        outcome: None,
1150                    },
1151                    signature: *signature,
1152                    original_proposal: None,
1153                };
1154                let super_owner = original_proposal.owner();
1155                ensure!(
1156                    chain
1157                        .manager
1158                        .ownership
1159                        .get()
1160                        .super_owners
1161                        .contains(&super_owner),
1162                    WorkerError::InvalidOwner
1163                );
1164                if let Some(signer) = block.authenticated_signer {
1165                    // Check the authentication of the operations in the new block.
1166                    ensure!(signer == super_owner, WorkerError::InvalidSigner(signer));
1167                }
1168                original_proposal.check_signature()?;
1169            }
1170        }
1171        // Check if the chain is ready for this new block proposal.
1172        chain.tip_state.get().verify_block_chaining(block)?;
1173        if chain.manager.check_proposed_block(&proposal)? == manager::Outcome::Skip {
1174            // We already voted for this block.
1175            return Ok((self.chain_info_response(), NetworkActions::default()));
1176        }
1177
1178        // Make sure we remember that a proposal was signed, to determine the correct round to
1179        // propose in.
1180        if self.chain.manager.update_signed_proposal(&proposal) {
1181            self.save().await?;
1182        }
1183
1184        let published_blobs = self.load_proposal_blobs(&proposal).await?;
1185        let ProposalContent {
1186            block,
1187            round,
1188            outcome,
1189        } = content;
1190
1191        let local_time = self.storage.clock().current_time();
1192        ensure!(
1193            block.timestamp.duration_since(local_time) <= self.config.grace_period,
1194            WorkerError::InvalidTimestamp
1195        );
1196        self.storage.clock().sleep_until(block.timestamp).await;
1197        let local_time = self.storage.clock().current_time();
1198
1199        self.chain
1200            .remove_bundles_from_inboxes(block.timestamp, block.incoming_bundles())
1201            .await?;
1202        let outcome = if let Some(outcome) = outcome {
1203            outcome.clone()
1204        } else {
1205            self.execute_block(block, local_time, round.multi_leader(), &published_blobs)
1206                .await?
1207        };
1208
1209        ensure!(
1210            !round.is_fast() || !outcome.has_oracle_responses(),
1211            WorkerError::FastBlockUsingOracles
1212        );
1213        let chain = &mut self.chain;
1214        // Check if the counters of tip_state would be valid.
1215        chain
1216            .tip_state
1217            .get_mut()
1218            .update_counters(&block.transactions, &outcome.messages)?;
1219        // Verify that the resulting chain would have no unconfirmed incoming messages.
1220        chain.validate_incoming_bundles().await?;
1221        // Don't save the changes since the block is not confirmed yet.
1222        chain.rollback();
1223
1224        // Create the vote and store it in the chain state.
1225        let block = outcome.with(proposal.content.block.clone());
1226        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
1227        let blobs = self
1228            .get_required_blobs(proposal.expected_blob_ids(), &created_blobs)
1229            .await?;
1230        let key_pair = self.config.key_pair();
1231        let manager = &mut self.chain.manager;
1232        match manager.create_vote(proposal, block, key_pair, local_time, blobs)? {
1233            // Cache the value we voted on, so the client doesn't have to send it again.
1234            Some(Either::Left(vote)) => {
1235                self.block_values.insert(Cow::Borrowed(vote.value.inner()));
1236            }
1237            Some(Either::Right(vote)) => {
1238                self.block_values.insert(Cow::Borrowed(vote.value.inner()));
1239            }
1240            None => (),
1241        }
1242        self.save().await?;
1243        let actions = self.create_network_actions().await?;
1244        Ok((self.chain_info_response(), actions))
1245    }

1247    /// Prepares a [`ChainInfoResponse`] for a [`ChainInfoQuery`].
1248    pub(super) async fn prepare_chain_info_response(
1249        &mut self,
1250        query: ChainInfoQuery,
1251    ) -> Result<ChainInfoResponse, WorkerError> {
1252        self.ensure_is_active().await?;
1253        let chain = &self.chain;
1254        let mut info = ChainInfo::from(chain);
1255        if query.request_committees {
1256            info.requested_committees = Some(chain.execution_state.system.committees.get().clone());
1257        }
1258        if query.request_owner_balance == AccountOwner::CHAIN {
1259            info.requested_owner_balance = Some(*chain.execution_state.system.balance.get());
1260        } else {
1261            info.requested_owner_balance = chain
1262                .execution_state
1263                .system
1264                .balances
1265                .get(&query.request_owner_balance)
1266                .await?;
1267        }
1268        if let Some(next_block_height) = query.test_next_block_height {
1269            ensure!(
1270                chain.tip_state.get().next_block_height == next_block_height,
1271                WorkerError::UnexpectedBlockHeight {
1272                    expected_block_height: next_block_height,
1273                    found_block_height: chain.tip_state.get().next_block_height
1274                }
1275            );
1276        }
1277        if query.request_pending_message_bundles {
1278            let mut messages = Vec::new();
1279            let pairs = chain.inboxes.try_load_all_entries().await?;
1280            let action = if *chain.execution_state.system.closed.get() {
1281                MessageAction::Reject
1282            } else {
1283                MessageAction::Accept
1284            };
1285            for (origin, inbox) in pairs {
1286                for bundle in inbox.added_bundles.elements().await? {
1287                    messages.push(IncomingBundle {
1288                        origin,
1289                        bundle,
1290                        action,
1291                    });
1292                }
1293            }
1294
1295            info.requested_pending_message_bundles = messages;
1296        }
1297        let mut hashes = Vec::new();
1298        for height in query.request_sent_certificate_hashes_by_heights {
1299            hashes.extend(chain.block_hashes(height..=height).await?);
1300        }
1301        info.requested_sent_certificate_hashes = hashes;
1302        if let Some(start) = query.request_received_log_excluding_first_n {
1303            let start = usize::try_from(start).map_err(|_| ArithmeticError::Overflow)?;
1304            info.requested_received_log = chain.received_log.read(start..).await?;
1305        }
1306        if query.request_manager_values {
1307            info.manager.add_values(&chain.manager);
1308        }
1309        Ok(ChainInfoResponse::new(info, self.config.key_pair()))
1310    }

1313    async fn execute_block(
1314        &mut self,
1315        block: &ProposedBlock,
1316        local_time: Timestamp,
1317        round: Option<u32>,
1318        published_blobs: &[Blob],
1319    ) -> Result<BlockExecutionOutcome, WorkerError> {
1320        let outcome =
1321            Box::pin(
1322                self.chain
1323                    .execute_block(block, local_time, round, published_blobs, None),
1324            )
1325            .await?;
1326        self.execution_state_cache.insert_owned(
1327            &outcome.state_hash,
1328            self.chain
1329                .execution_state
1330                .with_context(|ctx| InactiveContext(ctx.base_key().clone()))
1331                .await,
1332        );
1333        Ok(outcome)
1334    }

1387/// Helper type for handling cross-chain updates.
1388pub(crate) struct CrossChainUpdateHelper<'a> {
1389    pub allow_messages_from_deprecated_epochs: bool,
1390    pub current_epoch: Epoch,
1391    pub committees: &'a BTreeMap<Epoch, Committee>,
1392}
1393
1394impl<'a> CrossChainUpdateHelper<'a> {
1395    /// Creates a new [`CrossChainUpdateHelper`].
1396    pub fn new<C>(config: &ChainWorkerConfig, chain: &'a ChainStateView<C>) -> Self
1397    where
1398        C: Context + Clone + Send + Sync + 'static,
1399    {
1400        CrossChainUpdateHelper {
1401            allow_messages_from_deprecated_epochs: config.allow_messages_from_deprecated_epochs,
1402            current_epoch: *chain.execution_state.system.epoch.get(),
1403            committees: chain.execution_state.system.committees.get(),
1404        }
1405    }
1406
1407    /// Checks basic invariants and deals with repeated heights and deprecated epochs.
1408    /// * Returns a range of message bundles that are both new to us and not relying on
1409    ///   an untrusted set of validators.
1410    /// * In the case of validators, if the epoch(s) of the highest bundles are not
1411    ///   trusted, we only accept bundles that contain messages that were already
1412    ///   executed by anticipation (i.e. received in certified blocks).
1413    /// * Basic invariants are checked for good measure. We still crucially trust
1414    ///   the worker of the sending chain to have verified and executed the blocks
1415    ///   correctly.
1416    pub fn select_message_bundles(
1417        &self,
1418        origin: &'a ChainId,
1419        recipient: ChainId,
1420        next_height_to_receive: BlockHeight,
1421        last_anticipated_block_height: Option<BlockHeight>,
1422        mut bundles: Vec<(Epoch, MessageBundle)>,
1423    ) -> Result<Vec<MessageBundle>, WorkerError> {
1424        let mut latest_height = None;
1425        let mut skipped_len = 0;
1426        let mut trusted_len = 0;
1427        for (i, (epoch, bundle)) in bundles.iter().enumerate() {
1428            // Make sure that heights are not decreasing.
1429            ensure!(
1430                latest_height <= Some(bundle.height),
1431                WorkerError::InvalidCrossChainRequest
1432            );
1433            latest_height = Some(bundle.height);
1434            // Check if the block has been received already.
1435            if bundle.height < next_height_to_receive {
1436                skipped_len = i + 1;
1437            }
1438            // Check if the height is trusted or the epoch is trusted.
1439            if self.allow_messages_from_deprecated_epochs
1440                || Some(bundle.height) <= last_anticipated_block_height
1441                || *epoch >= self.current_epoch
1442                || self.committees.contains_key(epoch)
1443            {
1444                trusted_len = i + 1;
1445            }
1446        }
1447        if skipped_len > 0 {
1448            let (_, sample_bundle) = &bundles[skipped_len - 1];
1449            debug!(
1450                "Ignoring repeated messages to {recipient:.8} from {origin:} at height {}",
1451                sample_bundle.height,
1452            );
1453        }
1454        if skipped_len < bundles.len() && trusted_len < bundles.len() {
1455            let (sample_epoch, sample_bundle) = &bundles[trusted_len];
1456            warn!(
1457                "Refusing messages to {recipient:.8} from {origin:} at height {} \
1458                 because the epoch {} is not trusted any more",
1459                sample_bundle.height, sample_epoch,
1460            );
1461        }
1462        let bundles = if skipped_len < trusted_len {
1463            bundles
1464                .drain(skipped_len..trusted_len)
1465                .map(|(_, bundle)| bundle)
1466                .collect()
1467        } else {
1468            vec![]
1469        };
1470        Ok(bundles)
1471    }
1472}
