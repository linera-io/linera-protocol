287    /// Tries to read the blobs from the chain manager or storage. Returns `None` if not found.
288    #[instrument(skip_all, fields(
289        chain_id = %self.chain_id()
290    ))]
291    async fn maybe_get_required_blobs(
292        &self,
293        blob_ids: impl IntoIterator<Item = BlobId>,
294        mut created_blobs: Option<BTreeMap<BlobId, Blob>>,
295    ) -> Result<BTreeMap<BlobId, Option<Blob>>, WorkerError> {
296        let maybe_blobs = blob_ids.into_iter().collect::<BTreeSet<_>>();
297        let mut maybe_blobs = maybe_blobs
298            .into_iter()
299            .map(|x| (x, None))
300            .collect::<Vec<(BlobId, Option<Blob>)>>();
301
302        if let Some(blob_map) = &mut created_blobs {
303            for (blob_id, value) in &mut maybe_blobs {
304                if let Some(blob) = blob_map.remove(blob_id) {
305                    *value = Some(blob);
306                }
307            }
308        }
309
310        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
311        let second_block_blobs = self.chain.manager.pending_blobs(&missing_blob_ids).await?;
312        for (index, blob) in missing_indices.into_iter().zip(second_block_blobs) {
313            maybe_blobs[index].1 = blob;
314        }
315
316        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
317        let third_block_blobs = self
318            .chain
319            .pending_validated_blobs
320            .multi_get(&missing_blob_ids)
321            .await?;
322        for (index, blob) in missing_indices.into_iter().zip(third_block_blobs) {
323            maybe_blobs[index].1 = blob;
324        }
325
326        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
327        if !missing_indices.is_empty() {
328            let all_entries_pending_blobs = self
329                .chain
330                .pending_proposed_blobs
331                .try_load_all_entries()
332                .await?;
333            for (index, blob_id) in missing_indices.into_iter().zip(missing_blob_ids) {
334                for (_, pending_blobs) in &all_entries_pending_blobs {
335                    if let Some(blob) = pending_blobs.get(&blob_id).await? {
336                        maybe_blobs[index].1 = Some(blob);
337                        break;
338                    }
339                }
340            }
341        }
342
343        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
344        let fourth_block_blobs = self.storage.read_blobs(&missing_blob_ids).await?;
345        for (index, blob) in missing_indices.into_iter().zip(fourth_block_blobs) {
346            maybe_blobs[index].1 = blob.map(Arc::unwrap_or_clone);
347        }
348        Ok(maybe_blobs.into_iter().collect())
349    }

377    /// Loads pending cross-chain requests, and adds `NewRound` notifications where appropriate.
378    async fn create_network_actions(
379        &self,
380        old_round: Option<Round>,
381    ) -> Result<NetworkActions, WorkerError> {
382        #[cfg(with_metrics)]
383        let _latency = metrics::CREATE_NETWORK_ACTIONS_LATENCY.measure_latency();
384        let mut heights_by_recipient = BTreeMap::<_, Vec<_>>::new();
385        let mut targets = self.chain.nonempty_outbox_chain_ids();
386        if let Some(chain_modes) = self.chain_modes.as_ref() {
387            let chain_modes = chain_modes
388                .read()
389                .expect("Panics should not happen while holding a lock to `chain_modes`");
390            // Only process outboxes for full chains (those whose inboxes we update).
391            targets.retain(|target| chain_modes.get(target).is_some_and(ListeningMode::is_full));
392        }
393        let outboxes = self.chain.load_outboxes(&targets).await?;
394        for (target, outbox) in targets.into_iter().zip(outboxes) {
395            let heights = outbox.queue.elements().await?;
396            heights_by_recipient.insert(target, heights);
397        }
398        let cross_chain_requests = self
399            .create_cross_chain_requests(heights_by_recipient)
400            .await?;
401        let mut notifications = Vec::new();
402        if let Some(old_round) = old_round {
403            let round = self.chain.manager.current_round();
404            if round > old_round {
405                let height = self.chain.tip_state.get().next_block_height;
406                notifications.push(Notification {
407                    chain_id: self.chain_id(),
408                    reason: Reason::NewRound { height, round },
409                });
410            }
411        }
412        Ok(NetworkActions {
413            cross_chain_requests,
414            notifications,
415        })
416    }

418    /// Returns confirmed blocks by hash, checking the cache first and batch-loading the rest
419    /// from storage. The order of the returned blocks matches the order of the input hashes.
420    async fn read_confirmed_blocks(
421        &self,
422        hashes: &[CryptoHash],
423    ) -> Result<Vec<Option<Arc<ConfirmedBlock>>>, WorkerError> {
424        let mut blocks = Vec::with_capacity(hashes.len());
425        let mut uncached_indices = Vec::new();
426        let mut uncached_hashes = Vec::new();
427
428        for (i, hash) in hashes.iter().enumerate() {
429            if let Some(hashed_block) = self.block_values.get(hash) {
430                blocks.push(Some(Arc::new(ConfirmedBlock::from_hashed(
431                    Arc::unwrap_or_clone(hashed_block),
432                ))));
433            } else {
434                blocks.push(None);
435                uncached_indices.push(i);
436                uncached_hashes.push(*hash);
437            }
438        }
439
440        if !uncached_hashes.is_empty() {
441            let from_storage = self.storage.read_confirmed_blocks(uncached_hashes).await?;
442            for (i, maybe_block) in uncached_indices.into_iter().zip(from_storage) {
443                if let Some(block) = &maybe_block {
444                    self.block_values
445                        .insert_hashed(Cow::Borrowed(block.inner()));
446                }
447                blocks[i] = maybe_block;
448            }
449        }
450
451        Ok(blocks)
452    }

454    #[instrument(skip_all, fields(
455        chain_id = %self.chain_id(),
456        num_recipients = %heights_by_recipient.len()
457    ))]
458    async fn create_cross_chain_requests(
459        &self,
460        heights_by_recipient: BTreeMap<ChainId, Vec<BlockHeight>>,
461    ) -> Result<Vec<CrossChainRequest>, WorkerError> {
462        // Load all the certificates we will need, regardless of the medium.
463        let heights = heights_by_recipient
464            .values()
465            .flatten()
466            .copied()
467            .collect::<BTreeSet<_>>();
468        let hashes = self.chain.block_hashes(heights.iter().copied()).await?;
469
470        let blocks = self.read_confirmed_blocks(&hashes).await?;
471
472        let mut height_to_blocks = HashMap::new();
473        for (block, hash) in blocks.into_iter().zip(hashes) {
474            let block = block.ok_or_else(|| WorkerError::ReadCertificatesError(vec![hash]))?;
475            height_to_blocks.insert(block.height(), block);
476        }
477
478        let sender = self.chain.chain_id();
479        let mut cross_chain_requests = Vec::new();
480        for (recipient, heights) in heights_by_recipient {
481            // Extract the predecessor height for this recipient from the first
482            // block's `previous_message_blocks`. This lets the recipient detect
483            // gaps even before it consumes the missing message.
484            let previous_height = heights.first().and_then(|first_height| {
485                let block = height_to_blocks.get(first_height)?;
486                let (_, prev_height) =
487                    block.block().body.previous_message_blocks.get(&recipient)?;
488                Some(*prev_height)
489            });
490            let mut bundles = Vec::new();
491            let mut bundles_size = 0;
492            for height in heights {
493                let Some(confirmed_block) = height_to_blocks.get(&height) else {
494                    tracing::warn!(
495                        %height,
496                        %recipient,
497                        "spurious entry in outbox; skipping this and higher sender blocks"
498                    );
499                    break;
500                };
501                let new_bundles = confirmed_block
502                    .block()
503                    .message_bundles_for(recipient, confirmed_block.inner().hash())
504                    .collect::<Vec<_>>();
505                let new_size = new_bundles
506                    .iter()
507                    .map(|(_epoch, bundle)| bundle.estimated_size())
508                    .sum::<usize>();
509                // If adding this block's bundles would exceed the chunk limit,
510                // stop here. Always include at least one block's bundles.
511                if bundles_size + new_size > self.config.cross_chain_message_chunk_limit {
512                    if bundles.is_empty() {
513                        warn!(
514                            "Single block at height {height} produces an UpdateRecipient \
515                            of ~{new_size} bytes, exceeding the chunk limit of {}",
516                            self.config.cross_chain_message_chunk_limit
517                        );
518                    } else {
519                        debug!(
520                            "Stopping cross-chain batch for {recipient} at height {height}: \
521                            adding ~{new_size} bytes would exceed chunk limit of {} \
522                            (current batch ~{bundles_size} bytes)",
523                            self.config.cross_chain_message_chunk_limit
524                        );
525                        break;
526                    }
527                }
528                bundles.extend(new_bundles);
529                bundles_size += new_size;
530            }
531            if !bundles.is_empty() {
532                cross_chain_requests.push(CrossChainRequest::UpdateRecipient {
533                    sender,
534                    recipient,
535                    bundles,
536                    previous_height,
537                });
538            }
539        }
540        Ok(cross_chain_requests)
541    }

543    /// Returns true if there are no more outgoing messages in flight up to the given
544    /// block height for all tracked chains (those whose inboxes we update).
545    #[instrument(skip_all, fields(
546        chain_id = %self.chain_id(),
547        height = %height
548    ))]
549    async fn all_messages_to_tracked_chains_delivered_up_to(
550        &self,
551        height: BlockHeight,
552    ) -> Result<bool, WorkerError> {
553        if self.chain.all_messages_delivered_up_to(height) {
554            return Ok(true);
555        }
556        let Some(chain_modes) = self.chain_modes.as_ref() else {
557            return Ok(false);
558        };
559        let mut targets = self.chain.nonempty_outbox_chain_ids();
560        {
561            let chain_modes = chain_modes.read().unwrap();
562            // Only consider full chains (those whose inboxes we update).
563            targets.retain(|target| chain_modes.get(target).is_some_and(ListeningMode::is_full));
564        }
565        let outboxes = self.chain.load_outboxes(&targets).await?;
566        for outbox in outboxes {
567            let front = outbox.queue.front().await?;
568            if front.is_some_and(|key| key <= height) {
569                return Ok(false);
570            }
571        }
572        Ok(true)
573    }

575    /// Processes a leader timeout issued for this multi-owner chain.
576    #[instrument(skip_all, fields(
577        chain_id = %self.chain_id(),
578        height = %certificate.inner().height()
579    ))]
580    pub(crate) async fn process_timeout(
581        &mut self,
582        certificate: TimeoutCertificate,
583    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
584        // Check that the chain is active and ready for this timeout.
585        // Verify the certificate. Returns a catch-all error to make client code more robust.
586        self.initialize_and_save_if_needed().await?;
587        let (chain_epoch, committee) = self.chain.current_committee().await?;
588        certificate.check(&committee)?;
589        if self
590            .chain
591            .tip_state
592            .get()
593            .already_validated_block(certificate.inner().height())?
594        {
595            return Ok((self.chain_info_response().await?, NetworkActions::default()));
596        }
597        ensure!(
598            certificate.inner().epoch() == chain_epoch,
599            WorkerError::InvalidEpoch {
600                chain_id: certificate.inner().chain_id(),
601                chain_epoch,
602                epoch: certificate.inner().epoch()
603            }
604        );
605        let old_round = self.chain.manager.current_round();
606        self.chain
607            .manager
608            .handle_timeout_certificate(certificate, self.storage.clock().current_time());
609        self.save().await?;
610        let actions = self.create_network_actions(Some(old_round)).await?;
611        Ok((self.chain_info_response().await?, actions))
612    }

620        block_height = %proposal.content.block.height
621    ))]
622    async fn load_proposal_blobs(
623        &mut self,
624        proposal: &BlockProposal,
625    ) -> Result<Vec<Blob>, WorkerError> {
626        let owner = proposal.owner();
627        let BlockProposal {
628            content:
629                ProposalContent {
630                    block,
631                    round,
632                    outcome: _,
633                },
634            original_proposal,
635            signature: _,
636        } = proposal;
637
638        let mut maybe_blobs = self
639            .maybe_get_required_blobs(proposal.required_blob_ids(), None)
640            .await?;
641        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
642        if !missing_blob_ids.is_empty() {
643            let chain = &mut self.chain;
644            if chain.ownership().await?.open_multi_leader_rounds {
645                // TODO(#3203): Allow multiple pending proposals on permissionless chains.
646                chain.pending_proposed_blobs.clear();
647            }
648            let validated = matches!(original_proposal, Some(OriginalProposal::Regular { .. }));
649            chain
650                .pending_proposed_blobs
651                .try_load_entry_mut(&owner)
652                .await?
653                .update(*round, validated, maybe_blobs)?;
654            self.save().await?;
655            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
656        }
657        let published_blobs = block
658            .published_blob_ids()
659            .iter()
660            .filter_map(|blob_id| maybe_blobs.remove(blob_id).flatten())
661            .collect::<Vec<_>>();
662        Ok(published_blobs)
663    }

665    /// Processes a validated block issued for this multi-owner chain.
666    #[instrument(skip_all, fields(
667        chain_id = %self.chain_id(),
668        block_height = %certificate.block().header.height
669    ))]
670    pub(crate) async fn process_validated_block(
671        &mut self,
672        certificate: ValidatedBlockCertificate,
673    ) -> Result<(ChainInfoResponse, NetworkActions, BlockOutcome), WorkerError> {
674        let block = certificate.block();
675
676        let header = &block.header;
677        let height = header.height;
678        // Check that the chain is active and ready for this validated block.
679        // Verify the certificate. Returns a catch-all error to make client code more robust.
680        self.initialize_and_save_if_needed().await?;
681        let tip_state = self.chain.tip_state.get();
682        ensure!(
683            header.height == tip_state.next_block_height,
684            ChainError::UnexpectedBlockHeight {
685                expected_block_height: tip_state.next_block_height,
686                found_block_height: header.height,
687            }
688        );
689        let (epoch, committee) = self.chain.current_committee().await?;
690        check_block_epoch(epoch, header.chain_id, header.epoch)?;
691        certificate.check(&committee)?;
692        let already_committed_block = self.chain.tip_state.get().already_validated_block(height)?;
693        let should_skip_validated_block = || {
694            self.chain
695                .manager
696                .check_validated_block(&certificate)
697                .map(|outcome| outcome == manager::Outcome::Skip)
698        };
699        if already_committed_block || should_skip_validated_block()? {
700            // If we just processed the same pending block, return the chain info unchanged.
701            return Ok((
702                self.chain_info_response().await?,
703                NetworkActions::default(),
704                BlockOutcome::Skipped,
705            ));
706        }
707
708        self.block_values
709            .insert_hashed(Cow::Borrowed(certificate.inner().inner()));
710        let required_blob_ids = block.required_blob_ids();
711        let maybe_blobs = self
712            .maybe_get_required_blobs(required_blob_ids, Some(block.created_blobs()))
713            .await?;
714        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
715        if !missing_blob_ids.is_empty() {
716            self.chain
717                .pending_validated_blobs
718                .update(certificate.round, true, maybe_blobs)?;
719            self.save().await?;
720            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
721        }
722        let blobs = maybe_blobs
723            .into_iter()
724            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
725            .collect();
726        let old_round = self.chain.manager.current_round();
727        self.chain.manager.create_final_vote(
728            certificate,
729            self.config.key_pair(),
730            self.storage.clock().current_time(),
731            blobs,
732        )?;
733        self.save().await?;
734        let actions = self.create_network_actions(Some(old_round)).await?;
735        Ok((
736            self.chain_info_response().await?,
737            actions,
738            BlockOutcome::Processed,
739        ))
740    }

742    /// Processes a confirmed block (aka a commit).
743    #[instrument(skip_all, fields(
744        chain_id = %certificate.block().header.chain_id,
745        height = %certificate.block().header.height,
746        block_hash = %certificate.hash(),
747    ))]
748    pub(crate) async fn process_confirmed_block(
749        &mut self,
750        certificate: ConfirmedBlockCertificate,
751        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
752    ) -> Result<(ChainInfoResponse, NetworkActions, BlockOutcome), WorkerError> {
753        let block = certificate.block();
754        let block_hash = certificate.hash();
755        let height = block.header.height;
756        let chain_id = block.header.chain_id;
757
758        // Check if we already processed this block.
759        let tip = self.chain.tip_state.get().clone();
760        if tip.next_block_height > height {
761            let actions = self.create_network_actions(None).await?;
762            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
763                .await;
764            return Ok((
765                self.chain_info_response().await?,
766                actions,
767                BlockOutcome::Skipped,
768            ));
769        }
770
771        // We haven't processed the block - verify the certificate first.
772        // Resolve the committee hash from the admin chain's epoch event stream; the
773        // chain-local `committees` map is just a cache of the same hashes for trusted
774        // epochs, so we can skip it and always hit the event stream.
775        let epoch = block.header.epoch;
776        let hash = self
777            .chain
778            .execution_state
779            .context()
780            .extra()
781            .get_committee_hashes(epoch..=epoch)
782            .await
783            .map_err(|error| {
784                ChainError::ExecutionError(Box::new(error), ChainExecutionContext::Block)
785            })?
786            .remove(&epoch)
787            .ok_or_else(|| {
788                ChainError::InternalError(format!(
789                    "missing committee for epoch {epoch}; this is a bug"
790                ))
791            })?;
792        let committee = self
793            .chain
794            .execution_state
795            .context()
796            .extra()
797            .get_or_load_committee_by_hash(hash)
798            .await
799            .map_err(|error| {
800                ChainError::ExecutionError(Box::new(error), ChainExecutionContext::Block)
801            })?;
802        certificate.check(&committee)?;
803
804        // Certificate check passed - which means the blobs the block requires are legitimate and
805        // we can take note of it, so that if any are missing, we will accept them when the client
806        // sends them.
807        let required_blob_ids = block.required_blob_ids();
808        let blobs_result = self
809            .get_required_blobs(required_blob_ids.iter().copied(), block.created_blobs())
810            .await
811            .map(|blobs| blobs.into_values().collect::<Vec<_>>());
812
813        if let Ok(blobs) = &blobs_result {
814            self.storage
815                .write_blobs_and_certificate(blobs, &certificate)
816                .await?;
817            let events = block
818                .body
819                .events
820                .iter()
821                .flatten()
822                .map(|event| (event.id(chain_id), event.value.clone()));
823            self.storage.write_events(events).await?;
824        }
825
826        // Update the blob state with last used certificate hash.
827        let blob_state = certificate.value().to_blob_state(blobs_result.is_ok());
828        let blob_ids = required_blob_ids.into_iter().collect::<Vec<_>>();
829        self.storage
830            .maybe_write_blob_states(&blob_ids, blob_state)
831            .await?;
832
833        let mut blobs = blobs_result?
834            .into_iter()
835            .map(|blob| (blob.id(), blob))
836            .collect::<BTreeMap<_, _>>();
837
838        // If this block is higher than the next expected block in this chain, we're going
839        // to have a gap: do not execute this block, only update the outboxes and return.
840        if tip.next_block_height < height {
841            // Update the outboxes and event streams.
842            let updated_event_streams = self.chain.preprocess_block(certificate.value()).await?;
843            // Persist chain.
844            self.save().await?;
845            let mut actions = self.create_network_actions(None).await?;
846            if !updated_event_streams.is_empty() {
847                actions.notifications.push(Notification {
848                    chain_id,
849                    reason: Reason::NewEvents {
850                        height,
851                        block_hash,
852                        event_streams: updated_event_streams,
853                    },
854                });
855            }
856            trace!("Preprocessed confirmed block {height}");
857            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
858                .await;
859            return Ok((
860                self.chain_info_response().await?,
861                actions,
862                BlockOutcome::Preprocessed,
863            ));
864        }
865
866        // This should always be true for valid certificates.
867        ensure!(
868            tip.block_hash == block.header.previous_block_hash,
869            WorkerError::InvalidBlockChaining
870        );
871
872        // If we got here, `height` is equal to `tip.next_block_height` and the block is
873        // properly chained. Verify that the chain is active and that the epoch we used for
874        // verifying the certificate is actually the active one on the chain.
875        self.initialize_and_save_if_needed().await?;
876        let (epoch, _) = self.chain.current_committee().await?;
877        check_block_epoch(epoch, chain_id, block.header.epoch)?;
878
879        let published_blobs = block
880            .published_blob_ids()
881            .iter()
882            .filter_map(|blob_id| blobs.remove(blob_id))
883            .collect::<Vec<_>>();
884
885        // Execute the block and update inboxes.
886        let local_time = self.storage.clock().current_time();
887        if block.header.timestamp.duration_since(local_time) > self.config.block_time_grace_period {
888            warn!(
889                block_timestamp = %block.header.timestamp,
890                %local_time,
891                "Confirmed block has a timestamp in the future beyond the block time grace period"
892            );
893        }
894        let chain = &mut self.chain;
895        chain
896            .remove_bundles_from_inboxes(
897                block.header.timestamp,
898                false,
899                block.body.incoming_bundles(),
900            )
901            .await?;
902        let confirmed_block = if let Some(mut execution_state) = self
903            .execution_state_cache
904            .as_ref()
905            .and_then(|cache| cache.remove(&block_hash))
906        {
907            chain.execution_state = execution_state
908                .with_context(|ctx| {
909                    chain
910                        .execution_state
911                        .context()
912                        .clone_with_base_key(ctx.base_key().bytes.clone())
913                })
914                .await;
915            certificate.into_value()
916        } else {
917            let (proposed_block, outcome) = certificate.into_value().into_block().into_proposal();
918            let oracle_responses = Some(outcome.oracle_responses.clone());
919            let (proposed_block, verified, _resource_tracker) = chain
920                .execute_block(
921                    proposed_block,
922                    local_time,
923                    None,
924                    &published_blobs,
925                    oracle_responses,
926                    BundleExecutionPolicy::committed(),
927                )
928                .await?;
929            // We should always agree on the messages and state hash.
930            if outcome != verified {
931                return Err(ChainError::CorruptedChainState(format!(
932                    "computed block outcome differs from the certificate.\n\
933                    Computed: {verified:#?}\n\
934                    Submitted: {outcome:#?}"
935                ))
936                .into());
937            }
938            ConfirmedBlock::new(Block::new(proposed_block, verified))
939        };
940
941        // Update the rest of the chain state.
942        let updated_streams = chain
943            .apply_confirmed_block(&confirmed_block, local_time)
944            .await?;
945        let mut actions = self.create_network_actions(None).await?;
946        trace!("Processed confirmed block {height}");
947        actions.notifications.push(Notification {
948            chain_id,
949            reason: Reason::NewBlock {
950                height,
951                hash: block_hash,
952            },
953        });
954        if !updated_streams.is_empty() {
955            actions.notifications.push(Notification {
956                chain_id,
957                reason: Reason::NewEvents {
958                    height,
959                    block_hash,
960                    event_streams: updated_streams,
961                },
962            });
963        }
964        // Persist chain.
965        self.save().await?;
966
967        self.block_values
968            .insert_hashed(Cow::Owned(confirmed_block.into_inner()));
969
970        self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
971            .await;
972
973        Ok((
974            self.chain_info_response().await?,
975            actions,
976            BlockOutcome::Processed,
977        ))
978    }

981    /// `height`.
982    #[instrument(level = "trace", skip(self, notify_when_messages_are_delivered))]
983    async fn register_delivery_notifier(
984        &self,
985        height: BlockHeight,
986        actions: &NetworkActions,
987        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
988    ) {

991                .cross_chain_requests
992                .iter()
993                .any(|request| request.has_messages_lower_or_equal_than(height))

995                self.delivery_notifier.register(height, notifier);
996            } else {
997                // No need to wait. Also, cross-chain requests may not trigger the
998                // notifier later, even if we register it.
999                if let Err(()) = notifier.send(()) {
1000                    warn!("Failed to notify message delivery to caller");
1001                }
1002            }
1003        }

1006    /// Updates the chain's inboxes, receiving messages from a cross-chain update.
1007    #[instrument(level = "debug", skip(self, bundles), fields(chain_id = %self.chain_id()))]
1008    pub(crate) async fn process_cross_chain_update(
1009        &mut self,
1010        origin: ChainId,
1011        bundles: Vec<(Epoch, MessageBundle)>,
1012        sender_previous_height: Option<BlockHeight>,
1013    ) -> Result<CrossChainUpdateResult, WorkerError> {
1014        // Only process certificates with relevant heights and epochs.
1015        let mut inbox = self.chain.inboxes.try_load_entry_mut(&origin).await?;
1016        let next_height_to_receive = inbox.next_block_height_to_receive()?;
1017        let last_anticipated_block_height = match inbox.removed_bundles.back().await? {
1018            Some(bundle) => Some(bundle.height),
1019            None => None,
1020        };
1021
1022        // Proactive gap detection: if the sender declares a predecessor height that
1023        // we haven't received yet, the inbox has a gap.
1024        if let Some(prev) = sender_previous_height {
1025            if prev >= next_height_to_receive {
1026                let chain_id = self.chain_id();
1027                if self.config.allow_revert_confirm && self.config.recovery_allowed_for(&chain_id) {
1028                    warn!(
1029                        %chain_id,
1030                        "Inbox gap detected from {origin}: \
1031                        sender declares previous height {prev} but we only have up to \
1032                        {next_height_to_receive}; requesting resend",
1033                    );
1034                    return Ok(CrossChainUpdateResult::GapDetected {
1035                        origin,
1036                        retransmit_from: next_height_to_receive,
1037                    });
1038                }
1039                return Err(ChainError::InboxGapDetected {
1040                    chain_id,
1041                    origin,
1042                    expected_height: prev,
1043                    actual_height: bundles.first().map(|(_, b)| b.height).unwrap_or_default(),
1044                }
1045                .into());
1046            }
1047        }
1048
1049        let helper = CrossChainUpdateHelper::new(&self.config, &self.chain);
1050        let recipient = self.chain_id();
1051        let bundles = helper.select_message_bundles(
1052            &origin,
1053            recipient,
1054            next_height_to_receive,
1055            last_anticipated_block_height,
1056            bundles,
1057        )?;
1058        let Some(last_updated_height) = bundles.last().map(|bundle| bundle.height) else {
1059            return Ok(CrossChainUpdateResult::NothingToDo);
1060        };
1061        // Process the received messages in certificates.
1062        let local_time = self.storage.clock().current_time();
1063        let mut previous_height = None;
1064        for bundle in bundles {
1065            let add_to_received_log = previous_height != Some(bundle.height);
1066            previous_height = Some(bundle.height);
1067            // Update the staged chain state with the received block.
1068            self.chain
1069                .receive_message_bundle_with_inbox(
1070                    &mut inbox,
1071                    &origin,
1072                    bundle,
1073                    local_time,
1074                    add_to_received_log,
1075                )
1076                .await?;
1077        }
1078        inbox.observe_size_metric();
1079        drop(inbox);
1080        if !self.config.allow_inactive_chains && !self.chain.is_active().await? {
1081            // Refuse to create a chain state if the chain is still inactive by
1082            // now. Accordingly, do not send a confirmation, so that the
1083            // cross-chain update is retried later.
1084            warn!(
1085                chain_id = %self.chain_id(),
1086                "Refusing to deliver messages from {origin} \
1087                at height {last_updated_height} because the recipient is still inactive",
1088            );
1089            return Ok(CrossChainUpdateResult::NothingToDo);
1090        }
1091        // Save the chain.
1092        self.save().await?;
1093        Ok(CrossChainUpdateResult::Updated(last_updated_height))
1094    }

1129    /// Handles a `RevertConfirm` request: walks backward through
1130    /// `previous_message_blocks` to find all block heights that sent messages to
1131    /// `recipient` starting from the latest down to `retransmit_from`, re-adds them
1132    /// to the outbox, and creates cross-chain update actions to resend the bundles.
1133    #[instrument(skip_all, fields(
1134        chain_id = %self.chain_id(),
1135        %recipient,
1136        %retransmit_from,
1137    ))]
1138    pub(crate) async fn handle_revert_confirm(
1139        &mut self,
1140        recipient: ChainId,
1141        retransmit_from: BlockHeight,
1142    ) -> Result<NetworkActions, WorkerError> {
1143        // 1. Walk backward through previous_message_blocks to collect all heights
1144        //    that sent messages to this recipient, from the latest down to retransmit_from.
1145        let Some(latest_height) = self
1146            .chain
1147            .execution_state
1148            .previous_message_blocks
1149            .get(&recipient)
1150            .await?
1151        else {
1152            warn!("RevertConfirm: no record of sending to {recipient}");
1153            return Ok(NetworkActions::default());
1154        };
1155
1156        let mut heights_to_re_add = Vec::new();
1157        let mut current_height = latest_height;
1158        while current_height >= retransmit_from {
1159            // We arrived at current_height via previous_message_blocks links, starting from the
1160            // chain state and following the links downwards. So these blocks should all be in
1161            // confirmed_log or preprocessed_blocks already.
1162            heights_to_re_add.push(current_height);
1163            // Load the block at current_height to find the previous message block
1164            let hash = match &*self.chain.block_hashes([current_height]).await? {
1165                [hash] => *hash,
1166                _ => {
1167                    return Err(WorkerError::ConfirmedBlockHashNotFound {
1168                        height: current_height,
1169                        chain_id: self.chain_id(),
1170                    })
1171                }
1172            };
1173            let block = self
1174                .read_confirmed_blocks(&[hash])
1175                .await?
1176                .pop()
1177                .flatten()
1178                .ok_or_else(|| WorkerError::LocalBlockNotFound {
1179                    height: current_height,
1180                    chain_id: self.chain_id(),
1181                })?;
1182            match block.block().body.previous_message_blocks.get(&recipient) {
1183                Some((_, prev_height)) if *prev_height >= retransmit_from => {
1184                    current_height = *prev_height;
1185                }
1186                _ => break,
1187            }
1188        }
1189
1190        // 2. Re-add the heights to the outbox.
1191        let new_heights = self
1192            .chain
1193            .outboxes
1194            .try_load_entry_mut(&recipient)
1195            .await?
1196            .revert(&heights_to_re_add)
1197            .await?;
1198
1199        if new_heights.is_empty() {
1200            debug!("RevertConfirm: all heights already in outbox for {recipient}");
1201            return Ok(NetworkActions::default());
1202        }
1203
1204        // 3. Update outbox_counters (+1 per new height for this one recipient).
1205        let new_heights_len = new_heights.len();
1206        for h in new_heights {
1207            *self.chain.outbox_counters.get_mut().entry(h).or_default() += 1;
1208        }
1209        self.chain.nonempty_outboxes.get_mut().insert(recipient);
1210
1211        // 4. Create cross-chain requests for this recipient.
1212        let actions = self
1213            .create_cross_chain_actions_for_recipient(recipient)
1214            .await?;
1215
1216        // 5. Save chain state.
1217        self.save().await?;
1218
1219        warn!(
1220            "RevertConfirm: re-added {new_heights_len} heights to outbox for {recipient}, \
1221            starting from height {retransmit_from}"
1222        );
1223
1224        Ok(actions)
1225    }

1227    /// If the config enables corruption recovery and the min-duration guard is
1228    /// satisfied, resets the chain state and re-executes all confirmed blocks.
1229    /// Returns `RevertConfirm` requests to dispatch, or `None` if no reset happened.
1230    pub(crate) async fn maybe_reset_corrupted_chain_state(
1231        &mut self,
1232    ) -> Result<Option<Vec<CrossChainRequest>>, WorkerError> {
1233        let Some(min_duration) = self.config.reset_on_corrupted_chain_state else {
1234            return Ok(None);
1235        };
1236        let chain_id = self.chain_id();
1237        if !self.config.recovery_allowed_for(&chain_id) {
1238            return Ok(None);
1239        }
1240        let local_time = self.storage.clock().current_time();
1241        let block_zero_time = *self.chain.block_zero_executed_at.get();
1242        let elapsed = local_time.duration_since(block_zero_time);
1243        if elapsed < min_duration {
1244            warn!(
1245                %chain_id, ?elapsed, ?min_duration,

1252        Ok(Some(self.reset_and_reexecute_chain().await?))

1255    /// Resets the chain state completely and re-executes all confirmed blocks from storage.
1256    /// Returns a `RevertConfirm` request for every known sender so they resend cross-chain
1257    /// messages that may have been lost during the reset.
1258    #[instrument(skip_all, fields(
1259        chain_id = %self.chain_id(),
1260    ))]
1261    pub(crate) async fn reset_and_reexecute_chain(
1262        &mut self,
1263    ) -> Result<Vec<CrossChainRequest>, WorkerError> {
1264        let chain_id = self.chain_id();
1265        let tip_height = self.chain.tip_state.get().next_block_height;
1266
1267        // 1. Collect all sender chain IDs and block hashes before clearing.
1268        let sender_ids = self.chain.inboxes.indices().await?;
1269        let hashes = self.chain.confirmed_log.read(..).await?;
1270        let preprocessed = self.chain.preprocessed_blocks.index_values().await?;
1271
1272        // 2. Snapshot safety-critical manager state so that we cannot be tricked
1273        //    into double-signing if the reset wipes votes we already cast.
1274        let manager_snapshot = ManagerSafetySnapshot::capture(&self.chain.manager).await?;
1275
1276        // 3. Wipe every key under this chain's storage root and reload a fresh
1277        //    `ChainStateView`. `ChainStateView::clear` + `save` would only reset
1278        //    the in-memory view; `HistoricallyHashableView` deliberately keeps
1279        //    its `stored_hash` across clears, so the historical hash would carry
1280        //    over from the pre-reset state and the replayed block outcomes would
1281        //    never match the original certificates' `state_hash`. Deleting the
1282        //    storage prefix and reloading is equivalent to creating the chain
1283        //    from scratch, which is what replay expects.
1284        self.wipe_and_reload_chain().await?;
1285        self.knows_chain_is_active = false;
1286        warn!(
1287            %chain_id,
1288            "Cleared chain state up to height {tip_height}; \
1289            re-executing all blocks"
1290        );
1291
1292        // 4. Re-load certificates one at a time by hash and re-process them.
1293        for (index, hash) in hashes.into_iter().enumerate() {
1294            let height = BlockHeight(index as u64);
1295            let cert = self
1296                .storage
1297                .read_certificate(hash)
1298                .await?
1299                .map(Arc::unwrap_or_clone)
1300                .ok_or_else(|| WorkerError::LocalBlockNotFound { height, chain_id })?;
1301            Box::pin(self.process_confirmed_block(cert, None)).await?;
1302        }
1303        for (height, hash) in preprocessed {
1304            let cert = self
1305                .storage
1306                .read_certificate(hash)
1307                .await?
1308                .map(Arc::unwrap_or_clone)
1309                .ok_or_else(|| WorkerError::LocalBlockNotFound { height, chain_id })?;
1310            Box::pin(self.process_confirmed_block(cert, None)).await?;
1311        }
1312
1313        // 5. Restore any previously cast votes and locking block so we cannot be
1314        //    asked to sign a conflicting statement at the same height/round. Votes
1315        //    in the manager always belong to the pending height (one past the tip),
1316        //    so restoring is only meaningful if re-execution landed at the same
1317        //    tip. Otherwise the restored state would refer to a stale pending
1318        //    height and could only break the manager's invariants without any
1319        //    safety benefit — so we drop the snapshot in that case.
1320        let new_tip_height = self.chain.tip_state.get().next_block_height;
1321        if new_tip_height == tip_height {
1322            manager_snapshot.restore(&mut self.chain.manager)?;
1323            self.save().await?;
1324        } else {
1325            warn!(
1326                %tip_height, %new_tip_height,
1327                "Dropping manager snapshot: pre-reset tip differs from post-reset tip"
1328            );
1329        }
1330
1331        // 6. Build RevertConfirm requests so each sender resends messages we may
1332        //    have lost during the reset.
1333        let revert_requests = sender_ids
1334            .into_iter()
1335            .map(|sender| CrossChainRequest::RevertConfirm {
1336                sender,
1337                recipient: chain_id,
1338                retransmit_from: BlockHeight::ZERO,
1339            })
1340            .collect::<Vec<_>>();
1341
1342        warn!(
1343            tip_height = %self.chain.tip_state.get().next_block_height,
1344            num_revert_confirms = revert_requests.len(),
1345            "Chain reset and re-executed; sending RevertConfirm to senders"
1346        );
1347
1348        Ok(revert_requests)
1349    }

1787    /// Validates and executes a block proposed to extend this chain.
1788    #[instrument(skip_all, fields(
1789        chain_id = %self.chain_id(),
1790        block_height = %proposal.content.block.height
1791    ))]
1792    pub(crate) async fn handle_block_proposal(
1793        &mut self,
1794        proposal: BlockProposal,
1795    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
1796        self.initialize_and_save_if_needed().await?;
1797        proposal
1798            .check_invariants()
1799            .map_err(|msg| WorkerError::InvalidBlockProposal(msg.to_string()))?;
1800        proposal.check_signature()?;
1801        let owner = proposal.owner();
1802        let BlockProposal {
1803            content,
1804            original_proposal,
1805            signature: _,
1806        } = &proposal;
1807        let block = &content.block;
1808        let chain = &self.chain;
1809        // Check if the chain is ready for this new block proposal.
1810        chain.tip_state.get().verify_block_chaining(block)?;
1811        // Check the epoch.
1812        let (epoch, committee) = chain.current_committee().await?;
1813        check_block_epoch(epoch, block.chain_id, block.epoch)?;
1814        let policy = committee.policy().clone();
1815        block.check_proposal_size(policy.maximum_block_proposal_size)?;
1816        // Check the authentication of the block.
1817        ensure!(
1818            chain.manager.can_propose(&owner, proposal.content.round),
1819            WorkerError::InvalidOwner
1820        );
1821        let old_round = self.chain.manager.current_round();
1822        match original_proposal {
1823            None => {
1824                if let Some(signer) = block.authenticated_owner {
1825                    // Check the authentication of the operations in the new block.
1826                    ensure!(signer == owner, WorkerError::InvalidSigner(owner));
1827                }
1828            }
1829            Some(OriginalProposal::Regular { certificate }) => {
1830                // Verify that this block has been validated by a quorum before.
1831                certificate.check(&committee)?;
1832            }
1833            Some(OriginalProposal::Fast(signature)) => {
1834                let original_proposal = BlockProposal {
1835                    content: ProposalContent {
1836                        block: content.block.clone(),
1837                        round: Round::Fast,
1838                        outcome: None,
1839                    },
1840                    signature: *signature,
1841                    original_proposal: None,
1842                };
1843                let super_owner = original_proposal.owner();
1844                ensure!(
1845                    chain
1846                        .manager
1847                        .ownership
1848                        .get()
1849                        .super_owners
1850                        .contains(&super_owner),
1851                    WorkerError::InvalidOwner
1852                );
1853                if let Some(signer) = block.authenticated_owner {
1854                    // Check the authentication of the operations in the new block.
1855                    ensure!(signer == super_owner, WorkerError::InvalidSigner(signer));
1856                }
1857                original_proposal.check_signature()?;
1858            }
1859        }
1860        if chain.manager.check_proposed_block(&proposal)? == manager::Outcome::Skip {
1861            // We already voted for this block.
1862            return Ok((self.chain_info_response().await?, NetworkActions::default()));
1863        }
1864        let local_time = self.storage.clock().current_time();
1865
1866        // Make sure we remember that a proposal was signed, to determine the correct round to
1867        // propose in.
1868        if self
1869            .chain
1870            .manager
1871            .update_signed_proposal(&proposal, local_time)
1872        {
1873            self.save().await?;
1874        }
1875
1876        let published_blobs = self.load_proposal_blobs(&proposal).await?;
1877        let ProposalContent {
1878            block,
1879            round,
1880            outcome,
1881        } = content;
1882
1883        if self.config.key_pair().is_some()
1884            && block.timestamp.duration_since(local_time) > self.config.block_time_grace_period
1885        {
1886            return Err(WorkerError::InvalidTimestamp {
1887                local_time,
1888                block_timestamp: block.timestamp,
1889                block_time_grace_period: self.config.block_time_grace_period,
1890            });
1891        }
1892        // Note: WorkerState::handle_block_proposal delays processing proposals with future
1893        // timestamps (within the grace period) before acquiring the chain lock. By the time
1894        // we reach here, the block timestamp should be in the past or very close to current time.
1895
1896        self.chain
1897            .remove_bundles_from_inboxes(block.timestamp, true, block.incoming_bundles())
1898            .await?;
1899        let block = if let Some(outcome) = outcome {
1900            outcome.clone().with(proposal.content.block.clone())
1901        } else {
1902            let (executed_block, _resource_tracker) = Box::pin(self.execute_block(
1903                block.clone(),
1904                local_time,
1905                round.multi_leader(),
1906                &published_blobs,
1907                BundleExecutionPolicy::committed(),
1908            ))
1909            .await?;
1910            executed_block
1911        };
1912
1913        ensure!(
1914            !round.is_fast() || !block.has_oracle_responses(),
1915            WorkerError::FastBlockUsingOracles
1916        );
1917        let chain = &mut self.chain;
1918        // Check if the counters of tip_state would be valid.
1919        chain
1920            .tip_state
1921            .get_mut()
1922            .update_counters(&block.body.transactions, &block.body.messages)?;
1923        // Don't save the changes since the block is not confirmed yet.
1924        chain.rollback();
1925
1926        // Create the vote and store it in the chain state.
1927        let blobs = self
1928            .get_required_blobs(proposal.expected_blob_ids(), block.created_blobs())
1929            .await?;
1930        let key_pair = self.config.key_pair();
1931        let manager = &mut self.chain.manager;
1932        match manager.create_vote(&proposal, block, key_pair, local_time, blobs)? {
1933            // Cache the value we voted on, so the client doesn't have to send it again.
1934            Some(Either::Left(vote)) => {
1935                self.block_values
1936                    .insert_hashed(Cow::Borrowed(vote.value.inner()));
1937            }
1938            Some(Either::Right(vote)) => {
1939                self.block_values
1940                    .insert_hashed(Cow::Borrowed(vote.value.inner()));
1941            }
1942            None => (),
1943        }
1944        self.save().await?;
1945        let actions = self.create_network_actions(Some(old_round)).await?;
1946        Ok((self.chain_info_response().await?, actions))
1947    }

1950    #[instrument(skip_all, fields(
1951        chain_id = %self.chain_id()
1952    ))]

1954        &mut self,
1955        query: ChainInfoQuery,
1956    ) -> Result<ChainInfoResponse, WorkerError> {
1957        self.initialize_and_save_if_needed().await?;
1958        let chain = &self.chain;
1959        let mut info = ChainInfo::from_chain_view(chain).await?;
1960        if query.request_committees {
1961            info.requested_committees = Some(chain.execution_state.system.committees.get().clone());
1962        }
1963        if query.request_owner_balance == AccountOwner::CHAIN {
1964            info.requested_owner_balance = Some(*chain.execution_state.system.balance.get());
1965        } else {
1966            info.requested_owner_balance = chain
1967                .execution_state
1968                .system
1969                .balances
1970                .get(&query.request_owner_balance)
1971                .await?;
1972        }
1973        if let Some(next_block_height) = query.test_next_block_height {
1974            // If not, send the same error as if a block with next_block_height was proposed.
1975            ensure!(
1976                chain.tip_state.get().next_block_height == next_block_height,
1977                WorkerError::UnexpectedBlockHeight {
1978                    expected_block_height: chain.tip_state.get().next_block_height,
1979                    found_block_height: next_block_height,
1980                }
1981            );
1982        }
1983        if query.request_pending_message_bundles {
1984            let mut bundles = Vec::new();
1985            let nonempty_origins: Vec<ChainId> =
1986                chain.nonempty_inboxes.get().iter().copied().collect();
1987            #[cfg(with_metrics)]
1988            metrics::NUM_INBOXES
1989                .with_label_values(&[])
1990                .observe(nonempty_origins.len() as f64);
1991            let action = if *chain.execution_state.system.closed.get() {
1992                MessageAction::Reject
1993            } else {
1994                MessageAction::Accept
1995            };
1996            let inboxes = chain.inboxes.try_load_entries(&nonempty_origins).await?;
1997            for (origin, inbox) in nonempty_origins.into_iter().zip(inboxes) {
1998                let inbox = inbox.ok_or_else(|| {
1999                    ChainError::InternalError(format!("Missing inbox for origin {origin}"))
2000                })?;
2001                for bundle in inbox.added_bundles.elements().await? {
2002                    bundles.push(IncomingBundle {
2003                        origin,
2004                        bundle,
2005                        action,
2006                    });
2007                }
2008            }
2009            let priority_origins = &self.config.priority_bundle_origins;
2010            bundles.sort_by(|a, b| {
2011                let a_priority = priority_origins.contains(&a.origin);
2012                let b_priority = priority_origins.contains(&b.origin);
2013                b_priority
2014                    .cmp(&a_priority)
2015                    .then(a.bundle.timestamp.cmp(&b.bundle.timestamp))
2016            });
2017            info.requested_pending_message_bundles = bundles;
2018        }
2019        let hashes = chain
2020            .block_hashes(query.request_sent_certificate_hashes_by_heights)
2021            .await?;
2022        info.requested_sent_certificate_hashes = hashes;
2023        if let Some(start) = query.request_received_log_excluding_first_n {
2024            let start = usize::try_from(start).map_err(|_| ArithmeticError::Overflow)?;
2025            let max_received_log_entries = self.config.chain_info_max_received_log_entries;
2026            let end = start
2027                .saturating_add(max_received_log_entries)
2028                .min(chain.received_log.count());
2029            info.requested_received_log = chain.received_log.read(start..end).await?;
2030        }
2031        if query.request_manager_values {
2032            info.manager.add_values(&chain.manager);
2033        }
2034        if !query.request_previous_event_blocks.is_empty() {
2035            let stream_ids = query.request_previous_event_blocks;
2036            let heights = chain
2037                .execution_state
2038                .previous_event_blocks
2039                .multi_get(&stream_ids)
2040                .await?;
2041            let mut indices = Vec::new();
2042            let mut streams_with_heights = Vec::new();
2043            for (stream_id, height) in stream_ids.into_iter().zip(heights) {
2044                if let Some(height) = height {
2045                    let index = usize::try_from(height.0).map_err(|_| ArithmeticError::Overflow)?;
2046                    indices.push(index);
2047                    streams_with_heights.push((stream_id, height));
2048                }
2049            }
2050            let hashes = chain.confirmed_log.multi_get(indices).await?;
2051            for (maybe_hash, (stream_id, height)) in hashes.into_iter().zip(streams_with_heights) {
2052                let hash = maybe_hash.ok_or_else(|| WorkerError::ConfirmedLogEntryNotFound {
2053                    height,
2054                    chain_id: info.chain_id,
2055                })?;
2056                info.requested_previous_event_blocks
2057                    .insert(stream_id, (height, hash));
2058            }
2059        }
2060        Ok(ChainInfoResponse::new(info, self.config.key_pair()))
2061    }

2220/// Helper type for handling cross-chain updates.
2221pub(crate) struct CrossChainUpdateHelper<'a> {
2222    pub(crate) allow_messages_from_deprecated_epochs: bool,
2223    pub(crate) current_epoch: Epoch,
2224    pub(crate) committees: &'a BTreeMap<Epoch, CryptoHash>,

2226
2227impl<'a> CrossChainUpdateHelper<'a> {
2228    /// Creates a new [`CrossChainUpdateHelper`].
2229    fn new<C>(config: &ChainWorkerConfig, chain: &'a ChainStateView<C>) -> Self
2230    where
2231        C: Context + Clone + 'static,
2232    {
2233        CrossChainUpdateHelper {
2234            allow_messages_from_deprecated_epochs: config.allow_messages_from_deprecated_epochs,
2235            current_epoch: *chain.execution_state.system.epoch.get(),
2236            committees: chain.execution_state.system.committees.get(),
2237        }

2240    /// Checks basic invariants and deals with repeated heights and deprecated epochs.
2241    /// * Returns a range of message bundles that are both new to us and not relying on
2242    ///   an untrusted set of validators.
2243    /// * In the case of validators, if the epoch(s) of the highest bundles are not
2244    ///   trusted, we only accept bundles that contain messages that were already
2245    ///   executed by anticipation (i.e. received in certified blocks).
2246    /// * Basic invariants are checked for good measure. We still crucially trust
2247    ///   the worker of the sending chain to have verified and executed the blocks
2248    ///   correctly.
2249    pub(crate) fn select_message_bundles(
2250        &self,
2251        origin: &'a ChainId,
2252        recipient: ChainId,
2253        next_height_to_receive: BlockHeight,
2254        last_anticipated_block_height: Option<BlockHeight>,
2255        mut bundles: Vec<(Epoch, MessageBundle)>,
2256    ) -> Result<Vec<MessageBundle>, WorkerError> {
2257        let mut latest_height = None;
2258        let mut skipped_len = 0;
2259        let mut trusted_len = 0;
2260        for (i, (epoch, bundle)) in bundles.iter().enumerate() {
2261            // Make sure that heights are not decreasing.
2262            ensure!(
2263                latest_height <= Some(bundle.height),
2264                WorkerError::InvalidCrossChainRequest
2265            );
2266            latest_height = Some(bundle.height);
2267            // Check if the block has been received already.
2268            if bundle.height < next_height_to_receive {
2269                skipped_len = i + 1;
2270            }
2271            // Check if the height is trusted or the epoch is trusted.
2272            if self.allow_messages_from_deprecated_epochs
2273                || Some(bundle.height) <= last_anticipated_block_height
2274                || *epoch >= self.current_epoch
2275                || self.committees.contains_key(epoch)
2276            {
2277                trusted_len = i + 1;
2278            }
2279        }
2280        if skipped_len > 0 {
2281            let (_, sample_bundle) = &bundles[skipped_len - 1];
2282            debug!(
2283                "Ignoring repeated messages to {recipient:.8} from {origin:} at height {}",
2284                sample_bundle.height,
2285            );
2286        }
2287        if skipped_len < bundles.len() && trusted_len < bundles.len() {
2288            let (sample_epoch, sample_bundle) = &bundles[trusted_len];
2289            warn!(
2290                "Refusing messages to {recipient:.8} from {origin:} at height {} \
2291                 because the epoch {} is not trusted any more",
2292                sample_bundle.height, sample_epoch,
2293            );
2294        }
2295        let bundles = if skipped_len < trusted_len {
2296            bundles
2297                .drain(skipped_len..trusted_len)
2298                .map(|(_, bundle)| bundle)
2299                .collect()
2300        } else {
2301            vec![]
2302        };
2303        Ok(bundles)
2304    }
