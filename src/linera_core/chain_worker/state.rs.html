134    /// Handles a request and applies it to the chain state.
135    #[instrument(skip_all, fields(chain_id = %self.chain_id()))]
136    pub(super) async fn handle_request(
137        &mut self,
138        request: ChainWorkerRequest<StorageClient::Context>,
139    ) {
140        tracing::trace!("Handling chain worker request: {request:?}");
141        // TODO(#2237): Spawn concurrent tasks for read-only operations
142        let responded = match request {
143            #[cfg(with_testing)]
144            ChainWorkerRequest::ReadCertificate { height, callback } => {
145                callback.send(self.read_certificate(height).await).is_ok()
146            }
147            ChainWorkerRequest::GetChainStateView { callback } => {
148                callback.send(self.chain_state_view().await).is_ok()
149            }
150            ChainWorkerRequest::QueryApplication {
151                query,
152                block_hash,
153                callback,
154            } => callback
155                .send(self.query_application(query, block_hash).await)
156                .is_ok(),
157            ChainWorkerRequest::DescribeApplication {
158                application_id,
159                callback,
160            } => callback
161                .send(self.describe_application(application_id).await)
162                .is_ok(),
163            ChainWorkerRequest::StageBlockExecution {
164                block,
165                round,
166                published_blobs,
167                callback,
168            } => callback
169                .send(
170                    self.stage_block_execution(block, round, &published_blobs)
171                        .await,
172                )
173                .is_ok(),
174            ChainWorkerRequest::StageBlockExecutionWithPolicy {
175                block,
176                round,
177                published_blobs,
178                policy,
179                callback,
180            } => {
181                let result = self
182                    .stage_block_execution_with_policy(block, round, &published_blobs, policy)
183                    .await;
184                callback.send(result).is_ok()
185            }
186            ChainWorkerRequest::ProcessTimeout {
187                certificate,
188                callback,
189            } => callback
190                .send(self.process_timeout(certificate).await)
191                .is_ok(),
192            ChainWorkerRequest::HandleBlockProposal { proposal, callback } => callback
193                .send(self.handle_block_proposal(proposal).await)
194                .is_ok(),
195            ChainWorkerRequest::ProcessValidatedBlock {
196                certificate,
197                callback,
198            } => callback
199                .send(self.process_validated_block(certificate).await)
200                .is_ok(),
201            ChainWorkerRequest::ProcessConfirmedBlock {
202                certificate,
203                notify_when_messages_are_delivered,
204                callback,
205            } => callback
206                .send(
207                    self.process_confirmed_block(certificate, notify_when_messages_are_delivered)
208                        .await,
209                )
210                .is_ok(),
211            ChainWorkerRequest::ProcessCrossChainUpdate {
212                origin,
213                bundles,
214                callback,
215            } => callback
216                .send(self.process_cross_chain_update(origin, bundles).await)
217                .is_ok(),
218            ChainWorkerRequest::ConfirmUpdatedRecipient {
219                recipient,
220                latest_height,
221                callback,
222            } => callback
223                .send(
224                    self.confirm_updated_recipient(recipient, latest_height)
225                        .await,
226                )
227                .is_ok(),
228            ChainWorkerRequest::HandleChainInfoQuery { query, callback } => callback
229                .send(self.handle_chain_info_query(query).await)
230                .is_ok(),
231            ChainWorkerRequest::DownloadPendingBlob { blob_id, callback } => callback
232                .send(self.download_pending_blob(blob_id).await)
233                .is_ok(),
234            ChainWorkerRequest::HandlePendingBlob { blob, callback } => {
235                callback.send(self.handle_pending_blob(blob).await).is_ok()
236            }
237            ChainWorkerRequest::UpdateReceivedCertificateTrackers {
238                new_trackers,
239                callback,
240            } => callback
241                .send(
242                    self.update_received_certificate_trackers(new_trackers)
243                        .await,
244                )
245                .is_ok(),
246            ChainWorkerRequest::GetPreprocessedBlockHashes {
247                start,
248                end,
249                callback,
250            } => callback
251                .send(self.get_preprocessed_block_hashes(start, end).await)
252                .is_ok(),
253            ChainWorkerRequest::GetInboxNextHeight { origin, callback } => callback
254                .send(self.get_inbox_next_height(origin).await)
255                .is_ok(),
256            ChainWorkerRequest::GetLockingBlobs { blob_ids, callback } => callback
257                .send(self.get_locking_blobs(blob_ids).await)
258                .is_ok(),
259            ChainWorkerRequest::GetBlockHashes { heights, callback } => {
260                callback.send(self.get_block_hashes(heights).await).is_ok()
261            }
262            ChainWorkerRequest::GetProposedBlobs { blob_ids, callback } => callback
263                .send(self.get_proposed_blobs(blob_ids).await)
264                .is_ok(),
265            ChainWorkerRequest::GetEventSubscriptions { callback } => {
266                callback.send(self.get_event_subscriptions().await).is_ok()
267            }
268            ChainWorkerRequest::GetNextExpectedEvent {
269                stream_id,
270                callback,
271            } => callback
272                .send(self.get_next_expected_event(stream_id).await)
273                .is_ok(),
274            ChainWorkerRequest::GetReceivedCertificateTrackers { callback } => callback
275                .send(self.get_received_certificate_trackers().await)
276                .is_ok(),
277            ChainWorkerRequest::GetTipStateAndOutboxInfo {
278                receiver_id,
279                callback,
280            } => callback
281                .send(self.get_tip_state_and_outbox_info(receiver_id).await)
282                .is_ok(),
283            ChainWorkerRequest::GetNextHeightToPreprocess { callback } => callback
284                .send(self.get_next_height_to_preprocess().await)
285                .is_ok(),
286        };
287
288        if !responded {
289            debug!("Callback for `ChainWorkerActor` was dropped before a response was sent");
290        }
291
292        // Roll back any unsaved changes to the chain state: If there was an error while trying
293        // to handle the request, the chain state might contain unsaved and potentially invalid
294        // changes. The next request needs to be applied to the chain state as it is in storage.
295        self.chain.rollback();
296    }

299    ///
300    /// The returned view holds a lock on the chain state, which prevents the worker from changing
301    /// it.
302    async fn chain_state_view(
303        &mut self,
304    ) -> Result<OwnedRwLockReadGuard<ChainStateView<StorageClient::Context>>, WorkerError> {
305        if self.shared_chain_view.is_none() {
306            self.shared_chain_view = Some(Arc::new(RwLock::new(self.chain.clone_unchecked()?)));
307        }
308
309        Ok(self
310            .shared_chain_view
311            .as_ref()
312            .expect("`shared_chain_view` should be initialized above")
313            .clone()
314            .read_owned()
315            .await)
316    }

318    /// Clears the shared chain view, and acquires and drops its write lock.
319    ///
320    /// This is the only place a write lock is acquired, and read locks are acquired in
321    /// the `chain_state_view` method, which has a `&mut self` receiver like this one.
322    /// That means that when this function returns, no readers will be waiting to acquire
323    /// the lock and it is safe to write the chain state to storage without any readers
324    /// having a stale view of it.
325    #[instrument(skip_all, fields(
326        chain_id = %self.chain_id()
327    ))]
328    pub(super) async fn clear_shared_chain_view(&mut self) {
329        if let Some(shared_chain_view) = self.shared_chain_view.take() {
330            let _: RwLockWriteGuard<_> = shared_chain_view.write().await;
331        }
332    }

334    /// Handles a [`ChainInfoQuery`], potentially voting on the next block.
335    #[tracing::instrument(level = "debug", skip(self))]
336    async fn handle_chain_info_query(
337        &mut self,
338        query: ChainInfoQuery,
339    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
340        let create_network_actions = query.create_network_actions;
341        if let Some((height, round)) = query.request_leader_timeout {
342            self.vote_for_leader_timeout(height, round).await?;
343        }
344        if query.request_fallback {
345            self.vote_for_fallback().await?;
346        }
347        let response = self.prepare_chain_info_response(query).await?;
348        // Trigger any outgoing cross-chain messages that haven't been confirmed yet.
349        let actions = if create_network_actions {
350            self.create_network_actions(None).await?
351        } else {
352            NetworkActions::default()
353        };
354        Ok((response, actions))
355    }

364            return Ok(blob);

371    /// missing.
372    #[instrument(skip_all, fields(

374    ))]
375    async fn get_required_blobs(
376        &self,
377        required_blob_ids: impl IntoIterator<Item = BlobId>,
378        created_blobs: &BTreeMap<BlobId, Blob>,
379    ) -> Result<BTreeMap<BlobId, Blob>, WorkerError> {
380        let maybe_blobs = self
381            .maybe_get_required_blobs(required_blob_ids, Some(created_blobs))
382            .await?;
383        let not_found_blob_ids = missing_blob_ids(&maybe_blobs);
384        ensure!(
385            not_found_blob_ids.is_empty(),
386            WorkerError::BlobsNotFound(not_found_blob_ids)
387        );
388        Ok(maybe_blobs
389            .into_iter()
390            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
391            .collect())
392    }

394    /// Tries to read the blobs from the chain manager or storage. Returns `None` if not found.
395    #[instrument(skip_all, fields(
396        chain_id = %self.chain_id()
397    ))]
398    async fn maybe_get_required_blobs(
399        &self,
400        blob_ids: impl IntoIterator<Item = BlobId>,
401        created_blobs: Option<&BTreeMap<BlobId, Blob>>,
402    ) -> Result<BTreeMap<BlobId, Option<Blob>>, WorkerError> {
403        let maybe_blobs = blob_ids.into_iter().collect::<BTreeSet<_>>();
404        let mut maybe_blobs = maybe_blobs
405            .into_iter()
406            .map(|x| (x, None))
407            .collect::<Vec<(BlobId, Option<Blob>)>>();
408
409        if let Some(blob_map) = created_blobs {
410            for (blob_id, value) in &mut maybe_blobs {
411                if let Some(blob) = blob_map.get(blob_id) {
412                    *value = Some(blob.clone());
413                }
414            }
415        }
416
417        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
418        let second_block_blobs = self.chain.manager.pending_blobs(&missing_blob_ids).await?;
419        for (index, blob) in missing_indices.into_iter().zip(second_block_blobs) {
420            maybe_blobs[index].1 = blob;
421        }
422
423        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
424        let third_block_blobs = self
425            .chain
426            .pending_validated_blobs
427            .multi_get(&missing_blob_ids)
428            .await?;
429        for (index, blob) in missing_indices.into_iter().zip(third_block_blobs) {
430            maybe_blobs[index].1 = blob;
431        }
432
433        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
434        if !missing_indices.is_empty() {
435            let all_entries_pending_blobs = self
436                .chain
437                .pending_proposed_blobs
438                .try_load_all_entries()
439                .await?;
440            for (index, blob_id) in missing_indices.into_iter().zip(missing_blob_ids) {
441                for (_, pending_blobs) in &all_entries_pending_blobs {
442                    if let Some(blob) = pending_blobs.get(&blob_id).await? {
443                        maybe_blobs[index].1 = Some(blob);
444                        break;
445                    }
446                }
447            }
448        }
449
450        let (missing_indices, missing_blob_ids) = missing_indices_blob_ids(&maybe_blobs);
451        let fourth_block_blobs = self.storage.read_blobs(&missing_blob_ids).await?;
452        for (index, blob) in missing_indices.into_iter().zip(fourth_block_blobs) {
453            maybe_blobs[index].1 = blob;
454        }
455        Ok(maybe_blobs.into_iter().collect())
456    }

458    /// Loads pending cross-chain requests, and adds `NewRound` notifications where appropriate.
459    #[instrument(skip_all, fields(
460        chain_id = %self.chain_id()
461    ))]
462    async fn create_network_actions(
463        &self,
464        old_round: Option<Round>,
465    ) -> Result<NetworkActions, WorkerError> {
466        #[cfg(with_metrics)]
467        let _latency = metrics::CREATE_NETWORK_ACTIONS_LATENCY.measure_latency();
468        let mut heights_by_recipient = BTreeMap::<_, Vec<_>>::new();
469        let mut targets = self.chain.nonempty_outbox_chain_ids();
470        if let Some(chain_modes) = self.chain_modes.as_ref() {
471            let chain_modes = chain_modes
472                .read()
473                .expect("Panics should not happen while holding a lock to `chain_modes`");
474            // Only process outboxes for full chains (those whose inboxes we update).
475            targets.retain(|target| chain_modes.get(target).is_some_and(ListeningMode::is_full));
476        }
477        let outboxes = self.chain.load_outboxes(&targets).await?;
478        for (target, outbox) in targets.into_iter().zip(outboxes) {
479            let heights = outbox.queue.elements().await?;
480            heights_by_recipient.insert(target, heights);
481        }
482        let cross_chain_requests = self
483            .create_cross_chain_requests(heights_by_recipient)
484            .await?;
485        let mut notifications = Vec::new();
486        if let Some(old_round) = old_round {
487            let round = self.chain.manager.current_round();
488            if round > old_round {
489                let height = self.chain.tip_state.get().next_block_height;
490                notifications.push(Notification {
491                    chain_id: self.chain_id(),
492                    reason: Reason::NewRound { height, round },
493                });
494            }
495        }
496        Ok(NetworkActions {
497            cross_chain_requests,
498            notifications,
499        })
500    }

502    #[instrument(skip_all, fields(
503        chain_id = %self.chain_id(),
504        num_recipients = %heights_by_recipient.len()
505    ))]
506    async fn create_cross_chain_requests(
507        &self,
508        heights_by_recipient: BTreeMap<ChainId, Vec<BlockHeight>>,
509    ) -> Result<Vec<CrossChainRequest>, WorkerError> {
510        // Load all the certificates we will need, regardless of the medium.
511        let heights = BTreeSet::from_iter(heights_by_recipient.values().flatten().copied());
512        let next_block_height = self.chain.tip_state.get().next_block_height;
513        let log_heights = heights
514            .range(..next_block_height)
515            .copied()
516            .map(usize::try_from)
517            .collect::<Result<Vec<_>, _>>()?;
518        let mut hashes = self
519            .chain
520            .confirmed_log
521            .multi_get(log_heights)
522            .await?
523            .into_iter()
524            .zip(&heights)
525            .map(|(maybe_hash, height)| {
526                maybe_hash.ok_or_else(|| WorkerError::ConfirmedLogEntryNotFound {
527                    height: *height,
528                    chain_id: self.chain_id(),
529                })
530            })
531            .collect::<Result<Vec<_>, _>>()?;
532        let requested_heights: Vec<BlockHeight> = heights
533            .range(next_block_height..)
534            .copied()
535            .collect::<Vec<BlockHeight>>();
536        for (height, hash) in self
537            .chain
538            .preprocessed_blocks
539            .multi_get_pairs(requested_heights)
540            .await?
541        {
542            let hash = hash.ok_or_else(|| WorkerError::PreprocessedBlocksEntryNotFound {
543                height,
544                chain_id: self.chain_id(),
545            })?;
546            hashes.push(hash);
547        }
548
549        let mut uncached_hashes = Vec::new();
550        let mut height_to_blocks: HashMap<BlockHeight, Hashed<Block>> = HashMap::new();
551
552        for hash in hashes {
553            if let Some(hashed_block) = self.block_values.get(&hash) {
554                height_to_blocks.insert(hashed_block.inner().header.height, hashed_block);
555            } else {
556                uncached_hashes.push(hash);
557            }
558        }
559
560        if !uncached_hashes.is_empty() {
561            let certificates = self.storage.read_certificates(&uncached_hashes).await?;
562            let certificates = match ResultReadCertificates::new(certificates, uncached_hashes) {
563                ResultReadCertificates::Certificates(certificates) => certificates,
564                ResultReadCertificates::InvalidHashes(hashes) => {
565                    return Err(WorkerError::ReadCertificatesError(hashes))
566                }
567            };
568
569            for cert in certificates {
570                let hashed_block = cert.into_value().into_inner();
571                let height = hashed_block.inner().header.height;
572                self.block_values.insert(Cow::Owned(hashed_block.clone()));
573                height_to_blocks.insert(height, hashed_block);
574            }
575        }
576
577        let mut cross_chain_requests = Vec::new();
578        for (recipient, heights) in heights_by_recipient {
579            let mut bundles = Vec::new();
580            for height in heights {
581                let hashed_block = height_to_blocks
582                    .get(&height)
583                    .ok_or_else(|| ChainError::InternalError("missing block".to_string()))?;
584                bundles.extend(
585                    hashed_block
586                        .inner()
587                        .message_bundles_for(recipient, hashed_block.hash()),
588                );
589            }
590            let request = CrossChainRequest::UpdateRecipient {
591                sender: self.chain.chain_id(),
592                recipient,
593                bundles,
594            };
595            cross_chain_requests.push(request);
596        }
597        Ok(cross_chain_requests)
598    }

600    /// Returns true if there are no more outgoing messages in flight up to the given
601    /// block height for all tracked chains (those whose inboxes we update).
602    #[instrument(skip_all, fields(
603        chain_id = %self.chain_id(),
604        height = %height
605    ))]
606    async fn all_messages_to_tracked_chains_delivered_up_to(
607        &self,
608        height: BlockHeight,
609    ) -> Result<bool, WorkerError> {
610        if self.chain.all_messages_delivered_up_to(height) {
611            return Ok(true);
612        }
613        let Some(chain_modes) = self.chain_modes.as_ref() else {
614            return Ok(false);
615        };
616        let mut targets = self.chain.nonempty_outbox_chain_ids();
617        {
618            let chain_modes = chain_modes.read().unwrap();
619            // Only consider full chains (those whose inboxes we update).
620            targets.retain(|target| chain_modes.get(target).is_some_and(ListeningMode::is_full));
621        }
622        let outboxes = self.chain.load_outboxes(&targets).await?;
623        for outbox in outboxes {
624            let front = outbox.queue.front().await?;
625            if front.is_some_and(|key| key <= height) {
626                return Ok(false);
627            }
628        }
629        Ok(true)
630    }

688                    round,

722    /// Processes a validated block issued for this multi-owner chain.
723    #[instrument(skip_all, fields(
724        chain_id = %self.chain_id(),
725        block_height = %certificate.block().header.height
726    ))]
727    async fn process_validated_block(
728        &mut self,
729        certificate: ValidatedBlockCertificate,
730    ) -> Result<(ChainInfoResponse, NetworkActions, BlockOutcome), WorkerError> {
731        let block = certificate.block();
732
733        let header = &block.header;
734        let height = header.height;
735        // Check that the chain is active and ready for this validated block.
736        // Verify the certificate. Returns a catch-all error to make client code more robust.
737        self.initialize_and_save_if_needed().await?;
738        let tip_state = self.chain.tip_state.get();
739        ensure!(
740            header.height == tip_state.next_block_height,
741            ChainError::UnexpectedBlockHeight {
742                expected_block_height: tip_state.next_block_height,
743                found_block_height: header.height,
744            }
745        );
746        let (epoch, committee) = self.chain.current_committee()?;
747        check_block_epoch(epoch, header.chain_id, header.epoch)?;
748        certificate.check(committee)?;
749        let already_committed_block = self.chain.tip_state.get().already_validated_block(height)?;
750        let should_skip_validated_block = || {
751            self.chain
752                .manager
753                .check_validated_block(&certificate)
754                .map(|outcome| outcome == manager::Outcome::Skip)
755        };
756        if already_committed_block || should_skip_validated_block()? {
757            // If we just processed the same pending block, return the chain info unchanged.
758            return Ok((
759                self.chain_info_response(),
760                NetworkActions::default(),
761                BlockOutcome::Skipped,
762            ));
763        }
764
765        self.block_values
766            .insert(Cow::Borrowed(certificate.inner().inner()));
767        let required_blob_ids = block.required_blob_ids();
768        let maybe_blobs = self
769            .maybe_get_required_blobs(required_blob_ids, Some(&block.created_blobs()))
770            .await?;
771        let missing_blob_ids = missing_blob_ids(&maybe_blobs);
772        if !missing_blob_ids.is_empty() {
773            self.chain
774                .pending_validated_blobs
775                .update(certificate.round, true, maybe_blobs)?;
776            self.save().await?;
777            return Err(WorkerError::BlobsNotFound(missing_blob_ids));
778        }
779        let blobs = maybe_blobs
780            .into_iter()
781            .filter_map(|(blob_id, maybe_blob)| Some((blob_id, maybe_blob?)))
782            .collect();
783        let old_round = self.chain.manager.current_round();
784        self.chain.manager.create_final_vote(
785            certificate,
786            self.config.key_pair(),
787            self.storage.clock().current_time(),
788            blobs,
789        )?;
790        self.save().await?;
791        let actions = self.create_network_actions(Some(old_round)).await?;
792        Ok((self.chain_info_response(), actions, BlockOutcome::Processed))
793    }

795    /// Processes a confirmed block (aka a commit).
796    #[instrument(skip_all, fields(
797        chain_id = %certificate.block().header.chain_id,
798        height = %certificate.block().header.height,
799        block_hash = %certificate.hash(),
800    ))]
801    async fn process_confirmed_block(
802        &mut self,
803        certificate: ConfirmedBlockCertificate,
804        notify_when_messages_are_delivered: Option<oneshot::Sender<()>>,
805    ) -> Result<(ChainInfoResponse, NetworkActions, BlockOutcome), WorkerError> {
806        let block = certificate.block();
807        let block_hash = certificate.hash();
808        let height = block.header.height;
809        let chain_id = block.header.chain_id;
810
811        // Check that the chain is active and ready for this confirmation.
812        let tip = self.chain.tip_state.get().clone();
813        if tip.next_block_height > height {
814            // We already processed this block.
815            let actions = self.create_network_actions(None).await?;
816            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
817                .await;
818            return Ok((self.chain_info_response(), actions, BlockOutcome::Skipped));
819        }
820
821        // We haven't processed the block - verify the certificate first
822        let epoch = block.header.epoch;
823        // Get the committee for the block's epoch from storage.
824        if let Some(committee) = self
825            .chain
826            .execution_state
827            .system
828            .committees
829            .get()
830            .get(&epoch)
831        {
832            certificate.check(committee)?;
833        } else {
834            let committee = self
835                .chain
836                .execution_state
837                .context()
838                .extra()
839                .get_committees(epoch..=epoch)
840                .await
841                .map_err(|error| {
842                    ChainError::ExecutionError(Box::new(error), ChainExecutionContext::Block)
843                })?
844                .remove(&epoch)
845                .ok_or_else(|| {
846                    ChainError::InternalError(format!(
847                        "missing committee for epoch {epoch}; this is a bug"
848                    ))
849                })?;
850            certificate.check(&committee)?;
851        }
852
853        // Certificate check passed - which means the blobs the block requires are legitimate and
854        // we can take note of it, so that if any are missing, we will accept them when the client
855        // sends them.
856        let required_blob_ids = block.required_blob_ids();
857        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
858        let blobs_result = self
859            .get_required_blobs(required_blob_ids.iter().copied(), &created_blobs)
860            .await
861            .map(|blobs| blobs.into_values().collect::<Vec<_>>());
862
863        if let Ok(blobs) = &blobs_result {
864            self.storage
865                .write_blobs_and_certificate(blobs, &certificate)
866                .await?;
867            let events = block
868                .body
869                .events
870                .iter()
871                .flatten()
872                .map(|event| (event.id(chain_id), event.value.clone()));
873            self.storage.write_events(events).await?;
874        }
875
876        // Update the blob state with last used certificate hash.
877        let blob_state = certificate.value().to_blob_state(blobs_result.is_ok());
878        let blob_ids = required_blob_ids.into_iter().collect::<Vec<_>>();
879        self.storage
880            .maybe_write_blob_states(&blob_ids, blob_state)
881            .await?;
882
883        let mut blobs = blobs_result?
884            .into_iter()
885            .map(|blob| (blob.id(), blob))
886            .collect::<BTreeMap<_, _>>();
887
888        // If this block is higher than the next expected block in this chain, we're going
889        // to have a gap: do not execute this block, only update the outboxes and return.
890        if tip.next_block_height < height {
891            // Update the outboxes and event streams.
892            let updated_event_streams = self.chain.preprocess_block(certificate.value()).await?;
893            // Persist chain.
894            self.save().await?;
895            let mut actions = self.create_network_actions(None).await?;
896            if !updated_event_streams.is_empty() {
897                actions.notifications.push(Notification {
898                    chain_id,
899                    reason: Reason::NewEvents {
900                        height,
901                        hash: certificate.hash(),
902                        event_streams: updated_event_streams,
903                    },
904                });
905            }
906            trace!("Preprocessed confirmed block {height} on chain {chain_id:.8}");
907            self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
908                .await;
909            return Ok((
910                self.chain_info_response(),
911                actions,
912                BlockOutcome::Preprocessed,
913            ));
914        }
915
916        // This should always be true for valid certificates.
917        ensure!(
918            tip.block_hash == block.header.previous_block_hash,
919            WorkerError::InvalidBlockChaining
920        );
921
922        // If we got here, `height` is equal to `tip.next_block_height` and the block is
923        // properly chained. Verify that the chain is active and that the epoch we used for
924        // verifying the certificate is actually the active one on the chain.
925        self.initialize_and_save_if_needed().await?;
926        let (epoch, _) = self.chain.current_committee()?;
927        check_block_epoch(epoch, chain_id, block.header.epoch)?;
928
929        let published_blobs = block
930            .published_blob_ids()
931            .iter()
932            .filter_map(|blob_id| blobs.remove(blob_id))
933            .collect::<Vec<_>>();
934
935        // Execute the block and update inboxes.
936        let local_time = self.storage.clock().current_time();
937        if block.header.timestamp.duration_since(local_time) > self.config.block_time_grace_period {
938            warn!(
939                block_timestamp = %block.header.timestamp,
940                %local_time,
941                "Confirmed block has a timestamp in the future beyond the block time grace period"
942            );
943        }
944        let chain = &mut self.chain;
945        chain
946            .remove_bundles_from_inboxes(
947                block.header.timestamp,
948                false,
949                block.body.incoming_bundles(),
950            )
951            .await?;
952        let oracle_responses = Some(block.body.oracle_responses.clone());
953        let (proposed_block, outcome) = block.clone().into_proposal();
954        let verified_outcome =
955            if let Some(mut execution_state) = self.execution_state_cache.remove(&block_hash) {
956                chain.execution_state = execution_state
957                    .with_context(|ctx| {
958                        chain
959                            .execution_state
960                            .context()
961                            .clone_with_base_key(ctx.base_key().bytes.clone())
962                    })
963                    .await;
964                outcome.clone()
965            } else {
966                let (_, verified, _resource_tracker) = chain
967                    .execute_block(
968                        proposed_block,
969                        local_time,
970                        None,
971                        &published_blobs,
972                        oracle_responses,
973                        BundleExecutionPolicy::Abort,
974                    )
975                    .await?;
976                verified
977            };
978        // We should always agree on the messages and state hash.
979        ensure!(
980            outcome == verified_outcome,
981            WorkerError::IncorrectOutcome {
982                submitted: Box::new(outcome),
983                computed: Box::new(verified_outcome),
984            }
985        );
986
987        // Update the rest of the chain state.
988        let updated_streams = chain
989            .apply_confirmed_block(certificate.value(), local_time)
990            .await?;
991        let mut actions = self.create_network_actions(None).await?;
992        trace!("Processed confirmed block {height} on chain {chain_id:.8}");
993        let hash = certificate.hash();
994        actions.notifications.push(Notification {
995            chain_id,
996            reason: Reason::NewBlock { height, hash },
997        });
998        if !updated_streams.is_empty() {
999            actions.notifications.push(Notification {
1000                chain_id,
1001                reason: Reason::NewEvents {
1002                    height,
1003                    hash,
1004                    event_streams: updated_streams,
1005                },
1006            });
1007        }
1008        // Persist chain.
1009        self.save().await?;
1010
1011        self.block_values
1012            .insert(Cow::Owned(certificate.into_inner().into_inner()));
1013
1014        self.register_delivery_notifier(height, &actions, notify_when_messages_are_delivered)
1015            .await;
1016
1017        Ok((self.chain_info_response(), actions, BlockOutcome::Processed))
1018    }

1046    /// Updates the chain's inboxes, receiving messages from a cross-chain update.
1047    #[instrument(level = "trace", skip(self, bundles))]
1048    async fn process_cross_chain_update(
1049        &mut self,
1050        origin: ChainId,
1051        bundles: Vec<(Epoch, MessageBundle)>,
1052    ) -> Result<Option<BlockHeight>, WorkerError> {
1053        // Only process certificates with relevant heights and epochs.
1054        let next_height_to_receive = self.chain.next_block_height_to_receive(&origin).await?;
1055        let last_anticipated_block_height =
1056            self.chain.last_anticipated_block_height(&origin).await?;
1057        let helper = CrossChainUpdateHelper::new(&self.config, &self.chain);
1058        let recipient = self.chain_id();
1059        let bundles = helper.select_message_bundles(
1060            &origin,
1061            recipient,
1062            next_height_to_receive,
1063            last_anticipated_block_height,
1064            bundles,
1065        )?;
1066        let Some(last_updated_height) = bundles.last().map(|bundle| bundle.height) else {
1067            return Ok(None);
1068        };
1069        // Process the received messages in certificates.
1070        let local_time = self.storage.clock().current_time();
1071        let mut previous_height = None;
1072        for bundle in bundles {
1073            let add_to_received_log = previous_height != Some(bundle.height);
1074            previous_height = Some(bundle.height);
1075            // Update the staged chain state with the received block.
1076            self.chain
1077                .receive_message_bundle(&origin, bundle, local_time, add_to_received_log)
1078                .await?;
1079        }
1080        if !self.config.allow_inactive_chains && !self.chain.is_active() {
1081            // Refuse to create a chain state if the chain is still inactive by
1082            // now. Accordingly, do not send a confirmation, so that the
1083            // cross-chain update is retried later.
1084            warn!(
1085                "Refusing to deliver messages to {recipient:?} from {origin:?} \
1086                at height {last_updated_height} because the recipient is still inactive",
1087            );
1088            return Ok(None);
1089        }
1090        // Save the chain.
1091        self.save().await?;
1092        Ok(Some(last_updated_height))
1093    }

1541    /// Validates and executes a block proposed to extend this chain.
1542    #[instrument(skip_all, fields(
1543        chain_id = %self.chain_id(),
1544        block_height = %proposal.content.block.height
1545    ))]
1546    async fn handle_block_proposal(
1547        &mut self,
1548        proposal: BlockProposal,
1549    ) -> Result<(ChainInfoResponse, NetworkActions), WorkerError> {
1550        self.initialize_and_save_if_needed().await?;
1551        proposal
1552            .check_invariants()
1553            .map_err(|msg| WorkerError::InvalidBlockProposal(msg.to_string()))?;
1554        proposal.check_signature()?;
1555        let owner = proposal.owner();
1556        let BlockProposal {
1557            content,
1558            original_proposal,
1559            signature: _,
1560        } = &proposal;
1561        let block = &content.block;
1562        let chain = &self.chain;
1563        // Check if the chain is ready for this new block proposal.
1564        chain.tip_state.get().verify_block_chaining(block)?;
1565        // Check the epoch.
1566        let (epoch, committee) = chain.current_committee()?;
1567        check_block_epoch(epoch, block.chain_id, block.epoch)?;
1568        let policy = committee.policy().clone();
1569        block.check_proposal_size(policy.maximum_block_proposal_size)?;
1570        // Check the authentication of the block.
1571        ensure!(
1572            chain.manager.can_propose(&owner, proposal.content.round),
1573            WorkerError::InvalidOwner
1574        );
1575        let old_round = self.chain.manager.current_round();
1576        match original_proposal {
1577            None => {
1578                if let Some(signer) = block.authenticated_owner {
1579                    // Check the authentication of the operations in the new block.
1580                    ensure!(signer == owner, WorkerError::InvalidSigner(owner));
1581                }
1582            }
1583            Some(OriginalProposal::Regular { certificate }) => {
1584                // Verify that this block has been validated by a quorum before.
1585                certificate.check(committee)?;
1586            }
1587            Some(OriginalProposal::Fast(signature)) => {
1588                let original_proposal = BlockProposal {
1589                    content: ProposalContent {
1590                        block: content.block.clone(),
1591                        round: Round::Fast,
1592                        outcome: None,
1593                    },
1594                    signature: *signature,
1595                    original_proposal: None,
1596                };
1597                let super_owner = original_proposal.owner();
1598                ensure!(
1599                    chain
1600                        .manager
1601                        .ownership
1602                        .get()
1603                        .super_owners
1604                        .contains(&super_owner),
1605                    WorkerError::InvalidOwner
1606                );
1607                if let Some(signer) = block.authenticated_owner {
1608                    // Check the authentication of the operations in the new block.
1609                    ensure!(signer == super_owner, WorkerError::InvalidSigner(signer));
1610                }
1611                original_proposal.check_signature()?;
1612            }
1613        }
1614        if chain.manager.check_proposed_block(&proposal)? == manager::Outcome::Skip {
1615            // We already voted for this block.
1616            return Ok((self.chain_info_response(), NetworkActions::default()));
1617        }
1618        let local_time = self.storage.clock().current_time();
1619
1620        // Make sure we remember that a proposal was signed, to determine the correct round to
1621        // propose in.
1622        if self
1623            .chain
1624            .manager
1625            .update_signed_proposal(&proposal, local_time)
1626        {
1627            self.save().await?;
1628        }
1629
1630        let published_blobs = self.load_proposal_blobs(&proposal).await?;
1631        let ProposalContent {
1632            block,
1633            round,
1634            outcome,
1635        } = content;
1636
1637        if self.config.key_pair().is_some()
1638            && block.timestamp.duration_since(local_time) > self.config.block_time_grace_period
1639        {
1640            return Err(WorkerError::InvalidTimestamp {
1641                local_time,
1642                block_timestamp: block.timestamp,
1643                block_time_grace_period: self.config.block_time_grace_period,
1644            });
1645        }
1646        // Note: The actor delays processing proposals with future timestamps (within the grace
1647        // period) so that other requests can be handled in the meantime. By the time we reach
1648        // here, the block timestamp should be in the past or very close to the current time.
1649
1650        self.chain
1651            .remove_bundles_from_inboxes(block.timestamp, true, block.incoming_bundles())
1652            .await?;
1653        let block = if let Some(outcome) = outcome {
1654            outcome.clone().with(proposal.content.block.clone())
1655        } else {
1656            let (executed_block, _resource_tracker) = Box::pin(self.execute_block(
1657                block.clone(),
1658                local_time,
1659                round.multi_leader(),
1660                &published_blobs,
1661                BundleExecutionPolicy::Abort,
1662            ))
1663            .await?;
1664            executed_block
1665        };
1666
1667        ensure!(
1668            !round.is_fast() || !block.has_oracle_responses(),
1669            WorkerError::FastBlockUsingOracles
1670        );
1671        let chain = &mut self.chain;
1672        // Check if the counters of tip_state would be valid.
1673        chain
1674            .tip_state
1675            .get_mut()
1676            .update_counters(&block.body.transactions, &block.body.messages)?;
1677        // Don't save the changes since the block is not confirmed yet.
1678        chain.rollback();
1679
1680        // Create the vote and store it in the chain state.
1681        let created_blobs: BTreeMap<_, _> = block.iter_created_blobs().collect();
1682        let blobs = self
1683            .get_required_blobs(proposal.expected_blob_ids(), &created_blobs)
1684            .await?;
1685        let key_pair = self.config.key_pair();
1686        let manager = &mut self.chain.manager;
1687        match manager.create_vote(proposal, block, key_pair, local_time, blobs)? {
1688            // Cache the value we voted on, so the client doesn't have to send it again.
1689            Some(Either::Left(vote)) => {
1690                self.block_values.insert(Cow::Borrowed(vote.value.inner()));
1691            }
1692            Some(Either::Right(vote)) => {
1693                self.block_values.insert(Cow::Borrowed(vote.value.inner()));
1694            }
1695            None => (),
1696        }
1697        self.save().await?;
1698        let actions = self.create_network_actions(Some(old_round)).await?;
1699        Ok((self.chain_info_response(), actions))
1700    }

1864        .collect()
1865}

1884/// Helper type for handling cross-chain updates.
1885pub(crate) struct CrossChainUpdateHelper<'a> {
1886    pub(crate) allow_messages_from_deprecated_epochs: bool,
1887    pub(crate) current_epoch: Epoch,
1888    pub(crate) committees: &'a BTreeMap<Epoch, Committee>,
1889}
1890
1891impl<'a> CrossChainUpdateHelper<'a> {
1892    /// Creates a new [`CrossChainUpdateHelper`].
1893    fn new<C>(config: &ChainWorkerConfig, chain: &'a ChainStateView<C>) -> Self
1894    where
1895        C: Context + Clone + 'static,
1896    {
1897        CrossChainUpdateHelper {
1898            allow_messages_from_deprecated_epochs: config.allow_messages_from_deprecated_epochs,
1899            current_epoch: *chain.execution_state.system.epoch.get(),
1900            committees: chain.execution_state.system.committees.get(),
1901        }
1902    }
1903
1904    /// Checks basic invariants and deals with repeated heights and deprecated epochs.
1905    /// * Returns a range of message bundles that are both new to us and not relying on
1906    ///   an untrusted set of validators.
1907    /// * In the case of validators, if the epoch(s) of the highest bundles are not
1908    ///   trusted, we only accept bundles that contain messages that were already
1909    ///   executed by anticipation (i.e. received in certified blocks).
1910    /// * Basic invariants are checked for good measure. We still crucially trust
1911    ///   the worker of the sending chain to have verified and executed the blocks
1912    ///   correctly.
1913    pub(crate) fn select_message_bundles(
1914        &self,
1915        origin: &'a ChainId,
1916        recipient: ChainId,
1917        next_height_to_receive: BlockHeight,
1918        last_anticipated_block_height: Option<BlockHeight>,
1919        mut bundles: Vec<(Epoch, MessageBundle)>,
1920    ) -> Result<Vec<MessageBundle>, WorkerError> {
1921        let mut latest_height = None;
1922        let mut skipped_len = 0;
1923        let mut trusted_len = 0;
1924        for (i, (epoch, bundle)) in bundles.iter().enumerate() {
1925            // Make sure that heights are not decreasing.
1926            ensure!(
1927                latest_height <= Some(bundle.height),
1928                WorkerError::InvalidCrossChainRequest
1929            );
1930            latest_height = Some(bundle.height);
1931            // Check if the block has been received already.
1932            if bundle.height < next_height_to_receive {
1933                skipped_len = i + 1;
1934            }
1935            // Check if the height is trusted or the epoch is trusted.
1936            if self.allow_messages_from_deprecated_epochs
1937                || Some(bundle.height) <= last_anticipated_block_height
1938                || *epoch >= self.current_epoch
1939                || self.committees.contains_key(epoch)
1940            {
1941                trusted_len = i + 1;
1942            }
1943        }
1944        if skipped_len > 0 {
1945            let (_, sample_bundle) = &bundles[skipped_len - 1];
1946            debug!(
1947                "Ignoring repeated messages to {recipient:.8} from {origin:} at height {}",
1948                sample_bundle.height,
1949            );
1950        }
1951        if skipped_len < bundles.len() && trusted_len < bundles.len() {
1952            let (sample_epoch, sample_bundle) = &bundles[trusted_len];
1953            warn!(
1954                "Refusing messages to {recipient:.8} from {origin:} at height {} \
1955                 because the epoch {} is not trusted any more",
1956                sample_bundle.height, sample_epoch,
1957            );
1958        }
1959        let bundles = if skipped_len < trusted_len {
1960            bundles
1961                .drain(skipped_len..trusted_len)
1962                .map(|(_, bundle)| bundle)
1963                .collect()
1964        } else {
1965            vec![]
1966        };
1967        Ok(bundles)
1968    }
1969}
