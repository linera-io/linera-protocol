4//! Helpers for managing chain worker state behind an `Arc<RwLock<…>>`.
5//!
6//! Tokio's [`RwLock`] is write-preferring: once a writer is waiting, new readers
7//! queue behind it. This prevents read-only queries from starving write operations
8//! (block proposals, certificate handling, etc.).
9
10use std::{
11    ops::{Deref, DerefMut},
12    sync::{
13        atomic::{AtomicU64, Ordering},
14        Arc,
15    },
16};
17
18/// An atomically-updatable timestamp backed by microseconds since the Unix epoch.
19///
20/// This wraps the raw `AtomicU64` microsecond encoding so that call sites work
21/// exclusively with [`Duration`] and never see the underlying representation.
22pub(crate) struct AtomicTimestamp(AtomicU64);
23
24impl AtomicTimestamp {
25    /// Creates a new `AtomicTimestamp` set to the current time.
26    pub(crate) fn now() -> Self {
27        Self(AtomicU64::new(Self::current_micros()))
28    }
29
30    /// Updates the stored timestamp to the current time.
31    pub(crate) fn store_now(&self) {
32        self.0.store(Self::current_micros(), Ordering::Relaxed);
33    }
34
35    /// Returns how long has passed since the stored timestamp.
36    pub(crate) fn elapsed(&self) -> Duration {
37        let last = self.0.load(Ordering::Relaxed);
38        let now = Self::current_micros();
39        Duration::from_micros(now.saturating_sub(last))
40    }
41
42    fn current_micros() -> u64 {
43        linera_base::time::SystemTime::now()
44            .duration_since(linera_base::time::UNIX_EPOCH)
45            .map(|d| d.as_micros() as u64)
46            .unwrap_or(0)
47    }
48}

62/// A write guard that automatically rolls back uncommitted chain state changes on drop.
63///
64/// This ensures cancellation safety: if a write operation's future is dropped before
65/// completion, any uncommitted state changes are rolled back rather than leaked.
66pub(crate) struct RollbackGuard<S: Storage + Clone + 'static>(
67    tokio::sync::OwnedRwLockWriteGuard<ChainWorkerState<S>>,
68);
69
70impl<S: Storage + Clone + 'static> Deref for RollbackGuard<S> {
71    type Target = ChainWorkerState<S>;
72
73    fn deref(&self) -> &Self::Target {
74        &self.0
75    }
76}
77
78impl<S: Storage + Clone + 'static> DerefMut for RollbackGuard<S> {
79    fn deref_mut(&mut self) -> &mut Self::Target {
80        &mut self.0
81    }
82}
83
84impl<S: Storage + Clone + 'static> Drop for RollbackGuard<S> {
85    fn drop(&mut self) {
86        self.0.rollback();
87    }
88}

90/// The endpoint and background task for a long-lived service runtime.
91pub(crate) struct ServiceRuntimeActor {
92    pub(crate) task: web_thread_pool::Task<()>,
93    pub(crate) endpoint: ServiceRuntimeEndpoint,
94}
95
96impl ServiceRuntimeActor {
97    /// Spawns a blocking task to execute the service runtime actor.
98    pub(crate) async fn spawn(
99        chain_id: ChainId,
100        thread_pool: &linera_execution::ThreadPool,
101    ) -> Self {
102        let (execution_state_sender, incoming_execution_requests) =
103            futures::channel::mpsc::unbounded();
104        let (runtime_request_sender, runtime_request_receiver) = std::sync::mpsc::channel();
105
106        Self {
107            endpoint: ServiceRuntimeEndpoint {
108                incoming_execution_requests,
109                runtime_request_sender,
110            },
111            task: thread_pool
112                .run((), move |()| async move {
113                    // The dummy context is overwritten by `prepare_for_query`
114                    // before the first actual query is executed.
115                    ServiceSyncRuntime::new(
116                        execution_state_sender,
117                        QueryContext {
118                            chain_id,
119                            next_block_height: BlockHeight(0),
120                            local_time: Timestamp::from(0),
121                        },
122                    )
123                    .run(&runtime_request_receiver)
124                })
125                .await,
126        }
127    }
128}

130/// Wraps a [`ChainWorkerState`] in an `Arc<RwLock<…>>` and spawns a keep-alive task
131/// if a TTL is configured.
132pub(crate) fn create_chain_worker<S: Storage + Clone + 'static>(
133    state: ChainWorkerState<S>,
134    is_tracked: bool,
135    config: &ChainWorkerConfig,
136) -> Arc<RwLock<ChainWorkerState<S>>> {
137    let last_access = state.last_access_arc();
138    let chain_id = state.chain().chain_id();
139    let arc = Arc::new(RwLock::new(state));
140    let ttl = if is_tracked {
141        config.ttl
142    } else {
143        config.sender_chain_ttl
144    };
145    if let Some(ttl) = ttl {
146        spawn_keep_alive(chain_id, Arc::clone(&arc), last_access, ttl);
147    }
148    arc
149}

151/// Acquires a read lock, updating the last-access timestamp.
152pub(crate) async fn read_lock<S: Storage + Clone + 'static>(
153    state: &Arc<RwLock<ChainWorkerState<S>>>,
154) -> Result<OwnedRwLockReadGuard<ChainWorkerState<S>>, WorkerError> {
155    let guard = state.clone().read_owned().await;
156    guard.check_not_poisoned()?;
157    guard.touch();
158    Ok(guard)
159}
160
161/// Acquires a read lock, initializing the chain if needed.
162///
163/// First acquires a read lock and checks if the chain is already known to be active.
164/// If not, drops the read lock, runs the initializing write phase on a detached task
165/// (so caller cancellation cannot tear the save halfway), then re-acquires the read
166/// lock.
167pub(crate) async fn read_lock_initialized<S: Storage + Clone + 'static>(
168    state: &Arc<RwLock<ChainWorkerState<S>>>,
169) -> Result<OwnedRwLockReadGuard<ChainWorkerState<S>>, crate::worker::WorkerError> {
170    {
171        let guard = read_lock(state).await?;
172        if guard.knows_chain_is_active() {
173            return Ok(guard);
174        }
175    }
176    let state_for_task = state.clone();
177    linera_base::task::run_detached(async move {
178        let mut guard = write_lock(&state_for_task).await?;
179        guard.initialize_and_save_if_needed().await
180    })
181    .await?;
182    read_lock(state).await
183}

185/// Acquires a write lock, updating the last-access timestamp.
186///
187/// Returns a [`RollbackGuard`] that automatically rolls back uncommitted changes
188/// when dropped, ensuring cancellation safety.
189pub(crate) async fn write_lock<S: Storage + Clone + 'static>(
190    state: &Arc<RwLock<ChainWorkerState<S>>>,
191) -> Result<RollbackGuard<S>, WorkerError> {
192    let guard = RollbackGuard(state.clone().write_owned().await);
193    guard.check_not_poisoned()?;
194    guard.touch();
195    Ok(guard)
196}

198/// Spawns a background task that keeps the chain state alive for at least `ttl`
199/// after the last access. When the state has been idle for the full TTL, the task
200/// drops the state if it holds the only strong reference.
201fn spawn_keep_alive<S: Storage + Clone + 'static>(
202    chain_id: ChainId,
203    mut state: Arc<RwLock<ChainWorkerState<S>>>,
204    last_access: Arc<AtomicTimestamp>,
205    ttl: Duration,
206) {
207    linera_base::Task::spawn(async move {
208        loop {
209            while let Some(remaining) = ttl
210                .checked_sub(last_access.elapsed())
211                .filter(|remaining| *remaining > Duration::ZERO)
212            {
213                // Touched recently — sleep for the remaining time.
214                linera_base::time::timer::sleep(remaining).await;
215            }
216            // Idle long enough. Drop our strong reference if it's the only one.
217            match Arc::try_unwrap(state) {
218                Ok(rw_lock) => {
219                    tracing::debug!(%chain_id, "Dropping chain worker");
220                    // We have sole ownership — extract the state and
221                    // shut down the service runtime gracefully.
222                    let mut worker_state = RwLock::into_inner(rw_lock);
223                    let task = worker_state.clear_service_runtime();
224                    drop(worker_state);
225                    if let Some(task) = task {
226                        if let Err(err) = task.await {
227                            tracing::warn!(%err, "Failed to shut down service runtime");
228                        }
229                    }
230                    break;
231                }
232                Err(arc) => {
233                    arc.read().await.touch();
234                    state = arc;
235                }
236            }
237        }
238    })
239    .forget();
