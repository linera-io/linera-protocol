133/// Policies for automatically handling incoming messages.
134#[derive(Clone, Debug)]
135pub struct MessagePolicy {
136    /// The blanket policy applied to all messages.
137    blanket: BlanketMessagePolicy,
138    /// A collection of chains which restrict the origin of messages to be
139    /// accepted. `Option::None` means that messages from all chains are accepted. An empty
140    /// `HashSet` denotes that messages from no chains are accepted.
141    restrict_chain_ids_to: Option<HashSet<ChainId>>,
142}
143
144#[derive(Copy, Clone, Debug, clap::ValueEnum)]
145pub enum BlanketMessagePolicy {
146    /// Automatically accept all incoming messages. Reject them only if execution fails.
147    Accept,
148    /// Automatically reject tracked messages, ignore or skip untracked messages, but accept
149    /// protected ones.
150    Reject,
151    /// Don't include any messages in blocks, and don't make any decision whether to accept or
152    /// reject.
153    Ignore,
154}
155
156impl MessagePolicy {
157    pub fn new(
158        blanket: BlanketMessagePolicy,
159        restrict_chain_ids_to: Option<HashSet<ChainId>>,
160    ) -> Self {
161        Self {
162            blanket,
163            restrict_chain_ids_to,
164        }
165    }
166
167    #[cfg(with_testing)]
168    pub fn new_accept_all() -> Self {
169        Self {
170            blanket: BlanketMessagePolicy::Accept,
171            restrict_chain_ids_to: None,
172        }
173    }
174
175    #[instrument(level = "trace", skip(self))]
176    fn must_handle(&self, bundle: &mut IncomingBundle) -> bool {
177        if self.is_reject() {
178            if bundle.bundle.is_skippable() {
179                return false;
180            } else if !bundle.bundle.is_protected() {
181                bundle.action = MessageAction::Reject;
182            }
183        }
184        match &self.restrict_chain_ids_to {
185            None => true,
186            Some(chains) => chains.contains(&bundle.origin),
187        }
188    }
189
190    #[instrument(level = "trace", skip(self))]
191    fn is_ignore(&self) -> bool {
192        matches!(self.blanket, BlanketMessagePolicy::Ignore)
193    }
194
195    #[instrument(level = "trace", skip(self))]
196    fn is_reject(&self) -> bool {
197        matches!(self.blanket, BlanketMessagePolicy::Reject)
198    }
199}

209/// Defines how we listen to a chain:
210/// - do we care about every block notification?
211/// - or do we only care about blocks containing events from some particular streams?
212#[derive(Debug, Clone, PartialEq, Eq)]
213pub enum ListeningMode {
214    /// Listen to everything.
215    FullChain,
216    /// Only listen to blocks which contain events from those streams.
217    EventsOnly(BTreeSet<StreamId>),
218}
219
220impl PartialOrd for ListeningMode {
221    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
222        match (self, other) {
223            (ListeningMode::FullChain, ListeningMode::FullChain) => Some(Ordering::Equal),
224            (ListeningMode::FullChain, _) => Some(Ordering::Greater),
225            (_, ListeningMode::FullChain) => Some(Ordering::Less),
226            (ListeningMode::EventsOnly(events_a), ListeningMode::EventsOnly(events_b)) => {
227                if events_a.is_superset(events_b) {
228                    Some(Ordering::Greater)
229                } else if events_b.is_superset(events_a) {
230                    Some(Ordering::Less)
231                } else {
232                    None
233                }
234            }
235        }
236    }
237}
238
239impl ListeningMode {
240    pub fn extend(&mut self, other: Option<ListeningMode>) {
241        match (self, other) {
242            (_, None) => (),
243            (ListeningMode::FullChain, _) => (),
244            (mode, Some(ListeningMode::FullChain)) => {
245                *mode = ListeningMode::FullChain;
246            }
247            (
248                ListeningMode::EventsOnly(self_events),
249                Some(ListeningMode::EventsOnly(other_events)),
250            ) => {
251                self_events.extend(other_events);
252            }
253        }
254    }
255}

375    /// Fetches the chain description blob if needed, and returns the chain info.
376    async fn fetch_chain_info(
377        &self,
378        chain_id: ChainId,
379        validators: &[RemoteNode<Env::ValidatorNode>],
380    ) -> Result<Box<ChainInfo>, chain_client::Error> {
381        match self.local_node.chain_info(chain_id).await {
382            Ok(info) => Ok(info),
383            Err(LocalNodeError::BlobsNotFound(blob_ids)) => {
384                // Make sure the admin chain is up to date.
385                self.synchronize_chain_state(self.admin_id).await?;
386                // If the chain is missing then the error is a WorkerError
387                // and so a BlobsNotFound
388                self.update_local_node_with_blobs_from(blob_ids, validators)
389                    .await?;
390                Ok(self.local_node.chain_info(chain_id).await?)
391            }
392            Err(err) => Err(err.into()),
393        }
394    }

396    fn weighted_select(
397        remaining_validators: &mut Vec<RemoteNode<Env::ValidatorNode>>,
398        remaining_weights: &mut Vec<u64>,
399    ) -> Option<RemoteNode<Env::ValidatorNode>> {
400        if remaining_weights.is_empty() {
401            return None;
402        }
403        let dist = WeightedIndex::new(remaining_weights.clone()).unwrap();
404        let idx = dist.sample(&mut rand::thread_rng());
405        remaining_weights.remove(idx);
406        Some(remaining_validators.remove(idx))
407    }

409    /// Downloads and processes all certificates up to (excluding) the specified height.
410    #[instrument(level = "trace", skip(self))]
411    async fn download_certificates(
412        &self,
413        chain_id: ChainId,
414        target_next_block_height: BlockHeight,
415    ) -> Result<Box<ChainInfo>, chain_client::Error> {
416        let (_, committee) = self.admin_committee().await?;
417        let mut remaining_validators = self.make_nodes(&committee)?;
418        let mut info = self
419            .fetch_chain_info(chain_id, &remaining_validators)
420            .await?;
421        // Determining the weights of the validators
422        let mut remaining_weights = remaining_validators
423            .iter()
424            .map(|validator| {
425                let validator_state = committee.validators.get(&validator.public_key).unwrap();
426                validator_state.votes
427            })
428            .collect::<Vec<_>>();
429
430        while let Some(remote_node) =
431            Self::weighted_select(&mut remaining_validators, &mut remaining_weights)
432        {
433            if target_next_block_height <= info.next_block_height {
434                return Ok(info);
435            }
436            match self
437                .download_certificates_from(&remote_node, chain_id, target_next_block_height)
438                .await
439            {
440                Err(err) => info!(
441                    "Failed to download certificates from validator {:?}: {err}",
442                    remote_node.public_key
443                ),
444                Ok(Some(new_info)) => info = new_info,
445                Ok(None) => {}
446            }
447        }
448        ensure!(
449            target_next_block_height <= info.next_block_height,
450            chain_client::Error::CannotDownloadCertificates {
451                chain_id,
452                target_next_block_height,
453            }
454        );
455        Ok(info)
456    }

458    /// Downloads and processes all certificates up to (excluding) the specified height from the
459    /// given validator.
460    #[instrument(level = "trace", skip_all)]
461    async fn download_certificates_from(
462        &self,
463        remote_node: &RemoteNode<Env::ValidatorNode>,
464        chain_id: ChainId,
465        stop: BlockHeight,
466    ) -> Result<Option<Box<ChainInfo>>, chain_client::Error> {
467        let mut last_info = None;
468        // First load any blocks from local storage, if available.
469        let chain_info = self.local_node.chain_info(chain_id).await?;
470        let mut next_height = chain_info.next_block_height;
471        let hashes = self
472            .local_node
473            .get_preprocessed_block_hashes(chain_id, next_height, stop)
474            .await?;
475        let certificates = self
476            .storage_client()
477            .read_certificates(hashes.clone())
478            .await?;
479        let certificates = match ResultReadCertificates::new(certificates, hashes) {
480            ResultReadCertificates::Certificates(certificates) => certificates,
481            ResultReadCertificates::InvalidHashes(hashes) => {
482                return Err(chain_client::Error::ReadCertificatesError(hashes))
483            }
484        };
485        for certificate in certificates {
486            last_info = Some(self.handle_certificate(certificate).await?.info);
487        }
488        // Now download the rest in batches from the remote node.
489        while next_height < stop {
490            // TODO(#2045): Analyze network errors instead of using a fixed batch size.
491            let limit = u64::from(stop)
492                .checked_sub(u64::from(next_height))
493                .ok_or(ArithmeticError::Overflow)?
494                .min(self.options.certificate_download_batch_size);
495
496            let certificates = self
497                .requests_scheduler
498                .download_certificates(remote_node, chain_id, next_height, limit)
499                .await?;
500            let Some(info) = self.process_certificates(remote_node, certificates).await? else {
501                break;
502            };
503            assert!(info.next_block_height > next_height);
504            next_height = info.next_block_height;
505            last_info = Some(info);
506        }
507        Ok(last_info)
508    }

510    async fn download_blobs(
511        &self,
512        remote_nodes: &[RemoteNode<Env::ValidatorNode>],
513        blob_ids: &[BlobId],
514    ) -> Result<(), chain_client::Error> {
515        let blobs = &self
516            .requests_scheduler

518            .await?
519            .ok_or_else(|| {
520                chain_client::Error::RemoteNodeError(NodeError::BlobsNotFound(blob_ids.to_vec()))
521            })?;
522        self.local_node.store_blobs(blobs).await.map_err(Into::into)
523    }

525    /// Tries to process all the certificates, requesting any missing blobs from the given node.
526    /// Returns the chain info of the last successfully processed certificate.
527    #[instrument(level = "trace", skip_all)]
528    async fn process_certificates(
529        &self,
530        remote_node: &RemoteNode<Env::ValidatorNode>,
531        certificates: Vec<ConfirmedBlockCertificate>,
532    ) -> Result<Option<Box<ChainInfo>>, chain_client::Error> {
533        let mut info = None;
534        let required_blob_ids: Vec<_> = certificates
535            .iter()
536            .flat_map(|certificate| certificate.value().required_blob_ids())
537            .collect();
538
539        match self
540            .local_node
541            .read_blob_states_from_storage(&required_blob_ids)
542            .await
543        {
544            Err(LocalNodeError::BlobsNotFound(blob_ids)) => {
545                self.download_blobs(&[remote_node.clone()], &blob_ids)
546                    .await?;
547            }
548            x => {
549                x?;
550            }
551        }
552
553        for certificate in certificates {
554            info = Some(
555                match self.handle_certificate(certificate.clone()).await {
556                    Err(LocalNodeError::BlobsNotFound(blob_ids)) => {
557                        self.download_blobs(&[remote_node.clone()], &blob_ids)
558                            .await?;
559                        self.handle_certificate(certificate).await?
560                    }
561                    x => x?,
562                }
563                .info,
564            );
565        }
566
567        // Done with all certificates.
568        Ok(info)
569    }

625    /// Ensures that the client has the `ChainDescription` blob corresponding to this
626    /// client's `ChainId`.
627    pub async fn get_chain_description(
628        &self,
629        chain_id: ChainId,
630    ) -> Result<ChainDescription, chain_client::Error> {
631        let chain_desc_id = BlobId::new(chain_id.0, BlobType::ChainDescription);
632        let blob = self
633            .local_node
634            .storage_client()
635            .read_blob(chain_desc_id)
636            .await?;
637        if let Some(blob) = blob {
638            // We have the blob - return it.
639            return Ok(bcs::from_bytes(blob.bytes())?);
640        };
641        // Recover history from the current validators, according to the admin chain.
642        self.synchronize_chain_state(self.admin_id).await?;
643        let nodes = self.validator_nodes().await?;
644        let blob = self
645            .update_local_node_with_blobs_from(vec![chain_desc_id], &nodes)
646            .await?
647            .pop()
648            .unwrap(); // Returns exactly as many blobs as passed-in IDs.
649        Ok(bcs::from_bytes(blob.bytes())?)
650    }

652    /// Updates the latest block and next block height and round information from the chain info.
653    #[instrument(level = "trace", skip_all, fields(chain_id = format!("{:.8}", info.chain_id)))]
654    fn update_from_info(&self, info: &ChainInfo) {
655        self.chains.pin().update(info.chain_id, |state| {
656            let mut state = state.clone_for_update_unchecked();

750    /// Broadcasts certified blocks and optionally a block proposal, certificate or
751    /// leader timeout request.
752    ///
753    /// In that case, it verifies that the validator votes are for the provided value,
754    /// and returns a certificate.
755    #[instrument(level = "trace", skip_all)]
756    async fn communicate_chain_action<T: CertificateValue>(
757        &self,
758        committee: &Committee,
759        action: CommunicateAction,
760        value: T,
761    ) -> Result<GenericCertificate<T>, chain_client::Error> {
762        let nodes = self.make_nodes(committee)?;
763        let ((votes_hash, votes_round), votes) = communicate_with_quorum(
764            &nodes,
765            committee,
766            |vote: &LiteVote| (vote.value.value_hash, vote.round),
767            |remote_node| {
768                let mut updater = ValidatorUpdater {
769                    remote_node,
770                    local_node: self.local_node.clone(),
771                    admin_id: self.admin_id,
772                };
773                let action = action.clone();
774                Box::pin(async move { updater.send_chain_update(action).await })
775            },
776            self.options.grace_period,
777        )
778        .await?;
779        ensure!(
780            (votes_hash, votes_round) == (value.hash(), action.round()),
781            chain_client::Error::UnexpectedQuorum {
782                hash: votes_hash,
783                round: votes_round,
784                expected_hash: value.hash(),
785                expected_round: action.round(),
786            }
787        );
788        // Certificate is valid because
789        // * `communicate_with_quorum` ensured a sufficient "weight" of
790        // (non-error) answers were returned by validators.
791        // * each answer is a vote signed by the expected validator.
792        let certificate = LiteCertificate::try_from_votes(votes)
793            .ok_or_else(|| {
794                chain_client::Error::InternalError(
795                    "Vote values or rounds don't match; this is a bug",
796                )
797            })?
798            .with_value(value)
799            .ok_or_else(|| {
800                chain_client::Error::ProtocolError("A quorum voted for an unexpected value")
801            })?;
802        Ok(certificate)
803    }

805    /// Processes the confirmed block certificate in the local node without checking signatures.
806    /// Also downloads and processes all ancestors that are still missing.
807    #[instrument(level = "trace", skip_all)]
808    async fn receive_certificate_with_checked_signatures(
809        &self,
810        certificate: ConfirmedBlockCertificate,
811    ) -> Result<(), chain_client::Error> {
812        let certificate = Box::new(certificate);
813        let block = certificate.block();
814        // Recover history from the network.
815        self.download_certificates(block.header.chain_id, block.header.height)
816            .await?;
817        // Process the received operations. Download required hashed certificate values if
818        // necessary.
819        if let Err(err) = self.process_certificate(certificate.clone()).await {
820            match &err {
821                LocalNodeError::BlobsNotFound(blob_ids) => {
822                    self.download_blobs(&self.validator_nodes().await?, blob_ids)
823                        .await
824                        .map_err(|_| err)?;
825                    self.process_certificate(certificate).await?;
826                }
827                _ => {
828                    // The certificate is not as expected. Give up.
829                    warn!("Failed to process network hashed certificate value");
830                    return Err(err.into());
831                }
832            }
833        }
834
835        Ok(())
836    }

838    /// Processes the confirmed block in the local node, possibly without executing it.
839    #[instrument(level = "trace", skip_all)]
840    #[allow(dead_code)] // Otherwise CI fails when built for docker.
841    async fn receive_sender_certificate(
842        &self,
843        certificate: ConfirmedBlockCertificate,
844        mode: ReceiveCertificateMode,
845        nodes: Option<Vec<RemoteNode<Env::ValidatorNode>>>,
846    ) -> Result<(), chain_client::Error> {
847        // Verify the certificate before doing any expensive networking.
848        let (max_epoch, committees) = self.admin_committees().await?;
849        if let ReceiveCertificateMode::NeedsCheck = mode {
850            Self::check_certificate(max_epoch, &committees, &certificate)?.into_result()?;
851        }
852        // Recover history from the network.
853        let nodes = if let Some(nodes) = nodes {
854            nodes
855        } else {
856            self.validator_nodes().await?
857        };
858        if let Err(err) = self.handle_certificate(certificate.clone()).await {
859            match &err {
860                LocalNodeError::BlobsNotFound(blob_ids) => {
861                    self.download_blobs(&nodes, blob_ids).await?;
862                    self.handle_certificate(certificate.clone()).await?;
863                }
864                _ => {
865                    // The certificate is not as expected. Give up.
866                    warn!("Failed to process network hashed certificate value");
867                    return Err(err.into());
868                }
869            }
870        }
871
872        Ok(())
873    }

875    /// Downloads and processes certificates for sender chain blocks.
876    #[instrument(level = "trace", skip_all)]
877    async fn download_and_process_sender_chain(
878        &self,
879        sender_chain_id: ChainId,
880        nodes: &[RemoteNode<Env::ValidatorNode>],
881        received_log: &ReceivedLogs,
882        mut remote_heights: Vec<BlockHeight>,
883        sender: mpsc::UnboundedSender<ChainAndHeight>,
884    ) {
885        let (max_epoch, committees) = match self.admin_committees().await {
886            Ok(result) => result,
887            Err(error) => {
888                error!(%error, %sender_chain_id, "could not read admin committees");
889                return;
890            }
891        };
892        let committees_ref = &committees;
893        let mut nodes = nodes.to_vec();
894        while !remote_heights.is_empty() {
895            let remote_heights_ref = &remote_heights;
896            nodes.shuffle(&mut rand::thread_rng());
897            let certificates = match communicate_concurrently(
898                &nodes,
899                async move |remote_node| {
900                    let mut remote_heights = remote_heights_ref.clone();
901                    // No need trying to download certificates the validator didn't have in their
902                    // log - we'll retry downloading the remaining ones next time we loop.
903                    remote_heights.retain(|height| {
904                        received_log.validator_has_block(
905                            &remote_node.public_key,
906                            sender_chain_id,
907                            *height,
908                        )
909                    });
910                    if remote_heights.is_empty() {
911                        // It makes no sense to return `Ok(_)` if we aren't going to try downloading
912                        // anything from the validator - let the function try the other validators
913                        return Err(());
914                    }
915                    let certificates = self
916                        .requests_scheduler
917                        .download_certificates_by_heights(
918                            &remote_node,
919                            sender_chain_id,
920                            remote_heights,
921                        )
922                        .await
923                        .map_err(|_| ())?;
924                    let mut certificates_with_check_results = vec![];
925                    for cert in certificates {
926                        if let Ok(check_result) =
927                            Self::check_certificate(max_epoch, committees_ref, &cert)
928                        {
929                            certificates_with_check_results
930                                .push((cert, check_result.into_result().is_ok()));
931                        } else {
932                            // Invalid signature - the validator is faulty
933                            return Err(());
934                        }
935                    }
936                    Ok(certificates_with_check_results)
937                },
938                |errors| {
939                    errors
940                        .into_iter()
941                        .map(|(validator, _error)| validator)
942                        .collect::<BTreeSet<_>>()
943                },
944                self.options.certificate_batch_download_timeout,
945            )
946            .await
947            {
948                Ok(certificates_with_check_results) => certificates_with_check_results,
949                Err(faulty_validators) => {
950                    // filter out faulty validators and retry if any are left
951                    nodes.retain(|node| !faulty_validators.contains(&node.public_key));
952                    if nodes.is_empty() {
953                        info!(
954                            chain_id = %sender_chain_id,
955                            "could not download certificates for chain - no more correct validators left"
956                        );
957                        return;
958                    }
959                    continue;
960                }
961            };
962
963            trace!(
964                chain_id = %sender_chain_id,
965                num_certificates = %certificates.len(),
966                "received certificates",
967            );
968
969            let mut to_remove_from_queue = BTreeSet::new();
970
971            for (certificate, check_result) in certificates {
972                let hash = certificate.hash();
973                let chain_id = certificate.block().header.chain_id;
974                let height = certificate.block().header.height;
975                if !check_result {
976                    // The certificate was correctly signed, but we were missing a committee to
977                    // validate it properly - do not receive it, but also do not attempt to
978                    // re-download it.
979                    to_remove_from_queue.insert(height);
980                    continue;
981                }
982                // We checked the certificates right after downloading them.
983                let mode = ReceiveCertificateMode::AlreadyChecked;
984                if let Err(error) = self
985                    .receive_sender_certificate(certificate, mode, None)
986                    .await
987                {
988                    warn!(%error, %hash, "Received invalid certificate");
989                } else {
990                    to_remove_from_queue.insert(height);
991                    if let Err(error) = sender.send(ChainAndHeight { chain_id, height }) {
992                        error!(
993                            %chain_id,
994                            %height,
995                            %error,
996                            "failed to send chain and height over the channel",
997                        );
998                    }
999                }
1000            }
1001
1002            remote_heights.retain(|height| !to_remove_from_queue.contains(height));
1003        }
1004        trace!(
1005            chain_id = %sender_chain_id,
1006            "find_received_certificates: finished processing chain",
1007        );
1008    }

1048    /// Downloads a specific sender block and recursively downloads any earlier blocks
1049    /// that also sent a message to our chain, based on `previous_message_blocks`.
1050    ///
1051    /// This ensures that we have all the sender blocks needed to preprocess the target block
1052    /// and put the messages to our chain into the outbox.
1053    async fn download_sender_block_with_sending_ancestors(
1054        &self,
1055        receiver_chain_id: ChainId,
1056        sender_chain_id: ChainId,
1057        height: BlockHeight,
1058        remote_node: &RemoteNode<Env::ValidatorNode>,
1059    ) -> Result<(), chain_client::Error> {
1060        let next_outbox_height = self
1061            .local_node
1062            .next_outbox_heights(&[sender_chain_id], receiver_chain_id)
1063            .await?
1064            .get(&sender_chain_id)
1065            .copied()
1066            .unwrap_or(BlockHeight::ZERO);
1067        let (max_epoch, committees) = self.admin_committees().await?;
1068
1069        // Recursively collect all certificates we need, following
1070        // the chain of previous_message_blocks back to next_outbox_height.
1071        let mut certificates = BTreeMap::new();
1072        let mut current_height = height;
1073
1074        // Stop if we've reached the height we've already processed.
1075        while current_height >= next_outbox_height {
1076            // Download the certificate for this height.
1077            let downloaded = self
1078                .requests_scheduler
1079                .download_certificates_by_heights(
1080                    remote_node,
1081                    sender_chain_id,
1082                    vec![current_height],
1083                )
1084                .await?;
1085            let Some(certificate) = downloaded.into_iter().next() else {
1086                return Err(chain_client::Error::CannotDownloadMissingSenderBlock {
1087                    chain_id: sender_chain_id,
1088                    height: current_height,
1089                });
1090            };
1091
1092            // Validate the certificate.
1093            Client::<Env>::check_certificate(max_epoch, &committees, &certificate)?
1094                .into_result()?;
1095
1096            // Check if there's a previous message block to our chain.
1097            let block = certificate.block();
1098            let next_height = block
1099                .body
1100                .previous_message_blocks
1101                .get(&receiver_chain_id)
1102                .map(|(_prev_hash, prev_height)| *prev_height);
1103
1104            // Store this certificate.
1105            certificates.insert(current_height, certificate);
1106
1107            if let Some(prev_height) = next_height {
1108                // Continue with the previous block.
1109                current_height = prev_height;
1110            } else {
1111                // No more dependencies.
1112                break;
1113            }
1114        }
1115
1116        if certificates.is_empty() {
1117            self.local_node
1118                .retry_pending_cross_chain_requests(sender_chain_id)
1119                .await?;
1120        }
1121
1122        // Process certificates in ascending block height order (BTreeMap keeps them sorted).
1123        for certificate in certificates.into_values() {
1124            self.receive_sender_certificate(
1125                certificate,
1126                ReceiveCertificateMode::AlreadyChecked,
1127                Some(vec![remote_node.clone()]),
1128            )
1129            .await?;
1130        }
1131
1132        Ok(())
1133    }

1135    #[instrument(
1136        level = "trace", skip_all,
1137        fields(certificate_hash = ?incoming_certificate.hash()),
1138    )]
1139    fn check_certificate(
1140        highest_known_epoch: Epoch,
1141        committees: &BTreeMap<Epoch, Committee>,
1142        incoming_certificate: &ConfirmedBlockCertificate,
1143    ) -> Result<CheckCertificateResult, NodeError> {
1144        let block = incoming_certificate.block();
1145        // Check that certificates are valid w.r.t one of our trusted committees.
1146        if block.header.epoch > highest_known_epoch {
1147            return Ok(CheckCertificateResult::FutureEpoch);
1148        }
1149        if let Some(known_committee) = committees.get(&block.header.epoch) {
1150            // This epoch is recognized by our chain. Let's verify the
1151            // certificate.
1152            incoming_certificate.check(known_committee)?;
1153            Ok(CheckCertificateResult::New)
1154        } else {
1155            // We don't accept a certificate from a committee that was retired.
1156            Ok(CheckCertificateResult::OldEpoch)
1157        }
1158    }

1202    /// Downloads any certificates from the specified validator that we are missing for the given
1203    /// chain, and processes them.
1204    #[instrument(level = "trace", skip(self, remote_node, chain_id))]
1205    async fn synchronize_chain_state_from(
1206        &self,
1207        remote_node: &RemoteNode<Env::ValidatorNode>,
1208        chain_id: ChainId,
1209    ) -> Result<(), chain_client::Error> {
1210        let mut local_info = self.local_node.chain_info(chain_id).await?;
1211        let query = ChainInfoQuery::new(chain_id).with_manager_values();
1212        let remote_info = remote_node.handle_chain_info_query(query).await?;
1213        if let Some(new_info) = self
1214            .download_certificates_from(remote_node, chain_id, remote_info.next_block_height)
1215            .await?
1216        {
1217            local_info = new_info;
1218        };
1219
1220        // If we are at the same height as the remote node, we also update our chain manager.
1221        if local_info.next_block_height != remote_info.next_block_height {
1222            debug!(
1223                "Synced from validator {}; but remote height is {} and local height is {}",
1224                remote_node.public_key, remote_info.next_block_height, local_info.next_block_height
1225            );
1226            return Ok(());
1227        };
1228
1229        if let Some(timeout) = remote_info.manager.timeout {
1230            self.handle_certificate(*timeout).await?;
1231        }
1232        let mut proposals = Vec::new();
1233        if let Some(proposal) = remote_info.manager.requested_signed_proposal {
1234            proposals.push(*proposal);
1235        }
1236        if let Some(proposal) = remote_info.manager.requested_proposed {
1237            proposals.push(*proposal);
1238        }
1239        if let Some(locking) = remote_info.manager.requested_locking {
1240            match *locking {
1241                LockingBlock::Fast(proposal) => {
1242                    proposals.push(proposal);
1243                }
1244                LockingBlock::Regular(cert) => {
1245                    let hash = cert.hash();
1246                    if let Err(err) = self.try_process_locking_block_from(remote_node, cert).await {
1247                        debug!(
1248                            "Skipping locked block {hash} from validator {} at height {}: {err}",
1249                            remote_node.public_key, local_info.next_block_height,
1250                        );
1251                    }
1252                }
1253            }
1254        }
1255        'proposal_loop: for proposal in proposals {
1256            let owner: AccountOwner = proposal.owner();
1257            if let Err(mut err) = self
1258                .local_node
1259                .handle_block_proposal(proposal.clone())
1260                .await
1261            {
1262                if let LocalNodeError::BlobsNotFound(_) = &err {
1263                    let required_blob_ids = proposal.required_blob_ids().collect::<Vec<_>>();
1264                    if !required_blob_ids.is_empty() {
1265                        let mut blobs = Vec::new();
1266                        for blob_id in required_blob_ids {
1267                            let blob_content = match self
1268                                .requests_scheduler
1269                                .download_pending_blob(remote_node, chain_id, blob_id)
1270                                .await
1271                            {
1272                                Ok(content) => content,
1273                                Err(err) => {
1274                                    info!(
1275                                        "Skipping proposal from {owner} and validator {} at \
1276                                        height {}; failed to download {blob_id}: {err}",
1277                                        remote_node.public_key, local_info.next_block_height
1278                                    );
1279                                    continue 'proposal_loop;
1280                                }
1281                            };
1282                            blobs.push(Blob::new(blob_content));
1283                        }
1284                        self.local_node
1285                            .handle_pending_blobs(chain_id, blobs)
1286                            .await?;
1287                        // We found the missing blobs: retry.
1288                        if let Err(new_err) = self
1289                            .local_node
1290                            .handle_block_proposal(proposal.clone())
1291                            .await
1292                        {
1293                            err = new_err;
1294                        } else {
1295                            continue;
1296                        }
1297                    }
1298                    if let LocalNodeError::BlobsNotFound(blob_ids) = &err {
1299                        self.update_local_node_with_blobs_from(
1300                            blob_ids.clone(),
1301                            &[remote_node.clone()],
1302                        )
1303                        .await?;
1304                        // We found the missing blobs: retry.
1305                        if let Err(new_err) = self
1306                            .local_node
1307                            .handle_block_proposal(proposal.clone())
1308                            .await
1309                        {
1310                            err = new_err;
1311                        } else {
1312                            continue;
1313                        }
1314                    }
1315                }
1316                while let LocalNodeError::WorkerError(WorkerError::ChainError(chain_err)) = &err {
1317                    if let ChainError::MissingCrossChainUpdate {
1318                        chain_id,
1319                        origin,
1320                        height,
1321                    } = &**chain_err
1322                    {
1323                        self.download_sender_block_with_sending_ancestors(
1324                            *chain_id,
1325                            *origin,
1326                            *height,
1327                            remote_node,
1328                        )
1329                        .await?;
1330                        // Retry
1331                        if let Err(new_err) = self
1332                            .local_node
1333                            .handle_block_proposal(proposal.clone())
1334                            .await
1335                        {
1336                            err = new_err;
1337                        } else {
1338                            continue 'proposal_loop;
1339                        }
1340                    } else {
1341                        break;
1342                    }
1343                }
1344
1345                debug!(
1346                    "Skipping proposal from {owner} and validator {} at height {}: {err}",
1347                    remote_node.public_key, local_info.next_block_height
1348                );
1349            }
1350        }
1351        Ok(())
1352    }

1354    async fn try_process_locking_block_from(
1355        &self,
1356        remote_node: &RemoteNode<Env::ValidatorNode>,
1357        certificate: GenericCertificate<ValidatedBlock>,
1358    ) -> Result<(), chain_client::Error> {
1359        let chain_id = certificate.inner().chain_id();
1360        let certificate = Box::new(certificate);
1361        match self.process_certificate(certificate.clone()).await {
1362            Err(LocalNodeError::BlobsNotFound(blob_ids)) => {
1363                let mut blobs = Vec::new();
1364                for blob_id in blob_ids {
1365                    let blob_content = self
1366                        .requests_scheduler
1367                        .download_pending_blob(remote_node, chain_id, blob_id)
1368                        .await?;
1369                    blobs.push(Blob::new(blob_content));
1370                }
1371                self.local_node
1372                    .handle_pending_blobs(chain_id, blobs)
1373                    .await?;
1374                self.process_certificate(certificate).await?;
1375                Ok(())
1376            }
1377            Err(err) => Err(err.into()),
1378            Ok(()) => Ok(()),
1379        }
1380    }

1382    /// Downloads and processes from the specified validators a confirmed block certificates that
1383    /// use the given blobs. If this succeeds, the blob will be in our storage.
1384    async fn update_local_node_with_blobs_from(
1385        &self,
1386        blob_ids: Vec<BlobId>,
1387        remote_nodes: &[RemoteNode<Env::ValidatorNode>],
1388    ) -> Result<Vec<Blob>, chain_client::Error> {
1389        let timeout = self.options.blob_download_timeout;
1390        // Deduplicate IDs.
1391        let blob_ids = blob_ids.into_iter().collect::<BTreeSet<_>>();
1392        stream::iter(blob_ids.into_iter().map(|blob_id| {
1393            communicate_concurrently(
1394                remote_nodes,
1395                async move |remote_node| {
1396                    let certificate = self
1397                        .requests_scheduler
1398                        .download_certificate_for_blob(&remote_node, blob_id)
1399                        .await?;
1400                    self.receive_sender_certificate(
1401                        certificate,
1402                        ReceiveCertificateMode::NeedsCheck,
1403                        Some(vec![remote_node.clone()]),
1404                    )
1405                    .await?;
1406                    let blob = self
1407                        .local_node
1408                        .storage_client()
1409                        .read_blob(blob_id)
1410                        .await?
1411                        .ok_or_else(|| LocalNodeError::BlobsNotFound(vec![blob_id]))?;
1412                    Result::<_, chain_client::Error>::Ok(blob)
1413                },
1414                move |_| chain_client::Error::from(NodeError::BlobsNotFound(vec![blob_id])),
1415                timeout,
1416            )
1417        }))
1418        .buffer_unordered(self.options.max_joined_tasks)
1419        .collect::<Vec<_>>()
1420        .await
1421        .into_iter()
1422        .collect()
1423    }

1425    /// Attempts to execute the block locally. If any incoming message execution fails, that
1426    /// message is rejected and execution is retried, until the block accepts only messages
1427    /// that succeed.
1428    // TODO(#2806): Measure how failing messages affect the execution times.
1429    #[tracing::instrument(level = "trace", skip(self, block))]
1430    async fn stage_block_execution_and_discard_failing_messages(
1431        &self,
1432        mut block: ProposedBlock,
1433        round: Option<u32>,
1434        published_blobs: Vec<Blob>,
1435    ) -> Result<(Block, ChainInfoResponse), chain_client::Error> {
1436        loop {
1437            let result = self
1438                .stage_block_execution(block.clone(), round, published_blobs.clone())
1439                .await;
1440            if let Err(chain_client::Error::LocalNodeError(LocalNodeError::WorkerError(
1441                WorkerError::ChainError(chain_error),
1442            ))) = &result
1443            {
1444                if let ChainError::ExecutionError(
1445                    error,
1446                    ChainExecutionContext::IncomingBundle(index),
1447                ) = &**chain_error
1448                {
1449                    let transaction = block
1450                        .transactions
1451                        .get_mut(*index as usize)
1452                        .expect("Transaction at given index should exist");
1453                    let Transaction::ReceiveMessages(incoming_bundle) = transaction else {
1454                        panic!(
1455                            "Expected incoming bundle at transaction index {}, found operation",
1456                            index
1457                        );
1458                    };
1459                    ensure!(
1460                        !incoming_bundle.bundle.is_protected(),
1461                        chain_client::Error::BlockProposalError(
1462                            "Protected incoming message failed to execute locally"
1463                        )
1464                    );
1465                    if incoming_bundle.action == MessageAction::Reject {
1466                        return result;
1467                    }
1468                    // Reject the faulty message from the block and continue.
1469                    // TODO(#1420): This is potentially a bit heavy-handed for
1470                    // retryable errors.
1471                    info!(
1472                        %error, %index, origin = ?incoming_bundle.origin,
1473                        "Message bundle failed to execute locally and will be rejected."
1474                    );
1475                    incoming_bundle.action = MessageAction::Reject;
1476                    continue;
1477                }
1478            }
1479            return result;
1480        }
1481    }

1483    /// Attempts to execute the block locally. If any attempt to read a blob fails, the blob is
1484    /// downloaded and execution is retried.
1485    #[instrument(level = "trace", skip(self, block))]
1486    async fn stage_block_execution(
1487        &self,
1488        block: ProposedBlock,
1489        round: Option<u32>,
1490        published_blobs: Vec<Blob>,
1491    ) -> Result<(Block, ChainInfoResponse), chain_client::Error> {
1492        loop {
1493            let result = self
1494                .local_node
1495                .stage_block_execution(block.clone(), round, published_blobs.clone())
1496                .await;
1497            if let Err(LocalNodeError::BlobsNotFound(blob_ids)) = &result {
1498                let validators = self.validator_nodes().await?;
1499                self.update_local_node_with_blobs_from(blob_ids.clone(), &validators)
1500                    .await?;
1501                continue; // We found the missing blob: retry.
1502            }
1503            if let Ok((block, _)) = &result {
1504                let hash = CryptoHash::new(block);
1505                let notification = Notification {
1506                    chain_id: block.header.chain_id,
1507                    reason: Reason::BlockExecuted {
1508                        height: block.header.height,
1509                        hash,
1510                    },
1511                };
1512                self.notifier.notify(&[notification]);
1513            }
1514            return Ok(result?);
1515        }
1516    }
1517}

1519/// Performs `f` in parallel on multiple nodes, starting with a quadratically increasing delay on
1520/// each subsequent node. Returns error `err` if all of the nodes fail.
1521async fn communicate_concurrently<'a, A, E1, E2, F, G, R, V>(
1522    nodes: &[RemoteNode<A>],
1523    f: F,
1524    err: G,
1525    timeout: Duration,
1526) -> Result<V, E2>
1527where
1528    F: Clone + FnOnce(RemoteNode<A>) -> R,
1529    RemoteNode<A>: Clone,
1530    G: FnOnce(Vec<(ValidatorPublicKey, E1)>) -> E2,
1531    R: Future<Output = Result<V, E1>> + 'a,
1532{
1533    let mut stream = nodes
1534        .iter()
1535        .zip(0..)
1536        .map(|(remote_node, i)| {
1537            let fun = f.clone();
1538            let node = remote_node.clone();
1539            async move {
1540                linera_base::time::timer::sleep(timeout * i * i).await;
1541                fun(node).await.map_err(|err| (remote_node.public_key, err))
1542            }
1543        })
1544        .collect::<FuturesUnordered<_>>();
1545    let mut errors = vec![];
1546    while let Some(maybe_result) = stream.next().await {
1547        match maybe_result {
1548            Ok(result) => return Ok(result),
1549            Err(error) => errors.push(error),
1550        };
1551    }
1552    Err(err(errors))
1553}

1607/// Creates a compressed Contract, Service and bytecode.
1608#[cfg(not(target_arch = "wasm32"))]
1609pub async fn create_bytecode_blobs(
1610    contract: Bytecode,
1611    service: Bytecode,
1612    vm_runtime: VmRuntime,
1613) -> (Vec<Blob>, ModuleId) {
1614    match vm_runtime {
1615        VmRuntime::Wasm => {
1616            let (compressed_contract, compressed_service) =
1617                tokio::task::spawn_blocking(move || (contract.compress(), service.compress()))
1618                    .await
1619                    .expect("Compression should not panic");
1620            let contract_blob = Blob::new_contract_bytecode(compressed_contract);
1621            let service_blob = Blob::new_service_bytecode(compressed_service);
1622            let module_id =
1623                ModuleId::new(contract_blob.id().hash, service_blob.id().hash, vm_runtime);
1624            (vec![contract_blob, service_blob], module_id)
1625        }
1626        VmRuntime::Evm => {
1627            let compressed_contract = contract.compress();
1628            let evm_contract_blob = Blob::new_evm_bytecode(compressed_contract);
1629            let module_id = ModuleId::new(
1630                evm_contract_blob.id().hash,
1631                evm_contract_blob.id().hash,
1632                vm_runtime,
1633            );
1634            (vec![evm_contract_blob], module_id)
1635        }
1636    }
