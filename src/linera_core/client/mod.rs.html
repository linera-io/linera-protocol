164/// Defines what type of notifications we should process for a chain:
165/// - do we fully participate in consensus and download sender chains?
166/// - or do we only follow the chain's blocks without participating?
167/// - or do we only care about blocks containing events from some particular streams?
168#[derive(Debug, Clone, PartialEq, Eq)]
169pub enum ListeningMode {
170    /// Listen to everything: all blocks for the chain and all blocks from sender chains,
171    /// and participate in rounds.
172    FullChain,
173    /// Listen to all blocks for the chain, but don't download sender chain blocks or participate
174    /// in rounds. Use this when interested in the chain's state but not intending to propose
175    /// blocks (e.g., because we're not a chain owner).
176    FollowChain,
177    /// Only listen to blocks which contain events from those streams.
178    EventsOnly(BTreeSet<StreamId>),
179}
180
181impl PartialOrd for ListeningMode {
182    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
183        match (self, other) {
184            (ListeningMode::FullChain, ListeningMode::FullChain) => Some(Ordering::Equal),
185            (ListeningMode::FullChain, _) => Some(Ordering::Greater),
186            (_, ListeningMode::FullChain) => Some(Ordering::Less),
187            (ListeningMode::FollowChain, ListeningMode::FollowChain) => Some(Ordering::Equal),
188            (ListeningMode::FollowChain, ListeningMode::EventsOnly(_)) => Some(Ordering::Greater),
189            (ListeningMode::EventsOnly(_), ListeningMode::FollowChain) => Some(Ordering::Less),
190            (ListeningMode::EventsOnly(a), ListeningMode::EventsOnly(b)) => {
191                if a == b {
192                    Some(Ordering::Equal)
193                } else if a.is_superset(b) {
194                    Some(Ordering::Greater)
195                } else if b.is_superset(a) {
196                    Some(Ordering::Less)
197                } else {
198                    None
199                }
200            }
201        }
202    }
203}

205impl ListeningMode {
206    /// Returns whether a notification with this reason should be processed under this listening
207    /// mode.
208    pub fn is_relevant(&self, reason: &Reason) -> bool {
209        match (reason, self) {
210            // NewEvents is only processed in EventsOnly mode, the other modes depend on
211            // the NewBlock notification.
212            (Reason::NewEvents { .. }, ListeningMode::FollowChain | ListeningMode::FullChain) => {
213                false
214            }
215            // FullChain processes everything.
216            (_, ListeningMode::FullChain) => true,
217            // FollowChain processes new blocks on the chain itself, including blocks that
218            // produced events.
219            (Reason::NewBlock { .. }, ListeningMode::FollowChain) => true,
220            (_, ListeningMode::FollowChain) => false,
221            // EventsOnly only processes events from relevant streams.
222            (Reason::NewEvents { event_streams, .. }, ListeningMode::EventsOnly(relevant)) => {
223                relevant.intersection(event_streams).next().is_some()
224            }
225            (_, ListeningMode::EventsOnly(_)) => false,
226        }
227    }
228
229    /// Widens this mode to also cover the given mode, keeping the more inclusive of the two.
230    pub fn extend(&mut self, other: Option<ListeningMode>) {
231        match (self, other) {
232            (_, None) => (),
233            (ListeningMode::FullChain, _) => (),
234            (mode, Some(ListeningMode::FullChain)) => {
235                *mode = ListeningMode::FullChain;
236            }
237            (ListeningMode::FollowChain, _) => (),
238            (mode, Some(ListeningMode::FollowChain)) => {
239                *mode = ListeningMode::FollowChain;
240            }
241            (
242                ListeningMode::EventsOnly(self_events),
243                Some(ListeningMode::EventsOnly(other_events)),
244            ) => {
245                self_events.extend(other_events);
246            }
247        }
248    }
249
250    /// Returns whether this mode implies follow-only behavior (i.e., not participating in
251    /// consensus rounds).
252    pub fn is_follow_only(&self) -> bool {
253        !matches!(self, ListeningMode::FullChain)
254    }
255
256    /// Returns whether this is a full chain mode (synchronizing sender chains and updating
257    /// inboxes).
258    pub fn is_full(&self) -> bool {
259        matches!(self, ListeningMode::FullChain)
260    }
261
262    /// Returns whether this mode requires synchronizing the chain's own state.
263    pub fn should_sync_chain_state(&self) -> bool {
264        match self {
265            ListeningMode::FullChain | ListeningMode::FollowChain => true,
266            ListeningMode::EventsOnly(_) => false,
267        }
268    }
269}

271/// The per-chain [`ListeningMode`]s tracked by a local node, together with a memoized
272/// [`Hashed`] of the fully-tracked subset.
273///
274/// The hash is the version of the outbox index (see
275/// [`ChainStateView::outbox_index_tracked_hash`]). Caching it here — recomputed only when a chain
276/// newly becomes `FullChain`, which is rare and client-side — lets every cross-chain operation
277/// compare and reconcile the index in `O(1)` instead of rehashing the tracked set each time.
278///
279/// [`ChainStateView::outbox_index_tracked_hash`]: linera_chain::ChainStateView
280#[derive(Debug)]
281pub struct ChainModes {
282    modes: BTreeMap<ChainId, ListeningMode>,
283    full: Arc<Hashed<ChainIdSet>>,
284}
285
286impl Default for ChainModes {
287    fn default() -> Self {
288        Self::new(BTreeMap::new())
289    }
290}
291
292impl ChainModes {
293    /// Builds the listening modes from `modes`, computing the tracked-set hash once.
294    pub fn new(modes: BTreeMap<ChainId, ListeningMode>) -> Self {
295        let full = Self::compute_full(&modes);
296        Self { modes, full }
297    }
298
299    fn compute_full(modes: &BTreeMap<ChainId, ListeningMode>) -> Arc<Hashed<ChainIdSet>> {
300        Arc::new(Hashed::new(ChainIdSet(
301            modes
302                .iter()
303                .filter(|(_, mode)| mode.is_full())
304                .map(|(id, _)| *id)
305                .collect(),
306        )))
307    }
308
309    /// The fully-tracked chains together with their memoized hash (`O(1)` clone).
310    pub fn full(&self) -> Arc<Hashed<ChainIdSet>> {
311        self.full.clone()
312    }
313
314    /// Returns the listening mode for `chain_id`, if it is tracked.
315    pub fn get(&self, chain_id: &ChainId) -> Option<&ListeningMode> {
316        self.modes.get(chain_id)
317    }
318
319    /// Merges `mode` into the entry for `chain_id` — monotonic in the listening-mode order, so it
320    /// never weakens an existing entry — and returns the resulting mode. The tracked-set hash is
321    /// recomputed only if the chain newly became `FullChain`.
322    pub fn extend_mode(&mut self, chain_id: ChainId, mode: ListeningMode) -> ListeningMode {
323        let entry = self
324            .modes
325            .entry(chain_id)
326            .or_insert_with(|| ListeningMode::EventsOnly(BTreeSet::new()));
327        let was_full = entry.is_full();
328        entry.extend(Some(mode));
329        let result = entry.clone();
330        if !was_full && result.is_full() {
331            self.full = Self::compute_full(&self.modes);
332        }
333        result
334    }
335
336    /// Stops tracking `chain_id`, removing its entry entirely. Returns the removed mode, if any.
337    /// The tracked-set hash is recomputed only if the removed chain was `FullChain`.
338    pub fn remove_mode(&mut self, chain_id: &ChainId) -> Option<ListeningMode> {
339        let removed = self.modes.remove(chain_id)?;
340        if removed.is_full() {
341            self.full = Self::compute_full(&self.modes);
342        }
343        Some(removed)
344    }
345}

379    /// Creates a new `Client` with a new cache and notifiers.
380    #[instrument(level = "trace", skip_all)]
381    #[expect(clippy::too_many_arguments)]
382    pub fn new(
383        environment: Env,
384        admin_chain_id: ChainId,
385        long_lived_services: bool,
386        chain_modes: impl IntoIterator<Item = (ChainId, ListeningMode)>,
387        name: impl Into<String>,
388        chain_worker_ttl: Option<Duration>,
389        sender_chain_worker_ttl: Option<Duration>,
390        cross_chain_batch_size_limit: usize,
391        options: chain_client::Options,
392        block_cache_size: usize,
393        execution_state_cache_size: usize,
394        requests_scheduler_config: &requests_scheduler::RequestsSchedulerConfig,
395    ) -> Self {
396        let mut modes = chain_modes.into_iter().collect::<BTreeMap<_, _>>();
397        // The client needs the admin chain fully synced for epoch tracking, so
398        // promote it (or insert) into `FullChain`. `extend` is monotonic in the
399        // listening mode order, so this never weakens an existing entry.
400        modes
401            .entry(admin_chain_id)
402            .or_insert(ListeningMode::FullChain)
403            .extend(Some(ListeningMode::FullChain));
404        let chain_modes = Arc::new(RwLock::new(ChainModes::new(modes)));
405        let config = ChainWorkerConfig {
406            nickname: name.into(),
407            long_lived_services,
408            allow_inactive_chains: true,
409            ttl: chain_worker_ttl,
410            sender_chain_ttl: sender_chain_worker_ttl,
411            block_cache_size,
412            execution_state_cache_size,
413            cross_chain_batch_size_limit,
414            ..ChainWorkerConfig::default()
415        };
416        let state = WorkerState::new(
417            environment.storage().clone(),
418            config,
419            Some(chain_modes.clone()),
420        );
421        let clock = environment.storage().clock().clone();
422        let local_node = LocalNodeClient::new(state);
423        let requests_scheduler = Arc::new(RequestsScheduler::new(
424            vec![],
425            requests_scheduler_config,
426            clock,
427        ));
428
429        Self {
430            environment,
431            local_node,
432            requests_scheduler,
433            chains: papaya::HashMap::new(),
434            admin_chain_id,
435            chain_modes,
436            notifier: Arc::new(ChannelNotifier::default()),
437            options,
438        }

630        chain_id: ChainId,
631        target_next_block_height: BlockHeight,
632    ) -> Result<Box<ChainInfo>, chain_client::Error> {
633        let validators = self.validator_nodes().await?;
634        let mut info = Box::pin(self.fetch_chain_info(chain_id, &validators)).await?;
635        if target_next_block_height <= info.next_block_height {
636            return Ok(info);
637        }
638        info = self
639            .load_local_certificates(chain_id, target_next_block_height, None)
640            .await?;
641        let mut next_height = info.next_block_height;
642        // Download remaining batches using all validators with staggered fallback.
643        while next_height < target_next_block_height {
644            let limit = u64::from(target_next_block_height)
645                .checked_sub(u64::from(next_height))
646                .ok_or(ArithmeticError::Overflow)?
647                .min(self.options.certificate_download_batch_size);
648            let certificates = self
649                .requests_scheduler
650                .download_certificates_from_validators(
651                    &validators,
652                    chain_id,
653                    next_height,
654                    limit,
655                    self.options.certificate_batch_download_hedge_delay,
656                )
657                .await?;
658            let Some(new_info) = self
659                .process_certificates(
660                    &validators,
661                    certificates,
662                    None,
663                    ProcessConfirmedBlockMode::Execute,
664                )
665                .await?
666            else {
667                break;
668            };
669            assert!(new_info.next_block_height > next_height);
670            next_height = new_info.next_block_height;
671            info = new_info;
672        }
673        ensure!(
674            target_next_block_height <= info.next_block_height,
675            chain_client::Error::CannotDownloadCertificates {
676                chain_id,
677                target_next_block_height,
678            }
679        );
680        Ok(info)
681    }

683    /// Loads and processes certificates from local storage for the given chain, from the
684    /// current local height up to `end`. Returns the chain info after processing.
685    /// If `until_block_time` is `Some`, stops before processing any certificate whose
686    /// block timestamp is >= the given value (exclusive).
687    async fn load_local_certificates(
688        &self,
689        chain_id: ChainId,
690        end: BlockHeight,
691        until_block_time: Option<Timestamp>,
692    ) -> Result<Box<ChainInfo>, chain_client::Error> {
693        let mut last_info = self.local_node.chain_info(chain_id).await?;
694        let next_height = last_info.next_block_height;
695        let hashes = self
696            .local_node
697            .get_preprocessed_block_hashes(chain_id, next_height, end)
698            .await?;
699        let certificates = self.storage_client().read_certificates(&hashes).await?;
700        let certificates = match ResultReadCertificates::new(certificates, hashes) {
701            ResultReadCertificates::Certificates(certificates) => certificates,
702            ResultReadCertificates::InvalidHashes(hashes) => {
703                return Err(chain_client::Error::ReadCertificatesError(hashes))
704            }
705        };
706        for certificate in certificates {
707            if let Some(until) = until_block_time {
708                if certificate.value().block().header.timestamp >= until {
709                    break;
710                }
711            }
712            last_info = self.handle_certificate(certificate).await?.info;
713        }
714        Ok(last_info)
715    }

717    /// Downloads and processes certificates from the given validator.
718    ///
719    /// Stops when either condition is met:
720    /// - `stop`: the local chain has reached that height (exclusive).
721    /// - `until_block_time`: the next block's timestamp is >= that value (exclusive).
722    #[instrument(level = "trace", skip_all)]
723    async fn download_certificates_from(
724        &self,
725        remote_node: &RemoteNode<Env::ValidatorNode>,
726        chain_id: ChainId,
727        stop: BlockHeight,
728        until_block_time: Option<Timestamp>,
729    ) -> Result<Box<ChainInfo>, chain_client::Error> {
730        let mut last_info = self
731            .load_local_certificates(chain_id, stop, until_block_time)
732            .await?;
733        let mut next_height = last_info.next_block_height;
734
735        if next_height >= stop {
736            return Ok(last_info);
737        }
738
739        // Download remaining certificates from the remote node using a pipelined
740        // sliding window. A background task downloads up to `max_concurrent_batch_downloads`
741        // batches concurrently and sends them through a channel for sequential processing.
742        #[cfg(not(web))]
743        type CertificateBatchFuture = std::pin::Pin<
744            Box<dyn Future<Output = Result<Vec<ConfirmedBlockCertificate>, NodeError>> + Send>,
745        >;
746        #[cfg(web)]
747        type CertificateBatchFuture = std::pin::Pin<
748            Box<dyn Future<Output = Result<Vec<ConfirmedBlockCertificate>, NodeError>>>,
749        >;
750
751        let max_concurrent = self.options.max_concurrent_batch_downloads;
752        let batch_size = self.options.certificate_download_batch_size;
753        let (sender, mut receiver) = tokio::sync::mpsc::channel(max_concurrent);
754        let scheduler = self.requests_scheduler.clone();
755        let remote = remote_node.clone();
756
757        let download_task = linera_base::Task::spawn(async move {
758            let mut download_height = next_height;
759            let mut in_flight = FuturesOrdered::<CertificateBatchFuture>::new();
760
761            let try_enqueue = |in_flight: &mut FuturesOrdered<CertificateBatchFuture>,
762                               download_height: &mut BlockHeight| {
763                if *download_height >= stop {
764                    return;
765                }
766                let limit = u64::from(stop)
767                    .saturating_sub(u64::from(*download_height))
768                    .min(batch_size);
769                let height = *download_height;
770                let scheduler = scheduler.clone();
771                let remote = remote.clone();
772                in_flight.push_back(Box::pin(async move {
773                    scheduler
774                        .download_certificates(&remote, chain_id, height, limit)
775                        .await
776                }));
777                *download_height = BlockHeight(u64::from(*download_height) + limit);
778            };
779
780            while in_flight.len() < max_concurrent && download_height < stop {
781                try_enqueue(&mut in_flight, &mut download_height);
782            }
783
784            while let Some(result) = in_flight.next().await {
785                if sender.send(result).await.is_err() {
786                    break;
787                }
788                try_enqueue(&mut in_flight, &mut download_height);
789            }
790        });
791
792        // Process downloaded batches sequentially.
793        while let Some(result) = receiver.recv().await {
794            let certificates = result?;
795            let Some(info) = self
796                .process_certificates(
797                    slice::from_ref(remote_node),
798                    certificates,
799                    until_block_time,
800                    ProcessConfirmedBlockMode::Execute,
801                )
802                .await?
803            else {
804                break;
805            };
806            assert!(info.next_block_height >= next_height);
807            next_height = info.next_block_height;
808            last_info = info;
809        }
810        // Await the downloader so any panic inside the spawned task surfaces here
811        // instead of being silently swallowed when the channel closes.
812        download_task.await;
813        Ok(last_info)
814    }

835    /// Downloads the publisher chain certificates that contain the given events,
836    /// using the event block height index on validators. Queries a validator for
837    /// the block heights, downloads those certificates, and processes them — all
838    /// as one atomic unit per validator attempt, with staggered fallback.
839    #[instrument(level = "trace", skip_all)]
840    pub(crate) async fn download_certificates_for_events(
841        &self,
842        event_ids: &[EventId],
843    ) -> Result<(), chain_client::Error> {
844        let mut validators = self.validator_nodes().await?;
845        let hedge_delay = self.options.certificate_batch_download_hedge_delay;
846        let mut remaining_event_ids = event_ids.to_vec();
847
848        while !remaining_event_ids.is_empty() {
849            let remaining_ref = &remaining_event_ids;
850            validators.shuffle(&mut rand::thread_rng());
851            let result = communicate_concurrently(
852                &validators,
853                move |remote_node| {
854                    let validator_key = remote_node.public_key;
855                    let validator_address = remote_node.address();
856                    Box::pin(async move {
857                        // Query this validator for the block heights.
858                        let heights = remote_node
859                            .node
860                            .event_block_heights(remaining_ref.to_vec())
861                            .await?;
862
863                        // Separate resolved and unresolved events.
864                        let mut chain_heights = BTreeMap::<_, BTreeSet<_>>::new();
865                        let mut expected_events = BTreeMap::<_, HashSet<EventId>>::new();
866                        let mut unresolved = Vec::new();
867                        for (event_id, maybe_height) in remaining_ref.iter().zip(heights) {
868                            if let Some(height) = maybe_height {
869                                chain_heights
870                                    .entry(event_id.chain_id)
871                                    .or_default()
872                                    .insert(height);
873                                expected_events
874                                    .entry((event_id.chain_id, height))
875                                    .or_default()
876                                    .insert(event_id.clone());
877                            } else {
878                                unresolved.push(event_id.clone());
879                            }
880                        }
881                        if chain_heights.is_empty() {
882                            // This validator has no useful information.
883                            return Err(chain_client::Error::from(NodeError::EventsNotFound(remaining_ref.clone())));
884                        }
885
886                        // Download certificates and verify them.
887                        let mut checked_certificates = Vec::<ConfirmedBlockCertificate>::new();
888                        for (chain_id, heights) in chain_heights {
889                            let heights_vec = heights.into_iter().collect::<Vec<_>>();
890                            let certificates = self
891                                .requests_scheduler
892                                .download_certificates_by_heights(
893                                    &remote_node,
894                                    chain_id,
895                                    heights_vec,
896                                )
897                                .await?;
898                            for cert in &certificates {
899                                // Verify the block contains the expected events.
900                                let block = cert.block();
901                                let block_event_ids = block.event_ids().collect::<HashSet<_>>();
902                                if let Some(expected_event_ids) =
903                                    expected_events.get(&(chain_id, block.header.height))
904                                {
905                                    if !expected_event_ids.is_subset(&block_event_ids) {
906                                        tracing::debug!(
907                                            %validator_address, ?expected_event_ids, ?block_event_ids,
908                                            "validator lied about events in block."
909                                        );
910                                        return Err(NodeError::UnexpectedCertificateValue.into());
911                                    }
912                                }
913                            }
914                            for cert in certificates {
915                                self.check_certificate(&cert)
916                                    .await
917                                    .map_err(|error| {
918                                        tracing::debug!(
919                                            %validator_address, %error,
920                                            "invalid certificate"
921                                        );
922                                        error
923                                    })?
924                                    .into_result()
925                                    .map_err(|error| {
926                                        tracing::debug!(
927                                            %validator_address, %error,
928                                            "could not check certificate"
929                                        );
930                                        error
931                                    })?;
932                                checked_certificates.push(cert);
933                            }
934                        }
935                        Ok((checked_certificates, unresolved, validator_key))
936                    })
937                },
938                hedge_delay,
939                self.storage_client().clock(),
940            )
941            .await;
942
943            match result {
944                Ok((certificates, unresolved, validator_key)) => {
945                    for certificate in certificates {
946                        let mode = ReceiveCertificateMode::AlreadyChecked;
947                        self.receive_sender_certificate(
948                            self.storage_client().cache_certificate(certificate),
949                            mode,
950                            None,
951                        )
952                        .await?;
953                    }
954                    validators.retain(|node| node.public_key != validator_key);
955                    remaining_event_ids = unresolved;
956                }
957                Err(errors) => {
958                    for (validator, error) in &errors {
959                        warn!(
960                            %validator,
961                            %error,
962                            "failed to download event certificates from validator",
963                        );
964                    }
965                    // All validators failed; no point retrying.
966                    return Err(NodeError::EventsNotFound(remaining_event_ids).into());
967                }
968            }
969        }
970        Ok(())
971    }

973    /// Downloads the checkpoint certificate at `checkpoint_height` from `remote_node`
974    /// and processes it locally, if our chain isn't already past that height. The
975    /// worker's `process_confirmed_block` recognises the gap-plus-checkpoint case and
976    /// installs the chain's execution state from the checkpoint blob before re-running
977    /// the certificate.
978    ///
979    /// The certificate's signatures are still verified against the committee resolved
980    /// from the admin chain's epoch event stream, so the remote node is trusted only
981    /// to point us at a height — not to forge the snapshot itself.
982    #[instrument(level = "trace", skip_all)]
983    async fn bootstrap_chain_from_checkpoint(
984        &self,
985        remote_node: &RemoteNode<Env::ValidatorNode>,
986        chain_id: ChainId,
987        checkpoint_height: BlockHeight,
988    ) -> Result<(), chain_client::Error> {
989        let local_next = match self.local_node.chain_info(chain_id).await {
990            Ok(info) => info.next_block_height,
991            // A freshly-created follower whose storage doesn't yet hold this
992            // chain's description blob: treat as height 0 and let the
993            // checkpoint cert install the snapshot. The chain description is
994            // the only blob `chain_info` ever needs.
995            Err(LocalNodeError::BlobsNotFound(_)) => BlockHeight::ZERO,
996            Err(err) => return Err(err.into()),
997        };
998        if local_next > checkpoint_height {
999            return Ok(());
1000        }
1001        let certificates = remote_node
1002            .download_certificates_by_heights(chain_id, vec![checkpoint_height])
1003            .await?;
1004        if certificates.is_empty() {
1005            // The validator advertised a checkpoint height it can't actually serve;
1006            // skip and let the regular sync path take over.
1007            return Ok(());
1008        }
1009        // The first attempt at processing the checkpoint cert will fall into the
1010        // worker's `BlocksNotFound` pre-check if pre-checkpoint sender blocks are
1011        // missing; `handle_certificate_with_retry` downloads them by hash and
1012        // retries, so by the time `process_certificates` returns the chain has
1013        // both its restored state and every certified sender block in storage.
1014        self.process_certificates(
1015            slice::from_ref(remote_node),
1016            certificates,
1017            None,
1018            ProcessConfirmedBlockMode::Execute,
1019        )
1020        .await?;
1021        Ok(())
1022    }

1024    /// Tries to process all the certificates, requesting any missing blobs from the given nodes.
1025    /// Returns the chain info of the last successfully processed certificate.
1026    /// If `until_block_time` is `Some`, stops before processing any certificate whose
1027    /// block timestamp is greater or equal than the given value.
1028    #[instrument(level = "trace", skip_all)]
1029    async fn process_certificates(
1030        &self,
1031        remote_nodes: &[RemoteNode<Env::ValidatorNode>],
1032        certificates: Vec<ConfirmedBlockCertificate>,
1033        until_block_time: Option<Timestamp>,
1034        mode: ProcessConfirmedBlockMode,
1035    ) -> Result<Option<Box<ChainInfo>>, chain_client::Error> {
1036        let mut info = None;
1037        // Blobs created by these certs are already embedded in the downloaded
1038        // block bodies, so they don't need to be fetched from a validator. The
1039        // chain worker resolves them from `Block::created_blobs()` during
1040        // `handle_certificate`.
1041        let created_blob_ids = certificates
1042            .iter()
1043            .flat_map(|certificate| certificate.value().block().created_blob_ids())
1044            .collect::<BTreeSet<BlobId>>();
1045        let required_blob_ids = certificates
1046            .iter()
1047            .flat_map(|certificate| certificate.value().required_blob_ids())
1048            .filter(|blob_id| !created_blob_ids.contains(blob_id))
1049            .collect::<Vec<_>>();
1050
1051        match self
1052            .local_node
1053            .read_blob_states_from_storage(&required_blob_ids)
1054            .await
1055        {
1056            Err(LocalNodeError::BlobsNotFound(blob_ids)) => {
1057                self.download_blobs(remote_nodes, &blob_ids).await?;
1058            }
1059            x => {
1060                x?;
1061            }
1062        }
1063
1064        for certificate in certificates {
1065            if let Some(until) = until_block_time {
1066                if certificate.value().block().header.timestamp >= until {
1067                    break;
1068                }
1069            }
1070            let response = self
1071                .handle_certificate_with_retry(&certificate, remote_nodes, mode)
1072                .await?;
1073            info = Some(response.info);
1074        }
1075
1076        Ok(info)
1077    }

1079    /// Calls `handle_confirmed_certificate`, retrying with any missing blobs (downloaded
1080    /// from `nodes`) and any missing events (downloaded from the publisher
1081    /// chains via the current validators).
1082    async fn handle_certificate_with_retry(
1083        &self,
1084        certificate: &ConfirmedBlockCertificate,
1085        nodes: &[RemoteNode<Env::ValidatorNode>],
1086        mode: ProcessConfirmedBlockMode,
1087    ) -> Result<ChainInfoResponse, chain_client::Error> {
1088        let mut downloaded_blobs = HashSet::<BlobId>::new();
1089        let mut downloaded_blocks = HashSet::<CryptoHash>::new();
1090        let mut events = EventSetDownloader::new(self);
1091        loop {
1092            let result = self
1093                .handle_confirmed_certificate(certificate.clone(), mode)
1094                .await;
1095            if let Err(LocalNodeError::BlobsNotFound(blob_ids)) = &result {
1096                let new_blobs = filter_new(blob_ids, &downloaded_blobs);
1097                if !new_blobs.is_empty() {
1098                    self.download_blobs(nodes, &new_blobs).await?;
1099                    downloaded_blobs.extend(new_blobs);
1100                    continue;
1101                }
1102            }
1103            if let Err(LocalNodeError::BlocksNotFound(hashes)) = &result {
1104                let new_blocks = filter_new(hashes, &downloaded_blocks);
1105                if !new_blocks.is_empty() {
1106                    self.download_pre_checkpoint_blocks(nodes, &new_blocks)
1107                        .await?;
1108                    downloaded_blocks.extend(new_blocks);
1109                    continue;
1110                }
1111            }
1112            if let Err(LocalNodeError::EventsNotFound(event_ids)) = &result {
1113                if events.download_new(event_ids).await? {
1114                    continue;
1115                }
1116            }
1117            return Ok(result?);
1118        }
1119    }

1121    /// Downloads each missing pre-checkpoint sender block from `nodes` and feeds it
1122    /// through the local worker. The worker's trust-mark accept path verifies the
1123    /// cert against its own epoch's committee and writes it to storage. Routing
1124    /// through `handle_certificate_with_retry` ensures the sender block's own
1125    /// blob/event dependencies (e.g. a `ChainDescription` blob or an admin-chain
1126    /// epoch event for a revoked epoch) get resolved before the cert is accepted.
1127    async fn download_pre_checkpoint_blocks(
1128        &self,
1129        nodes: &[RemoteNode<Env::ValidatorNode>],
1130        hashes: &[CryptoHash],
1131    ) -> Result<(), chain_client::Error> {
1132        for hash in hashes {
1133            let mut last_error = None;
1134            for node in nodes {
1135                match node.node.download_certificate(*hash).await {
1136                    Ok(certificate) => {
1137                        Box::pin(self.handle_certificate_with_retry(
1138                            &certificate,
1139                            nodes,
1140                            ProcessConfirmedBlockMode::Auto,
1141                        ))
1142                        .await?;
1143                        last_error = None;
1144                        break;
1145                    }
1146                    Err(error) => last_error = Some(error),
1147                }
1148            }
1149            if let Some(error) = last_error {
1150                return Err(error.into());
1151            }
1152        }
1153        Ok(())
1154    }

1306    /// Submits a block proposal to the validators.
1307    #[instrument(level = "trace", skip_all)]
1308    async fn submit_block_proposal<T: ProcessableCertificate>(
1309        self: &Arc<Self>,
1310        committee: Arc<Committee>,
1311        proposal: Box<BlockProposal>,
1312        value: T,
1313    ) -> Result<GenericCertificate<T>, chain_client::Error> {
1314        debug!(
1315            round = %proposal.content.round,
1316            "Submitting block proposal to validators"
1317        );
1318
1319        // Check if the block timestamp is in the future and log INFO.
1320        let block_timestamp = proposal.content.block.timestamp;
1321        let local_time = self.local_node.storage_client().clock().current_time();
1322        if block_timestamp > local_time {
1323            info!(
1324                chain_id = %proposal.content.block.chain_id,
1325                %block_timestamp,
1326                %local_time,
1327                "Block timestamp is in the future; waiting until it can be proposed",
1328            );
1329        }
1330
1331        // Create channel for clock skew reports from validators.
1332        let (clock_skew_sender, mut clock_skew_receiver) = mpsc::unbounded_channel();
1333        let submit_action = CommunicateAction::SubmitBlock {
1334            proposal,
1335            blob_ids: value.required_blob_ids().into_iter().collect(),
1336            clock_skew_sender,
1337        };
1338
1339        // Spawn a task to monitor clock skew reports and warn if threshold is reached.
1340        let validity_threshold = committee.validity_threshold();
1341        let committee_clone = committee.clone();
1342        let clock_skew_check_handle = linera_base::Task::spawn(async move {
1343            let mut skew_weight = 0u64;
1344            let mut min_skew = TimeDelta::MAX;
1345            let mut max_skew = TimeDelta::ZERO;
1346            while let Some((public_key, clock_skew)) = clock_skew_receiver.recv().await {
1347                if clock_skew.as_micros() > 0 {
1348                    skew_weight += committee_clone.weight(&public_key);
1349                    min_skew = min_skew.min(clock_skew);
1350                    max_skew = max_skew.max(clock_skew);
1351                    if skew_weight >= validity_threshold {
1352                        warn!(
1353                            skew_weight,
1354                            validity_threshold,
1355                            min_skew_ms = min_skew.as_micros() / 1000,
1356                            max_skew_ms = max_skew.as_micros() / 1000,
1357                            "A validity threshold of validators reported clock skew; \
1358                             consider checking your system clock",
1359                        );
1360                        return;
1361                    }
1362                }
1363            }
1364        });
1365
1366        let certificate = self
1367            .communicate_chain_action(&committee, submit_action, value)
1368            .await?;
1369
1370        clock_skew_check_handle.await;
1371
1372        self.handle_certificate(certificate.clone()).await?;
1373        Ok(certificate)
1374    }

1410    /// Broadcasts certified blocks and optionally a block proposal, certificate or
1411    /// leader timeout request.
1412    ///
1413    /// In that case, it verifies that the validator votes are for the provided value,
1414    /// and returns a certificate.
1415    #[instrument(level = "trace", skip_all)]
1416    async fn communicate_chain_action<T: CertificateValue>(
1417        self: &Arc<Self>,
1418        committee: &Committee,
1419        action: CommunicateAction,
1420        value: T,
1421    ) -> Result<GenericCertificate<T>, chain_client::Error> {
1422        let nodes = self.make_nodes(committee)?;
1423        let ((votes_hash, votes_round), votes) = communicate_with_quorum(
1424            &nodes,
1425            committee,
1426            |vote: &LiteVote| (vote.value.value_hash, vote.round),
1427            |remote_node| {
1428                let mut updater = ValidatorUpdater {
1429                    remote_node,
1430                    client: self.clone(),
1431                    admin_chain_id: self.admin_chain_id,
1432                };
1433                let action = action.clone();
1434                Box::pin(async move { updater.send_chain_update(action).await })
1435            },
1436            self.options.quorum_grace_period,
1437        )
1438        .await?;
1439        ensure!(
1440            (votes_hash, votes_round) == (value.hash(), action.round()),
1441            chain_client::Error::UnexpectedQuorum {
1442                hash: votes_hash,
1443                round: votes_round,
1444                expected_hash: value.hash(),
1445                expected_round: action.round(),
1446            }
1447        );
1448        // Certificate is valid because
1449        // * `communicate_with_quorum` ensured a sufficient "weight" of
1450        // (non-error) answers were returned by validators.
1451        // * each answer is a vote signed by the expected validator.
1452        let certificate = LiteCertificate::try_from_votes(votes)
1453            .ok_or_else(|| {
1454                chain_client::Error::InternalError(
1455                    "Vote values or rounds don't match; this is a bug",
1456                )
1457            })?
1458            .with_value(value)
1459            .ok_or_else(|| {
1460                chain_client::Error::ProtocolError("A quorum voted for an unexpected value")
1461            })?;
1462        Ok(certificate)
1463    }

1467    #[instrument(level = "trace", skip_all)]
1468    async fn receive_certificate_with_checked_signatures(
1469        &self,
1470        certificate: ConfirmedBlockCertificate,
1471        mode: ProcessConfirmedBlockMode,
1472    ) -> Result<(), chain_client::Error> {

1474        // Recover history from the network.
1475        self.download_certificates(block.header.chain_id, block.header.height)

1485    /// Processes the confirmed block in the local node. Whether it is fully
1486    /// executed or only preprocessed depends on the chain's [`ListeningMode`]:
1487    /// chains we follow get executed, untracked or events-only chains are only
1488    /// preprocessed.
1489    #[instrument(level = "trace", skip_all)]
1490    // Returns a boxed future rather than being an `async fn`: this function recurses
1491    // (admin-chain sync -> sender-certificate processing -> here), and boxing it at this
1492    // boundary lets the concurrent admin-chain self-heal below satisfy its future bound
1493    // despite the recursion. The future is `Send` off `web` (see the type alias).
1494    fn receive_sender_certificate(
1495        &self,
1496        certificate: CacheArc<ConfirmedBlockCertificate>,
1497        mode: ReceiveCertificateMode,
1498        nodes: Option<Vec<RemoteNode<Env::ValidatorNode>>>,
1499    ) -> ReceiveSenderCertificateFuture<'_> {
1500        Box::pin(async move {
1501            // Verify the certificate before doing any expensive networking.
1502            if let ReceiveCertificateMode::NeedsCheck = mode {
1503                let mut check_result = self.check_certificate(&certificate).await?;
1504                if matches!(check_result, CheckCertificateResult::FutureEpoch) {
1505                    // The certificate is from an epoch our local view of the admin chain
1506                    // hasn't caught up to yet. Catch up and check again instead of failing.
1507                    // Prefer the nodes that gave us the certificate: they evidently know
1508                    // the newer epoch even if our own committee view is stale or its
1509                    // members are unreachable. Race them concurrently rather than blocking
1510                    // on a slow one, then fall back to the known committee. A sync only
1511                    // counts if it actually made the epoch known.
1512                    let admin_chain_id = self.admin_chain_id;
1513                    let epoch = certificate.block().header.epoch;
1514                    info!(
1515                        %epoch,
1516                        "certificate is from an unknown epoch; synchronizing the admin chain"
1517                    );
1518                    let synced_from_serving_node = if let Some(nodes) = &nodes {
1519                        let certificate = &certificate;
1520                        communicate_concurrently(
1521                            nodes,
1522                            |node| {
1523                                Box::pin(async move {
1524                                    self.synchronize_chain_state_from(&node, admin_chain_id)
1525                                        .await?;
1526                                    match self.check_certificate(certificate).await? {
1527                                        CheckCertificateResult::FutureEpoch => {
1528                                            Err(chain_client::Error::CommitteeSynchronizationError)
1529                                        }
1530                                        _ => Ok(()),
1531                                    }
1532                                })
1533                            },
1534                            self.options.blob_download_hedge_delay,
1535                            self.storage_client().clock(),
1536                        )
1537                        .await
1538                        .is_ok()
1539                    } else {
1540                        false
1541                    };
1542                    if synced_from_serving_node {
1543                        check_result = self.check_certificate(&certificate).await?;
1544                    }
1545                    if matches!(check_result, CheckCertificateResult::FutureEpoch) {
1546                        Box::pin(self.synchronize_chain_state(admin_chain_id)).await?;
1547                        check_result = self.check_certificate(&certificate).await?;
1548                    }
1549                }
1550                check_result.into_result()?;
1551            }
1552            // Recover history from the network.
1553            let nodes = if let Some(nodes) = nodes {
1554                nodes
1555            } else {
1556                self.validator_nodes().await?
1557            };
1558            let processing_mode = if self
1559                .chain_mode(certificate.value().chain_id())
1560                .is_some_and(|m| m.should_sync_chain_state())
1561            {
1562                ProcessConfirmedBlockMode::Auto
1563            } else {
1564                ProcessConfirmedBlockMode::Preprocess
1565            };
1566            self.handle_certificate_with_retry(&certificate, &nodes, processing_mode)
1567                .await?;
1568
1569            Ok(())
1570        })
1571    }

1573    /// Downloads and processes certificates for sender chain blocks.
1574    #[instrument(level = "debug", skip_all, fields(chain_id = %sender_chain_id))]
1575    async fn download_and_process_sender_chain(
1576        &self,
1577        sender_chain_id: ChainId,
1578        nodes: &[RemoteNode<Env::ValidatorNode>],
1579        received_log: &ReceivedLogs,
1580        mut remote_heights: Vec<BlockHeight>,
1581        sender: mpsc::UnboundedSender<ChainAndHeight>,
1582    ) {
1583        let mut nodes = nodes.to_vec();
1584        while !remote_heights.is_empty() {
1585            // Check local storage first — certificates may already be available from
1586            // a prior sync cycle, another receiver chain, or a concurrent notification.
1587            if let Ok(local_certs) = self
1588                .storage_client()
1589                .read_certificates_by_heights(sender_chain_id, &remote_heights)
1590                .await
1591            {
1592                let mut still_needed = Vec::new();
1593                for (height, maybe_cert) in remote_heights.iter().copied().zip(local_certs) {
1594                    if let Some(certificate) = maybe_cert {
1595                        let chain_id = certificate.block().header.chain_id;
1596                        if let Err(error) = sender.send(ChainAndHeight { chain_id, height }) {
1597                            error!(
1598                                %chain_id, %height, %error,
1599                                "failed to send chain and height over the channel",
1600                            );
1601                        }
1602                    } else {
1603                        still_needed.push(height);
1604                    }
1605                }
1606                remote_heights = still_needed;
1607                if remote_heights.is_empty() {
1608                    break;
1609                }
1610            }
1611
1612            let remote_heights_ref = &remote_heights;
1613            let certificates = match communicate_concurrently(
1614                &nodes,
1615                async move |remote_node| {
1616                    let mut remote_heights = remote_heights_ref.clone();
1617                    // No need trying to download certificates the validator didn't have in their
1618                    // log - we'll retry downloading the remaining ones next time we loop.
1619                    remote_heights.retain(|height| {
1620                        received_log.validator_has_block(
1621                            &remote_node.public_key,
1622                            sender_chain_id,
1623                            *height,
1624                        )
1625                    });
1626                    if remote_heights.is_empty() {
1627                        // It makes no sense to return `Ok(_)` if we aren't going to try downloading
1628                        // anything from the validator - let the function try the other validators
1629                        return Err(NodeError::MissingCertificateValue);
1630                    }
1631                    let certificates = self
1632                        .requests_scheduler
1633                        .download_certificates_by_heights(
1634                            &remote_node,
1635                            sender_chain_id,
1636                            remote_heights,
1637                        )
1638                        .await?;
1639                    let mut certificates_with_check_results = vec![];
1640                    for cert in certificates {
1641                        let check_result = self.check_certificate(&cert).await?;
1642                        certificates_with_check_results
1643                            .push((cert, check_result.into_result().is_ok()));
1644                    }
1645                    Ok(certificates_with_check_results)
1646                },
1647                self.options.certificate_batch_download_hedge_delay,
1648                self.storage_client().clock(),
1649            )
1650            .await
1651            {
1652                Ok(certificates_with_check_results) => certificates_with_check_results,
1653                Err(errors) => {
1654                    let faulty_validators = errors
1655                        .into_iter()
1656                        .map(|(validator, error)| {
1657                            warn!(
1658                                %validator,
1659                                %sender_chain_id,
1660                                %error,
1661                                "failed to download certificates from validator",
1662                            );
1663                            validator
1664                        })
1665                        .collect::<BTreeSet<_>>();
1666                    // filter out faulty validators and retry if any are left
1667                    nodes.retain(|node| !faulty_validators.contains(&node.public_key));
1668                    if nodes.is_empty() {
1669                        info!(
1670                            chain_id = %sender_chain_id,
1671                            "could not download certificates for chain - no more correct validators left"
1672                        );
1673                        return;
1674                    }
1675                    continue;
1676                }
1677            };
1678
1679            trace!(
1680                num_certificates = %certificates.len(),
1681                "received certificates",
1682            );
1683
1684            let mut to_remove_from_queue = BTreeSet::new();
1685
1686            for (certificate, check_result) in certificates {
1687                let hash = certificate.hash();
1688                let chain_id = certificate.block().header.chain_id;
1689                let height = certificate.block().header.height;
1690                if !check_result {
1691                    // The certificate was correctly signed, but we were missing a committee to
1692                    // validate it properly - do not receive it, but also do not attempt to
1693                    // re-download it.
1694                    to_remove_from_queue.insert(height);
1695                    continue;
1696                }
1697                // We checked the certificates right after downloading them.
1698                let mode = ReceiveCertificateMode::AlreadyChecked;
1699                if let Err(error) = self
1700                    .receive_sender_certificate(
1701                        self.storage_client().cache_certificate(certificate),
1702                        mode,
1703                        None,
1704                    )
1705                    .await
1706                {
1707                    warn!(%error, %hash, "Received invalid certificate");
1708                } else {
1709                    to_remove_from_queue.insert(height);
1710                    if let Err(error) = sender.send(ChainAndHeight { chain_id, height }) {
1711                        error!(
1712                            %chain_id,
1713                            %height,
1714                            %error,
1715                            "failed to send chain and height over the channel",
1716                        );
1717                    }
1718                }
1719            }
1720
1721            remote_heights.retain(|height| !to_remove_from_queue.contains(height));
1722        }
1723        trace!("find_received_certificates: finished processing chain");
1724    }

1764    /// Downloads a specific sender block and recursively downloads any earlier blocks
1765    /// that also sent a message to our chain, based on `previous_message_blocks`.
1766    ///
1767    /// This ensures that we have all the sender blocks needed to preprocess the target block
1768    /// and put the messages to our chain into the outbox.
1769    async fn download_sender_block_with_sending_ancestors(
1770        &self,
1771        receiver_chain_id: ChainId,
1772        sender_chain_id: ChainId,
1773        height: BlockHeight,
1774        remote_node: &RemoteNode<Env::ValidatorNode>,
1775    ) -> Result<(), chain_client::Error> {
1776        let next_outbox_height = self
1777            .local_node
1778            .next_outbox_heights(&[sender_chain_id], receiver_chain_id)
1779            .await?
1780            .get(&sender_chain_id)
1781            .copied()
1782            .unwrap_or(BlockHeight::ZERO);
1783
1784        // Recursively collect all certificates we need, following
1785        // the chain of previous_message_blocks back to next_outbox_height.
1786        let mut certificates = BTreeMap::new();
1787        let mut current_height = height;
1788        // On the first iteration we only have a height; subsequent iterations
1789        // also carry the hash from `previous_message_blocks`.
1790        let mut current_hash: Option<CryptoHash> = None;
1791
1792        // Stop if we've reached the height we've already processed.
1793        while current_height >= next_outbox_height {
1794            // Try local storage first — avoids a validator round-trip when
1795            // the certificate was already downloaded by a prior sync cycle,
1796            // another receiver chain, or a concurrent notification handler.
1797            let certificate = if let Some(local) = self
1798                .try_read_local_certificate(sender_chain_id, current_height, current_hash)
1799                .await?
1800            {
1801                local
1802            } else {
1803                let downloaded = self
1804                    .requests_scheduler
1805                    .download_certificates_by_heights(
1806                        remote_node,
1807                        sender_chain_id,
1808                        vec![current_height],
1809                    )
1810                    .await?;
1811                let Some(certificate) = downloaded.into_iter().next() else {
1812                    return Err(chain_client::Error::CannotDownloadMissingSenderBlock {
1813                        chain_id: sender_chain_id,
1814                        height: current_height,
1815                    });
1816                };
1817                self.storage_client().cache_certificate(certificate)
1818            };
1819
1820            // Validate the certificate.
1821            self.check_certificate(&certificate).await?.into_result()?;
1822
1823            // Check if there's a previous message block to our chain.
1824            let block = certificate.block();
1825            let next = block
1826                .body
1827                .previous_message_blocks
1828                .get(&receiver_chain_id)
1829                .map(|(prev_hash, prev_height)| (*prev_hash, *prev_height));
1830
1831            // Store this certificate.
1832            certificates.insert(current_height, certificate);
1833
1834            if let Some((prev_hash, prev_height)) = next {
1835                // Continue with the previous block (now with its hash for local lookup).
1836                current_height = prev_height;
1837                current_hash = Some(prev_hash);
1838            } else {
1839                // No more dependencies.
1840                break;
1841            }
1842        }
1843
1844        if certificates.is_empty() {
1845            self.retry_pending_cross_chain_requests(sender_chain_id)
1846                .await?;
1847        }
1848
1849        // Process certificates in ascending block height order (BTreeMap keeps them sorted).
1850        for certificate in certificates.into_values() {
1851            self.receive_sender_certificate(
1852                certificate,
1853                ReceiveCertificateMode::AlreadyChecked,
1854                Some(vec![remote_node.clone()]),
1855            )
1856            .await?;
1857        }
1858
1859        Ok(())

1862    /// Downloads event-bearing blocks for the given streams by walking the
1863    /// `previous_event_blocks` linked list backwards from `height`, stopping when we
1864    /// reach blocks that are already executed locally or whose events we already track.
1865    async fn download_event_bearing_blocks(
1866        &self,
1867        publisher_chain_id: ChainId,
1868        initial_blocks: BTreeSet<(BlockHeight, CryptoHash)>,
1869        local_next_block_height: BlockHeight,
1870        subscribed_streams: &BTreeSet<StreamId>,
1871        remote_node: &RemoteNode<Env::ValidatorNode>,
1872    ) -> Result<(), chain_client::Error> {
1873        if initial_blocks.is_empty() {
1874            return Ok(());
1875        }
1876
1877        let mut certificates = BTreeMap::new();
1878        let mut blocks_to_fetch = initial_blocks;
1879        let next_expected_events = self
1880            .local_node
1881            .next_expected_events(
1882                publisher_chain_id,
1883                subscribed_streams.iter().cloned().collect(),
1884            )
1885            .await?;
1886
1887        while let Some((current_height, current_hash)) = blocks_to_fetch.pop_last() {
1888            if current_height < local_next_block_height {
1889                continue; // Already executed locally.
1890            }
1891            if certificates.contains_key(&current_height) {
1892                continue;
1893            }
1894
1895            let certificate = if let Some(certificate) =
1896                self.storage_client().read_certificate(current_hash).await?
1897            {
1898                certificate
1899            } else {
1900                let downloaded = self
1901                    .requests_scheduler
1902                    .download_certificates(remote_node, publisher_chain_id, current_height, 1)
1903                    .await?;
1904                let Some(certificate) = downloaded.into_iter().next() else {
1905                    tracing::debug!(
1906                        validator = remote_node.address(),
1907                        %publisher_chain_id,
1908                        height = %current_height,
1909                        "failed to download event publisher block"
1910                    );
1911                    continue;
1912                };
1913
1914                self.check_certificate(&certificate).await?.into_result()?;
1915
1916                self.storage_client().cache_certificate(certificate)
1917            };
1918
1919            let block = certificate.block();
1920            // Walk previous_event_blocks for subscribed streams.
1921            for stream_id in subscribed_streams {
1922                if let Some((prev_hash, prev_height)) =
1923                    block.body.previous_event_blocks.get(stream_id)
1924                {
1925                    if next_expected_events.get(stream_id).is_some_and(|index| {
1926                        block
1927                            .body
1928                            .events
1929                            .iter()
1930                            .flatten()
1931                            .find(|event| event.stream_id == *stream_id)
1932                            .is_some_and(|event| event.index == *index)
1933                    }) {
1934                        continue;
1935                    }
1936                    if !certificates.contains_key(prev_height) {
1937                        blocks_to_fetch.insert((*prev_height, *prev_hash));
1938                    }
1939                }
1940            }
1941
1942            certificates.insert(current_height, certificate);
1943        }
1944
1945        // Process in ascending height order.
1946        for certificate in certificates.into_values() {
1947            self.receive_sender_certificate(
1948                certificate,
1949                ReceiveCertificateMode::AlreadyChecked,
1950                Some(vec![remote_node.clone()]),
1951            )
1952            .await?;
1953        }
1954
1955        Ok(())
1956    }

2365    /// Attempts to execute the block locally. If any incoming message execution fails, that
2366    /// message is rejected and execution is retried, until the block accepts only messages
2367    /// that succeed.

2369    /// Attempts to execute the block locally with a specified policy for handling bundle failures.
2370    /// If any attempt to read a blob fails, the blob is downloaded and execution is retried.
2371    ///
2372    /// Returns the modified block (bundles may be rejected/removed based on the policy)
2373    /// and the execution result.
2374    #[instrument(level = "trace", skip(self, block))]
2375    async fn stage_block_execution(
2376        &self,
2377        block: ProposedBlock,
2378        round: Option<u32>,
2379        published_blobs: Vec<Blob>,
2380        policy: BundleExecutionPolicy,
2381    ) -> Result<(Block, ChainInfoResponse, HashSet<ChainId>), chain_client::Error> {
2382        let mut events = EventSetDownloader::new(self);
2383        loop {
2384            let result = self
2385                .local_node
2386                .stage_block_execution(
2387                    block.clone(),
2388                    round,
2389                    published_blobs.clone(),
2390                    policy.clone(),
2391                )
2392                .await;
2393            if let Err(LocalNodeError::BlobsNotFound(blob_ids)) = &result {
2394                let validators = self.validator_nodes().await?;
2395                self.update_local_node_with_blobs_from(blob_ids.clone(), &validators)
2396                    .await?;
2397                continue; // We found the missing blob: retry.
2398            }
2399            if let Err(LocalNodeError::EventsNotFound(event_ids)) = &result {
2400                if events.download_new(event_ids).await? {
2401                    continue; // We downloaded new publisher chain data: retry.
2402                }
2403                // All reported events were already downloaded; don't loop forever.
2404            }
2405            if let Ok((_, executed_block, _, _, _)) = &result {
2406                let hash = executed_block.hash();

2408                    chain_id: executed_block.header.chain_id,

2410                        height: executed_block.header.height,
2411                        hash,
2412                    },
2413                };
2414                self.notifier.notify(&[notification]);
2415            }
2416            let (
2417                _modified_block,
2418                executed_block,
2419                response,
2420                _resource_tracker,
2421                never_reject_origins,
2422            ) = result?;
2423            return Ok((executed_block, response, never_reject_origins));
2424        }

2440/// events the loop has already downloaded and the [`Client`] used to fetch new

2442/// `EventsNotFound` and continue the loop only if it returns `true`.

2445    downloaded: HashSet<EventId>,
2446}

2456    /// If any of `event_ids` haven't been downloaded yet, fetches the publisher
2457    /// certificates that contain them and returns `true`. Returns `false`
2458    /// (without downloading) if every reported event has already been
2459    /// downloaded — that prevents an infinite retry loop when the events are
2460    /// genuinely unavailable.
2461    pub(crate) async fn download_new(
2462        &mut self,
2463        event_ids: &[EventId],
2464    ) -> Result<bool, chain_client::Error> {
2465        let new_events = filter_new(event_ids, &self.downloaded);
2466        if new_events.is_empty() {
2467            return Ok(false);
2468        }
2469        Box::pin(self.client.download_certificates_for_events(&new_events)).await?;
2470        self.downloaded.extend(new_events);
2471        Ok(true)
2472    }

2482/// Races `operation` across peers with a hedged, **failure-responsive** fan-out, returning the
2483/// first `Ok` (or every error if all attempts fail).
2484///
2485/// `first_peer` is tried immediately. `next_peer` then supplies additional peers, each started
2486/// either *immediately* when an in-flight attempt fails (no point waiting once we know an attempt
2487/// failed), or after the in-flight attempt has run for `hedge_schedule(started)` without answering
2488/// — hedging against a slow peer by racing an extra request; the slow attempt is **not** cancelled.
2489/// `hedge_schedule` maps the number of attempts started so far to the delay before starting the
2490/// next one (e.g. `|k| delay * k` for a linearly-growing stagger). All sleeping is on `clock`, so
2491/// this runs in (possibly simulated) time; freezing the clock disables the slow-peer hedge while
2492/// still advancing on every failure.
2493pub(crate) async fn hedged_fan_out<Peer, T, Err, NextPeer, NextFut, Op, OpFut>(
2494    first_peer: Peer,
2495    mut next_peer: NextPeer,
2496    operation: Op,
2497    hedge_schedule: impl Fn(usize) -> Duration,
2498    // `Sync` so `clock.sleep_for(..)` yields a `Send` future, as required when this fan-out is
2499    // spawned (e.g. background certificate downloads). All storage clocks are `Sync`.
2500    clock: &(impl linera_storage::Clock + Sync),
2501) -> Result<T, Vec<Err>>
2502where
2503    NextPeer: FnMut() -> NextFut,
2504    NextFut: Future<Output = Option<Peer>>,
2505    Op: Fn(Peer) -> OpFut,
2506    OpFut: Future<Output = Result<T, Err>>,
2507{
2508    use futures::future::{select, Either};
2509
2510    let mut in_flight = FuturesUnordered::new();
2511    let mut errors = vec![];
2512    let mut started = 0usize;
2513    let arm = |started: usize| clock.sleep_for(hedge_schedule(started));
2514
2515    in_flight.push(operation(first_peer));
2516    started += 1;
2517    let mut hedge = arm(started);
2518
2519    loop {
2520        if in_flight.is_empty() {
2521            // Nothing running: start the next peer, or stop if there are none left.
2522            match next_peer().await {
2523                Some(peer) => {
2524                    in_flight.push(operation(peer));
2525                    started += 1;
2526                    hedge = arm(started);
2527                }
2528                None => return Err(errors),
2529            }
2530            continue;
2531        }
2532        match select(in_flight.next(), hedge).await {
2533            // An attempt succeeded.
2534            Either::Left((Some(Ok(value)), _)) => return Ok(value),
2535            // An attempt failed: try the next peer right away rather than waiting out the hedge.
2536            Either::Left((Some(Err(error)), pending_hedge)) => {
2537                errors.push(error);
2538                hedge = pending_hedge;
2539                if let Some(peer) = next_peer().await {
2540                    in_flight.push(operation(peer));
2541                    started += 1;
2542                    hedge = arm(started);
2543                }
2544            }
2545            // All in-flight attempts drained; the loop top decides what to do next.
2546            Either::Left((None, pending_hedge)) => hedge = pending_hedge,
2547            // An attempt is slow: hedge by starting another peer, if any remain.
2548            Either::Right(((), _)) => match next_peer().await {
2549                Some(peer) => {
2550                    in_flight.push(operation(peer));
2551                    started += 1;
2552                    hedge = arm(started);
2553                }
2554                None => break,
2555            },
2556        }
2557    }
2558
2559    // No more peers to start; just wait for the remaining in-flight attempts.
2560    while let Some(result) = in_flight.next().await {
2561        match result {
2562            Ok(value) => return Ok(value),
2563            Err(error) => errors.push(error),
2564        }
2565    }
2566    Err(errors)
2567}

2569/// Performs `f` on the validators with a hedged, staggered fan-out (see [`hedged_fan_out`]),
2570/// returning the first `Ok` result, or every `(validator, error)` pair if all of them fail.
2571///
2572/// The hedge before starting the n-th validator grows quadratically with `n`, so we stay
2573/// reluctant to fan out to many validators at once when one is merely slow.
2574async fn communicate_concurrently<A, E, F, R, V>(
2575    nodes: &[RemoteNode<A>],
2576    f: F,
2577    hedge_delay: Duration,
2578    clock: &(impl linera_storage::Clock + Sync),
2579) -> Result<V, Vec<(ValidatorPublicKey, E)>>
2580where
2581    F: Clone + FnOnce(RemoteNode<A>) -> R,
2582    RemoteNode<A>: Clone,
2583    R: Future<Output = Result<V, E>>,
2584{
2585    let mut nodes = nodes.to_vec();
2586    nodes.shuffle(&mut rand::thread_rng());
2587    let mut nodes = nodes.into_iter();
2588    let Some(first_peer) = nodes.next() else {
2589        return Err(vec![]);
2590    };
2591    hedged_fan_out(
2592        first_peer,
2593        move || std::future::ready(nodes.next()),
2594        |node: RemoteNode<A>| {
2595            let fun = f.clone();
2596            async move {
2597                let public_key = node.public_key;
2598                fun(node).await.map_err(|err| (public_key, err))
2599            }
2600        },
2601        |started| {
2602            let k = u32::try_from(started).unwrap_or(u32::MAX);
2603            hedge_delay.saturating_mul(k).saturating_mul(k)
2604        },
2605        clock,
2606    )
2607    .await
2608}

2662/// `ApplicationFormats` blob built from the BCS-encoded `Formats` description

2695                vm_runtime,
2696                formats_blob_hash,
2697            );
2698            (vec![evm_contract_blob], module_id)
2699        }
2700    };
2701    if let Some(blob) = formats_blob {
2702        blobs.push(blob);
2703    }
2704    (blobs, module_id)
