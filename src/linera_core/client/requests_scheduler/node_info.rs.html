6use super::scoring::ScoringWeights;
7use crate::{environment::Environment, remote_node::RemoteNode};

9/// Tracks performance metrics and request capacity for a validator node using
10/// Exponential Moving Averages (EMA) for adaptive scoring.
11///
12/// This struct wraps a `RemoteNode` with performance tracking that adapts quickly
13/// to changing network conditions. The scoring system uses EMAs to weight recent
14/// performance more heavily than historical data.
15#[derive(Debug, Clone)]
16pub(super) struct NodeInfo<Env: Environment> {
17    /// The underlying validator node connection
18    pub(super) node: RemoteNode<Env::ValidatorNode>,
19
20    /// Exponential Moving Average of latency in milliseconds
21    /// Adapts quickly to changes in response time
22    ema_latency_ms: f64,
23
24    /// Exponential Moving Average of success rate (0.0 to 1.0)
25    /// Tracks recent success/failure patterns
26    ema_success_rate: f64,
27
28    /// Total number of requests processed (for monitoring and cold-start handling)
29    total_requests: u64,
30
31    /// Configuration for scoring weights
32    weights: ScoringWeights,
33
34    /// EMA smoothing factor (0 < alpha < 1)
35    /// Higher values give more weight to recent observations
36    alpha: f64,
37
38    /// Maximum expected latency in milliseconds for score normalization
39    max_expected_latency_ms: f64,
40}
41
42impl<Env: Environment> NodeInfo<Env> {
43    /// Creates a new `NodeInfo` with custom configuration.
44    pub(super) fn with_config(
45        node: RemoteNode<Env::ValidatorNode>,
46        weights: ScoringWeights,
47        alpha: f64,
48        max_expected_latency_ms: f64,
49    ) -> Self {
50        assert!(alpha > 0.0 && alpha < 1.0, "Alpha must be in (0, 1) range");
51        Self {
52            node,
53            ema_latency_ms: 100.0, // Start with reasonable latency expectation
54            ema_success_rate: 1.0, // Start optimistically with 100% success
55            total_requests: 0,
56            weights,
57            alpha,
58            max_expected_latency_ms,
59        }
60    }
61
62    /// Calculates a normalized performance score (0.0 to 1.0) using weighted metrics.
63    ///
64    /// The score combines three normalized components:
65    /// - **Latency score**: Inversely proportional to EMA latency
66    /// - **Success score**: Directly proportional to EMA success rate
67    /// - **Load score**: Inversely proportional to current load
68    ///
69    /// Returns a score from 0.0 to 1.0, where higher values indicate better performance.
70    pub(super) async fn calculate_score(&self) -> f64 {
71        // 1. Normalize Latency (lower is better, so we invert)
72        let latency_score = 1.0
73            - (self.ema_latency_ms.min(self.max_expected_latency_ms)
74                / self.max_expected_latency_ms);
75
76        // 2. Success Rate is already normalized [0, 1]
77        let success_score = self.ema_success_rate;
78
79        // 4. Apply cold-start penalty for nodes with very few requests
80        let confidence_factor = (self.total_requests as f64 / 10.0).min(1.0);
81
82        // 5. Combine with weights
83        let raw_score =
84            (self.weights.latency * latency_score) + (self.weights.success * success_score);
85
86        // Apply confidence factor to penalize nodes with too few samples
87        raw_score * (0.5 + 0.5 * confidence_factor)
88    }
89
90    /// Updates performance metrics using Exponential Moving Average.
91    ///
92    /// # Arguments
93    /// - `success`: Whether the request completed successfully
94    /// - `response_time_ms`: The request's response time in milliseconds
95    ///
96    /// Uses EMA formula: new_value = (alpha * observation) + ((1 - alpha) * old_value)
97    /// This gives more weight to recent observations while maintaining some history.
98    pub(super) fn update_metrics(&mut self, success: bool, response_time_ms: u64) {
99        let response_time_f64 = response_time_ms as f64;
100
101        // Update latency EMA
102        self.ema_latency_ms =
103            (self.alpha * response_time_f64) + ((1.0 - self.alpha) * self.ema_latency_ms);
104
105        // Update success rate EMA
106        let success_value = if success { 1.0 } else { 0.0 };
107        self.ema_success_rate =
108            (self.alpha * success_value) + ((1.0 - self.alpha) * self.ema_success_rate);
109
110        self.total_requests += 1;
111    }
112
113    /// Returns the current EMA success rate.
114    pub(super) fn ema_success_rate(&self) -> f64 {
115        self.ema_success_rate
116    }
117
118    /// Returns the total number of requests processed.
119    pub(super) fn total_requests(&self) -> u64 {
120        self.total_requests
121    }
122}
