15/// Tracks in-flight requests to deduplicate concurrent requests for the same data.
16///
17/// This structure manages a map of request keys to in-flight entries, each containing
18/// broadcast senders for notifying waiters when a request completes, as well as timing
19/// information and alternative data sources.
20#[derive(Debug, Clone)]
21pub(super) struct InFlightTracker<N> {
22    /// Maps request keys to in-flight entries containing broadcast senders and metadata
23    entries: Arc<tokio::sync::RwLock<HashMap<RequestKey, InFlightEntry<N>>>>,
24    /// Maximum duration before an in-flight request is considered stale and deduplication is skipped
25    timeout: Duration,
26}
27
28impl<N: Clone> InFlightTracker<N> {
29    /// Creates a new `InFlightTracker` with the specified timeout.
30    ///
31    /// # Arguments
32    /// - `timeout`: Maximum duration before an in-flight request is considered too old to deduplicate against
33    pub(super) fn new(timeout: Duration) -> Self {
34        Self {
35            entries: Arc::new(tokio::sync::RwLock::new(HashMap::new())),
36            timeout,
37        }
38    }
39
40    /// Attempts to subscribe to an existing in-flight request (exact or subsuming match).
41    ///
42    /// Searches for either an exact key match or a subsuming request (whose result would
43    /// contain all the data needed by this request). Returns information about which type
44    /// of match was found, along with subscription details.
45    ///
46    /// # Arguments
47    /// - `key`: The request key to look up
48    ///
49    /// # Returns
50    /// - `None`: No matching in-flight request found. Also returned if the found request is stale (exceeds timeout).
51    /// - `Some(InFlightMatch::Subsuming { key, outcome })`: Subsuming request found
52    pub(super) async fn try_subscribe(&self, key: &RequestKey) -> Option<InFlightMatch> {
53        let in_flight = self.entries.read().await;
54
55        if let Some(entry) = in_flight.get(key) {
56            let elapsed = Instant::now().duration_since(entry.started_at);
57
58            if elapsed <= self.timeout {
59                return Some(InFlightMatch::Exact(Subscribed(entry.sender.subscribe())));
60            }
61        }
62
63        // Sometimes a request key may not have the exact match but may be subsumed by a larger one.
64        for (in_flight_key, entry) in in_flight.iter() {
65            if in_flight_key.subsumes(key) {
66                let elapsed = Instant::now().duration_since(entry.started_at);
67
68                if elapsed <= self.timeout {
69                    return Some(InFlightMatch::Subsuming {
70                        key: in_flight_key.clone(),
71                        outcome: Subscribed(entry.sender.subscribe()),
72                    });
73                }
74            }
75        }
76
77        None
78    }
79
80    /// Inserts a new in-flight request entry.
81    ///
82    /// Creates a new broadcast channel and in-flight entry for the given key,
83    /// marking the start time as now.
84    ///
85    /// # Arguments
86    /// - `key`: The request key to insert
87    pub(super) async fn insert_new(&self, key: RequestKey) {
88        let (sender, _receiver) = broadcast::channel(1);
89        let mut in_flight = self.entries.write().await;
90
91        in_flight.insert(
92            key,
93            InFlightEntry {
94                sender,
95                started_at: Instant::now(),
96                alternative_peers: Arc::new(tokio::sync::RwLock::new(Vec::new())),
97            },
98        );
99    }
100
101    /// Completes an in-flight request by removing it and broadcasting the result.
102    ///
103    /// Removes the entry for the given key and broadcasts the result to all waiting
104    /// subscribers. Logs the number of waiters that received the notification.
105    ///
106    /// # Arguments
107    /// - `key`: The request key to complete
108    /// - `result`: The result to broadcast to waiters
109    pub(super) async fn complete_and_broadcast(
110        &self,
111        key: &RequestKey,
112        result: Arc<Result<RequestResult, NodeError>>,
113    ) -> usize {
114        let mut in_flight = self.entries.write().await;
115
116        if let Some(entry) = in_flight.remove(key) {
117            let waiter_count = entry.sender.receiver_count();
118            tracing::trace!(
119                key = ?key,
120                waiters = waiter_count,
121                "request completed; broadcasting result to waiters",
122            );
123            if waiter_count != 0 {
124                if let Err(err) = entry.sender.send(result) {
125                    tracing::warn!(
126                        key = ?key,
127                        error = ?err,
128                        "failed to broadcast result to waiters"
129                    );
130                }
131            }
132            return waiter_count;
133        }
134        0
135    }
136
137    /// Registers an alternative peer for an in-flight request.
138    ///
139    /// If an entry exists for the given key, registers the peer as an alternative source
140    /// (if not already registered).
141    ///
142    /// # Arguments
143    /// - `key`: The request key
144    /// - `peer`: The peer to register as an alternative
145    pub(super) async fn add_alternative_peer(&self, key: &RequestKey, peer: N)
146    where
147        N: PartialEq + Eq,
148    {
149        if let Some(entry) = self.entries.read().await.get(key) {
150            // Register this peer as an alternative source if not already present
151            {
152                let mut alt_peers = entry.alternative_peers.write().await;
153                if !alt_peers.contains(&peer) {
154                    alt_peers.push(peer);
155                }
156            }
157        }
158    }
159
160    /// Retrieves the list of alternative peers registered for an in-flight request.
161    ///
162    /// Returns a clone of the alternative peers list if an entry exists for the given key.
163    ///
164    /// # Arguments
165    /// - `key`: The request key to look up
166    ///
167    /// # Returns
168    /// - `Vec<N>`: List of alternative peers (empty if no entry exists)
169    pub(super) async fn get_alternative_peers(&self, key: &RequestKey) -> Option<Vec<N>> {
170        let in_flight = self.entries.read().await;
171
172        let entry = in_flight.get(key)?;
173        let peers = entry.alternative_peers.read().await;
174        Some(peers.clone())
175    }
176}
177
178/// Type of in-flight request match found.
179#[derive(Debug)]
180pub(super) enum InFlightMatch {
181    /// Exact key match found
182    Exact(Subscribed),
183    /// Subsuming key match found (larger request that contains this request)
184    Subsuming {
185        /// The key of the subsuming request
186        key: RequestKey,
187        /// Outcome of attempting to subscribe
188        outcome: Subscribed,
189    },
190}
191
192/// Outcome of attempting to subscribe to an in-flight request.
193/// Successfully subscribed; receiver will be notified when request completes
194#[derive(Debug)]
195pub(super) struct Subscribed(pub(super) broadcast::Receiver<Arc<Result<RequestResult, NodeError>>>);
196
197/// In-flight request entry that tracks when the request was initiated.
198#[derive(Debug)]
199pub(super) struct InFlightEntry<N> {
200    /// Broadcast sender for notifying waiters when the request completes
201    sender: broadcast::Sender<Arc<Result<RequestResult, NodeError>>>,
202    /// Time when this request was initiated
203    started_at: Instant,
204    /// Alternative peers that can provide this data if the primary request fails
205    alternative_peers: Arc<tokio::sync::RwLock<Vec<N>>>,
206}
