11/// Cached result entry with timestamp for TTL expiration
12#[derive(Debug, Clone)]
13pub(super) struct CacheEntry<R> {
14    result: Arc<R>,
15    cached_at: Instant,
16}
17
18/// Cache for request results with TTL-based expiration and LRU eviction.
19///
20/// This cache supports:
21/// - Exact match lookups
22/// - Subsumption-based lookups (larger requests can satisfy smaller ones)
23/// - TTL-based expiration
24/// - LRU eviction
25#[derive(Debug, Clone)]
26pub(super) struct RequestsCache<K, R> {
27    /// Cache of recently completed requests with their results and timestamps.
28    /// Used to avoid re-executing requests for the same data within the TTL window.
29    cache: Arc<tokio::sync::RwLock<HashMap<K, CacheEntry<R>>>>,
30    /// Time-to-live for cached entries. Entries older than this duration are considered expired.
31    cache_ttl: Duration,
32    /// Maximum number of entries to store in the cache. When exceeded, oldest entries are evicted (LRU).
33    max_cache_size: usize,
34}
35
36impl<K, R> RequestsCache<K, R>
37where
38    K: Eq + std::hash::Hash + std::fmt::Debug + Clone + SubsumingKey<R>,
39    R: Clone + std::fmt::Debug,
40{
41    /// Creates a new `RequestsCache` with the specified TTL and maximum size.
42    ///
43    /// # Arguments
44    /// - `cache_ttl`: Time-to-live for cached entries
45    /// - `max_cache_size`: Maximum number of entries in the cache
46    pub(super) fn new(cache_ttl: Duration, max_cache_size: usize) -> Self {
47        Self {
48            cache: Arc::new(tokio::sync::RwLock::new(HashMap::new())),
49            cache_ttl,
50            max_cache_size,
51        }
52    }
53
54    /// Attempts to retrieve a cached result for the given key.
55    ///
56    /// This method performs both exact match lookups and subsumption-based lookups.
57    /// If a larger request that contains all the data needed by this request is cached,
58    /// we can extract the subset result instead of making a new request.
59    ///
60    /// # Returns
61    /// - `Some(T)` if a cached result is found (either exact or subsumed)
62    /// - `None` if no suitable cached result exists
63    pub(super) async fn get<T>(&self, key: &K) -> Option<T>
64    where
65        T: TryFrom<R>,
66    {
67        let cache = self.cache.read().await;
68
69        // Check cache for exact match first
70        if let Some(entry) = cache.get(key) {
71            tracing::trace!(
72                key = ?key,
73                "cache hit (exact match) - returning cached result"
74            );
75            #[cfg(with_metrics)]
76            metrics::REQUEST_CACHE_HIT.inc();
77            return T::try_from((*entry.result).clone()).ok();
78        }
79
80        // Check cache for subsuming requests
81        for (cached_key, entry) in cache.iter() {
82            if cached_key.subsumes(key) {
83                if let Some(extracted) = key.try_extract_result(cached_key, &entry.result) {
84                    tracing::trace!(
85                        key = ?key,
86                        "cache hit (subsumption) - extracted result from larger cached request"
87                    );
88                    #[cfg(with_metrics)]
89                    metrics::REQUEST_CACHE_HIT.inc();
90                    return T::try_from(extracted).ok();
91                }
92            }
93        }
94
95        None
96    }
97
98    /// Stores a result in the cache with LRU eviction if cache is full.
99    ///
100    /// If the cache is at capacity, this method removes the oldest expired entries first.
101    /// Entries are considered "oldest" based on their cached_at timestamp.
102    ///
103    /// # Arguments
104    /// - `key`: The request key to cache
105    /// - `result`: The result to cache
106    pub(super) async fn store(&self, key: K, result: Arc<R>) {
107        self.evict_expired_entries().await; // Clean up expired entries first
108        let mut cache = self.cache.write().await;
109        // Insert new entry
110        cache.insert(
111            key.clone(),
112            CacheEntry {
113                result,
114                cached_at: Instant::now(),
115            },
116        );
117        tracing::trace!(
118            key = ?key,
119            "stored result in cache"
120        );
121    }
122
123    /// Removes all cache entries that are older than the configured cache TTL.
124    ///
125    /// This method scans the cache and removes entries where the time elapsed since
126    /// `cached_at` exceeds `cache_ttl`. It's useful for explicitly cleaning up stale
127    /// cache entries rather than relying on lazy expiration checks.
128    ///
129    /// # Returns
130    /// The number of entries that were evicted
131    async fn evict_expired_entries(&self) -> usize {
132        let mut cache = self.cache.write().await;
133        let now = Instant::now();
134        // Not strictly smaller b/c we want to add a new entry after eviction.
135        if cache.len() < self.max_cache_size {
136            return 0; // No need to evict if under max size
137        }
138        let mut expired_keys = 0usize;
139
140        cache.retain(|_key, entry| {
141            if now.duration_since(entry.cached_at) > self.cache_ttl {
142                expired_keys += 1;
143                false
144            } else {
145                true
146            }
147        });
148
149        if expired_keys > 0 {
150            tracing::trace!(count = expired_keys, "evicted expired cache entries");
151        }
152
153        expired_keys
154    }
155}

157/// Trait for request keys that support subsumption-based matching and result extraction.
158pub(super) trait SubsumingKey<R> {
159    /// Checks if this request fully subsumes another request.
160    ///
161    /// Request `self` subsumes request `other` if `self`'s result would contain all the data that
162    /// `other`'s result would contain. This means `other`'s request is redundant if `self` is already
163    /// in-flight or cached.
164    fn subsumes(&self, other: &Self) -> bool;
165
166    /// Attempts to extract a subset result for this request from a larger request's result.
167    ///
168    /// This is used when a request A subsumes this request B. We can extract B's result
169    /// from A's result by filtering the certificates to only those requested by B.
170    ///
171    /// # Arguments
172    /// - `from`: The key of the larger request that subsumes this one
173    /// - `result`: The result from the larger request
174    ///
175    /// # Returns
176    /// - `Some(result)` with the extracted subset if possible
177    /// - `None` if extraction is not possible (wrong variant, different chain, etc.)
178    fn try_extract_result(&self, from: &Self, result: &R) -> Option<R>;
179}

189    // Mock key type for testing: represents a range request [start, end]
190    #[derive(Debug, Clone, PartialEq, Eq, Hash)]
191    struct RangeKey {
192        start: u64,
193        end: u64,
194    }
195
196    // Mock result type: vector of values in the range
197    #[derive(Debug, Clone, PartialEq)]
198    struct RangeResult(Vec<u64>);
199
200    impl SubsumingKey<RangeResult> for RangeKey {
201        fn subsumes(&self, other: &Self) -> bool {
202            // This range subsumes another if it contains the other's range
203            self.start <= other.start && self.end >= other.end
204        }
205
206        fn try_extract_result(&self, from: &Self, result: &RangeResult) -> Option<RangeResult> {
207            if !from.subsumes(self) {
208                return None;
209            }
210            // Extract values that fall within our range
211            let filtered: Vec<u64> = result
212                .0
213                .iter()
214                .filter(|&&v| v >= self.start && v <= self.end)
215                .copied()
216                .collect();
217            Some(RangeResult(filtered))
218        }
219    }

296    #[tokio::test]
297    async fn test_eviction_when_exceeding_max_size() {
298        let cache_size = 3u64;
299        let cache = RequestsCache::new(Duration::from_millis(50), cache_size as usize);
300
301        // Fill cache to max size
302        for i in 0..cache_size {
303            let key = RangeKey {
304                start: i * 10,
305                end: i * 10,
306            };
307            cache.store(key, Arc::new(RangeResult(vec![i * 10]))).await;
308            // Small delay to ensure different timestamps
309            tokio::time::sleep(tokio::time::Duration::from_millis(5)).await;
310        }
311
312        // Wait for first entry to expire
313        tokio::time::sleep(tokio::time::Duration::from_millis(60)).await;
314
315        // Cache is now at max size (3) with expired entries, so next store triggers eviction.
316        let key_4 = RangeKey {
317            start: 100,
318            end: 100,
319        };
320        cache
321            .store(key_4.clone(), Arc::new(RangeResult(vec![100])))
322            .await;
323
324        let cache_guard = cache.cache.read().await;
325        // Expired entries should have been evicted
326        let first_key = RangeKey { start: 0, end: 5 };
327        assert!(!cache_guard.contains_key(&first_key));
328
329        // Latest entries should still be there
330        assert!(cache_guard.contains_key(&key_4));
331    }
332    #[tokio::test]
333    async fn test_subsumption_with_extraction_failure_tries_next() {
334        // Mock key that subsumes but extraction returns None
335        #[derive(Debug, Clone, PartialEq, Eq, Hash)]
336        struct FailingKey {
337            id: u64,
338            always_fail_extraction: bool,
339        }
340
341        #[derive(Debug, Clone, PartialEq)]
342        struct SimpleResult(u64);
343
344        impl SubsumingKey<SimpleResult> for FailingKey {
345            fn subsumes(&self, other: &Self) -> bool {
346                self.id >= other.id
347            }
348
349            fn try_extract_result(
350                &self,
351                from: &Self,
352                _result: &SimpleResult,
353            ) -> Option<SimpleResult> {
354                if from.always_fail_extraction {
355                    None
356                } else {
357                    Some(SimpleResult(self.id))
358                }
359            }
360        }
361
362        let cache = RequestsCache::<FailingKey, SimpleResult>::new(Duration::from_secs(60), 10);
363
364        // Store entry that subsumes but fails extraction
365        let failing_key = FailingKey {
366            id: 10,
367            always_fail_extraction: true,
368        };
369        cache.store(failing_key, Arc::new(SimpleResult(10))).await;
370
371        // Store entry that subsumes and succeeds extraction
372        let working_key = FailingKey {
373            id: 20,
374            always_fail_extraction: false,
375        };
376        cache.store(working_key, Arc::new(SimpleResult(20))).await;
377
378        // Request should find the working one
379        let target_key = FailingKey {
380            id: 5,
381            always_fail_extraction: false,
382        };
383        let retrieved: Option<SimpleResult> = cache.get(&target_key).await;
384
385        assert!(retrieved.is_some());
386    }
