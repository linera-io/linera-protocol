29/// A range of block heights as used in `ChainInfoQuery`.
30#[derive(Clone, Debug, Serialize, Deserialize)]
31#[cfg_attr(with_testing, derive(test_strategy::Arbitrary, Eq, PartialEq))]
32pub struct BlockHeightRange {
33    /// Starting point
34    pub start: BlockHeight,
35    /// Optional limit on the number of elements.
36    #[debug(skip_if = Option::is_none)]
37    pub limit: Option<u64>,
38}
39
40impl BlockHeightRange {
41    /// Creates a range containing only the single specified block height.
42    pub fn single(start: BlockHeight) -> BlockHeightRange {
43        let limit = Some(1);
44        BlockHeightRange { start, limit }
45    }
46
47    /// Creates a range starting at the specified block height and containing up to `limit` elements.
48    pub fn multi(start: BlockHeight, limit: u64) -> BlockHeightRange {
49        BlockHeightRange {
50            start,
51            limit: Some(limit),
52        }
53    }
54
55    /// Returns the highest block height in the range.
56    pub fn highest(&self) -> BlockHeight {
57        self.limit
58            .map_or(self.start, |limit| BlockHeight(self.start.0 + limit - 1))
59    }
60}

62/// Request information about a chain.
63#[derive(Clone, Debug, Serialize, Deserialize)]
64#[cfg_attr(with_testing, derive(test_strategy::Arbitrary, Eq, PartialEq))]
65pub struct ChainInfoQuery {
66    /// The chain ID.
67    pub chain_id: ChainId,
68    /// Optionally test that the block height is the one expected.
69    #[debug(skip_if = Option::is_none)]
70    pub test_next_block_height: Option<BlockHeight>,
71    /// Request the balance of a given [`AccountOwner`].
72    pub request_owner_balance: AccountOwner,
73    /// Query the current committees.
74    #[debug(skip_if = Not::not)]
75    pub request_committees: bool,
76    /// Query the received messages that are waiting to be picked in the next block.
77    #[debug(skip_if = Not::not)]
78    pub request_pending_message_bundles: bool,
79    /// Query a range of certificate hashes sent from the chain.
80    //  dev: this field is left and unused to maintain backwards compatibility
81    //  after hotfixing Testnet Conway.
82    #[debug(skip_if = Option::is_none)]
83    pub request_sent_certificate_hashes_in_range: Option<BlockHeightRange>,
84    /// Query new certificate sender chain IDs and block heights received from the chain.
85    #[debug(skip_if = Option::is_none)]
86    pub request_received_log_excluding_first_n: Option<u64>,
87    /// Query values from the chain manager, not just votes.
88    #[debug(skip_if = Not::not)]
89    pub request_manager_values: bool,
90    /// Include a timeout vote for the specified round, if appropriate.
91    #[debug(skip_if = Option::is_none)]
92    pub request_leader_timeout: Option<(BlockHeight, Round)>,
93    /// Include a vote to switch to fallback mode, if appropriate.
94    #[debug(skip_if = Not::not)]
95    pub request_fallback: bool,
96    /// Query for certificate hashes at block heights.
97    #[debug(skip_if = Vec::is_empty)]
98    pub request_sent_certificate_hashes_by_heights: Vec<BlockHeight>,
99    #[serde(default = "default_true")]
100    pub create_network_actions: bool,
101}

178#[derive(Clone, Debug, Serialize, Deserialize)]
179#[cfg_attr(with_testing, derive(Eq, PartialEq))]
180pub struct ChainInfo {
181    /// The chain ID.
182    pub chain_id: ChainId,
183    /// The number identifying the current configuration.
184    pub epoch: Epoch,
185    /// The chain description.
186    #[debug(skip_if = Option::is_none)]
187    pub description: Option<ChainDescription>,
188    /// The state of the chain authentication.
189    pub manager: Box<ChainManagerInfo>,
190    /// The current balance.
191    pub chain_balance: Amount,
192    /// The last block hash, if any.
193    #[debug(skip_if = Option::is_none)]
194    pub block_hash: Option<CryptoHash>,
195    /// The earliest possible timestamp for the next block.
196    pub timestamp: Timestamp,
197    /// The height after the latest block in the chain.
198    pub next_block_height: BlockHeight,
199    /// The hash of the current execution state.
200    #[debug(skip_if = Option::is_none)]
201    pub state_hash: Option<CryptoHash>,
202    /// The requested owner balance, if any.
203    #[debug(skip_if = Option::is_none)]
204    pub requested_owner_balance: Option<Amount>,
205    /// The current committees.
206    #[debug(skip_if = Option::is_none)]
207    pub requested_committees: Option<BTreeMap<Epoch, Committee>>,
208    /// The received messages that are waiting be picked in the next block (if requested).
209    #[debug(skip_if = Vec::is_empty)]
210    pub requested_pending_message_bundles: Vec<IncomingBundle>,
211    /// The response to `request_sent_certificate_hashes_in_range`
212    #[debug(skip_if = Vec::is_empty)]
213    pub requested_sent_certificate_hashes: Vec<CryptoHash>,
214    /// The current number of received certificates (useful for `request_received_log_excluding_first_n`)
215    pub count_received_log: usize,
216    /// The response to `request_received_certificates_excluding_first_n`
217    #[debug(skip_if = Vec::is_empty)]
218    pub requested_received_log: Vec<ChainAndHeight>,
219}

252/// An internal request between chains within a validator.
253#[derive(Clone, Debug, Serialize, Deserialize)]
254#[cfg_attr(with_testing, derive(Eq, PartialEq))]
255pub enum CrossChainRequest {
256    /// Communicate a number of confirmed blocks from the sender to the recipient.
257    /// Blocks must be given by increasing heights.
258    UpdateRecipient {
259        sender: ChainId,
260        recipient: ChainId,
261        bundles: Vec<(Epoch, MessageBundle)>,
262    },
263    /// Acknowledge the height of the highest confirmed blocks communicated with `UpdateRecipient`.
264    ConfirmUpdatedRecipient {
265        sender: ChainId,
266        recipient: ChainId,
267        latest_height: BlockHeight,
268    },
269}
270
271impl CrossChainRequest {
272    /// Where to send the cross-chain request.
273    pub fn target_chain_id(&self) -> ChainId {
274        use CrossChainRequest::*;
275        match self {
276            UpdateRecipient { recipient, .. } => *recipient,
277            ConfirmUpdatedRecipient { sender, .. } => *sender,
278        }
279    }
280
281    /// Returns true if the cross-chain request has messages lower or equal than `height`.
282    pub fn has_messages_lower_or_equal_than(&self, height: BlockHeight) -> bool {
283        match self {
284            CrossChainRequest::UpdateRecipient { bundles, .. } => {
285                debug_assert!(bundles.windows(2).all(|w| w[0].1.height <= w[1].1.height));
286                matches!(bundles.first(), Some((_, h)) if h.height <= height)
287            }
288            _ => false,
289        }
290    }
291}

293impl<C, S> From<&ChainStateView<C>> for ChainInfo
294where
295    C: Context<Extra = ChainRuntimeContext<S>> + Clone + 'static,
296    ChainRuntimeContext<S>: ExecutionRuntimeContext,
297{
298    fn from(view: &ChainStateView<C>) -> Self {
299        let system_state = &view.execution_state.system;
300        let tip_state = view.tip_state.get();
301        ChainInfo {
302            chain_id: view.chain_id(),
303            epoch: *system_state.epoch.get(),
304            description: system_state.description.get().clone(),
305            manager: Box::new(ChainManagerInfo::from(&view.manager)),
306            chain_balance: *system_state.balance.get(),
307            block_hash: tip_state.block_hash,
308            next_block_height: tip_state.next_block_height,
309            timestamp: *view.execution_state.system.timestamp.get(),
310            state_hash: *view.execution_state_hash.get(),
311            requested_committees: None,
312            requested_owner_balance: None,
313            requested_pending_message_bundles: Vec::new(),
314            requested_sent_certificate_hashes: Vec::new(),
315            count_received_log: view.received_log.count(),
316            requested_received_log: Vec::new(),
317        }
318    }
319}

321impl ChainInfoResponse {
322    pub fn new(info: impl Into<ChainInfo>, key_pair: Option<&ValidatorSecretKey>) -> Self {
323        let info = Box::new(info.into());
324        let signature = key_pair.map(|kp| ValidatorSignature::new(&*info, kp));
325        Self { info, signature }
326    }
327
328    /// Signs the [`ChainInfo`] stored inside this [`ChainInfoResponse`] with the provided
329    /// [`ValidatorSecretKey`].
330    pub fn sign(&mut self, key_pair: &ValidatorSecretKey) {
331        self.signature = Some(ValidatorSignature::new(&*self.info, key_pair));
332    }
333
334    pub fn check(&self, public_key: ValidatorPublicKey) -> Result<(), CryptoError> {
335        match self.signature.as_ref() {
336            Some(sig) => sig.check(&*self.info, public_key),
337            None => Err(CryptoError::MissingValidatorSignature),
338        }
339    }
340}

351/// The outcome of trying to commit a list of operations to the chain.
352#[derive(Debug)]
353pub enum ClientOutcome<T> {
354    /// The operations were committed successfully.
355    Committed(T),
356    /// We are not the round leader and cannot do anything. Try again at the specified time
357    /// or whenever the round or block height changes.
358    WaitForTimeout(RoundTimeout),
359    /// A different block was committed at the current block height.
360    Conflict(Box<ConfirmedBlockCertificate>),
361}
362
363#[derive(Debug)]
364pub struct RoundTimeout {
365    pub timestamp: Timestamp,
366    pub current_round: Round,
367    pub next_block_height: BlockHeight,
368}
369
370impl fmt::Display for RoundTimeout {
371    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
372        write!(
373            f,
374            "{} at height {} times out at {}",
375            self.current_round, self.next_block_height, self.timestamp
376        )
377    }
378}
379
380impl<T> ClientOutcome<T> {
381    #[cfg(with_testing)]
382    pub fn unwrap(self) -> T {
383        match self {
384            ClientOutcome::Committed(t) => t,
385            ClientOutcome::WaitForTimeout(timeout) => panic!("unexpected timeout: {timeout}"),
386            ClientOutcome::Conflict(certificate) => {
387                panic!("unexpected conflict: {}", certificate.hash())
388            }
389        }
390    }
391
392    pub fn expect(self, msg: &'static str) -> T {
393        match self {
394            ClientOutcome::Committed(t) => t,
395            ClientOutcome::WaitForTimeout(_) | ClientOutcome::Conflict(_) => panic!("{}", msg),
396        }
397    }
398
399    pub fn map<F, S>(self, f: F) -> ClientOutcome<S>
400    where
401        F: FnOnce(T) -> S,
402    {
403        match self {
404            ClientOutcome::Committed(t) => ClientOutcome::Committed(f(t)),
405            ClientOutcome::WaitForTimeout(timeout) => ClientOutcome::WaitForTimeout(timeout),
406            ClientOutcome::Conflict(certificate) => ClientOutcome::Conflict(certificate),
407        }
408    }
409
410    pub fn try_map<F, S>(self, f: F) -> Result<ClientOutcome<S>, chain_client::Error>
411    where
412        F: FnOnce(T) -> Result<S, chain_client::Error>,
413    {
414        match self {
415            ClientOutcome::Committed(t) => Ok(ClientOutcome::Committed(f(t)?)),
416            ClientOutcome::WaitForTimeout(timeout) => Ok(ClientOutcome::WaitForTimeout(timeout)),
417            ClientOutcome::Conflict(certificate) => Ok(ClientOutcome::Conflict(certificate)),
418        }
419    }
