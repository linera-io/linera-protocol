28/// A range of block heights as used in `ChainInfoQuery`.
29#[derive(Clone, Debug, Serialize, Deserialize)]
30#[cfg_attr(with_testing, derive(test_strategy::Arbitrary, Eq, PartialEq))]
31pub struct BlockHeightRange {
32    /// Starting point
33    pub start: BlockHeight,
34    /// Optional limit on the number of elements.
35    #[debug(skip_if = Option::is_none)]
36    pub limit: Option<u64>,
37}
38
39impl BlockHeightRange {
40    /// Creates a range containing only the single specified block height.
41    pub fn single(start: BlockHeight) -> BlockHeightRange {
42        let limit = Some(1);
43        BlockHeightRange { start, limit }
44    }
45
46    /// Creates a range starting at the specified block height and containing up to `limit` elements.
47    pub fn multi(start: BlockHeight, limit: u64) -> BlockHeightRange {
48        BlockHeightRange {
49            start,
50            limit: Some(limit),
51        }
52    }
53
54    /// Returns the highest block height in the range.
55    pub fn highest(&self) -> BlockHeight {
56        self.limit
57            .map_or(self.start, |limit| BlockHeight(self.start.0 + limit - 1))
58    }
59}

61/// Request information about a chain.
62#[derive(Clone, Debug, Serialize, Deserialize)]
63#[cfg_attr(with_testing, derive(test_strategy::Arbitrary, Eq, PartialEq))]
64pub struct ChainInfoQuery {
65    /// The chain ID.
66    pub chain_id: ChainId,
67    /// Optionally test that the block height is the one expected.
68    #[debug(skip_if = Option::is_none)]
69    pub test_next_block_height: Option<BlockHeight>,
70    /// Request the balance of a given [`AccountOwner`].
71    pub request_owner_balance: AccountOwner,
72    /// Query the current committees.
73    #[debug(skip_if = Not::not)]
74    pub request_committees: bool,
75    /// Query the received messages that are waiting to be picked in the next block.
76    #[debug(skip_if = Not::not)]
77    pub request_pending_message_bundles: bool,
78    /// Query a range of certificate hashes sent from the chain.
79    //  dev: this field is left and unused to maintain backwards compatibility
80    //  after hotfixing Testnet Conway.
81    #[debug(skip_if = Option::is_none)]
82    pub request_sent_certificate_hashes_in_range: Option<BlockHeightRange>,
83    /// Query new certificate sender chain IDs and block heights received from the chain.
84    #[debug(skip_if = Option::is_none)]
85    pub request_received_log_excluding_first_n: Option<u64>,
86    /// Query values from the chain manager, not just votes.
87    #[debug(skip_if = Not::not)]
88    pub request_manager_values: bool,
89    /// Include a timeout vote for the specified round, if appropriate.
90    #[debug(skip_if = Option::is_none)]
91    pub request_leader_timeout: Option<(BlockHeight, Round)>,
92    /// Include a vote to switch to fallback mode, if appropriate.
93    #[debug(skip_if = Not::not)]
94    pub request_fallback: bool,
95    /// Query for certificate hashes at block heights.
96    #[debug(skip_if = Vec::is_empty)]
97    pub request_sent_certificate_hashes_by_heights: Vec<BlockHeight>,
98    #[serde(default = "default_true")]
99    pub create_network_actions: bool,
100}
101

177#[derive(Clone, Debug, Serialize, Deserialize)]
178#[cfg_attr(with_testing, derive(Eq, PartialEq))]
179pub struct ChainInfo {
180    /// The chain ID.
181    pub chain_id: ChainId,
182    /// The number identifying the current configuration.
183    pub epoch: Epoch,
184    /// The chain description.
185    #[debug(skip_if = Option::is_none)]
186    pub description: Option<ChainDescription>,
187    /// The state of the chain authentication.
188    pub manager: Box<ChainManagerInfo>,
189    /// The current balance.
190    pub chain_balance: Amount,
191    /// The last block hash, if any.
192    #[debug(skip_if = Option::is_none)]
193    pub block_hash: Option<CryptoHash>,
194    /// The earliest possible timestamp for the next block.
195    pub timestamp: Timestamp,
196    /// The height after the latest block in the chain.
197    pub next_block_height: BlockHeight,
198    /// The hash of the current execution state.
199    #[debug(skip_if = Option::is_none)]
200    pub state_hash: Option<CryptoHash>,
201    /// The requested owner balance, if any.
202    #[debug(skip_if = Option::is_none)]
203    pub requested_owner_balance: Option<Amount>,
204    /// The current committees.
205    #[debug(skip_if = Option::is_none)]
206    pub requested_committees: Option<BTreeMap<Epoch, Committee>>,
207    /// The received messages that are waiting be picked in the next block (if requested).
208    #[debug(skip_if = Vec::is_empty)]
209    pub requested_pending_message_bundles: Vec<IncomingBundle>,
210    /// The response to `request_sent_certificate_hashes_in_range`
211    #[debug(skip_if = Vec::is_empty)]
212    pub requested_sent_certificate_hashes: Vec<CryptoHash>,
213    /// The current number of received certificates (useful for `request_received_log_excluding_first_n`)
214    pub count_received_log: usize,
215    /// The response to `request_received_certificates_excluding_first_n`
216    #[debug(skip_if = Vec::is_empty)]
217    pub requested_received_log: Vec<ChainAndHeight>,
218}

251/// An internal request between chains within a validator.
252#[derive(Clone, Debug, Serialize, Deserialize)]
253#[cfg_attr(with_testing, derive(Eq, PartialEq))]
254pub enum CrossChainRequest {
255    /// Communicate a number of confirmed blocks from the sender to the recipient.
256    /// Blocks must be given by increasing heights.
257    UpdateRecipient {
258        sender: ChainId,
259        recipient: ChainId,
260        bundles: Vec<(Epoch, MessageBundle)>,
261    },
262    /// Acknowledge the height of the highest confirmed blocks communicated with `UpdateRecipient`.
263    ConfirmUpdatedRecipient {
264        sender: ChainId,
265        recipient: ChainId,
266        latest_height: BlockHeight,
267    },
268}
269
270impl CrossChainRequest {
271    /// Where to send the cross-chain request.
272    pub fn target_chain_id(&self) -> ChainId {
273        use CrossChainRequest::*;
274        match self {
275            UpdateRecipient { recipient, .. } => *recipient,
276            ConfirmUpdatedRecipient { sender, .. } => *sender,
277        }
278    }
279
280    /// Returns true if the cross-chain request has messages lower or equal than `height`.
281    pub fn has_messages_lower_or_equal_than(&self, height: BlockHeight) -> bool {
282        match self {
283            CrossChainRequest::UpdateRecipient { bundles, .. } => {
284                debug_assert!(bundles.windows(2).all(|w| w[0].1.height <= w[1].1.height));
285                matches!(bundles.first(), Some((_, h)) if h.height <= height)
286            }
287            _ => false,
288        }
289    }
290}

292impl<C, S> From<&ChainStateView<C>> for ChainInfo
293where
294    C: Context<Extra = ChainRuntimeContext<S>> + Clone + Send + Sync + 'static,
295    ChainRuntimeContext<S>: ExecutionRuntimeContext,
296{
297    fn from(view: &ChainStateView<C>) -> Self {
298        let system_state = &view.execution_state.system;
299        let tip_state = view.tip_state.get();
300        ChainInfo {
301            chain_id: view.chain_id(),
302            epoch: *system_state.epoch.get(),
303            description: system_state.description.get().clone(),
304            manager: Box::new(ChainManagerInfo::from(&view.manager)),
305            chain_balance: *system_state.balance.get(),
306            block_hash: tip_state.block_hash,
307            next_block_height: tip_state.next_block_height,
308            timestamp: *view.execution_state.system.timestamp.get(),
309            state_hash: *view.execution_state_hash.get(),
310            requested_committees: None,
311            requested_owner_balance: None,
312            requested_pending_message_bundles: Vec::new(),
313            requested_sent_certificate_hashes: Vec::new(),
314            count_received_log: view.received_log.count(),
315            requested_received_log: Vec::new(),
316        }
317    }
318}

320impl ChainInfoResponse {
321    pub fn new(info: impl Into<ChainInfo>, key_pair: Option<&ValidatorSecretKey>) -> Self {
322        let info = Box::new(info.into());
323        let signature = key_pair.map(|kp| ValidatorSignature::new(&*info, kp));
324        Self { info, signature }
325    }
326
327    /// Signs the [`ChainInfo`] stored inside this [`ChainInfoResponse`] with the provided
328    /// [`ValidatorSecretKey`].
329    pub fn sign(&mut self, key_pair: &ValidatorSecretKey) {
330        self.signature = Some(ValidatorSignature::new(&*self.info, key_pair));
331    }
332
333    pub fn check(&self, public_key: ValidatorPublicKey) -> Result<(), CryptoError> {
334        match self.signature.as_ref() {
335            Some(sig) => sig.check(&*self.info, public_key),
336            None => Err(CryptoError::MissingValidatorSignature),
337        }
338    }
339}

350/// The outcome of trying to commit a list of operations to the chain.
351#[derive(Debug)]
352pub enum ClientOutcome<T> {
353    /// The operations were committed successfully.
354    Committed(T),
355    /// We are not the round leader and cannot do anything. Try again at the specified time
356    /// or whenever the round or block height changes.
357    WaitForTimeout(RoundTimeout),
358}
359
360#[derive(Debug)]
361pub struct RoundTimeout {
362    pub timestamp: Timestamp,
363    pub current_round: Round,
364    pub next_block_height: BlockHeight,
365}
366
367impl<T> ClientOutcome<T> {
368    #[cfg(with_testing)]
369    pub fn unwrap(self) -> T {
370        match self {
371            ClientOutcome::Committed(t) => t,
372            ClientOutcome::WaitForTimeout(timeout) => panic!("Unexpected timeout: {timeout:?}"),
373        }
374    }
375
376    pub fn expect(self, msg: &'static str) -> T {
377        match self {
378            ClientOutcome::Committed(t) => t,
379            ClientOutcome::WaitForTimeout(_) => panic!("{}", msg),
380        }
381    }
382
383    pub fn map<F, S>(self, f: F) -> ClientOutcome<S>
384    where
385        F: FnOnce(T) -> S,
386    {
387        match self {
388            ClientOutcome::Committed(t) => ClientOutcome::Committed(f(t)),
389            ClientOutcome::WaitForTimeout(timeout) => ClientOutcome::WaitForTimeout(timeout),
390        }
391    }
392
393    pub fn try_map<F, S>(self, f: F) -> Result<ClientOutcome<S>, chain_client::Error>
394    where
395        F: FnOnce(T) -> Result<S, chain_client::Error>,
396    {
397        match self {
398            ClientOutcome::Committed(t) => Ok(ClientOutcome::Committed(f(t)?)),
399            ClientOutcome::WaitForTimeout(timeout) => Ok(ClientOutcome::WaitForTimeout(timeout)),
400        }
401    }
