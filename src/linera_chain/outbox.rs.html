36// The number of block heights in a bucket
37// The `BlockHeight` has just 8 bytes so the size is constant.
38// This means that by choosing a size of 1000, we have a
39// reasonable size that will not create any memory issues.
40const BLOCK_HEIGHT_BUCKET_SIZE: usize = 1000;
41
42/// The state of an outbox
43/// * An outbox is used to send messages to another chain.
44/// * Internally, this is implemented as a FIFO queue of (increasing) block heights.
45///   Messages are contained in blocks, together with destination information, so currently
46///   we just send the certified blocks over and let the receivers figure out what were the
47///   messages for them.
48/// * When marking block heights as received, messages at lower heights are also marked (i.e. dequeued).
49#[cfg_attr(with_graphql, derive(async_graphql::SimpleObject))]
50#[derive(Debug, ClonableView, View)]
51pub struct OutboxStateView<C>
52where
53    C: Context + Send + Sync + 'static,
54{
55    /// The minimum block height accepted in the future.
56    pub next_height_to_schedule: RegisterView<C, BlockHeight>,
57    /// Keep sending these certified blocks of ours until they are acknowledged by
58    /// receivers.
59    pub queue: BucketQueueView<C, BlockHeight, BLOCK_HEIGHT_BUCKET_SIZE>,
60}
61
62impl<C> OutboxStateView<C>
63where
64    C: Context + Clone + Send + Sync + 'static,
65{
66    /// Schedules a message at the given height if we haven't already.
67    /// Returns true if a change was made.
68    pub(crate) fn schedule_message(
69        &mut self,
70        height: BlockHeight,
71    ) -> Result<bool, ArithmeticError> {
72        if height < *self.next_height_to_schedule.get() {
73            return Ok(false);
74        }
75        self.next_height_to_schedule.set(height.try_add_one()?);
76        self.queue.push_back(height);
77        #[cfg(with_metrics)]
78        metrics::OUTBOX_SIZE
79            .with_label_values(&[])
80            .observe(self.queue.count() as f64);
81        Ok(true)
82    }
83
84    /// Marks all messages as received up to the given height.
85    /// Returns true if a change was made.
86    pub(crate) async fn mark_messages_as_received(
87        &mut self,
88        height: BlockHeight,
89    ) -> Result<Vec<BlockHeight>, ViewError> {
90        let mut updates = Vec::new();
91        while let Some(h) = self.queue.front().copied() {
92            if h > height {
93                break;
94            }
95            self.queue.delete_front().await?;
96            updates.push(h);
97        }
98        #[cfg(with_metrics)]
99        metrics::OUTBOX_SIZE
100            .with_label_values(&[])
101            .observe(self.queue.count() as f64);
102        Ok(updates)
103    }
104}
