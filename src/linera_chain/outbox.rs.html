37/// The state of an outbox
38/// * An outbox is used to send messages to another chain.
39/// * Internally, this is implemented as a FIFO queue of (increasing) block heights.
40///   Messages are contained in blocks, together with destination information, so currently
41///   we just send the certified blocks over and let the receivers figure out what were the
42///   messages for them.
43/// * When marking block heights as received, messages at lower heights are also marked (i.e. dequeued).
44#[cfg_attr(with_graphql, derive(async_graphql::SimpleObject))]
45#[derive(Debug, ClonableView, View, Allocative)]
46#[allocative(bound = "C")]
47pub struct OutboxStateView<C>
48where
49    C: Context + 'static,
50{
51    /// The minimum block height accepted in the future.
52    pub next_height_to_schedule: RegisterView<C, BlockHeight>,
53    /// Keep sending these certified blocks of ours until they are acknowledged by
54    /// receivers.
55    pub queue: QueueView<C, BlockHeight>,
56}
57
58impl<C> OutboxStateView<C>
59where
60    C: Context + Clone + 'static,
61{
62    /// Schedules a message at the given height if we haven't already.
63    /// Returns true if a change was made.
64    pub(crate) fn schedule_message(
65        &mut self,
66        height: BlockHeight,
67    ) -> Result<bool, ArithmeticError> {
68        if height < *self.next_height_to_schedule.get() {
69            return Ok(false);
70        }
71        self.next_height_to_schedule.set(height.try_add_one()?);
72        self.queue.push_back(height);
73        #[cfg(with_metrics)]
74        metrics::OUTBOX_SIZE
75            .with_label_values(&[])
76            .observe(self.queue.count() as f64);
77        Ok(true)
78    }
79
80    /// Marks all messages as received up to the given height.
81    /// Returns the heights that were newly marked as received.
82    pub(crate) async fn mark_messages_as_received(
83        &mut self,
84        height: BlockHeight,
85    ) -> Result<Vec<BlockHeight>, ViewError> {
86        let mut updates = Vec::new();
87        while let Some(h) = self.queue.front().await? {
88            if h > height {
89                break;
90            }
91            self.queue.delete_front();
92            updates.push(h);
93        }
94        #[cfg(with_metrics)]
95        metrics::OUTBOX_SIZE
96            .with_label_values(&[])
97            .observe(self.queue.count() as f64);
98        Ok(updates)
99    }
100}
