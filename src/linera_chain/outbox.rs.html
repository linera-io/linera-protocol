39/// The state of an outbox
40/// * An outbox is used to send messages to another chain.
41/// * Internally, this is implemented as a FIFO queue of (increasing) block heights.
42///   Messages are contained in blocks, together with destination information, so currently
43///   we just send the certified blocks over and let the receivers figure out what were the
44///   messages for them.
45/// * When marking block heights as received, messages at lower heights are also marked (i.e. dequeued).
46#[cfg_attr(with_graphql, derive(async_graphql::SimpleObject))]
47#[derive(Debug, ClonableView, View, Allocative)]
48#[allocative(bound = "C")]
49pub struct OutboxStateView<C>
50where
51    C: Context + 'static,
52{
53    /// The minimum block height accepted in the future.
54    pub next_height_to_schedule: RegisterView<C, BlockHeight>,
55    /// Keep sending these certified blocks of ours until they are acknowledged by
56    /// receivers.
57    pub queue: QueueView<C, BlockHeight>,
58}
59
60impl<C> OutboxStateView<C>
61where
62    C: Context + Clone + 'static,
63{
64    /// Schedules a message at the given height if we haven't already.
65    /// Returns true if a change was made.
66    pub(crate) fn schedule_message(
67        &mut self,
68        height: BlockHeight,
69    ) -> Result<bool, ArithmeticError> {
70        if height < *self.next_height_to_schedule.get() {
71            return Ok(false);
72        }
73        self.next_height_to_schedule.set(height.try_add_one()?);
74        self.queue.push_back(height);
75        Ok(true)
76    }
77
78    /// Re-adds heights to the outbox queue by merging them with existing entries.
79    /// Used by the `RevertConfirm` mechanism to undo a previous confirmation.
80    /// Returns the heights that were newly added.
81    pub async fn revert(
82        &mut self,
83        heights_to_add: &[BlockHeight],
84    ) -> Result<Vec<BlockHeight>, ViewError> {
85        let existing = self.queue.elements().await?;
86        let mut all_heights = existing.iter().copied().collect::<BTreeSet<_>>();
87        let new_heights = heights_to_add
88            .iter()
89            .filter(|h| !all_heights.contains(h))
90            .copied()
91            .collect::<Vec<_>>();
92        if new_heights.is_empty() {
93            return Ok(Vec::new());
94        }
95        all_heights.extend(new_heights.iter().copied());
96        self.clear();
97        for h in &all_heights {
98            self.schedule_message(*h).map_err(ViewError::from)?;
99        }
100        Ok(new_heights)
101    }
102
103    /// Marks all messages as received up to the given height.
104    /// Returns the heights that were newly marked as received.
105    pub(crate) async fn mark_messages_as_received(
106        &mut self,
107        height: BlockHeight,
108    ) -> Result<Vec<BlockHeight>, ViewError> {
109        let mut updates = Vec::new();
110        while let Some(h) = self.queue.front().await? {
111            if h > height {
112                break;
113            }
114            self.queue.delete_front();
115            updates.push(h);
116        }
117        #[cfg(with_metrics)]
118        metrics::OUTBOX_SIZE
119            .with_label_values(&[])
120            .observe(self.queue.count() as f64);
121        Ok(updates)
122    }
123}
