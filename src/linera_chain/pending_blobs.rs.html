21/// The pending blobs belonging to a block that can't be processed without them.
22#[cfg_attr(with_graphql, derive(async_graphql::SimpleObject))]
23#[derive(Debug, View, ClonableView)]
24pub struct PendingBlobsView<C>
25where
26    C: Clone + Context + Send + Sync + 'static,
27{
28    /// The round in which the block is validated.
29    pub round: RegisterView<C, Round>,
30    /// Whether these blobs were already validated.
31    ///
32    /// This is only `false` for _new_ block proposals, not when re-proposing blocks from earlier
33    /// rounds or when handling validated block certificates. If it is false, the pending blobs are
34    /// only the ones published by the new block, not the ones that are only read.
35    pub validated: RegisterView<C, bool>,
36    /// The map of blobs needed to process the block.
37    pub pending_blobs: MapView<C, BlobId, Option<Blob>>,
38}
39
40impl<C> PendingBlobsView<C>
41where
42    C: Clone + Context + Send + Sync + 'static,
43{
44    pub async fn get(&self, blob_id: &BlobId) -> Result<Option<Blob>, ViewError> {
45        Ok(self.pending_blobs.get(blob_id).await?.flatten())
46    }
47
48    /// Inserts the blob. Returns whether the blob was required by the pending block.
49    pub async fn maybe_insert(&mut self, blob: &Blob) -> Result<bool, ViewError> {
50        let blob_id = blob.id();
51        let Some(maybe_blob) = self.pending_blobs.get_mut(&blob_id).await? else {
52            return Ok(false);
53        };
54        if maybe_blob.is_none() {
55            *maybe_blob = Some(blob.clone());
56        }
57        Ok(true)
58    }
59
60    pub fn update(
61        &mut self,
62        round: Round,
63        validated: bool,
64        maybe_blobs: BTreeMap<BlobId, Option<Blob>>,
65    ) -> Result<(), ChainError> {
66        let existing_round = *self.round.get();
67        ensure!(
68            existing_round <= round,
69            ChainError::InsufficientRound(existing_round)
70        );
71        if existing_round < round {
72            self.clear();
73            self.round.set(round);
74            self.validated.set(validated);
75        }
76        for (blob_id, maybe_blob) in maybe_blobs {
77            self.pending_blobs.insert(&blob_id, maybe_blob)?;
78        }
79        Ok(())
80    }
81}
