22/// The pending blobs belonging to a block that can't be processed without them.
23#[cfg_attr(with_graphql, derive(async_graphql::SimpleObject))]
24#[derive(Debug, View, ClonableView, Allocative)]
25#[allocative(bound = "C")]
26pub struct PendingBlobsView<C>
27where
28    C: Clone + Context,
29{
30    /// The round in which the block is validated.
31    pub round: RegisterView<C, Round>,
32    /// Whether these blobs were already validated.
33    ///
34    /// This is only `false` for _new_ block proposals, not when re-proposing blocks from earlier
35    /// rounds or when handling validated block certificates. If it is false, the pending blobs are
36    /// only the ones published by the new block, not the ones that are only read.
37    pub validated: RegisterView<C, bool>,
38    /// The map of blobs needed to process the block.
39    pub pending_blobs: MapView<C, BlobId, Option<Blob>>,
40}
41
42impl<C> PendingBlobsView<C>
43where
44    C: Clone + Context,
45{
46    pub async fn multi_get(&self, blob_ids: &[BlobId]) -> Result<Vec<Option<Blob>>, ViewError> {
47        Ok(self
48            .pending_blobs
49            .multi_get(blob_ids)
50            .await?
51            .into_iter()
52            .map(|x| x.flatten())
53            .collect())
54    }
55
56    pub async fn get(&self, blob_id: &BlobId) -> Result<Option<Blob>, ViewError> {
57        Ok(self.pending_blobs.get(blob_id).await?.flatten())
58    }
59
60    /// Inserts the blob. Returns whether the blob was required by the pending block.
61    pub async fn maybe_insert(&mut self, blob: &Blob) -> Result<bool, ViewError> {
62        let blob_id = blob.id();
63        let Some(maybe_blob) = self.pending_blobs.get_mut(&blob_id).await? else {
64            return Ok(false);
65        };
66        if maybe_blob.is_none() {
67            *maybe_blob = Some(blob.clone());
68        }
69        Ok(true)
70    }
71
72    pub fn update(
73        &mut self,
74        round: Round,
75        validated: bool,
76        maybe_blobs: BTreeMap<BlobId, Option<Blob>>,
77    ) -> Result<(), ChainError> {
78        let existing_round = *self.round.get();
79        ensure!(
80            existing_round <= round,
81            ChainError::InsufficientRound(existing_round)
82        );
83        if existing_round < round {
84            self.clear();
85            self.round.set(round);
86            self.validated.set(validated);
87        }
88        for (blob_id, maybe_blob) in maybe_blobs {
89            self.pending_blobs.insert(&blob_id, maybe_blob)?;
90        }
91        Ok(())
92    }
93}
