32/// Tracks execution of transactions within a block.
33/// Captures the resource policy, produced messages, oracle responses and events.
34#[derive(Debug)]
35pub struct BlockExecutionTracker<'resources, 'blobs> {
36    chain_id: ChainId,
37    block_height: BlockHeight,
38    timestamp: Timestamp,
39    authenticated_owner: Option<AccountOwner>,
40    resource_controller: &'resources mut ResourceController<Option<AccountOwner>, ResourceTracker>,
41    local_time: Timestamp,
42    #[debug(skip_if = Option::is_none)]
43    replaying_oracle_responses: Option<Vec<Vec<OracleResponse>>>,
44    next_application_index: u32,
45    next_chain_index: u32,
46    #[debug(skip_if = Vec::is_empty)]
47    oracle_responses: Vec<Vec<OracleResponse>>,
48    #[debug(skip_if = Vec::is_empty)]
49    events: Vec<Vec<Event>>,
50    #[debug(skip_if = Vec::is_empty)]
51    blobs: Vec<Vec<Blob>>,
52    #[debug(skip_if = Vec::is_empty)]
53    messages: Vec<Vec<OutgoingMessage>>,
54    #[debug(skip_if = Vec::is_empty)]
55    operation_results: Vec<OperationResult>,
56    // Index of the currently executed transaction in a block.
57    transaction_index: u32,
58
59    // Blobs published in the block.
60    published_blobs: BTreeMap<BlobId, &'blobs Blob>,
61
62    // We expect the number of outcomes to be equal to the number of transactions in the block.
63    expected_outcomes_count: usize,
64}
65
66impl<'resources, 'blobs> BlockExecutionTracker<'resources, 'blobs> {
67    /// Creates a new BlockExecutionTracker.
68    pub fn new(
69        resource_controller: &'resources mut ResourceController<
70            Option<AccountOwner>,
71            ResourceTracker,
72        >,
73        published_blobs: BTreeMap<BlobId, &'blobs Blob>,
74        local_time: Timestamp,
75        replaying_oracle_responses: Option<Vec<Vec<OracleResponse>>>,
76        proposal: &ProposedBlock,
77    ) -> Result<Self, ChainError> {
78        resource_controller
79            .track_block_size(EMPTY_BLOCK_SIZE)
80            .with_execution_context(ChainExecutionContext::Block)?;
81
82        Ok(Self {
83            chain_id: proposal.chain_id,
84            block_height: proposal.height,
85            timestamp: proposal.timestamp,
86            authenticated_owner: proposal.authenticated_owner,
87            resource_controller,
88            local_time,
89            replaying_oracle_responses,
90            next_application_index: 0,
91            next_chain_index: 0,
92            oracle_responses: Vec::new(),
93            events: Vec::new(),
94            blobs: Vec::new(),
95            messages: Vec::new(),
96            operation_results: Vec::new(),
97            transaction_index: 0,
98            published_blobs,
99            expected_outcomes_count: proposal.transactions.len(),
100        })
101    }

103    /// Executes a transaction in the context of the block.
104    #[instrument(skip_all, fields(
105        chain_id = %self.chain_id,
106        block_height = %self.block_height,
107    ))]
108    pub async fn execute_transaction<C>(
109        &mut self,
110        transaction: &Transaction,
111        round: Option<u32>,
112        chain: &mut ExecutionStateView<C>,
113    ) -> Result<(), ChainError>
114    where
115        C: Context + Clone + Send + Sync + 'static,
116        C::Extra: ExecutionRuntimeContext,
117    {
118        let chain_execution_context = self.chain_execution_context(transaction);
119        let mut txn_tracker = self.new_transaction_tracker()?;
120
121        match transaction {
122            Transaction::ReceiveMessages(incoming_bundle) => {
123                self.resource_controller_mut()
124                    .track_block_size_of(&incoming_bundle)
125                    .with_execution_context(chain_execution_context)?;
126                for posted_message in incoming_bundle.messages() {
127                    Box::pin(self.execute_message_in_block(
128                        chain,
129                        posted_message,
130                        incoming_bundle,
131                        round,
132                        &mut txn_tracker,
133                    ))
134                    .await?;
135                }
136            }
137            Transaction::ExecuteOperation(operation) => {
138                self.resource_controller_mut()
139                    .with_state(&mut chain.system)
140                    .await?
141                    .track_block_size_of(&operation)
142                    .with_execution_context(chain_execution_context)?;
143                #[cfg(with_metrics)]
144                let _operation_latency = metrics::OPERATION_EXECUTION_LATENCY.measure_latency();
145                let context = OperationContext {
146                    chain_id: self.chain_id,
147                    height: self.block_height,
148                    round,
149                    authenticated_owner: self.authenticated_owner,
150                    timestamp: self.timestamp,
151                };
152                let mut actor =
153                    ExecutionStateActor::new(chain, &mut txn_tracker, self.resource_controller);
154                Box::pin(actor.execute_operation(context, operation.clone()))
155                    .await
156                    .with_execution_context(chain_execution_context)?;
157                self.resource_controller_mut()
158                    .with_state(&mut chain.system)
159                    .await?
160                    .track_operation(operation)
161                    .with_execution_context(chain_execution_context)?;
162            }
163        }
164
165        let txn_outcome = txn_tracker
166            .into_outcome()
167            .with_execution_context(chain_execution_context)?;
168        self.process_txn_outcome(txn_outcome, &mut chain.system, chain_execution_context)
169            .await?;
170        Ok(())
171    }

173    /// Returns a new TransactionTracker for the current transaction.
174    fn new_transaction_tracker(&mut self) -> Result<TransactionTracker, ChainError> {
175        Ok(TransactionTracker::new(
176            self.local_time,
177            self.transaction_index,
178            self.next_application_index,
179            self.next_chain_index,
180            self.oracle_responses()?,
181            &self.blobs,
182        ))
183    }

185    /// Executes a message as part of an incoming bundle in a block.
186    async fn execute_message_in_block<C>(
187        &mut self,
188        chain: &mut ExecutionStateView<C>,
189        posted_message: &PostedMessage,
190        incoming_bundle: &IncomingBundle,
191        round: Option<u32>,
192        txn_tracker: &mut TransactionTracker,
193    ) -> Result<(), ChainError>
194    where
195        C: Context + Clone + Send + Sync + 'static,
196        C::Extra: ExecutionRuntimeContext,
197    {
198        #[cfg(with_metrics)]
199        let _message_latency = metrics::MESSAGE_EXECUTION_LATENCY.measure_latency();
200        let context = MessageContext {
201            chain_id: self.chain_id,
202            origin: incoming_bundle.origin,
203            is_bouncing: posted_message.is_bouncing(),
204            height: self.block_height,
205            round,
206            authenticated_owner: posted_message.authenticated_owner,
207            refund_grant_to: posted_message.refund_grant_to,
208            timestamp: self.timestamp,
209        };
210        let mut grant = posted_message.grant;
211        match incoming_bundle.action {
212            MessageAction::Accept => {
213                let chain_execution_context =
214                    ChainExecutionContext::IncomingBundle(txn_tracker.transaction_index());
215                // Once a chain is closed, accepting incoming messages is not allowed.
216                ensure!(!chain.system.closed.get(), ChainError::ClosedChain);
217
218                let mut actor =
219                    ExecutionStateActor::new(chain, txn_tracker, self.resource_controller);
220                Box::pin(actor.execute_message(
221                    context,
222                    posted_message.message.clone(),
223                    (grant > Amount::ZERO).then_some(&mut grant),
224                ))
225                .await
226                .with_execution_context(chain_execution_context)?;
227                actor
228                    .send_refund(context, grant)
229                    .with_execution_context(chain_execution_context)?;
230            }
231            MessageAction::Reject => {
232                // If rejecting a message fails, the entire block proposal should be
233                // scrapped.
234                ensure!(
235                    !posted_message.is_protected() || *chain.system.closed.get(),
236                    ChainError::CannotRejectMessage {
237                        chain_id: self.chain_id,
238                        origin: incoming_bundle.origin,
239                        posted_message: Box::new(posted_message.clone()),
240                    }
241                );
242                let mut actor =
243                    ExecutionStateActor::new(chain, txn_tracker, self.resource_controller);
244                if posted_message.is_tracked() {
245                    // Bounce the message.
246                    actor
247                        .bounce_message(context, grant, posted_message.message.clone())
248                        .with_execution_context(ChainExecutionContext::Block)?;
249                } else {
250                    // Nothing to do except maybe refund the grant.
251                    actor
252                        .send_refund(context, grant)
253                        .with_execution_context(ChainExecutionContext::Block)?;
254                }
255            }
256        }
257        Ok(())
258    }

260    /// Returns oracle responses for the current transaction.
261    fn oracle_responses(&self) -> Result<Option<Vec<OracleResponse>>, ChainError> {
262        if let Some(responses) = self.replaying_oracle_responses.as_ref() {
263            match responses.get(self.transaction_index as usize) {
264                Some(responses) => Ok(Some(responses.clone())),
265                None => Err(ChainError::MissingOracleResponseList),
266            }
267        } else {
268            Ok(None)
269        }
270    }

272    /// Processes the transaction outcome.
273    ///
274    /// Updates block tracker with indexes for the next messages, applications, etc.
275    /// so that the execution of the next transaction doesn't overwrite the previous ones.
276    ///
277    /// Tracks the resources used by the transaction - size of the incoming and outgoing messages, blobs, etc.
278    pub async fn process_txn_outcome<C>(
279        &mut self,
280        txn_outcome: TransactionOutcome,
281        view: &mut SystemExecutionStateView<C>,
282        context: ChainExecutionContext,
283    ) -> Result<(), ChainError>
284    where
285        C: Context + Clone + Send + Sync + 'static,
286    {
287        let mut resource_controller = self.resource_controller.with_state(view).await?;
288
289        for message_out in &txn_outcome.outgoing_messages {
290            if message_out.kind == MessageKind::Bouncing {
291                continue; // Bouncing messages are free.
292            }
293            resource_controller
294                .track_message(&message_out.message)
295                .with_execution_context(context)?;
296        }
297
298        resource_controller
299            .track_block_size_of(&(
300                &txn_outcome.oracle_responses,
301                &txn_outcome.outgoing_messages,
302                &txn_outcome.events,
303                &txn_outcome.blobs,
304            ))
305            .with_execution_context(context)?;
306
307        // Account for blobs published by this transaction directly.
308        for blob in &txn_outcome.blobs {
309            resource_controller
310                .track_blob_published(blob)
311                .with_execution_context(context)?;
312        }
313
314        // Account for blobs published indirectly but referenced by the transaction.
315        for blob_id in &txn_outcome.blobs_published {
316            if let Some(blob) = self.published_blobs.get(blob_id) {
317                resource_controller
318                    .track_blob_published(blob)
319                    .with_execution_context(context)?;
320            } else {
321                return Err(ChainError::InternalError(format!(
322                    "Missing published blob {blob_id}"
323                )));
324            }
325        }
326
327        self.resource_controller
328            .track_block_size_of(&(&txn_outcome.operation_result))
329            .with_execution_context(context)?;
330
331        self.next_application_index = txn_outcome.next_application_index;
332        self.next_chain_index = txn_outcome.next_chain_index;
333        self.oracle_responses.push(txn_outcome.oracle_responses);
334        self.events.push(txn_outcome.events);
335        self.blobs.push(txn_outcome.blobs);
336        self.messages.push(txn_outcome.outgoing_messages);
337        if matches!(context, ChainExecutionContext::Operation(_)) {
338            self.operation_results
339                .push(OperationResult(txn_outcome.operation_result));
340        }
341        self.transaction_index += 1;
342        Ok(())
343    }

345    /// Returns recipient chain IDs for outgoing messages in the block.
346    pub fn recipients(&self) -> BTreeSet<ChainId> {
347        self.messages
348            .iter()
349            .flatten()
350            .map(|msg| msg.destination)
351            .collect()
352    }
353
354    /// Returns stream IDs for events published in the block.
355    pub fn event_streams(&self) -> BTreeSet<StreamId> {
356        self.events
357            .iter()
358            .flatten()
359            .map(|event| event.stream_id.clone())
360            .collect()
361    }

363    /// Returns a mutable reference to the resource controller.
364    pub fn resource_controller_mut(
365        &mut self,
366    ) -> &mut ResourceController<Option<AccountOwner>, ResourceTracker> {
367        self.resource_controller
368    }

370    /// Finalizes the execution and returns the collected results.
371    ///
372    /// This method should be called after all transactions have been processed.
373    /// Panics if the number of lists of oracle responses, outgoing messages,
374    /// events, or blobs does not match the expected counts.
375    pub fn finalize(self) -> FinalizeExecutionResult {
376        // Asserts that the number of outcomes matches the expected count.
377        assert_eq!(self.oracle_responses.len(), self.expected_outcomes_count);
378        assert_eq!(self.messages.len(), self.expected_outcomes_count);
379        assert_eq!(self.events.len(), self.expected_outcomes_count);
380        assert_eq!(self.blobs.len(), self.expected_outcomes_count);
381
382        #[cfg(with_metrics)]
383        crate::chain::metrics::track_block_metrics(&self.resource_controller.tracker);
384
385        (
386            self.messages,
387            self.oracle_responses,
388            self.events,
389            self.blobs,
390            self.operation_results,
391        )
392    }
393
394    /// Returns the execution context for the current transaction.
395    fn chain_execution_context(&self, transaction: &Transaction) -> ChainExecutionContext {
396        match transaction {
397            Transaction::ReceiveMessages(_) => {
398                ChainExecutionContext::IncomingBundle(self.transaction_index)
399            }
400            Transaction::ExecuteOperation(_) => {
401                ChainExecutionContext::Operation(self.transaction_index)
402            }
403        }
404    }
