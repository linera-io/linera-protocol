33/// Tracks execution of transactions within a block.
34/// Captures the resource policy, produced messages, oracle responses and events.
35#[derive(Debug)]
36pub struct BlockExecutionTracker<'resources, 'blobs> {
37    chain_id: ChainId,
38    /// The protocol phase this block is executed in (staging a proposal, validating a
39    /// received proposal, or executing a confirmed certificate). Recorded on execution
40    /// spans and used to label execution metrics.
41    phase: BlockExecutionPhase,
42    block_height: BlockHeight,
43    timestamp: Timestamp,
44    authenticated_owner: Option<AccountOwner>,
45    resource_controller: &'resources mut ResourceController<Option<AccountOwner>, ResourceTracker>,
46    local_time: Timestamp,
47    #[debug(skip_if = Option::is_none)]
48    replaying_oracle_responses: Option<Vec<Vec<OracleResponse>>>,
49    next_application_index: u32,
50    next_chain_index: u32,
51    #[debug(skip_if = Vec::is_empty)]
52    oracle_responses: Vec<Vec<OracleResponse>>,
53    #[debug(skip_if = Vec::is_empty)]
54    events: Vec<Vec<Event>>,
55    #[debug(skip_if = Vec::is_empty)]
56    blobs: Vec<Vec<Blob>>,
57    #[debug(skip_if = Vec::is_empty)]
58    messages: Vec<Vec<OutgoingMessage>>,
59    #[debug(skip_if = Vec::is_empty)]
60    operation_results: Vec<OperationResult>,
61    // Index of the currently executed transaction in a block.
62    transaction_index: u32,
63
64    // Blobs published in the block.
65    published_blobs: BTreeMap<BlobId, &'blobs Blob>,
66
67    // Checkpoint inputs computed pre-block (state dump split into blobs and the per-origin
68    // inbox cursors), to be handed to the matching `SystemOperation::Checkpoint`
69    // operation handler when it runs.
70    #[debug(skip_if = Option::is_none)]
71    prepared_checkpoint: Option<PreparedCheckpoint>,
72}

102            oracle_responses: Vec::new(),
103            events: Vec::new(),
104            blobs: Vec::new(),
105            messages: Vec::new(),
106            operation_results: Vec::new(),
107            transaction_index: 0,
108            published_blobs,
109            prepared_checkpoint: None,
110        })

119    /// Executes a transaction in the context of the block.
120    #[instrument(skip_all, fields(
121        chain_id = %self.chain_id,
122        block_height = %self.block_height,
123        phase = self.phase.as_str(),
124        transaction_index = %self.transaction_index,
125        transaction_type = %transaction.as_ref(),
126    ))]
127    pub async fn execute_transaction<C>(
128        &mut self,
129        transaction: &Transaction,
130        round: Option<u32>,
131        chain: &mut ExecutionStateView<C>,
132    ) -> Result<(), ChainError>
133    where
134        C: Context + Clone + 'static,
135        C::Extra: ExecutionRuntimeContext,
136    {
137        let chain_execution_context = self.chain_execution_context(transaction);
138        let mut txn_tracker = self.new_transaction_tracker()?;
139
140        match transaction {
141            Transaction::ReceiveMessages(incoming_bundle) => {
142                self.resource_controller_mut()
143                    .track_block_size_of(&incoming_bundle)
144                    .with_execution_context(chain_execution_context)?;
145                for posted_message in incoming_bundle.messages() {
146                    Box::pin(self.execute_message_in_block(
147                        chain,
148                        posted_message,
149                        incoming_bundle,
150                        round,
151                        &mut txn_tracker,
152                    ))
153                    .await?;
154                }
155                let progress = chain.system.progress.get_mut();
156                progress.num_incoming_bundles = progress
157                    .num_incoming_bundles
158                    .checked_add(1)
159                    .ok_or(ArithmeticError::Overflow)?;
160            }
161            Transaction::ExecuteOperation(operation) => {
162                self.resource_controller_mut()
163                    .with_state(&mut chain.system)
164                    .await?
165                    .track_block_size_of(&operation)
166                    .with_execution_context(chain_execution_context)?;
167                #[cfg(with_metrics)]
168                let operation_latency =
169                    metrics::OPERATION_EXECUTION_LATENCY.with_label_values(&[self.phase.as_str()]);
170                #[cfg(with_metrics)]
171                let _operation_latency = operation_latency.measure_latency_us();
172                let context = OperationContext {
173                    chain_id: self.chain_id,
174                    height: self.block_height,
175                    round,
176                    authenticated_owner: self.authenticated_owner,
177                    timestamp: self.timestamp,
178                };
179                let mut actor =
180                    ExecutionStateActor::new(chain, &mut txn_tracker, self.resource_controller);
181                Box::pin(actor.execute_operation(context, operation.clone()))
182                    .await
183                    .with_execution_context(chain_execution_context)?;
184                self.resource_controller_mut()
185                    .with_state(&mut chain.system)
186                    .await?
187                    .track_operation(operation)
188                    .with_execution_context(chain_execution_context)?;
189                let progress = chain.system.progress.get_mut();
190                progress.num_operations = progress
191                    .num_operations
192                    .checked_add(1)
193                    .ok_or(ArithmeticError::Overflow)?;
194            }
195        }
196
197        let txn_outcome = txn_tracker
198            .into_outcome()
199            .with_execution_context(chain_execution_context)?;
200        let progress = chain.system.progress.get_mut();
201        progress.num_outgoing_messages = progress
202            .num_outgoing_messages
203            .checked_add(
204                u32::try_from(txn_outcome.outgoing_messages.len())
205                    .map_err(|_| ArithmeticError::Overflow)?,
206            )
207            .ok_or(ArithmeticError::Overflow)?;
208        self.process_txn_outcome(txn_outcome, &mut chain.system, chain_execution_context)
209            .await?;
210        Ok(())
211    }

213    /// Returns a new TransactionTracker for the current transaction.
214    fn new_transaction_tracker(&self) -> Result<TransactionTracker, ChainError> {
215        let mut tracker = TransactionTracker::new(
216            self.local_time,
217            self.transaction_index,
218            self.next_application_index,
219            self.next_chain_index,
220            self.oracle_responses()?,
221            &self.blobs,
222        );
223        // Cloning is cheap — blob bytes are behind an `Arc`.
224        if let Some(prepared) = self.prepared_checkpoint.as_ref() {
225            tracker.set_prepared_checkpoint(prepared.clone());
226        }
227        Ok(tracker)
228    }

230    /// Executes a message as part of an incoming bundle in a block.
231    #[instrument(skip_all, fields(
232        chain_id = %self.chain_id,
233        block_height = %self.block_height,
234        origin = %incoming_bundle.origin,
235        action = ?incoming_bundle.action,
236        is_bouncing = %posted_message.is_bouncing(),
237    ))]
238    async fn execute_message_in_block<C>(
239        &mut self,
240        chain: &mut ExecutionStateView<C>,
241        posted_message: &PostedMessage,
242        incoming_bundle: &IncomingBundle,
243        round: Option<u32>,
244        txn_tracker: &mut TransactionTracker,
245    ) -> Result<(), ChainError>
246    where
247        C: Context + Clone + 'static,
248        C::Extra: ExecutionRuntimeContext,
249    {
250        #[cfg(with_metrics)]
251        let message_latency =
252            metrics::MESSAGE_EXECUTION_LATENCY.with_label_values(&[self.phase.as_str()]);
253        #[cfg(with_metrics)]
254        let _message_latency = message_latency.measure_latency_us();
255        let context = MessageContext {
256            chain_id: self.chain_id,
257            origin: incoming_bundle.origin,
258            origin_certificate_hash: incoming_bundle.bundle.certificate_hash,
259            origin_timestamp: incoming_bundle.bundle.timestamp,
260            is_bouncing: posted_message.is_bouncing(),
261            height: self.block_height,
262            round,
263            authenticated_owner: posted_message.authenticated_owner,
264            refund_grant_to: posted_message.refund_grant_to,
265            timestamp: self.timestamp,
266        };
267        let mut grant = posted_message.grant;
268        match incoming_bundle.action {
269            MessageAction::Accept => {
270                let chain_execution_context =
271                    ChainExecutionContext::IncomingBundle(txn_tracker.transaction_index());
272                // Once a chain is closed, accepting incoming messages is not allowed.
273                ensure!(!chain.system.closed.get(), ChainError::ClosedChain);
274
275                let mut actor =
276                    ExecutionStateActor::new(chain, txn_tracker, self.resource_controller);
277                Box::pin(actor.execute_message(
278                    context,
279                    posted_message.message.clone(),
280                    (grant > Amount::ZERO).then_some(&mut grant),
281                ))
282                .await
283                .with_execution_context(chain_execution_context)?;
284                actor
285                    .send_refund(context, grant)
286                    .with_execution_context(chain_execution_context)?;
287            }
288            MessageAction::Reject => {
289                // If rejecting a message fails, the entire block proposal should be
290                // scrapped.
291                ensure!(
292                    !posted_message.is_protected() || *chain.system.closed.get(),
293                    ChainError::CannotRejectMessage {
294                        chain_id: self.chain_id,
295                        origin: incoming_bundle.origin,
296                        posted_message: Box::new(posted_message.clone()),
297                    }
298                );
299                debug!(
300                    chain_id = %self.chain_id,
301                    origin = %incoming_bundle.origin,
302                    "Rejecting incoming message"
303                );
304                let mut actor =
305                    ExecutionStateActor::new(chain, txn_tracker, self.resource_controller);
306                if posted_message.is_tracked() {
307                    // Bounce the message.
308                    actor
309                        .bounce_message(context, grant, posted_message.message.clone())
310                        .with_execution_context(ChainExecutionContext::Block)?;
311                } else {
312                    // Nothing to do except maybe refund the grant.
313                    actor
314                        .send_refund(context, grant)
315                        .with_execution_context(ChainExecutionContext::Block)?;
316                }
317            }
318        }
319        // Record that this origin has sent us a real message; we owe them a
320        // `SystemMessage::CheckpointAck` at our next checkpoint. `CheckpointAck`
321        // messages themselves are excluded so they don't keep the notification
322        // ping-pong alive forever.
323        if !posted_message.message.is_checkpoint_ack() {
324            chain
325                .system
326                .pending_checkpoint_ack_targets
327                .insert(&incoming_bundle.origin)?;
328        }
329        Ok(())
330    }

332    /// Returns oracle responses for the current transaction.
333    fn oracle_responses(&self) -> Result<Option<Vec<OracleResponse>>, ChainError> {
334        if let Some(responses) = self.replaying_oracle_responses.as_ref() {
335            match responses.get(self.transaction_index as usize) {
336                Some(responses) => Ok(Some(responses.clone())),
337                None => Err(ChainError::MissingOracleResponseList),
338            }
339        } else {
340            Ok(None)
341        }

344    /// Processes the transaction outcome.
345    ///
346    /// Updates block tracker with indexes for the next messages, applications, etc.
347    /// so that the execution of the next transaction doesn't overwrite the previous ones.
348    ///
349    /// Tracks the resources used by the transaction - size of the incoming and outgoing messages, blobs, etc.
350    #[instrument(skip_all, fields(
351        chain_id = %self.chain_id,
352        block_height = %self.block_height,
353        transaction_index = %self.transaction_index,
354        outgoing_messages_count = %txn_outcome.outgoing_messages.len(),
355        events_count = %txn_outcome.events.len(),
356        blobs_count = %txn_outcome.blobs.len(),
357    ))]
358    pub async fn process_txn_outcome<C>(
359        &mut self,
360        txn_outcome: TransactionOutcome,
361        view: &mut SystemExecutionStateView<C>,
362        context: ChainExecutionContext,
363    ) -> Result<(), ChainError>
364    where
365        C: Context + Clone + 'static,
366    {
367        let mut resource_controller = self.resource_controller.with_state(view).await?;
368
369        for message_out in &txn_outcome.outgoing_messages {
370            if message_out.kind == MessageKind::Bouncing {
371                continue; // Bouncing messages are free.
372            }
373            if let Message::User { application_id, .. } = &message_out.message {
374                if resource_controller.policy().is_free_app(application_id) {
375                    continue; // Outgoing message fees are waived for free apps.
376                }
377            }
378            resource_controller
379                .track_message(&message_out.message)
380                .with_execution_context(context)?;
381        }
382
383        resource_controller
384            .track_block_size_of(&(
385                &txn_outcome.oracle_responses,
386                &txn_outcome.outgoing_messages,
387                &txn_outcome.events,
388                &txn_outcome.blobs,
389            ))
390            .with_execution_context(context)?;
391
392        // Account for blobs published by this transaction directly.
393        for blob in &txn_outcome.blobs {
394            if txn_outcome.free_blob_ids.contains(&blob.id()) {
395                continue; // Blob publishing fees are waived for free apps.
396            }
397            resource_controller
398                .track_blob_published(blob)
399                .with_execution_context(context)?;
400        }
401
402        // Account for blobs published indirectly but referenced by the transaction.
403        for blob_id in &txn_outcome.blobs_published {
404            if let Some(blob) = self.published_blobs.get(blob_id) {
405                resource_controller
406                    .track_blob_published(blob)
407                    .with_execution_context(context)?;
408            } else {
409                return Err(ChainError::InternalError(format!(
410                    "Missing published blob {blob_id}"
411                )));
412            }
413        }
414
415        self.resource_controller
416            .track_block_size_of(&(&txn_outcome.operation_result))
417            .with_execution_context(context)?;
418
419        self.next_application_index = txn_outcome.next_application_index;
420        self.next_chain_index = txn_outcome.next_chain_index;
421        self.oracle_responses.push(txn_outcome.oracle_responses);
422        self.events.push(txn_outcome.events);
423        self.blobs.push(txn_outcome.blobs);
424        self.messages.push(txn_outcome.outgoing_messages);
425        if matches!(context, ChainExecutionContext::Operation(_)) {
426            self.operation_results
427                .push(OperationResult(txn_outcome.operation_result));
428        }
429        self.transaction_index += 1;
430        Ok(())
431    }

433    /// Returns recipient chain IDs for outgoing messages in the block.
434    pub fn recipients(&self) -> BTreeSet<ChainId> {
435        self.messages
436            .iter()
437            .flatten()
438            .map(|msg| msg.destination)
439            .collect()
440    }

442    /// For each recipient that received at least one non-`SystemMessage::CheckpointAck`
443    /// message in this block, returns the transaction indices that produced such a
444    /// message. The chain-level bookkeeping (`previous_message_blocks`,
445    /// `unfinalized_message_blocks`) is updated only for these recipients, so that
446    /// `CheckpointAck`-only blocks don't pin a slot that would never get trimmed (the
447    /// recipient never acknowledges back). The transaction indices feed
448    /// `unfinalized_message_blocks` as full `(block.height, tx_index)` cursors, so a
449    /// later ack with a cursor past the last unfinalized bundle can fully evict the
450    /// recipient.
451    pub fn non_checkpoint_ack_tx_indices(&self) -> BTreeMap<ChainId, BTreeSet<u32>> {
452        let mut result = BTreeMap::<ChainId, BTreeSet<u32>>::new();
453        for (tx_idx, msgs) in (0u32..).zip(&self.messages) {
454            for msg in msgs {
455                if !msg.message.is_checkpoint_ack() {
456                    result.entry(msg.destination).or_default().insert(tx_idx);
457                }
458            }
459        }
460        result
461    }

479    /// Creates a checkpoint of the tracker's mutable state.
480    ///
481    /// This captures all state that could be modified during transaction execution,
482    /// allowing restoration if execution fails.
483    pub fn create_checkpoint(&self) -> TrackerCheckpoint {
484        TrackerCheckpoint {
485            resource_tracker: self.resource_controller.tracker,
486            next_application_index: self.next_application_index,
487            next_chain_index: self.next_chain_index,
488            transaction_index: self.transaction_index,
489            oracle_responses_len: self.oracle_responses.len(),
490            events_len: self.events.len(),
491            blobs_len: self.blobs.len(),
492            messages_len: self.messages.len(),
493            operation_results_len: self.operation_results.len(),
494        }
495    }
496
497    /// Restores the tracker's mutable state from a checkpoint.
498    ///
499    /// This reverts all state to what it was when the checkpoint was saved,
500    /// as if the failed transaction execution never happened.
501    pub fn restore_checkpoint(&mut self, checkpoint: &TrackerCheckpoint) {
502        // Destructure to ensure all fields are handled (compiler will warn on new fields).
503        let TrackerCheckpoint {
504            resource_tracker,
505            next_application_index,
506            next_chain_index,
507            transaction_index,
508            oracle_responses_len,
509            events_len,
510            blobs_len,
511            messages_len,
512            operation_results_len,
513        } = checkpoint;
514
515        self.resource_controller.tracker = *resource_tracker;
516        self.next_application_index = *next_application_index;
517        self.next_chain_index = *next_chain_index;
518        self.transaction_index = *transaction_index;
519        self.oracle_responses.truncate(*oracle_responses_len);
520        self.events.truncate(*events_len);
521        self.blobs.truncate(*blobs_len);
522        self.messages.truncate(*messages_len);
523        self.operation_results.truncate(*operation_results_len);
524    }

526    /// Finalizes the execution and returns the collected results.
527    ///
528    /// This method should be called after all transactions have been processed.
529    /// The `expected_outcomes_count` should be the number of transactions in the final block.
530    /// Panics if the number of lists of oracle responses, outgoing messages,
531    /// events, or blobs does not match the expected counts.
532    pub fn finalize(self, expected_outcomes_count: usize) -> FinalizeExecutionResult {
533        // Asserts that the number of outcomes matches the expected count.
534        assert_eq!(self.oracle_responses.len(), expected_outcomes_count);
535        assert_eq!(self.messages.len(), expected_outcomes_count);
536        assert_eq!(self.events.len(), expected_outcomes_count);
537        assert_eq!(self.blobs.len(), expected_outcomes_count);
538
539        #[cfg(with_metrics)]
540        crate::chain::metrics::track_block_metrics(&self.resource_controller.tracker, self.phase);
541
542        let resource_tracker = self.resource_controller.tracker;
543
544        (
545            self.messages,
546            self.oracle_responses,
547            self.events,
548            self.blobs,
549            self.operation_results,
550            resource_tracker,
551        )
552    }
553
554    /// Returns the execution context for the current transaction.
555    fn chain_execution_context(&self, transaction: &Transaction) -> ChainExecutionContext {
556        match transaction {
557            Transaction::ReceiveMessages(_) => {
558                ChainExecutionContext::IncomingBundle(self.transaction_index)
559            }
560            Transaction::ExecuteOperation(_) => {
561                ChainExecutionContext::Operation(self.transaction_index)
562            }
563        }
564    }
