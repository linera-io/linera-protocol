32/// Tracks execution of transactions within a block.
33/// Captures the resource policy, produced messages, oracle responses and events.
34#[derive(Debug)]
35pub struct BlockExecutionTracker<'resources, 'blobs> {
36    chain_id: ChainId,
37    block_height: BlockHeight,
38    timestamp: Timestamp,
39    authenticated_owner: Option<AccountOwner>,
40    resource_controller: &'resources mut ResourceController<Option<AccountOwner>, ResourceTracker>,
41    local_time: Timestamp,
42    #[debug(skip_if = Option::is_none)]
43    replaying_oracle_responses: Option<Vec<Vec<OracleResponse>>>,
44    next_application_index: u32,
45    next_chain_index: u32,
46    #[debug(skip_if = Vec::is_empty)]
47    oracle_responses: Vec<Vec<OracleResponse>>,
48    #[debug(skip_if = Vec::is_empty)]
49    events: Vec<Vec<Event>>,
50    #[debug(skip_if = Vec::is_empty)]
51    blobs: Vec<Vec<Blob>>,
52    #[debug(skip_if = Vec::is_empty)]
53    messages: Vec<Vec<OutgoingMessage>>,
54    #[debug(skip_if = Vec::is_empty)]
55    operation_results: Vec<OperationResult>,
56    // Index of the currently executed transaction in a block.
57    transaction_index: u32,
58
59    // Blobs published in the block.
60    published_blobs: BTreeMap<BlobId, &'blobs Blob>,
61}

92            messages: Vec::new(),
93            operation_results: Vec::new(),
94            transaction_index: 0,
95            published_blobs,

97    }

99    /// Executes a transaction in the context of the block.
100    #[instrument(skip_all, fields(
101        chain_id = %self.chain_id,
102        block_height = %self.block_height,
103        transaction_index = %self.transaction_index,
104        transaction_type = %transaction.as_ref(),
105    ))]
106    pub async fn execute_transaction<C>(
107        &mut self,
108        transaction: &Transaction,
109        round: Option<u32>,
110        chain: &mut ExecutionStateView<C>,
111    ) -> Result<(), ChainError>
112    where
113        C: Context + Clone + 'static,
114        C::Extra: ExecutionRuntimeContext,
115    {
116        let chain_execution_context = self.chain_execution_context(transaction);
117        let mut txn_tracker = self.new_transaction_tracker()?;
118
119        match transaction {
120            Transaction::ReceiveMessages(incoming_bundle) => {
121                self.resource_controller_mut()
122                    .track_block_size_of(&incoming_bundle)
123                    .with_execution_context(chain_execution_context)?;
124                for posted_message in incoming_bundle.messages() {
125                    Box::pin(self.execute_message_in_block(
126                        chain,
127                        posted_message,
128                        incoming_bundle,
129                        round,
130                        &mut txn_tracker,
131                    ))
132                    .await?;
133                }
134            }
135            Transaction::ExecuteOperation(operation) => {
136                self.resource_controller_mut()
137                    .with_state(&mut chain.system)
138                    .await?
139                    .track_block_size_of(&operation)
140                    .with_execution_context(chain_execution_context)?;
141                #[cfg(with_metrics)]
142                let _operation_latency = metrics::OPERATION_EXECUTION_LATENCY.measure_latency_us();
143                let context = OperationContext {
144                    chain_id: self.chain_id,
145                    height: self.block_height,
146                    round,
147                    authenticated_owner: self.authenticated_owner,
148                    timestamp: self.timestamp,
149                };
150                let mut actor =
151                    ExecutionStateActor::new(chain, &mut txn_tracker, self.resource_controller);
152                Box::pin(actor.execute_operation(context, operation.clone()))
153                    .await
154                    .with_execution_context(chain_execution_context)?;
155                self.resource_controller_mut()
156                    .with_state(&mut chain.system)
157                    .await?
158                    .track_operation(operation)
159                    .with_execution_context(chain_execution_context)?;
160            }
161        }
162
163        let txn_outcome = txn_tracker
164            .into_outcome()
165            .with_execution_context(chain_execution_context)?;
166        self.process_txn_outcome(txn_outcome, &mut chain.system, chain_execution_context)
167            .await?;
168        Ok(())
169    }

183    /// Executes a message as part of an incoming bundle in a block.
184    #[instrument(skip_all, fields(
185        chain_id = %self.chain_id,
186        block_height = %self.block_height,
187        origin = %incoming_bundle.origin,
188        action = ?incoming_bundle.action,
189        is_bouncing = %posted_message.is_bouncing(),
190    ))]
191    async fn execute_message_in_block<C>(
192        &mut self,
193        chain: &mut ExecutionStateView<C>,
194        posted_message: &PostedMessage,
195        incoming_bundle: &IncomingBundle,
196        round: Option<u32>,
197        txn_tracker: &mut TransactionTracker,
198    ) -> Result<(), ChainError>
199    where
200        C: Context + Clone + 'static,
201        C::Extra: ExecutionRuntimeContext,
202    {
203        #[cfg(with_metrics)]
204        let _message_latency = metrics::MESSAGE_EXECUTION_LATENCY.measure_latency_us();
205        let context = MessageContext {
206            chain_id: self.chain_id,
207            origin: incoming_bundle.origin,
208            is_bouncing: posted_message.is_bouncing(),
209            height: self.block_height,
210            round,
211            authenticated_owner: posted_message.authenticated_owner,
212            refund_grant_to: posted_message.refund_grant_to,
213            timestamp: self.timestamp,
214        };
215        let mut grant = posted_message.grant;
216        match incoming_bundle.action {
217            MessageAction::Accept => {
218                let chain_execution_context =
219                    ChainExecutionContext::IncomingBundle(txn_tracker.transaction_index());
220                // Once a chain is closed, accepting incoming messages is not allowed.
221                ensure!(!chain.system.closed.get(), ChainError::ClosedChain);
222
223                let mut actor =
224                    ExecutionStateActor::new(chain, txn_tracker, self.resource_controller);
225                Box::pin(actor.execute_message(
226                    context,
227                    posted_message.message.clone(),
228                    (grant > Amount::ZERO).then_some(&mut grant),
229                ))
230                .await
231                .with_execution_context(chain_execution_context)?;
232                actor
233                    .send_refund(context, grant)
234                    .with_execution_context(chain_execution_context)?;
235            }
236            MessageAction::Reject => {
237                // If rejecting a message fails, the entire block proposal should be
238                // scrapped.
239                ensure!(
240                    !posted_message.is_protected() || *chain.system.closed.get(),
241                    ChainError::CannotRejectMessage {
242                        chain_id: self.chain_id,
243                        origin: incoming_bundle.origin,
244                        posted_message: Box::new(posted_message.clone()),
245                    }
246                );
247                let mut actor =
248                    ExecutionStateActor::new(chain, txn_tracker, self.resource_controller);
249                if posted_message.is_tracked() {
250                    // Bounce the message.
251                    actor
252                        .bounce_message(context, grant, posted_message.message.clone())
253                        .with_execution_context(ChainExecutionContext::Block)?;
254                } else {
255                    // Nothing to do except maybe refund the grant.
256                    actor
257                        .send_refund(context, grant)
258                        .with_execution_context(ChainExecutionContext::Block)?;
259                }
260            }
261        }
262        Ok(())
263    }

265    /// Returns oracle responses for the current transaction.
266    fn oracle_responses(&self) -> Result<Option<Vec<OracleResponse>>, ChainError> {
267        if let Some(responses) = self.replaying_oracle_responses.as_ref() {
268            match responses.get(self.transaction_index as usize) {
269                Some(responses) => Ok(Some(responses.clone())),
270                None => Err(ChainError::MissingOracleResponseList),
271            }
272        } else {
273            Ok(None)
274        }
275    }

277    /// Processes the transaction outcome.
278    ///
279    /// Updates block tracker with indexes for the next messages, applications, etc.
280    /// so that the execution of the next transaction doesn't overwrite the previous ones.
281    ///
282    /// Tracks the resources used by the transaction - size of the incoming and outgoing messages, blobs, etc.
283    #[instrument(skip_all, fields(
284        chain_id = %self.chain_id,
285        block_height = %self.block_height,
286        transaction_index = %self.transaction_index,
287        outgoing_messages_count = %txn_outcome.outgoing_messages.len(),
288        events_count = %txn_outcome.events.len(),
289        blobs_count = %txn_outcome.blobs.len(),
290    ))]
291    pub async fn process_txn_outcome<C>(
292        &mut self,
293        txn_outcome: TransactionOutcome,
294        view: &mut SystemExecutionStateView<C>,
295        context: ChainExecutionContext,
296    ) -> Result<(), ChainError>
297    where
298        C: Context + Clone + 'static,
299    {
300        let mut resource_controller = self.resource_controller.with_state(view).await?;
301
302        for message_out in &txn_outcome.outgoing_messages {
303            if message_out.kind == MessageKind::Bouncing {
304                continue; // Bouncing messages are free.
305            }
306            if let Message::User { application_id, .. } = &message_out.message {
307                if resource_controller.policy().is_free_app(application_id) {
308                    continue; // Outgoing message fees are waived for free apps.
309                }
310            }
311            resource_controller
312                .track_message(&message_out.message)
313                .with_execution_context(context)?;
314        }
315
316        resource_controller
317            .track_block_size_of(&(
318                &txn_outcome.oracle_responses,
319                &txn_outcome.outgoing_messages,
320                &txn_outcome.events,
321                &txn_outcome.blobs,
322            ))
323            .with_execution_context(context)?;
324
325        // Account for blobs published by this transaction directly.
326        for blob in &txn_outcome.blobs {
327            if txn_outcome.free_blob_ids.contains(&blob.id()) {
328                continue; // Blob publishing fees are waived for free apps.
329            }
330            resource_controller
331                .track_blob_published(blob)
332                .with_execution_context(context)?;
333        }
334
335        // Account for blobs published indirectly but referenced by the transaction.
336        for blob_id in &txn_outcome.blobs_published {
337            if let Some(blob) = self.published_blobs.get(blob_id) {
338                resource_controller
339                    .track_blob_published(blob)
340                    .with_execution_context(context)?;
341            } else {
342                return Err(ChainError::InternalError(format!(
343                    "Missing published blob {blob_id}"
344                )));
345            }
346        }
347
348        self.resource_controller
349            .track_block_size_of(&(&txn_outcome.operation_result))
350            .with_execution_context(context)?;
351
352        self.next_application_index = txn_outcome.next_application_index;
353        self.next_chain_index = txn_outcome.next_chain_index;
354        self.oracle_responses.push(txn_outcome.oracle_responses);
355        self.events.push(txn_outcome.events);
356        self.blobs.push(txn_outcome.blobs);
357        self.messages.push(txn_outcome.outgoing_messages);
358        if matches!(context, ChainExecutionContext::Operation(_)) {
359            self.operation_results
360                .push(OperationResult(txn_outcome.operation_result));
361        }
362        self.transaction_index += 1;
363        Ok(())
364    }

366    /// Returns recipient chain IDs for outgoing messages in the block.
367    pub fn recipients(&self) -> BTreeSet<ChainId> {
368        self.messages
369            .iter()
370            .flatten()
371            .map(|msg| msg.destination)
372            .collect()
373    }
374
375    /// Returns stream IDs for events published in the block.
376    pub fn event_streams(&self) -> BTreeSet<StreamId> {
377        self.events
378            .iter()
379            .flatten()
380            .map(|event| event.stream_id.clone())
381            .collect()
382    }

391    /// Creates a checkpoint of the tracker's mutable state.
392    ///
393    /// This captures all state that could be modified during transaction execution,
394    /// allowing restoration if execution fails.
395    pub fn create_checkpoint(&self) -> TrackerCheckpoint {
396        TrackerCheckpoint {
397            resource_tracker: self.resource_controller.tracker,
398            next_application_index: self.next_application_index,
399            next_chain_index: self.next_chain_index,
400            transaction_index: self.transaction_index,
401            oracle_responses_len: self.oracle_responses.len(),
402            events_len: self.events.len(),
403            blobs_len: self.blobs.len(),
404            messages_len: self.messages.len(),
405            operation_results_len: self.operation_results.len(),
406        }
407    }
408
409    /// Restores the tracker's mutable state from a checkpoint.
410    ///
411    /// This reverts all state to what it was when the checkpoint was saved,
412    /// as if the failed transaction execution never happened.
413    pub fn restore_checkpoint(&mut self, checkpoint: TrackerCheckpoint) {
414        // Destructure to ensure all fields are handled (compiler will warn on new fields).
415        let TrackerCheckpoint {
416            resource_tracker,
417            next_application_index,
418            next_chain_index,
419            transaction_index,
420            oracle_responses_len,
421            events_len,
422            blobs_len,
423            messages_len,
424            operation_results_len,
425        } = checkpoint;
426
427        self.resource_controller.tracker = resource_tracker;
428        self.next_application_index = next_application_index;
429        self.next_chain_index = next_chain_index;
430        self.transaction_index = transaction_index;
431        self.oracle_responses.truncate(oracle_responses_len);
432        self.events.truncate(events_len);
433        self.blobs.truncate(blobs_len);
434        self.messages.truncate(messages_len);
435        self.operation_results.truncate(operation_results_len);
436    }

438    /// Finalizes the execution and returns the collected results.
439    ///
440    /// This method should be called after all transactions have been processed.
441    /// The `expected_outcomes_count` should be the number of transactions in the final block.
442    /// Panics if the number of lists of oracle responses, outgoing messages,
443    /// events, or blobs does not match the expected counts.
444    pub fn finalize(self, expected_outcomes_count: usize) -> FinalizeExecutionResult {
445        // Asserts that the number of outcomes matches the expected count.
446        assert_eq!(self.oracle_responses.len(), expected_outcomes_count);
447        assert_eq!(self.messages.len(), expected_outcomes_count);
448        assert_eq!(self.events.len(), expected_outcomes_count);
449        assert_eq!(self.blobs.len(), expected_outcomes_count);
450
451        #[cfg(with_metrics)]
452        crate::chain::metrics::track_block_metrics(&self.resource_controller.tracker);
453
454        let resource_tracker = self.resource_controller.tracker;
455
456        (
457            self.messages,
458            self.oracle_responses,
459            self.events,
460            self.blobs,
461            self.operation_results,
462            resource_tracker,
463        )
464    }
465
466    /// Returns the execution context for the current transaction.
467    fn chain_execution_context(&self, transaction: &Transaction) -> ChainExecutionContext {
468        match transaction {
469            Transaction::ReceiveMessages(_) => {
470                ChainExecutionContext::IncomingBundle(self.transaction_index)
471            }
472            Transaction::ExecuteOperation(_) => {
473                ChainExecutionContext::Operation(self.transaction_index)
474            }
475        }
476    }
