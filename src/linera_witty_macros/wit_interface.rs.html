18/// Returns the code generated for implementing `WitInterface` to generate WIT snippets for an
19/// interface.
20pub fn generate<'input, Functions>(
21    wit_package: &LitStr,
22    wit_name: LitStr,
23    functions: Functions,
24) -> TokenStream
25where
26    Functions: IntoIterator,
27    Functions::IntoIter: Clone,
28    FunctionInformation<'input>: From<Functions::Item>,
29{
30    let functions = functions.into_iter().map(FunctionInformation::from);
31
32    let dependencies = functions
33        .clone()
34        .flat_map(|function| function.dependencies().cloned().collect::<Vec<_>>());
35
36    let wit_functions = functions.map(|function| function.wit_declaration());
37
38    quote! {
39        type Dependencies = linera_witty::HList![#( #dependencies ),*];
40
41        fn wit_package() -> &'static str {
42            #wit_package
43        }
44
45        fn wit_name() -> &'static str {
46            #wit_name
47        }
48
49        fn wit_functions() -> Vec<String> {
50            vec![
51                #( #wit_functions ),*
52            ]
53        }
54    }
55}

57/// Information from a function relevant to deriving the [`WitInterface`] trait.
58pub struct FunctionInformation<'input> {
59    name: &'input Ident,
60    parameter_names: Vec<&'input Ident>,
61    parameter_types: Vec<&'input Type>,
62    output_type: Option<Type>,
63}
64
65impl<'input> FunctionInformation<'input> {
66    /// Creates a new [`FunctionInformation`] instance from its signature.
67    pub fn new(
68        name: &'input Ident,
69        inputs: impl IntoIterator<Item = &'input FnArg>,
70        output: ReturnType,
71    ) -> Self {
72        let (parameter_names, parameter_types) = inputs
73            .into_iter()
74            .map(|argument| {
75                let FnArg::Typed(PatType { pat, ty, .. }) = argument else {
76                    abort!(argument, "`self` is not supported in imported functions");
77                };
78
79                let Pat::Ident(PatIdent { ident, .. }) = pat.as_ref() else {
80                    abort!(
81                        pat,
82                        "Only named parameters are supported in imported functions"
83                    );
84                };
85
86                (ident, ty.as_ref())
87            })
88            .unzip();
89
90        let output_type = match output {
91            ReturnType::Default => None,
92            ReturnType::Type(_arrow, return_type) => Some(*return_type),
93        };
94
95        FunctionInformation {
96            name,
97            parameter_names,
98            parameter_types,
99            output_type,
100        }
101    }
102
103    /// Returns the types used in the function signature.
104    pub fn dependencies(&self) -> impl Iterator<Item = &'_ Type> {
105        self.parameter_types
106            .clone()
107            .into_iter()
108            .chain(&self.output_type)
109    }
110
111    /// Returns a [`LitStr`] with the kebab-case WIT name of the function.
112    pub fn wit_name(&self) -> LitStr {
113        LitStr::new(&self.name.to_string().to_kebab_case(), self.name.span())
114    }
115
116    /// Returns the code to generate a [`String`] with the WIT declaration of the function.
117    pub fn wit_declaration(&self) -> TokenStream {
118        let wit_name = self.wit_name();
119
120        let parameters = self.parameter_names.iter().zip(&self.parameter_types).map(
121            |(parameter_name, parameter_type)| {
122                let parameter_wit_name = LitStr::new(
123                    &parameter_name.to_string().to_kebab_case(),
124                    parameter_name.span(),
125                );
126
127                quote! {
128                    #parameter_wit_name.into(),
129                    ": ".into(),
130                    <#parameter_type as linera_witty::WitType>::wit_type_name()
131                }
132            },
133        );
134
135        let commas = iter::repeat_n(
136            Some(quote! { ", ".into(), }),
137            self.parameter_names.len().saturating_sub(1),
138        )
139        .chain(Some(None));
140
141        let output = self
142            .output_type
143            .as_ref()
144            .map(|output_type| {
145                quote! { " -> ".into(), <#output_type as linera_witty::WitType>::wit_type_name() }
146            })
147            .into_iter();
148
149        quote! {
150            [
151                std::borrow::Cow::Borrowed("    "),
152                #wit_name.into(),
153                ": func(".into(),
154                #( #parameters, #commas )*
155                ")".into(),
156                #( #output, )*
157                ";".into(),
158            ]
159            .as_slice()
160            .join("")
161        }
162    }
163}

165impl<'input> From<&'_ wit_import::FunctionInformation<'input>> for FunctionInformation<'input> {
166    fn from(imported_function: &'_ wit_import::FunctionInformation<'input>) -> Self {
167        let signature = &imported_function.function.sig;
168
169        FunctionInformation::new(
170            &signature.ident,
171            &signature.inputs,
172            signature.output.clone(),
173        )
174    }
175}

177#[cfg(with_wit_export)]
178impl<'input> From<&'_ wit_export::FunctionInformation<'input>> for FunctionInformation<'input> {
179    fn from(exported_function: &'_ wit_export::FunctionInformation<'input>) -> Self {
180        let signature = &exported_function.function.sig;
181
182        let inputs = signature
183            .inputs
184            .iter()
185            .skip(if exported_function.is_reentrant { 1 } else { 0 });
186
187        let mut output = signature.output.clone();
188
189        if exported_function.call_early_return.is_some() {
190            let ReturnType::Type(_arrow, return_type) = &mut output else {
191                abort!(output, "Missing `Result` in fallible function return type");
192            };
193
194            let Some(actual_output) = wit_export::ok_type_inside_result(&*return_type).cloned()
195            else {
196                abort!(
197                    output,
198                    "Missing `Ok` result type in fallible function return type"
199                );
200            };
201
202            if is_unit_type(&actual_output) {
203                output = ReturnType::Default;
204            } else {
205                **return_type = actual_output;
206            }
207        }
208
209        FunctionInformation::new(&signature.ident, inputs, output)
210    }
211}

213/// Returns `true` if `the_type` is the unit `()` type.
214#[cfg(with_wit_export)]
215pub fn is_unit_type(the_type: &Type) -> bool {
216    matches!(the_type, Type::Tuple(tuple) if tuple.elems.is_empty())
217}
