16/// Returns the body of the `WitLoad` implementation for the Rust `struct` with the specified
17/// `fields`.
18pub fn derive_for_struct<'input>(fields: impl Into<FieldsInformation<'input>>) -> TokenStream {
19    let fields = fields.into();
20    let fields_hlist_binding = fields.hlist_bindings();
21    let fields_hlist_type = fields.hlist_type();
22    let construction = fields.construction();
23    let fallback_bindings = fields.bindings_for_skipped_fields();
24
25    quote! {
26        fn load<Instance>(
27            memory: &linera_witty::Memory<'_, Instance>,
28            mut location: linera_witty::GuestPointer,
29        ) -> Result<Self, linera_witty::RuntimeError>
30        where
31            Instance: linera_witty::InstanceWithMemory,
32            <Instance::Runtime as linera_witty::Runtime>::Memory:
33                linera_witty::RuntimeMemory<Instance>,
34        {
35            let #fields_hlist_binding =
36                <#fields_hlist_type as linera_witty::WitLoad>::load(memory, location)?;
37
38            #fallback_bindings
39
40            Ok(Self #construction)
41        }
42
43        fn lift_from<Instance>(
44            flat_layout: <Self::Layout as linera_witty::Layout>::Flat,
45            memory: &linera_witty::Memory<'_, Instance>,
46        ) -> Result<Self, linera_witty::RuntimeError>
47        where
48            Instance: linera_witty::InstanceWithMemory,
49            <Instance::Runtime as linera_witty::Runtime>::Memory:
50                linera_witty::RuntimeMemory<Instance>,
51        {
52            let #fields_hlist_binding =
53                <#fields_hlist_type as linera_witty::WitLoad>::lift_from(flat_layout, memory)?;
54
55            #fallback_bindings
56
57            Ok(Self #construction)
58        }
59    }
60}

62/// Returns the body of the `WitLoad` implementation for the Rust `enum` with the specified
63/// `variants`.
64pub fn derive_for_enum<'variants>(
65    name: &Ident,
66    variants: impl DoubleEndedIterator<Item = &'variants Variant> + Clone,
67) -> TokenStream {
68    let variant_count = variants.clone().count();
69    let variant_fields = variants
70        .clone()
71        .map(|variant| FieldsInformation::from(&variant.fields))
72        .collect::<Vec<_>>();
73    let variant_types = variant_fields.iter().map(FieldsInformation::hlist_type);
74    let variants = variants.zip(&variant_fields).enumerate();
75
76    let discriminant_type = if variant_count <= u8::MAX.into() {
77        quote! { u8 }
78    } else if variant_count <= u16::MAX.into() {
79        quote! { u16 }
80    } else if variant_count <= u32::MAX as usize {
81        quote! { u32 }
82    } else {
83        abort!(name, "Too many variants in `enum`");
84    };
85
86    let align_to_cases = variant_types.fold(quote! {}, |location, variant_type| {
87        quote! {
88            #location.after_padding_for::<#variant_type>()
89        }
90    });
91
92    let load_variants = variants.clone().map(|(index, (variant, fields))| {
93        let variant_name = &variant.ident;
94        let index = LitInt::new(&index.to_string(), variant_name.span());
95        let field_bindings = fields.hlist_bindings();
96        let fields_type = fields.hlist_type();
97        let construction = fields.construction();
98        let fallback_bindings = fields.bindings_for_skipped_fields();
99
100        quote! {
101            #index => {
102                let #field_bindings =
103                    <#fields_type as linera_witty::WitLoad>::load(memory, location)?;
104
105                #fallback_bindings
106
107                Ok(#name::#variant_name #construction)
108            }
109        }
110    });
111
112    let lift_variants = variants.map(|(index, (variant, fields))| {
113        let variant_name = &variant.ident;
114        let index = LitInt::new(&index.to_string(), variant_name.span());
115        let field_bindings = fields.hlist_bindings();
116        let fields_type = fields.hlist_type();
117        let construction = fields.construction();
118        let fallback_bindings = fields.bindings_for_skipped_fields();
119
120        quote! {
121            #index => {
122                let #field_bindings = <#fields_type as linera_witty::WitLoad>::lift_from(
123                    linera_witty::JoinFlatLayouts::from_joined(flat_layout),
124                    memory,
125                )?;
126
127                #fallback_bindings
128
129                Ok(#name::#variant_name #construction)
130            }
131        }
132    });
133
134    quote! {
135        fn load<Instance>(
136            memory: &linera_witty::Memory<'_, Instance>,
137            mut location: linera_witty::GuestPointer,
138        ) -> Result<Self, linera_witty::RuntimeError>
139        where
140            Instance: linera_witty::InstanceWithMemory,
141            <Instance::Runtime as linera_witty::Runtime>::Memory:
142                linera_witty::RuntimeMemory<Instance>,
143        {
144            let discriminant = <#discriminant_type as linera_witty::WitLoad>::load(
145                memory,
146                location,
147            )?;
148            location = location.after::<#discriminant_type>() #align_to_cases;
149
150            match discriminant {
151                #( #load_variants )*
152                discriminant => Err(linera_witty::RuntimeError::InvalidVariant {
153                    type_name: ::std::any::type_name::<Self>(),
154                    discriminant: discriminant.into(),
155                }),
156            }
157        }
158
159        fn lift_from<Instance>(
160            linera_witty::hlist_pat![discriminant_flat_type, ...flat_layout]:
161                <Self::Layout as linera_witty::Layout>::Flat,
162            memory: &linera_witty::Memory<'_, Instance>,
163        ) -> Result<Self, linera_witty::RuntimeError>
164        where
165            Instance: linera_witty::InstanceWithMemory,
166            <Instance::Runtime as linera_witty::Runtime>::Memory:
167                linera_witty::RuntimeMemory<Instance>,
168        {
169            let discriminant = <#discriminant_type as linera_witty::WitLoad>::lift_from(
170                linera_witty::hlist![discriminant_flat_type],
171                memory,
172            )?;
173
174            match discriminant {
175                #( #lift_variants )*
176                discriminant => Err(linera_witty::RuntimeError::InvalidVariant {
177                    type_name: ::std::any::type_name::<Self>(),
178                    discriminant: discriminant.into(),
179                }),
180            }
181        }
182    }
183}
