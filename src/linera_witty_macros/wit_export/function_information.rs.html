4//! Extraction of information and generation of code related to a single exported host function.

16/// Pieces of information extracted from a function's definition.
17pub struct FunctionInformation<'input> {
18    pub(crate) function: &'input ImplItemFn,
19    pub(crate) is_reentrant: bool,
20    pub(crate) call_early_return: Option<Token![?]>,
21    wit_name: String,
22    parameter_bindings: TokenStream,
23    interface_type: TokenStream,
24}
25
26impl<'input> FunctionInformation<'input> {
27    /// Parses a function definition from an [`ImplItem`] and collects pieces of information into a
28    /// [`FunctionInformation`] instance.
29    pub fn from_item(item: &'input ImplItem, caller_type_parameter: Option<&'input Ident>) -> Self {
30        match item {
31            ImplItem::Fn(function) => FunctionInformation::new(function, caller_type_parameter),
32            ImplItem::Const(const_item) => abort!(
33                const_item.ident,
34                "Const items are not supported in exported types"
35            ),
36            ImplItem::Type(type_item) => abort!(
37                type_item.ident,
38                "Type items are not supported in exported types"
39            ),
40            ImplItem::Macro(macro_item) => abort!(
41                macro_item.mac.path,
42                "Macro items are not supported in exported types"
43            ),
44            _ => abort!(item, "Only function items are supported in exported types"),
45        }
46    }
47
48    /// Parses a function definition and collects pieces of information into a
49    /// [`FunctionInformation`] instance.
50    pub fn new(function: &'input ImplItemFn, caller_type: Option<&'input Ident>) -> Self {
51        let wit_name = function.sig.ident.to_string().to_kebab_case();
52        let is_reentrant = Self::is_reentrant(&function.sig)
53            || Self::uses_caller_parameter(&function.sig, caller_type);
54        let (parameter_bindings, parameter_types) =
55            Self::parse_parameters(is_reentrant, function.sig.inputs.iter());
56        let (results, is_fallible) = Self::parse_output(&function.sig.output);
57
58        let interface_type = quote_spanned! { function.sig.span() =>
59            (linera_witty::HList![#parameter_types], #results)
60        };
61
62        FunctionInformation {
63            function,
64            is_reentrant,
65            call_early_return: is_fallible.then(|| Token![?](Span::call_site())),
66            wit_name,
67            parameter_bindings,
68            interface_type,
69        }
70    }
71
72    /// Checks if a function should be considered as a reentrant function.
73    ///
74    /// A reentrant function has a generic type parameter that's used as the type of the first
75    /// parameter.
76    fn is_reentrant(signature: &Signature) -> bool {
77        if signature.generics.params.len() != 1 {
78            return false;
79        }
80
81        let Some(GenericParam::Type(generic_type)) = signature.generics.params.first() else {
82            return false;
83        };
84
85        Self::first_parameter_is_caller(signature, &generic_type.ident)
86    }
87
88    /// Checks if a function uses a `caller_type` in the first parameter.
89    ///
90    /// If it does, the function is assumed to be reentrant.
91    fn uses_caller_parameter(signature: &Signature, caller_type: Option<&Ident>) -> bool {
92        if let Some(caller_type) = caller_type {
93            Self::first_parameter_is_caller(signature, caller_type)
94        } else {
95            false
96        }
97    }
98
99    /// Checks if the type of a function's first parameter is the `caller_type`.
100    fn first_parameter_is_caller(signature: &Signature, caller_type: &Ident) -> bool {
101        let Some(first_parameter) = signature.inputs.first() else {
102            return false;
103        };
104
105        let FnArg::Typed(PatType {
106            ty: first_parameter_type,
107            ..
108        }) = first_parameter
109        else {
110            abort!(
111                first_parameter,
112                "`self` parameters aren't supported by Witty"
113            );
114        };
115
116        let Type::Reference(TypeReference {
117            mutability: Some(_),
118            elem: referenced_type,
119            ..
120        }) = &**first_parameter_type
121        else {
122            return false;
123        };
124
125        let Type::Path(TypePath { path, .. }) = &**referenced_type else {
126            return false;
127        };
128
129        path.is_ident(caller_type)
130    }
131
132    /// Parses a function's parameters and returns the generated code with a list of bindings to the
133    /// parameters and a list of the parameters types.
134    fn parse_parameters(
135        is_reentrant: bool,
136        inputs: impl Iterator<Item = &'input FnArg> + Clone,
137    ) -> (TokenStream, TokenStream) {
138        let parameters = inputs
139            .skip(if is_reentrant { 1 } else { 0 })
140            .map(|input| match input {
141                FnArg::Typed(parameter) => parameter,
142                FnArg::Receiver(receiver) => abort!(
143                    receiver.self_token,
144                    "Exported interfaces can not have `self` parameters"
145                ),
146            });
147
148        let bindings = parameters.clone().map(|parameter| &parameter.pat);
149        let types = parameters.map(|parameter| &parameter.ty);
150
151        (quote! { #( #bindings ),* }, quote! { #( #types ),* })
152    }
153
154    /// Parses a function's return type, returning the type to use as the WIT result and whether
155    /// the function is fallible.
156    fn parse_output(output: &ReturnType) -> (TokenStream, bool) {
157        match output {
158            ReturnType::Default => (quote_spanned! { output.span() => () }, false),
159            ReturnType::Type(_, return_type) => match ok_type_inside_result(return_type) {
160                Some(inner_type) => (inner_type.to_token_stream(), true),
161                None => (return_type.to_token_stream(), false),
162            },
163        }
164    }

166    /// Generates the code to export a host function using the Wasmer runtime.
167    #[cfg(with_wasmer)]
168    pub fn generate_for_wasmer(
169        &self,
170        namespace: &LitStr,
171        type_name: &Ident,
172        caller: &Type,
173    ) -> TokenStream {
174        let input_to_guest_parameters = quote! {
175            linera_witty::wasmer::WasmerParameters::from_wasmer(input)
176        };
177        let guest_results_to_output = quote! {
178            linera_witty::wasmer::WasmerResults::into_wasmer(guest_results)
179        };
180        let output_results_trait = quote! { linera_witty::wasmer::WasmerResults };
181
182        self.generate(
183            namespace,
184            type_name,
185            caller,
186            input_to_guest_parameters,
187            guest_results_to_output,
188            output_results_trait,
189        )
190    }

240    /// Generates the code to export using a host function.
241    fn generate(
242        &self,
243        namespace: &LitStr,
244        type_name: &Ident,
245        caller: &Type,
246        input_to_guest_parameters: TokenStream,
247        guest_results_to_output: TokenStream,
248        output_results_trait: TokenStream,
249    ) -> TokenStream {
250        let wit_name = &self.wit_name;
251        let interface_type = &self.interface_type;
252        let host_parameters = &self.parameter_bindings;
253        let call_early_return = &self.call_early_return;
254        let function_name = &self.function.sig.ident;
255        let caller_parameter = self.is_reentrant.then(|| quote! { &mut caller, });
256
257        let output_type = quote_spanned! { self.function.sig.output.span() =>
258            <
259                <
260                    #interface_type as linera_witty::ExportedFunctionInterface
261                >::GuestResults as #output_results_trait
262            >::Results
263        };
264
265        quote_spanned! { self.function.span() =>
266            linera_witty::ExportFunction::export(
267                target,
268                #namespace,
269                #wit_name,
270                #[allow(clippy::type_complexity)]
271                |mut caller: #caller, input| -> Result<#output_type, linera_witty::RuntimeError> {
272                    type Interface = #interface_type;
273
274                    let guest_parameters = #input_to_guest_parameters;
275                    let (linera_witty::hlist_pat![#host_parameters], result_storage) =
276                        <Interface as linera_witty::ExportedFunctionInterface>::lift_parameters(
277                            guest_parameters,
278                            &linera_witty::InstanceWithMemory::memory(&mut caller)?,
279                        )?;
280
281                    #[allow(clippy::let_unit_value)]
282                    let host_results = #type_name::#function_name(
283                        #caller_parameter
284                        #host_parameters
285                    ) #call_early_return;
286                    let guest_results =
287                        <Interface as linera_witty::ExportedFunctionInterface>::lower_results(
288                            host_results,
289                            result_storage,
290                            &mut linera_witty::InstanceWithMemory::memory(&mut caller)?,
291                        )?;
292
293                    #[allow(clippy::unit_arg)]
294                    Ok(#guest_results_to_output)
295                }
296            )?;
297        }
298    }
299}

301/// Returns the type inside the `Ok` variant of the `maybe_result_type`.
302///
303/// The type is only considered if it's a [`Result`] type with `RuntimeError` as its error variant.
304pub(crate) fn ok_type_inside_result(maybe_result_type: &Type) -> Option<&Type> {
305    let Type::Path(TypePath { qself: None, path }) = maybe_result_type else {
306        return None;
307    };
308
309    let (ok_type, error_type) = result_type_arguments(path)?;
310
311    if let Type::Path(TypePath { qself: None, path }) = error_type {
312        if !path.is_ident("RuntimeError") {
313            return None;
314        }
315    } else {
316        return None;
317    }
318
319    Some(ok_type)
320}

322/// Returns the generic type arguments of the [`Result`] type in `result_path`.
323fn result_type_arguments(result_path: &Path) -> Option<(&Type, &Type)> {
324    if !type_is_result(result_path) {
325        return None;
326    }
327
328    let PathArguments::AngleBracketed(type_arguments) = &result_path.segments.last()?.arguments
329    else {
330        return None;
331    };
332
333    if type_arguments.args.len() != 2 {
334        return None;
335    }
336
337    let mut arguments = type_arguments.args.iter();
338
339    let GenericArgument::Type(ok_type) = arguments.next()? else {
340        return None;
341    };
342
343    let GenericArgument::Type(error_type) = arguments.next()? else {
344        return None;
345    };
346
347    Some((ok_type, error_type))
348}

350/// Checks if `result_path` is a [`Result`] type.
351fn type_is_result(result_path: &Path) -> bool {
352    let segment_count = result_path.segments.len();
353
354    if segment_count == 1 {
355        result_path.leading_colon.is_none() && path_matches_segments(result_path, &["Result"])
356    } else if result_path.segments.len() == 3 {
357        path_matches_segments(result_path, &["std", "result", "Result"])
358    } else {
359        false
360    }
361}
362
363/// Checks if `path` matches the provided path `segments`.
364fn path_matches_segments(path: &Path, segments: &[&str]) -> bool {
365    if path.segments.len() != segments.len() {
366        return false;
367    }
368
369    for (index, (segment, expected)) in path.segments.iter().zip(segments).enumerate() {
370        let with_type_parameters = index == segments.len() - 1;
371
372        if !is_path_segment(segment, expected, with_type_parameters) {
373            return false;
374        }
375    }
376
377    true
378}

380/// Checks if `segment` is the `expected_identifier` and if it should have generic type parameters.
381fn is_path_segment(
382    segment: &PathSegment,
383    expected_identifier: &str,
384    with_type_parameters: bool,
385) -> bool {
386    let arguments_are_correct = if with_type_parameters {
387        matches!(segment.arguments, PathArguments::AngleBracketed(_))
388    } else {
389        matches!(segment.arguments, PathArguments::None)
390    };
391
392    segment.ident == expected_identifier && arguments_are_correct
393}
