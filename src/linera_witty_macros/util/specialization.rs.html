30/// Collected specializations to apply before deriving a trait for a type.
31#[derive(Debug)]
32pub struct Specializations(Vec<Specialization>);
33
34impl FromIterator<Specialization> for Specializations {
35    fn from_iter<I>(specializations: I) -> Self
36    where
37        I: IntoIterator<Item = Specialization>,
38    {
39        Specializations(specializations.into_iter().collect())
40    }
41}
42
43impl Specializations {
44    /// Changes the [`DeriveInput`] based on the specializations requested through the
45    /// `witty_specialize_with` attributes.
46    ///
47    /// The [`DeriveInput`] is changed so that its `where` clause and field types are specialized.
48    /// Returns the [`Specializations`] instance created from parsing the `witty_specialize_with`
49    /// attributes from the [`DeriveInput`].
50    pub fn prepare_derive_input(input: &mut DeriveInput) -> Self {
51        let this: Self = Self::parse_specialization_attributes(&input.attrs).collect();
52
53        for specialization in &this.0 {
54            specialization.apply_to_derive_input(input);
55        }
56
57        this
58    }
59
60    /// Creates a list of [`Specialization`]s based on the `witty_specialize_with` attributes found
61    /// in the provided `attributes`.
62    fn parse_specialization_attributes(
63        attributes: &[Attribute],
64    ) -> impl Iterator<Item = Specialization> {
65        let mut specializations = Vec::new();
66
67        let abort_with_error = |span: Span| -> ! {
68            abort!(
69                span,
70                "Failed to parse Witty specialization attribute. \
71                Expected: `#[witty_specialize_with(TypeParam = Type, ...)]`."
72            );
73        };
74
75        for attribute in attributes {
76            match &attribute.meta {
77                Meta::List(MetaList {
78                    path,
79                    delimiter,
80                    tokens,
81                }) if path.is_ident("witty_specialize_with")
82                    && matches!(delimiter, MacroDelimiter::Paren(_)) =>
83                {
84                    let parser = Punctuated::<Specialization, Token![,]>::parse_separated_nonempty;
85                    specializations.push(
86                        parser
87                            .parse2(tokens.clone())
88                            .unwrap_or_else(|_| abort_with_error(attribute.span()))
89                            .into_iter(),
90                    );
91                }
92                _ => {}
93            }
94        }
95
96        specializations.into_iter().flatten()
97    }
98
99    /// Specializes the types in the [`Generics`] representation.
100    #[cfg(with_wit_export)]
101    pub fn apply_to_generics(&self, generics: &mut Generics) {
102        for specialization in &self.0 {
103            specialization.apply_to_generics(generics);
104        }
105    }
106
107    /// Specializes the types in the `target_type`, either itself or its type parameters.
108    #[cfg(with_wit_export)]
109    pub fn apply_to_type(&self, target_type: &mut Type) {
110        for specialization in &self.0 {
111            specialization.change_types_in_type(target_type);
112        }
113    }
114
115    /// Retrieves the information related to generics from the provided [`Generics`] after
116    /// applying the specializations from this instance.
117    ///
118    /// Returns the generic parameters for the `impl` block, the generic arguments for the target
119    /// type, and the where clause to use.
120    pub fn split_generics_from<'generics>(
121        &self,
122        generics: &'generics Generics,
123    ) -> (
124        TokenStream,
125        Option<AngleBracketedGenericArguments>,
126        Option<&'generics WhereClause>,
127    ) {
128        let (_original_generic_parameters, original_type_generics, where_clause) =
129            generics.split_for_impl();
130        let type_generics = self.specialize_type_generics(original_type_generics);
131        let generic_parameters = self.clean_generic_parameters(generics.clone());
132
133        (
134            generic_parameters.into_token_stream(),
135            type_generics,
136            where_clause,
137        )
138    }
139
140    /// Specializes the target type's generic parameters.
141    fn specialize_type_generics(
142        &self,
143        type_generics: impl ToTokens,
144    ) -> Option<AngleBracketedGenericArguments> {
145        let mut generic_type = syn::parse_quote!(TheType #type_generics);
146
147        for specialization in &self.0 {
148            specialization.change_types_in_type(&mut generic_type);
149        }
150
151        match generic_type {
152            Type::Path(TypePath {
153                qself: None,
154                path:
155                    Path {
156                        leading_colon: None,
157                        segments,
158                    },
159            }) if segments.len() == 1 => {
160                let segment = segments
161                    .into_iter()
162                    .next()
163                    .expect("Missing custom type's path");
164                assert_eq!(segment.ident, "TheType");
165
166                match segment.arguments {
167                    PathArguments::None => None,
168                    PathArguments::AngleBracketed(arguments) => Some(arguments),
169                    PathArguments::Parenthesized(_) => {
170                        unreachable!("Custom type has unexpected function type parameters")
171                    }
172                }
173            }
174            _ => unreachable!("Parsed custom type literal is incorrect"),
175        }
176    }
177
178    /// Returns the generic parameters from the [`Generics`] information after applying the
179    /// specializations from this instance.
180    fn clean_generic_parameters(&self, mut generics: Generics) -> TokenStream {
181        let original_generic_types = mem::take(&mut generics.params);
182        let parameters_to_remove: HashSet<Ident> = self
183            .0
184            .iter()
185            .map(|specialization| specialization.type_parameter.clone())
186            .collect();
187
188        generics
189            .params
190            .extend(
191                original_generic_types
192                    .into_iter()
193                    .filter(|generic_type| match generic_type {
194                        GenericParam::Lifetime(_) | GenericParam::Const(_) => true,
195                        GenericParam::Type(type_parameter) => {
196                            !parameters_to_remove.contains(&type_parameter.ident)
197                        }
198                    }),
199            );
200
201        let (generic_parameters, _incorrect_type_generics, _unaltered_where_clause) =
202            generics.split_for_impl();
203
204        generic_parameters.into_token_stream()
205    }
206}

208/// A single specialization of a generic type parameter.
209pub struct Specialization {
210    /// The type parameter to be specialized.
211    type_parameter: Ident,
212    /// The type to use as the specialized argument.
213    specialized_type: Type,
214}
215
216impl Parse for Specialization {
217    fn parse(input: ParseStream) -> parse::Result<Self> {
218        let type_parameter = input.parse()?;
219        let _: Token![=] = input.parse()?;
220        let specialized_type = input.parse()?;
221
222        Ok(Specialization {
223            type_parameter,
224            specialized_type,
225        })
226    }
227}
228
229impl Specialization {
230    /// Creates a new specialization for the `type_parameter`, to specialize it into the
231    /// `specialized_type`.
232    #[cfg(with_wit_export)]
233    pub fn new(type_parameter: Ident, specialized_type: Type) -> Self {
234        Specialization {
235            type_parameter,
236            specialized_type,
237        }
238    }
239
240    /// Replaces a type parameter in the [`DeriveInput`] with a specialized type.
241    ///
242    /// Note that the specialization is only done to the `where` clause and the type's fields. The
243    /// types generic parameters needs to be changed separately (see
244    /// [`Specializations::specialize_type_generics`].
245    pub fn apply_to_derive_input(&self, input: &mut DeriveInput) {
246        self.apply_to_generics(&mut input.generics);
247        self.change_types_in_fields(&mut input.data);
248    }
249
250    /// Replaces a type parameter in the [`Generics`] representation with a specialized type.
251    pub fn apply_to_generics(&self, generics: &mut Generics) {
252        self.remove_from_where_clause(generics.where_clause.as_mut());
253        self.change_types_in_where_clause(generics.where_clause.as_mut());
254    }
255
256    /// Removes from a [`WhereClause`] all predicates for the [`Self::type_parameter`] that this
257    /// specialization targets.
258    fn remove_from_where_clause(&self, maybe_where_clause: Option<&mut WhereClause>) {
259        if let Some(WhereClause { predicates, .. }) = maybe_where_clause {
260            let original_predicates = mem::take(predicates);
261
262            predicates.extend(
263                original_predicates
264                    .into_iter()
265                    .filter(|predicate| !self.affects_predicate(predicate)),
266            );
267        }
268    }
269
270    /// Returns [`true`] if this [`Specialization`] affects the `predicate`.
271    fn affects_predicate(&self, predicate: &WherePredicate) -> bool {
272        let WherePredicate::Type(PredicateType {
273            bounded_ty: Type::Path(type_path),
274            ..
275        }) = predicate
276        else {
277            return false;
278        };
279        let mut type_path = type_path;
280
281        while let Some(inner_type) = &type_path.qself {
282            type_path = match &*inner_type.ty {
283                Type::Path(path) => path,
284                _ => return false,
285            };
286        }
287
288        let Some(segment) = type_path.path.segments.first() else {
289            return false;
290        };
291
292        segment.ident == self.type_parameter && matches!(segment.arguments, PathArguments::None)
293    }
294
295    /// Replaces the [`Self::type_parameter`] with the [`Self::specialized_type`] inside the
296    /// predicates of the [`WhereClause`].
297    fn change_types_in_where_clause(&self, maybe_where_clause: Option<&mut WhereClause>) {
298        let type_predicates = maybe_where_clause
299            .map(|where_clause| where_clause.predicates.iter_mut())
300            .into_iter()
301            .flatten()
302            .filter_map(|predicate| match predicate {
303                WherePredicate::Type(type_predicate) => Some(type_predicate),
304                _ => None,
305            });
306
307        for predicate in type_predicates {
308            self.change_types_in_type(&mut predicate.bounded_ty);
309            self.change_types_in_bounds(predicate.bounds.iter_mut());
310        }
311    }
312
313    /// Replaces the [`Self::type_parameter`] with the [`Self::specialized_type`] inside the
314    /// fields of the [`DeriveInput`]'s [`Data`].
315    fn change_types_in_fields(&self, data: &mut Data) {
316        let fields: Box<dyn Iterator<Item = &mut Field>> = match data {
317            Data::Struct(DataStruct {
318                fields: Fields::Named(fields),
319                ..
320            })
321            | Data::Union(DataUnion { fields, .. }) => Box::new(fields.named.iter_mut()),
322            Data::Struct(DataStruct {
323                fields: Fields::Unnamed(fields),
324                ..
325            }) => Box::new(fields.unnamed.iter_mut()),
326            Data::Enum(DataEnum { variants, .. }) => Box::new(
327                variants
328                    .iter_mut()
329                    .flat_map(|variant| variant.fields.iter_mut()),
330            ),
331            Data::Struct(_) => Box::new(None.into_iter()),
332        };
333
334        for Field { ty, .. } in fields {
335            self.change_types_in_type(ty);
336        }
337    }
338
339    /// Replaces the [`Self::type_parameter`] with the [`Self::specialized_type`] inside the
340    /// generic type parameter bounds.
341    fn change_types_in_bounds<'bound>(
342        &self,
343        bounds: impl Iterator<Item = &'bound mut TypeParamBound>,
344    ) {
345        for bound in bounds {
346            if let TypeParamBound::Trait(trait_bound) = bound {
347                self.change_types_in_path(&mut trait_bound.path);
348            }
349        }
350    }
351
352    /// Replaces the [`Self::type_parameter`] with the [`Self::specialized_type`] inside the
353    /// provided [`Type`].
354    pub fn change_types_in_type(&self, the_type: &mut Type) {
355        match the_type {
356            Type::Array(TypeArray { elem, .. })
357            | Type::Group(TypeGroup { elem, .. })
358            | Type::Paren(TypeParen { elem, .. })
359            | Type::Ptr(TypePtr { elem, .. })
360            | Type::Reference(TypeReference { elem, .. })
361            | Type::Slice(TypeSlice { elem, .. }) => self.change_types_in_type(elem.as_mut()),
362            Type::BareFn(TypeBareFn { inputs, output, .. }) => self.change_types_in_function_type(
363                inputs.iter_mut().map(|bare_fn_arg| &mut bare_fn_arg.ty),
364                output,
365            ),
366            Type::ImplTrait(TypeImplTrait { bounds, .. }) => {
367                self.change_types_in_bounds(bounds.iter_mut())
368            }
369            Type::Path(TypePath { qself: None, path }) if path.is_ident(&self.type_parameter) => {
370                *the_type = self.specialized_type.clone();
371            }
372            Type::Path(TypePath { qself, path }) => {
373                if let Some(QSelf { ty, .. }) = qself {
374                    self.change_types_in_type(ty);
375                }
376                self.change_types_in_path(path);
377            }
378            Type::Tuple(TypeTuple { elems, .. }) => {
379                for element in elems {
380                    self.change_types_in_type(element);
381                }
382            }
383            _ => {}
384        }
385    }
386
387    /// Replaces the [`Self::type_parameter`] with the [`Self::specialized_type`] inside the
388    /// [`Path`]'s type arguments.
389    fn change_types_in_path(&self, path: &mut Path) {
390        for segment in &mut path.segments {
391            match &mut segment.arguments {
392                PathArguments::None => {}
393                PathArguments::AngleBracketed(angle_bracketed) => {
394                    self.change_types_in_angle_bracketed_generic_arguments(angle_bracketed)
395                }
396                PathArguments::Parenthesized(function_type) => self.change_types_in_function_type(
397                    function_type.inputs.iter_mut(),
398                    &mut function_type.output,
399                ),
400            }
401        }
402    }
403
404    /// Replaces the [`Self::type_parameter`] with the [`Self::specialized_type`] inside the
405    /// [`AngleBracketedGenericArguments`] of a [`PathSegment`][`syn::PathSegment`].
406    fn change_types_in_angle_bracketed_generic_arguments(
407        &self,
408        arguments: &mut AngleBracketedGenericArguments,
409    ) {
410        for argument in &mut arguments.args {
411            match argument {
412                GenericArgument::Type(the_type) => self.change_types_in_type(the_type),
413                GenericArgument::AssocType(AssocType { generics, ty, .. }) => {
414                    if let Some(arguments) = generics {
415                        self.change_types_in_angle_bracketed_generic_arguments(arguments);
416                    }
417                    self.change_types_in_type(ty);
418                }
419                GenericArgument::AssocConst(AssocConst {
420                    generics: Some(arguments),
421                    ..
422                }) => {
423                    self.change_types_in_angle_bracketed_generic_arguments(arguments);
424                }
425                GenericArgument::Constraint(Constraint {
426                    generics, bounds, ..
427                }) => {
428                    if let Some(arguments) = generics {
429                        self.change_types_in_angle_bracketed_generic_arguments(arguments);
430                    }
431                    self.change_types_in_bounds(bounds.iter_mut());
432                }
433                _ => {}
434            }
435        }
436    }
437
438    /// Replaces the [`Self::type_parameter`] with the [`Self::specialized_type`] inside a
439    /// function's input parameter [`Type`]s and output [`ReturnType`].
440    /// [`Path`]'s type arguments.
441    fn change_types_in_function_type<'input>(
442        &self,
443        inputs: impl Iterator<Item = &'input mut Type>,
444        output: &mut ReturnType,
445    ) {
446        for ty in inputs {
447            self.change_types_in_type(ty);
448        }
449
450        if let ReturnType::Type(_, return_type) = output {
451            self.change_types_in_type(return_type.as_mut());
452        }
453    }
454}

456impl Debug for Specialization {
457    fn fmt(&self, formatter: &mut Formatter) -> fmt::Result {
458        formatter
459            .debug_struct("Specialization")
460            .field("type_parameter", &self.type_parameter)
461            .field(
462                "specialized_type",
463                &self.specialized_type.to_token_stream().to_string(),
464            )
465            .finish()
466    }
