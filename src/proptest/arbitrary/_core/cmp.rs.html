10//! Arbitrary implementations for `std::cmp`.
11
12use core::cmp::{Ordering, Reverse};
13
14use crate::strategy::{Just, TupleUnion, WA};
15
16wrap_ctor!(Reverse, Reverse);
17
18type WAJO = WA<Just<Ordering>>;
19arbitrary!(Ordering, TupleUnion<(WAJO, WAJO, WAJO)>;
20    prop_oneof![
21        Just(Ordering::Equal),
22        Just(Ordering::Less),
23        Just(Ordering::Greater)
24    ]
25);
26
27#[cfg(test)]
28mod test {
29    no_panic_test!(
30        reverse => Reverse<u8>,
31        ordering => Ordering
32    );
33}
