8/// Any error that happens when opening a stream.
9#[derive(Debug)]
10pub enum Error {
11    /// the [`io::Error`](IoError) returned by the os when opening the file
12    Io(IoError),
13    #[cfg(feature = "http")]
14    /// the HTTP response code and message returned by the sever
15    ///
16    /// code 499 may be returned in some instances when the connection to
17    /// the server did not complete.
18    Http {
19        /// [HTTP status code](https://en.wikipedia.org/wiki/List_of_HTTP_status_codes#4xx_client_errors)
20        code: u16,
21        /// the error message returned by the server
22        message: String,
23    },
24}
25
26/// A result with a [`clio::Error`](Error)
27pub type Result<T> = std::result::Result<T, Error>;

29macro_rules! io_error {
30    ($func_name:ident, $unix:ident, $win:ident => ($kind:ident, $des:literal)) => {
31        // When io_error_more graduates from nightly these can use the right kind directly
32        #[cfg(unix)]
33        pub(crate) fn $func_name() -> Error {
34            Error::Io(IoError::from_raw_os_error(libc::$unix))
35        }
36        #[cfg(windows)]
37        pub(crate) fn $func_name() -> Error {
38            Error::Io(IoError::from_raw_os_error(
39                windows_sys::Win32::Foundation::$win as i32,
40            ))
41        }
42        #[cfg(not(any(unix, windows)))]
43        pub(crate) fn $func_name() -> Error {
44            Error::Io(IoError::new(ErrorKind::$kind, $des))
45        }
46    };
47}

49impl Error {
50    pub(crate) fn to_os_string(&self, path: &OsStr) -> OsString {
51        let mut str = OsString::new();
52        str.push("Error opening ");
53        str.push(path);
54        str.push(": ");
55        str.push(self.to_string());
56        str
57    }
58
59    /// Returns the corresponding [`ErrorKind`] for this error.
60    pub fn kind(&self) -> ErrorKind {
61        match self {
62            Error::Io(err) => err.kind(),
63            #[cfg(feature = "http")]
64            Error::Http { code, message: _ } => match code {
65                404 | 410 => ErrorKind::NotFound,
66                401 | 403 => ErrorKind::PermissionDenied,
67                _ => ErrorKind::Other,
68            },
69        }
70    }
71
72    pub(crate) fn other(message: &'static str) -> Self {
73        Error::Io(IoError::new(ErrorKind::Other, message))
74    }
75
76    io_error!(seek_error, ESPIPE, ERROR_BROKEN_PIPE => (Other, "Cannot seek on stream"));
77    io_error!(dir_error, EISDIR, ERROR_INVALID_NAME => (PermissionDenied, "Is a directory"));
78    io_error!(not_dir_error, ENOTDIR, ERROR_ACCESS_DENIED => (PermissionDenied, "Is not a Directory"));
79    io_error!(permission_error, EACCES, ERROR_ACCESS_DENIED => (PermissionDenied, "Permission denied"));
80    io_error!(not_found_error, ENOENT, ERROR_FILE_NOT_FOUND => (NotFound, "The system cannot find the path specified."));
81}

107impl From<Error> for IoError {
108    fn from(err: Error) -> Self {
109        match err {
110            Error::Io(err) => err,
111            #[cfg(feature = "http")]
112            Error::Http { .. } => IoError::new(err.kind(), err.to_string()),
113        }
114    }
115}

129impl Display for Error {
130    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::result::Result<(), std::fmt::Error> {
131        match self {
132            Error::Io(err) => err.fmt(f),
133            #[cfg(feature = "http")]
134            Error::Http { code, message } => write!(f, "{}: {}", code, message),
135        }
136    }
