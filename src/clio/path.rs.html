16/// A builder for [Input](crate::Input) and [Output](crate::Output).
17///
18/// It is designed to be used to get files related to the one passed in.
19///
20/// e.g. Take an [Input](crate::Input) of `/tmp/foo.svg` and have a default [Output](crate::Output) of `/tmp/foo.png`
21///
22/// ```no_run
23/// use clio::{Input, Output};
24///
25/// let input = Input::new("/tmp/foo.svg")?;
26/// let mut output_path = input.path().clone();
27/// output_path.set_extension("png");
28/// let output = output_path.create()?;
29///
30/// assert_eq!(output.path().as_os_str().to_string_lossy(), "/tmp/foo.png");
31/// # Ok::<(), clio::Error>(())
32/// ```
33/// Unlike [InputPath](crate::InputPath) and [OutputPath](crate::OutputPath) it does not
34/// validate the path until you try creating/opening it.
35///
36/// However you can add extra validation using the [`clap`] parser.
37/// ```
38/// # #[cfg(feature="clap-parse")]{
39/// use clap::Parser;
40/// use clio::ClioPath;
41///
42/// #[derive(Parser)]
43/// struct Opt {
44///     /// path to input file, use '-' for stdin
45///     #[clap(value_parser)]
46///     input_path: ClioPath,
47///
48///     /// path to output file, use '-' for stdout point to directory to use default name
49///     #[clap(value_parser = clap::value_parser!(ClioPath).default_name("out.bin"))]
50///     output_file: ClioPath,
51///
52///     /// path to directory
53///     #[clap(value_parser = clap::value_parser!(ClioPath).exists().is_dir())]
54///     log_dir: ClioPath,
55/// }
56/// # }
57/// ```
58#[derive(Debug, PartialEq, Eq, Clone)]
59pub struct ClioPath {
60    pub(crate) path: ClioPathEnum,
61    pub(crate) atomic: bool,
62}
63
64#[derive(Debug, PartialEq, Eq, Clone)]
65pub enum InOut {
66    In,
67    Out,
68}

70#[derive(Debug, PartialEq, Eq, Clone)]
71pub(crate) enum ClioPathEnum {
72    /// stdin or stdout from a cli arg of `'-'`
73    Std(Option<InOut>),
74    /// a path to local file which may or may not exist
75    Local(PathBuf),
76    #[cfg(feature = "http")]
77    /// a http URL to a file on the web
78    Http(Url),
79}
80
81impl ClioPathEnum {
82    fn new(path: &OsStr, io: Option<InOut>) -> Result<Self> {
83        #[cfg(feature = "http")]
84        if is_http(path) {
85            return Ok(ClioPathEnum::Http(try_to_url(path)?));
86        }
87
88        if path == "-" {
89            Ok(ClioPathEnum::Std(io))
90        } else {
91            Ok(ClioPathEnum::Local(path.into()))
92        }
93    }
94}

124        ClioPath {

133    pub(crate) fn with_path_mut<F, O>(&mut self, update: F) -> O
134    where
135        O: Default,
136        F: FnOnce(&mut PathBuf) -> O,
137    {
138        match &mut self.path {
139            ClioPathEnum::Std(_) => O::default(),
140            ClioPathEnum::Local(path) => update(path),
141            #[cfg(feature = "http")]
142            ClioPathEnum::Http(url) => {
143                let mut path = Path::new(url.path()).to_owned();
144                let r = update(&mut path);
145                url.set_path(&path.to_string_lossy());
146                r
147            }
148        }
149    }

151    /// Updates [`self.file_name`](Path::file_name) to `file_name`.
152    ///
153    /// see [`PathBuf::set_file_name`] for more details
154    ///
155    /// # Examples
156    ///
157    /// ```
158    /// use clio::ClioPath;
159    ///
160    /// let mut buf = ClioPath::new("/")?;
161    /// assert!(buf.file_name() == None);
162    /// buf.set_file_name("bar");
163    /// assert!(buf == ClioPath::new("/bar")?);
164    /// assert!(buf.file_name().is_some());
165    /// buf.set_file_name("baz.txt");
166    /// assert!(buf == ClioPath::new("/baz.txt")?);

168    ///     let mut p = ClioPath::new("https://example.com/bar.html?x=y#p2")?;
169    ///     p.set_file_name("baz.txt");
170    ///     assert_eq!(Some("https://example.com/baz.txt?x=y#p2"), p.as_os_str().to_str());  
171    /// }
172    ///
173    /// # Ok::<(), clio::Error>(())
174    /// ```
175    pub fn set_file_name<S: AsRef<OsStr>>(&mut self, file_name: S) {
176        self.with_path_mut(|path| path.set_file_name(file_name))
177    }

208    /// Adds an extension to the end of the [`self.file_name`](Path::file_name).
209    ///
210    /// # Examples
211    ///
212    /// ```
213    /// use clio::ClioPath;
214    ///
215    /// let mut p = ClioPath::new("/tmp/log.txt")?;
216    ///
217    /// p.add_extension("gz");
218    /// assert_eq!(ClioPath::new("/tmp/log.txt.gz")?, p);
219    /// # Ok::<(), clio::Error>(())
220    /// ```
221    ///
222    /// ```
223    /// use clio::ClioPath;
224    ///
225    /// let mut p = ClioPath::new("/tmp/log")?;
226    /// p.add_extension("gz");
227    /// assert_eq!(ClioPath::new("/tmp/log.gz")?, p);
228    /// # Ok::<(), clio::Error>(())
229    /// ```
230    pub fn add_extension<S: AsRef<OsStr>>(&mut self, extension: S) -> bool {
231        if self.file_name().is_some() && !self.ends_with_slash() {
232            if let Some(existing) = self.extension() {
233                let mut existing = existing.to_os_string();
234                existing.push(".");
235                existing.push(extension);
236                self.with_path_mut(|path| path.set_extension(existing))
237            } else {
238                self.with_path_mut(|path| path.set_extension(extension))
239            }
240        } else {
241            false
242        }
243    }

268    /// Creates an owned [`ClioPath`] with `path` adjoined to `self`.
269    ///
270    /// If `path` is absolute, it replaces the current path.
271    ///
272    /// See [`PathBuf::push`] for more details on what it means to adjoin a path.
273    ///
274    /// # Examples
275    ///
276    /// ```
277    /// use clio::ClioPath;
278    ///
279    /// assert_eq!(ClioPath::new("/etc")?.join("passwd"), ClioPath::new("/etc/passwd")?);
280    /// assert_eq!(ClioPath::new("/etc")?.join("/bin/sh"), ClioPath::new("/bin/sh")?);
281    ///
282    /// #[cfg(feature = "http")] {
283    ///     let mut p = ClioPath::new("https://example.com/tmp?x=y#p2")?;
284    ///     p.push("file.bk");
285    ///     assert_eq!(
286    ///         ClioPath::new("https://example.com/tmp?x=y#p2")?.join("file.bk"),
287    ///         ClioPath::new("https://example.com/tmp/file.bk?x=y#p2")?);
288    /// }
289    /// # Ok::<(), clio::Error>(())
290    /// ```
291    pub fn join<P: AsRef<Path>>(&mut self, path: P) -> Self {
292        let mut new = self.clone();
293        new.push(path);
294        new
295    }

297    /// Returns true if this path is stdin/stout i.e. it was created with `-`

302    /// Returns true if this [`is_std`](Self::is_std) and it would connect to a tty
303    pub fn is_tty(&self) -> bool {
304        match self.path {
305            ClioPathEnum::Std(Some(InOut::In)) => std::io::stdin().is_terminal(),
306            ClioPathEnum::Std(Some(InOut::Out)) => std::io::stdout().is_terminal(),
307            ClioPathEnum::Std(None) => {
308                std::io::stdin().is_terminal() || std::io::stdout().is_terminal()
309            }
310            _ => false,
311        }
312    }

315    /// as opposed to point to stdin/stout or a URL
316    pub fn is_local(&self) -> bool {
317        matches!(self.path, ClioPathEnum::Local(_))

320    pub(crate) fn is_fifo(&self) -> bool {
321        match &self.path {
322            ClioPathEnum::Local(path) => {
323                if let Ok(meta) = path.metadata() {
324                    is_fifo(&meta)
325                } else {
326                    false
327                }
328            }
329            ClioPathEnum::Std(_) => true,
330            #[cfg(feature = "http")]
331            ClioPathEnum::Http(_) => false,
332        }
333    }

335    /// Returns `true` if this path ends with a `/`
336    ///
337    /// A trailing slash is often used by command line arguments
338    /// to refer to a directory that may not exist.
339    /// e.g. `cp foo /tmp/`
340    pub fn ends_with_slash(&self) -> bool {
341        cfg_if::cfg_if! {
342            if #[cfg(unix)] {
343                use std::os::unix::ffi::OsStrExt;
344                self.path().as_os_str().as_bytes().ends_with(b"/")
345            } else if #[cfg(windows)] {
346                use std::os::windows::ffi::OsStrExt;
347                self.path().as_os_str().encode_wide().last() == Some('/' as u16)
348            } else {
349                self.path().as_os_str().to_string_lossy().ends_with("/")
350            }
351        }
352    }

354    /// If this is a folder returns all the files that match the filter found by looking recursively
355    /// Otherwise returns just this path
356    /// ```no_run
357    /// use clio::has_extension;
358    /// use clio::ClioPath;
359    ///
360    /// let dir = ClioPath::new("/tmp/foo")?;
361    /// for txt_file in dir.files(has_extension("txt"))? {
362    ///     txt_file.open()?;
363    /// }
364    /// # Ok::<(), clio::Error>(())
365    /// ```
366    pub fn files<P>(self, mut predicate: P) -> Result<Vec<ClioPath>>
367    where
368        P: FnMut(&ClioPath) -> bool,
369    {
370        if self.is_local() {
371            let mut result = vec![];
372            for entry in WalkDir::new(self.path()).follow_links(true) {
373                let entry = entry?;
374                if entry.file_type().is_file() {
375                    let path = ClioPath::local(entry.into_path());
376                    if predicate(&path) {
377                        result.push(path);
378                    }
379                }
380            }
381            Ok(result)
382        } else {
383            Ok(vec![self])
384        }
385    }

407    /// A path represented by this [`ClioPath`]
408    /// If it is `-` and it is no known if it is in or out then the path will be `-`
409    /// If it is `-` and it is known to be in/out then it will be the pseudo device  e.g `/dev/stdin`
410    /// If it is a url it will be the path part of the url
411    /// ```
412    /// use clio::{ClioPath, OutputPath};
413    /// use std::path::Path;
414    ///
415    /// let p = ClioPath::new("-")?;
416    ///
417    /// assert_eq!(Path::new("-"), p.path());
418    ///
419    /// let stdout = OutputPath::new("-")?;
420    /// let p:&ClioPath = stdout.path();
421    /// assert_eq!(Path::new("/dev/stdout"), p.path());
422    ///
423    /// #[cfg(feature = "http")] {
424    ///     let p = ClioPath::new("https://example.com/foo/bar.html?x=y#p2")?;
425    ///
426    ///     assert_eq!(Path::new("/foo/bar.html"), p.path());  
427    /// }
428    /// # Ok::<(), clio::Error>(())
429    /// ```
430    pub fn path(&self) -> &Path {
431        match &self.path {
432            ClioPathEnum::Std(None) => Path::new("-"),
433            ClioPathEnum::Std(Some(InOut::In)) => Path::new("/dev/stdin"),
434            ClioPathEnum::Std(Some(InOut::Out)) => Path::new("/dev/stdout"),
435            ClioPathEnum::Local(path) => path.as_path(),
436            #[cfg(feature = "http")]
437            ClioPathEnum::Http(url) => Path::new(url.path()),
438        }
439    }

441    pub(crate) fn safe_parent(&self) -> Option<&Path> {
442        match &self.path {
443            ClioPathEnum::Local(path) => {
444                let parent = path.parent()?;
445                if parent == Path::new("") {
446                    Some(Path::new("."))
447                } else {
448                    Some(parent)
449                }
450            }
451            _ => None,
452        }
453    }
