21pub fn load_accounts<
22    EVM: EvmTr<Precompiles: PrecompileProvider<EVM::Context>>,
23    ERROR: From<<<EVM::Context as ContextTr>::Db as Database>::Error>,
24>(
25    evm: &mut EVM,
26) -> Result<(), ERROR> {
27    let (context, precompiles) = evm.ctx_precompiles();
28
29    let gen_spec = context.cfg().spec();
30    let spec = gen_spec.clone().into();
31    // sets eth spec id in journal
32    context.journal().set_spec_id(spec);
33    let precompiles_changed = precompiles.set_spec(gen_spec);
34    let empty_warmed_precompiles = context.journal().precompile_addresses().is_empty();
35
36    if precompiles_changed || empty_warmed_precompiles {
37        // load new precompile addresses into journal.
38        // When precompiles addresses are changed we reset the warmed hashmap to those new addresses.
39        context
40            .journal()
41            .warm_precompiles(precompiles.warm_addresses().collect());
42    }
43
44    // Load coinbase
45    // EIP-3651: Warm COINBASE. Starts the `COINBASE` address warm
46    if spec.is_enabled_in(SpecId::SHANGHAI) {
47        let coinbase = context.block().beneficiary();
48        context.journal().warm_account(coinbase);
49    }
50
51    // Load access list
52    let (tx, journal) = context.tx_journal();
53    // legacy is only tx type that does not have access list.
54    if tx.tx_type() != TransactionType::Legacy {
55        if let Some(access_list) = tx.access_list() {
56            for item in access_list {
57                let address = item.address();
58                let mut storage = item.storage_slots().peekable();
59                if storage.peek().is_none() {
60                    journal.warm_account(*address);
61                } else {
62                    journal.warm_account_and_storage(
63                        *address,
64                        storage.map(|i| StorageKey::from_be_bytes(i.0)),
65                    )?;
66                }
67            }
68        }
69    }
70
71    Ok(())
72}

74#[inline]
75pub fn validate_account_nonce_and_code(
76    caller_info: &mut AccountInfo,
77    tx_nonce: u64,
78    bump_nonce: bool,
79    is_eip3607_disabled: bool,
80    is_nonce_check_disabled: bool,
81) -> Result<(), InvalidTransaction> {
82    // EIP-3607: Reject transactions from senders with deployed code
83    // This EIP is introduced after london but there was no collision in past
84    // so we can leave it enabled always
85    if !is_eip3607_disabled {
86        let bytecode = match caller_info.code.as_ref() {
87            Some(code) => code,
88            None => &Bytecode::default(),
89        };
90        // Allow EOAs whose code is a valid delegation designation,
91        // i.e. 0xef0100 || address, to continue to originate transactions.
92        if !bytecode.is_empty() && !bytecode.is_eip7702() {
93            return Err(InvalidTransaction::RejectCallerWithCode);
94        }
95    }
96
97    // Check that the transaction's nonce is correct
98    if !is_nonce_check_disabled {
99        let tx = tx_nonce;
100        let state = caller_info.nonce;
101        match tx.cmp(&state) {
102            Ordering::Greater => {
103                return Err(InvalidTransaction::NonceTooHigh { tx, state });
104            }
105            Ordering::Less => {
106                return Err(InvalidTransaction::NonceTooLow { tx, state });
107            }
108            _ => {}
109        }
110    }
111
112    // Bump the nonce for calls. Nonce for CREATE will be bumped in `handle_create`.
113    if bump_nonce {
114        // Nonce is already checked
115        caller_info.nonce = caller_info.nonce.saturating_add(1);
116    }
117
118    Ok(())
119}

121#[inline]
122pub fn validate_against_state_and_deduct_caller<
123    CTX: ContextTr,
124    ERROR: From<InvalidTransaction> + From<<CTX::Db as Database>::Error>,
125>(
126    context: &mut CTX,
127) -> Result<(), ERROR> {
128    let basefee = context.block().basefee() as u128;
129    let blob_price = context.block().blob_gasprice().unwrap_or_default();
130    let is_balance_check_disabled = context.cfg().is_balance_check_disabled();
131    let is_eip3607_disabled = context.cfg().is_eip3607_disabled();
132    let is_nonce_check_disabled = context.cfg().is_nonce_check_disabled();
133
134    let (tx, journal) = context.tx_journal();
135
136    // Load caller's account.
137    let caller_account = journal.load_account_code(tx.caller())?.data;
138
139    validate_account_nonce_and_code(
140        &mut caller_account.info,
141        tx.nonce(),
142        tx.kind().is_call(),
143        is_eip3607_disabled,
144        is_nonce_check_disabled,
145    )?;
146
147    let max_balance_spending = tx.max_balance_spending()?;
148
149    // Check if account has enough balance for `gas_limit * max_fee`` and value transfer.
150    // Transfer will be done inside `*_inner` functions.
151    if is_balance_check_disabled {
152        // Make sure the caller's balance is at least the value of the transaction.
153        caller_account.info.balance = caller_account.info.balance.max(tx.value());
154    } else if max_balance_spending > caller_account.info.balance {
155        return Err(InvalidTransaction::LackOfFundForMaxFee {
156            fee: Box::new(max_balance_spending),
157            balance: Box::new(caller_account.info.balance),
158        }
159        .into());
160    } else {
161        let effective_balance_spending = tx
162            .effective_balance_spending(basefee, blob_price)
163            .expect("effective balance is always smaller than max balance so it can't overflow");
164
165        // subtracting max balance spending with value that is going to be deducted later in the call.
166        let gas_balance_spending = effective_balance_spending - tx.value();
167
168        caller_account.info.balance = caller_account
169            .info
170            .balance
171            .saturating_sub(gas_balance_spending);
172    }
173
174    // Touch account so we know it is changed.
175    caller_account.mark_touch();
176    Ok(())
177}

179/// Apply EIP-7702 auth list and return number gas refund on already created accounts.
180#[inline]
181pub fn apply_eip7702_auth_list<
182    CTX: ContextTr,
183    ERROR: From<InvalidTransaction> + From<<CTX::Db as Database>::Error>,
184>(
185    context: &mut CTX,
186) -> Result<u64, ERROR> {
187    let tx = context.tx();
188    // Return if there is no auth list.
189    if tx.tx_type() != TransactionType::Eip7702 {
190        return Ok(0);
191    }
192
193    let chain_id = context.cfg().chain_id();
194    let (tx, journal) = context.tx_journal();
195
196    let mut refunded_accounts = 0;
197    for authorization in tx.authorization_list() {
198        // 1. Verify the chain id is either 0 or the chain's current ID.
199        let auth_chain_id = authorization.chain_id();
200        if !auth_chain_id.is_zero() && auth_chain_id != U256::from(chain_id) {
201            continue;
202        }
203
204        // 2. Verify the `nonce` is less than `2**64 - 1`.
205        if authorization.nonce() == u64::MAX {
206            continue;
207        }
208
209        // recover authority and authorized addresses.
210        // 3. `authority = ecrecover(keccak(MAGIC || rlp([chain_id, address, nonce])), y_parity, r, s]`
211        let Some(authority) = authorization.authority() else {
212            continue;
213        };
214
215        // warm authority account and check nonce.
216        // 4. Add `authority` to `accessed_addresses` (as defined in [EIP-2929](./eip-2929.md).)
217        let mut authority_acc = journal.load_account_code(authority)?;
218
219        // 5. Verify the code of `authority` is either empty or already delegated.
220        if let Some(bytecode) = &authority_acc.info.code {
221            // if it is not empty and it is not eip7702
222            if !bytecode.is_empty() && !bytecode.is_eip7702() {
223                continue;
224            }
225        }
226
227        // 6. Verify the nonce of `authority` is equal to `nonce`. In case `authority` does not exist in the trie, verify that `nonce` is equal to `0`.
228        if authorization.nonce() != authority_acc.info.nonce {
229            continue;
230        }
231
232        let loaded_not_existing = authority_acc.is_loaded_as_not_existing();
233        let is_not_touched = !authority_acc.is_touched();
234        let loaded_not_existing_not_touched = loaded_not_existing && is_not_touched;
235        // 7. Add `PER_EMPTY_ACCOUNT_COST - PER_AUTH_BASE_COST` gas to the global refund counter if `authority` exists in the trie.
236        if !(authority_acc.is_empty() && loaded_not_existing_not_touched) {
237            refunded_accounts += 1;
238        }
239
240        // 8. Set the code of `authority` to be `0xef0100 || address`. This is a delegation designation.
241        //  * As a special case, if `address` is `0x0000000000000000000000000000000000000000` do not write the designation.
242        //    Clear the accounts code and reset the account's code hash to the empty hash `0xc5d2460186f7233c927e7db2dcc703c0e500b653ca82273b7bfad8045d85a470`.
243        let address = authorization.address();
244        let (bytecode, hash) = if address.is_zero() {
245            (Bytecode::default(), KECCAK_EMPTY)
246        } else {
247            let bytecode = Bytecode::new_eip7702(address);
248            let hash = bytecode.hash_slow();
249            (bytecode, hash)
250        };
251        authority_acc.info.code_hash = hash;
252        authority_acc.info.code = Some(bytecode);
253
254        // 9. Increase the nonce of `authority` by one.
255        authority_acc.info.nonce = authority_acc.info.nonce.saturating_add(1);
256        authority_acc.mark_touch();
257    }
258
259    let refunded_gas =
260        refunded_accounts * (eip7702::PER_EMPTY_ACCOUNT_COST - eip7702::PER_AUTH_BASE_COST);
261
262    Ok(refunded_gas)
