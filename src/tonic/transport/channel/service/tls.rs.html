22#[derive(Clone)]
23pub(crate) struct TlsConnector {

25    domain: Arc<ServerName<'static>>,

30impl TlsConnector {
31    #[allow(clippy::too_many_arguments)]
32    pub(crate) fn new(
33        ca_certs: Vec<Certificate>,
34        trust_anchors: Vec<TrustAnchor<'static>>,
35        identity: Option<Identity>,
36        domain: &str,
37        assume_http2: bool,
38        use_key_log: bool,
39        timeout: Option<Duration>,
40        #[cfg(feature = "tls-native-roots")] with_native_roots: bool,
41        #[cfg(feature = "tls-webpki-roots")] with_webpki_roots: bool,
42    ) -> Result<Self, crate::BoxError> {
43        fn with_provider(
44            provider: Arc<crypto::CryptoProvider>,
45        ) -> ConfigBuilder<ClientConfig, WantsVerifier> {
46            ClientConfig::builder_with_provider(provider)
47                .with_safe_default_protocol_versions()
48                .unwrap()
49        }
50
51        #[allow(unreachable_patterns)]
52        let builder = match crypto::CryptoProvider::get_default() {
53            Some(provider) => with_provider(provider.clone()),
54            #[cfg(feature = "tls-ring")]
55            None => with_provider(Arc::new(crypto::ring::default_provider())),
56            #[cfg(feature = "tls-aws-lc")]
57            None => with_provider(Arc::new(crypto::aws_lc_rs::default_provider())),
58            // somehow tls is enabled, but neither of the crypto features are enabled.
59            _ => ClientConfig::builder(),
60        };
61
62        let mut roots = RootCertStore::from_iter(trust_anchors);
63
64        #[cfg(feature = "tls-native-roots")]
65        if with_native_roots {
66            let rustls_native_certs::CertificateResult { certs, errors, .. } =
67                rustls_native_certs::load_native_certs();
68            if !errors.is_empty() {
69                tracing::debug!("errors occurred when loading native certs: {errors:?}");
70            }
71            if certs.is_empty() {
72                return Err(TlsError::NativeCertsNotFound.into());
73            }
74            roots.add_parsable_certificates(certs);
75        }
76
77        #[cfg(feature = "tls-webpki-roots")]
78        if with_webpki_roots {
79            roots.extend(webpki_roots::TLS_SERVER_ROOTS.iter().cloned());
80        }
81
82        for cert in ca_certs {
83            roots.add_parsable_certificates(convert_certificate_to_pki_types(&cert)?);
84        }
85
86        let builder = builder.with_root_certificates(roots);
87        let mut config = match identity {
88            Some(identity) => {
89                let (client_cert, client_key) = convert_identity_to_pki_types(&identity)?;
90                builder.with_client_auth_cert(client_cert, client_key)?
91            }
92            None => builder.with_no_client_auth(),
93        };
94
95        if use_key_log {
96            config.key_log = Arc::new(tokio_rustls::rustls::KeyLogFile::new());
97        }
98
99        config.alpn_protocols.push(ALPN_H2.into());
100        Ok(Self {
101            config: Arc::new(config),
102            domain: Arc::new(ServerName::try_from(domain)?.to_owned()),
103            assume_http2,
104            timeout,
105        })
106    }
107
108    pub(crate) async fn connect<I>(&self, io: I) -> Result<BoxedIo, crate::BoxError>
109    where
110        I: AsyncRead + AsyncWrite + Send + Unpin + 'static,
111    {
112        let conn_fut =
113            RustlsConnector::from(self.config.clone()).connect(self.domain.as_ref().to_owned(), io);
114        let io = match self.timeout {
115            Some(timeout) => time::timeout(timeout, conn_fut)
116                .await
117                .map_err(|_| TlsError::HandshakeTimeout)?,
118            None => conn_fut.await,
119        }?;
120
121        // Generally we require ALPN to be negotiated, but if the user has
122        // explicitly set `assume_http2` to true, we'll allow it to be missing.
123        let (_, session) = io.get_ref();
124        let alpn_protocol = session.alpn_protocol();
125        if !(alpn_protocol == Some(ALPN_H2) || self.assume_http2) {
126            return Err(TlsError::H2NotNegotiated.into());
127        }
128        Ok(BoxedIo::new(TokioIo::new(io)))
129    }
130}
