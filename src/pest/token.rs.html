10use crate::position::Position;
11
12/// A token generated by a `Parser`.
13#[derive(Clone, Debug, Eq, Hash, PartialEq)]
14pub enum Token<'i, R> {
15    /// The starting `Position` of a matched `Rule`
16    Start {
17        /// matched rule
18        rule: R,
19        /// starting position
20        pos: Position<'i>,
21    },
22    /// The ending `Position` of a matched `Rule`
23    End {
24        /// matched rule
25        rule: R,
26        /// ending position
27        pos: Position<'i>,
28    },
29}
