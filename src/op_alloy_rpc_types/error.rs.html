22//! Superchain consensus errors
23
24use derive_more;
25
26/// Supervisor data availability error codes.
27///
28/// Specs: <https://specs.optimism.io/interop/supervisor.html#protocol-specific-error-codes>
29#[derive(thiserror::Error, Debug, Clone, Copy, PartialEq, Eq, derive_more::TryFrom)]
30#[repr(i32)]
31#[try_from(repr)]
32pub enum SuperchainDAError {
33    //-------------------------------- -3205XX NOT_FOUND errors --------------------------------//
34    /// Happens when we try to retrieve data that is not available (pruned).
35    /// It may also happen if we erroneously skip data, that was not considered a conflict, if the
36    /// DB is corrupted.
37    #[error("data was skipped or pruned and is not available")]
38    SkippedData = -320500,
39
40    /// Happens when a chain is unknown, not in the dependency set.
41    #[error("unsupported chain id")]
42    UnknownChain = -320501,
43
44    //--------------------------------- -3206XX ALREADY_EXISTS ---------------------------------//
45    /// Happens when we know for sure that there is different canonical data.
46    #[error("conflicting data exists in the database")]
47    ConflictingData = -320600,
48
49    /// Happens when data is accepted as compatible, but did not change anything.
50    /// This happens when a node is deriving an L2 block we already know of being
51    /// derived from the given source,
52    /// but without path to skip forward to newer source blocks without doing the known
53    /// derivation work first.
54    #[error("data is already known and didn't change anything")]
55    IneffectiveData = -320601,
56
57    //-------------------------- -3209XX FAILED_PRECONDITION errors ----------------------------//
58    /// Happens when you try to add data to the DB, but it does not actually fit onto
59    /// the latest data.
60    /// (by being too old or new).
61    #[error("data is out of order (too old or new)")]
62    OutOfOrder = -320900,
63
64    /// Happens when something was assumed from the DB, but then invalidated due to e.g. a reorg.
65    #[error("invalidated read")]
66    InvalidatedRead = -32901,
67
68    /// Happens when we know for sure that a replacement block is needed before progress
69    /// can be made.
70    #[error("waiting for replacement block before progress can be made")]
71    AwaitingReplacement = -320901,
72
73    //--------------------------------- -3210XX ABORTED errors ---------------------------------//
74    /// Happens when we fail to rewind the chain (reorg response).
75    #[error("rewind failed")]
76    RewindFailed = -321000,
77
78    // -3211XX OUT_OF_RANGE errors
79    /// Happens when data is accessed, but access is not allowed, because of a limited
80    /// scope.
81    /// E.g. when limiting scope to L2 blocks derived from a specific subset of the L1
82    /// chain.
83    #[error("data access not allowed due to limited scope")]
84    OutOfScope = -321100,
85
86    //------------------------------ -3212XX UNIMPLEMENTED errors ------------------------------//
87    /// Happens when you try to get the previous block of the first block.
88    /// E.g. when trying to determine the previous source block for the first L1 block
89    /// in the database.
90    #[error("cannot get parent of first block in database")]
91    NoParentForFirstBlock = -321200,
92
93    //------------------------------- -3214XX UNAVAILABLE errors -------------------------------//
94    /// Happens when data is just not yet available.
95    #[error("data is not yet available (from the future)")]
96    FutureData = -321401,
97
98    //-------------------------------- -3215XX DATA_LOSS errors --------------------------------//
99    /// Happens when we search the DB, know the data may be there, but is not (e.g.
100    /// different revision).
101    #[error("data may exist but was not found (possibly different revision)")]
102    MissedData = -321500,
103
104    /// Happens when the underlying DB has some I/O issue.
105    #[error("underlying database has I/O issues or is corrupted")]
106    DataCorruption = -321501,
107}
108
109#[cfg(feature = "jsonrpsee")]
110impl From<SuperchainDAError> for jsonrpsee::types::ErrorObjectOwned {
111    fn from(err: SuperchainDAError) -> Self {
112        use crate::alloc::string::ToString;
113
114        jsonrpsee::types::ErrorObjectOwned::owned(err as i32, err.to_string(), None::<()>)
115    }
116}
