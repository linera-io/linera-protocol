13/// OP Transaction type
14#[derive(
15    Clone, Debug, PartialEq, Eq, Serialize, Deserialize, derive_more::Deref, derive_more::DerefMut,
16)]
17#[cfg_attr(all(any(test, feature = "arbitrary"), feature = "k256"), derive(arbitrary::Arbitrary))]

20    into = "tx_serde::TransactionSerdeHelper<T>",

24    /// Ethereum Transaction Types
25    #[deref]

28
29    /// Nonce for deposit transactions. Only present in RPC responses.
30    pub deposit_nonce: Option<u64>,
31
32    /// Deposit receipt version for deposit transactions post-canyon
33    pub deposit_receipt_version: Option<u64>,
34}

36impl<T: OpTransaction + TransactionTrait> Transaction<T> {
37    /// Converts a consensus `tx` with an additional context `tx_info` into an RPC [`Transaction`].
38    pub fn from_transaction(tx: Recovered<T>, tx_info: OpTransactionInfo) -> Self {
39        let base_fee = tx_info.inner.base_fee;
40        let effective_gas_price = if tx.is_deposit() {
41            // For deposits, we must always set the `gasPrice` field to 0 in rpc
42            // deposit tx don't have a gas price field, but serde of `Transaction` will take care of
43            // it
44            0
45        } else {
46            base_fee
47                .map(|base_fee| {
48                    tx.effective_tip_per_gas(base_fee).unwrap_or_default() + base_fee as u128
49                })
50                .unwrap_or_else(|| tx.max_fee_per_gas())
51        };
52
53        Self {
54            inner: alloy_rpc_types_eth::Transaction {
55                inner: tx,
56                block_hash: tx_info.inner.block_hash,
57                block_number: tx_info.inner.block_number,
58                transaction_index: tx_info.inner.index,
59                effective_gas_price: Some(effective_gas_price),
60            },
61            deposit_nonce: tx_info.deposit_meta.deposit_nonce,
62            deposit_receipt_version: tx_info.deposit_meta.deposit_receipt_version,
63        }
64    }
65}

203mod tx_serde {
204    //! Helper module for serializing and deserializing OP [`Transaction`].
205    //!
206    //! This is needed because we might need to deserialize the `from` field into both
207    //! [`alloy_consensus::transaction::Recovered::signer`] which resides in
208    //! [`alloy_rpc_types_eth::Transaction::inner`] and [`op_alloy_consensus::TxDeposit::from`].
209    //!
210    //! Additionally, we need similar logic for the `gasPrice` field
211    use super::*;
212    use alloy_consensus::transaction::Recovered;
213    use op_alloy_consensus::OpTransaction;
214    use serde::de::Error;
215
216    /// Helper struct which will be flattened into the transaction and will only contain `from`
217    /// field if inner [`OpTxEnvelope`] did not consume it.
218    #[derive(Serialize, Deserialize)]
219    struct OptionalFields {
220        #[serde(default, skip_serializing_if = "Option::is_none")]
221        from: Option<Address>,
222        #[serde(
223            default,
224            rename = "gasPrice",
225            skip_serializing_if = "Option::is_none",
226            with = "alloy_serde::quantity::opt"
227        )]
228        effective_gas_price: Option<u128>,
229        #[serde(
230            default,
231            rename = "nonce",
232            skip_serializing_if = "Option::is_none",
233            with = "alloy_serde::quantity::opt"
234        )]
235        deposit_nonce: Option<u64>,
236    }
237
238    #[derive(Serialize, Deserialize)]
239    #[serde(rename_all = "camelCase")]
240    pub(crate) struct TransactionSerdeHelper<T> {
241        #[serde(flatten)]
242        inner: T,
243        #[serde(default)]
244        block_hash: Option<BlockHash>,
245        #[serde(default, with = "alloy_serde::quantity::opt")]
246        block_number: Option<u64>,
247        #[serde(default, with = "alloy_serde::quantity::opt")]
248        transaction_index: Option<u64>,
249        #[serde(
250            default,
251            skip_serializing_if = "Option::is_none",
252            with = "alloy_serde::quantity::opt"
253        )]
254        deposit_receipt_version: Option<u64>,
255
256        #[serde(flatten)]
257        other: OptionalFields,
258    }
259
260    impl<T: TransactionTrait + OpTransaction> From<Transaction<T>> for TransactionSerdeHelper<T> {
261        fn from(value: Transaction<T>) -> Self {
262            let Transaction {
263                inner:
264                    alloy_rpc_types_eth::Transaction {
265                        inner,
266                        block_hash,
267                        block_number,
268                        transaction_index,
269                        effective_gas_price,
270                    },
271                deposit_receipt_version,
272                deposit_nonce,
273            } = value;
274
275            // if inner transaction is a deposit, then don't serialize `from` directly
276            let from = if inner.as_deposit().is_some() { None } else { Some(inner.signer()) };
277
278            // if inner transaction has its own `gasPrice` don't serialize it in this struct.
279            let effective_gas_price = effective_gas_price.filter(|_| inner.gas_price().is_none());
280
281            Self {
282                inner: inner.into_inner(),
283                block_hash,
284                block_number,
285                transaction_index,
286                deposit_receipt_version,
287                other: OptionalFields { from, effective_gas_price, deposit_nonce },
288            }
289        }
290    }
291
292    impl<T: TransactionTrait + OpTransaction> TryFrom<TransactionSerdeHelper<T>> for Transaction<T> {
293        type Error = serde_json::Error;
294
295        fn try_from(value: TransactionSerdeHelper<T>) -> Result<Self, Self::Error> {
296            let TransactionSerdeHelper {
297                inner,
298                block_hash,
299                block_number,
300                transaction_index,
301                deposit_receipt_version,
302                other,
303            } = value;
304
305            // Try to get `from` field from inner envelope or from `MaybeFrom`, otherwise return
306            // error
307            let from = if let Some(from) = other.from {
308                from
309            } else {
310                inner
311                    .as_deposit()
312                    .map(|v| v.from)
313                    .ok_or_else(|| serde_json::Error::custom("missing `from` field"))?
314            };
315
316            // Only serialize deposit_nonce if inner transaction is deposit to avoid duplicated keys
317            let deposit_nonce = other.deposit_nonce.filter(|_| inner.is_deposit());
318
319            let effective_gas_price = other.effective_gas_price.or(inner.gas_price());
320
321            Ok(Self {
322                inner: alloy_rpc_types_eth::Transaction {
323                    inner: Recovered::new_unchecked(inner, from),
324                    block_hash,
325                    block_number,
326                    transaction_index,
327                    effective_gas_price,
328                },
329                deposit_receipt_version,
330                deposit_nonce,
331            })
332        }
333    }
334}
