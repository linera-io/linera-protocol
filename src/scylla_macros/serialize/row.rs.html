10#[derive(FromAttributes)]
11#[darling(attributes(scylla))]
12struct Attributes {
13    #[darling(rename = "crate")]
14    crate_path: Option<syn::Path>,
15
16    #[darling(default)]
17    flavor: Flavor,
18
19    // If true, then the type checking code won't verify the column names.
20    // Columns will be matched to struct fields based solely on the order.
21    //
22    // This annotation only works if `enforce_order` flavor is specified.
23    #[darling(default)]
24    skip_name_checks: bool,
25}
26
27impl Attributes {
28    fn crate_path(&self) -> syn::Path {
29        self.crate_path
30            .as_ref()
31            .map(|p| parse_quote!(#p::_macro_internal))
32            .unwrap_or_else(|| parse_quote!(::scylla::_macro_internal))
33    }
34}

36struct Field {
37    ident: syn::Ident,
38    typ: syn::Type,
39    attrs: FieldAttributes,
40}
41
42impl Field {
43    fn column_name(&self) -> String {
44        match &self.attrs.rename {
45            Some(name) => name.clone(),
46            None => self.ident.to_string(),
47        }
48    }
49}

51#[derive(FromAttributes)]
52#[darling(attributes(scylla))]
53struct FieldAttributes {
54    // If set, then serializes from the column with this particular name
55    // instead of the Rust field name.
56    rename: Option<String>,
57
58    // If set, then the field is inlined into the struct, that is, it is
59    // serialized using `SerializeRow`, with the serialized fields flattened
60    // into (possibly multiple) columns as if they were in the parent struct.
61    #[darling(default)]
62    flatten: bool,
63
64    // If true, then the field is not serialized at all, but simply ignored.
65    // All other attributes are ignored.
66    #[darling(default)]
67    skip: bool,
68}
69
70struct Context {
71    attributes: Attributes,
72    fields: Vec<Field>,
73    struct_name: syn::Ident,
74    generics: syn::Generics,
75}

77pub(crate) fn derive_serialize_row(tokens_input: TokenStream) -> Result<syn::ItemImpl, syn::Error> {
78    let input: syn::DeriveInput = syn::parse(tokens_input)?;
79    let struct_name = input.ident.clone();
80    let named_fields = crate::parser::parse_named_fields(&input, "SerializeRow")?;
81    let (impl_generics, ty_generics, where_clause) = input.generics.split_for_impl();
82    let attributes = Attributes::from_attributes(&input.attrs)?;
83
84    let crate_path = attributes.crate_path();
85    let implemented_trait: syn::Path = parse_quote!(#crate_path::SerializeRow);
86
87    let fields = named_fields
88        .named
89        .iter()
90        .map(|f| {
91            FieldAttributes::from_attributes(&f.attrs).map(|attrs| Field {
92                ident: f.ident.clone().unwrap(),
93                typ: f.ty.clone(),
94                attrs,
95            })
96        })
97        // Filter the fields now instead of at the places that use them later
98        // as it's less error prone - we just filter in one place instead of N places.
99        .filter(|f| f.as_ref().map(|f| !f.attrs.skip).unwrap_or(true))
100        .collect::<Result<_, _>>()?;
101    let ctx = Context {
102        attributes,
103        fields,
104        struct_name: struct_name.clone(),
105        generics: input.generics.clone(),
106    };
107    ctx.validate(&input.ident)?;
108
109    let gen: Box<dyn Generator> = match ctx.attributes.flavor {
110        Flavor::MatchByName => Box::new(ColumnSortingGenerator { ctx: &ctx }),
111        Flavor::EnforceOrder => Box::new(ColumnOrderedGenerator { ctx: &ctx }),
112    };
113
114    let serialize_item = gen.generate_serialize();
115    let is_empty_item = gen.generate_is_empty();
116
117    let res = parse_quote! {
118        #[automatically_derived]
119        impl #impl_generics #implemented_trait for #struct_name #ty_generics #where_clause {
120            #serialize_item
121            #is_empty_item
122        }
123    };
124    Ok(res)
125}
126
127impl Context {
128    fn validate(&self, struct_ident: &syn::Ident) -> Result<(), syn::Error> {
129        let mut errors = darling::Error::accumulator();
130
131        if self.attributes.skip_name_checks {
132            // Skipping name checks is only available in enforce_order mode
133            if self.attributes.flavor != Flavor::EnforceOrder {
134                let err = darling::Error::custom(
135                    "the `skip_name_checks` attribute is only allowed with the `enforce_order` flavor",
136                )
137                .with_span(struct_ident);
138                errors.push(err);
139            }
140
141            // `rename` annotations don't make sense with skipped name checks
142            for field in self.fields.iter() {
143                if field.attrs.rename.is_some() {
144                    let err = darling::Error::custom(
145                        "the `rename` annotations don't make sense with `skip_name_checks` attribute",
146                    )
147                    .with_span(&field.ident);
148                    errors.push(err);
149                }
150            }
151        }
152
153        // Check that no renames are attempted on flattened fields
154        let rename_flatten_errors = self
155            .fields
156            .iter()
157            .filter(|f| f.attrs.flatten && f.attrs.rename.is_some())
158            .map(|f| {
159                darling::Error::custom(
160                    "`rename` and `flatten` annotations do not make sense together",
161                )
162                .with_span(&f.ident)
163            });
164        errors.extend(rename_flatten_errors);
165
166        // Check for name collisions
167        let mut used_names = HashMap::<String, &Field>::new();
168        for field in self.fields.iter() {
169            let column_name = field.column_name();
170            if let Some(other_field) = used_names.get(&column_name) {
171                let other_field_ident = &other_field.ident;
172                let msg = format!("the column / bind marker name `{column_name}` used by this struct field is already used by field `{other_field_ident}`");
173                let err = darling::Error::custom(msg).with_span(&field.ident);
174                errors.push(err);
175            } else {
176                used_names.insert(column_name, field);
177            }
178        }
179
180        errors.finish()?;
181        Ok(())
182    }
183}

190// Generates an implementation of the trait which sorts the columns according
191// to how they are defined in prepared statement metadata.
192struct ColumnSortingGenerator<'a> {
193    ctx: &'a Context,
194}
195
196impl Generator for ColumnSortingGenerator<'_> {
197    fn generate_serialize(&self) -> syn::TraitItemFn {
198        // Serializing by name requires:
199        //
200        // 1. Defining a partial struct: a struct that keeps references to each serializable field and
201        // tracks completion of these fields
202        //
203        // 2. Implement serialization for the partial struct: When serializing a column check if this
204        // column is one of our nonflattened fields, a nested field inside a flattened struct, or
205        // not relevant to this struct. If relevant to this struct, typecheck it.
206        //
207        // 3. Implement SerializeRowByName: Creates an instance of the partial struct.
208        //
209        // 4. Implement SerializeRow: simply forwards to ser::row::ByName which will be in charge of
210        // asking for a partial view of our struct and one by one sending columns to serialize to it
211        // until it is done or an error occurs
212
213        let crate_path = self.ctx.attributes.crate_path();
214        let struct_name = &self.ctx.struct_name;
215        // 1. Defining a partial struct
216        let partial_struct_name = syn::Ident::new(
217            &format!("_{struct_name}ScyllaSerPartial"),
218            struct_name.span(),
219        );
220        let mut partial_generics = self.ctx.generics.clone();
221        let partial_lt: syn::Lifetime = syn::parse_quote!('scylla_ser_partial);
222
223        if !self.ctx.fields.is_empty() {
224            // all fields have to outlive the partial struct lifetime
225            partial_generics
226                .params
227                .iter_mut()
228                .filter_map(|p| match p {
229                    syn::GenericParam::Lifetime(lt) => Some(lt),
230                    _ => None,
231                })
232                .for_each(|lt| {
233                    lt.bounds.push(partial_lt.clone());
234                });
235
236            // now add the partial struct lifetime
237            partial_generics
238                .params
239                .push(syn::GenericParam::Lifetime(syn::LifetimeParam {
240                    attrs: vec![],
241                    lifetime: partial_lt.clone(),
242                    colon_token: None,
243                    bounds: syn::punctuated::Punctuated::new(),
244                }));
245        }
246
247        let flattened: Vec<_> = self.ctx.fields.iter().filter(|f| f.attrs.flatten).collect();
248        let flattened_fields: Vec<_> = flattened.iter().map(|f| &f.ident).collect();
249        let flattened_types: Vec<_> = flattened.iter().map(|f| &f.typ).collect();
250        let flattened_visited_flag_names: Vec<_> = flattened_fields
251            .iter()
252            .map(|ident| format_ident!("__visited_flag_{}", ident))
253            .collect();
254
255        let nonflattened: Vec<_> = self
256            .ctx
257            .fields
258            .iter()
259            .filter(|f| !f.attrs.flatten)
260            .collect();
261        let nonflattened_fields: Vec<_> = nonflattened.iter().map(|f| &f.ident).collect();
262        let nonflattened_visited_flag_names: Vec<_> = nonflattened_fields
263            .iter()
264            .map(|ident| format_ident!("__visited_flag_{}", ident))
265            .collect();
266        let nonflattened_types: Vec<_> = nonflattened.iter().map(|f| &f.typ).collect();
267
268        let partial_struct: syn::ItemStruct = parse_quote! {
269            pub struct #partial_struct_name #partial_generics {
270                #(#nonflattened_fields: &#partial_lt #nonflattened_types,)*
271                #(#flattened_fields: <#flattened_types as #crate_path::SerializeRowByName>::Partial<#partial_lt>,)*
272                #(#flattened_visited_flag_names: bool,)*
273                #(#nonflattened_visited_flag_names: bool,)*
274                remaining_count: usize,
275            }
276        };
277
278        // 2. Implement serialization for the partial struct
279        let nonflattened_columns: Vec<_> = nonflattened.iter().map(|f| f.column_name()).collect();
280        let (impl_generics, ty_generics, where_clause) = self.ctx.generics.split_for_impl();
281        let serialize_field_block: syn::Block = if self.ctx.fields.is_empty() {
282            parse_quote! {{
283                ::std::result::Result::Ok(#crate_path::ser::row::FieldStatus::NotUsed)
284            }}
285        } else {
286            parse_quote! {{
287                match spec.name() {
288                    // first check if the spec name matches a non-flattened column
289                    #(#nonflattened_columns => {
290                        #crate_path::ser::row::serialize_column::<#struct_name #ty_generics>(
291                            &self.#nonflattened_fields, spec, writer,
292                        )?;
293                        if !self.#nonflattened_visited_flag_names {
294                            self.#nonflattened_visited_flag_names = true;
295                            self.remaining_count -= 1;
296                        }
297                    })*
298                    // if not, then check if any flattened field has a column for it
299                    _ => 'flatten_try: {
300                        #({
301                            match <<#flattened_types as #crate_path::SerializeRowByName>::Partial<#partial_lt> as #crate_path::PartialSerializeRowByName>::serialize_field(&mut self.#flattened_fields, spec, writer)? {
302                                // there is a column and the field is done
303                                #crate_path::ser::row::FieldStatus::Done => {
304                                    if !self.#flattened_visited_flag_names {
305                                        self.#flattened_visited_flag_names = true;
306                                        self.remaining_count -= 1;
307                                    }
308                                    break 'flatten_try;
309                                }
310                                // there is a column in this flattened field but we need more
311                                // columns for this field
312                                #crate_path::ser::row::FieldStatus::NotDone => {
313                                    return ::std::result::Result::Ok(#crate_path::ser::row::FieldStatus::NotDone);
314                                }
315                                // there wasn't any column on this field -- try the next one
316                                #crate_path::ser::row::FieldStatus::NotUsed => {}
317                            };
318                        })*
319
320                        // we didn't break out of 'flatten_try so the column didn't match any
321                        // flattened field
322                        return ::std::result::Result::Ok(#crate_path::ser::row::FieldStatus::NotUsed);
323                    }
324                }
325
326                // report if we are done; so that any parent struct knows if this field is done in
327                // case this field is flattened
328                ::std::result::Result::Ok(if self.remaining_count == 0 {
329                    #crate_path::ser::row::FieldStatus::Done
330                } else {
331                    #crate_path::ser::row::FieldStatus::NotDone
332                })
333            }}
334        };
335
336        let (partial_impl_generics, partial_ty_generics, partial_where_clause) =
337            partial_generics.split_for_impl();
338        let partial_serialize: syn::ItemImpl = parse_quote! {
339            impl #partial_impl_generics #crate_path::PartialSerializeRowByName for #partial_struct_name #partial_ty_generics #partial_where_clause {
340                fn serialize_field(
341                    &mut self,
342                    spec: &#crate_path::ColumnSpec,
343                    writer: &mut #crate_path::RowWriter<'_>,
344                ) -> ::std::result::Result<#crate_path::ser::row::FieldStatus, #crate_path::SerializationError> {
345                    #serialize_field_block
346                }
347
348                fn check_missing(self) -> ::std::result::Result<(), #crate_path::SerializationError> {
349                    if self.remaining_count == 0 {
350                        return ::std::result::Result::Ok(());
351                    }
352
353                    #(if !self.#nonflattened_visited_flag_names {
354                        return ::std::result::Result::Err(#crate_path::ser::row::mk_typck_err::<#struct_name #ty_generics>(#crate_path::BuiltinRowTypeCheckErrorKind::NoColumnWithName {
355                            name: <_ as ::std::borrow::ToOwned>::to_owned(#nonflattened_columns),
356                        }))
357                    })*
358
359                    // if what is missing is a flattened field then report that error
360                    #(if !self.#flattened_visited_flag_names {
361                        return <<#flattened_types as #crate_path::SerializeRowByName>::Partial<#partial_lt> as #crate_path::PartialSerializeRowByName>::check_missing(self.#flattened_fields)
362                    })*
363
364                    ::std::unreachable!()
365                }
366            }
367        };
368
369        // 3. Implement SerializeRowByName
370        let num_fields = flattened_visited_flag_names.len() + nonflattened_visited_flag_names.len();
371        let serialize_by_name: syn::ItemImpl = parse_quote! {
372            impl #impl_generics #crate_path::SerializeRowByName for #struct_name #ty_generics #where_clause {
373                type Partial<#partial_lt> = #partial_struct_name #partial_ty_generics where Self: #partial_lt;
374
375                fn partial(&self) -> Self::Partial<'_> {
376                    use ::std::iter::FromIterator as _;
377
378                    #partial_struct_name {
379                        #(#nonflattened_fields: &self.#nonflattened_fields,)*
380                        #(#flattened_fields: <_ as #crate_path::SerializeRowByName>::partial(&self.#flattened_fields),)*
381                        #(#nonflattened_visited_flag_names: false,)*
382                        #(#flattened_visited_flag_names: false,)*
383                        remaining_count: #num_fields,
384                    }
385                }
386            }
387        };
388
389        // 4. Implement SerializeRow
390        parse_quote! {
391            fn serialize<'_scylla_ser_row_writer_buffer>(
392                &self,
393                ctx: &#crate_path::RowSerializationContext,
394                writer: &mut #crate_path::RowWriter<'_scylla_ser_row_writer_buffer>,
395            ) -> ::std::result::Result<(), #crate_path::SerializationError> {
396                #partial_struct
397                #partial_serialize
398
399                #[expect(non_local_definitions)]
400                #serialize_by_name
401
402                #crate_path::ser::row::ByName::<Self>::serialize(#crate_path::ser::row::ByName(self), ctx, writer)
403            }
404        }
405    }
406
407    fn generate_is_empty(&self) -> syn::TraitItemFn {
408        let is_empty = self.ctx.fields.is_empty();
409        parse_quote! {
410            #[inline]
411            fn is_empty(&self) -> bool {
412                #is_empty
413            }
414        }
415    }
416}

418// Generates an implementation of the trait which requires the columns
419// to be placed in the same order as they are defined in the struct.
420struct ColumnOrderedGenerator<'a> {
421    ctx: &'a Context,
422}
423
424impl Generator for ColumnOrderedGenerator<'_> {
425    fn generate_serialize(&self) -> syn::TraitItemFn {
426        let crate_path = self.ctx.attributes.crate_path();
427        let struct_name = &self.ctx.struct_name;
428        let (impl_generics, ty_generics, where_clause) = self.ctx.generics.split_for_impl();
429        let column_serializers = self.ctx.fields.iter().map(|f| -> syn::Stmt {
430            let field = &f.ident;
431            if f.attrs.flatten {
432                syn::parse_quote! {
433                    <_ as #crate_path::SerializeRowInOrder>::serialize_in_order(&self.#field, columns, writer)?;
434                }
435            } else {
436                let column = f.column_name();
437                let enforce_name = !self.ctx.attributes.skip_name_checks;
438                syn::parse_quote! {
439                    #crate_path::ser::row::NextColumnSerializer::serialize::<Self, #enforce_name>(columns, #column, &self.#field, writer)?;
440                }
441            }
442        });
443
444        parse_quote! {
445            fn serialize<'_scylla_ser_row_writer_buffer>(
446                &self,
447                ctx: &#crate_path::RowSerializationContext,
448                writer: &mut #crate_path::RowWriter<'_scylla_ser_row_writer_buffer>,
449            ) -> ::std::result::Result<(), #crate_path::SerializationError> {
450                #[expect(non_local_definitions)]
451                impl #impl_generics #crate_path::SerializeRowInOrder for #struct_name #ty_generics #where_clause {
452                    fn serialize_in_order(
453                        &self,
454                        columns: &mut #crate_path::ser::row::NextColumnSerializer<'_, '_>,
455                        writer: &mut #crate_path::RowWriter<'_>,
456                    ) -> ::std::result::Result<(), #crate_path::SerializationError> {
457                        #(#column_serializers)*
458                        ::std::result::Result::Ok(())
459                    }
460                }
461
462                #crate_path::ser::row::InOrder::<Self>::serialize(#crate_path::ser::row::InOrder(self), ctx, writer)
463            }
464        }
465    }
466
467    fn generate_is_empty(&self) -> syn::TraitItemFn {
468        let is_empty = self.ctx.fields.is_empty();
469        parse_quote! {
470            #[inline]
471            fn is_empty(&self) -> bool {
472                #is_empty
473            }
474        }
475    }
