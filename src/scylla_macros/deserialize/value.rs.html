12#[derive(FromAttributes)]
13#[darling(attributes(scylla))]
14struct StructAttrs {
15    #[darling(rename = "crate")]
16    crate_path: Option<syn::Path>,
17
18    #[darling(default)]
19    flavor: Flavor,
20
21    // If true, then the type checking code won't verify the UDT field names.
22    // UDT fields will be matched to struct fields based solely on the order.
23    //
24    // This annotation only works if `enforce_order` is specified.
25    #[darling(default)]
26    skip_name_checks: bool,
27
28    // If true, then the type checking code will require that the UDT does not
29    // contain excess fields at its suffix. Otherwise, if UDT has some fields
30    // at its suffix that do not correspond to Rust struct's fields,
31    // they will be ignored. With true, an error will be raised.
32    #[darling(default)]
33    forbid_excess_udt_fields: bool,
34}
35
36impl DeserializeCommonStructAttrs for StructAttrs {
37    fn crate_path(&self) -> Option<&syn::Path> {
38        self.crate_path.as_ref()
39    }
40}

42#[derive(FromField)]
43#[darling(attributes(scylla))]
44struct Field {
45    // If true, then the field is not parsed at all, but it is initialized
46    // with Default::default() instead. All other attributes are ignored.
47    #[darling(default)]
48    skip: bool,
49
50    // If true, then - if this field is missing from the UDT fields metadata
51    // - it will be initialized to Default::default().
52    #[darling(default)]
53    #[darling(rename = "allow_missing")]
54    default_when_missing: bool,
55
56    // If true, then - if this field is present among UDT fields metadata
57    // but at the same time missing from serialized data or set to null
58    // - it will be initialized to Default::default().
59    #[darling(default)]
60    default_when_null: bool,
61
62    // If set, then deserializes from the UDT field with this particular name
63    // instead of the Rust field name.
64    #[darling(default)]
65    rename: Option<String>,
66
67    ident: Option<syn::Ident>,
68    ty: syn::Type,
69}
70
71impl DeserializeCommonFieldAttrs for Field {
72    fn needs_default(&self) -> bool {
73        self.skip || self.default_when_missing
74    }
75
76    fn deserialize_target(&self) -> &syn::Type {
77        &self.ty
78    }
79}

81// derive(DeserializeValue) for the DeserializeValue trait
82pub(crate) fn deserialize_value_derive(
83    tokens_input: TokenStream,
84) -> Result<syn::ItemImpl, syn::Error> {
85    let input = syn::parse(tokens_input)?;
86
87    let implemented_trait: syn::Path = parse_quote!(DeserializeValue);
88    let implemented_trait_name = implemented_trait
89        .segments
90        .last()
91        .unwrap()
92        .ident
93        .unraw()
94        .to_string();
95    let constraining_trait = implemented_trait.clone();
96    let s = StructDesc::new(&input, &implemented_trait_name, constraining_trait)?;
97
98    validate_attrs(&s.attrs, s.fields())?;
99
100    let items = [
101        s.generate_type_check_method().into(),
102        s.generate_deserialize_method().into(),
103    ];
104
105    Ok(s.generate_impl(implemented_trait, items))
106}

108fn validate_attrs(attrs: &StructAttrs, fields: &[Field]) -> Result<(), darling::Error> {
109    let mut errors = darling::Error::accumulator();
110
111    if attrs.skip_name_checks {
112        // Skipping name checks is only available in enforce_order mode
113        if attrs.flavor != Flavor::EnforceOrder {
114            let error = darling::Error::custom(
115                "attribute <skip_name_checks> requires <flavor = enforce_order>.",
116            );
117            errors.push(error);
118        }
119
120        // Fields with `allow_missing` are only permitted at the end of the
121        // struct, i.e. no field without `allow_missing` and `skip` is allowed
122        // to be after any field with `allow_missing`.
123        let invalid_default_when_missing_field = fields
124            .iter()
125            .rev()
126            // Skip the whole suffix of <allow_missing> and <skip>.
127            .skip_while(|field| !field.is_required())
128            // skip_while finished either because the iterator is empty or it found a field without both <allow_missing> and <skip>.
129            // In either case, there aren't allowed to be any more fields with `allow_missing`.
130            .find(|field| field.default_when_missing);
131        if let Some(invalid) = invalid_default_when_missing_field {
132            let error =
133                darling::Error::custom(
134                    "when <skip_name_checks> is on, fields with <allow_missing> are only permitted at the end of the struct, \
135                          i.e. no field without <allow_missing> and <skip> is allowed to be after any field with <allow_missing>."
136            ).with_span(&invalid.ident);
137            errors.push(error);
138        }
139
140        // <rename> annotations don't make sense with skipped name checks
141        for field in fields {
142            if field.rename.is_some() {
143                let err = darling::Error::custom(
144                    "<rename> annotations don't make sense with <skip_name_checks> attribute",
145                )
146                .with_span(&field.ident);
147                errors.push(err);
148            }
149        }
150    } else {
151        // Detect name collisions caused by <rename>.
152        let mut used_names = HashMap::<String, &Field>::new();
153        for field in fields {
154            let udt_field_name = field.udt_field_name();
155            if let Some(other_field) = used_names.get(&udt_field_name) {
156                let other_field_ident = other_field.ident.as_ref().unwrap();
157                let msg = format!("the UDT field name `{udt_field_name}` used by this struct field is already used by field `{other_field_ident}`");
158                let err = darling::Error::custom(msg).with_span(&field.ident);
159                errors.push(err);
160            } else {
161                used_names.insert(udt_field_name, field);
162            }
163        }
164    }
165
166    errors.finish()
167}

169impl Field {
170    // Returns whether this field is mandatory for deserialization.
171    fn is_required(&self) -> bool {
172        !self.skip && !self.default_when_missing
173    }
174
175    // The name of UDT field corresponding to this Rust struct field
176    fn udt_field_name(&self) -> String {
177        match self.rename.as_ref() {
178            Some(rename) => rename.to_owned(),
179            None => self.ident.as_ref().unwrap().unraw().to_string(),
180        }
181    }
182
183    // A Rust literal representing the name of this field
184    fn cql_name_literal(&self) -> syn::LitStr {
185        syn::LitStr::new(&self.udt_field_name(), Span::call_site())
186    }
187}

191impl StructDesc {
192    /// Generates an expression which extracts the UDT fields or returns an error.
193    fn generate_extract_fields_from_type(&self, typ_expr: syn::Expr) -> syn::Expr {
194        let macro_internal = &self.struct_attrs().macro_internal_path();
195        parse_quote!(
196            match #typ_expr {
197                #macro_internal::ColumnType::UserDefinedType { definition, .. } => &definition.field_types,
198                other => return ::std::result::Result::Err(
199                    #macro_internal::mk_value_typck_err::<Self>(
200                        &other,
201                        #macro_internal::DeserUdtTypeCheckErrorKind::NotUdt,
202                    )
203                ),
204            }
205        )
206    }
207
208    fn generate_type_check_method(&self) -> syn::ImplItemFn {
209        match self.attrs.flavor {
210            Flavor::MatchByName => TypeCheckUnorderedGenerator(self).generate(),
211            Flavor::EnforceOrder => TypeCheckAssumeOrderGenerator(self).generate(),
212        }
213    }
214
215    fn generate_deserialize_method(&self) -> syn::ImplItemFn {
216        match self.attrs.flavor {
217            Flavor::MatchByName => DeserializeUnorderedGenerator(self).generate(),
218            Flavor::EnforceOrder => DeserializeAssumeOrderGenerator(self).generate(),
219        }
220    }

223struct TypeCheckAssumeOrderGenerator<'sd>(&'sd StructDesc);
224
225impl TypeCheckAssumeOrderGenerator<'_> {
226    // Generates name and type validation for given Rust struct's field.
227    fn generate_field_validation(&self, rust_field_idx: usize, field: &Field) -> syn::Expr {
228        let macro_internal = self.0.struct_attrs().macro_internal_path();
229        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
230        let rust_field_name = field.cql_name_literal();
231        let rust_field_typ = field.deserialize_target();
232        let default_when_missing = field.default_when_missing;
233        let skip_name_checks = self.0.attrs.skip_name_checks;
234
235        // Action performed in case of field name mismatch.
236        let name_mismatch: syn::Expr = if default_when_missing {
237            parse_quote! {
238                {
239                    // If the Rust struct's field is marked as `default_when_missing`, then let's assume
240                    // optimistically that the remaining UDT fields match required Rust struct fields.
241                    // For that, store the read UDT field to be fit against the next Rust struct field.
242                    saved_cql_field = ::std::option::Option::Some(next_cql_field);
243                    break 'verifications; // Skip type verification, because the UDT field is absent.
244                }
245            }
246        } else {
247            parse_quote! {
248                {
249                    // Error - required value for field not present among the CQL fields.
250                    return ::std::result::Result::Err(
251                        #macro_internal::mk_value_typck_err::<Self>(
252                            typ,
253                            #macro_internal::DeserUdtTypeCheckErrorKind::FieldNameMismatch {
254                                position: #rust_field_idx,
255                                rust_field_name: <_ as ::std::borrow::ToOwned>::to_owned(#rust_field_name),
256                                db_field_name: <_ as ::std::clone::Clone>::clone(cql_field_name).into_owned(),
257                            }
258                        )
259                    );
260                }
261            }
262        };
263
264        // Optional name verification.
265        let name_verification: Option<syn::Expr> = (!skip_name_checks).then(|| {
266            parse_quote! {
267                if #rust_field_name != cql_field_name {
268                    // The read UDT field is not the one expected by the Rust struct.
269                    #name_mismatch
270                }
271            }
272        });
273
274        parse_quote! {
275            'field: {
276                let next_cql_field = match saved_cql_field
277                    // We may have a stored CQL UDT field that did not match the previous Rust struct's field.
278                    .take()
279                    // If not, simply fetch another CQL UDT field from the iterator.
280                    .or_else(|| ::std::iter::Iterator::next(&mut cql_field_iter)) {
281                        ::std::option::Option::Some(cql_field) => cql_field,
282                        // In case the Rust field allows default-initialisation and there are no more CQL fields,
283                        // simply assume it's going to be default-initialised.
284                        ::std::option::Option::None if #default_when_missing => break 'field,
285                        ::std::option::Option::None => return ::std::result::Result::Err(too_few_fields()),
286                };
287                let (cql_field_name, cql_field_typ) = next_cql_field;
288
289                'verifications: {
290                    // Verify the name (unless `skip_name_checks` is specified)
291                    // In a specific case when this Rust field is going to be default-initialised
292                    // due to no corresponding CQL UDT field, the below type verification will be skipped.
293                    #name_verification
294
295                    // Verify the type
296                    <#rust_field_typ as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::type_check(cql_field_typ)
297                        .map_err(|err| #macro_internal::mk_value_typck_err::<Self>(
298                            typ,
299                            #macro_internal::DeserUdtTypeCheckErrorKind::FieldTypeCheckFailed {
300                                field_name: <_ as ::std::borrow::ToOwned>::to_owned(#rust_field_name),
301                                err,
302                            }
303                        ))?;
304                }
305            }
306        }
307    }
308
309    // Generates the type_check method for when ensure_order == true.
310    fn generate(&self) -> syn::ImplItemFn {
311        // The generated method will:
312        // - Check that every required field appears on the list in the same order as struct fields
313        // - Every type on the list is correct
314
315        let macro_internal = self.0.struct_attrs().macro_internal_path();
316
317        let extract_fields_expr = self.0.generate_extract_fields_from_type(parse_quote!(typ));
318
319        let required_fields_iter = || self.0.fields().iter().filter(|f| f.is_required());
320
321        let required_field_count = required_fields_iter().count();
322        let required_field_count_lit =
323            syn::LitInt::new(&required_field_count.to_string(), Span::call_site());
324
325        let required_fields_names = required_fields_iter().map(|field| field.ident.as_ref());
326
327        let nonskipped_fields_iter = || {
328            self.0
329                .fields()
330                .iter()
331                // It is important that we enumerate **before** filtering, because otherwise we would not
332                // count the skipped fields, which might be confusing.
333                .enumerate()
334                .filter(|(_idx, f)| !f.skip)
335        };
336
337        let field_validations =
338            nonskipped_fields_iter().map(|(idx, field)| self.generate_field_validation(idx, field));
339
340        let check_excess_udt_fields: Option<syn::Expr> =
341            self.0.attrs.forbid_excess_udt_fields.then(|| {
342                parse_quote! {
343                    if let ::std::option::Option::Some((cql_field_name, cql_field_typ)) = saved_cql_field
344                        .take()
345                        .or_else(|| ::std::iter::Iterator::next(&mut cql_field_iter)) {
346                        return ::std::result::Result::Err(#macro_internal::mk_value_typck_err::<Self>(
347                            typ,
348                            #macro_internal::DeserUdtTypeCheckErrorKind::ExcessFieldInUdt {
349                                db_field_name: <_ as ::std::clone::Clone>::clone(cql_field_name).into_owned(),
350                            }
351                        ));
352                    }
353                }
354            });
355
356        parse_quote! {
357            fn type_check(
358                typ: &#macro_internal::ColumnType,
359            ) -> ::std::result::Result<(), #macro_internal::TypeCheckError> {
360                // Extract information about the field types from the UDT
361                // type definition.
362                let fields = #extract_fields_expr;
363
364                let too_few_fields = || #macro_internal::mk_value_typck_err::<Self>(
365                    typ,
366                    #macro_internal::DeserUdtTypeCheckErrorKind::TooFewFields {
367                        required_fields: ::std::vec![
368                            #(stringify!(#required_fields_names),)*
369                        ],
370                        present_fields: ::std::iter::Iterator::collect(::std::iter::Iterator::map(fields.iter(), |(name, _typ)| ::std::clone::Clone::clone(name).into_owned())),
371                    }
372                );
373
374                // Verify that the field count is correct
375                if fields.len() < #required_field_count_lit {
376                    return ::std::result::Result::Err(too_few_fields());
377                }
378
379                let mut cql_field_iter = fields.iter();
380                // A CQL UDT field that has already been fetched from the field iterator,
381                // but not yet matched to a Rust struct field (because the previous
382                // Rust struct field didn't match it and had #[allow_missing] specified).
383                let mut saved_cql_field = ::std::option::Option::None::<
384                    &(::std::borrow::Cow<'_, str>, #macro_internal::ColumnType),
385                >;
386                #(
387                    #field_validations
388                )*
389
390                #check_excess_udt_fields
391
392                // All is good!
393                ::std::result::Result::Ok(())
394            }
395        }
396    }
397}

399struct DeserializeAssumeOrderGenerator<'sd>(&'sd StructDesc);
400
401impl DeserializeAssumeOrderGenerator<'_> {
402    fn generate_finalize_field(&self, field: &Field) -> syn::Expr {
403        if field.skip {
404            // Skipped fields are initialized with Default::default()
405            return parse_quote! {
406                ::std::default::Default::default()
407            };
408        }
409
410        let macro_internal = self.0.struct_attrs().macro_internal_path();
411        let cql_name_literal = field.cql_name_literal();
412        let deserializer = field.deserialize_target();
413        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
414        let default_when_missing = field.default_when_missing;
415        let default_when_null = field.default_when_null;
416        let skip_name_checks = self.0.attrs.skip_name_checks;
417
418        let deserialize: syn::Expr = parse_quote! {
419            <#deserializer as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::deserialize(cql_field_typ, value)
420                .map_err(|err| #macro_internal::mk_value_deser_err::<Self>(
421                    typ,
422                    #macro_internal::UdtDeserializationErrorKind::FieldDeserializationFailed {
423                        field_name: <_ as ::std::borrow::ToOwned>::to_owned(#cql_name_literal),
424                        err,
425                    }
426                ))?
427        };
428
429        let maybe_default_deserialize: syn::Expr = if default_when_null {
430            parse_quote! {
431                if value.is_none() {
432                    ::std::default::Default::default()
433                } else {
434                    #deserialize
435                }
436            }
437        } else {
438            parse_quote! {
439                #deserialize
440            }
441        };
442
443        // Action performed in case of field name mismatch.
444        let name_mismatch: syn::Expr = if default_when_missing {
445            parse_quote! {
446                {
447                    // If the Rust struct's field is marked as `default_when_missing`, then let's assume
448                    // optimistically that the remaining UDT fields match required Rust struct fields.
449                    // For that, store the read UDT field to be fit against the next Rust struct field.
450                    saved_cql_field = ::std::option::Option::Some(next_cql_field);
451
452                    ::std::default::Default::default()
453                }
454            }
455        } else {
456            parse_quote! {
457                {
458                    ::std::panic!(
459                        "type check should have prevented this scenario - field name mismatch! Rust field name {}, CQL field name {}",
460                        #cql_name_literal,
461                        cql_field_name
462                    );
463                }
464            }
465        };
466
467        let maybe_name_check_and_deserialize_or_save: syn::Expr = if skip_name_checks {
468            parse_quote! {
469                #maybe_default_deserialize
470            }
471        } else {
472            parse_quote! {
473                if #cql_name_literal == cql_field_name {
474                    #maybe_default_deserialize
475                } else {
476                    #name_mismatch
477                }
478            }
479        };
480
481        let no_more_fields: syn::Expr = if default_when_missing {
482            parse_quote! {
483                ::std::default::Default::default()
484            }
485        } else {
486            parse_quote! {
487                // Type check has ensured that there are enough CQL UDT fields.
488                ::std::panic!("Too few CQL UDT fields - type check should have prevented this scenario!")
489            }
490        };
491
492        parse_quote! {
493            {
494                let maybe_next_cql_field = saved_cql_field
495                    .take()
496                    .map(::std::result::Result::Ok)
497                    .or_else(|| {
498                        ::std::option::Option::map(
499                            ::std::iter::Iterator::next(&mut cql_field_iter),
500                            |(specs, value_res)| value_res.map(|value| (specs, value)))
501                    })
502                    .transpose()
503                    // Propagate deserialization errors.
504                    .map_err(|err| #macro_internal::mk_value_deser_err::<Self>(
505                        typ,
506                        #macro_internal::UdtDeserializationErrorKind::FieldDeserializationFailed {
507                            field_name: <_ as ::std::borrow::ToOwned>::to_owned(#cql_name_literal),
508                            err,
509                        }
510                    ))?;
511
512                if let ::std::option::Option::Some(next_cql_field) = maybe_next_cql_field {
513                    let ((cql_field_name, cql_field_typ), value) = next_cql_field;
514
515                    // The value can be either
516                    // - None - missing from the serialized representation
517                    // - Some(None) - present in the serialized representation but null
518                    // For now, we treat both cases as "null".
519                    let value = value.flatten();
520
521                    #maybe_name_check_and_deserialize_or_save
522                } else {
523                    #no_more_fields
524                }
525            }
526        }
527    }
528
529    fn generate(&self) -> syn::ImplItemFn {
530        // We can assume that type_check was called.
531
532        let macro_internal = self.0.struct_attrs().macro_internal_path();
533        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
534        let fields = self.0.fields();
535
536        let field_idents = fields.iter().map(|f| f.ident.as_ref().unwrap());
537        let field_finalizers = fields.iter().map(|f| self.generate_finalize_field(f));
538
539        let iterator_type: syn::Type =
540            parse_quote!(#macro_internal::UdtIterator<#frame_lifetime, #metadata_lifetime>);
541
542        parse_quote! {
543            fn deserialize(
544                typ: &#metadata_lifetime #macro_internal::ColumnType<#metadata_lifetime>,
545                v: ::std::option::Option<#macro_internal::FrameSlice<#frame_lifetime>>,
546            ) -> ::std::result::Result<Self, #macro_internal::DeserializationError> {
547                // Create an iterator over the fields of the UDT.
548                let mut cql_field_iter = <#iterator_type as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::deserialize(typ, v)
549                    .map_err(#macro_internal::value_deser_error_replace_rust_name::<Self>)?;
550
551                // This is to hold another field that already popped up from the field iterator but appeared to not match
552                // the expected nonrequired field. Therefore, that field is stored here, while the expected field
553                // is default-initialized.
554                let mut saved_cql_field = ::std::option::Option::None::<(
555                    &(::std::borrow::Cow<'_, str>, #macro_internal::ColumnType),
556                    ::std::option::Option<::std::option::Option<#macro_internal::FrameSlice>>
557                )>;
558
559                ::std::result::Result::Ok(Self {
560                    #(#field_idents: #field_finalizers,)*
561                })
562            }
563        }
564    }
565}

567struct TypeCheckUnorderedGenerator<'sd>(&'sd StructDesc);
568
569impl TypeCheckUnorderedGenerator<'_> {
570    // An identifier for a bool variable that represents whether given
571    // field was already visited during type check
572    fn visited_flag_variable(field: &Field) -> syn::Ident {
573        quote::format_ident!("visited_{}", field.ident.as_ref().unwrap().unraw())
574    }
575
576    // Generates a declaration of a "visited" flag for the purpose of type check.
577    // We generate it even if the flag is not required in order to protect
578    // from fields appearing more than once
579    fn generate_visited_flag_decl(field: &Field) -> Option<syn::Stmt> {
580        (!field.skip).then(|| {
581            let visited_flag = Self::visited_flag_variable(field);
582            parse_quote! {
583                let mut #visited_flag = false;
584            }
585        })
586    }
587
588    // Generates code that, given variable `typ`, type-checks given field
589    fn generate_type_check(&self, field: &Field) -> Option<syn::Block> {
590        (!field.skip).then(|| {
591            let macro_internal = self.0.struct_attrs().macro_internal_path();
592            let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
593            let visited_flag = Self::visited_flag_variable(field);
594            let typ = field.deserialize_target();
595            let cql_name_literal = field.cql_name_literal();
596            let decrement_if_required: Option<syn::Stmt> = field
597                .is_required()
598                .then(|| parse_quote! {remaining_required_cql_fields -= 1;});
599
600            parse_quote! {
601                {
602                    if !#visited_flag {
603                        <#typ as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::type_check(cql_field_typ)
604                            .map_err(|err| #macro_internal::mk_value_typck_err::<Self>(
605                                typ,
606                                #macro_internal::DeserUdtTypeCheckErrorKind::FieldTypeCheckFailed {
607                                    field_name: <_ as ::std::clone::Clone>::clone(cql_field_name).into_owned(),
608                                    err,
609                                }
610                            ))?;
611                        #visited_flag = true;
612                        #decrement_if_required
613                    } else {
614                        return ::std::result::Result::Err(
615                            #macro_internal::mk_value_typck_err::<Self>(
616                                typ,
617                                #macro_internal::DeserUdtTypeCheckErrorKind::DuplicatedField {
618                                    field_name: <_ as ::std::borrow::ToOwned>::to_owned(#cql_name_literal),
619                                }
620                            )
621                        )
622                    }
623                }
624            }
625        })
626    }
627
628    // Generates code that appends the field name if it is missing.
629    // The generated code is used to construct a nice error message.
630    fn generate_append_name(field: &Field) -> Option<syn::Block> {
631        field.is_required().then(|| {
632            let visited_flag = Self::visited_flag_variable(field);
633            let cql_name_literal = field.cql_name_literal();
634            parse_quote!(
635                {
636                    if !#visited_flag {
637                        missing_fields.push(#cql_name_literal);
638                    }
639                }
640            )
641        })
642    }
643
644    // Generates the type_check method for when ensure_order == false.
645    fn generate(&self) -> syn::ImplItemFn {
646        // The generated method will:
647        // - Check that every required field appears on the list exactly once, in any order
648        // - Every type on the list is correct
649
650        let macro_internal = &self.0.struct_attrs().macro_internal_path();
651        let forbid_excess_udt_fields = self.0.attrs.forbid_excess_udt_fields;
652        let rust_fields = self.0.fields();
653        let visited_field_declarations = rust_fields
654            .iter()
655            .flat_map(Self::generate_visited_flag_decl);
656        let type_check_blocks = rust_fields.iter().flat_map(|f| self.generate_type_check(f));
657        let append_name_blocks = rust_fields.iter().flat_map(Self::generate_append_name);
658        let rust_nonskipped_field_names = rust_fields
659            .iter()
660            .filter(|f| !f.skip)
661            .map(|f| f.cql_name_literal());
662        let required_cql_field_count = rust_fields.iter().filter(|f| f.is_required()).count();
663        let required_cql_field_count_lit =
664            syn::LitInt::new(&required_cql_field_count.to_string(), Span::call_site());
665        let extract_cql_fields_expr = self.0.generate_extract_fields_from_type(parse_quote!(typ));
666
667        // If UDT contains a field with an unknown name, an error is raised iff
668        // `forbid_excess_udt_fields` attribute is specified.
669        let excess_udt_field_action: syn::Expr = if forbid_excess_udt_fields {
670            parse_quote! {
671                return ::std::result::Result::Err(
672                    #macro_internal::mk_value_typck_err::<Self>(
673                        typ,
674                        #macro_internal::DeserUdtTypeCheckErrorKind::ExcessFieldInUdt {
675                            db_field_name: <_ as ::std::borrow::ToOwned>::to_owned(unknown),
676                        }
677                    )
678                )
679            }
680        } else {
681            parse_quote! {
682                // We ignore excess UDT fields, as this facilitates the process of adding new fields
683                // to a UDT in running production cluster & clients.
684                ()
685            }
686        };
687
688        parse_quote! {
689            fn type_check(
690                typ: &#macro_internal::ColumnType,
691            ) -> ::std::result::Result<(), #macro_internal::TypeCheckError> {
692                // Extract information about the field types from the UDT
693                // type definition.
694                let cql_fields = #extract_cql_fields_expr;
695
696                // Counts down how many required fields are remaining
697                let mut remaining_required_cql_fields: ::std::primitive::usize = #required_cql_field_count_lit;
698
699                // For each required field, generate a "visited" boolean flag
700                #(#visited_field_declarations)*
701
702                for (cql_field_name, cql_field_typ) in cql_fields {
703                    // Pattern match on the name and verify that the type is correct.
704                    match ::std::ops::Deref::deref(cql_field_name) {
705                        #(#rust_nonskipped_field_names => #type_check_blocks,)*
706                        unknown => #excess_udt_field_action,
707                    }
708                }
709
710                if remaining_required_cql_fields > 0 {
711                    // If there are some missing required fields, generate an error
712                    // which contains missing field names
713                    let mut missing_fields = ::std::vec::Vec::<&'static str>::with_capacity(remaining_required_cql_fields);
714                    #(#append_name_blocks)*
715                    return ::std::result::Result::Err(
716                        #macro_internal::mk_value_typck_err::<Self>(
717                            typ,
718                            #macro_internal::DeserUdtTypeCheckErrorKind::ValuesMissingForUdtFields {
719                                field_names: missing_fields,
720                            }
721                        )
722                    )
723                }
724
725                ::std::result::Result::Ok(())
726            }
727        }
728    }
729}
730
731struct DeserializeUnorderedGenerator<'sd>(&'sd StructDesc);
732
733impl DeserializeUnorderedGenerator<'_> {
734    /// An identifier for a variable that is meant to store the parsed variable
735    /// before being ultimately moved to the struct on deserialize.
736    fn deserialize_field_variable(field: &Field) -> syn::Ident {
737        quote::format_ident!("f_{}", field.ident.as_ref().unwrap().unraw())
738    }
739
740    /// Generates an expression which produces a value ready to be put into a field
741    /// of the target structure.
742    fn generate_finalize_field(&self, field: &Field) -> syn::Expr {
743        if field.skip {
744            // Skipped fields are initialized with Default::default()
745            return parse_quote!(::std::default::Default::default());
746        }
747
748        let deserialize_field = Self::deserialize_field_variable(field);
749        if field.default_when_missing {
750            // Generate Default::default if the field was missing
751            parse_quote! {
752                #deserialize_field.unwrap_or_default()
753            }
754        } else {
755            let cql_name_literal = field.cql_name_literal();
756            parse_quote! {
757                #deserialize_field.unwrap_or_else(|| ::std::panic!(
758                    "field {} missing in UDT - type check should have prevented this!",
759                    #cql_name_literal
760                ))
761            }
762        }
763    }
764
765    /// Generates code that performs deserialization when the raw field
766    /// is being processed.
767    fn generate_deserialization(&self, field: &Field) -> Option<syn::Expr> {
768        (!field.skip).then(|| {
769            let macro_internal = self.0.struct_attrs().macro_internal_path();
770            let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
771            let deserialize_field = Self::deserialize_field_variable(field);
772            let cql_name_literal = field.cql_name_literal();
773            let deserializer = field.deserialize_target();
774
775            let do_deserialize: syn::Expr = parse_quote! {
776                <#deserializer as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::deserialize(cql_field_typ, value)
777                .map_err(|err| #macro_internal::mk_value_deser_err::<Self>(
778                    typ,
779                    #macro_internal::UdtDeserializationErrorKind::FieldDeserializationFailed {
780                        field_name: <_ as ::std::borrow::ToOwned>::to_owned(#cql_name_literal),
781                        err,
782                    }
783                ))?
784            };
785
786            let deserialize_action: syn::Expr = if field.default_when_null {
787                parse_quote! {
788                    if value.is_some() {
789                        #do_deserialize
790                    } else {
791                        ::std::default::Default::default()
792                    }
793                }
794            } else {
795                do_deserialize
796            };
797
798            parse_quote! {
799                {
800                    assert!(
801                        #deserialize_field.is_none(),
802                        "duplicated field {} - type check should have prevented this!",
803                        stringify!(#deserialize_field)
804                    );
805
806                    // The value can be either
807                    // - None - missing from the serialized representation
808                    // - Some(None) - present in the serialized representation but null
809                    // For now, we treat both cases as "null".
810                    let value = value.flatten();
811
812                    #deserialize_field = ::std::option::Option::Some(
813                        #deserialize_action
814                    );
815                }
816            }
817        })
818    }
819
820    // Generate a declaration of a variable that temporarily keeps
821    // the deserialized value
822    fn generate_deserialize_field_decl(field: &Field) -> Option<syn::Stmt> {
823        (!field.skip).then(|| {
824            let deserialize_field = Self::deserialize_field_variable(field);
825            parse_quote! {
826                let mut #deserialize_field = ::std::option::Option::None;
827            }
828        })
829    }
830
831    fn generate(&self) -> syn::ImplItemFn {
832        let macro_internal = self.0.struct_attrs().macro_internal_path();
833        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
834        let fields = self.0.fields();
835
836        let deserialize_field_decls = fields.iter().map(Self::generate_deserialize_field_decl);
837        let deserialize_blocks = fields.iter().flat_map(|f| self.generate_deserialization(f));
838        let rust_field_idents = fields.iter().map(|f| f.ident.as_ref().unwrap());
839        let rust_nonskipped_field_names = fields
840            .iter()
841            .filter(|f| !f.skip)
842            .map(|f| f.cql_name_literal());
843
844        let field_finalizers = fields.iter().map(|f| self.generate_finalize_field(f));
845
846        let iterator_type: syn::Type = parse_quote! {
847            #macro_internal::UdtIterator<#frame_lifetime, #metadata_lifetime>
848        };
849
850        // TODO: Allow collecting unrecognized fields into some special field
851
852        parse_quote! {
853            fn deserialize(
854                typ: &#metadata_lifetime #macro_internal::ColumnType<#metadata_lifetime>,
855                v: ::std::option::Option<#macro_internal::FrameSlice<#frame_lifetime>>,
856            ) -> ::std::result::Result<Self, #macro_internal::DeserializationError> {
857                // Create an iterator over the fields of the UDT.
858                let cql_field_iter = <#iterator_type as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::deserialize(typ, v)
859                    .map_err(#macro_internal::value_deser_error_replace_rust_name::<Self>)?;
860
861                // Generate fields that will serve as temporary storage
862                // for the fields' values. Those are of type Option<FieldType>.
863                #(#deserialize_field_decls)*
864
865                for item in cql_field_iter {
866                    let ((cql_field_name, cql_field_typ), value_res) = item;
867                    let value = value_res.map_err(|err| #macro_internal::mk_value_deser_err::<Self>(
868                        typ,
869                        #macro_internal::UdtDeserializationErrorKind::FieldDeserializationFailed {
870                            field_name: ::std::clone::Clone::clone(cql_field_name).into_owned(),
871                            err,
872                        }
873                    ))?;
874                    // Pattern match on the field name and deserialize.
875                    match ::std::ops::Deref::deref(cql_field_name) {
876                        #(#rust_nonskipped_field_names => #deserialize_blocks,)*
877                        unknown => {
878                            // Assuming we type checked sucessfully, this must be an excess field.
879                            // Let's skip it.
880                        },
881                    }
882                }
883
884                // Create the final struct. The finalizer expressions convert
885                // the temporary storage fields to the final field values.
886                // For example, if a field is missing but marked as
887                // `default_when_null` it will create a default value, otherwise
888                // it will report an error.
889                ::std::result::Result::Ok(Self {
890                    #(#rust_field_idents: #field_finalizers,)*
891                })
892            }
893        }
894    }
895}
