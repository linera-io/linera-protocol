12#[derive(FromAttributes)]
13#[darling(attributes(scylla))]
14struct StructAttrs {
15    #[darling(rename = "crate")]
16    crate_path: Option<syn::Path>,
17
18    #[darling(default)]
19    flavor: Flavor,

21    // If true, then the type checking code won't verify the column names.
22    // Columns will be matched to struct fields based solely on the order.

35#[derive(FromField)]
36#[darling(attributes(scylla))]
37struct Field {
38    // If true, then the field is not parsed at all, but it is initialized
39    // with Default::default() instead. All other attributes are ignored.
40    #[darling(default)]
41    skip: bool,
42
43    // If set, then deserialization will look for the column with given name
44    // and deserialize it to this Rust field, instead of just using the Rust
45    // field name.
46    #[darling(default)]
47    rename: Option<String>,
48
49    ident: Option<syn::Ident>,
50    ty: syn::Type,
51}

63// derive(DeserializeRow) for the new DeserializeRow trait
64pub(crate) fn deserialize_row_derive(
65    tokens_input: proc_macro::TokenStream,
66) -> Result<syn::ItemImpl, syn::Error> {
67    let input = syn::parse(tokens_input)?;
68
69    let implemented_trait: syn::Path = parse_quote! { DeserializeRow };
70    let implemented_trait_name = implemented_trait
71        .segments
72        .last()
73        .unwrap()
74        .ident
75        .unraw()
76        .to_string();
77    let constraining_trait = parse_quote! { DeserializeValue };
78    let s = StructDesc::new(&input, &implemented_trait_name, constraining_trait)?;
79
80    validate_attrs(&s.attrs, &s.fields)?;
81
82    let items = [
83        s.generate_type_check_method().into(),
84        s.generate_deserialize_method().into(),
85    ];
86
87    Ok(s.generate_impl(implemented_trait, items))
88}

90fn validate_attrs(attrs: &StructAttrs, fields: &[Field]) -> Result<(), darling::Error> {
91    let mut errors = darling::Error::accumulator();
92
93    if attrs.skip_name_checks {
94        // Skipping name checks is only available in enforce_order mode
95        if attrs.flavor != Flavor::EnforceOrder {
96            let error =
97                darling::Error::custom("attribute <skip_name_checks> requires <enforce_order>.");
98            errors.push(error);
99        }
100
101        // <rename> annotations don't make sense with skipped name checks
102        for field in fields {
103            if field.rename.is_some() {
104                let err = darling::Error::custom(
105                    "<rename> annotations don't make sense with <skip_name_checks> attribute",
106                )
107                .with_span(&field.ident);
108                errors.push(err);
109            }
110        }
111    } else {
112        // Detect name collisions caused by `rename`.
113        let mut used_names = HashMap::<String, &Field>::new();
114        for field in fields {
115            let column_name = field.column_name();
116            if let Some(other_field) = used_names.get(&column_name) {
117                let other_field_ident = other_field.ident.as_ref().unwrap();
118                let msg = format!("the column name `{column_name}` used by this struct field is already used by field `{other_field_ident}`");
119                let err = darling::Error::custom(msg).with_span(&field.ident);
120                errors.push(err);
121            } else {
122                used_names.insert(column_name, field);
123            }
124        }
125    }
126
127    errors.finish()
128}

130impl Field {
131    // Returns whether this field is mandatory for deserialization.
132    fn is_required(&self) -> bool {
133        !self.skip
134    }
135
136    // The name of the column corresponding to this Rust struct field
137    fn column_name(&self) -> String {
138        match self.rename.as_ref() {
139            Some(rename) => rename.to_owned(),
140            None => self.ident.as_ref().unwrap().unraw().to_string(),
141        }
142    }
143
144    // A Rust literal representing the name of this field
145    fn cql_name_literal(&self) -> syn::LitStr {
146        syn::LitStr::new(&self.column_name(), Span::call_site())
147    }
148}

152impl StructDesc {
153    fn generate_type_check_method(&self) -> syn::ImplItemFn {
154        match self.attrs.flavor {
155            Flavor::MatchByName => TypeCheckUnorderedGenerator(self).generate(),
156            Flavor::EnforceOrder => TypeCheckAssumeOrderGenerator(self).generate(),
157        }
158    }
159
160    fn generate_deserialize_method(&self) -> syn::ImplItemFn {
161        match self.attrs.flavor {
162            Flavor::MatchByName => DeserializeUnorderedGenerator(self).generate(),
163            Flavor::EnforceOrder => DeserializeAssumeOrderGenerator(self).generate(),
164        }
165    }

168struct TypeCheckAssumeOrderGenerator<'sd>(&'sd StructDesc);
169
170impl TypeCheckAssumeOrderGenerator<'_> {
171    fn generate_name_verification(
172        &self,
173        field_index: usize, //  These two indices can be different because of `skip` attribute
174        column_index: usize, // applied to some field.
175        field: &Field,
176        column_spec: &syn::Ident,
177    ) -> Option<syn::Expr> {
178        (!self.0.attrs.skip_name_checks).then(|| {
179            let macro_internal = self.0.struct_attrs().macro_internal_path();
180            let rust_field_name = field.cql_name_literal();
181
182            parse_quote! {
183                if #column_spec.name() != #rust_field_name {
184                    return ::std::result::Result::Err(
185                        #macro_internal::mk_row_typck_err::<Self>(
186                            column_types_iter(),
187                            #macro_internal::DeserBuiltinRowTypeCheckErrorKind::ColumnNameMismatch {
188                                field_index: #field_index,
189                                column_index: #column_index,
190                                rust_column_name: #rust_field_name,
191                                db_column_name: ::std::borrow::ToOwned::to_owned(#column_spec.name()),
192                            }
193                        )
194                    );
195                }
196            }
197        })
198    }
199
200    fn generate(&self) -> syn::ImplItemFn {
201        // The generated method will check that the order and the types
202        // of the columns correspond fields' names/types.
203
204        let macro_internal = self.0.struct_attrs().macro_internal_path();
205        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
206
207        let required_fields_iter = || {
208            self.0
209                .fields()
210                .iter()
211                .enumerate()
212                .filter(|(_, f)| f.is_required())
213        };
214        let required_fields_count = required_fields_iter().count();
215        let required_fields_idents: Vec<_> = (0..required_fields_count)
216            .map(|i| quote::format_ident!("f_{}", i))
217            .collect();
218        let name_verifications = required_fields_iter()
219            .zip(required_fields_idents.iter().enumerate())
220            .map(|((field_idx, field), (col_idx, fidents))| {
221                self.generate_name_verification(field_idx, col_idx, field, fidents)
222            });
223
224        let required_fields_deserializers =
225            required_fields_iter().map(|(_, f)| f.deserialize_target());
226        let numbers = 0usize..;
227
228        parse_quote! {
229            fn type_check(
230                specs: &[#macro_internal::ColumnSpec],
231            ) -> ::std::result::Result<(), #macro_internal::TypeCheckError> {
232                let column_types_iter = || ::std::iter::Iterator::map(specs.iter(), |spec| ::std::clone::Clone::clone(spec.typ()).into_owned());
233
234                match specs {
235                    [#(#required_fields_idents),*] => {
236                        #(
237                            // Verify the name (unless `skip_name_checks' is specified)
238                            #name_verifications
239
240                            // Verify the type
241                            <#required_fields_deserializers as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::type_check(#required_fields_idents.typ())
242                                .map_err(|err| #macro_internal::mk_row_typck_err::<Self>(
243                                    column_types_iter(),
244                                    #macro_internal::DeserBuiltinRowTypeCheckErrorKind::ColumnTypeCheckFailed {
245                                        column_index: #numbers,
246                                        column_name: ::std::borrow::ToOwned::to_owned(#required_fields_idents.name()),
247                                        err,
248                                    }
249                                ))?;
250                        )*
251                        ::std::result::Result::Ok(())
252                    },
253                    _ => ::std::result::Result::Err(
254                        #macro_internal::mk_row_typck_err::<Self>(
255                            column_types_iter(),
256                            #macro_internal::DeserBuiltinRowTypeCheckErrorKind::WrongColumnCount {
257                                rust_cols: #required_fields_count,
258                                cql_cols: specs.len(),
259                            }
260                        ),
261                    ),
262                }
263            }
264        }
265    }
266}

268struct DeserializeAssumeOrderGenerator<'sd>(&'sd StructDesc);
269
270impl DeserializeAssumeOrderGenerator<'_> {
271    fn generate_finalize_field(&self, field_index: usize, field: &Field) -> syn::Expr {
272        if field.skip {
273            // Skipped fields are initialized with Default::default()
274            return parse_quote!(::std::default::Default::default());
275        }
276
277        let macro_internal = self.0.struct_attrs().macro_internal_path();
278        let cql_name_literal = field.cql_name_literal();
279        let deserializer = field.deserialize_target();
280        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
281
282        let name_check: Option<syn::Stmt> = (!self.0.struct_attrs().skip_name_checks).then(|| parse_quote! {
283            if col.spec.name() != #cql_name_literal {
284                ::std::panic!(
285                    "Typecheck should have prevented this scenario - field-column name mismatch! Rust field name {}, CQL column name {}",
286                    #cql_name_literal,
287                    col.spec.name()
288                );
289            }
290        });
291
292        parse_quote!(
293            {
294                let col = ::std::iter::Iterator::next(&mut row)
295                    .expect("Typecheck should have prevented this scenario! Too few columns in the serialized data.")
296                    .map_err(#macro_internal::row_deser_error_replace_rust_name::<Self>)?;
297
298                #name_check
299
300                <#deserializer as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::deserialize(col.spec.typ(), col.slice)
301                    .map_err(|err| #macro_internal::mk_row_deser_err::<Self>(
302                        #macro_internal::BuiltinRowDeserializationErrorKind::ColumnDeserializationFailed {
303                            column_index: #field_index,
304                            column_name: <_ as ::std::borrow::ToOwned>::to_owned(col.spec.name()),
305                            err,
306                        }
307                    ))?
308            }
309        )
310    }
311
312    fn generate(&self) -> syn::ImplItemFn {
313        let macro_internal = self.0.struct_attrs().macro_internal_path();
314        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
315
316        let fields = self.0.fields();
317        let field_idents = fields.iter().map(|f| f.ident.as_ref().unwrap());
318        let field_finalizers = fields
319            .iter()
320            .enumerate()
321            .map(|(field_idx, f)| self.generate_finalize_field(field_idx, f));
322
323        parse_quote! {
324            fn deserialize(
325                mut row: #macro_internal::ColumnIterator<#frame_lifetime, #metadata_lifetime>,
326            ) -> ::std::result::Result<Self, #macro_internal::DeserializationError> {
327                ::std::result::Result::Ok(Self {
328                    #(#field_idents: #field_finalizers,)*
329                })
330            }
331        }
332    }
333}

335struct TypeCheckUnorderedGenerator<'sd>(&'sd StructDesc);
336
337impl TypeCheckUnorderedGenerator<'_> {
338    // An identifier for a bool variable that represents whether given
339    // field was already visited during type check
340    fn visited_flag_variable(field: &Field) -> syn::Ident {
341        quote::format_ident!("visited_{}", field.ident.as_ref().unwrap().unraw())
342    }
343
344    // Generates a declaration of a "visited" flag for the purpose of type check.
345    // We generate it even if the flag is not required in order to protect
346    // from fields appearing more than once
347    fn generate_visited_flag_decl(field: &Field) -> Option<syn::Stmt> {
348        (!field.skip).then(|| {
349            let visited_flag = Self::visited_flag_variable(field);
350            parse_quote! {
351                let mut #visited_flag = false;
352            }
353        })
354    }
355
356    // Generates code that, given variable `typ`, type-checks given field
357    fn generate_type_check(&self, field: &Field) -> Option<syn::Block> {
358        (!field.skip).then(|| {
359            let macro_internal = self.0.struct_attrs().macro_internal_path();
360            let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
361            let visited_flag = Self::visited_flag_variable(field);
362            let typ = field.deserialize_target();
363            let cql_name_literal = field.cql_name_literal();
364            let decrement_if_required: Option::<syn::Stmt> = field.is_required().then(|| parse_quote! {
365                remaining_required_fields -= 1;
366            });
367
368            parse_quote! {
369                {
370                    if !#visited_flag {
371                        <#typ as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::type_check(spec.typ())
372                            .map_err(|err| {
373                                #macro_internal::mk_row_typck_err::<Self>(
374                                    column_types_iter(),
375                                    #macro_internal::DeserBuiltinRowTypeCheckErrorKind::ColumnTypeCheckFailed {
376                                        column_index,
377                                        column_name: <_ as ::std::borrow::ToOwned>::to_owned(#cql_name_literal),
378                                        err,
379                                    }
380                                )
381                            })?;
382                        #visited_flag = true;
383                        #decrement_if_required
384                    } else {
385                        return ::std::result::Result::Err(
386                            #macro_internal::mk_row_typck_err::<Self>(
387                                column_types_iter(),
388                                #macro_internal::DeserBuiltinRowTypeCheckErrorKind::DuplicatedColumn {
389                                    column_index,
390                                    column_name: #cql_name_literal,
391                                }
392                            )
393                        )
394                    }
395                }
396            }
397        })
398    }
399
400    // Generates code that appends the flag name if it is missing.
401    // The generated code is used to construct a nice error message.
402    fn generate_append_name(field: &Field) -> Option<syn::Block> {
403        field.is_required().then(|| {
404            let visited_flag = Self::visited_flag_variable(field);
405            let cql_name_literal = field.cql_name_literal();
406            parse_quote! {
407                {
408                    if !#visited_flag {
409                        missing_fields.push(#cql_name_literal);
410                    }
411                }
412            }
413        })
414    }
415
416    fn generate(&self) -> syn::ImplItemFn {
417        let macro_internal = self.0.struct_attrs().macro_internal_path();
418
419        let fields = self.0.fields();
420        let visited_field_declarations = fields.iter().flat_map(Self::generate_visited_flag_decl);
421        let type_check_blocks = fields.iter().flat_map(|f| self.generate_type_check(f));
422        let append_name_blocks = fields.iter().flat_map(Self::generate_append_name);
423        let nonskipped_field_names = fields
424            .iter()
425            .filter(|f| !f.skip)
426            .map(|f| f.cql_name_literal());
427        let field_count_lit = fields.iter().filter(|f| f.is_required()).count();
428
429        parse_quote! {
430            fn type_check(
431                specs: &[#macro_internal::ColumnSpec],
432            ) -> ::std::result::Result<(), #macro_internal::TypeCheckError> {
433                // Counts down how many required fields are remaining
434                let mut remaining_required_fields: ::std::primitive::usize = #field_count_lit;
435
436                // For each required field, generate a "visited" boolean flag
437                #(#visited_field_declarations)*
438
439                let column_types_iter = || ::std::iter::Iterator::map(specs.iter(), |spec| ::std::clone::Clone::clone(spec.typ()).into_owned());
440
441                for (column_index, spec) in ::std::iter::Iterator::enumerate(specs.iter()) {
442                    // Pattern match on the name and verify that the type is correct.
443                    match spec.name() {
444                        #(#nonskipped_field_names => #type_check_blocks,)*
445                        _unknown => {
446                            return ::std::result::Result::Err(
447                                #macro_internal::mk_row_typck_err::<Self>(
448                                    column_types_iter(),
449                                    #macro_internal::DeserBuiltinRowTypeCheckErrorKind::ColumnWithUnknownName {
450                                        column_index,
451                                        column_name: <_ as ::std::borrow::ToOwned>::to_owned(spec.name())
452                                    }
453                                )
454                            )
455                        }
456                    }
457                }
458
459                if remaining_required_fields > 0 {
460                    // If there are some missing required fields, generate an error
461                    // which contains missing field names
462                    let mut missing_fields = ::std::vec::Vec::<&'static str>::with_capacity(remaining_required_fields);
463                    #(#append_name_blocks)*
464                    return ::std::result::Result::Err(
465                        #macro_internal::mk_row_typck_err::<Self>(
466                            column_types_iter(),
467                            #macro_internal::DeserBuiltinRowTypeCheckErrorKind::ValuesMissingForColumns {
468                                column_names: missing_fields
469                            }
470                        )
471                    )
472                }
473
474                ::std::result::Result::Ok(())
475            }
476        }
477    }
478}

480struct DeserializeUnorderedGenerator<'sd>(&'sd StructDesc);
481
482impl DeserializeUnorderedGenerator<'_> {
483    // An identifier for a variable that is meant to store the parsed variable
484    // before being ultimately moved to the struct on deserialize
485    fn deserialize_field_variable(field: &Field) -> syn::Ident {
486        quote::format_ident!("f_{}", field.ident.as_ref().unwrap().unraw())
487    }
488
489    // Generates an expression which produces a value ready to be put into a field
490    // of the target structure
491    fn generate_finalize_field(&self, field: &Field) -> syn::Expr {
492        if field.skip {
493            // Skipped fields are initialized with Default::default()
494            return parse_quote! {
495                ::std::default::Default::default()
496            };
497        }
498
499        let deserialize_field = Self::deserialize_field_variable(field);
500        let cql_name_literal = field.cql_name_literal();
501        parse_quote! {
502            #deserialize_field.unwrap_or_else(|| ::std::panic!(
503                "column {} missing in DB row - type check should have prevented this!",
504                #cql_name_literal
505            ))
506        }
507    }
508
509    // Generated code that performs deserialization when the raw field
510    // is being processed
511    fn generate_deserialization(&self, column_index: usize, field: &Field) -> syn::Expr {
512        assert!(!field.skip);
513        let macro_internal = self.0.struct_attrs().macro_internal_path();
514        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
515        let deserialize_field = Self::deserialize_field_variable(field);
516        let deserializer = field.deserialize_target();
517
518        parse_quote! {
519            {
520                assert!(
521                    #deserialize_field.is_none(),
522                    "duplicated column {} - type check should have prevented this!",
523                    stringify!(#deserialize_field)
524                );
525
526                #deserialize_field = ::std::option::Option::Some(
527                    <#deserializer as #macro_internal::DeserializeValue<#frame_lifetime, #metadata_lifetime>>::deserialize(col.spec.typ(), col.slice)
528                        .map_err(|err| {
529                            #macro_internal::mk_row_deser_err::<Self>(
530                                #macro_internal::BuiltinRowDeserializationErrorKind::ColumnDeserializationFailed {
531                                    column_index: #column_index,
532                                    column_name: <_ as ::std::borrow::ToOwned>::to_owned(col.spec.name()),
533                                    err,
534                                }
535                            )
536                        })?
537                );
538            }
539        }
540    }
541
542    // Generate a declaration of a variable that temporarily keeps
543    // the deserialized value
544    fn generate_deserialize_field_decl(field: &Field) -> Option<syn::Stmt> {
545        (!field.skip).then(|| {
546            let deserialize_field = Self::deserialize_field_variable(field);
547            parse_quote! {
548                let mut #deserialize_field = ::std::option::Option::None;
549            }
550        })
551    }
552
553    fn generate(&self) -> syn::ImplItemFn {
554        let macro_internal = self.0.struct_attrs().macro_internal_path();
555        let (frame_lifetime, metadata_lifetime) = self.0.constraint_lifetimes();
556        let fields = self.0.fields();
557
558        let deserialize_field_decls = fields
559            .iter()
560            .flat_map(Self::generate_deserialize_field_decl);
561        let deserialize_blocks = fields
562            .iter()
563            .filter(|f| !f.skip)
564            .enumerate()
565            .map(|(col_idx, f)| self.generate_deserialization(col_idx, f));
566        let field_idents = fields.iter().map(|f| f.ident.as_ref().unwrap());
567        let nonskipped_field_names = fields
568            .iter()
569            .filter(|&f| (!f.skip))
570            .map(|f| f.cql_name_literal());
571
572        let field_finalizers = fields.iter().map(|f| self.generate_finalize_field(f));
573
574        // TODO: Allow collecting unrecognized fields into some special field
575
576        parse_quote! {
577            fn deserialize(
578                row: #macro_internal::ColumnIterator<#frame_lifetime, #metadata_lifetime>,
579            ) -> ::std::result::Result<Self, #macro_internal::DeserializationError> {
580
581                // Generate fields that will serve as temporary storage
582                // for the fields' values. Those are of type Option<FieldType>.
583                #(#deserialize_field_decls)*
584
585                for col in row {
586                    let col = col.map_err(#macro_internal::row_deser_error_replace_rust_name::<Self>)?;
587                    // Pattern match on the field name and deserialize.
588                    match col.spec.name() {
589                        #(#nonskipped_field_names => #deserialize_blocks,)*
590                        unknown => ::std::unreachable!("Typecheck should have prevented this scenario! Unknown column name: {}", unknown),
591                    }
592                }
593
594                // Create the final struct. The finalizer expressions convert
595                // the temporary storage fields to the final field values.
596                // For example, if a field is missing but marked as
597                // `default_when_null` it will create a default value, otherwise
598                // it will report an error.
599                ::std::result::Result::Ok(Self {
600                    #(#field_idents: #field_finalizers,)*
601                })
602            }
603        }
604    }
