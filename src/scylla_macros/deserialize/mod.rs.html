8/// Common attributes that all deserialize impls should understand.
9trait DeserializeCommonStructAttrs {
10    /// The path to either `scylla` or `scylla_cql` crate.
11    fn crate_path(&self) -> Option<&syn::Path>;
12
13    /// The path to `macro_internal` module,
14    /// which contains exports used by macros.
15    fn macro_internal_path(&self) -> syn::Path {
16        match self.crate_path() {
17            Some(path) => parse_quote!(#path::_macro_internal),
18            None => parse_quote!(scylla::_macro_internal),
19        }
20    }
21}

23/// Provides access to attributes that are common to DeserializeValue
24/// and DeserializeRow traits.
25trait DeserializeCommonFieldAttrs {
26    /// Does the type of this field need Default to be implemented?
27    fn needs_default(&self) -> bool;
28
29    /// The type of the field, i.e. what this field deserializes to.
30    fn deserialize_target(&self) -> &syn::Type;
31}

33/// A structure helpful in implementing DeserializeValue and DeserializeRow.
34///
35/// It implements some common logic for both traits:
36/// - Generates a unique lifetime that binds all other lifetimes in both structs,
37/// - Adds appropriate trait bounds (DeserializeValue + Default)
38struct StructDescForDeserialize<Attrs, Field> {
39    name: syn::Ident,
40    attrs: Attrs,
41    fields: Vec<Field>,
42    constraint_trait: syn::Path,
43    constraint_lifetimes: (syn::Lifetime, syn::Lifetime),
44
45    generics: syn::Generics,
46}
47
48impl<Attrs, Field> StructDescForDeserialize<Attrs, Field>
49where
50    Attrs: FromAttributes + DeserializeCommonStructAttrs,
51    Field: FromField + DeserializeCommonFieldAttrs,
52{
53    fn new(
54        input: &syn::DeriveInput,
55        trait_name: &str,
56        constraint_trait: syn::Path,
57    ) -> Result<Self, syn::Error> {
58        let attrs = Attrs::from_attributes(&input.attrs)?;
59
60        // TODO: support structs with unnamed fields.
61        // A few things to consider:
62        // - such support would necessarily require `enforce_order` and `skip_name_checks` attributes to be passed,
63        // - either:
64        //      - the inner code would have to represent unnamed fields differently and handle the errors differently,
65        //      - or we could use `.0, .1` or `0`, `1` as names for consecutive fields, making representation and error handling uniform.
66        let fields = crate::parser::parse_named_fields(input, trait_name)?
67            .named
68            .iter()
69            .map(Field::from_field)
70            .collect::<Result<_, _>>()?;
71
72        let constraint_lifetimes = generate_pair_of_unique_lifetimes_for_impl(&input.generics);
73
74        Ok(Self {
75            name: input.ident.clone(),
76            attrs,
77            fields,
78            constraint_trait,
79            constraint_lifetimes,
80            generics: input.generics.clone(),
81        })
82    }
83
84    fn struct_attrs(&self) -> &Attrs {
85        &self.attrs
86    }
87
88    fn constraint_lifetimes(&self) -> &(syn::Lifetime, syn::Lifetime) {
89        &self.constraint_lifetimes
90    }
91
92    fn fields(&self) -> &[Field] {
93        &self.fields
94    }
95
96    fn generate_impl(
97        &self,
98        trait_: syn::Path,
99        items: impl IntoIterator<Item = syn::ImplItem>,
100    ) -> syn::ItemImpl {
101        let (frame_lifetime, metadata_lifetime) = self.constraint_lifetimes();
102        let (_, ty_generics, _) = self.generics.split_for_impl();
103        let impl_generics = &self.generics.params;
104
105        let macro_internal = self.attrs.macro_internal_path();
106        let struct_name = &self.name;
107        let predicates = generate_lifetime_constraints_for_impl(
108            &self.generics,
109            self.constraint_trait.clone(),
110            frame_lifetime,
111        )
112        .chain(generate_default_constraints(&self.fields));
113        let trait_: syn::Path = parse_quote!(#macro_internal::#trait_);
114        let items = items.into_iter();
115
116        parse_quote! {
117            #[automatically_derived]
118            impl<#frame_lifetime, #metadata_lifetime, #impl_generics>
119                #trait_<#frame_lifetime, #metadata_lifetime> for #struct_name #ty_generics
120            where #(#predicates),*
121            {
122                #(#items)*
123            }
124        }
125    }
126}

128/// Generates T: Default constraints for those fields that need it.
129fn generate_default_constraints<Field: DeserializeCommonFieldAttrs>(
130    fields: &[Field],
131) -> impl Iterator<Item = syn::WherePredicate> + '_ {
132    fields.iter().filter(|f| f.needs_default()).map(|f| {
133        let t = f.deserialize_target();
134        parse_quote!(#t: ::std::default::Default)
135    })
136}

138/// Helps introduce a lifetime to an `impl` definition that constrains
139/// other lifetimes and types.
140///
141/// The original use case is DeserializeValue and DeserializeRow. Both of those traits
142/// are parametrized with a lifetime. If T: DeserializeValue<'a> then this means
143/// that you can deserialize T as some CQL value from bytes that have
144/// lifetime 'a, similarly for DeserializeRow. In impls for those traits,
145/// an additional lifetime must be introduced and properly constrained.
146fn generate_lifetime_constraints_for_impl<'a>(
147    generics: &'a syn::Generics,
148    trait_full_name: syn::Path,
149    constraint_lifetime: &'a syn::Lifetime,
150) -> impl Iterator<Item = syn::WherePredicate> + 'a {
151    // Constrain the new lifetime with the existing lifetime parameters
152    //     'lifetime: 'a + 'b + 'c ...
153    let mut lifetimes = generics.lifetimes().map(|l| &l.lifetime).peekable();
154    let lifetime_constraints = std::iter::from_fn(move || {
155        let lifetimes = lifetimes.by_ref();
156        lifetimes
157            .peek()
158            .is_some()
159            .then::<syn::WherePredicate, _>(|| parse_quote!(#constraint_lifetime: #(#lifetimes)+*))
160    });
161
162    // For each type parameter T, constrain it like this:
163    //     T: DeserializeValue<'lifetime>,
164    let type_constraints = generics.type_params().map(move |t| {
165        let t_ident = &t.ident;
166        parse_quote!(#t_ident: #trait_full_name<#constraint_lifetime>)
167    });
168
169    lifetime_constraints.chain(type_constraints)
170}

172/// Generates a pair of new lifetime parameters, with a different name to any of the
173/// existing generic lifetimes.
174fn generate_pair_of_unique_lifetimes_for_impl(
175    generics: &syn::Generics,
176) -> (syn::Lifetime, syn::Lifetime) {
177    let mut constraint_lifetime_name = "'lifetime".to_string();
178    fn lifetime_occupied(generics: &syn::Generics, lifetime_name: &str) -> bool {
179        generics
180            .lifetimes()
181            .any(|l| l.lifetime.to_string() == lifetime_name)
182    }
183
184    while lifetime_occupied(generics, &constraint_lifetime_name) {
185        // Extend the lifetime name with another underscore.
186        constraint_lifetime_name += "_";
187    }
188
189    let lifetime1 = syn::Lifetime::new(&constraint_lifetime_name, Span::call_site());
190    constraint_lifetime_name += "_";
191
192    while lifetime_occupied(generics, &constraint_lifetime_name) {
193        // Extend the lifetime name with another underscore.
194        constraint_lifetime_name += "_";
195    }
196    let lifetime2 = syn::Lifetime::new(&constraint_lifetime_name, Span::call_site());
197
198    (lifetime1, lifetime2)
199}
