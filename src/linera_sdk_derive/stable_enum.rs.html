4//! Code generation for the `StableEnum` derive macro.
5//!
6//! Each variant of the input enum is assigned a `u32` "stable tag" computed at
7//! macro-expansion time from the variant's name:
8//!
9//! 1. Compute `Keccak-256(variant_name)`.
10//! 2. Read the first 4 bytes as a big-endian `u32`.
11//! 3. Mask the top 5 bits to `00001`, i.e. `(val & 0x07FF_FFFF) | 0x0800_0000`.
12//!
13//! Step 3 forces the tag into the range `[2^27, 2^28 - 1]`, which is exactly
14//! the range whose ULEB128 encoding is always 4 bytes.
15
16use proc_macro2::{Ident, Span, TokenStream as TokenStream2};
17use sha3::{Digest, Keccak256};
18use syn::{__private::quote::quote, spanned::Spanned, Error, Fields, ItemEnum, Result, Variant};
19
20/// Where the `StableEnumTrace` trait lives, from the perspective of the code
21/// being generated.
22pub enum CrateRoot {
23    /// Generated code lives outside `linera-sdk`: use `::linera_sdk::...`.
24    LineraSdk,
25    /// Generated code lives inside `linera-sdk`: use `crate::...`.
26    Crate,
27}
28
29impl CrateRoot {
30    fn trait_path(&self) -> TokenStream2 {
31        match self {
32            CrateRoot::LineraSdk => quote! { ::linera_sdk::formats::StableEnumTrace },
33            CrateRoot::Crate => quote! { crate::formats::StableEnumTrace },
34        }
35    }
36
37    /// Path to `serde`, routed through `linera_sdk`'s private re-export so
38    /// downstream crates don't need a direct `serde` dependency.
39    fn serde_path(&self) -> TokenStream2 {
40        match self {
41            CrateRoot::LineraSdk => quote! { ::linera_sdk::formats::__private::serde },
42            CrateRoot::Crate => quote! { crate::formats::__private::serde },
43        }
44    }
45
46    /// Path to `serde_reflection`, routed through `linera_sdk`'s private
47    /// re-export so downstream crates don't need a direct `serde-reflection`
48    /// dependency just to use `#[derive(StableEnum)]`.
49    fn serde_reflection_path(&self) -> TokenStream2 {
50        match self {
51            CrateRoot::LineraSdk => quote! { ::linera_sdk::formats::__private::serde_reflection },
52            CrateRoot::Crate => quote! { crate::formats::__private::serde_reflection },
53        }
54    }
55}

57/// Computes the stable tag for a variant name.
58fn compute_tag(variant_name: &str) -> u32 {
59    let hash = Keccak256::digest(variant_name.as_bytes());
60    let val = u32::from_be_bytes([hash[0], hash[1], hash[2], hash[3]]);
61    (val & 0x07FF_FFFF) | 0x0800_0000
62}
63
64/// Computes all variant tags, returning an error on a (vanishingly unlikely) collision.
65fn variant_tags(input: &ItemEnum) -> Result<Vec<(String, u32, &Variant)>> {
66    let mut out = Vec::with_capacity(input.variants.len());
67    for variant in &input.variants {
68        let name = variant.ident.to_string();
69        let tag = compute_tag(&name);
70        if let Some((other_name, _, _)) = out
71            .iter()
72            .find(|(_, t, _): &&(String, u32, &Variant)| *t == tag)
73        {
74            return Err(Error::new(
75                variant.span(),
76                format!(
77                    "Keccak-256 tag collision between variants `{other_name}` and `{name}` \
78                     (tag = {tag:#010x}). Rename one of the variants."
79                ),
80            ));
81        }
82        out.push((name, tag, variant));
83    }
84    Ok(out)
85}
86
87fn reject_generics(input: &ItemEnum) -> Result<()> {
88    if !input.generics.params.is_empty() || input.generics.where_clause.is_some() {
89        return Err(Error::new(
90            input.generics.span(),
91            "#[derive(StableEnum)] does not yet support generic enums",
92        ));
93    }
94    Ok(())
95}

97/// Emits the combined `Serialize` + `Deserialize` + `StableEnumTrace` impls.
98pub fn generate_all(input: &ItemEnum, crate_root: CrateRoot) -> Result<TokenStream2> {
99    let ser = generate_serialize(input, &crate_root)?;
100    let de = generate_deserialize(input, &crate_root)?;
101    let tr = generate_trace(input, &crate_root)?;
102    Ok(quote! {
103        #ser
104        #de
105        #tr
106    })
107}

109pub fn generate_serialize(input: &ItemEnum, crate_root: &CrateRoot) -> Result<TokenStream2> {
110    reject_generics(input)?;
111    let enum_ident = &input.ident;
112    let enum_name_str = enum_ident.to_string();
113    let tagged = variant_tags(input)?;
114    let serde = crate_root.serde_path();
115
116    let arms = tagged.iter().map(|(name, tag, variant)| {
117        let variant_ident = &variant.ident;
118        let name_lit = name.as_str();
119        match &variant.fields {
120            Fields::Unit => quote! {
121                Self::#variant_ident => #serde::Serializer::serialize_unit_variant(
122                    __serializer, #enum_name_str, #tag, #name_lit,
123                ),
124            },
125            Fields::Unnamed(fields) if fields.unnamed.len() == 1 => quote! {
126                Self::#variant_ident(__f0) => #serde::Serializer::serialize_newtype_variant(
127                    __serializer, #enum_name_str, #tag, #name_lit, __f0,
128                ),
129            },
130            Fields::Unnamed(fields) => {
131                let len = fields.unnamed.len();
132                let idents: Vec<Ident> = (0..len)
133                    .map(|i| Ident::new(&format!("__f{i}"), Span::call_site()))
134                    .collect();
135                quote! {
136                    Self::#variant_ident(#(#idents),*) => {
137                        let mut __tv = #serde::Serializer::serialize_tuple_variant(
138                            __serializer, #enum_name_str, #tag, #name_lit, #len,
139                        )?;
140                        #(
141                            #serde::ser::SerializeTupleVariant::serialize_field(&mut __tv, #idents)?;
142                        )*
143                        #serde::ser::SerializeTupleVariant::end(__tv)
144                    }
145                }
146            }
147            Fields::Named(fields) => {
148                let len = fields.named.len();
149                let field_idents: Vec<&Ident> = fields
150                    .named
151                    .iter()
152                    .map(|f| f.ident.as_ref().expect("named field has ident"))
153                    .collect();
154                let field_strs: Vec<String> = field_idents.iter().map(|i| i.to_string()).collect();
155                quote! {
156                    Self::#variant_ident { #(#field_idents),* } => {
157                        let mut __sv = #serde::Serializer::serialize_struct_variant(
158                            __serializer, #enum_name_str, #tag, #name_lit, #len,
159                        )?;
160                        #(
161                            #serde::ser::SerializeStructVariant::serialize_field(
162                                &mut __sv, #field_strs, #field_idents,
163                            )?;
164                        )*
165                        #serde::ser::SerializeStructVariant::end(__sv)
166                    }
167                }
168            }
169        }
170    });
171
172    Ok(quote! {
173        #[automatically_derived]
174        impl #serde::Serialize for #enum_ident {
175            fn serialize<__S>(&self, __serializer: __S) -> ::core::result::Result<__S::Ok, __S::Error>
176            where
177                __S: #serde::Serializer,
178            {
179                match self {
180                    #(#arms)*
181                }
182            }
183        }
184    })
185}

187pub fn generate_deserialize(input: &ItemEnum, crate_root: &CrateRoot) -> Result<TokenStream2> {
188    reject_generics(input)?;
189    let enum_ident = &input.ident;
190    let enum_name_str = enum_ident.to_string();
191    let tagged = variant_tags(input)?;
192    let serde = crate_root.serde_path();
193
194    let variant_name_lits: Vec<String> = tagged.iter().map(|(n, _, _)| n.clone()).collect();
195
196    let arms = tagged.iter().map(|(name, tag, variant)| {
197        let variant_ident = &variant.ident;
198        let name_lit = name.as_str();
199        match &variant.fields {
200            Fields::Unit => quote! {
201                #tag => {
202                    #serde::de::VariantAccess::unit_variant(__variant)?;
203                    ::core::result::Result::Ok(#enum_ident::#variant_ident)
204                }
205            },
206            Fields::Unnamed(fields) if fields.unnamed.len() == 1 => {
207                let ty = &fields.unnamed.first().unwrap().ty;
208                quote! {
209                    #tag => {
210                        let __inner = #serde::de::VariantAccess::newtype_variant::<#ty>(__variant)?;
211                        ::core::result::Result::Ok(#enum_ident::#variant_ident(__inner))
212                    }
213                }
214            }
215            Fields::Unnamed(fields) => {
216                let len = fields.unnamed.len();
217                let types = fields.unnamed.iter().map(|f| &f.ty).collect::<Vec<_>>();
218                let bindings: Vec<Ident> = (0..len)
219                    .map(|i| Ident::new(&format!("__f{i}"), Span::call_site()))
220                    .collect();
221                let next_elements = bindings.iter().zip(types.iter()).enumerate().map(|(i, (b, t))| {
222                    let expected = format!("tuple variant `{name_lit}` with {len} elements");
223                    quote! {
224                        let #b: #t = #serde::de::SeqAccess::next_element::<#t>(&mut __seq)?
225                            .ok_or_else(|| <__A::Error as #serde::de::Error>::invalid_length(#i, &#expected))?;
226                    }
227                });
228                let expecting_msg = format!("tuple variant `{name_lit}`");
229                quote! {
230                    #tag => {
231                        struct __TupleVisitor;
232                        impl<'de> #serde::de::Visitor<'de> for __TupleVisitor {
233                            type Value = (#(#types,)*);
234                            fn expecting(&self, __f: &mut ::core::fmt::Formatter) -> ::core::fmt::Result {
235                                __f.write_str(#expecting_msg)
236                            }
237                            fn visit_seq<__A>(self, mut __seq: __A) -> ::core::result::Result<Self::Value, __A::Error>
238                            where
239                                __A: #serde::de::SeqAccess<'de>,
240                            {
241                                #(#next_elements)*
242                                ::core::result::Result::Ok((#(#bindings,)*))
243                            }
244                        }
245                        let (#(#bindings,)*) = #serde::de::VariantAccess::tuple_variant(
246                            __variant, #len, __TupleVisitor,
247                        )?;
248                        ::core::result::Result::Ok(#enum_ident::#variant_ident(#(#bindings,)*))
249                    }
250                }
251            }
252            Fields::Named(fields) => {
253                let len = fields.named.len();
254                let field_idents: Vec<&Ident> = fields
255                    .named
256                    .iter()
257                    .map(|f| f.ident.as_ref().expect("named field has ident"))
258                    .collect();
259                let field_strs: Vec<String> = field_idents.iter().map(|i| i.to_string()).collect();
260                let types: Vec<_> = fields.named.iter().map(|f| &f.ty).collect();
261                let next_elements = field_idents
262                    .iter()
263                    .zip(types.iter())
264                    .enumerate()
265                    .map(|(i, (b, t))| {
266                        let expected = format!("struct variant `{name_lit}` with {len} fields");
267                        quote! {
268                            let #b: #t = #serde::de::SeqAccess::next_element::<#t>(&mut __seq)?
269                                .ok_or_else(|| <__A::Error as #serde::de::Error>::invalid_length(#i, &#expected))?;
270                        }
271                    });
272                let expecting_msg = format!("struct variant `{name_lit}`");
273                quote! {
274                    #tag => {
275                        struct __StructVisitor;
276                        impl<'de> #serde::de::Visitor<'de> for __StructVisitor {
277                            type Value = (#(#types,)*);
278                            fn expecting(&self, __f: &mut ::core::fmt::Formatter) -> ::core::fmt::Result {
279                                __f.write_str(#expecting_msg)
280                            }
281                            fn visit_seq<__A>(self, mut __seq: __A) -> ::core::result::Result<Self::Value, __A::Error>
282                            where
283                                __A: #serde::de::SeqAccess<'de>,
284                            {
285                                #(#next_elements)*
286                                ::core::result::Result::Ok((#(#field_idents,)*))
287                            }
288                        }
289                        const __FIELDS: &[&::core::primitive::str] = &[#(#field_strs),*];
290                        let (#(#field_idents,)*) = #serde::de::VariantAccess::struct_variant(
291                            __variant, __FIELDS, __StructVisitor,
292                        )?;
293                        ::core::result::Result::Ok(#enum_ident::#variant_ident { #(#field_idents,)* })
294                    }
295                }
296            }
297        }
298    });
299
300    let expecting_msg = format!("enum `{enum_name_str}` (stable-tagged)");
301
302    Ok(quote! {
303        #[automatically_derived]
304        impl<'de> #serde::Deserialize<'de> for #enum_ident {
305            fn deserialize<__D>(__deserializer: __D) -> ::core::result::Result<Self, __D::Error>
306            where
307                __D: #serde::Deserializer<'de>,
308            {
309                struct __OuterVisitor;
310                impl<'de> #serde::de::Visitor<'de> for __OuterVisitor {
311                    type Value = #enum_ident;
312                    fn expecting(&self, __f: &mut ::core::fmt::Formatter) -> ::core::fmt::Result {
313                        __f.write_str(#expecting_msg)
314                    }
315                    fn visit_enum<__A>(self, __access: __A) -> ::core::result::Result<Self::Value, __A::Error>
316                    where
317                        __A: #serde::de::EnumAccess<'de>,
318                    {
319                        let (__tag, __variant) =
320                            #serde::de::EnumAccess::variant::<::core::primitive::u32>(__access)?;
321                        match __tag {
322                            #(#arms)*
323                            __unknown => ::core::result::Result::Err(
324                                <__A::Error as #serde::de::Error>::custom(::std::format!(
325                                    "unknown stable variant tag for `{}`: {:#010x}",
326                                    #enum_name_str, __unknown,
327                                )),
328                            ),
329                        }
330                    }
331                }
332                const __VARIANTS: &[&::core::primitive::str] = &[#(#variant_name_lits),*];
333                #serde::Deserializer::deserialize_enum(
334                    __deserializer,
335                    #enum_name_str,
336                    __VARIANTS,
337                    __OuterVisitor,
338                )
339            }
340        }
341    })
342}

344pub fn generate_trace(input: &ItemEnum, crate_root: &CrateRoot) -> Result<TokenStream2> {
345    reject_generics(input)?;
346    let enum_ident = &input.ident;
347    let enum_name_str = enum_ident.to_string();
348    let tagged = variant_tags(input)?;
349    let trait_path = crate_root.trait_path();
350    let sr = crate_root.serde_reflection_path();
351
352    let variants_const_entries = tagged.iter().map(|(name, tag, _)| {
353        let n = name.as_str();
354        quote! { (#n, #tag) }
355    });
356
357    // For each variant, generate a block that:
358    //   1. calls trace_type_once for each field's type using the caller-supplied samples
359    //   2. constructs the variant and calls trace_value with our scratch samples
360    let blocks = tagged.iter().map(|(_, _, variant)| {
361        let variant_ident = &variant.ident;
362        match &variant.fields {
363            Fields::Unit => quote! {
364                let (__fmt, _) = #sr::Tracer::trace_value(
365                    __tracer,
366                    &mut __scratch,
367                    &#enum_ident::#variant_ident,
368                )?;
369                __format = ::core::option::Option::Some(__fmt);
370            },
371            Fields::Unnamed(fields) => {
372                let bindings: Vec<Ident> = (0..fields.unnamed.len())
373                    .map(|i| Ident::new(&format!("__field_{i}"), Span::call_site()))
374                    .collect();
375                let traces = bindings.iter().zip(fields.unnamed.iter()).map(|(b, f)| {
376                    let ty = &f.ty;
377                    quote! {
378                        let (_, #b): (_, #ty) =
379                            #sr::Tracer::trace_type_once(__tracer, __samples)?;
380                    }
381                });
382                quote! {
383                    #(#traces)*
384                    let (__fmt, _) = #sr::Tracer::trace_value(
385                        __tracer,
386                        &mut __scratch,
387                        &#enum_ident::#variant_ident(#(#bindings),*),
388                    )?;
389                    __format = ::core::option::Option::Some(__fmt);
390                }
391            }
392            Fields::Named(fields) => {
393                let field_idents: Vec<&Ident> = fields
394                    .named
395                    .iter()
396                    .map(|f| f.ident.as_ref().expect("named field has ident"))
397                    .collect();
398                let bindings: Vec<Ident> = field_idents
399                    .iter()
400                    .map(|i| Ident::new(&format!("__field_{i}"), i.span()))
401                    .collect();
402                let traces = bindings.iter().zip(fields.named.iter()).map(|(b, f)| {
403                    let ty = &f.ty;
404                    quote! {
405                        let (_, #b): (_, #ty) =
406                            #sr::Tracer::trace_type_once(__tracer, __samples)?;
407                    }
408                });
409                quote! {
410                    #(#traces)*
411                    let (__fmt, _) = #sr::Tracer::trace_value(
412                        __tracer,
413                        &mut __scratch,
414                        &#enum_ident::#variant_ident {
415                            #( #field_idents: #bindings ),*
416                        },
417                    )?;
418                    __format = ::core::option::Option::Some(__fmt);
419                }
420            }
421        }
422    });
423
424    Ok(quote! {
425        // The impl uses `serde_reflection`, which Linera applications only
426        // depend on for native (non-wasm) builds (e.g. format snapshot tests).
427        #[cfg(not(target_arch = "wasm32"))]
428        #[automatically_derived]
429        impl #trait_path for #enum_ident {
430            const STABLE_VARIANTS: &'static [(&'static ::core::primitive::str, ::core::primitive::u32)] = &[
431                #(#variants_const_entries),*
432            ];
433
434            fn trace_all_variants(
435                __tracer: &mut #sr::Tracer,
436                __samples: &#sr::Samples,
437            ) -> #sr::Result<#sr::Format> {
438                // `trace_value` requires `&mut Samples` even though we have no
439                // need to feed its recordings back to the caller, so we use a
440                // scratch instance here.
441                let mut __scratch = #sr::Samples::default();
442                let mut __format: ::core::option::Option<#sr::Format> =
443                    ::core::option::Option::None;
444                #({ #blocks })*
445                __format.ok_or_else(|| #sr::Error::Custom(
446                    ::std::format!(
447                        "stable enum `{}` has no variants",
448                        #enum_name_str,
449                    ),
450                ))
451            }
452        }
453    })
