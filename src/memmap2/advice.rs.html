5/// Values supported by [Mmap::advise] and [MmapMut::advise] functions.
6/// See [madvise()](https://man7.org/linux/man-pages/man2/madvise.2.html) map page.
7#[repr(i32)]
8#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
9pub enum Advice {
10    /// **MADV_NORMAL**
11    ///
12    /// No special treatment.  This is the default.
13    Normal = libc::MADV_NORMAL,
14
15    /// **MADV_RANDOM**
16    ///
17    /// Expect page references in random order.  (Hence, read
18    /// ahead may be less useful than normally.)
19    Random = libc::MADV_RANDOM,
20
21    /// **MADV_SEQUENTIAL**
22    ///
23    /// Expect page references in sequential order.  (Hence, pages
24    /// in the given range can be aggressively read ahead, and may
25    /// be freed soon after they are accessed.)
26    Sequential = libc::MADV_SEQUENTIAL,
27
28    /// **MADV_WILLNEED**
29    ///
30    /// Expect access in the near future.  (Hence, it might be a
31    /// good idea to read some pages ahead.)
32    WillNeed = libc::MADV_WILLNEED,
33
34    /// **MADV_DONTNEED**
35    ///
36    /// Do not expect access in the near future.  (For the time
37    /// being, the application is finished with the given range,
38    /// so the kernel can free resources associated with it.)
39    ///
40    /// After a successful MADV_DONTNEED operation, the semantics
41    /// of memory access in the specified region are changed:
42    /// subsequent accesses of pages in the range will succeed,
43    /// but will result in either repopulating the memory contents
44    /// from the up-to-date contents of the underlying mapped file
45    /// (for shared file mappings, shared anonymous mappings, and
46    /// shmem-based techniques such as System V shared memory
47    /// segments) or zero-fill-on-demand pages for anonymous
48    /// private mappings.
49    ///
50    /// Note that, when applied to shared mappings, MADV_DONTNEED
51    /// might not lead to immediate freeing of the pages in the
52    /// range.  The kernel is free to delay freeing the pages
53    /// until an appropriate moment.  The resident set size (RSS)
54    /// of the calling process will be immediately reduced
55    /// however.
56    ///
57    /// **MADV_DONTNEED** cannot be applied to locked pages, Huge TLB
58    /// pages, or VM_PFNMAP pages.  (Pages marked with the kernel-
59    /// internal VM_PFNMAP flag are special memory areas that are
60    /// not managed by the virtual memory subsystem.  Such pages
61    /// are typically created by device drivers that map the pages
62    /// into user space.)
63    DontNeed = libc::MADV_DONTNEED,
64
65    //
66    // The rest are Linux-specific
67    //
68    /// **MADV_FREE** - Linux (since Linux 4.5) and Darwin
69    ///
70    /// The application no longer requires the pages in the range
71    /// specified by addr and len.  The kernel can thus free these
72    /// pages, but the freeing could be delayed until memory
73    /// pressure occurs.  For each of the pages that has been
74    /// marked to be freed but has not yet been freed, the free
75    /// operation will be canceled if the caller writes into the
76    /// page.  After a successful MADV_FREE operation, any stale
77    /// data (i.e., dirty, unwritten pages) will be lost when the
78    /// kernel frees the pages.  However, subsequent writes to
79    /// pages in the range will succeed and then kernel cannot
80    /// free those dirtied pages, so that the caller can always
81    /// see just written data.  If there is no subsequent write,
82    /// the kernel can free the pages at any time.  Once pages in
83    /// the range have been freed, the caller will see zero-fill-
84    /// on-demand pages upon subsequent page references.
85    ///
86    /// The MADV_FREE operation can be applied only to private
87    /// anonymous pages (see mmap(2)).  In Linux before version
88    /// 4.12, when freeing pages on a swapless system, the pages
89    /// in the given range are freed instantly, regardless of
90    /// memory pressure.
91    #[cfg(any(target_os = "linux", target_os = "macos", target_os = "ios"))]
92    Free = libc::MADV_FREE,
93
94    /// **MADV_REMOVE** - Linux only (since Linux 2.6.16)
95    ///
96    /// Free up a given range of pages and its associated backing
97    /// store.  This is equivalent to punching a hole in the
98    /// corresponding byte range of the backing store (see
99    /// fallocate(2)).  Subsequent accesses in the specified
100    /// address range will see bytes containing zero.
101    ///
102    /// The specified address range must be mapped shared and
103    /// writable.  This flag cannot be applied to locked pages,
104    /// Huge TLB pages, or VM_PFNMAP pages.
105    ///
106    /// In the initial implementation, only tmpfs(5) was supported
107    /// **MADV_REMOVE**; but since Linux 3.5, any filesystem which
108    /// supports the fallocate(2) FALLOC_FL_PUNCH_HOLE mode also
109    /// supports MADV_REMOVE.  Hugetlbfs fails with the error
110    /// EINVAL and other filesystems fail with the error
111    /// EOPNOTSUPP.
112    #[cfg(target_os = "linux")]
113    Remove = libc::MADV_REMOVE,
114
115    /// **MADV_DONTFORK** - Linux only (since Linux 2.6.16)
116    ///
117    /// Do not make the pages in this range available to the child
118    /// after a fork(2).  This is useful to prevent copy-on-write
119    /// semantics from changing the physical location of a page if
120    /// the parent writes to it after a fork(2).  (Such page
121    /// relocations cause problems for hardware that DMAs into the
122    /// page.)
123    #[cfg(target_os = "linux")]
124    DontFork = libc::MADV_DONTFORK,
125
126    /// **MADV_DOFORK** - Linux only (since Linux 2.6.16)
127    ///
128    /// Undo the effect of MADV_DONTFORK, restoring the default
129    /// behavior, whereby a mapping is inherited across fork(2).
130    #[cfg(target_os = "linux")]
131    DoFork = libc::MADV_DOFORK,
132
133    /// **MADV_MERGEABLE** - Linux only (since Linux 2.6.32)
134    ///
135    /// Enable Kernel Samepage Merging (KSM) for the pages in the
136    /// range specified by addr and length.  The kernel regularly
137    /// scans those areas of user memory that have been marked as
138    /// mergeable, looking for pages with identical content.
139    /// These are replaced by a single write-protected page (which
140    /// is automatically copied if a process later wants to update
141    /// the content of the page).  KSM merges only private
142    /// anonymous pages (see mmap(2)).
143    ///
144    /// The KSM feature is intended for applications that generate
145    /// many instances of the same data (e.g., virtualization
146    /// systems such as KVM).  It can consume a lot of processing
147    /// power; use with care.  See the Linux kernel source file
148    /// Documentation/admin-guide/mm/ksm.rst for more details.
149    ///
150    /// The MADV_MERGEABLE and MADV_UNMERGEABLE operations are
151    /// available only if the kernel was configured with
152    /// CONFIG_KSM.
153    #[cfg(target_os = "linux")]
154    Mergeable = libc::MADV_MERGEABLE,
155
156    /// **MADV_UNMERGEABLE** - Linux only (since Linux 2.6.32)
157    ///
158    /// Undo the effect of an earlier MADV_MERGEABLE operation on
159    /// the specified address range; KSM unmerges whatever pages
160    /// it had merged in the address range specified by addr and
161    /// length.
162    #[cfg(target_os = "linux")]
163    Unmergeable = libc::MADV_UNMERGEABLE,
164
165    /// **MADV_HUGEPAGE** - Linux only (since Linux 2.6.38)
166    ///
167    /// Enable Transparent Huge Pages (THP) for pages in the range
168    /// specified by addr and length.  Currently, Transparent Huge
169    /// Pages work only with private anonymous pages (see
170    /// mmap(2)).  The kernel will regularly scan the areas marked
171    /// as huge page candidates to replace them with huge pages.
172    /// The kernel will also allocate huge pages directly when the
173    /// region is naturally aligned to the huge page size (see
174    /// posix_memalign(2)).
175    ///
176    /// This feature is primarily aimed at applications that use
177    /// large mappings of data and access large regions of that
178    /// memory at a time (e.g., virtualization systems such as
179    /// QEMU).  It can very easily waste memory (e.g., a 2 MB
180    /// mapping that only ever accesses 1 byte will result in 2 MB
181    /// of wired memory instead of one 4 KB page).  See the Linux
182    /// kernel source file
183    /// Documentation/admin-guide/mm/transhuge.rst for more
184    /// details.
185    ///
186    /// Most common kernels configurations provide MADV_HUGEPAGE-
187    /// style behavior by default, and thus MADV_HUGEPAGE is
188    /// normally not necessary.  It is mostly intended for
189    /// embedded systems, where MADV_HUGEPAGE-style behavior may
190    /// not be enabled by default in the kernel.  On such systems,
191    /// this flag can be used in order to selectively enable THP.
192    /// Whenever MADV_HUGEPAGE is used, it should always be in
193    /// regions of memory with an access pattern that the
194    /// developer knows in advance won't risk to increase the
195    /// memory footprint of the application when transparent
196    /// hugepages are enabled.
197    ///
198    /// The MADV_HUGEPAGE and MADV_NOHUGEPAGE operations are
199    /// available only if the kernel was configured with
200    /// CONFIG_TRANSPARENT_HUGEPAGE.
201    #[cfg(target_os = "linux")]
202    HugePage = libc::MADV_HUGEPAGE,
203
204    /// **MADV_NOHUGEPAGE** - Linux only (since Linux 2.6.38)
205    ///
206    /// Ensures that memory in the address range specified by addr
207    /// and length will not be backed by transparent hugepages.
208    #[cfg(target_os = "linux")]
209    NoHugePage = libc::MADV_NOHUGEPAGE,
210
211    /// **MADV_DONTDUMP** - Linux only (since Linux 3.4)
212    ///
213    /// Exclude from a core dump those pages in the range
214    /// specified by addr and length.  This is useful in
215    /// applications that have large areas of memory that are
216    /// known not to be useful in a core dump.  The effect of
217    /// **MADV_DONTDUMP** takes precedence over the bit mask that is
218    /// set via the `/proc/[pid]/coredump_filter` file (see
219    /// core(5)).
220    #[cfg(target_os = "linux")]
221    DontDump = libc::MADV_DONTDUMP,
222
223    /// **MADV_DODUMP** - Linux only (since Linux 3.4)
224    ///
225    /// Undo the effect of an earlier MADV_DONTDUMP.
226    #[cfg(target_os = "linux")]
227    DoDump = libc::MADV_DODUMP,
228
229    /// **MADV_HWPOISON** - Linux only (since Linux 2.6.32)
230    ///
231    /// Poison the pages in the range specified by addr and length
232    /// and handle subsequent references to those pages like a
233    /// hardware memory corruption.  This operation is available
234    /// only for privileged (CAP_SYS_ADMIN) processes.  This
235    /// operation may result in the calling process receiving a
236    /// SIGBUS and the page being unmapped.
237    ///
238    /// This feature is intended for testing of memory error-
239    /// handling code; it is available only if the kernel was
240    /// configured with CONFIG_MEMORY_FAILURE.
241    #[cfg(target_os = "linux")]
242    HwPoison = libc::MADV_HWPOISON,
243
244    /// **MADV_ZERO_WIRED_PAGES** - Darwin only
245    ///
246    /// Indicates that the application would like the wired pages in this address range to be
247    /// zeroed out if the address range is deallocated without first unwiring the pages (i.e.
248    /// a munmap(2) without a preceding munlock(2) or the application quits).  This is used
249    /// with madvise() system call.
250    #[cfg(any(target_os = "macos", target_os = "ios"))]
251    ZeroWiredPages = libc::MADV_ZERO_WIRED_PAGES,
252
253    /// **MADV_FREE_REUSABLE** - Darwin only
254    ///
255    /// Behaves like **MADV_FREE**, but the freed pages are accounted for in the RSS of the process.
256    #[cfg(any(target_os = "macos", target_os = "ios"))]
257    FreeReusable = libc::MADV_FREE_REUSABLE,
258
259    /// **MADV_FREE_REUSE** - Darwin only
260    ///
261    /// Marks a memory region previously freed by **MADV_FREE_REUSABLE** as non-reusable, accounts
262    /// for the pages in the RSS of the process. Pages that have been freed will be replaced by
263    /// zero-filled pages on demand, other pages will be left as is.
264    #[cfg(any(target_os = "macos", target_os = "ios"))]
265    FreeReuse = libc::MADV_FREE_REUSE,
266}
