37    /// Creates a new `MmapInner`.
38    ///
39    /// This is a thin wrapper around the `mmap` sytem call.
40    fn new(
41        len: usize,
42        prot: libc::c_int,
43        flags: libc::c_int,
44        file: RawFd,
45        offset: u64,
46    ) -> io::Result<MmapInner> {
47        let alignment = offset % page_size() as u64;
48        let aligned_offset = offset - alignment;
49        let aligned_len = len + alignment as usize;
50
51        // `libc::mmap` does not support zero-size mappings. POSIX defines:
52        //
53        // https://pubs.opengroup.org/onlinepubs/9699919799/functions/mmap.html
54        // > If `len` is zero, `mmap()` shall fail and no mapping shall be established.
55        //
56        // So if we would create such a mapping, crate a one-byte mapping instead:
57        let aligned_len = aligned_len.max(1);
58
59        // Note that in that case `MmapInner::len` is still set to zero,
60        // and `Mmap` will still dereferences to an empty slice.
61        //
62        // If this mapping is backed by an empty file, we create a mapping larger than the file.
63        // This is unusual but well-defined. On the same man page, POSIX further defines:
64        //
65        // > The `mmap()` function can be used to map a region of memory that is larger
66        // > than the current size of the object.
67        //
68        // (The object here is the file.)
69        //
70        // > Memory access within the mapping but beyond the current end of the underlying
71        // > objects may result in SIGBUS signals being sent to the process. The reason for this
72        // > is that the size of the object can be manipulated by other processes and can change
73        // > at any moment. The implementation should tell the application that a memory reference
74        // > is outside the object where this can be detected; otherwise, written data may be lost
75        // > and read data may not reflect actual data in the object.
76        //
77        // Because `MmapInner::len` is not incremented, this increment of `aligned_len`
78        // will not allow accesses past the end of the file and will not cause SIGBUS.
79        //
80        // (SIGBUS is still possible by mapping a non-empty file and then truncating it
81        // to a shorter size, but that is unrelated to this handling of empty files.)
82
83        unsafe {
84            let ptr = libc::mmap(
85                ptr::null_mut(),
86                aligned_len as libc::size_t,
87                prot,
88                flags,
89                file,
90                aligned_offset as libc::off_t,
91            );
92
93            if ptr == libc::MAP_FAILED {
94                Err(io::Error::last_os_error())
95            } else {
96                Ok(MmapInner {
97                    ptr: ptr.offset(alignment as isize),
98                    len,
99                })
100            }
101        }
102    }

177    pub fn flush(&self, offset: usize, len: usize) -> io::Result<()> {
178        let alignment = (self.ptr as usize + offset) % page_size();
179        let offset = offset as isize - alignment as isize;
180        let len = len + alignment;
181        let result =
182            unsafe { libc::msync(self.ptr.offset(offset), len as libc::size_t, libc::MS_SYNC) };
183        if result == 0 {
184            Ok(())
185        } else {
186            Err(io::Error::last_os_error())
187        }
188    }

203    fn mprotect(&mut self, prot: libc::c_int) -> io::Result<()> {
204        unsafe {
205            let alignment = self.ptr as usize % page_size();
206            let ptr = self.ptr.offset(-(alignment as isize));
207            let len = self.len + alignment;
208            let len = len.max(1);
209            if libc::mprotect(ptr, len, prot) == 0 {
210                Ok(())
211            } else {
212                Err(io::Error::last_os_error())
213            }
214        }
215    }

278impl Drop for MmapInner {
279    fn drop(&mut self) {
280        let alignment = self.ptr as usize % page_size();
281        let len = self.len + alignment;
282        let len = len.max(1);
283        // Any errors during unmapping/closing are ignored as the only way
284        // to report them would be through panicking which is highly discouraged
285        // in Drop impls, c.f. https://github.com/rust-lang/lang-team/issues/97
286        unsafe {
287            let ptr = self.ptr.offset(-(alignment as isize));
288            libc::munmap(ptr, len as libc::size_t);
289        }
290    }
291}

296fn page_size() -> usize {
297    static PAGE_SIZE: AtomicUsize = AtomicUsize::new(0);
298
299    match PAGE_SIZE.load(Ordering::Relaxed) {
300        0 => {
301            let page_size = unsafe { libc::sysconf(libc::_SC_PAGESIZE) as usize };
302
303            PAGE_SIZE.store(page_size, Ordering::Relaxed);
304
305            page_size
306        }
307        page_size => page_size,
308    }
309}

311pub fn file_len(file: RawFd) -> io::Result<u64> {
312    // SAFETY: We must not close the passed-in fd by dropping the File we create,
313    // we ensure this by immediately wrapping it in a ManuallyDrop.
314    unsafe {
315        let file = ManuallyDrop::new(File::from_raw_fd(file));
316        Ok(file.metadata()?.len())
317    }
