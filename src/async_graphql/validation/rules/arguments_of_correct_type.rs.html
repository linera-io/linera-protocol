15#[derive(Default)]
16pub struct ArgumentsOfCorrectType<'a> {
17    current_args: Option<&'a IndexMap<String, MetaInputValue>>,
18}
19
20impl<'a> Visitor<'a> for ArgumentsOfCorrectType<'a> {
21    fn enter_directive(
22        &mut self,
23        ctx: &mut VisitorContext<'a>,
24        directive: &'a Positioned<Directive>,
25    ) {
26        self.current_args = ctx
27            .registry
28            .directives
29            .get(directive.node.name.node.as_str())
30            .map(|d| &d.args);
31    }
32
33    fn exit_directive(
34        &mut self,
35        _ctx: &mut VisitorContext<'a>,
36        _directive: &'a Positioned<Directive>,
37    ) {
38        self.current_args = None;
39    }
40
41    fn enter_argument(
42        &mut self,
43        ctx: &mut VisitorContext<'a>,
44        name: &'a Positioned<Name>,
45        value: &'a Positioned<Value>,
46    ) {
47        if let Some(arg) = self
48            .current_args
49            .and_then(|args| args.get(name.node.as_str()))
50        {
51            let value = value
52                .node
53                .clone()
54                .into_const_with(|var_name| {
55                    ctx.variables
56                        .and_then(|variables| variables.get(&var_name))
57                        .cloned()
58                        .ok_or(())
59                })
60                .ok();
61
62            if let Some(reason) = value.and_then(|value| {
63                is_valid_input_value(
64                    ctx.registry,
65                    &arg.ty,
66                    &value,
67                    QueryPathNode {
68                        parent: None,
69                        segment: QueryPathSegment::Name(&arg.name),
70                    },
71                )
72            }) {
73                ctx.report_error(
74                    vec![name.pos],
75                    format!("Invalid value for argument {}", reason),
76                );
77            }
78        }
79    }
80
81    fn enter_field(&mut self, ctx: &mut VisitorContext<'a>, field: &'a Positioned<Field>) {
82        self.current_args = ctx
83            .parent_type()
84            .and_then(|p| p.field_by_name(&field.node.name.node))
85            .map(|f| &f.args);
86    }
87
88    fn exit_field(&mut self, _ctx: &mut VisitorContext<'a>, _field: &'a Positioned<Field>) {
89        self.current_args = None;
90    }
91}
