4use std::{collections::HashMap, sync::OnceLock};
5
6use prometheus::{
7    core::{Collector, Desc},
8    proto::{Counter, Gauge, LabelPair, Metric, MetricFamily, MetricType},
9};
10use tokio::runtime::Handle;
11
12struct TokioRuntimeCollector {
13    handle: Handle,
14    workers: Desc,
15    alive_tasks: Desc,
16    global_queue: Desc,
17    busy_seconds: Desc,
18    parks: Desc,
19    park_unparks: Desc,
20}
21
22impl TokioRuntimeCollector {
23    fn new(handle: Handle) -> Self {
24        let desc = |name: &str, help: &str, labels: Vec<String>| {
25            Desc::new(name.into(), help.into(), labels, HashMap::new())
26                .expect("static metric descriptor is always valid")
27        };
28        let workers = desc(
29            "linera_tokio_workers",
30            "Number of worker threads in the tokio runtime.",
31            vec![],
32        );
33        let alive_tasks = desc(
34            "linera_tokio_alive_tasks",
35            "Number of tasks currently alive in the tokio runtime.",
36            vec![],
37        );
38        let global_queue = desc(
39            "linera_tokio_global_queue_depth",
40            "Number of tasks in the runtime's global injection queue.",
41            vec![],
42        );
43        let worker_label = vec!["worker".to_string()];
44        let busy_seconds = desc(
45            "linera_tokio_worker_busy_seconds_total",
46            "Cumulative time each worker has spent executing tasks, in seconds.",
47            worker_label.clone(),
48        );
49        let parks = desc(
50            "linera_tokio_worker_parks_total",
51            "Cumulative number of times each worker has parked (ran out of work).",
52            worker_label.clone(),
53        );
54        let park_unparks = desc(
55            "linera_tokio_worker_park_unparks_total",
56            "Monotonically increasing count of park and unpark events combined per worker. \
57             An odd value means the worker is currently parked.",
58            worker_label,
59        );
60        Self {
61            handle,
62            workers,
63            alive_tasks,
64            global_queue,
65            busy_seconds,
66            parks,
67            park_unparks,
68        }
69    }
70}
71
72impl Collector for TokioRuntimeCollector {
73    fn desc(&self) -> Vec<&Desc> {
74        vec![
75            &self.workers,
76            &self.alive_tasks,
77            &self.global_queue,
78            &self.busy_seconds,
79            &self.parks,
80            &self.park_unparks,
81        ]
82    }
83
84    fn collect(&self) -> Vec<MetricFamily> {
85        let m = self.handle.metrics();
86        let num_workers = m.num_workers();
87        let mut families = Vec::with_capacity(6);
88
89        families.push(gauge_family(&self.workers, num_workers as f64));
90        families.push(gauge_family(&self.alive_tasks, m.num_alive_tasks() as f64));
91        families.push(gauge_family(
92            &self.global_queue,
93            m.global_queue_depth() as f64,
94        ));
95
96        #[cfg(target_has_atomic = "64")]
97        {
98            let mut busy = counter_family(&self.busy_seconds);
99            let mut parks = counter_family(&self.parks);
100            let mut park_unparks = counter_family(&self.park_unparks);
101            for i in 0..num_workers {
102                let label = worker_label(i);
103                busy.mut_metric().push(counter_metric(
104                    m.worker_total_busy_duration(i).as_secs_f64(),
105                    &label,
106                ));
107                parks
108                    .mut_metric()
109                    .push(counter_metric(m.worker_park_count(i) as f64, &label));
110                park_unparks
111                    .mut_metric()
112                    .push(counter_metric(m.worker_park_unpark_count(i) as f64, &label));
113            }
114            families.push(busy);
115            families.push(parks);
116            families.push(park_unparks);
117        }
118
119        families
120    }
121}

123fn gauge_family(desc: &Desc, value: f64) -> MetricFamily {
124    let mut gauge = Gauge::new();
125    gauge.set_value(value);
126    let mut metric = Metric::new();
127    metric.set_gauge(gauge);
128    let mut family = MetricFamily::new();
129    family.set_name(desc.fq_name.clone());
130    family.set_help(desc.help.clone());
131    family.set_field_type(MetricType::GAUGE);
132    family.mut_metric().push(metric);
133    family
134}
135
136fn counter_family(desc: &Desc) -> MetricFamily {
137    let mut family = MetricFamily::new();
138    family.set_name(desc.fq_name.clone());
139    family.set_help(desc.help.clone());
140    family.set_field_type(MetricType::COUNTER);
141    family
142}
143
144fn counter_metric(value: f64, labels: &[LabelPair]) -> Metric {
145    let mut counter = Counter::new();
146    counter.set_value(value);
147    let mut metric = Metric::new();
148    metric.set_counter(counter);
149    metric.set_label(labels.to_vec().into());
150    metric
151}
152
153fn worker_label(i: usize) -> [LabelPair; 1] {
154    let mut lp = LabelPair::new();
155    lp.set_name("worker".into());
156    lp.set_value(i.to_string());
157    [lp]
158}

160/// Must be called from within a tokio runtime context.
161pub(crate) fn register() {
162    static ONCE: OnceLock<()> = OnceLock::new();
163    ONCE.get_or_init(|| {
164        let collector = Box::new(TokioRuntimeCollector::new(Handle::current()));
165        if let Err(e) = prometheus::register(collector) {
166            tracing::warn!("failed to register tokio runtime metrics: {e}");
167        }
168    });
169}
