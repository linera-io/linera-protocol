5/// Controls how the resource version parameter is applied for list calls
6///
7/// Not specifying a `VersionMatch` strategy will give you different semantics
8/// depending on what `resource_version`, `limit`, `continue_token` you include with the list request.
9///
10/// See <https://kubernetes.io/docs/reference/using-api/api-concepts/#semantics-for-get-and-list> for details.
11#[derive(Clone, Debug, PartialEq)]
12pub enum VersionMatch {
13    /// Returns data at least as new as the provided resource version.
14    ///
15    /// The newest available data is preferred, but any data not older than the provided resource version may be served.
16    /// This guarantees that the collection's resource version is not older than the requested resource version,
17    /// but does not make any guarantee about the resource version of any of the items in that collection.
18    ///
19    /// ### Any Version
20    /// A degenerate, but common sub-case of `NotOlderThan` is when used together with `resource_version` "0".
21    ///
22    /// It is possible for a "0" resource version request to return data at a much older resource version
23    /// than the client has previously observed, particularly in HA configurations, due to partitions or stale caches.
24    /// Clients that cannot tolerate this should not use this semantic.
25    NotOlderThan,
26
27    /// Return data at the exact resource version provided.
28    ///
29    /// If the provided resource version  is unavailable, the server responds with HTTP 410 "Gone".
30    /// For list requests to servers that honor the resource version Match parameter, this guarantees that the collection's
31    /// resource version  is the same as the resource version  you requested in the query string.
32    /// That guarantee does not apply to the resource version  of any items within that collection.
33    ///
34    /// Note that `Exact` cannot be used with resource version "0". For the most up-to-date list; use `Unset`.
35    Exact,
36}

38/// Common query parameters used in list/delete calls on collections
39#[derive(Clone, Debug, Default, PartialEq)]
40pub struct ListParams {
41    /// A selector to restrict the list of returned objects by their labels.
42    ///
43    /// Defaults to everything if `None`.
44    pub label_selector: Option<String>,
45
46    /// A selector to restrict the list of returned objects by their fields.
47    ///
48    /// Defaults to everything if `None`.
49    pub field_selector: Option<String>,
50
51    /// Timeout for the list/watch call.
52    ///
53    /// This limits the duration of the call, regardless of any activity or inactivity.
54    pub timeout: Option<u32>,
55
56    /// Limit the number of results.
57    ///
58    /// If there are more results, the server will respond with a continue token which can be used to fetch another page
59    /// of results. See the [Kubernetes API docs](https://kubernetes.io/docs/reference/using-api/api-concepts/#retrieving-large-results-sets-in-chunks)
60    /// for pagination details.
61    pub limit: Option<u32>,
62
63    /// Fetch a second page of results.
64    ///
65    /// After listing results with a limit, a continue token can be used to fetch another page of results.
66    pub continue_token: Option<String>,
67
68    /// Determines how resourceVersion is matched applied to list calls.
69    pub version_match: Option<VersionMatch>,
70
71    /// An explicit resourceVersion using the given `VersionMatch` strategy
72    ///
73    /// See <https://kubernetes.io/docs/reference/using-api/api-concepts/#resource-versions> for details.
74    pub resource_version: Option<String>,
75}
76
77impl ListParams {
78    pub(crate) fn validate(&self) -> Result<(), Error> {
79        if let Some(rv) = &self.resource_version {
80            if self.version_match == Some(VersionMatch::Exact) && rv == "0" {
81                return Err(Error::Validation(
82                    "A non-zero resource_version is required when using an Exact match".into(),
83                ));
84            }
85        } else if self.version_match.is_some() {
86            return Err(Error::Validation(
87                "A resource_version is required when using an explicit match".into(),
88            ));
89        }
90        Ok(())
91    }
92
93    // Partially populate query parameters (needs resourceVersion out of band)
94    pub(crate) fn populate_qp(&self, qp: &mut form_urlencoded::Serializer<String>) {
95        if let Some(fields) = &self.field_selector {
96            qp.append_pair("fieldSelector", fields);
97        }
98        if let Some(labels) = &self.label_selector {
99            qp.append_pair("labelSelector", labels);
100        }
101        if let Some(limit) = &self.limit {
102            qp.append_pair("limit", &limit.to_string());
103        }
104        if let Some(continue_token) = &self.continue_token {
105            qp.append_pair("continue", continue_token);
106        } else {
107            // When there's a continue token, we don't want to set resourceVersion
108            if let Some(rv) = &self.resource_version {
109                if rv != "0" || self.limit.is_none() {
110                    qp.append_pair("resourceVersion", rv.as_str());
111
112                    match &self.version_match {
113                        None => {}
114                        Some(VersionMatch::NotOlderThan) => {
115                            qp.append_pair("resourceVersionMatch", "NotOlderThan");
116                        }
117                        Some(VersionMatch::Exact) => {
118                            qp.append_pair("resourceVersionMatch", "Exact");
119                        }
120                    }
121                }
122            }
123        }
124    }
125}

127/// Builder interface to ListParams
128///
129/// Usage:
130/// ```
131/// use kube::api::ListParams;
132/// let lp = ListParams::default()
133///     .match_any()
134///     .timeout(60)
135///     .labels("kubernetes.io/lifecycle=spot");

137impl ListParams {

139    ///
140    /// This limits the duration of the call, regardless of any activity or inactivity.

148    /// Configure the selector to restrict the list of returned objects by their fields.
149    ///
150    /// Defaults to everything.

152    /// The server only supports a limited number of field queries per type.

155        self.field_selector = Some(field_selector.to_string());
156        self
157    }
158
159    /// Configure the selector to restrict the list of returned objects by their labels.
160    ///
161    /// Defaults to everything.
162    /// Supports `=`, `==`, `!=`, and can be comma separated: `key1=value1,key2=value2`.
163    #[must_use]
164    pub fn labels(mut self, label_selector: &str) -> Self {
165        self.label_selector = Some(label_selector.to_string());
166        self
167    }
168
169    /// Sets a result limit.
170    #[must_use]
171    pub fn limit(mut self, limit: u32) -> Self {
172        self.limit = Some(limit);
173        self
174    }
175
176    /// Sets a continue token.
177    #[must_use]
178    pub fn continue_token(mut self, token: &str) -> Self {
179        self.continue_token = Some(token.to_string());
180        self
181    }
182
183    /// Sets the resource version
184    #[must_use]
185    pub fn at(mut self, resource_version: &str) -> Self {
186        self.resource_version = Some(resource_version.into());
187        self
188    }
189
190    /// Sets an arbitary resource version match strategy
191    ///
192    /// A non-default strategy such as `VersionMatch::Exact` or `VersionMatch::NotOlderThan`
193    /// requires an explicit `resource_version` set to pass request validation.
194    #[must_use]
195    pub fn matching(mut self, version_match: VersionMatch) -> Self {
196        self.version_match = Some(version_match);
197        self
198    }
199
200    /// Use the semantic "any" resource version strategy
201    ///
202    /// This is a less taxing variant of the default list, returning data at any resource version.
203    /// It will prefer the newest avialable resource version, but strong consistency is not required;
204    /// data at any resource version may be served.
205    /// It is possible for the request to return data at a much older resource version than the client
206    /// has previously observed, particularly in high availability configurations, due to partitions or stale caches.
207    /// Clients that cannot tolerate this should not use this semantic.
208    #[must_use]
209    pub fn match_any(self) -> Self {
210        self.matching(VersionMatch::NotOlderThan).at("0")
211    }

249/// The validation directive to use for `fieldValidation` when using server-side apply.
250#[derive(Clone, Debug)]
251pub enum ValidationDirective {
252    /// Strict mode will fail any invalid manifests.
253    ///
254    /// This will fail the request with a BadRequest error if any unknown fields would be dropped from the
255    /// object, or if any duplicate fields are present. The error returned from the server will contain
256    /// all unknown and duplicate fields encountered.
257    Strict,
258    /// Warn mode will return a warning for invalid manifests.
259    ///
260    /// This will send a warning via the standard warning response header for each unknown field that
261    /// is dropped from the object, and for each duplicate field that is encountered. The request will
262    /// still succeed if there are no other errors, and will only persist the last of any duplicate fields.
263    Warn,
264    /// Ignore mode will silently ignore any problems.
265    ///
266    /// This will ignore any unknown fields that are silently dropped from the object, and will ignore
267    /// all but the last duplicate field that the decoder encounters.
268    Ignore,
269}
270
271impl ValidationDirective {
272    /// Returns the string format of the directive
273    pub fn as_str(&self) -> &str {
274        match self {
275            Self::Strict => "Strict",
276            Self::Warn => "Warn",
277            Self::Ignore => "Ignore",
278        }
279    }
280}

282/// Common query parameters used in watch calls on collections
283#[derive(Clone, Debug, PartialEq)]
284pub struct WatchParams {
285    /// A selector to restrict returned objects by their labels.
286    ///
287    /// Defaults to everything if `None`.
288    pub label_selector: Option<String>,
289
290    /// A selector to restrict returned objects by their fields.
291    ///
292    /// Defaults to everything if `None`.
293    pub field_selector: Option<String>,
294
295    /// Timeout for the watch call.
296    ///
297    /// This limits the duration of the call, regardless of any activity or inactivity.
298    /// If unset for a watch call, we will use 290s.
299    /// We limit this to 295s due to [inherent watch limitations](https://github.com/kubernetes/kubernetes/issues/6513).
300    pub timeout: Option<u32>,
301
302    /// Enables watch events with type "BOOKMARK".
303    ///
304    /// Servers that do not implement bookmarks ignore this flag and
305    /// bookmarks are sent at the server's discretion. Clients should not
306    /// assume bookmarks are returned at any specific interval, nor may they
307    /// assume the server will send any BOOKMARK event during a session.
308    /// If this is not a watch, this field is ignored.
309    /// If the feature gate WatchBookmarks is not enabled in apiserver,
310    /// this field is ignored.
311    pub bookmarks: bool,
312
313    /// Kubernetes 1.27 Streaming Lists
314    /// `sendInitialEvents=true` may be set together with `watch=true`.
315    /// In that case, the watch stream will begin with synthetic events to
316    /// produce the current state of objects in the collection. Once all such
317    /// events have been sent, a synthetic "Bookmark" event  will be sent.
318    /// The bookmark will report the ResourceVersion (RV) corresponding to the
319    /// set of objects, and be marked with `"k8s.io/initial-events-end": "true"` annotation.
320    /// Afterwards, the watch stream will proceed as usual, sending watch events
321    /// corresponding to changes (subsequent to the RV) to objects watched.
322    ///
323    /// When `sendInitialEvents` option is set, we require `resourceVersionMatch`
324    /// option to also be set. The semantic of the watch request is as following:
325    /// - `resourceVersionMatch` = NotOlderThan
326    ///   is interpreted as "data at least as new as the provided `resourceVersion`"
327    ///   and the bookmark event is send when the state is synced
328    ///   to a `resourceVersion` at least as fresh as the one provided by the ListOptions.
329    ///   If `resourceVersion` is unset, this is interpreted as "consistent read" and the
330    ///   bookmark event is send when the state is synced at least to the moment
331    ///   when request started being processed.
332    /// - `resourceVersionMatch` set to any other value or unset
333    ///   Invalid error is returned.
334    pub send_initial_events: bool,
335}
336
337impl WatchParams {
338    pub(crate) fn validate(&self) -> Result<(), Error> {
339        if let Some(to) = &self.timeout {
340            // https://github.com/kubernetes/kubernetes/issues/6513
341            if *to >= 295 {
342                return Err(Error::Validation("WatchParams::timeout must be < 295s".into()));
343            }
344        }
345        if self.send_initial_events && !self.bookmarks {
346            return Err(Error::Validation(
347                "WatchParams::bookmarks must be set when using send_initial_events".into(),
348            ));
349        }
350        Ok(())
351    }
352
353    // Partially populate query parameters (needs resourceVersion out of band)
354    pub(crate) fn populate_qp(&self, qp: &mut form_urlencoded::Serializer<String>) {
355        qp.append_pair("watch", "true");
356
357        // https://github.com/kubernetes/kubernetes/issues/6513
358        qp.append_pair("timeoutSeconds", &self.timeout.unwrap_or(290).to_string());
359
360        if let Some(fields) = &self.field_selector {
361            qp.append_pair("fieldSelector", fields);
362        }
363        if let Some(labels) = &self.label_selector {
364            qp.append_pair("labelSelector", labels);
365        }
366        if self.bookmarks {
367            qp.append_pair("allowWatchBookmarks", "true");
368        }
369        if self.send_initial_events {
370            qp.append_pair("sendInitialEvents", "true");
371            qp.append_pair("resourceVersionMatch", "NotOlderThan");
372        }
373    }
374}
375
376impl Default for WatchParams {
377    /// Default `WatchParams` without any constricting selectors
378    fn default() -> Self {
379        Self {
380            // bookmarks stable since 1.17, and backwards compatible
381            bookmarks: true,
382
383            label_selector: None,
384            field_selector: None,
385            timeout: None,
386            send_initial_events: false,
387        }
388    }
389}

485/// Common query parameters for put/post calls
486#[derive(Default, Clone, Debug, PartialEq)]
487pub struct PostParams {
488    /// Whether to run this as a dry run
489    pub dry_run: bool,
490    /// fieldManager is a name of the actor that is making changes
491    pub field_manager: Option<String>,
492}
493
494impl PostParams {
495    pub(crate) fn populate_qp(&self, qp: &mut form_urlencoded::Serializer<String>) {
496        if self.dry_run {
497            qp.append_pair("dryRun", "All");
498        }
499        if let Some(ref fm) = self.field_manager {
500            qp.append_pair("fieldManager", fm);
501        }
502    }
503
504    pub(crate) fn validate(&self) -> Result<(), Error> {
505        if let Some(field_manager) = &self.field_manager {
506            // Implement the easy part of validation, in future this may be extended to provide validation as in go code
507            // For now it's fine, because k8s API server will return an error
508            if field_manager.len() > 128 {
509                return Err(Error::Validation(
510                    "Failed to validate PostParams::field_manager!".into(),
511                ));
512            }
513        }
514        Ok(())
515    }

518/// Describes changes that should be applied to a resource
519///
520/// Takes arbitrary serializable data for all strategies except `Json`.
521///
522/// We recommend using ([server-side](https://kubernetes.io/blog/2020/04/01/kubernetes-1.18-feature-server-side-apply-beta-2)) `Apply` patches on new kubernetes releases.
523///
524/// See [kubernetes patch docs](https://kubernetes.io/docs/tasks/run-application/update-api-object-kubectl-patch/#use-a-json-merge-patch-to-update-a-deployment) for the older patch types.
525///
526/// Note that patches have different effects on different fields depending on their merge strategies.
527/// These strategies are configurable when deriving your [`CustomResource`](https://docs.rs/kube-derive/*/kube_derive/derive.CustomResource.html#customizing-schemas).
528///
529/// # Creating a patch via serde_json
530/// ```
531/// use kube::api::Patch;
532/// let patch = serde_json::json!({
533///     "apiVersion": "v1",
534///     "kind": "Pod",
535///     "metadata": {
536///         "name": "blog"
537///     },
538///     "spec": {
539///         "activeDeadlineSeconds": 5
540///     }
541/// });
542/// let patch = Patch::Apply(&patch);
543/// ```
544/// # Creating a patch from a type
545/// ```
546/// use kube::api::Patch;
547/// use k8s_openapi::api::rbac::v1::Role;
548/// use k8s_openapi::apimachinery::pkg::apis::meta::v1::ObjectMeta;
549/// let r = Role {
550///     metadata: ObjectMeta { name: Some("user".into()), ..ObjectMeta::default() },
551///     rules: Some(vec![])
552/// };
553/// let patch = Patch::Apply(&r);
554/// ```
555#[non_exhaustive]
556#[derive(Debug, PartialEq, Clone)]
557pub enum Patch<T: Serialize> {
558    /// [Server side apply](https://kubernetes.io/docs/reference/using-api/api-concepts/#server-side-apply)
559    ///
560    /// Requires kubernetes >= 1.16
561    Apply(T),
562
563    /// [JSON patch](https://kubernetes.io/docs/tasks/run-application/update-api-object-kubectl-patch/#use-a-json-merge-patch-to-update-a-deployment)
564    ///
565    /// Using this variant will require you to explicitly provide a type for `T` at the moment.
566    ///
567    /// # Example
568    ///
569    /// ```
570    /// use kube::api::Patch;
571    /// let json_patch = json_patch::Patch(vec![]);
572    /// let patch = Patch::Json::<()>(json_patch);
573    /// ```
574    #[cfg(feature = "jsonpatch")]
575    #[cfg_attr(docsrs, doc(cfg(feature = "jsonpatch")))]
576    Json(json_patch::Patch),
577
578    /// [JSON Merge patch](https://kubernetes.io/docs/tasks/run-application/update-api-object-kubectl-patch/#use-a-json-merge-patch-to-update-a-deployment)
579    Merge(T),
580    /// [Strategic JSON Merge patch](https://kubernetes.io/docs/tasks/run-application/update-api-object-kubectl-patch/#use-a-strategic-merge-patch-to-update-a-deployment)
581    Strategic(T),
582}
583
584impl<T: Serialize> Patch<T> {
585    pub(crate) fn is_apply(&self) -> bool {
586        matches!(self, Patch::Apply(_))
587    }
588
589    pub(crate) fn content_type(&self) -> &'static str {
590        match &self {
591            Self::Apply(_) => "application/apply-patch+yaml",
592            #[cfg(feature = "jsonpatch")]
593            #[cfg_attr(docsrs, doc(cfg(feature = "jsonpatch")))]
594            Self::Json(_) => "application/json-patch+json",
595            Self::Merge(_) => "application/merge-patch+json",
596            Self::Strategic(_) => "application/strategic-merge-patch+json",
597        }
598    }
599}
600
601impl<T: Serialize> Patch<T> {
602    pub(crate) fn serialize(&self) -> Result<Vec<u8>, serde_json::Error> {
603        match self {
604            Self::Apply(p) => serde_json::to_vec(p),
605            #[cfg(feature = "jsonpatch")]
606            #[cfg_attr(docsrs, doc(cfg(feature = "jsonpatch")))]
607            Self::Json(p) => serde_json::to_vec(p),
608            Self::Strategic(p) => serde_json::to_vec(p),
609            Self::Merge(p) => serde_json::to_vec(p),
610        }
611    }
612}

614/// Common query parameters for patch calls

628impl PatchParams {
629    pub(crate) fn validate<P: Serialize>(&self, patch: &Patch<P>) -> Result<(), Error> {
630        if let Some(field_manager) = &self.field_manager {
631            // Implement the easy part of validation, in future this may be extended to provide validation as in go code
632            // For now it's fine, because k8s API server will return an error
633            if field_manager.len() > 128 {
634                return Err(Error::Validation(
635                    "Failed to validate PatchParams::field_manager!".into(),
636                ));
637            }
638        }
639        if self.force && !patch.is_apply() {
640            return Err(Error::Validation(
641                "PatchParams::force only works with Patch::Apply".into(),
642            ));
643        }
644        Ok(())
645    }

647    pub(crate) fn populate_qp(&self, qp: &mut form_urlencoded::Serializer<String>) {
648        if self.dry_run {
649            qp.append_pair("dryRun", "All");
650        }
651        if self.force {
652            qp.append_pair("force", "true");
653        }
654        if let Some(ref fm) = self.field_manager {
655            qp.append_pair("fieldManager", fm);
656        }
657        if let Some(sv) = &self.field_validation {
658            qp.append_pair("fieldValidation", sv.as_str());
659        }
660    }

668        }

712/// Common query parameters for delete calls
713#[derive(Default, Clone, Serialize, Debug, PartialEq)]
714#[serde(rename_all = "camelCase")]
715pub struct DeleteParams {
716    /// When present, indicates that modifications should not be persisted.
717    #[serde(
718        serialize_with = "dry_run_all_ser",
719        skip_serializing_if = "std::ops::Not::not"
720    )]
721    pub dry_run: bool,
722
723    /// The duration in seconds before the object should be deleted.
724    ///
725    /// Value must be non-negative integer. The value zero indicates delete immediately.
726    /// If this value is `None`, the default grace period for the specified type will be used.
727    /// Defaults to a per object value if not specified. Zero means delete immediately.
728    #[serde(skip_serializing_if = "Option::is_none")]
729    pub grace_period_seconds: Option<u32>,
730
731    /// Whether or how garbage collection is performed.
732    ///
733    /// The default policy is decided by the existing finalizer set in
734    /// `metadata.finalizers`, and the resource-specific default policy.
735    #[serde(skip_serializing_if = "Option::is_none")]
736    pub propagation_policy: Option<PropagationPolicy>,
737
738    /// Condtions that must be fulfilled before a deletion is carried out
739    ///
740    /// If not possible, a `409 Conflict` status will be returned.
741    #[serde(skip_serializing_if = "Option::is_none")]
742    pub preconditions: Option<Preconditions>,
743}
744
745impl DeleteParams {
746    /// Construct `DeleteParams` with `PropagationPolicy::Background`.
747    ///
748    /// This allows the garbage collector to delete the dependents in the background.
749    pub fn background() -> Self {
750        Self {
751            propagation_policy: Some(PropagationPolicy::Background),
752            ..Self::default()
753        }
754    }
755
756    /// Construct `DeleteParams` with `PropagationPolicy::Foreground`.
757    ///
758    /// This is a cascading policy that deletes all dependents in the foreground.
759    pub fn foreground() -> Self {
760        Self {
761            propagation_policy: Some(PropagationPolicy::Foreground),
762            ..Self::default()
763        }
764    }
765
766    /// Construct `DeleteParams` with `PropagationPolicy::Orphan`.
767    ///
768    ///
769    /// This orpans the dependents.
770    pub fn orphan() -> Self {
771        Self {
772            propagation_policy: Some(PropagationPolicy::Orphan),
773            ..Self::default()
774        }
775    }
776
777    /// Perform a dryRun only
778    #[must_use]
779    pub fn dry_run(mut self) -> Self {
780        self.dry_run = true;
781        self
782    }
783
784    /// Set the duration in seconds before the object should be deleted.
785    #[must_use]
786    pub fn grace_period(mut self, secs: u32) -> Self {
787        self.grace_period_seconds = Some(secs);
788        self
789    }
790
791    /// Set the condtions that must be fulfilled before a deletion is carried out.
792    #[must_use]
793    pub fn preconditions(mut self, preconditions: Preconditions) -> Self {
794        self.preconditions = Some(preconditions);
795        self
796    }
797
798    pub(crate) fn is_default(&self) -> bool {
799        !self.dry_run
800            && self.grace_period_seconds.is_none()
801            && self.propagation_policy.is_none()
802            && self.preconditions.is_none()
803    }

806// dryRun serialization differ when used as body parameters and query strings:
807// query strings are either true/false
808// body params allow only: missing field, or ["All"]
809// The latter is a very awkward API causing users to do to
810// dp.dry_run = vec!["All".into()];
811// just to turn on dry_run..
812// so we hide this detail for now.
813fn dry_run_all_ser<S>(t: &bool, s: S) -> std::result::Result<S::Ok, S::Error>
814where
815    S: serde::ser::Serializer,
816{
817    use serde::ser::SerializeTuple;
818    match t {
819        true => {
820            let mut map = s.serialize_tuple(1)?;
821            map.serialize_element("All")?;
822            map.end()
823        }
824        false => s.serialize_none(),
825    }
826}
