8#[derive(Debug, Error)]
9#[error("failed to parse group version: {0}")]
10/// Failed to parse group version
11pub struct ParseGroupVersionError(pub String);

13/// Core information about an API Resource.
14#[derive(Deserialize, Serialize, Debug, Clone, PartialEq, Eq, Hash)]
15pub struct GroupVersionKind {
16    /// API group
17    pub group: String,
18    /// Version
19    pub version: String,
20    /// Kind
21    pub kind: String,
22}
23
24impl GroupVersionKind {
25    /// Construct from explicit group, version, and kind
26    pub fn gvk(group_: &str, version_: &str, kind_: &str) -> Self {
27        let version = version_.to_string();
28        let group = group_.to_string();
29        let kind = kind_.to_string();
30
31        Self { group, version, kind }
32    }
33}
34
35impl TryFrom<&TypeMeta> for GroupVersionKind {
36    type Error = ParseGroupVersionError;
37
38    fn try_from(tm: &TypeMeta) -> Result<Self, Self::Error> {
39        Ok(GroupVersion::from_str(&tm.api_version)?.with_kind(&tm.kind))
40    }
41}
42impl TryFrom<TypeMeta> for GroupVersionKind {
43    type Error = ParseGroupVersionError;
44

50/// Core information about a family of API Resources
51#[derive(Deserialize, Serialize, Debug, Clone, PartialEq, Eq, Hash)]
52pub struct GroupVersion {
53    /// API group
54    pub group: String,
55    /// Version
56    pub version: String,
57}
58
59impl GroupVersion {
60    /// Construct from explicit group and version
61    pub fn gv(group_: &str, version_: &str) -> Self {
62        let version = version_.to_string();
63        let group = group_.to_string();
64        Self { group, version }
65    }
66
67    /// Upgrade a GroupVersion to a GroupVersionKind
68    pub fn with_kind(self, kind: &str) -> GroupVersionKind {
69        GroupVersionKind {
70            group: self.group,
71            version: self.version,
72            kind: kind.into(),
73        }
74    }
75}
76
77impl FromStr for GroupVersion {
78    type Err = ParseGroupVersionError;
79
80    fn from_str(gv: &str) -> Result<Self, Self::Err> {
81        let gvsplit = gv.splitn(2, '/').collect::<Vec<_>>();
82        let (group, version) = match *gvsplit.as_slice() {
83            [g, v] => (g.to_string(), v.to_string()), // standard case
84            [v] => ("".to_string(), v.to_string()),   // core v1 case
85            _ => return Err(ParseGroupVersionError(gv.into())),
86        };
87        Ok(Self { group, version })
88    }
89}
90
91impl GroupVersion {
92    /// Generate the apiVersion string used in a kind's yaml
93    pub fn api_version(&self) -> String {
94        if self.group.is_empty() {
95            self.version.clone()
96        } else {
97            format!("{}/{}", self.group, self.version)
98        }
99    }
100}
101impl GroupVersionKind {

104        if self.group.is_empty() {
105            self.version.clone()

108        }
109    }

112/// Represents a type-erased object resource.
113#[derive(Deserialize, Serialize, Debug, Clone, PartialEq, Eq, Hash)]
114pub struct GroupVersionResource {
115    /// API group
116    pub group: String,
117    /// Version
118    pub version: String,
119    /// Resource
120    pub resource: String,
121    /// Concatenation of group and version
122    #[serde(default)]
123    api_version: String,
124}
125
126impl GroupVersionResource {
127    /// Set the api group, version, and the plural resource name.
128    pub fn gvr(group_: &str, version_: &str, resource_: &str) -> Self {
129        let version = version_.to_string();
130        let group = group_.to_string();
131        let resource = resource_.to_string();
132        let api_version = if group.is_empty() {
133            version.to_string()
134        } else {
135            format!("{group}/{version}")
136        };
137
138        Self {
139            group,
140            version,
141            resource,
142            api_version,
143        }
144    }
145}

147#[cfg(test)]
148mod tests {
149    #[test]
150    fn gvk_yaml() {
151        use crate::{GroupVersionKind, TypeMeta};
152        let input = r#"---
153apiVersion: kube.rs/v1
154kind: Example
155metadata:
156  name: doc1
157"#;
158        let tm: TypeMeta = serde_yaml::from_str(input).unwrap();
159        let gvk = GroupVersionKind::try_from(&tm).unwrap(); // takes ref
160        let gvk2: GroupVersionKind = tm.try_into().unwrap(); // takes value
161        assert_eq!(gvk.kind, "Example");
162        assert_eq!(gvk.group, "kube.rs");
163        assert_eq!(gvk.version, "v1");
164        assert_eq!(gvk.kind, gvk2.kind);
165    }
