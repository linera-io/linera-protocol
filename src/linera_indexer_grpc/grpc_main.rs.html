4//! An example of a gRPC indexer server with multiple database backend support.
5
6use clap::Parser;
7use linera_indexer::{
8    common::IndexerError,
9    db::{postgres::PostgresDatabase, sqlite::SqliteDatabase},
10    grpc::IndexerGrpcServer,
11};
12
13#[derive(Parser, Debug)]
14#[command(version = linera_version::VersionInfo::default_clap_str())]
15#[command(group = clap::ArgGroup::new("database").required(true).multiple(false))]
16struct Args {
17    /// The port of the gRPC indexer server
18    #[arg(long, default_value = "8081")]
19    port: u16,
20
21    /// Use in-memory SQLite database (data is lost on restart)
22    #[arg(long, group = "database")]
23    memory: bool,
24
25    /// Use SQLite database with file persistence
26    #[arg(long, group = "database", value_name = "PATH")]
27    sqlite: Option<String>,
28
29    /// Use PostgreSQL database
30    #[arg(long, group = "database", value_name = "URL")]
31    postgres: Option<String>,
32}
33
34#[tokio::main]
35async fn main() -> Result<(), IndexerError> {
36    let env_filter = tracing_subscriber::EnvFilter::builder()
37        .with_default_directive(tracing_subscriber::filter::LevelFilter::INFO.into())
38        .from_env_lossy();
39    tracing_subscriber::fmt()
40        .with_writer(std::io::stderr)
41        .with_env_filter(env_filter)
42        .init();
43
44    let args = Args::parse();
45
46    // Start server with the selected database backend
47    if args.memory {
48        tracing::info!("Starting indexer with in-memory SQLite database");
49        let database = SqliteDatabase::new("sqlite::memory:").await?;
50        let grpc_server = IndexerGrpcServer::new(database);
51        grpc_server
52            .serve(args.port)
53            .await
54            .map_err(IndexerError::Other)?;
55    } else if let Some(path) = args.sqlite {
56        tracing::info!(?path, "Starting indexer with SQLite database");
57        let database = SqliteDatabase::new(&path).await?;
58        let grpc_server = IndexerGrpcServer::new(database);
59        grpc_server
60            .serve(args.port)
61            .await
62            .map_err(IndexerError::Other)?;
63    } else if let Some(url) = args.postgres {
64        tracing::info!(?url, "Starting indexer with PostgreSQL database");
65        let database = PostgresDatabase::new(&url).await?;
66        let grpc_server = IndexerGrpcServer::new(database);
67        grpc_server
68            .serve(args.port)
69            .await
70            .map_err(IndexerError::Other)?;
71    } else {
72        return Err(IndexerError::Other(
73            "No database backend specified. Use --memory, --sqlite, or --postgres".into(),
74        ));
75    }
76
77    Ok(())
78}
