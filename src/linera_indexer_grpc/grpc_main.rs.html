4//! An example of a gRPC indexer server with SQLite storage.
5
6use clap::Parser;
7use linera_indexer::{common::IndexerError, db::sqlite::SqliteDatabase, grpc::IndexerGrpcServer};
8
9#[derive(Parser, Debug)]
10#[command(version = linera_version::VersionInfo::default_clap_str())]
11struct Args {
12    /// The port of the gRPC indexer server
13    #[arg(long, default_value = "8081")]
14    port: u16,
15    /// SQLite database file path
16    #[arg(long, default_value = "indexer.db")]
17    database_path: String,
18}
19
20#[tokio::main]
21async fn main() -> Result<(), IndexerError> {
22    let env_filter = tracing_subscriber::EnvFilter::builder()
23        .with_default_directive(tracing_subscriber::filter::LevelFilter::INFO.into())
24        .from_env_lossy();
25    tracing_subscriber::fmt()
26        .with_writer(std::io::stderr)
27        .with_env_filter(env_filter)
28        .init();
29
30    let args = Args::parse();
31
32    // Initialize SQLite database
33    let database = SqliteDatabase::new(args.database_path.as_str()).await?;
34
35    // Create and start gRPC server
36    let grpc_server = IndexerGrpcServer::new(database);
37    grpc_server
38        .serve(args.port)
39        .await
40        .map_err(IndexerError::Other)?;
41
42    Ok(())
43}
