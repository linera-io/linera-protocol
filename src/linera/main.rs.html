104/// Check if an error is retryable (HTTP 502, 503, 504, timeouts, connection errors)
105fn is_retryable_error(err: &anyhow::Error) -> bool {
106    // Check for reqwest errors in the error chain
107    if let Some(reqwest_err) = err.downcast_ref::<reqwest::Error>() {
108        // Check for retryable HTTP status codes (502, 503, 504)
109        if let Some(status) = reqwest_err.status() {
110            return status == reqwest::StatusCode::BAD_GATEWAY
111                || status == reqwest::StatusCode::SERVICE_UNAVAILABLE
112                || status == reqwest::StatusCode::GATEWAY_TIMEOUT;
113        }
114        // Check for connection errors or timeouts
115        return reqwest_err.is_timeout() || reqwest_err.is_connect();
116    }
117    false
118}

120/// Retry a faucet operation with exponential backoff
121async fn retry_faucet_operation<F, Fut, T>(operation: F) -> anyhow::Result<T>
122where
123    F: Fn() -> Fut,
124    Fut: std::future::Future<Output = anyhow::Result<T>>,
125{
126    let max_retries = 5;
127    let mut attempt = 0;
128
129    loop {
130        attempt += 1;
131        match operation().await {
132            Ok(result) => return Ok(result),
133            Err(err) if attempt < max_retries && is_retryable_error(&err) => {
134                let backoff_ms = 100 * 2_u64.pow(attempt - 1);
135                warn!(
136                    "Faucet operation failed with retryable error (attempt {}/{}): {:?}. Retrying after {}ms",
137                    attempt, max_retries, err, backoff_ms
138                );
139                tokio::time::sleep(Duration::from_millis(backoff_ms)).await;
140            }
141            Err(err) => return Err(err),
142        }
143    }
144}

146fn read_json(string: Option<String>, path: Option<PathBuf>) -> anyhow::Result<Vec<u8>> {
147    let value = match (string, path) {
148        (Some(_), Some(_)) => bail!("cannot have both a json string and file"),
149        (Some(s), None) => serde_json::from_str(&s)?,
150        (None, Some(path)) => {
151            let s = fs_err::read_to_string(path)?;
152            serde_json::from_str(&s)?
153        }
154        (None, None) => Value::Null,
155    };
156    Ok(serde_json::to_vec(&value)?)
157}

1916) -> anyhow::Result<Option<linera_service::tracing::chrome::ChromeTraceGuard>> {
1917    if options.chrome_trace_exporter {
1918        let trace_file_path = options.chrome_trace_file.as_deref().map_or_else(
1919            || format!("{}.trace.json", options.command.log_file_name()),
1920            |s| s.to_string(),
1921        );
1922        let writer = std::fs::File::create(&trace_file_path)?;
1923        Ok(Some(linera_service::tracing::chrome::init(
1924            &options.command.log_file_name(),
1925            writer,
1926        )))
1927    } else {
1928        // Always use opentelemetry::init to ensure baggage propagation is set up.
1929        // This enables traffic_type labeling (organic vs synthetic) to work regardless
1930        // of whether traces are being exported to OTLP.
1931        linera_service::tracing::opentelemetry::init(
1932            &options.command.log_file_name(),
1933            options.otlp_exporter_endpoint.as_deref(),
1934        );
1935        Ok(None)
1936    }
1937}

1944fn main() -> anyhow::Result<process::ExitCode> {
1945    let options = Options::init();
1946    let mut runtime = if options.tokio_threads == Some(1) {
1947        tokio::runtime::Builder::new_current_thread()
1948    } else {
1949        let mut builder = tokio::runtime::Builder::new_multi_thread();
1950
1951        if let Some(threads) = options.tokio_threads {
1952            builder.worker_threads(threads);
1953        }
1954
1955        builder
1956    };
1957
1958    // The default stack size 2 MiB causes some stack overflows in ValidatorUpdater methods.
1959    runtime.thread_stack_size(4 << 20);
1960    if let Some(blocking_threads) = options.tokio_blocking_threads {
1961        runtime.max_blocking_threads(blocking_threads);
1962    }
1963
1964    let span = tracing::info_span!("linera::main");
1965    if let Some(wallet_id) = &options.with_wallet {
1966        span.record("wallet_id", wallet_id);
1967    }
1968
1969    let result = runtime
1970        .enable_all()
1971        .build()?
1972        .block_on(run(&options).instrument(span));
1973
1974    Ok(match result {
1975        Ok(0) => process::ExitCode::SUCCESS,
1976        Ok(code) => process::ExitCode::from(code as u8),
1977        Err(msg) => {
1978            error!("Error is {:?}", msg);
1979            process::ExitCode::FAILURE
1980        }
1981    })
1982}
