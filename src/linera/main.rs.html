101fn read_json(string: Option<String>, path: Option<PathBuf>) -> anyhow::Result<Vec<u8>> {
102    let value = match (string, path) {
103        (Some(_), Some(_)) => bail!("cannot have both a json string and file"),
104        (Some(s), None) => serde_json::from_str(&s)?,
105        (None, Some(path)) => {
106            let s = fs_err::read_to_string(path)?;
107            serde_json::from_str(&s)?
108        }
109        (None, None) => Value::Null,
110    };
111    Ok(serde_json::to_vec(&value)?)
112}

599                    .map(|(_validator_name, network_address)| {
600                        let context = context.clone();
601                        let chains = chains.clone();
602                        async move {
603                            let validator =

606                            for chain_id in &chains {
607                                let chain = context.make_chain_client(*chain_id);
608                                Box::pin(chain.sync_validator(validator.clone())).await?;
609                            }
610                            anyhow::Result::<()>::Ok(())
611                        }
612                    })
613                    .collect::<Vec<_>>();
614
615                // Wait for all validator sync tasks to complete
616                futures::future::try_join_all(tasks).await?;
617
618                let time_total = time_start.elapsed();
