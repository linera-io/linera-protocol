24/// A message.
25#[derive(Debug, Deserialize, Serialize)]
26pub enum Message {
27    /// Credits the given `target` account, unless the message is bouncing, in which case
28    /// `source` is credited instead.
29    Credit {
30        /// Target account to credit amount to
31        target: AccountOwner,
32        /// Amount to be credited
33        amount: Amount,
34        /// Source account to remove amount from
35        source: AccountOwner,
36    },
37
38    /// Withdraws from the given account and starts a transfer to the target account.
39    Withdraw {
40        /// Account to withdraw from
41        owner: AccountOwner,
42        /// Amount to be withdrawn
43        amount: Amount,
44        /// Target account to transfer amount to
45        target_account: Account,
46    },
47}

49#[derive(Clone, Debug, Deserialize, Serialize)]
50pub struct OwnerSpender {
51    /// Account to withdraw from
52    pub owner: AccountOwner,
53    /// Account to do the withdrawing
54    pub spender: AccountOwner,
55}
56
57scalar!(OwnerSpender);
58
59impl OwnerSpender {
60    pub fn new(owner: AccountOwner, spender: AccountOwner) -> Self {
61        if owner == spender {
62            panic!("owner should be different from spender");
63        }
64        Self { owner, spender }
65    }
66}

81/// An operation
82#[derive(Debug, Deserialize, Serialize, GraphQLMutationRoot)]
83pub enum Operation {
84    /// Requests an account balance.
85    Balance {
86        /// Owner to query the balance for
87        owner: AccountOwner,
88    },
89    /// Requests this fungible token's ticker symbol.
90    TickerSymbol,
91    /// Approve the transfer of tokens
92    Approve {
93        /// Owner to transfer from
94        owner: AccountOwner,
95        /// The spender account
96        spender: AccountOwner,
97        /// Maximum amount to be transferred
98        allowance: Amount,
99    },
100    /// Transfers tokens from a (locally owned) account to a (possibly remote) account.
101    Transfer {
102        /// Owner to transfer from
103        owner: AccountOwner,
104        /// Amount to be transferred
105        amount: Amount,
106        /// Target account to transfer the amount to
107        target_account: Account,
108    },
109    /// Transfers tokens from a (locally owned) account to a (possibly remote) account by using the allowance.
110    TransferFrom {
111        /// Owner to transfer from
112        owner: AccountOwner,
113        /// The spender of the amount.
114        spender: AccountOwner,
115        /// Amount to be transferred
116        amount: Amount,
117        /// Target account to transfer the amount to
118        target_account: Account,
119    },
120    /// Same as `Transfer` but the source account may be remote. Depending on its
121    /// configuration, the target chain may take time or refuse to process
122    /// the message.
123    Claim {
124        /// Source account to claim amount from
125        source_account: Account,
126        /// Amount to be claimed
127        amount: Amount,
128        /// Target account to claim the amount into
129        target_account: Account,
130    },
131}

133/// Creates a fungible token application and distributes `initial_amounts` to new individual
134/// chains.
135#[cfg(all(any(test, feature = "test"), not(target_arch = "wasm32")))]
136pub async fn create_with_accounts(
137    validator: &TestValidator,
138    module_id: ModuleId<FungibleTokenAbi, Parameters, InitialState>,
139    initial_amounts: impl IntoIterator<Item = Amount>,
140) -> (
141    ApplicationId<FungibleTokenAbi>,
142    Vec<(ActiveChain, AccountOwner, Amount)>,
143) {
144    let mut token_chain = validator.new_chain().await;
145    let mut initial_state = InitialStateBuilder::default();
146
147    let accounts = stream::iter(initial_amounts)
148        .then(|initial_amount| async move {
149            let chain = validator.new_chain().await;
150            let account = AccountOwner::from(chain.public_key());
151
152            (chain, account, initial_amount)
153        })
154        .collect::<Vec<_>>()
155        .await;
156
157    for (_chain, account, initial_amount) in &accounts {
158        initial_state = initial_state.with_account(*account, *initial_amount);
159    }
160
161    let params = Parameters::new("FUN");
162    let application_id = token_chain
163        .create_application(module_id, params, initial_state.build(), vec![])
164        .await;
165
166    for (chain, account, initial_amount) in &accounts {
167        let claim_certificate = chain
168            .add_block(|block| {
169                block.with_operation(
170                    application_id,
171                    Operation::Claim {
172                        source_account: Account {
173                            chain_id: token_chain.id(),
174                            owner: *account,
175                        },
176                        amount: *initial_amount,
177                        target_account: Account {
178                            chain_id: chain.id(),
179                            owner: *account,
180                        },
181                    },
182                );
183            })
184            .await;
185
186        assert_eq!(claim_certificate.outgoing_message_count(), 1);
187
188        let transfer_certificate = token_chain
189            .add_block(|block| {
190                block.with_messages_from(&claim_certificate);
191            })
192            .await;
193
194        assert_eq!(transfer_certificate.outgoing_message_count(), 1);
195
196        chain
197            .add_block(|block| {
198                block.with_messages_from(&transfer_certificate);
199            })
200            .await;
201    }
202
203    (application_id, accounts)
204}

206/// Queries the balance of an account owned by `account_owner` on a specific `chain`.
207#[cfg(all(any(test, feature = "test"), not(target_arch = "wasm32")))]
208pub async fn query_account(
209    application_id: ApplicationId<FungibleTokenAbi>,
210    chain: &ActiveChain,
211    account_owner: AccountOwner,
212) -> Option<Amount> {
213    let query = format!(
214        "query {{ accounts {{ entry(key: {}) {{ value }} }} }}",
215        account_owner.to_value()
216    );
217    let QueryOutcome { response, .. } = chain.graphql_query(application_id, query).await;
218    let balance = response.pointer("/accounts/entry/value")?.as_str()?;
219
220    Some(
221        balance
222            .parse()
223            .expect("Account balance cannot be parsed as a number"),
224    )
225}
