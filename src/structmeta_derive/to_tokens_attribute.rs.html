1use proc_macro2::Span;
2use syn::{parse::Parse, spanned::Spanned, LitStr, Result, Token};
3
4pub struct ToTokensAttribute {
5    pub dump: Option<Span>,
6    pub token: Vec<LitStr>,
7}
8impl Parse for ToTokensAttribute {
9    fn parse(input: syn::parse::ParseStream) -> Result<Self> {
10        let args = input.parse_terminated(ToTokensAttributeArg::parse, Token![,])?;
11        let mut token = Vec::new();
12        let mut dump = None;
13        for arg in args.into_iter() {
14            match arg {
15                ToTokensAttributeArg::Token(token_value) => {
16                    token.push(token_value);
17                }
18                ToTokensAttributeArg::Dump(kw_dump) => {
19                    if dump.is_none() {
20                        dump = Some(kw_dump.span());
21                    }
22                }
23            }
24        }
25        Ok(Self { dump, token })
26    }
27}

29mod kw {
30    use syn::custom_keyword;
31    custom_keyword!(dump);
32}
33
34enum ToTokensAttributeArg {
35    Token(LitStr),
36    Dump(kw::dump),
37}
38impl Parse for ToTokensAttributeArg {
39    fn parse(input: syn::parse::ParseStream) -> Result<Self> {
40        if input.peek(LitStr) {
41            Ok(Self::Token(input.parse()?))
42        } else if input.peek(kw::dump) {
43            Ok(Self::Dump(input.parse()?))
44        } else {
45            Err(input.error("expected string literal."))
46        }
47    }
48}

50pub fn to_close(c: char) -> char {
51    match c {
52        '(' => ')',
53        '[' => ']',
54        '{' => '}',
55        _ => panic!("not found closing delimiter for {c}"),
56    }
