17#[derive(Error, Debug)]
18pub enum IndexerError {
19    #[error(transparent)]
20    ViewError(#[from] linera_views::ViewError),
21    #[error(transparent)]
22    ReqwestError(#[from] reqwest::Error),
23    #[error(transparent)]
24    GraphQLError(#[from] graphql_ws_client::Error),
25    #[error(transparent)]
26    TungsteniteError(#[from] Box<async_tungstenite::tungstenite::Error>),
27    #[error(transparent)]
28    InvalidHeader(#[from] InvalidHeaderValue),
29    #[error(transparent)]
30    IoError(#[from] std::io::Error),
31    #[error(transparent)]
32    ParserError(#[from] AddrParseError),
33    #[error("Null GraphQL data: {0:?}")]
34    NullData(Option<Vec<graphql_client::Error>>),
35    #[error("Block not found: {0:?}")]
36    NotFound(Option<CryptoHash>),
37    #[error("Unknown plugin: {0}")]
38    UnknownPlugin(String),
39    #[error(transparent)]
40    ConversionError(linera_service_graphql_client::ConversionError),
41    #[error("Plugin is already registered")]
42    PluginAlreadyRegistered,
43    #[error("Open exclusive error")]
44    OpenExclusiveError,
45    #[error("Other error: {0}")]
46    Other(#[from] Box<dyn std::error::Error + Send + Sync>),
47
48    #[cfg(feature = "rocksdb")]
49    #[error(transparent)]
50    RocksDbError(#[from] linera_views::rocks_db::RocksDbStoreError),
51    #[cfg(feature = "scylladb")]
52    #[error(transparent)]
53    ScyllaDbError(#[from] Box<linera_views::scylla_db::ScyllaDbStoreError>),
54}
55
56impl From<SqliteError> for IndexerError {
57    fn from(error: SqliteError) -> Self {
58        Self::Other(Box::new(error).into())
59    }
60}
61
62impl From<crate::db::postgres::PostgresError> for IndexerError {
63    fn from(error: crate::db::postgres::PostgresError) -> Self {
64        Self::Other(Box::new(error).into())
65    }
66}
67
68impl From<async_tungstenite::tungstenite::Error> for IndexerError {
69    fn from(error: async_tungstenite::tungstenite::Error) -> Self {
70        Box::new(error).into()
71    }
72}
73
74#[cfg(feature = "scylladb")]
75impl From<linera_views::scylla_db::ScyllaDbStoreError> for IndexerError {
76    fn from(error: linera_views::scylla_db::ScyllaDbStoreError) -> Self {
77        Box::new(error).into()
78    }
79}
