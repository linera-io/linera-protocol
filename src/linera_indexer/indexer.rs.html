34#[derive(RootView, Allocative)]

46pub struct Indexer<D>
47where
48    D: KeyValueDatabase,
49{
50    pub state: State<ViewContext<(), D::Store>>,
51    pub plugins: BTreeMap<String, Box<dyn Plugin<D>>>,
52}
53
54#[derive(Debug)]
55enum LatestBlock {
56    LatestHash(CryptoHash),
57    StartHeight(BlockHeight),
58}
59
60impl<D> Indexer<D>
61where
62    D: KeyValueDatabase + Clone + Send + Sync + 'static,
63    D::Store: KeyValueStore + Clone + Send + Sync + 'static,
64    D::Error: Send + Sync + std::error::Error + 'static,
65{
66    /// Loads the indexer using a database backend with an `indexer` prefix.
67    pub async fn load(database: D) -> Result<Self, IndexerError> {
68        let state = State(
69            plugin::load::<D, StateView<ViewContext<(), D::Store>>>(database, "indexer").await?,
70        );
71        Ok(Indexer {
72            state,
73            plugins: BTreeMap::new(),
74        })
75    }
76
77    /// Processes one block: registers the block in all the plugins and saves the state of
78    /// the indexer.
79    pub async fn process_value(
80        &self,
81        state: &mut StateView<ViewContext<(), D::Store>>,
82        value: &ConfirmedBlock,
83    ) -> Result<(), IndexerError> {
84        for plugin in self.plugins.values() {
85            plugin.register(value).await?
86        }
87        let chain_id = value.chain_id();
88        let hash = value.hash();
89        let height = value.height();
90        info!("save {:?}: {:?} ({})", chain_id, hash, height);
91        state.chains.insert(&chain_id, (hash, height))?;
92        state.save().await.map_err(IndexerError::ViewError)
93    }
94
95    /// Processes a `NewBlock` notification: processes all blocks from the latest
96    /// registered to the one in the notification in the corresponding chain.
97    pub async fn process(
98        &self,
99        listener: &Listener,
100        value: &ConfirmedBlock,
101    ) -> Result<(), IndexerError> {
102        let chain_id = value.chain_id();
103        let hash = value.hash();
104        let height = value.height();
105        if height < listener.start {
106            return Ok(());
107        }
108        // Hold the lock only long enough to read the current chain state.
109        let latest_block = {
110            let state = self.state.0.lock().await;
111            match state.chains.get(&chain_id).await? {
112                None => LatestBlock::StartHeight(listener.start),
113                Some((last_hash, last_height)) => {
114                    if last_hash == hash || last_height >= height {
115                        return Ok(());
116                    }
117                    LatestBlock::LatestHash(last_hash)
118                }
119            }
120        };
121        info!("process {:?}: {:?} ({})", chain_id, hash, height);
122
123        // Collect all missing blocks via network I/O without holding the lock.
124        let mut values = Vec::new();
125        let mut value = value.clone();
126        loop {
127            let header = &value.block().header;
128            values.push(value.clone());
129            let next_hash = header
130                .previous_block_hash
131                .filter(|&hash| match &latest_block {
132                    LatestBlock::LatestHash(latest_hash) => *latest_hash != hash,
133                    LatestBlock::StartHeight(start) => header.height > *start,
134                });
135            if let Some(hash) = next_hash {
136                value = listener.service.get_value(chain_id, Some(hash)).await?;
137            } else {
138                break;
139            }
140        }
141
142        // Re-acquire the lock only for writing.
143        let mut state = self.state.0.lock().await;
144        values.reverse();
145        for value in &values {
146            self.process_value(&mut state, value).await?
147        }
148        Ok(())
149    }
150
151    pub async fn init(&self, listener: &Listener, chain_id: ChainId) -> Result<(), IndexerError> {
152        match listener.service.get_value(chain_id, None).await {
153            Err(IndexerError::NotFound(_)) => Ok(()),
154            result => self.process(listener, &result?).await,
155        }
156    }
157
158    /// Produces the GraphQL schema for the indexer or for a certain plugin
159    pub fn sdl(&self, plugin: Option<String>) -> Result<String, IndexerError> {
160        if let Some(plugin_name) = plugin {
161            self.plugins
162                .get(&plugin_name)
163                .map(|plugin| plugin.sdl())
164                .ok_or(IndexerError::UnknownPlugin(plugin_name))
165        } else {
166            Ok(self.state.clone().schema().sdl())
167        }
168    }
169
170    /// Registers a new plugin in the indexer
171    pub async fn add_plugin(
172        &mut self,
173        plugin: impl Plugin<D> + 'static,
174    ) -> Result<(), IndexerError> {
175        let name = plugin.name();
176        match self.plugins.entry(name.clone()) {
177            btree_map::Entry::Occupied(_) => return Err(IndexerError::PluginAlreadyRegistered),
178            btree_map::Entry::Vacant(entry) => {
179                entry.insert(Box::new(plugin));
180            }
181        }
182        let mut state = self.state.0.lock().await;
183        state.plugins.insert(&name)?;
184        Ok(())
185    }
186
187    /// Handles queries made to the root of the indexer
188    async fn handler(
189        schema: Extension<StateSchema<D::Store>>,
190        req: GraphQLRequest,
191    ) -> GraphQLResponse {
192        schema.execute(req.into_inner()).await.into()
193    }
194
195    /// Registers the handler to an Axum router
196    pub fn route(&self, app: Option<Router>) -> Router {
197        let app = app.unwrap_or_default();
198        app.route("/", get(graphiql).post(Self::handler))
199            .layer(Extension(self.state.clone().schema()))
200            .layer(CorsLayer::permissive())
201    }
202}

204#[derive(SimpleObject)]
205pub struct HighestBlock {
206    chain: ChainId,
207    block: Option<CryptoHash>,
208    height: Option<BlockHeight>,
209}
210
211#[async_graphql::Object(cache_control(no_cache))]
212impl<C> State<C>
213where
214    C: Context + Clone + Send + Sync + 'static,
215{
216    /// Gets the plugins registered in the indexer
217    pub async fn plugins(&self) -> Result<Vec<String>, IndexerError> {
218        let state = self.0.lock().await;
219        Ok(state.plugins.indices().await?)
220    }
221
222    /// Gets the latest blocks registered for each chain handled by the indexer
223    pub async fn state(&self) -> Result<Vec<HighestBlock>, IndexerError> {
224        let state = self.0.lock().await;
225        let chains = state.chains.indices().await?;
226        let mut result = Vec::new();
227        for chain in chains {
228            let (block, height) = match state.chains.get(&chain).await? {
229                Some((hash, h)) => (Some(hash), Some(h)),
230                None => (None, None),
231            };
232            result.push(HighestBlock {
233                chain,
234                block,
235                height,
236            });
237        }
238        Ok(result)
239    }
240}
