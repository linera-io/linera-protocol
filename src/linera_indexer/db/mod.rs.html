21/// Trait defining the database operations for the indexer
22#[async_trait]
23pub trait IndexerDatabase: Send + Sync {
24    type Error;
25
26    type Transaction<'a>: Send + Sync;
27
28    /// Atomically store a block with its required blobs
29    /// This is the high-level API that can be implemented in terms of the other methods
30    async fn store_block_with_blobs(
31        &self,
32        block_hash: &CryptoHash,
33        chain_id: &ChainId,
34        height: BlockHeight,
35        timestamp: Timestamp,
36        block_data: &[u8],
37        blobs: &[(BlobId, Vec<u8>)],
38    ) -> Result<(), Self::Error> {
39        // Start atomic transaction
40        let mut tx = self.begin_transaction().await?;
41
42        // Insert all blobs first
43        for (blob_id, blob_data) in blobs {
44            self.insert_blob_tx(&mut tx, blob_id, blob_data).await?;
45        }
46
47        // Insert the block
48        self.insert_block_tx(&mut tx, block_hash, chain_id, height, timestamp, block_data)
49            .await?;
50
51        // Commit transaction - this is the only point where data becomes visible
52        self.commit_transaction(tx).await
53    }
54
55    /// Starts a new transaction.
56    async fn begin_transaction(&self) -> Result<Self::Transaction<'_>, Self::Error>;
57
58    /// Inserts a blob within a transaction.
59    async fn insert_blob_tx(
60        &self,
61        tx: &mut Self::Transaction<'_>,
62        blob_id: &BlobId,
63        data: &[u8],
64    ) -> Result<(), Self::Error>;
65
66    /// Inserts a block within a transaction.
67    async fn insert_block_tx(
68        &self,
69        tx: &mut Self::Transaction<'_>,
70        hash: &CryptoHash,
71        chain_id: &ChainId,
72        height: BlockHeight,
73        timestamp: Timestamp,
74        data: &[u8],
75    ) -> Result<(), Self::Error>;
76
77    /// Commits a transaction.
78    async fn commit_transaction(&self, tx: Self::Transaction<'_>) -> Result<(), Self::Error>;
79
80    /// Gets a block by hash.
81    async fn get_block(&self, hash: &CryptoHash) -> Result<Vec<u8>, Self::Error>;
82
83    /// Gets a blob by blob ID.
84    async fn get_blob(&self, blob_id: &BlobId) -> Result<Vec<u8>, Self::Error>;
85
86    /// Gets the latest block for a chain.
87    async fn get_latest_block_for_chain(
88        &self,
89        chain_id: &ChainId,
90    ) -> Result<Option<(CryptoHash, BlockHeight, Vec<u8>)>, Self::Error>;
91
92    /// Gets blocks for a chain within a height range.
93    async fn get_blocks_for_chain_range(
94        &self,
95        chain_id: &ChainId,
96        start_height: BlockHeight,
97        end_height: BlockHeight,
98    ) -> Result<Vec<(CryptoHash, BlockHeight, Vec<u8>)>, Self::Error>;
99
100    /// Checks if a blob exists.
101    async fn blob_exists(&self, blob_id: &BlobId) -> Result<bool, Self::Error>;
102
103    /// Checks if a block exists.
104    async fn block_exists(&self, hash: &CryptoHash) -> Result<bool, Self::Error>;
105
106    /// Gets incoming bundles for a specific block.
107    async fn get_incoming_bundles_for_block(
108        &self,
109        block_hash: &CryptoHash,
110    ) -> Result<Vec<(i64, IncomingBundleInfo)>, Self::Error>;
111
112    /// Gets posted messages for a specific bundle.
113    async fn get_posted_messages_for_bundle(
114        &self,
115        bundle_id: i64,
116    ) -> Result<Vec<PostedMessageInfo>, Self::Error>;
117
118    /// Gets all bundles from a specific origin chain.
119    async fn get_bundles_from_origin_chain(
120        &self,
121        origin_chain_id: &ChainId,
122    ) -> Result<Vec<(CryptoHash, i64, IncomingBundleInfo)>, Self::Error>;
123}

125/// Information about an incoming bundle (denormalized for queries)
126#[derive(Debug, Clone)]
127pub struct IncomingBundleInfo {
128    pub bundle_index: usize,
129    pub origin_chain_id: ChainId,
130    pub action: MessageAction,
131    pub source_height: BlockHeight,
132    pub source_timestamp: Timestamp,
133    pub source_cert_hash: CryptoHash,
134    pub transaction_index: u32,
135}
136
137/// Information about a posted message (with serialized complex fields)
138#[derive(Debug, Clone)]
139pub struct PostedMessageInfo {
140    pub message_index: u32,
141    pub authenticated_owner_data: Option<String>,
142    pub grant_amount: String,
143    pub refund_grant_to_data: Option<String>,
144    pub message_kind: String,
145    pub message_data: Vec<u8>,
146}
