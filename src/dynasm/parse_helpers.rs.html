9pub trait ParseOpt: Sized {
10    fn parse(input: parse::ParseStream) -> parse::Result<Option<Self>>;
11}
12
13pub trait ParseOptExt {
14    /// Parses a syntax tree node of type `T`, advancing the position of our
15    /// parse stream past it if it was found.
16    fn parse_opt<T: ParseOpt>(&self) -> parse::Result<Option<T>>;
17}
18
19impl<'a> ParseOptExt for parse::ParseBuffer<'a> {
20    fn parse_opt<T: ParseOpt>(&self) -> parse::Result<Option<T>> {
21        T::parse(self)
22    }
23}

25/// Tries to parse an ident that has a specific name as a keyword. Returns true if it worked.
26pub fn eat_pseudo_keyword(input: parse::ParseStream, kw: &str) -> bool {
27    input.step(|cursor| {
28        if let Some((ident, rest)) = cursor.ident() {
29            if ident == kw {
30                return Ok(((), rest));
31            }
32        }
33        Err(cursor.error("expected identifier"))
34    }).is_ok()
35}

37/// parses an ident, but instead of syn's Parse impl it does also parse keywords as idents
38pub fn parse_ident_or_rust_keyword(input: parse::ParseStream) -> parse::Result<syn::Ident> {
39    input.step(|cursor| {
40        if let Some((ident, rest)) = cursor.ident() {
41            return Ok((ident, rest));
42        }
43        Err(cursor.error("expected identifier"))
44    })
45}

47/// checks if an expression is simply an ident, and if so, returns a clone of it.
48pub fn as_ident(expr: &syn::Expr) -> Option<&syn::Ident> {
49    let path = match *expr {
50        syn::Expr::Path(syn::ExprPath {ref path, qself: None, ..}) => path,
51        _ => return None
52    };
53
54    if path.leading_colon.is_some() || path.segments.len() != 1 {
55        return None;
56    }
57
58    let segment = &path.segments[0];
59    if segment.arguments != syn::PathArguments::None {
60        return None;
61    }
62
63    Some(&segment.ident)
64}

66/// checks if an expression is a simple literal, allowing us to perform compile-time analysis of an expression
67pub fn as_lit(expr: &syn::Expr) -> Option<&syn::Lit> {
68    // strip any wrapping Group nodes due to delimiting
69    let mut inner = expr;
70    while let syn::Expr::Group(syn::ExprGroup { expr, .. }) = inner {
71        inner = expr;
72    }
73
74    match inner {
75        syn::Expr::Lit(syn::ExprLit { ref lit, .. } ) => Some(lit),
76        _ => None
77    }
78}

80/// checks if an expression is a literal with possible negation
81pub fn as_lit_with_negation(expr: &syn::Expr) -> Option<(&syn::Lit, bool)> {
82    // strip any wrapping Group nodes due to delimiting
83    let mut inner = expr;
84    while let syn::Expr::Group(syn::ExprGroup { expr, .. }) = inner {
85        inner = expr;
86    }
87
88    match inner {
89        syn::Expr::Lit(syn::ExprLit { ref lit, .. } ) => Some((lit, false)),
90        syn::Expr::Unary(syn::ExprUnary { op: syn::UnOp::Neg(_), ref expr, .. } ) => {
91            match &**expr {
92                syn::Expr::Lit(syn::ExprLit { ref lit, .. } ) => Some((lit, true)),
93                _ => None
94            }
95        }
96        _ => None
97    }
98}

100/// checks if an expression is a constant number literal
101pub fn as_number(expr: &syn::Expr) -> Option<u64> {
102    match as_lit(expr)?  {
103        syn::Lit::Int(i) => i.base10_parse().ok(),
104        _ => None
105    }
106}
107
108/// checks if an expression is a signed number literal
109pub fn as_signed_number(expr: &syn::Expr) -> Option<i64> {
110    let (expr, negated) = as_lit_with_negation(expr)?;
111    match expr {
112        syn::Lit::Int(i) => if let Ok(value) = i.base10_parse::<u64>() {
113            let value: i64 = value.try_into().ok()?;
114            Some (if negated {-value} else {value})
115        } else {
116            None
117        },
118        _ => None
119    }
120}

122/// checks if an expression is a constant float literal
123pub fn as_float(expr: &syn::Expr) -> Option<f64> {
124    let (expr, negated) = as_lit_with_negation(expr)?;
125    match expr {
126        syn::Lit::Float(i) => i.base10_parse::<f64>().ok().map(|i| if negated { -i } else { i } ),
127        _ => None
128    }
