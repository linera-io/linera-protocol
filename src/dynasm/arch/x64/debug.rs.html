6pub fn format_opdata_list(name: &str, data: &[Opdata]) -> String {
7    let mut forms = Vec::new();
8    for data in data {
9        forms.extend(format_opdata(name, data));
10    }
11    forms.join("\n")
12}

14pub fn format_opdata(name: &str, data: &Opdata) -> Vec<String> {
15    let opsizes = if data.flags.contains(Flags::AUTO_SIZE) {&b"qwd"[..]}
16             else if data.flags.contains(Flags::AUTO_NO32) {&b"qw"[..]}
17             else if data.flags.contains(Flags::AUTO_REXW) {&b"qd"[..]}
18             else if data.flags.contains(Flags::AUTO_VEXL) {&b"ho"[..]}
19             else if name == "monitorx"                    {&b"qwd"[..]}
20             else                                          {&b"!"[..]};
21
22    let mut forms = Vec::new();
23    for opsize in opsizes.iter().cloned() {
24        let mut buf = String::new();
25        buf.push_str(">>> ");
26        buf.push_str(name);
27        let mut first = true;
28        for (ty, size) in FormatStringIterator::new(data.args) {
29            if first {
30                buf.push_str(" ");
31                first = false;
32            } else {
33                buf.push_str(", ");
34            }
35            buf.push_str(&format_arg(ty, size, opsize))
36        }
37        if data.flags.contains(Flags::X86_ONLY) {
38            for _ in buf.len() .. 45 {
39                buf.push(' ');
40            }
41            buf.push_str(" (x86 only)");
42        }
43        if !data.features.is_empty() {
44            for _ in buf.len() .. 45 {
45                buf.push(' ');
46            }
47            buf.push_str(&format!(" ({})", data.features));
48        }
49        forms.push(buf);
50    }
51    forms
52}

54static REGS: [&str; 16] = ["a",  "d",  "c",   "b",   "bp",  "sp",  "si",  "di",
55                                   "r8", "r9", "r10", "r11", "r12", "r13", "r14", "r15"];
56static SEGREGS: [&str; 6] = ["es", "cs", "ss", "ds", "fs", "gs"];
57
58fn format_arg(ty: u8, mut size: u8, opsize: u8) -> Cow<'static, str> {
59    if size == b'*' {
60        size = if opsize == b'q' && (ty == b'i' || ty == b'o') {
61            b'd'
62        } else {
63            opsize
64        };
65    }
66
67    fn format_size(size: u8) -> &'static str {
68        match size {
69            b'b' => "8",
70            b'w' => "16",
71            b'd' => "32",
72            b'f' => "48",
73            b'q' => "64",
74            b'p' => "80",
75            b'o' => "128",
76            b'h' => "256",
77            _ => ""
78        }
79    }
80
81    match ty {
82        b'i' => format!("imm{}",      format_size(size)).into(),
83        b'o' => format!("rel{}off",   format_size(size)).into(),
84        b'm' => format!("mem{}",      format_size(size)).into(),
85        b'k' => format!("vm32addr{}", format_size(size)).into(),
86        b'l' => format!("vm64addr{}", format_size(size)).into(),
87        b'r' => format!("reg{}",      format_size(size)).into(),
88        b'f' => "st".into(),
89        b'x' => "mm".into(),
90        b'y' => (if size == b'h' {"ymm"} else {"xmm"}).into(),
91        b's' => "segreg".into(),
92        b'c' => "creg".into(),
93        b'd' => "dreg".into(),
94        b'b' => "bndreg".into(),
95        b'v' => format!("reg/mem{}", format_size(size)).into(),
96        b'u' => format!("mm/mem{}", format_size(size)).into(),
97        b'w' => format!("{}mm/mem{}", if size == b'h' {"y"} else {"x"}, format_size(size)).into(),
98        b'A'..=b'P' => {
99            let i = ty as usize - 'A' as usize;
100            match size {
101                b'b' => if i < 4 { format!("{}l", REGS[i]).into() }
102                   else if i < 8 { REGS[i].into() }
103                   else          { format!("{}b", REGS[i]).into() },
104                b'w' => if i < 4 { format!("{}x", REGS[i]).into() }
105                   else if i < 8 { REGS[i].into() }
106                   else          { format!("{}w", REGS[i]).into() },
107                b'd' => if i < 4 { format!("e{}x",REGS[i]).into() }
108                   else if i < 8 { format!("e{}", REGS[i]).into() }
109                   else          { format!("{}d", REGS[i]).into() },
110                b'q' => if i < 4 { format!("r{}x",REGS[i]).into() }
111                   else          { format!("r{}", REGS[i]).into() },
112                _ => panic!("invalid formatting data")
113            }
114        },
115        b'Q'..=b'V' => SEGREGS[ty as usize - 'Q' as usize].into(),
116        b'W' => "cr8".into(),
117        b'X' => "st0".into(),
118        _ => panic!("invalid formatting data")
119    }
120}

122#[cfg(feature = "dynasm_opmap")]
123pub fn create_opmap() -> String {
124    let mut s = String::new();
125
126    let mut mnemnonics: Vec<_> = super::x64data::mnemnonics().cloned().collect();
127    mnemnonics.sort();
128
129    for mnemnonic in mnemnonics {
130        // get the data for this mnemnonic
131        let data = super::x64data::get_mnemnonic_data(mnemnonic).unwrap();
132        // format the data for the opmap docs
133        let mut formats = data.into_iter()
134            .map(|x| format_opdata(mnemnonic, x))
135            .flat_map(|x| x)
136            .map(|x| x.replace(">>> ", ""))
137            .collect::<Vec<_>>();
138        formats.sort();
139
140        // push mnemnonic name as title
141        s.push_str("### ");
142        s.push_str(mnemnonic);
143        s.push_str("\n```insref\n");
144
145        // push the formats
146        s.push_str(&formats.join("\n"));
147        s.push_str("\n```\n");
148    }
149    s
150}
