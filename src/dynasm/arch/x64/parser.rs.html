20// parses a full instruction
21// syntax for a single op: PREFIX* ident (SIZE? expr ("," SIZE? expr)*)? ";"
22pub(super) fn parse_instruction(ctx: &mut Context, input: parse::ParseStream) -> parse::Result<(Instruction, Vec<RawArg>)> {
23    let span = input.cursor().span();
24
25    let mut ops = Vec::new();
26
27    // read prefixes + op
28    let mut op = parse_ident_or_rust_keyword(input)?;
29    while is_prefix(&op) {
30        ops.push(op);
31        op = parse_ident_or_rust_keyword(input)?;
32    }
33    ops.push(op);
34
35    // parse (sizehint? expr),*
36    let mut args = Vec::new();
37
38    if !(input.is_empty() || input.peek(Token![;])) {
39        args.push(parse_arg(ctx, input)?);
40
41        while input.peek(Token![,]) {
42            let _: Token![,] = input.parse()?;
43
44            args.push(parse_arg(ctx, input)?);
45        }
46    }
47
48    // let span = span.join(input.cursor().span()); // FIXME can't join spans ATM
49
50    Ok((
51        Instruction {
52            idents: ops,
53            span
54        },
55        args
56    ))
57}

59/// checks if the given ident is a valid x86 prefix
60fn is_prefix(ident: &syn::Ident) -> bool {
61    const PREFIXES: [&str; 12] = [
62        "lock",
63        "rep", "repe", "repz",
64        "repne", "repnz",
65        "ss", "cs", "ds", "es", "fs", "gs"
66    ];
67
68    PREFIXES.contains(&ident.to_string().as_str())
69}

71/// if a size hint is present in the parse stream, returning the indicated size
72fn eat_size_hint(ctx: &Context, input: parse::ParseStream) -> Option<Size> {
73    const X86_SIZES: [(&str, Size); 9] = [
74        ("BYTE", Size::BYTE),
75        ("WORD", Size::WORD),
76        ("DWORD", Size::DWORD),
77        ("AWORD", Size::DWORD),
78        ("FWORD", Size::FWORD),
79        ("QWORD", Size::QWORD),
80        ("TWORD", Size::PWORD),
81        ("OWORD", Size::OWORD),
82        ("YWORD", Size::HWORD)
83    ];
84    const X64_SIZES: [(&str, Size); 9] = [
85        ("BYTE", Size::BYTE),
86        ("WORD", Size::WORD),
87        ("DWORD", Size::DWORD),
88        ("FWORD", Size::FWORD),
89        ("AWORD", Size::QWORD),
90        ("QWORD", Size::QWORD),
91        ("TWORD", Size::PWORD),
92        ("OWORD", Size::OWORD),
93        ("YWORD", Size::HWORD)
94    ];
95
96    let sizes = match ctx.mode {
97        X86Mode::Protected => &X86_SIZES,
98        X86Mode::Long      => &X64_SIZES
99    };
100    for &(kw, size) in sizes {
101        if eat_pseudo_keyword(input, kw) {
102            return Some(size);
103        }
104    }
105    None
106}

108/// tries to parse a full arg definition
109fn parse_arg(ctx: &mut Context, input: parse::ParseStream) -> parse::Result<RawArg> {
110    // sizehint
111    let size = eat_size_hint(ctx, input);
112
113    let _start = input.cursor().span(); // FIXME can't join spans yet
114
115    // bare label
116    if let Some(jump) = input.parse_opt()? {
117        return Ok(RawArg::JumpTarget {
118            jump,
119            size
120        })
121    }
122
123    // indirect
124    if input.peek(syn::token::Bracket) {
125        let span = input.cursor().span();
126        let inner;
127        let _ = syn::bracketed!(inner in input);
128        let inner = &inner;
129
130        // label
131        if let Some(jump) = inner.parse_opt()? {
132            return Ok(RawArg::IndirectJumpTarget {
133                jump,
134                size
135            })
136        }
137
138        // memory reference
139        let nosplit = eat_pseudo_keyword(inner, "NOSPLIT");
140        let disp_size = eat_size_hint(ctx, inner);
141        let expr: syn::Expr = inner.parse()?;
142
143        // split the expression into the different (displacement, register, scaled register) components
144        let items = parse_adds(ctx, expr);
145
146        return Ok(RawArg::IndirectRaw {
147            span,
148            nosplit,
149            value_size: size,
150            disp_size,
151            items
152        })
153
154    }
155
156    // it's a normal (register/immediate/typemapped) operand
157    let arg: syn::Expr = input.parse()?;
158
159    // typemapped: expr => type [expr] . ident
160    if input.peek(Token![=>]) {
161        let _: Token![=>] = input.parse()?;
162
163        let base = if let Some((_, base)) = parse_reg(ctx, &arg) {
164            base
165        } else {
166            emit_error!(arg, "Expected register");
167            return Ok(RawArg::Invalid);
168        };
169
170        let ty: syn::Path = input.parse()?;
171
172        // any attribute, register as index and immediate in index
173        let mut nosplit = false;
174        let mut disp_size = None;
175
176        let items = if input.peek(syn::token::Bracket) {
177            let inner;
178            let _brackets = syn::bracketed!(inner in input);
179            let inner = &inner;
180
181            nosplit = eat_pseudo_keyword(inner, "NOSPLIT");
182            disp_size = eat_size_hint(ctx, inner);
183            let index_expr: syn::Expr = inner.parse()?;
184
185            parse_adds(ctx, index_expr)
186        } else {
187            Vec::new()
188        };
189
190        let attr: Option<syn::Ident> = if input.peek(Token![.]) {
191            let _: Token![.] = input.parse()?;
192            Some(input.parse()?)
193        } else {
194            None
195        };
196
197        return Ok(RawArg::TypeMappedRaw {
198            span: arg.span(), // FIXME can't join spans yet
199            nosplit,
200            value_size: size,
201            disp_size,
202            base_reg: base,
203            scale: ty,
204            scaled_items: items,
205            attribute: attr,
206        });
207    }
208
209    // direct register
210    if let Some((span, reg)) = parse_reg(ctx, &arg) {
211        if size.is_some() {
212            emit_error!(span, "size hint with direct register");
213        }
214        return Ok(RawArg::Direct {
215            reg,
216            span
217        })
218    }
219
220    // immediate
221    Ok(RawArg::Immediate {
222        value: arg,
223        size
224    })
225}

227/// checks if an expression is interpretable as a register reference.
228fn parse_reg(ctx: &Context, expr: &syn::Expr) -> Option<(Span, Register)> {
229    if let Some(path) = as_ident(expr) {
230        // static register names
231
232        let name = path.to_string();
233        let mut name = name.as_str();
234        if let Some(x) = ctx.state.invocation_context.aliases.get(name) {
235            name = x;
236        }
237
238        let (reg, size) = match ctx.mode {
239            X86Mode::Long      => X64_REGISTERS.get(&name).cloned(),
240            X86Mode::Protected => X86_REGISTERS.get(&name).cloned()
241        }?;
242
243        Some((
244            path.span(),
245            Register::new_static(size, reg)
246        ))
247
248    } else if let syn::Expr::Call(syn::ExprCall {ref func, ref args, ..}) = expr {
249        // dynamically chosen registers ( ident(expr) )
250        if args.len() != 1 {
251            return None;
252        }
253
254        let called = if let Some(called) = as_ident(&&*func) {
255            called
256        } else {
257            return None;
258        };
259
260        let name = called.to_string();
261        let name = name.as_str();
262        let (size, family) = match ctx.mode {
263            X86Mode::Long      => X64_FAMILIES.get(&name).cloned(),
264            X86Mode::Protected => X86_FAMILIES.get(&name).cloned()
265        }?;
266
267        Some((
268            expr.span(), // FIXME:can't join spans atm
269            Register::new_dynamic(size, family, args[0].clone())
270        ))
271    } else {
272        None
273    }
274}

276/// splits an expression into different components of a memory reference.
277fn parse_adds(ctx: &Context, expr: syn::Expr) -> Vec<MemoryRefItem> {
278    let mut adds = Vec::new();
279    collect_adds(expr, &mut adds);
280
281    let mut items = Vec::new();
282
283    // detect what kind of equation we're dealing with
284    for node in adds {
285        // simple reg
286        if let Some((_, reg)) = parse_reg(ctx, &node) {
287            items.push(MemoryRefItem::Register(reg));
288            continue;
289        }
290        if let syn::Expr::Binary(syn::ExprBinary { op: syn::BinOp::Mul(_), ref left, ref right, .. } ) = node {
291            // reg * const
292            if let Some((_, reg)) = parse_reg(ctx, left) {
293                if let syn::Expr::Lit(syn::ExprLit {ref lit, ..}) = **right {
294                    if let syn::Lit::Int(lit) = lit {
295                        if let Ok(value) = lit.base10_parse::<isize>() {
296                            items.push(MemoryRefItem::ScaledRegister(reg, value));
297                            continue;
298                        }
299                    }
300                }
301            } // const * reg
302            if let Some((_, reg)) = parse_reg(ctx, right) {
303                if let syn::Expr::Lit(syn::ExprLit {ref lit, ..}) = **left {
304                    if let syn::Lit::Int(lit) = lit {
305                        if let Ok(value) = lit.base10_parse::<isize>() {
306                            items.push(MemoryRefItem::ScaledRegister(reg, value));
307                            continue;
308                        }
309                    }
310                }
311            }
312        }
313        items.push(MemoryRefItem::Displacement(node));
314    }
315
316    items
317}

319/// Takes an expression and splits all added (and subtracted) components
320fn collect_adds(node: syn::Expr, collection: &mut Vec<syn::Expr>) {
321    if let syn::Expr::Binary(syn::ExprBinary { op: syn::BinOp::Add(_), left, right, .. } ) = node {
322        collect_adds(*left, collection);
323        collect_adds(*right, collection);
324    } else if let syn::Expr::Binary(syn::ExprBinary { op: syn::BinOp::Sub(sub), left, right, .. } ) = node {
325        collect_adds(*left, collection);
326        collection.push(syn::Expr::Unary(syn::ExprUnary { op: syn::UnOp::Neg(sub), expr: right, attrs: Vec::new() } ));
327    } else {
328        collection.push(node);
329    }
330}
