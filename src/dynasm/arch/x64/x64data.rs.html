9macro_rules! constify {
10    ($t:ty, $e:expr) => { {const C: &$t = &$e; C} }

13macro_rules! OpInner {
14    ($fmt:expr, $ops:expr, $reg:expr)          => { Opdata {args: $fmt, ops: constify!([u8], $ops), reg: $reg, flags: Flags::DEFAULT,  features: Features::X64_IMPLICIT}  };
15    ($fmt:expr, $ops:expr, $reg:expr, $f:expr) => { Opdata {args: $fmt, ops: constify!([u8], $ops), reg: $reg, flags: Flags::make($f), features: Features::X64_IMPLICIT}  };
16    ($fmt:expr, $ops:expr, $reg:expr, $f:expr, $ft:expr) => { Opdata {args: $fmt, ops: constify!([u8], $ops), reg: $reg, flags: Flags::make($f), features: Features::make($ft)}  };
17
18}

38bitflags! {
39    pub struct Flags: u32 {
40        const DEFAULT   = 0x0000_0000; // this instruction has default encoding
41        const VEX_OP    = 0x0000_0001; // this instruction requires a VEX prefix to be encoded
42        const XOP_OP    = 0x0000_0002; // this instruction requires a XOP prefix to be encoded
43        const IMM_OP    = 0x0000_0004; // this instruction encodes the final opcode byte in the immediate position, like 3DNow! ops.
44
45        // note: the first 4 in this block are mutually exclusive
46        const AUTO_SIZE = 0x0000_0008; // 16 bit -> OPSIZE , 32-bit -> None     , 64-bit -> REX.W/VEX.W/XOP.W
47        const AUTO_NO32 = 0x0000_0010; // 16 bit -> OPSIZE , 32-bit -> None(x86), 64-bit -> None(x64)
48        const AUTO_REXW = 0x0000_0020; // 16 bit -> illegal, 32-bit -> None     , 64-bit -> REX.W/VEX.W/XOP.W
49        const AUTO_VEXL = 0x0000_0040; // 128bit -> None   , 256bit -> VEX.L
50        const WORD_SIZE = 0x0000_0080; // implies opsize prefix
51        const WITH_REXW = 0x0000_0100; // implies REX.W/VEX.W/XOP.W
52        const WITH_VEXL = 0x0000_0200; // implies VEX.L/XOP.L
53        const EXACT_SIZE= 0x0000_0400; // operands with unknown sizes cannot be assumed to match
54
55        const PREF_66   = WORD_SIZE;   // mandatory prefix (same as WORD_SIZE)
56        const PREF_67   = 0x0000_0800; // mandatory prefix (same as SMALL_ADDRESS)
57        const PREF_F0   = 0x0000_1000; // mandatory prefix (same as LOCK)
58        const PREF_F2   = 0x0000_2000; // mandatory prefix (REPNE)
59        const PREF_F3   = 0x0000_4000; // mandatory prefix (REP)
60
61        const LOCK      = 0x0000_8000; // user lock prefix is valid with this instruction
62        const REP       = 0x0001_0000; // user rep prefix is valid with this instruction
63        const REPE      = 0x0002_0000;
64
65        const SHORT_ARG = 0x0004_0000; // a register argument is encoded in the last byte of the opcode
66        const ENC_MR    = 0x0008_0000; // select alternate arg encoding
67        const ENC_VM    = 0x0010_0000; // select alternate arg encoding
68        const ENC_MIB   = 0x0020_0000; // A special encoding using the SIB to specify an immediate and two registers
69        const X86_ONLY  = 0x0040_0000; // instructions available in protected mode, but not long mode
70    }
71}

73impl Flags {
74    const fn make(bits: u32) -> Flags {
75        Flags { bits }
76    }

147impl Display for Features {
148    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
149        let mut keys = Vec::new();
150        if self.contains(Features::FPU)   { keys.push("fpu"); }
151        if self.contains(Features::MMX)   { keys.push("mmx"); }
152        if self.contains(Features::TDNOW) { keys.push("tdnow"); }
153        if self.contains(Features::SSE)   { keys.push("sse"); }
154        if self.contains(Features::SSE2)  { keys.push("sse2"); }
155        if self.contains(Features::SSE3)  { keys.push("sse3"); }
156        if self.contains(Features::VMX)   { keys.push("vmx"); }
157        if self.contains(Features::SSSE3) { keys.push("ssse3"); }
158        if self.contains(Features::SSE4A) { keys.push("sse4a"); }
159        if self.contains(Features::SSE41) { keys.push("sse41"); }
160        if self.contains(Features::SSE42) { keys.push("sse42"); }
161        if self.contains(Features::SSE5)  { keys.push("sse5"); }
162        if self.contains(Features::AVX)   { keys.push("avx"); }
163        if self.contains(Features::AVX2)  { keys.push("avx2"); }
164        if self.contains(Features::FMA)   { keys.push("fma"); }
165        if self.contains(Features::BMI1)  { keys.push("bmi1"); }
166        if self.contains(Features::BMI2)  { keys.push("bmi2"); }
167        if self.contains(Features::TBM)   { keys.push("tbm"); }
168        if self.contains(Features::RTM)   { keys.push("rtm"); }
169        if self.contains(Features::INVPCID) { keys.push("invpcid"); }
170        if self.contains(Features::MPX)   { keys.push("mpx"); }
171        if self.contains(Features::SHA)   { keys.push("sha"); }
172        if self.contains(Features::PREFETCHWT1) { keys.push("prefetchwt1"); }
173        if self.contains(Features::CYRIX) { keys.push("cyrix"); }
174        if self.contains(Features::AMD)   { keys.push("amd"); }
175        for (i, k) in keys.into_iter().enumerate() {
176            if i != 0 {
177                f.write_str(", ")?;
178            }
179            f.write_str(k)?;
180        }
181        Ok(())
182    }
