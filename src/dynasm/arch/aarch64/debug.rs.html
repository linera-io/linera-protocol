8#[cfg(feature = "dynasm_opmap")]
9pub fn create_opmap() -> String {
10    let mut s = String::new();
11
12    let mut mnemnonics: Vec<_> = super::aarch64data::mnemnonics().cloned().collect();
13    mnemnonics.sort();
14
15    for mnemnonic in mnemnonics {
16        // get the data for this mnemnonic
17        let data = super::aarch64data::get_mnemonic_data(mnemnonic).unwrap();
18        // format the data for the opmap docs
19        let formats = data.into_iter()
20            .map(|x| format_opdata(mnemnonic, x))
21            .flat_map(|x| x)
22            .map(|x| x.replace(">>> ", ""))
23            .collect::<Vec<_>>();
24
25        // push mnemnonic name as title
26        write!(s, "### {}\n```insref\n{}\n```\n", mnemnonic, formats.join("\n")).unwrap();
27    }
28    s
29}

64pub fn format_opdata(name: &str, data: &Opdata) -> Vec<String> {
65
66    let has_simd_full_width = data.matchers.iter().any(|m| match m {
67        Matcher::V(_) | Matcher::RegList(_, _) => true,
68        _ => false
69    });
70
71    let form_count = 1 + has_simd_full_width as u8;
72    let mut forms = Vec::new();
73
74    for i in 0 .. form_count {
75        let mut buf = format!(">>> {}", name);
76
77        let (constraints, names) = match constraints_and_names(data) {
78            Ok(o) => o,
79            Err(e) => panic!("Encountered a faulty op listing for {}: {}", name, e)
80        };
81
82        let mut first = true;
83        let mut after_dot = false;
84        let mut end_count = 0;
85        let mut names = &names[..];
86
87        for matcher in data.matchers {
88            if let Matcher::End = matcher {
89                end_count += 1;
90                buf.push_str(" {");
91                continue;
92            } else if let Matcher::Dot = matcher {
93                after_dot = true;
94                buf.push_str(".");
95                continue;
96            }
97
98            if first {
99                if !after_dot {
100                    buf.push_str(" ");
101                    first = false;
102                }
103                after_dot = false;
104            } else {
105                buf.push_str(", ");
106            }
107
108            let (arg_names, rest) = names.split_at(matcher.flatarg_count());
109            names = rest;
110
111            match matcher {
112                Matcher::Dot => (),
113                Matcher::Lit(s) => write!(buf, "{}", s).unwrap(),
114                Matcher::LitInt(v) => write!(buf, "{}", v).unwrap(),
115                Matcher::LitFloat(v) => write!(buf, "{}", v).unwrap(),
116                Matcher::Ident => write!(buf, "{}", arg_names[0]).unwrap(),
117                Matcher::Cond => write!(buf, "<cond>").unwrap(),
118                Matcher::Imm => write!(buf, "#{}", arg_names[0]).unwrap(),
119                Matcher::W =>   write!(buf, "W{}", arg_names[0]).unwrap(),
120                Matcher::X =>   write!(buf, "X{}", arg_names[0]).unwrap(),
121                Matcher::WSP => write!(buf, "W{}|WSP", arg_names[0]).unwrap(),
122                Matcher::XSP => write!(buf, "X{}|SP", arg_names[0]).unwrap(),
123                Matcher::B =>   write!(buf, "B{}", arg_names[0]).unwrap(),
124                Matcher::H =>   write!(buf, "H{}", arg_names[0]).unwrap(),
125                Matcher::S =>   write!(buf, "S{}", arg_names[0]).unwrap(),
126                Matcher::D =>   write!(buf, "D{}", arg_names[0]).unwrap(),
127                Matcher::Q =>   write!(buf, "Q{}", arg_names[0]).unwrap(),
128                Matcher::V(s) => {
129                    let width = if i == 0 { 16 } else { 8 };
130                    write!(buf, "V{}.{}{}", arg_names[0], size_to_string(*s), width / s.in_bytes()).unwrap();
131                },
132                Matcher::VStatic(s, c) => write!(buf, "V{}.{}{}", arg_names[0], size_to_string(*s), c).unwrap(),
133                Matcher::VElement(s) => write!(buf, "V{}.{}[{}]", arg_names[0], size_to_string(*s), arg_names[1]).unwrap(),
134                Matcher::VElementStatic(s, element) => write!(buf, "V{}.{}[{}]", arg_names[0], size_to_string(*s), element).unwrap(),
135                Matcher::VStaticElement(s, c) => write!(buf, "V{}.{}{}[{}]", arg_names[0], size_to_string(*s), c, arg_names[1]).unwrap(),
136                Matcher::RegList(a, s) => {
137                    let width = if i == 0 { 16 } else { 8 };
138                    write!(buf, "{{V{}.{}{} * {}}}", arg_names[0], size_to_string(*s), width / s.in_bytes(), a).unwrap();
139                },
140                Matcher::RegListStatic(a, s, c) => write!(buf, "{{V{}.{}{} * {}}}", arg_names[0], size_to_string(*s), c, a).unwrap(),
141                Matcher::RegListElement(a, s) =>   write!(buf, "{{V{}.{} * {}}}[{}]", arg_names[0], size_to_string(*s), a, arg_names[1]).unwrap(),
142                Matcher::Offset => buf.push_str(&arg_names[0]),
143                Matcher::RefBase =>   write!(buf, "[X{}|SP]", arg_names[0]).unwrap(),
144                Matcher::RefOffset => write!(buf, "[X{}|SP {{, #{} }} ]", arg_names[0], arg_names[1]).unwrap(),
145                Matcher::RefPre =>    write!(buf, "[X{}|SP, #{}]!", arg_names[0], arg_names[1]).unwrap(),
146                Matcher::RefIndex =>  write!(buf, "[X{}|SP, W{}|X{} {{ , UXTW|LSL|SXTW|SXTX {{ #{} }} }} ]", arg_names[0], arg_names[1], arg_names[1], arg_names[3]).unwrap(),
147                Matcher::LitMod(m) => {
148                    buf.push_str(m.as_str());
149                    if !m.expr_required() {
150                        write!(buf, " {{ #{} }}", arg_names[0]).unwrap();
151                    } else {
152                        write!(buf, " #{}", arg_names[0]).unwrap();
153                    }
154                },
155                Matcher::Mod(mods) => {
156                    let mut required = false;
157                    let mut unsigned_extends = String::new();
158                    let mut signed_extends   = String::new();
159                    let mut rest = Vec::new();
160                    for m in *mods {
161                        required = required || m.expr_required();
162                        match m {
163                            Modifier::LSL | Modifier::LSR | Modifier::ASR | Modifier::ROR | Modifier::MSL => rest.push(m.as_str()),
164                            Modifier::SXTX | Modifier::SXTW | Modifier::SXTH | Modifier::SXTB => signed_extends.push(m.as_str().chars().nth(3).unwrap()),
165                            Modifier::UXTX | Modifier::UXTW | Modifier::UXTH | Modifier::UXTB => unsigned_extends.push(m.as_str().chars().nth(3).unwrap()),
166                        }
167                    }
168                    if !unsigned_extends.is_empty() {
169                        if unsigned_extends.len() > 1 {
170                            unsigned_extends = format!("UXT[{}]", unsigned_extends);
171                        } else {
172                            unsigned_extends = format!("UXT{}", unsigned_extends);
173                        }
174                        rest.push(&unsigned_extends);
175                    }
176                    if !signed_extends.is_empty() {
177                        if signed_extends.len() > 1 {
178                            signed_extends = format!("SXT[{}]", signed_extends);
179                        } else {
180                            signed_extends = format!("SXT{}", signed_extends);
181                        }
182                        rest.push(&signed_extends);
183                    }
184                    buf.push_str(&rest.join("|"));
185
186                    if !required {
187                        write!(buf, " {{ #{} }}", arg_names[1]).unwrap();
188                    } else {
189                        write!(buf, " #{}", arg_names[1]).unwrap();
190                    }
191                },
192                Matcher::End => ()
193            }
194
195        }
196
197        for _ in 0 .. end_count {
198            buf.push_str(" }");
199        }
200
201        if let Some(c) = constraints {
202            let mut len = c.len() + buf.len();
203            while len < 100 {
204                buf.push(' ');
205                len += 1;
206            }
207            buf.push_str(&c);
208        }
209
210        forms.push(buf);
211    }
212
213    forms
214}

216pub fn size_to_string(size: Size) -> &'static str {
217    match size {
218        Size::BYTE => "B",
219        Size::WORD => "H",
220        Size::DWORD => "S",
221        Size::QWORD => "D",
222        Size::OWORD => "Q",
223        _ => unimplemented!()
224    }
225}

227fn constraints_and_names(opdata: &Opdata) -> Result<(Option<String>, Vec<String>), &'static str> {
228    let data = group_opdata(opdata)?;
229    let constraints = format_constraints(&data);
230    let names = data.into_iter().map(|a| a.name.unwrap_or_else(|| "?".into())).collect();
231    Ok((constraints, names))
232}

234fn group_opdata(opdata: &Opdata) -> Result<Vec<ArgWithCommands>, &'static str> {
235    let args = flatten_matchers(opdata.matchers);
236    let (max_cursor, commands) = group_commands(opdata.commands);
237
238    if args.len() != max_cursor {
239        return Err("arg / command count mismatch");
240    }
241
242    let mut args: Vec<_> = args.into_iter().map(|(arg, can_be_default)| ArgWithCommands {
243        arg,
244        can_be_default,
245        commands: Vec::new(),
246        name: None
247    }).collect();
248
249    for (command, idx) in commands {
250        args[idx].commands.push(command);
251    }
252
253    // validate the commands - argtypes
254    check_command_sanity(&args)?;
255
256    name_args(&mut args);
257
258    Ok(args)
259}

278/// Take a matcher array and return a vector of the types of flat arg each should produce
279fn flatten_matchers(matchers: &[Matcher]) -> Vec<(FlatArgTy, bool)> {
280    let mut args = Vec::new();
281    let mut default = false;
282
283    for matcher in matchers {
284        match matcher {
285            Matcher::Dot
286            | Matcher::Lit(_)
287            | Matcher::LitInt(_)
288            | Matcher::LitFloat(_) => (),
289            Matcher::Ident
290            | Matcher::Cond
291            | Matcher::Imm => args.push((FlatArgTy::Immediate, default)),
292            Matcher::W
293            | Matcher::X
294            | Matcher::WSP
295            | Matcher::XSP
296            | Matcher::B
297            | Matcher::H
298            | Matcher::S
299            | Matcher::D
300            | Matcher::Q => args.push((FlatArgTy::Direct, default)),
301            Matcher::V(_)
302            | Matcher::VStatic(_, _)
303            | Matcher::VElementStatic(_, _)
304            | Matcher::RegList(_, _)
305            | Matcher::RegListStatic(_, _, _) => args.push((FlatArgTy::Direct, default)),
306            Matcher::VElement(_)
307            | Matcher::VStaticElement(_, _)
308            | Matcher::RegListElement(_, _) => {
309                args.push((FlatArgTy::Direct, default));
310                args.push((FlatArgTy::Immediate, default));
311            },
312            Matcher::Offset => args.push((FlatArgTy::JumpTarget, default)),
313            Matcher::RefBase => args.push((FlatArgTy::Direct, default)),
314            Matcher::RefOffset => {
315                args.push((FlatArgTy::Direct, default));
316                args.push((FlatArgTy::Immediate, true));
317            },
318            Matcher::RefPre => {
319                args.push((FlatArgTy::Direct, default));
320                args.push((FlatArgTy::Immediate, default));
321            },
322            Matcher::RefIndex => {
323                args.push((FlatArgTy::Direct, default));
324                args.push((FlatArgTy::Direct, default));
325                args.push((FlatArgTy::Modifier, true));
326                args.push((FlatArgTy::Immediate, true));
327            },
328            Matcher::LitMod(_) => {
329                args.push((FlatArgTy::Immediate, true));
330            },
331            Matcher::Mod(_) => {
332                args.push((FlatArgTy::Modifier, default));
333                args.push((FlatArgTy::Immediate, true));
334            },
335            Matcher::End => default = true,
336        }
337    }
338    args
339}

341/// Take a commands slice and calculate the expected amount of args / a vec of command, argidx
342fn group_commands(commands: &[Command]) -> (usize, Vec<(Command, usize)>) {
343    let mut cursor = 0;
344    let mut command_idx = Vec::new();
345
346    for command in commands {
347        match command {
348            Command::A => {
349                cursor += 1;
350                continue;
351            },
352            Command::C => {
353                cursor -= 1;
354                continue;
355            },
356            Command::Rwidth(_) => {
357                continue;
358            },
359            _ => ()
360        }
361
362        command_idx.push((*command, cursor));
363        match command {
364            Command::R(_)
365            | Command::REven(_)
366            | Command::R4(_)
367            | Command::RNoZr(_)
368            | Command::RNext
369            | Command::Ubits(_, _)
370            | Command::Uscaled(_, _, _)
371            | Command::Ulist(_, _)
372            | Command::Urange(_, _, _)
373            | Command::Usub(_, _, _)
374            | Command::Unegmod(_, _)
375            | Command::Usumdec(_, _)
376            | Command::Ufields(_)
377            | Command::Sbits(_, _)
378            | Command::Sscaled(_, _,_)
379            | Command::Special(_, _)
380            | Command::Rotates(_)
381            | Command::ExtendsW(_)
382            | Command::ExtendsX(_)
383            | Command::Cond(_)
384            | Command::CondInv(_)
385            | Command::LitList(_, _)
386            | Command::Offset(_) => cursor += 1,
387            _ => ()
388        }
389    }
390
391    (cursor, command_idx)
392}

394/// checks if the commands for each arg type make sense
395fn check_command_sanity(args: &[ArgWithCommands]) -> Result<(), &'static str> {
396    for arg in args {
397        if arg.commands.is_empty() {
398            return Err("Arg with no commands")
399        }
400
401        for command in &arg.commands {
402            let check = match command {
403                Command::R(_)
404                | Command::REven(_)
405                | Command::R4(_)
406                | Command::RNoZr(_)
407                | Command::RNext => arg.arg == FlatArgTy::Direct,
408                Command::Ubits(_, _)
409                | Command::Uscaled(_, _, _)
410                | Command::Ulist(_, _)
411                | Command::Urange(_, _, _)
412                | Command::Usub(_, _, _)
413                | Command::Unegmod(_, _)
414                | Command::Usumdec(_, _)
415                | Command::Ufields(_)
416                | Command::Sbits(_, _)
417                | Command::Sscaled(_, _,_)
418                | Command::BUbits(_)
419                | Command::BUsum(_)
420                | Command::BSscaled(_, _)
421                | Command::BUrange(_, _)
422                | Command::Uslice(_, _, _)
423                | Command::Sslice(_, _, _)
424                | Command::Special(_, _) => arg.arg == FlatArgTy::Immediate,
425                Command::Cond(_)
426                | Command::CondInv(_)
427                | Command::LitList(_, _) => arg.arg == FlatArgTy::Lit || arg.arg == FlatArgTy::Immediate,
428                Command::Offset(_) => arg.arg == FlatArgTy::JumpTarget,
429                Command::Rotates(_)
430                | Command::ExtendsW(_)
431                | Command::ExtendsX(_) => arg.arg == FlatArgTy::Modifier,
432                Command::A
433                | Command::C
434                | Command::Rwidth(_) => unreachable!()
435            };
436
437            if !check {
438                return Err("command / argtype mismatch");
439            }
440
441            let check = match command {
442                Command::R(_)
443                | Command::Ubits(_, _)
444                | Command::Uscaled(_, _, _)
445                | Command::Uslice(_, _, _)
446                | Command::Urange(_, _, _)
447                | Command::Ulist(_, _)
448                | Command::Ufields(_)
449                | Command::Sbits(_, _)
450                | Command::Sscaled(_, _, _)
451                | Command::Sslice(_, _, _)
452                | Command::BUbits(_)
453                | Command::BUsum(_)
454                | Command::BSscaled(_, _)
455                | Command::Rotates(_)
456                | Command::ExtendsW(_)
457                | Command::ExtendsX(_) => true,
458                Command::R4(_)
459                | Command::RNoZr(_)
460                | Command::REven(_)
461                | Command::RNext
462                | Command::Usub(_, _, _)
463                | Command::Unegmod(_, _)
464                | Command::Usumdec(_, _)
465                | Command::BUrange(_, _)
466                | Command::Special(_, _)
467                | Command::Cond(_)
468                | Command::CondInv(_)
469                | Command::LitList(_, _)
470                | Command::Offset(_) => !arg.can_be_default,
471                Command::A
472                | Command::C
473                | Command::Rwidth(_) => unreachable!()
474            };
475
476            if !check {
477                return Err("default mismatch");
478            }
479        }
480    }
481
482    Ok(())
483}

485/// assign names to the args being used
486fn name_args(args: &mut [ArgWithCommands]) {
487    // iirc no op uses more than 4 unconstrained literals / immediates
488    let reg_name_list = ["n", "m", "a", "b"];
489    let mut reg_name_idx = 0;
490    let imm_name_list = ["", "1", "2", "3"];
491    let mut imm_name_idx = 0;
492
493    for arg in args {
494        match arg.arg {
495            FlatArgTy::Direct => {
496                match &arg.commands[0] {
497                    Command::R(_)
498                    | Command::REven(_)
499                    | Command::RNoZr(_)
500                    | Command::R4(_) => {
501                        arg.name = Some(reg_name_list[reg_name_idx].to_string());
502                        reg_name_idx += 1;
503                    },
504                    Command::RNext => {
505                        arg.name = Some(format!("{}+1", reg_name_list[reg_name_idx - 1]));
506                    },
507                    _ => unreachable!()
508                }
509            },
510            FlatArgTy::Immediate => {
511                match &arg.commands[0] {
512                    Command::Cond(_)
513                    | Command::CondInv(_) => arg.name = None,
514                    Command::LitList(_, name) => arg.name = Some(name.trim_end_matches('S').to_lowercase()),
515                    Command::Ubits(_, _)
516                    | Command::Uscaled(_, _, _)
517                    | Command::Ulist(_, _)
518                    | Command::Urange(_, _, _)
519                    | Command::Usub(_, _, _)
520                    | Command::Unegmod(_, _)
521                    | Command::Usumdec(_, _)
522                    | Command::Ufields(_)
523                    | Command::BUbits(_)
524                    | Command::BUsum(_)
525                    | Command::BUrange(_, _)
526                    | Command::Uslice(_, _, _) => {
527                        arg.name = Some(format!("uimm{}", imm_name_list[imm_name_idx]));
528                        imm_name_idx += 1;
529                    },
530                    Command::Sbits(_, _)
531                    | Command::Sscaled(_, _,_)
532                    | Command::BSscaled(_, _)
533                    | Command::Sslice(_, _, _) => {
534                        arg.name = Some(format!("simm{}", imm_name_list[imm_name_idx]));
535                        imm_name_idx += 1;
536                    },
537                    Command::Special(_, _) => {
538                        arg.name = Some(format!("imm{}", imm_name_list[imm_name_idx]));
539                        imm_name_idx += 1;
540                    },
541                    _ => unreachable!()
542                }
543            },
544            FlatArgTy::Modifier => arg.name = None,
545            FlatArgTy::JumpTarget => match &arg.commands[0] {
546                Command::Offset(_) => arg.name = Some("<offset>".to_string()),
547                _ => unreachable!()
548            },
549            FlatArgTy::Lit => match &arg.commands[0] {
550                Command::Cond(_)
551                | Command::CondInv(_) => arg.name = None,
552                Command::LitList(_, name) => arg.name = Some(name.trim_end_matches('S').to_lowercase()),
553                _ => unreachable!()
554            }
555        }
556    }
557}

559fn format_constraints(args: &[ArgWithCommands]) -> Option<String> {
560    let mut constraints = String::new();
561    let mut prevname = "?";
562
563    for arg in args {
564        if let Some(ref name) = arg.name {
565            emit_constraints(name, prevname, &arg.commands, &mut constraints);
566            prevname = name;
567        }
568    }
569
570    if constraints.is_empty() {
571        None
572    } else {
573        let len = constraints.len();
574        Some(format!(" ({})", &constraints[0 .. len - 2]))
575    }
576}

578fn emit_constraints(name: &str, prevname: &str, commands: &[Command], buf: &mut String) {
579    for command in commands {
580        match command {
581            Command::R4(_) => write!(buf, "{} is 0-15", name),
582            Command::RNoZr(_) => write!(buf, "{} is 0-30", name),
583            Command::REven(_) => write!(buf, "{} is even", name),
584            Command::Ubits(_, bits)
585            | Command::BUbits(bits) => write!(buf, "#{} < {}", name, 1u32 << bits),
586            Command::Uscaled(_, bits, scale) => write!(buf, "#{} < {}, #{} = {} * N", name, 1u32 << (bits + scale), name, 1u32 << scale),
587            Command::Ulist(_, list) => {
588                let numbers = list.iter().map(|n| n.to_string()).collect::<Vec<_>>().join(", ");
589                write!(buf, "#{} = [{}]", name, numbers)
590            },
591            Command::Urange(_, min, max)
592            | Command::BUrange(min, max) => write!(buf, "{} <= #{} <= {}", min, name, max),
593            Command::Usub(_, bits, addval) => write!(buf, "{} <= #{} <= {}", u32::from(*addval) + 1 - (1u32 << bits), name, addval),
594            Command::Unegmod(_, bits) => write!(buf, "0 <= #{} < {}", name, 1u32 << bits),
595            Command::Usumdec(_, bits)
596            | Command::BUsum(bits) => write!(buf, "1 <= #{} <= {} - {}", name, 1u32 << bits, prevname),
597            Command::Ufields(fields) => write!(buf, "#{} < {}", name, 1u32 << fields.len()),
598            Command::Sbits(_, bits) => write!(buf, "-{} <= #{} < {}", 1u32 << (bits - 1), name, 1u32 << (bits - 1)),
599            Command::Sscaled(_, bits, scale)
600            | Command::BSscaled(bits, scale) => write!(buf, "-{} <= #{} < {}, #{} = {} * N", 1u32 << (bits + scale - 1), name, 1u32 << (bits + scale - 1), name, 1u32 << scale),
601            Command::Special(_, SpecialComm::WIDE_IMMEDIATE_W)
602            | Command::Special(_, SpecialComm::WIDE_IMMEDIATE_X)
603            | Command::Special(_, SpecialComm::INVERTED_WIDE_IMMEDIATE_W)
604            | Command::Special(_, SpecialComm::INVERTED_WIDE_IMMEDIATE_X) => write!(buf, "#{} is a wide immediate", name),
605            Command::Special(_, SpecialComm::LOGICAL_IMMEDIATE_W)
606            | Command::Special(_, SpecialComm::LOGICAL_IMMEDIATE_X) => write!(buf, "#{} is a logical immediate", name),
607            Command::Special(_, SpecialComm::FLOAT_IMMEDIATE)
608            | Command::Special(_, SpecialComm::SPLIT_FLOAT_IMMEDIATE) => write!(buf, "#{} is a floating point immediate", name),
609            Command::Special(_, SpecialComm::STRETCHED_IMMEDIATE) => write!(buf, "#{} is a stretched immediate", name),
610            Command::Offset(Relocation::B) => write!(buf, "offset is 26 bit, 4-byte aligned"),
611            Command::Offset(Relocation::BCOND) => write!(buf, "offset is 19 bit, 4-byte aligned"),
612            Command::Offset(Relocation::ADR) => write!(buf, "offset is 21 bit"),
613            Command::Offset(Relocation::ADRP) => write!(buf, "offset is 21 bit, 4K-page aligned"),
614            Command::Offset(Relocation::TBZ) => write!(buf, "offset is 14 bit, 4-byte aligned"),
615            Command::Offset(Relocation::LITERAL32) => write!(buf, "offset is 32 bit>"),
616            Command::Offset(Relocation::LITERAL64) => write!(buf, "offset is 64 bit>"),
617            _ => continue
618        }.unwrap();
619
620        write!(buf, ", ").unwrap();
621        break;
622    }
623}

625#[cfg(feature = "dynasm_extract")]
626pub fn extract_opdata(name: &str, data: &Opdata) -> Vec<String> {
627
628    let has_simd_full_width = data.matchers.iter().any(|m| match m {
629        Matcher::V(_) | Matcher::RegList(_, _) => true,
630        _ => false
631    });
632
633    let form_count = 1 + has_simd_full_width as u8;
634    let mut forms = Vec::new();
635
636    for i in 0 .. form_count {
637        let mut buf = format!("\"{}", name);
638
639        let mut first = true;
640        let mut after_dot = false;
641        let mut end_count = 0;
642        let mut arg_idx = 0;
643
644        let grouped = group_opdata(data).unwrap();
645        let mut constraints = extract_constraints(&grouped);
646
647        for matcher in data.matchers {
648            if let Matcher::End = matcher {
649                end_count += 1;
650                buf.push_str(" <");
651                continue;
652            } else if let Matcher::Dot = matcher {
653                after_dot = true;
654                buf.push_str(".");
655                continue;
656            }
657
658            if first {
659                if !after_dot {
660                    buf.push_str(" ");
661                    first = false;
662                }
663                after_dot = false;
664            } else {
665                buf.push_str(", ");
666            }
667
668            match matcher {
669                Matcher::Dot => (),
670                Matcher::Lit(s) => write!(buf, "{}", s).unwrap(),
671                Matcher::LitInt(v) => write!(buf, "{}", v).unwrap(),
672                Matcher::LitFloat(v) => write!(buf, "{:.5}", v).unwrap(),
673                Matcher::Ident
674                | Matcher::Cond => write!(buf, "<Ident,{}>", arg_idx).unwrap(),
675                Matcher::Imm => write!(buf, "<Imm,{}>", arg_idx).unwrap(),
676                Matcher::W =>   write!(buf, "<W,{}>", arg_idx).unwrap(),
677                Matcher::X =>   write!(buf, "<X,{}>", arg_idx).unwrap(),
678                Matcher::WSP => write!(buf, "<WSP,{}>", arg_idx).unwrap(),
679                Matcher::XSP => write!(buf, "<XSP,{}>", arg_idx).unwrap(),
680                Matcher::B =>   write!(buf, "<B,{}>", arg_idx).unwrap(),
681                Matcher::H =>   write!(buf, "<H,{}>", arg_idx).unwrap(),
682                Matcher::S =>   write!(buf, "<S,{}>", arg_idx).unwrap(),
683                Matcher::D =>   write!(buf, "<D,{}>", arg_idx).unwrap(),
684                Matcher::Q =>   write!(buf, "<Q,{}>", arg_idx).unwrap(),
685                Matcher::V(s) => {
686                    let width = if i == 0 { 16 } else { 8 };
687                    write!(buf, "<V,{}>.{}{}", arg_idx, size_to_string(*s), width / s.in_bytes()).unwrap();
688                },
689                Matcher::VStatic(s, c) => write!(buf, "<V,{}>.{}{}", arg_idx, size_to_string(*s), c).unwrap(),
690                Matcher::VElement(s) => write!(buf, "<V,{}>.{}[<Imm,{}>]", arg_idx, size_to_string(*s), arg_idx + 1).unwrap(),
691                Matcher::VElementStatic(s, element) => write!(buf, "<V,{}>.{}[{}]", arg_idx, size_to_string(*s), element).unwrap(),
692                Matcher::VStaticElement(s, c) => write!(buf, "<V,{}>.{}{}[<Imm,{}>]", arg_idx, size_to_string(*s), c, arg_idx + 1).unwrap(),
693                Matcher::RegList(a, s) => {
694                    let width = if i == 0 { 16 } else { 8 };
695                    write!(buf, "{{<V,{}>.{}{} * {}}}", arg_idx, size_to_string(*s), width / s.in_bytes(), a).unwrap();
696                },
697                Matcher::RegListStatic(a, s, c) => write!(buf, "{{<V,{}>.{}{} * {}}}", arg_idx, size_to_string(*s), c, a).unwrap(),
698                Matcher::RegListElement(a, s) =>   write!(buf, "{{<V,{}>.{} * {}}}[<Imm,{}>]", arg_idx, size_to_string(*s), a, arg_idx + 1).unwrap(),
699                Matcher::Offset => write!(buf, "<Off,{}>", arg_idx).unwrap(),
700                Matcher::RefBase =>   write!(buf, "[<XSP,{}>]", arg_idx).unwrap(),
701                Matcher::RefOffset => write!(buf, "[<XSP,{}> <, <Imm,{}> > ]", arg_idx, arg_idx + 1).unwrap(),
702                Matcher::RefPre =>    write!(buf, "[<XSP,{}>, <Imm,{}>]!", arg_idx, arg_idx + 1).unwrap(),
703                Matcher::RefIndex => {
704                    constraints.push(format!("{}: ModWX()", arg_idx + 2));
705                    write!(buf, "[<XSP,{}>, <WX,{}> < , <Mod,{}> < <Imm,{}> > > ]", arg_idx, arg_idx + 1, arg_idx + 2, arg_idx + 3).unwrap();
706                },
707                Matcher::LitMod(m) => {
708                    buf.push_str(m.as_str());
709                    if !m.expr_required() {
710                        write!(buf, " {{ <Imm,{}> }}", arg_idx).unwrap();
711                    } else {
712                        write!(buf, " <Imm,{}>", arg_idx).unwrap();
713                    }
714                },
715                Matcher::Mod(mods) => {
716                    let mut required = false;
717                    let mut options = Vec::new();
718                    for m in *mods {
719                        required = required || m.expr_required();
720                        options.push(format!("\"{}\"", m.as_str()));
721                    }
722
723                    constraints.push(format!("{}: List({})", arg_idx, options.join(", ")));
724
725                    if !required {
726                        write!(buf, "<Mod,{}> < <Imm,{}> >", arg_idx, arg_idx + 1).unwrap();
727                    } else {
728                        write!(buf, "<Mod,{}> <Imm,{}>", arg_idx, arg_idx + 1).unwrap();
729                    }
730                },
731                Matcher::End => ()
732            }
733
734            arg_idx += matcher.flatarg_count();
735        }
736
737        for _ in 0 .. end_count {
738            buf.push_str(" >");
739        }
740
741        write!(buf, "\"\t {{{}}}", constraints.join(", ")).unwrap();
742
743        forms.push(buf);
744    }
745
746    forms
747}

749#[cfg(feature = "dynasm_extract")]
750fn extract_constraints(args: &[ArgWithCommands]) -> Vec<String> {
751    use super::aarch64data::{COND_MAP, SPECIAL_IDENT_MAP};
752
753    let mut constraints = Vec::new();
754    for (i, arg) in args.iter().enumerate() {
755        for command in &arg.commands {
756            let constraint = match command {
757                Command::R(_) => format!("R(32)"),
758                Command::REven(_) => format!("R(32, 2)"),
759                Command::RNoZr(_) => format!("R(31)"),
760                Command::R4(_) => format!("R(16)"),
761                Command::RNext => format!("RNext()"),
762                Command::Ubits(_, bits)
763                | Command::BUbits(bits) => format!("Range(0, {}, 1)", 1u32 << bits),
764                Command::Uscaled(_, bits, scale) => format!("Range(0, {}, {})", 1u32 << (bits + scale), 1u32 << scale),
765                Command::Ulist(_, list) => {
766                    let numbers = list.iter().map(|n| n.to_string()).collect::<Vec<_>>().join(", ");
767                    format!("List({})", numbers)
768                },
769                Command::Urange(_, min, max)
770                | Command::BUrange(min, max) => format!("Range({}, {}+1, 1)", min, max),
771                Command::Usub(_, bits, addval) => format!("Range({}, {}+1, 1)", *addval as u32 + 1 - (1u32 << bits), addval),
772                Command::Unegmod(_, bits) => format!("Range(0, {}, 1)", 1u32 << bits),
773                Command::Usumdec(_, bits)
774                | Command::BUsum(bits) => format!("Range2(1, {}+1, 1)", 1u32 << bits),
775                Command::Ufields(fields) => format!("Range(0, {}, 1)", 1u32 << fields.len()),
776                Command::Sbits(_, bits) => format!("Range(-{}, {}, 1)", 1u32 << (bits - 1), 1u32 << (bits - 1)),
777                Command::Sscaled(_, bits, scale)
778                | Command::BSscaled(bits, scale) => format!("Range(-{}, {}, {})", 1u32 << (bits + scale - 1), 1u32 << (bits + scale - 1), 1u32 << scale),
779                Command::Special(_, SpecialComm::WIDE_IMMEDIATE_W) => format!("Special('wide_w')"),
780                | Command::Special(_, SpecialComm::WIDE_IMMEDIATE_X) => format!("Special('wide_x')"),
781                | Command::Special(_, SpecialComm::INVERTED_WIDE_IMMEDIATE_W) => format!("Special('inverted_w')"),
782                | Command::Special(_, SpecialComm::INVERTED_WIDE_IMMEDIATE_X) => format!("Special('inverted_x')"),
783                Command::Special(_, SpecialComm::LOGICAL_IMMEDIATE_W) => format!("Special('logical_w')"),
784                | Command::Special(_, SpecialComm::LOGICAL_IMMEDIATE_X) => format!("Special('logical_x')"),
785                Command::Special(_, SpecialComm::FLOAT_IMMEDIATE)
786                | Command::Special(_, SpecialComm::SPLIT_FLOAT_IMMEDIATE) => format!("Special('float')"),
787                Command::Special(_, SpecialComm::STRETCHED_IMMEDIATE) => format!("Special('stretched')"),
788                Command::Offset(Relocation::B) => format!("Range(-{}, {}, {})", 1<<27, 1<<27, 4),
789                Command::Offset(Relocation::BCOND) => format!("Range(-{}, {}, {})", 1<<18, 1<<18, 4),
790                Command::Offset(Relocation::ADR) => format!("Range(-{}, {}, {})", 1<<20, 1<<20, 1),
791                Command::Offset(Relocation::ADRP) => format!("Range(-{}, {}, {})", 1u64<<32, 1u64<<32, 4096),
792                Command::Offset(Relocation::TBZ) => format!("Range(-{}, {}, {})", 1<<15, 1<<15, 4),
793                Command::Offset(Relocation::LITERAL32) => format!("Range(-{}, {}, {})", 1<<31, 1<<31, 1),
794                Command::Offset(Relocation::LITERAL64) => format!("Range(-{}, {}, {})", 1u64<<63, 1u64<<63, 1),
795                Command::Cond(_) => {
796                    let keys: Vec<_> = COND_MAP.keys().map(|k| format!("\"{}\"", k)).collect();
797                    format!("List({})", keys.join(", "))
798                },
799                Command::CondInv(_) => {
800                    let keys: Vec<_> = COND_MAP.iter().filter_map(|(k, v)| if *v < 14 { Some(format!("\"{}\"", k)) } else { None }).collect();
801                    format!("List({})", keys.join(", "))
802                },
803                Command::LitList(_, name) => {
804                    let keys: Vec<_> = SPECIAL_IDENT_MAP[name].keys().map(|k| format!("\"{}\"", k)).collect();
805                    format!("List({})", keys.join(", "))
806                }
807                _ => continue
808            };
809            constraints.push(format!("{}: {}", i, constraint));
810
811            break;
812        }
813    }
814    constraints
815}
