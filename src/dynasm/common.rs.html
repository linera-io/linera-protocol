12/// Enum representing the result size of a value/expression/register/etc in bytes.
13/// Uses the NASM syntax for sizes (a word is 16 bits)
14#[derive(Debug, PartialOrd, PartialEq, Ord, Eq, Hash, Clone, Copy)]
15pub enum Size {
16    BYTE  = 1,
17    WORD  = 2,
18    DWORD = 4,
19    FWORD = 6,
20    QWORD = 8,
21    PWORD = 10,
22    OWORD = 16,
23    HWORD = 32,
24}
25
26impl Size {
27    pub fn in_bytes(self) -> u8 {
28        self as u8
29    }
30
31    pub fn as_literal(self) -> syn::Ident {
32        syn::Ident::new(match self {
33            Size::BYTE  => "i8",
34            Size::WORD  => "i16",
35            Size::DWORD => "i32",
36            Size::FWORD => "i48",
37            Size::QWORD => "i64",
38            Size::PWORD => "i80",
39            Size::OWORD => "i128",
40            Size::HWORD => "i256"
41        }, Span::mixed_site())
42    }
43}

49#[derive(Debug, Clone)]
50pub struct Jump {
51    pub kind: JumpKind,
52    pub offset: Option<syn::Expr>
53}
54
55#[derive(Debug, Clone)]
56pub enum JumpKind {
57    // note: these symbol choices try to avoid stuff that is a valid starting symbol for parse_expr
58    // in order to allow the full range of expressions to be used. the only currently existing ambiguity is
59    // with the symbol <, as this symbol is also the starting symbol for the universal calling syntax <Type as Trait>.method(args)
60    Global(syn::Ident),   // -> label (["+" "-"] offset)?
61    Backward(syn::Ident), //  > label (["+" "-"] offset)?
62    Forward(syn::Ident),  //  < label (["+" "-"] offset)?
63    Dynamic(syn::Expr),   // =>expr | => (expr) (["+" "-"] offset)?
64    Bare(syn::Expr)       // jump to this address
65}
66
67impl ParseOpt for Jump {
68    fn parse(input: parse::ParseStream) -> parse::Result<Option<Jump>> {
69        // extern label
70        if eat_pseudo_keyword(input, "extern") {
71            let expr: syn::Expr = input.parse()?;
72
73            return Ok(Some(Jump { kind: JumpKind::Bare(expr), offset: None }));
74        }
75
76        // -> global_label
77        let kind = if input.peek(Token![->]) {
78            let _: Token![->] = input.parse()?;
79            let name: syn::Ident = input.parse()?;
80
81            JumpKind::Global(name)
82
83        // > forward_label
84        } else if input.peek(Token![>]) {
85            let _: Token![>] = input.parse()?;
86            let name: syn::Ident = input.parse()?;
87
88            JumpKind::Forward(name)
89
90        // < backwards_label
91        } else if input.peek(Token![<]) {
92            let _: Token![<] = input.parse()?;
93            let name: syn::Ident = input.parse()?;
94
95            JumpKind::Backward(name)
96
97        // => dynamic_label
98        } else if input.peek(Token![=>]) {
99            let _: Token![=>] = input.parse()?;
100
101            let expr: syn::Expr = if input.peek(syn::token::Paren) {
102                let inner;
103                let _ = syn::parenthesized!(inner in input);
104                let inner = &inner;
105
106                inner.parse()?
107            } else {
108                input.parse()?
109            };
110
111            JumpKind::Dynamic(expr)
112
113        // nothing
114        } else {
115            return Ok(None);
116        };
117
118        // parse optional offset
119        let offset = if input.peek(Token![-]) || input.peek(Token![+]) {
120            if input.peek(Token![+]) {
121                let _: Token![+] = input.parse()?;
122            }
123
124            let expr: syn::Expr = input.parse()?;
125            Some(expr)
126
127        } else {
128            None
129        };
130
131        Ok(Some(Jump::new(kind, offset)))
132    }
133}

135impl Jump {
136    pub fn new(kind: JumpKind, offset: Option<syn::Expr>) -> Jump {
137        Jump {
138            kind,
139            offset
140        }
141    }

143    /// Takes a jump and encodes it as a relocation starting `start_offset` bytes ago, relative to `ref_offset`.
144    /// Any data detailing the type of relocation emitted should be contained in `data`, which is emitted as a tuple of u8's.
145    pub fn encode(self, field_offset: u8, ref_offset: u8, data: &[u8]) -> Stmt {
146        let span = self.span();
147
148        let target_offset = delimited(if let Some(offset) = self.offset {
149            quote!(#offset)
150        } else {
151            quote!(0isize)
152        });
153
154        // Create a relocation descriptor, containing all information about the actual jump except for the target itself.
155        let relocation = Relocation {
156            target_offset,
157            field_offset,
158            ref_offset,
159            kind: serialize::expr_tuple_of_u8s(span, data)
160        };
161        match self.kind {
162            JumpKind::Global(ident) => Stmt::GlobalJumpTarget(ident, relocation),
163            JumpKind::Backward(ident) => Stmt::BackwardJumpTarget(ident, relocation),
164            JumpKind::Forward(ident) => Stmt::ForwardJumpTarget(ident, relocation),
165            JumpKind::Dynamic(expr) => Stmt::DynamicJumpTarget(delimited(expr), relocation),
166            JumpKind::Bare(expr) => Stmt::BareJumpTarget(delimited(expr), relocation),
167        }
168    }
169
170    pub fn span(&self) -> Span {
171        match &self.kind {
172            JumpKind::Global(ident) => ident.span(),
173            JumpKind::Backward(ident) => ident.span(),
174            JumpKind::Forward(ident) => ident.span(),
175            JumpKind::Dynamic(expr) => expr.span(),
176            JumpKind::Bare(expr) => expr.span(),
177        }
178    }
179}

192/// An abstract representation of a dynasm runtime statement to be emitted
193#[derive(Debug, Clone)]
194pub enum Stmt {
195    // simply push data into the instruction stream. unsigned
196    Const(u64, Size),
197    // push data that is stored inside of an expression. unsigned
198    ExprUnsigned(TokenTree, Size),
199    // push signed data into the instruction stream. signed
200    ExprSigned(TokenTree, Size),
201
202    // extend the instruction stream with unsigned bytes
203    Extend(Vec<u8>),
204    // extend the instruction stream with unsigned bytes
205    ExprExtend(TokenTree),
206    // align the instruction stream to some alignment
207    Align(TokenTree, TokenTree),
208
209    // label declarations
210    GlobalLabel(syn::Ident),
211    LocalLabel(syn::Ident),
212    DynamicLabel(TokenTree),
213
214    // and their respective relocations (as expressions as they differ per assembler).
215    GlobalJumpTarget(syn::Ident, Relocation),
216    ForwardJumpTarget(syn::Ident, Relocation),
217    BackwardJumpTarget(syn::Ident, Relocation),
218    DynamicJumpTarget(TokenTree, Relocation),
219    BareJumpTarget(TokenTree, Relocation),
220
221    // a statement that provides some information for the next statement,
222    // and should therefore not be reordered with it
223    PrefixStmt(TokenTree),
224
225    // a random statement that has to be inserted between assembly hunks
226    Stmt(TokenTree)
227}
228
229// convenience methods
230impl Stmt {
231    #![allow(dead_code)]
232
233    pub fn u8(value: u8) -> Stmt {
234        Stmt::Const(u64::from(value), Size::BYTE)
235    }
236
237    pub fn u16(value: u16) -> Stmt {
238        Stmt::Const(u64::from(value), Size::WORD)
239    }
240
241    pub fn u32(value: u32) -> Stmt {
242        Stmt::Const(u64::from(value), Size::DWORD)
243    }
244
245    pub fn u64(value: u64) -> Stmt {
246        Stmt::Const(value, Size::QWORD)
247    }
248}

251// Makes a None-delimited TokenTree item out of anything that can be converted to tokens.
252// This is a useful shortcut to escape issues around not-properly delimited tokenstreams
253// because it is guaranteed to be parsed back properly to its source ast at type-level.
254pub fn delimited<T: ToTokens>(expr: T) -> TokenTree {
255    let span = expr.span();
256    let mut group = proc_macro2::Group::new(
257        proc_macro2::Delimiter::None, expr.into_token_stream()
258    );
259    group.set_span(span);
260    proc_macro2::TokenTree::Group(group)
261}

264/// Create a bitmask with `scale` bits set
265pub fn bitmask(scale: u8) -> u32 {
266    1u32.checked_shl(u32::from(scale)).unwrap_or(0).wrapping_sub(1)
267}

270/// Create a bitmask with `scale` bits set
