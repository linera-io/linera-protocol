17/// An ID, which allows to differentiate states received from a Channel.
18/// Elements with a bigger state ID (`id > otherId`) have been published more
19/// recently into the Channel.
20#[derive(Copy, Clone, Debug, Default, PartialEq, Eq, Ord, PartialOrd)]
21pub struct StateId(u64);
22
23impl StateId {
24    /// Returns the initial StateId, which is guaranteed to return the
25    /// oldest buffered value available.
26    pub fn new() -> Self {
27        StateId(0)
28    }
29}

31/// Tracks how the future had interacted with the channel
32#[derive(PartialEq, Debug)]
33pub enum RecvPollState {
34    /// The task is not registered at the wait queue at the channel
35    Unregistered,
36    /// The task was added to the wait queue at the channel.
37    Registered,
38}

40/// Tracks the channel futures waiting state.
41/// Access to this struct is synchronized through the channel.
42#[derive(Debug)]
43pub struct RecvWaitQueueEntry {
44    /// The task handle of the waiting task
45    task: Option<Waker>,
46    /// Current polling state
47    state: RecvPollState,
48    /// The minimum state ID we are waiting for
49    state_id: StateId,
50}
51
52impl RecvWaitQueueEntry {
53    /// Creates a new RecvWaitQueueEntry
54    pub fn new(state_id: StateId) -> RecvWaitQueueEntry {
55        RecvWaitQueueEntry {
56            task: None,
57            state_id,
58            state: RecvPollState::Unregistered,
59        }
60    }

63/// Adapter trait that allows Futures to generically interact with Channel
64/// implementations via dynamic dispatch.
65pub trait ChannelReceiveAccess<T> {
66    unsafe fn receive_or_register(
67        &self,
68        wait_node: &mut ListNode<RecvWaitQueueEntry>,
69        cx: &mut Context<'_>,
70    ) -> Poll<Option<(StateId, T)>>;
71
72    fn remove_receive_waiter(
73        &self,
74        wait_node: &mut ListNode<RecvWaitQueueEntry>,
75    );
76}

78/// A Future that is returned by the `receive` function on a state broadcast channel.
79/// The future gets resolved with `Some((state_id, state))` when a value could be
80/// received from the channel.
81///
82/// `state` represents the new state which had been retrieved from the channel.
83///
84/// `state_id` is the [`StateId`] which can be passed as a parameter to
85/// `receive()` in order to fetch the next state from the channel.
86///
87/// If the channels gets closed and no items are still enqueued inside the
88/// channel, the future will resolve to `None`.
89#[must_use = "futures do nothing unless polled"]
90pub struct StateReceiveFuture<'a, MutexType, T>
91where
92    T: Clone,
93{
94    /// The channel that is associated with this StateReceiveFuture
95    channel: Option<&'a dyn ChannelReceiveAccess<T>>,
96    /// Node for waiting on the channel
97    wait_node: ListNode<RecvWaitQueueEntry>,
98    /// Marker for mutex type
99    _phantom: PhantomData<MutexType>,
100}
101
102// Safety: Channel futures can be sent between threads as long as the underlying
103// channel is thread-safe (Sync), which allows to poll/register/unregister from
104// a different thread.
105unsafe impl<'a, MutexType: Sync, T: Clone + Send> Send
106    for StateReceiveFuture<'a, MutexType, T>
107{
108}

118impl<'a, MutexType, T: Clone> Future for StateReceiveFuture<'a, MutexType, T> {
119    type Output = Option<(StateId, T)>;
120
121    fn poll(
122        self: Pin<&mut Self>,
123        cx: &mut Context<'_>,
124    ) -> Poll<Option<(StateId, T)>> {
125        // It might be possible to use Pin::map_unchecked here instead of the two unsafe APIs.
126        // However this didn't seem to work for some borrow checker reasons
127
128        // Safety: The next operations are safe, because Pin promises us that
129        // the address of the wait queue entry inside StateReceiveFuture is stable,
130        // and we don't move any fields inside the future until it gets dropped.
131        let mut_self: &mut StateReceiveFuture<MutexType, T> =
132            unsafe { Pin::get_unchecked_mut(self) };
133
134        let channel = mut_self
135            .channel
136            .expect("polled StateReceiveFuture after completion");
137
138        let poll_res =
139            unsafe { channel.receive_or_register(&mut mut_self.wait_node, cx) };
140
141        if poll_res.is_ready() {
142            // A value was available
143            mut_self.channel = None;
144        }
145
146        poll_res
147    }
148}

158impl<'a, MutexType, T: Clone> Drop for StateReceiveFuture<'a, MutexType, T> {
159    fn drop(&mut self) {
160        // If this StateReceiveFuture has been polled and it was added to the
161        // wait queue at the channel, it must be removed before dropping.
162        // Otherwise the channel would access invalid memory.
163        if let Some(channel) = self.channel {
164            channel.remove_receive_waiter(&mut self.wait_node);
165        }
166    }
167}

169fn wake_waiters(waiters: &mut LinkedList<RecvWaitQueueEntry>) {
170    // Remove all waiters from the waiting list in reverse order and wake them.
171    // We reverse the waiter list, so that the oldest waker (which is
172    // at the end of the list), gets woken first and has the best
173    // chance to grab the channel value.
174    waiters.reverse_drain(|waiter| {
175        if let Some(handle) = waiter.task.take() {
176            handle.wake();
177        }
178        waiter.state = RecvPollState::Unregistered;
179    });
180}

182/// Internal state of the state broadcast channel
183struct ChannelState<T> {
184    /// Whether the channel was actively closed
185    is_closed: bool,
186    /// The ID of the next state.
187    state_id: StateId,
188    /// The value which is stored inside the channel
189    value: Option<T>,
190    /// The list of waiters, which are waiting for the channel to get fulfilled
191    waiters: LinkedList<RecvWaitQueueEntry>,
192}
193
194impl<T> ChannelState<T>
195where
196    T: Clone,
197{
198    fn new() -> ChannelState<T> {
199        ChannelState::<T> {
200            is_closed: false,
201            state_id: StateId(0),
202            value: None,
203            waiters: LinkedList::new(),
204        }
205    }
206
207    /// Writes a single value to the channel.
208    /// If the maximum amount of values had been written, the new value will be rejected.
209    fn send(&mut self, value: T) -> Result<(), ChannelSendError<T>> {
210        if self.is_closed || self.state_id.0 == core::u64::MAX {
211            return Err(ChannelSendError(value));
212        }
213
214        self.value = Some(value);
215        self.state_id.0 += 1;
216
217        // Wakeup all waiters
218        wake_waiters(&mut self.waiters);
219
220        Ok(())
221    }
222
223    fn close(&mut self) -> CloseStatus {
224        if self.is_closed {
225            return CloseStatus::AlreadyClosed;
226        }
227        self.is_closed = true;
228
229        // Wakeup all waiters
230        wake_waiters(&mut self.waiters);
231
232        CloseStatus::NewlyClosed
233    }
234
235    fn try_receive(&mut self, state_id: StateId) -> Option<(StateId, T)> {
236        let val = self.value.as_ref()?;
237        if state_id < self.state_id {
238            Some((self.state_id, val.clone()))
239        } else {
240            None
241        }
242    }
243
244    /// Tries to read the value from the channel.
245    /// If the value isn't available yet, the StateReceiveFuture gets added to the
246    /// wait queue at the channel, and will be signalled once ready.
247    /// This function is only safe as long as the `wait_node`s address is guaranteed
248    /// to be stable until it gets removed from the queue.
249    unsafe fn receive_or_register(
250        &mut self,
251        wait_node: &mut ListNode<RecvWaitQueueEntry>,
252        cx: &mut Context<'_>,
253    ) -> Poll<Option<(StateId, T)>> {
254        match wait_node.state {
255            RecvPollState::Unregistered => {
256                // The caller must wait for a value if either there is no value
257                // available yet, or if the value isn't newer than what the
258                // caller requested.
259                let val_to_deliver = match &self.value {
260                    Some(ref v) if wait_node.state_id < self.state_id => {
261                        Some(v.clone())
262                    }
263                    Some(_) | None => None,
264                };
265
266                match val_to_deliver {
267                    Some(v) => {
268                        // A value that satisfies the caller is available.
269                        Poll::Ready(Some((self.state_id, v)))
270                    }
271                    None => {
272                        // Check if something was written into the channel before
273                        // or the channel was closed.
274                        if self.is_closed {
275                            Poll::Ready(None)
276                        } else {
277                            // Added the task to the wait queue
278                            wait_node.task = Some(cx.waker().clone());
279                            wait_node.state = RecvPollState::Registered;
280                            self.waiters.add_front(wait_node);
281                            Poll::Pending
282                        }
283                    }
284                }
285            }
286            RecvPollState::Registered => {
287                // Since the channel wakes up all waiters and moves their states
288                // to unregistered there can't be any value in the channel in this state.
289                // However the caller might have passed a different `Waker`.
290                // In this case we need to update it.
291                update_waker_ref(&mut wait_node.task, cx);
292                Poll::Pending
293            }
294        }
295    }
296
297    fn remove_waiter(&mut self, wait_node: &mut ListNode<RecvWaitQueueEntry>) {
298        // StateReceiveFuture only needs to get removed if it had been added to
299        // the wait queue of the channel. This has happened in the RecvPollState::Waiting case.
300        if let RecvPollState::Registered = wait_node.state {
301            // Safety: Due to the state, we know that the node must be part
302            // of the waiter list
303            if !unsafe { self.waiters.remove(wait_node) } {
304                // Panic if the address isn't found. This can only happen if the contract was
305                // violated, e.g. the RecvWaitQueueEntry got moved after the initial poll.
306                panic!("Future could not be removed from wait queue");
307            }
308            wait_node.state = RecvPollState::Unregistered;
309        }
310    }
311}

313/// A channel which can be used to synchronize the state between a sender an
314/// arbitrary number of receivers.
315///
316/// The sender can publish its state.
317///
318/// The receivers can wait for state updates by announcing the most recent state
319/// that is already known to them.
320pub struct GenericStateBroadcastChannel<MutexType: RawMutex, T> {
321    inner: Mutex<MutexType, ChannelState<T>>,
322}
323
324// The channel can be sent to other threads as long as it's not borrowed and the
325// value in it can be sent to other threads.
326unsafe impl<MutexType: RawMutex + Send, T: Send> Send
327    for GenericStateBroadcastChannel<MutexType, T>
328{
329}
330// The channel is thread-safe as long as a thread-safe mutex is used
331unsafe impl<MutexType: RawMutex + Sync, T: Send> Sync
332    for GenericStateBroadcastChannel<MutexType, T>

334}

344impl<MutexType: RawMutex, T> GenericStateBroadcastChannel<MutexType, T>
345where
346    T: Clone,
347{
348    /// Creates a new State Broadcast Channel in the given state
349    pub fn new() -> GenericStateBroadcastChannel<MutexType, T>
350    where
351        T: Clone,
352    {
353        GenericStateBroadcastChannel {
354            inner: Mutex::new(ChannelState::new()),
355        }
356    }
357
358    /// Writes a single value to the channel.
359    ///
360    /// This will notify waiters about the availability of the value.
361    /// If the maximum amount of values had been written to the channel,
362    /// or if the channel is closed, the new value will be rejected and
363    /// returned inside the error variant.
364    pub fn send(&self, value: T) -> Result<(), ChannelSendError<T>> {
365        self.inner.lock().send(value)
366    }
367
368    /// Closes the channel.
369    ///
370    /// This will notify waiters about closure, by fulfilling pending `Future`s
371    /// with `None`.
372    /// `send(value)` attempts which follow this call will fail with a
373    /// [`ChannelSendError`].
374    pub fn close(&self) -> CloseStatus {
375        self.inner.lock().close()
376    }
377
378    /// Returns a future that gets fulfilled when a value is written to the channel
379    /// or the channel is closed.
380    /// `state_id` specifies the minimum state ID that should be retrieved
381    /// by the `receive` operation.
382    ///
383    /// The returned [`StateReceiveFuture`] will get fulfilled with the
384    /// retrieved value as well as the [`StateId`] which is required to retrieve
385    /// the following state.
386    pub fn receive(
387        &self,
388        state_id: StateId,
389    ) -> StateReceiveFuture<MutexType, T> {
390        StateReceiveFuture {
391            channel: Some(self),
392            wait_node: ListNode::new(RecvWaitQueueEntry::new(state_id)),
393            _phantom: PhantomData,
394        }
395    }
396
397    /// Attempt to retrieve a value whose `StateId` is greater than the one provided.
398    ///
399    /// Returns `None` if no value is found in the channel, or if the current `StateId`
400    /// of the value is less or equal to the one provided.
401    pub fn try_receive(&self, state_id: StateId) -> Option<(StateId, T)> {
402        self.inner.lock().try_receive(state_id)
403    }
404}

700        pub fn generic_state_broadcast_channel<MutexType, T>() -> (
701            GenericStateSender<MutexType, T>,
702            GenericStateReceiver<MutexType, T>,
703        )
704        where
705            MutexType: RawMutex,
706            T: Clone + Send,
707        {
708            let inner = alloc::sync::Arc::new(
709                GenericStateBroadcastChannelSharedState {
710                    channel: GenericStateBroadcastChannel::new(),
711                    senders: AtomicUsize::new(1),
712                    receivers: AtomicUsize::new(1),
713                },
714            );
715
716            let sender = GenericStateSender {
717                inner: inner.clone(),
718            };
719            let receiver = GenericStateReceiver { inner };
720
721            (sender, receiver)
722        }
723
724        impl<MutexType, T> GenericStateSender<MutexType, T>
725        where
726            MutexType: RawMutex + 'static,
727            T: Clone,
728        {
729            /// Writes a single value to the channel.
730            ///
731            /// This will notify waiters about the availability of the value.
732            /// If a value had been written to the channel before, or if the
733            /// channel is closed, the new value will be rejected and
734            /// returned inside the error variant.
735            pub fn send(&self, value: T) -> Result<(), ChannelSendError<T>> {
736                self.inner.channel.send(value)
737            }
738        }
739
740        impl<MutexType, T> GenericStateReceiver<MutexType, T>
741        where
742            MutexType: RawMutex + 'static,
743            T: Clone,
744        {
745            /// Returns a future that gets fulfilled when a value is written to the channel
746            /// or the channel is closed.
747            /// `state_id` specifies the minimum state ID that should be retrieved
748            /// by the `receive` operation.
749            ///
750            /// The returned [`StateReceiveFuture`] will get fulfilled with the
751            /// retrieved value as well as the [`StateId`] which is required to retrieve
752            /// the following state
753            pub fn receive(
754                &self,
755                state_id: StateId,
756            ) -> StateReceiveFuture<MutexType, T> {
757                StateReceiveFuture {
758                    channel: Some(self.inner.clone()),
759                    wait_node: ListNode::new(RecvWaitQueueEntry::new(state_id)),
760                    _phantom: PhantomData,
761                }
762            }
763
764            /// Attempt to retrieve a value whose `StateId` is greater than the one provided.
765            ///
766            /// Returns `None` if no value is found in the channel, or if the current `StateId`
767            /// of the value is less or equal to the one provided.
768            pub fn try_receive(
769                &self,
770                state_id: StateId,
771            ) -> Option<(StateId, T)> {
772                self.inner.channel.try_receive(state_id)
773            }
774        }
