7pub(crate) trait RetainExt {
8    type Item<'a>;
9
10    fn retain<F>(&mut self, f: F) -> &mut Self
11    where
12        F: for<'a> FnMut(Self::Item<'a>) -> bool;
13
14    fn try_retain<F>(&mut self, mut f: F) -> Result<&mut Self>
15    where
16        F: for<'a> FnMut(Self::Item<'a>) -> Result<bool>,
17    {
18        let mut filter_err = None;
19
20        let result = self.retain(|value| {
21            if filter_err.is_some() {
22                return false;
23            }
24
25            f(value).unwrap_or_else(|err| {
26                filter_err = Some(err);
27                false
28            })
29        });
30
31        filter_err.map(Err).unwrap_or(Ok(result))
32    }
33}
34
35impl RetainExt for Structure<'_> {
36    type Item<'a> = &'a BindingInfo<'a>;
37
38    fn retain<F>(&mut self, mut f: F) -> &mut Self
39    where
40        F: for<'a> FnMut(&'a BindingInfo<'a>) -> bool,
41    {
42        self.filter(|value| f(value))
43    }
44}
