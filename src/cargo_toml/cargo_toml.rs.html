4//! This crate defines `struct`s that can be deserialized with Serde
5//! to load and inspect `Cargo.toml` metadata.
6//!
7//! See [`Manifest::from_slice`].
8//!
9//! Correct interpretation of the manifest requires two things:
10//!
11//! * List of files in order to auto-discover binaries, examples, benchmarks, and tests.
12//! * Potentially `Manifest` from parent directories that acts as a workspace root for inheritance of shared workspace information.
13//!
14//! The crate has methods for processing this information, but you will need to write some glue code to obtain it. See [`Manifest::complete_from_path_and_workspace`].

290    /// Parse `Cargo.toml`, and parse its `[package.metadata]` into a custom Serde-compatible type.
291    ///
292    /// It does not call [`Manifest::complete_from_path`], so may be missing implicit data.
293    #[inline]
294    pub fn from_slice_with_metadata(cargo_toml_content: &[u8]) -> Result<Self, Error> {
295        let cargo_toml_content = std::str::from_utf8(cargo_toml_content).map_err(|_| Error::Other("utf8"))?;
296        Self::from_slice_with_metadata_str(cargo_toml_content)
297    }
298
299    #[inline(never)]
300    fn from_slice_with_metadata_str(cargo_toml_content: &str) -> Result<Self, Error> {
301        let mut manifest: Self = toml::from_str(cargo_toml_content)?;
302
303        if let Some(package) = &mut manifest.package {
304            // This is a clumsy implementation of Cargo's rule that missing version defaults publish to false.
305            // Serde just doesn't support such relationship for default field values, so this will be incorrect
306            // for explicit `version = "0.0.0"` and `publish = true`.
307            if package.version.get().map_or(false, |v| v == "0.0.0") && package.publish.get().map_or(false, |p| p.is_default()) {
308                package.publish = Inheritable::Set(Publish::Flag(false));
309            }
310        }
311        Ok(manifest)
312    }
313
314    /// Parse contents from `Cargo.toml` file on disk, with custom Serde-compatible metadata type.
315    ///
316    /// Calls [`Manifest::complete_from_path`]
317    pub fn from_path_with_metadata<P: AsRef<Path>>(cargo_toml_path: P) -> Result<Self, Error> {
318        let cargo_toml_path = cargo_toml_path.as_ref();
319        let cargo_toml_content = fs::read_to_string(cargo_toml_path)?;
320        let mut manifest = Self::from_slice_with_metadata_str(&cargo_toml_content)?;
321        manifest.complete_from_path(cargo_toml_path)?;
322        Ok(manifest)
323    }

327    /// `Cargo.toml` doesn't contain explicit information about `[lib]` and `[[bin]]`,
328    /// which are inferred based on files on disk.
329    ///
330    /// This scans the disk to make the data in the manifest as complete as possible.
331    ///
332    /// It supports workspace inheritance and will search for a root workspace.
333    /// Use [`Manifest::complete_from_path_and_workspace`] to provide the workspace explicitly.
334    pub fn complete_from_path(&mut self, path: &Path) -> Result<(), Error> {
335        let manifest_dir = path.parent().ok_or(Error::Other("bad path"))?;
336        self.complete_from_abstract_filesystem::<Value, _>(Filesystem::new(manifest_dir), None)
337    }

339    /// [`Manifest::complete_from_path`], but allows passing workspace manifest explicitly.
340    ///
341    /// `workspace_manifest_and_path` is the root workspace manifest already parsed,
342    /// and the path is the path to the root workspace's directory.
343    /// If it's `None`, the root workspace will be discovered automatically.
344    pub fn complete_from_path_and_workspace<PackageMetadataTypeDoesNotMatterHere>(&mut self, package_manifest_path: &Path, workspace_manifest_and_path: Option<(&Manifest<PackageMetadataTypeDoesNotMatterHere>, &Path)>) -> Result<(), Error> {
345        let manifest_dir = package_manifest_path.parent().ok_or(Error::Other("bad path"))?;
346        self.complete_from_abstract_filesystem(Filesystem::new(manifest_dir), workspace_manifest_and_path)
347    }

349    /// `Cargo.toml` doesn't contain explicit information about `[lib]` and `[[bin]]`,
350    /// which are inferred based on files on disk.
351    ///
352    /// You can provide any implementation of directory scan, which doesn't have to
353    /// be reading straight from disk (might scan a tarball or a git repo, for example).
354    ///
355    /// If `workspace_manifest_and_path` is set, it will inherit from this workspace.
356    /// If it's `None`, it will try to find a workspace if needed.
357    pub fn complete_from_abstract_filesystem<PackageMetadataTypeDoesNotMatterHere, Fs: AbstractFilesystem>(
358        &mut self, fs: Fs, workspace_manifest_and_path: Option<(&Manifest<PackageMetadataTypeDoesNotMatterHere>, &Path)>
359    ) -> Result<(), Error> {
360        if let Some((ws, ws_path)) = workspace_manifest_and_path {
361            self._inherit_workspace(ws.workspace.as_ref(), ws_path)?;
362        } else if let Some(ws) = self.workspace.take() {
363            // Manifest may be both a workspace and a package
364            self._inherit_workspace(Some(&ws), Path::new(""))?;
365            self.workspace = Some(ws);
366        } else if self.needs_workspace_inheritance() {
367            let (ws_manifest, base_path) = fs.parse_root_workspace(self.package.as_ref().and_then(|p| p.workspace.as_deref()))?;
368            self._inherit_workspace(ws_manifest.workspace.as_ref(), &base_path)?;
369        }
370        self.complete_from_abstract_filesystem_inner(&fs)
371    }

373    /// If `true`, some fields are unavailable. If `false`, it's fully usable as-is.
374    ///
375    /// It is `false` in manifests that use workspace inheritance, but had their data completed from the root manifest already.
376    pub fn needs_workspace_inheritance(&self) -> bool {
377        self.package.as_ref().map_or(false, Package::needs_workspace_inheritance) ||
378        self.dependencies.values()
379            .chain(self.build_dependencies.values())
380            .chain(self.dev_dependencies.values())
381            .any(|dep| {
382                matches!(dep, Dependency::Inherited(_))
383            })
384    }

395    fn _inherit_workspace<Ignored>(&mut self, workspace: Option<&Workspace<Ignored>>, workspace_base_path: &Path) -> Result<(), Error> {
396        let workspace_base_path = if workspace_base_path.file_name() == Some("Cargo.toml".as_ref()) {
397            workspace_base_path.parent().ok_or(Error::Other("bad path"))?
398        } else {
399            workspace_base_path
400        };
401
402        inherit_dependencies(&mut self.dependencies, workspace, workspace_base_path)?;
403        inherit_dependencies(&mut self.build_dependencies, workspace, workspace_base_path)?;
404        inherit_dependencies(&mut self.dev_dependencies, workspace, workspace_base_path)?;
405
406        let package = match &mut self.package {
407            Some(p) => p,
408            None => return Ok(()),
409        };
410        if let Some(ws) = workspace.and_then(|w| w.package.as_ref()) {
411            Self::inherit_package_properties(package, ws, workspace_base_path)?;
412        }
413
414        if package.needs_workspace_inheritance() {
415            return Err(Error::WorkspaceIntegrity(format!("not all fields of `{}` have been present in workspace.package", package.name())));
416        }
417        Ok(())
418    }

420    fn inherit_package_properties(package: &mut Package<Metadata>, ws: &PackageTemplate, workspace_base_path: &Path) -> Result<(), Error> {
421        fn maybe_inherit<T: Clone>(to: Option<&mut Inheritable<T>>, from: Option<&T>) {
422            if let Some(from) = from {
423                if let Some(to) = to {
424                    to.inherit(from);
425                }
426            }
427        }
428        fn inherit<T: Clone>(to: &mut Inheritable<T>, from: Option<&T>) {
429            if let Some(from) = from {
430                to.inherit(from);
431            }
432        }
433        inherit(&mut package.authors, ws.authors.as_ref());
434        inherit(&mut package.categories, ws.categories.as_ref());
435        inherit(&mut package.edition, ws.edition.as_ref());
436        inherit(&mut package.exclude, ws.exclude.as_ref());
437        inherit(&mut package.include, ws.include.as_ref());
438        inherit(&mut package.keywords, ws.keywords.as_ref());
439        inherit(&mut package.version, ws.version.as_ref());
440        maybe_inherit(package.description.as_mut(), ws.description.as_ref());
441        maybe_inherit(package.documentation.as_mut(), ws.documentation.as_ref());
442        maybe_inherit(package.homepage.as_mut(), ws.homepage.as_ref());
443        maybe_inherit(package.license.as_mut(), ws.license.as_ref());
444        maybe_inherit(package.repository.as_mut(), ws.repository.as_ref());
445        maybe_inherit(package.rust_version.as_mut(), ws.rust_version.as_ref());
446        package.publish.inherit(&ws.publish);
447        match (&mut package.readme, &ws.readme) {
448            (r @ Inheritable::Inherited { .. }, flag @ OptionalFile::Flag(_)) => {
449                r.set(flag.clone());
450            },
451            (r @ Inheritable::Inherited { .. }, OptionalFile::Path(path)) => {
452                r.set(OptionalFile::Path(workspace_base_path.join(path)));
453            },
454            _ => {},
455        }
456        match (package.license_file.as_mut(), ws.license_file.as_ref()) {
457            (Some(f), Some(ws)) => f.set(workspace_base_path.join(ws)),
458            _ => {}
459        };
460        Ok(())
461    }

464    fn complete_from_abstract_filesystem_inner(&mut self, fs: &dyn AbstractFilesystem) -> Result<(), Error> {
465        let Some(package) = &self.package else { return Ok(()) };
466
467        let src = match fs.file_names_in("src") {
468            Ok(src) => src,
469            Err(err) if err.kind() == io::ErrorKind::NotFound => Default::default(),
470            Err(err) => return Err(err.into()),
471        };
472
473        if let Some(ref mut lib) = self.lib {
474            lib.required_features.clear(); // not applicable
475        }
476
477        let has_path = self.lib.as_ref().map_or(false, |l| l.path.is_some());
478        if !has_path && src.contains("lib.rs") {
479            let old_lib = self.lib.take().unwrap_or_default();
480            self.lib = Some(Product {
481                name: if let Some(name) = old_lib.name { Some(name)} else {Some(package.name.replace('-', "_"))},
482                path: Some("src/lib.rs".to_string()),
483                edition: *package.edition.get()?,
484                crate_type: vec!["rlib".to_string()],
485                ..old_lib
486            });
487        }
488
489        if package.autobins {
490            let mut bin = take(&mut self.bin);
491            let (fully_overrided, mut partial_overrided) = self.autoset(&mut bin, "src/bin", fs)?;
492            self.bin = bin;
493
494            if src.contains("main.rs") && !fully_overrided.contains("src/main.rs") {
495                let rel_path = "src/main.rs".to_string();
496                let name = &package.name;
497
498                let product = if let Some(mut product) = partial_overrided.remove(name) {
499                    product.path = Some(rel_path);
500                    product
501                } else {
502                    Product {
503                        name: Some(name.clone()),
504                        path: Some(rel_path),
505                        edition: *package.edition.get()?,
506                        ..Product::default()
507                    }
508                };
509                self.bin.push(product);
510            }
511        }
512
513        Self::sort_products(&mut self.bin);
514
515        if package.autoexamples {
516            let mut example = take(&mut self.example);
517            self.autoset(&mut example, "examples", fs)?;
518            self.example = example;
519        }
520
521        Self::sort_products(&mut self.example);
522
523        if package.autotests {
524            let mut test = take(&mut self.test);
525            self.autoset(&mut test, "tests", fs)?;
526            self.test = test;
527        }
528
529        Self::sort_products(&mut self.test);
530
531        if package.autobenches {
532            let mut bench = take(&mut self.bench);
533            self.autoset(&mut bench, "benches", fs)?;
534            self.bench = bench;
535        }
536
537        Self::sort_products(&mut self.bench);
538
539        let Some(package) = &mut self.package else { return Ok(()) };
540
541        if matches!(package.build, None | Some(OptionalFile::Flag(true))) && fs.file_names_in(".").map_or(false, |dir| dir.contains("build.rs")) {
542            package.build = Some(OptionalFile::Path("build.rs".into()));
543        }
544
545        if matches!(package.readme.get()?, OptionalFile::Flag(true)) {
546            let files = fs.file_names_in(".").ok();
547            if let Some(name) = files.as_ref().and_then(|dir| {
548                dir.get("README.md")
549                    .or_else(|| dir.get("README.txt"))
550                    .or_else(|| dir.get("README"))
551            }) {
552                package.readme = Inheritable::Set(OptionalFile::Path(PathBuf::from(&**name)));
553            }
554        }
555        Ok(())
556    }

562        dir: &str,
563        fs: &dyn AbstractFilesystem,
564    ) -> Result<(BTreeSet<String>, BTreeMap<String, Product>), Error> {
565        let fully_overrided: BTreeSet<_> = out.iter()
566            .filter_map(|product| product.path.clone())
567            .collect();
568
569        let mut partial_overrided: BTreeMap<String, Product> = out.iter()
570            .filter_map(|product| {
571                match (&product.path, &product.name)  {
572                    (None, Some(name)) => {
573                        Some((name.clone(), product.clone()))
574                    },
575                    _ => None
576                }
577            })
578            .collect();
579
580        // Remove partially overrided items
581        out.retain(|product| product.path.is_some());
582
583        if let Some(ref package) = self.package {
584            if let Ok(bins) = fs.file_names_in(dir) {
585                for name in bins {
586                    let rel_path = format!("{dir}/{name}");
587
588                    if name.ends_with(".rs") {
589                        if !fully_overrided.contains(&rel_path) {
590                            let name = name.trim_end_matches(".rs");
591
592                            let product = if let Some(mut product) = partial_overrided.remove(name) {
593                                product.path = Some(rel_path);
594                                product
595                            } else {
596                                Product {
597                                    name: Some(name.to_string()),
598                                    path: Some(rel_path),
599                                    edition: *package.edition.get()?,
600                                    ..Product::default()
601                                }
602                            };
603                            out.push(product);
604                        }
605                    } else if let Ok(sub) = fs.file_names_in(&rel_path) {
606                        let rel_path = format!("{rel_path}/main.rs");
607
608                        if sub.contains("main.rs") && !fully_overrided.contains(&rel_path) {
609                            let product = if let Some(mut product) = partial_overrided.remove(&*name) {
610                                product.path = Some(rel_path);
611                                product
612                            } else {
613                                Product {
614                                    name: Some(name.into()),
615                                    path: Some(rel_path),
616                                    edition: *package.edition.get()?,
617                                    ..Product::default()
618                                }
619                            };
620                            out.push(product);
621                        }
622                    }
623                }
624            }
625        }
626        Ok((fully_overrided, partial_overrided))
627    }
628
629    /// ensure bins are deterministic
630    fn sort_products(products: &mut [Product]) {
631        products.sort_unstable_by(|a, b| a.name.cmp(&b.name).then(a.path.cmp(&b.path)));
632    }
633
634    /// Panics if it's not a package (only a workspace).
635    ///
636    /// You can access `.package` field directly to handle the `Option`.
637    #[track_caller]
638    #[inline]
639    pub fn package(&self) -> &Package<Metadata> {
640        self.package.as_ref().expect("not a package")
641    }
642}
643
644fn inherit_dependencies<Ignored>(deps_to_inherit: &mut BTreeMap<String, Dependency>, workspace: Option<&Workspace<Ignored>>, workspace_base_path: &Path) -> Result<(), Error> {
645    for (key, dep) in deps_to_inherit {
646        if let Dependency::Inherited(overrides) = dep {
647            let template = workspace.and_then(|ws| ws.dependencies.get(key))
648                .ok_or_else(|| Error::WorkspaceIntegrity(format!("workspace dependencies are missing `{key}`")))?;
649            let mut overrides = overrides.clone();
650            *dep = template.clone();
651            if overrides.optional {
652                dep.try_detail_mut()?.optional = true;
653            }
654            if !overrides.features.is_empty() {
655                dep.try_detail_mut()?.features.append(&mut overrides.features);
656            }
657            if let Dependency::Detailed(dep) = dep {
658                dep.inherited = true;
659                if let Some(path) = &mut dep.path {
660                    *path = workspace_base_path.join(&path).display().to_string();
661                }
662            }
663        }
664    }
665    Ok(())
666}
