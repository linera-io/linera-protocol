1use std::error::Error as StdErr;
2use std::{fmt, io};
3
4/// In this crate's `Result`s.
5#[derive(Debug)]
6#[non_exhaustive]
7pub enum Error {
8    /// TOML parsing errors
9    Parse(toml::de::Error),
10    /// Filesystem access errors
11    Io(io::Error),
12    /// Manifest uses workspace inheritance, and the workspace failed to load
13    Workspace(Box<Error>),
14    /// Manifest uses workspace inheritance, and the data hasn't been inherited yet
15    InheritedUnknownValue,
16    /// Manifest uses workspace inheritance, but the root workspace is missing data
17    WorkspaceIntegrity(String),
18    /// ???
19    Other(&'static str),
20}
21
22impl StdErr for Error {
23    fn source(&self) -> Option<&(dyn StdErr + 'static)> {
24        match *self {
25            Error::Parse(ref err) => Some(err),
26            Error::Io(ref err) => Some(err),
27            Error::Workspace(ref err) => Some(err),
28            Error::Other(_) | Error::InheritedUnknownValue | Error::WorkspaceIntegrity(_) => None,
29        }
30    }
31}
32
33impl fmt::Display for Error {
34    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
35        match *self {
36            Error::Parse(ref err) => err.fmt(f),
37            Error::Io(ref err) => err.fmt(f),
38            Error::Other(msg) => f.write_str(msg),
39            Error::WorkspaceIntegrity(ref s) => f.write_str(s),
40            Error::Workspace(_) => f.write_str("can't load root workspace"),
41            Error::InheritedUnknownValue => f.write_str("value from workspace hasn't been set"),
42        }
43    }
44}
45
46impl Clone for Error {
47    fn clone(&self) -> Self {
48        match *self {
49            Error::Parse(ref err) => Error::Parse(err.clone()),
50            Error::Io(ref err) => Error::Io(io::Error::new(err.kind(), err.to_string())),
51            Error::Other(msg) => Error::Other(msg),
52            Error::WorkspaceIntegrity(ref msg) => Error::WorkspaceIntegrity(msg.clone()),
53            Error::Workspace(ref e) => Error::Workspace(e.clone()),
54            Error::InheritedUnknownValue => Error::InheritedUnknownValue,
55        }
56    }
57}
58
59impl From<toml::de::Error> for Error {
60    fn from(o: toml::de::Error) -> Self {
61        Error::Parse(o)
62    }
63}
64
65impl From<io::Error> for Error {
66    fn from(o: io::Error) -> Self {
67        Error::Io(o)
68    }
