8use serde::ser::{Serialize, Serializer};
9
10pub struct ArrayWrap<'a, T: Serialize, const N: usize> {
11    inner: &'a [T; N],
12}
13
14impl<'a, T: Serialize, const N: usize> ArrayWrap<'a, T, N> {
15    pub fn new(array: &'a [T; N]) -> ArrayWrap<'a, T, N> {
16        ArrayWrap { inner: array }
17    }
18}
19
20impl<T: Serialize, const N: usize> Serialize for ArrayWrap<'_, T, N> {
21    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
22    where
23        S: Serializer,
24    {
25        super::serialize(self.inner, serializer)
26    }
27}
