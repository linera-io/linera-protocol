10fn crate_path(errors: Option<&Errors>) -> Option<syn::Path> {
11    use proc_macro_crate::FoundCrate;
12    const CRATE_NAME: &str = "deluxe";
13
14    let crate_name = match proc_macro_crate::crate_name(CRATE_NAME) {
15        Ok(FoundCrate::Name(name)) => name,
16        Ok(FoundCrate::Itself) => CRATE_NAME.into(),
17        Err(e) => {
18            if let Some(errors) = errors {
19                errors.push(proc_macro2::Span::call_site(), e.to_string());
20            }
21            return None;
22        }
23    };
24
25    let ident = syn::Ident::new(&crate_name, proc_macro2::Span::call_site());
26    Some(syn::parse_quote! { ::#ident })
27}

29#[inline]
30pub fn get_crate_path(path: Option<Option<syn::Path>>, errors: &Errors) -> Option<syn::Path> {
31    match path {
32        Some(Some(path)) => Some(path),
33        Some(None) => crate_path(Some(errors)),
34        None => crate_path(None),
35    }
36}
