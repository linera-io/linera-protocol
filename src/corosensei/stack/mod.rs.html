1//! Details about stacks used by coroutines.
2//!
3//! In most cases, just using the default stack implementation is sufficient.
4//! However it is possible to use a custom `Stack` implementation if more
5//! control is desired.
6
7use core::num::NonZeroUsize;
8
9pub mod valgrind;
10
11cfg_if::cfg_if! {
12    if #[cfg(all(feature = "default-stack", unix))] {
13        mod unix;
14        pub use self::unix::DefaultStack;
15    } else if #[cfg(all(feature = "default-stack", windows))] {
16        mod windows;
17        pub use self::windows::DefaultStack;
18    } else {
19        /// Dummy stack for platforms that do not provide a default stack.
20        ///
21        /// This is only here for use as a default generic parameter.
22        pub struct DefaultStack;
23    }
24}
25
26/// Type to represent a stack address.
27pub type StackPointer = NonZeroUsize;
28
29/// Required stack alignment at function call boundaries.
30pub const STACK_ALIGNMENT: usize = crate::arch::STACK_ALIGNMENT;
31
32/// Minimum size of a stack, excluding guard pages.
33pub const MIN_STACK_SIZE: usize = 4096;
34
35/// A trait for objects that hold ownership of a stack.
36///
37/// # Safety
38///
39/// To preserve memory safety, a stack must have a guard page which catches
40/// stack overflows. It must also contain at least [`MIN_STACK_SIZE`] bytes of
41/// usable memory.
42///
43/// # Windows
44///
45/// On Windows, this trait has some extra methods which are needed because the
46/// OS is much more involved with stack management than other platforms.
47pub unsafe trait Stack {
48    /// Returns the base address of the stack. This is the highest address since
49    /// stacks grow downwards on most modern architectures.
50    ///
51    /// Must be aligned to [`STACK_ALIGNMENT`].
52    fn base(&self) -> StackPointer;
53
54    /// Returns the maximum limit address of the stack. This is the lowest
55    /// address since stacks grow downwards on most modern architectures.
56    ///
57    /// This limit must include any guard pages in the stack.
58    ///
59    /// Must be aligned to [`STACK_ALIGNMENT`].
60    fn limit(&self) -> StackPointer;
61
62    /// On Windows, certain fields must be updated in the Thread Environment
63    /// Block when switching to another stack. This function returns the values
64    /// that must be assigned for this stack.
65    #[cfg(windows)]
66    fn teb_fields(&self) -> StackTebFields;
67
68    /// Updates the stack's copy of TEB fields which may have changed while
69    /// executing code on the stack.
70    #[cfg(windows)]
71    fn update_teb_fields(&mut self, stack_limit: usize, guaranteed_stack_bytes: usize);
72}
73
74/// Fields in the Thread Environment Block (TEB) which must be updated when
75/// switching to a different stack. These are the same fields that are updated
76/// by the `SwitchToFiber` function in the Windows API.
77#[cfg(windows)]
78#[derive(Clone, Copy, Debug)]
79#[allow(non_snake_case)]
80#[allow(missing_docs)]
81pub struct StackTebFields {
82    pub StackBase: usize,
83    pub StackLimit: usize,
84    pub DeallocationStack: usize,
85    pub GuaranteedStackBytes: usize,
86}
87
88/// A mutable reference to a stack can be used as a stack. The lifetime of the
89/// resulting coroutine will be bound to that of the reference.
90unsafe impl<'a, S: Stack> Stack for &'a mut S {
91    #[inline]
92    fn base(&self) -> StackPointer {
93        (**self).base()
94    }
95
96    #[inline]
97    fn limit(&self) -> StackPointer {
98        (**self).limit()
99    }
100
101    #[inline]
102    #[cfg(windows)]
103    fn teb_fields(&self) -> StackTebFields {
104        (**self).teb_fields()
105    }
106
107    #[inline]
108    #[cfg(windows)]
109    fn update_teb_fields(&mut self, stack_limit: usize, guaranteed_stack_bytes: usize) {
110        (**self).update_teb_fields(stack_limit, guaranteed_stack_bytes)
111    }
112}
