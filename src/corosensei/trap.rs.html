13/// Helper type to force a trapping coroutine to return from a trap handler.
14///
15/// This type does not use a lifetime so that it can be placed in static
16/// thread-local storage for access by a trap handler. However it is UB to call
17/// `setup_trap_return` past the lifetime of the originating coroutine.
18///
19/// See [`ScopedCoroutine::trap_handler`] for more details.
20#[derive(Clone, Copy)]
21pub struct CoroutineTrapHandler<Return> {
22    pub(crate) stack_base: StackPointer,
23    pub(crate) stack_limit: StackPointer,
24    pub(crate) marker: PhantomData<fn(Return)>,
25}
26
27impl<Return> CoroutineTrapHandler<Return> {
28    /// Checks whether the stack pointer at the point where a trap occurred is
29    /// within the coroutine that this `CoroutineTrapHandler` was produced from.
30    /// This check includes any guard pages on the stack and will therefore
31    /// still return true in the case of a stack overflow.
32    ///
33    /// The result of this function is only meaningful if the coroutine has not
34    /// been dropped yet.
35    pub fn stack_ptr_in_bounds(&self, stack_ptr: usize) -> bool {
36        stack_ptr >= self.stack_limit.get() && stack_ptr < self.stack_base.get()
37    }
38
39    /// Sets up the stack and register state so that once the trap handler
40    /// returns, the interrupted coroutine will execute the given function and
41    /// then exit the coroutine with the result as a `CoroutineResult::Return`.
42    ///
43    /// This function works by setting up the coroutine stack for a call to a
44    /// special handler function. The returned `TrapHandlerRegs` indicate which
45    /// registers the trap handler must update in the interrupted register
46    /// context so that control is properly passed to this handler function.
47    ///
48    /// The stack is reset by this function: all data currently on the coroutine
49    /// stack is discarded and the full stack is available for use by the given
50    /// function.
51    ///
52    /// # Panics
53    ///
54    /// It is safe for the provided function to panic, this will simply cause
55    /// the panic to unwind back to the parent of the coroutine. However no
56    /// further traps may happen after this.
57    ///
58    /// # Safety
59    ///
60    /// Using this feature safely has very strict requirements. The coroutine
61    /// must not write to any memory outside of itself except through volatile
62    /// operations or assembly. Effectively, the coroutine must yield to its
63    /// parent for any operation which requires interactions with its
64    /// environment.
65    ///
66    /// Additionally, the exception context must have its stack pointer within
67    /// the interrupted coroutine, as checked by `stack_ptr_in_bounds`. Since
68    /// this function modifies the coroutine stack, the trap handler *must*
69    /// update the interrupted register context as specified in the
70    /// `TrapHandlerRegs`.
71    pub unsafe fn setup_trap_handler<F>(&self, f: F) -> TrapHandlerRegs
72    where
73        F: FnOnce() -> Return,
74        F: 'static,
75    {
76        initial_func_abi! {
77            // This function is executed on the coroutine stack once execution
78            // is resumed from the trap handler.
79            unsafe fn trap_handler<F: FnOnce() -> Return, Return>(
80                f: *mut F,
81                parent_link: *mut StackPointer
82            ) -> ! {
83                // After returning from the exception handler we may have an
84                // invalid SEH exception chain. We need to reset it to the
85                // exception record at the root of the stack.
86                #[cfg(all(windows, target_arch = "x86"))]
87                arch::reset_seh_handler(parent_link);
88
89                // This must be called after a stack overflow exception, but it
90                // doesn't hurt to call it for other exception types as well.
91                #[cfg(windows)]
92                reset_guard_page();
93
94                let result = crate::unwind::catch_unwind_at_root(f.read());
95
96                let mut result = ManuallyDrop::new(result);
97                arch::switch_and_reset(util::encode_val(&mut result), parent_link);
98            }
99        }
100
101        // Move the value into the coroutine stack so that it is still available
102        // after the trap handler returns. Then defer to the arch-specific code
103        // to set up the register state for a call.
104        arch::setup_trap_trampoline(self.stack_base, f, trap_handler::<F, Return>)
105    }
106}

108/// On Windows, the guard pages at the end of a stack must be reset after a
109/// stack overflow exception has occurred. This function must be called before
110/// executing new code otherwise future stack overflow may not be caught.
111///
112/// For more information, see the [`_resetstkoflw`] function in the Windows
113/// documentation.
114///
115/// [`_resetstkoflw`]: https://docs.microsoft.com/en-us/cpp/c-runtime-library/reference/resetstkoflw
116#[cfg(windows)]
117fn reset_guard_page() {
118    extern "C" {
119        fn _resetstkoflw() -> i32;
120    }
121
122    if unsafe { _resetstkoflw() } == 0 {
123        panic!("failed to restore stack guard page");
124    }
125}
