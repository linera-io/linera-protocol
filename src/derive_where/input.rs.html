17/// Parsed input.
18pub struct Input<'a> {
19	/// Path to `derive_where` if set by `#[derive_where(crate = ...)]`.
20	pub crate_: Option<Path>,
21	/// `derive_where` attributes on the item.
22	pub derive_wheres: Vec<DeriveWhere>,
23	/// Generics necessary to define for an `impl`.
24	pub generics: SplitGenerics<'a>,
25	/// Fields or variants of this item.
26	pub item: Item<'a>,
27}
28
29impl<'a> Input<'a> {
30	/// Create [`Input`] from `proc_macro_derive` parameter.
31	pub fn from_input(
32		span: Span,
33		DeriveInput {
34			attrs,
35			ident,
36			generics,
37			data,
38			..
39		}: &'a DeriveInput,
40	) -> Result<Self> {
41		// Parse `Attribute`s on item.
42		let ItemAttr {
43			crate_,
44			skip_inner,
45			derive_wheres,
46			incomparable,
47		} = ItemAttr::from_attrs(span, data, attrs)?;
48
49		// Find if `incomparable` is specified on any item/variant.
50		let mut found_incomparable = incomparable.0.is_some();
51
52		// Extract fields and variants of this item.
53		let item = match &data {
54			syn::Data::Struct(data) => Data::from_struct(
55				span,
56				&derive_wheres,
57				skip_inner,
58				incomparable,
59				ident,
60				&data.fields,
61			)
62			.map(Item::Item)?,
63			syn::Data::Enum(data) => {
64				#[cfg(not(feature = "nightly"))]
65				let discriminant = Discriminant::parse(attrs, &data.variants)?;
66
67				let variants = data
68					.variants
69					.iter()
70					.map(|variant| Data::from_variant(ident, &derive_wheres, variant))
71					.collect::<Result<Vec<Data>>>()?;
72
73				// Find if a default option is specified on a variant.
74				let mut found_default = false;
75
76				// While searching for a default option, check for duplicates.
77				for variant in &variants {
78					if let Some(span) = variant.default_span() {
79						if found_default {
80							return Err(Error::default_duplicate(span));
81						} else {
82							found_default = true;
83						}
84					}
85					if let (Some(item), Some(variant)) = (incomparable.0, variant.incomparable.0) {
86						return Err(Error::incomparable_on_item_and_variant(item, variant));
87					}
88					found_incomparable |= variant.is_incomparable();
89				}
90
91				// Make sure a variant has the `option` attribute if `Default` is being
92				// implemented.
93				if !found_default
94					&& derive_wheres
95						.iter()
96						.any(|derive_where| derive_where.contains(Trait::Default))
97				{
98					return Err(Error::default_missing(span));
99				}
100
101				// Empty enums aren't allowed unless they implement `Default` or are
102				// incomparable.
103				if !found_default
104					&& !found_incomparable
105					&& variants.iter().all(|variant| match variant.fields() {
106						Either::Left(fields) => fields.fields.is_empty(),
107						Either::Right(_) => true,
108					}) {
109					return Err(Error::item_empty(span));
110				}
111
112				Item::Enum {
113					#[cfg(not(feature = "nightly"))]
114					discriminant,
115					ident,
116					variants,
117					incomparable,
118				}
119			}
120			syn::Data::Union(data) => Data::from_union(
121				span,
122				&derive_wheres,
123				skip_inner,
124				incomparable,
125				ident,
126				&data.fields,
127			)
128			.map(Item::Item)?,
129		};
130
131		// Don't allow generic constraints be the same as generics on item unless there
132		// is a use-case for it.
133		// Count number of generic type parameters.
134		let generics_len = generics
135			.params
136			.iter()
137			.filter(|generic_param| match generic_param {
138				GenericParam::Type(_) => true,
139				GenericParam::Lifetime(_) | GenericParam::Const(_) => false,
140			})
141			.count();
142
143		'outer: for derive_where in &derive_wheres {
144			// No point in starting to compare both if not even the length is the same.
145			// This can be easily circumvented by doing the following:
146			// `#[derive_where(..; T: Clone)]`, or `#[derive_where(..; T, T)]`, which
147			// apparently is valid Rust syntax: `where T: Clone, T: Clone`, we are only here
148			// to help though.
149			if derive_where.generics.len() != generics_len {
150				continue;
151			}
152
153			// No point in starting to check if there is no use-case if a custom bound was
154			// used, which is a use-case.
155			if derive_where.any_custom_bound() {
156				continue;
157			}
158
159			// Check if every generic type parameter present on the item is defined in this
160			// `DeriveWhere`.
161			for generic_param in &generics.params {
162				// Only check generic type parameters.
163				if let GenericParam::Type(type_param) = generic_param {
164					if !derive_where.has_type_param(&type_param.ident) {
165						continue 'outer;
166					}
167				}
168			}
169
170			// The `for` loop should short-circuit to the `'outer` loop if not all generic
171			// type parameters were found.
172
173			// Don't allow no use-case compared to std `derive`.
174			for (span, trait_) in derive_where.spans.iter().zip(&derive_where.traits) {
175				// `Default` is used on an enum.
176				if trait_ == Trait::Default && item.is_enum() {
177					continue;
178				}
179
180				// Any field is skipped with a corresponding `Trait`.
181				if item.any_skip_trait(***trait_) {
182					continue;
183				}
184
185				// Any variant is marked as incomparable.
186				if found_incomparable {
187					continue;
188				}
189
190				#[cfg(feature = "zeroize")]
191				{
192					// `Zeroize(crate = ..)`, `ZeroizeOnDrop(crate = ..)` or
193					// `ZeroizeOnDrop(no_drop)` is used.
194					if let DeriveTrait::Zeroize(Zeroize { crate_: Some(_) })
195					| DeriveTrait::ZeroizeOnDrop(ZeroizeOnDrop {
196						crate_: Some(_), ..
197					})
198					| DeriveTrait::ZeroizeOnDrop(ZeroizeOnDrop { no_drop: true, .. }) = *trait_
199					{
200						continue;
201					}
202
203					// `Zeroize(fqs)` is used on any field.
204					if trait_ == Trait::Zeroize && item.any_fqs() {
205						continue;
206					}
207				}
208
209				return Err(Error::use_case(*span));
210			}
211		}
212
213		let generics = SplitGenerics::new(generics);
214
215		Ok(Self {
216			crate_,
217			derive_wheres,
218			generics,
219			item,
220		})
221	}
222}

224/// Stores output of [`Generics::split_for_impl()`].
225pub struct SplitGenerics<'a> {
226	/// Necessary generic definitions.
227	pub imp: ImplGenerics<'a>,
228	/// Generics on the type itself.
229	pub ty: TypeGenerics<'a>,
230	/// `where` clause.
231	pub where_clause: Option<&'a WhereClause>,
232}
233
234impl<'a> SplitGenerics<'a> {
235	/// Creates a [`SplitGenerics`] from [`Generics`].
236	fn new(generics: &'a Generics) -> Self {
237		let (imp, ty, where_clause) = generics.split_for_impl();
238
239		SplitGenerics {
240			imp,
241			ty,
242			where_clause,
243		}
244	}
245}
