1//! Parsing implementation for `#[derive_where(crate = ...)]`.
2
3use proc_macro2::Span;
4use syn::{spanned::Spanned, Expr, ExprLit, ExprPath, Lit, Meta, Path, Result};
5
6use crate::{util, Error, DERIVE_WHERE};
7
8/// Parses `#[derive_where(crate = ...)]`.
9pub fn parse_crate(meta: Meta) -> Result<(Path, Span)> {
10	if let Meta::NameValue(name_value) = meta {
11		let path = match &name_value.value {
12			Expr::Lit(ExprLit {
13				lit: Lit::Str(lit_str),
14				..
15			}) => match lit_str.parse::<Path>() {
16				Ok(path) => path,
17				Err(error) => return Err(Error::path(lit_str.span(), error)),
18			},
19			Expr::Path(ExprPath { path, .. }) => path.clone(),
20			_ => return Err(Error::option_syntax(name_value.value.span())),
21		};
22
23		if path == util::path_from_strs(&[DERIVE_WHERE]) {
24			Err(Error::path_unnecessary(
25				path.span(),
26				&format!("::{}", DERIVE_WHERE),
27			))
28		} else {
29			Ok((path, name_value.span()))
30		}
31	} else {
32		Err(Error::option_syntax(meta.span()))
33	}
34}
