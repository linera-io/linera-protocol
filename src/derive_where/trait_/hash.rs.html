13/// [`TraitImpl`] for [`Hash`](trait@std::hash::Hash).
14pub struct Hash;
15
16impl TraitImpl for Hash {
17	fn as_str() -> &'static str {
18		"Hash"
19	}
20
21	fn default_derive_trait() -> DeriveTrait {
22		DeriveTrait::Hash
23	}
24
25	fn path(&self) -> syn::Path {
26		util::path_from_strs(&["core", "hash", "Hash"])
27	}
28
29	fn build_signature(
30		&self,
31		_derive_where: &DeriveWhere,
32		_item: &Item,
33		_generics: &SplitGenerics<'_>,
34		body: &TokenStream,
35	) -> TokenStream {
36		quote! {
37			fn hash<__H: ::core::hash::Hasher>(&self, __state: &mut __H) {
38				match self {
39					#body
40				}
41			}
42		}
43	}
44
45	fn build_body(&self, _derive_where: &DeriveWhere, data: &Data) -> TokenStream {
46		let self_pattern = data.self_pattern();
47		let trait_path = self.path();
48
49		// Add hashing the variant if this is an enum.
50		let discriminant = if let DataType::Variant { .. } = data.type_ {
51			Some(quote! { #trait_path::hash(&::core::mem::discriminant(self), __state); })
52		} else {
53			None
54		};
55
56		match data.simple_type() {
57			SimpleType::Struct(_) | SimpleType::Tuple(_) => {
58				let self_ident = data.iter_self_ident(**self);
59
60				quote! {
61					#self_pattern => {
62						#discriminant
63						#(#trait_path::hash(#self_ident, __state);)*
64					}
65				}
66			}
67			SimpleType::Unit(_) => {
68				quote! {
69					#self_pattern => {
70						#discriminant
71					}
72				}
73			}
74			SimpleType::Union => unreachable!("unexpected trait for union"),
75		}
76	}
77}
78
79impl Deref for Hash {
80	type Target = Trait;
81
82	fn deref(&self) -> &Self::Target {
83		&Trait::Hash
84	}
85}
