11/// [`TraitImpl`] for [`Eq`](trait@std::cmp::Eq).
12pub struct Eq;
13
14impl TraitImpl for Eq {
15	fn as_str() -> &'static str {
16		"Eq"
17	}
18
19	fn default_derive_trait() -> DeriveTrait {
20		DeriveTrait::Eq
21	}
22
23	fn path(&self) -> Path {
24		util::path_from_strs(&["core", "cmp", "Eq"])
25	}
26
27	fn additional_impl(&self) -> Option<(Path, TokenStream)> {
28		Some((self.path(), quote! {}))
29	}
30
31	fn impl_item(
32		&self,
33		_: Option<&Path>,
34		_: &DeriveInput,
35		imp: &ImplGenerics<'_>,
36		ident: &Ident,
37		ty: &TypeGenerics<'_>,
38		where_clause: &Option<Cow<'_, WhereClause>>,
39		body: TokenStream,
40	) -> TokenStream {
41		quote! {
42			const _: () = {
43				trait DeriveWhereAssertEq {
44					fn assert(&self);
45				}
46
47				impl #imp DeriveWhereAssertEq for #ident #ty
48				#where_clause
49				{
50					#body
51				}
52			};
53		}
54	}
55
56	fn build_signature(
57		&self,
58		_derive_where: &DeriveWhere,
59		_item: &Item,
60		_generics: &SplitGenerics<'_>,
61		body: &TokenStream,
62	) -> TokenStream {
63		quote! {
64			fn assert(&self) {
65				struct __AssertEq<__T: ::core::cmp::Eq + ?::core::marker::Sized>(::core::marker::PhantomData<__T>);
66
67				#body
68			}
69		}
70	}
71
72	fn build_body(&self, _derive_where: &DeriveWhere, data: &Data) -> TokenStream {
73		let types = data.iter_fields(**self).map(|field| field.type_);
74
75		quote! {
76			#(let _: __AssertEq<#types>;)*
77		}
78	}
79}
