7// minimum size of a parallelized chunk
8#[allow(unused)]
9#[cfg(feature = "parallel")]
10const MIN_PARALLEL_CHUNK_SIZE: usize = 1 << 7;

13pub(crate) fn bitreverse(mut n: u32, l: u32) -> u32 {
14    let mut r = 0;
15    for _ in 0..l {
16        r = (r << 1) | (n & 1);
17        n >>= 1;
18    }
19    r
20}

26pub(crate) fn compute_powers_and_mul_by_const_serial<F: Field>(
27    size: usize,
28    root: F,
29    c: F,
30) -> Vec<F> {
31    let mut value = c;
32    (0..size)
33        .map(|_| {
34            let old_value = value;
35            value *= root;
36            old_value
37        })
38        .collect()
39}

41#[allow(unused)]
42#[cfg(feature = "parallel")]
43pub(crate) fn compute_powers<F: Field>(size: usize, g: F) -> Vec<F> {
44    if size < MIN_PARALLEL_CHUNK_SIZE {
45        return compute_powers_serial(size, g);
46    }
47    // compute the number of threads we will be using.
48    use ark_std::cmp::{max, min};
49    let num_cpus_available = rayon::current_num_threads();
50    let num_elem_per_thread = max(size / num_cpus_available, MIN_PARALLEL_CHUNK_SIZE);
51    let num_cpus_used = size / num_elem_per_thread;
52
53    // Split up the powers to compute across each thread evenly.
54    let res: Vec<F> = (0..num_cpus_used)
55        .into_par_iter()
56        .flat_map(|i| {
57            let offset = g.pow(&[(i * num_elem_per_thread) as u64]);
58            // Compute the size that this chunks' output should be
59            // (num_elem_per_thread, unless there are less than num_elem_per_thread elements remaining)
60            let num_elements_to_compute = min(size - i * num_elem_per_thread, num_elem_per_thread);
61            let res = compute_powers_and_mul_by_const_serial(num_elements_to_compute, g, offset);
62            res
63        })
64        .collect();
65    res
66}

68#[cfg(feature = "parallel")]
69fn log2_floor(num: usize) -> u32 {
70    if num == 0 {
71        0
72    } else {
73        1usize.leading_zeros() - num.leading_zeros()
74    }
75}

77#[cfg(feature = "parallel")]
78pub(crate) fn best_fft<T: DomainCoeff<F>, F: FftField>(
79    a: &mut [T],
80    omega: F,
81    log_n: u32,
82    serial_fft: fn(&mut [T], F, u32),
83) {
84    let num_cpus = rayon::current_num_threads();
85    let log_cpus = log2_floor(num_cpus);
86    if log_n <= log_cpus {
87        serial_fft(a, omega, log_n);
88    } else {
89        parallel_fft(a, omega, log_n, log_cpus, serial_fft);
90    }
91}

104#[cfg(feature = "parallel")]
105pub(crate) fn parallel_fft<T: DomainCoeff<F>, F: FftField>(
106    a: &mut [T],
107    omega: F,
108    log_n: u32,
109    log_cpus: u32,
110    serial_fft: fn(&mut [T], F, u32),
111) {
112    assert!(log_n >= log_cpus);
113    // For documentation purposes, comments explain things
114    // as though `a` is a polynomial that we are trying to evaluate.
115
116    // Partition `a` equally into the number of threads.
117    // each partition is then of size m / num_threads.
118    let m = a.len();
119    let num_threads = 1 << (log_cpus as usize);
120    let num_cosets = num_threads;
121    assert_eq!(m % num_threads, 0);
122    let coset_size = m / num_threads;
123
124    // We compute the FFT non-mutatively first in tmp first,
125    // and then shuffle it back into a.
126    // The evaluations are going to be arranged in cosets, each of size |a| / num_threads.
127    // so the first coset is (1, g^{num_cosets}, g^{2*num_cosets}, etc.)
128    // the second coset is (g, g^{1 + num_cosets}, g^{1 + 2*num_cosets}, etc.)
129    // These are cosets with generator g^{num_cosets}, and varying shifts.
130    let mut tmp = vec![vec![T::zero(); coset_size]; num_cosets];
131    let new_omega = omega.pow(&[num_cosets as u64]);
132    let new_two_adicity = ark_ff::utils::k_adicity(2, coset_size as u64);
133
134    // For each coset, we first build a polynomial of degree |coset size|,
135    // whose evaluations over coset k will agree with the evaluations of a over the coset.
136    // Denote the kth such polynomial as poly_k
137    tmp.par_iter_mut()
138        .enumerate()
139        .for_each(|(k, kth_poly_coeffs)| {
140            // Shuffle into a sub-FFT
141            let omega_k = omega.pow(&[k as u64]);
142            let omega_step = omega.pow(&[(k * coset_size) as u64]);
143
144            let mut elt = F::one();
145            // Construct kth_poly_coeffs, which is a polynomial whose evaluations on this coset
146            // should equal the evaluations of a on this coset.
147            // `kth_poly_coeffs[i] = sum_{c in num_cosets} g^{k * (i + c * |coset|)} * a[i + c * |coset|]`
148            // Where c represents the index of the coset being considered.
149            // multiplying by g^{k*i} corresponds to the shift for just being in a different coset.
150            //
151            // TODO: Come back and improve the speed, and make this a more 'normal'
152            // Cooley-Tukey. This appears to be an FFT of the polynomial
153            // `P(x) = sum_{c in |coset|} a[i + c |coset|] * x^c`
154            // onto this coset.
155            // However this is being evaluated in time O(N) instead of time
156            // O(|coset|log(|coset|)). If this understanding is the case, its not
157            // doing standard Cooley-Tukey. At the moment, this has time complexity
158            // of at least 2*N field mul's per thread, so we will be getting
159            // pretty bad parallelism. Exact complexity per thread atm is
160            // `2N + (N/num threads)log(N/num threads)` field muls Compare to the time
161            // complexity of serial is Nlog(N) field muls), with log(N) in [15, 25]
162            for i in 0..coset_size {
163                for c in 0..num_threads {
164                    let idx = i + (c * coset_size);
165                    // t = the value of a corresponding to the ith element of the sth coset.
166                    let mut t = a[idx];
167                    // elt = g^{k * idx}
168                    t *= elt;
169                    kth_poly_coeffs[i] += t;
170                    elt *= &omega_step;
171                }
172                elt *= &omega_k;
173            }
174
175            // Perform sub-FFT
176            // Since the sub-FFT is mutative, after this point
177            // `kth_poly_coeffs` should be renamed `kth_coset_evals`
178            serial_fft(kth_poly_coeffs, new_omega, new_two_adicity);
179        });
180
181    // shuffle the values computed above into a
182    // The evaluations of a should be ordered as (1, g, g^2, ...)
183    a.iter_mut()
184        .enumerate()
185        .for_each(|(i, a)| *a = tmp[i % num_cosets][i / num_cosets]);
186}

188/// An iterator over the elements of a domain.
189pub struct Elements<F: FftField> {
190    pub(crate) cur_elem: F,
191    pub(crate) cur_pow: u64,
192    pub(crate) size: u64,
193    pub(crate) group_gen: F,
194}
195
196impl<F: FftField> Iterator for Elements<F> {
197    type Item = F;
198    fn next(&mut self) -> Option<F> {
199        if self.cur_pow == self.size {
200            None
201        } else {
202            let cur_elem = self.cur_elem;
203            self.cur_elem *= &self.group_gen;
204            self.cur_pow += 1;
205            Some(cur_elem)
206        }
207    }
208}
