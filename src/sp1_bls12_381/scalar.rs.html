24/// Represents an element of the scalar field $\mathbb{F}_q$ of the BLS12-381 elliptic
25/// curve construction.
26// The internal representation of this type is four 64-bit unsigned
27// integers in little-endian order. `Scalar` values are always in
28// Montgomery form; i.e., Scalar(a) = aR mod q, with R = 2^256.
29#[derive(Clone, Copy, Eq)]
30pub struct Scalar(pub [u64; 4]);

274    /// Attempts to convert a little-endian byte representation of
275    /// a scalar into a `Scalar`, failing if the input is not canonical.
276    pub fn from_bytes(bytes: &[u8; 32]) -> CtOption<Scalar> {
277        let mut tmp = Scalar([0, 0, 0, 0]);
278
279        tmp.0[0] = u64::from_le_bytes(<[u8; 8]>::try_from(&bytes[0..8]).unwrap());
280        tmp.0[1] = u64::from_le_bytes(<[u8; 8]>::try_from(&bytes[8..16]).unwrap());
281        tmp.0[2] = u64::from_le_bytes(<[u8; 8]>::try_from(&bytes[16..24]).unwrap());
282        tmp.0[3] = u64::from_le_bytes(<[u8; 8]>::try_from(&bytes[24..32]).unwrap());
283
284        // Try to subtract the modulus
285        let (_, borrow) = sbb(tmp.0[0], MODULUS.0[0], 0);
286        let (_, borrow) = sbb(tmp.0[1], MODULUS.0[1], borrow);
287        let (_, borrow) = sbb(tmp.0[2], MODULUS.0[2], borrow);
288        let (_, borrow) = sbb(tmp.0[3], MODULUS.0[3], borrow);
289
290        // If the element is smaller than MODULUS then the
291        // subtraction will underflow, producing a borrow value
292        // of 0xffff...ffff. Otherwise, it'll be zero.
293        let is_some = (borrow as u8) & 1;
294
295        // Convert to Montgomery form by computing
296        // (a.R^0 * R^2) / R = a.R
297        tmp *= &R2;
298
299        CtOption::new(tmp, Choice::from(is_some))
300    }

302    /// Converts an element of `Scalar` into a byte representation in
303    /// little-endian byte order.
304    pub fn to_bytes(&self) -> [u8; 32] {
305        // Turn into canonical form by computing
306        // (a.R) / R = a
307        let tmp = Scalar::montgomery_reduce(self.0[0], self.0[1], self.0[2], self.0[3], 0, 0, 0, 0);
308
309        let mut res = [0; 32];
310        res[0..8].copy_from_slice(&tmp.0[0].to_le_bytes());
311        res[8..16].copy_from_slice(&tmp.0[1].to_le_bytes());
312        res[16..24].copy_from_slice(&tmp.0[2].to_le_bytes());
313        res[24..32].copy_from_slice(&tmp.0[3].to_le_bytes());
314
315        res
316    }

333    fn from_u512(limbs: [u64; 8]) -> Scalar {
334        // We reduce an arbitrary 512-bit number by decomposing it into two 256-bit digits
335        // with the higher bits multiplied by 2^256. Thus, we perform two reductions
336        //
337        // 1. the lower bits are multiplied by R^2, as normal
338        // 2. the upper bits are multiplied by R^2 * 2^256 = R^3
339        //
340        // and computing their sum in the field. It remains to see that arbitrary 256-bit
341        // numbers can be placed into Montgomery form safely using the reduction. The
342        // reduction works so long as the product is less than R=2^256 multiplied by
343        // the modulus. This holds because for any `c` smaller than the modulus, we have
344        // that (2^256 - 1)*c is an acceptable product for the reduction. Therefore, the
345        // reduction always works so long as `c` is in the field; in this case it is either the
346        // constant `R2` or `R3`.
347        let d0 = Scalar([limbs[0], limbs[1], limbs[2], limbs[3]]);
348        let d1 = Scalar([limbs[4], limbs[5], limbs[6], limbs[7]]);
349        // Convert to Montgomery form
350        d0 * R2 + d1 * R3
351    }

360    /// CPU version of the square operation. Necessary to prevent syscalls in unconstrained mode.
361    fn cpu_square(&self) -> Scalar {
362        let (r1, carry) = mac(0, self.0[0], self.0[1], 0);
363        let (r2, carry) = mac(0, self.0[0], self.0[2], carry);
364        let (r3, r4) = mac(0, self.0[0], self.0[3], carry);
365
366        let (r3, carry) = mac(r3, self.0[1], self.0[2], 0);
367        let (r4, r5) = mac(r4, self.0[1], self.0[3], carry);
368
369        let (r5, r6) = mac(r5, self.0[2], self.0[3], 0);
370
371        let r7 = r6 >> 63;
372        let r6 = (r6 << 1) | (r5 >> 63);
373        let r5 = (r5 << 1) | (r4 >> 63);
374        let r4 = (r4 << 1) | (r3 >> 63);
375        let r3 = (r3 << 1) | (r2 >> 63);
376        let r2 = (r2 << 1) | (r1 >> 63);
377        let r1 = r1 << 1;
378
379        let (r0, carry) = mac(0, self.0[0], self.0[0], 0);
380        let (r1, carry) = adc(0, r1, carry);
381        let (r2, carry) = mac(r2, self.0[1], self.0[1], carry);
382        let (r3, carry) = adc(0, r3, carry);
383        let (r4, carry) = mac(r4, self.0[2], self.0[2], carry);
384        let (r5, carry) = adc(0, r5, carry);
385        let (r6, carry) = mac(r6, self.0[3], self.0[3], carry);
386        let (r7, _) = adc(0, r7, carry);
387
388        Scalar::montgomery_reduce(r0, r1, r2, r3, r4, r5, r6, r7)
389    }

405    /// Exponentiates `self` by `by`, where `by` is a
406    /// little-endian order integer exponent.
407    pub fn pow(&self, by: &[u64; 4]) -> Self {
408        let mut res = Self::one();
409        for e in by.iter().rev() {
410            for i in (0..64).rev() {
411                res = res.square();
412                let mut tmp = res;
413                tmp.mul_inp(self);
414                res.conditional_assign(&tmp, (((*e >> i) & 0x1) as u8).into());
415            }
416        }
417        res
418    }

420    /// Exponentiates `self` by `by`, where `by` is a
421    /// little-endian order integer exponent.
422    ///
423    /// **This operation is variable time with respect
424    /// to the exponent.** If the exponent is fixed,
425    /// this operation is effectively constant time.
426    pub fn pow_vartime(&self, by: &[u64; 4]) -> Self {
427        let mut res = Self::one();
428        for e in by.iter().rev() {
429            for i in (0..64).rev() {
430                res = res.square();
431
432                if ((*e >> i) & 1) == 1 {
433                    res.mul_inp(self);
434                }
435            }
436        }
437        res
438    }

540    pub fn invert(&self) -> CtOption<Self> {
541        #[cfg(target_os = "zkvm")]
542        {
543            if self.is_zero().into() {
544                return CtOption::new(Self::zero(), Choice::from(0u8));
545            }
546            unconstrained! {
547                let mut buf = [0u8; 32];
548                self.cpu_invert().map(|inv| {
549                    buf[0..32].copy_from_slice(&inv.to_bytes());
550                });
551                hint_slice(&buf);
552            }
553            let byte_vec = read_vec();
554
555            // Safety:
556            //
557            // - The byte_vec is guaranteed to be 32 bytes long because we just pushed it,
558            // and the executor always pushes to the front of the input buffer.
559            //
560            // `from_scalar` just clones the bytes before byte_vec is dropped.
561            let bytes = unsafe { &*(byte_vec.as_ptr() as *const [u8; 32]) };
562            let inv = Scalar::from_bytes(bytes).unwrap();
563
564            assert!(self * &inv == Scalar::one(), "Invalid hint: Scalar invert");
565            return CtOption::new(inv, Choice::from(1u8));
566        }
567        #[cfg(not(target_os = "zkvm"))]
568        {
569            self.cpu_invert()
570        }
571    }

573    #[inline(always)]
574    pub const fn montgomery_reduce(
575        r0: u64,
576        r1: u64,
577        r2: u64,
578        r3: u64,
579        r4: u64,
580        r5: u64,
581        r6: u64,
582        r7: u64,
583    ) -> Self {
584        // The Montgomery reduction here is based on Algorithm 14.32 in
585        // Handbook of Applied Cryptography
586        // <http://cacr.uwaterloo.ca/hac/about/chap14.pdf>.
587
588        let k = r0.wrapping_mul(INV);
589        let (_, carry) = mac(r0, k, MODULUS.0[0], 0);
590        let (r1, carry) = mac(r1, k, MODULUS.0[1], carry);
591        let (r2, carry) = mac(r2, k, MODULUS.0[2], carry);
592        let (r3, carry) = mac(r3, k, MODULUS.0[3], carry);
593        let (r4, carry2) = adc(r4, 0, carry);
594
595        let k = r1.wrapping_mul(INV);
596        let (_, carry) = mac(r1, k, MODULUS.0[0], 0);
597        let (r2, carry) = mac(r2, k, MODULUS.0[1], carry);
598        let (r3, carry) = mac(r3, k, MODULUS.0[2], carry);
599        let (r4, carry) = mac(r4, k, MODULUS.0[3], carry);
600        let (r5, carry2) = adc(r5, carry2, carry);
601
602        let k = r2.wrapping_mul(INV);
603        let (_, carry) = mac(r2, k, MODULUS.0[0], 0);
604        let (r3, carry) = mac(r3, k, MODULUS.0[1], carry);
605        let (r4, carry) = mac(r4, k, MODULUS.0[2], carry);
606        let (r5, carry) = mac(r5, k, MODULUS.0[3], carry);
607        let (r6, carry2) = adc(r6, carry2, carry);
608
609        let k = r3.wrapping_mul(INV);
610        let (_, carry) = mac(r3, k, MODULUS.0[0], 0);
611        let (r4, carry) = mac(r4, k, MODULUS.0[1], carry);
612        let (r5, carry) = mac(r5, k, MODULUS.0[2], carry);
613        let (r6, carry) = mac(r6, k, MODULUS.0[3], carry);
614        let (r7, _) = adc(r7, carry2, carry);
615
616        // Result may be within MODULUS of the correct value
617        #[allow(clippy::needless_borrow)]
618        (&Scalar([r4, r5, r6, r7])).sub(&MODULUS)
619    }
620
621    #[inline]
622    #[cfg(target_os = "zkvm")]
623    pub(crate) fn mul_r_inv_internal(&mut self) {
624        unsafe {
625            sys_bigint(
626                self.0.as_mut_ptr() as *mut [u32; 8],
627                0,
628                self.0.as_ptr() as *const [u32; 8],
629                &R_INV,
630                &MODULUS_LIMBS_32,
631            );
632        }

635    #[inline]
636    pub fn mul_inp(&mut self, rhs: &Scalar) {
637        cfg_if! {
638            if #[cfg(target_os = "zkvm")] {
639                unsafe {
640                    sys_bigint(
641                        self.0.as_mut_ptr() as *mut[u32; 8],
642                        0,
643                        self.0.as_ptr() as *const [u32; 8],
644                        rhs.0.as_ptr() as *const [u32; 8],
645                        &MODULUS_LIMBS_32,

656    /// CPU version of the multiplication operation. Necessary to prevent syscalls in unconstrained mode.
657    fn cpu_mul(&self, rhs: &Self) -> Self {
658        // Schoolbook multiplication
659
660        let (r0, carry) = mac(0, self.0[0], rhs.0[0], 0);
661        let (r1, carry) = mac(0, self.0[0], rhs.0[1], carry);
662        let (r2, carry) = mac(0, self.0[0], rhs.0[2], carry);
663        let (r3, r4) = mac(0, self.0[0], rhs.0[3], carry);
664
665        let (r1, carry) = mac(r1, self.0[1], rhs.0[0], 0);
666        let (r2, carry) = mac(r2, self.0[1], rhs.0[1], carry);
667        let (r3, carry) = mac(r3, self.0[1], rhs.0[2], carry);
668        let (r4, r5) = mac(r4, self.0[1], rhs.0[3], carry);
669
670        let (r2, carry) = mac(r2, self.0[2], rhs.0[0], 0);
671        let (r3, carry) = mac(r3, self.0[2], rhs.0[1], carry);
672        let (r4, carry) = mac(r4, self.0[2], rhs.0[2], carry);
673        let (r5, r6) = mac(r5, self.0[2], rhs.0[3], carry);
674
675        let (r3, carry) = mac(r3, self.0[3], rhs.0[0], 0);
676        let (r4, carry) = mac(r4, self.0[3], rhs.0[1], carry);
677        let (r5, carry) = mac(r5, self.0[3], rhs.0[2], carry);
678        let (r6, r7) = mac(r6, self.0[3], rhs.0[3], carry);
679
680        Scalar::montgomery_reduce(r0, r1, r2, r3, r4, r5, r6, r7)
681    }

683    /// Multiplies `rhs` by `self`, returning the result.
684    #[inline]
685    pub fn mul(&self, rhs: &Self) -> Self {
686        cfg_if! {
687            if #[cfg(target_os = "zkvm")] {
688                let mut res = *self;
689                res.mul_inp(rhs);
690                res
691            } else {
692                self.cpu_mul(rhs)
693            }
694        }
695    }

697    /// Subtracts `rhs` from `self`, returning the result.
698    #[inline]
699    pub const fn sub(&self, rhs: &Self) -> Self {
700        let (d0, borrow) = sbb(self.0[0], rhs.0[0], 0);
701        let (d1, borrow) = sbb(self.0[1], rhs.0[1], borrow);
702        let (d2, borrow) = sbb(self.0[2], rhs.0[2], borrow);
703        let (d3, borrow) = sbb(self.0[3], rhs.0[3], borrow);
704
705        // If underflow occurred on the final limb, borrow = 0xfff...fff, otherwise
706        // borrow = 0x000...000. Thus, we use it as a mask to conditionally add the modulus.
707        let (d0, carry) = adc(d0, MODULUS.0[0] & borrow, 0);
708        let (d1, carry) = adc(d1, MODULUS.0[1] & borrow, carry);
709        let (d2, carry) = adc(d2, MODULUS.0[2] & borrow, carry);
710        let (d3, _) = adc(d3, MODULUS.0[3] & borrow, carry);
711
712        Scalar([d0, d1, d2, d3])
713    }
714
715    /// Adds `rhs` to `self`, returning the result.
716    #[inline]
717    pub const fn add(&self, rhs: &Self) -> Self {
718        let (d0, carry) = adc(self.0[0], rhs.0[0], 0);
719        let (d1, carry) = adc(self.0[1], rhs.0[1], carry);
720        let (d2, carry) = adc(self.0[2], rhs.0[2], carry);
721        let (d3, _) = adc(self.0[3], rhs.0[3], carry);
722
723        // Attempt to subtract the modulus, to ensure the value
724        // is smaller than the modulus.
725
726        #[allow(clippy::needless_borrow)]
727        (&Scalar([d0, d1, d2, d3])).sub(&MODULUS)
728    }
729
730    /// Negates `self`.
731    #[inline]
732    pub const fn neg(&self) -> Self {
733        // Subtract `self` from `MODULUS` to negate. Ignore the final
734        // borrow because it cannot underflow; self is guaranteed to
735        // be in the field.
736        let (d0, borrow) = sbb(MODULUS.0[0], self.0[0], 0);
737        let (d1, borrow) = sbb(MODULUS.0[1], self.0[1], borrow);
738        let (d2, borrow) = sbb(MODULUS.0[2], self.0[2], borrow);
739        let (d3, _) = sbb(MODULUS.0[3], self.0[3], borrow);
740
741        // `tmp` could be `MODULUS` if `self` was zero. Create a mask that is
742        // zero if `self` was zero, and `u64::max_value()` if self was nonzero.
743        let mask = (((self.0[0] | self.0[1] | self.0[2] | self.0[3]) == 0) as u64).wrapping_sub(1);
744
745        Scalar([d0 & mask, d1 & mask, d2 & mask, d3 & mask])
746    }
747
748    #[inline]
749    pub fn divn(&self, mut n: u32) -> Scalar {
750        if n >= 256 {
751            return Scalar::from(0);
752        }
753
754        let mut out = *self;
755
756        while n >= 64 {
757            let mut t = 0;
758            for i in out.0.iter_mut().rev() {
759                core::mem::swap(&mut t, i);
760            }
761            n -= 64;
762        }
763
764        if n > 0 {
765            let mut t = 0;
766            for i in out.0.iter_mut().rev() {
767                let t2 = *i << (64 - n);
768                *i >>= n;
769                *i |= t;
770                t = t2;
771            }
772        }
773
774        out
775    }
776}
