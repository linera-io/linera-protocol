82impl ConstantTimeEq for G1Affine {
83    fn ct_eq(&self, other: &Self) -> Choice {
84        // The only cases in which two points are equal are
85        // 1. infinity is set on both
86        // 2. infinity is not set on both, and their coordinates are equal
87
88        (self.infinity & other.infinity)
89            | ((!self.infinity)
90                & (!other.infinity)
91                & self.x.ct_eq(&other.x)
92                & self.y.ct_eq(&other.y))
93    }

230    /// Serializes this element into compressed form. See [`notes::serialization`](crate::notes::serialization)
231    /// for details about how group elements are serialized.
232    pub fn to_compressed(&self) -> [u8; 48] {
233        // Strictly speaking, self.x is zero already when self.infinity is true, but
234        // to guard against implementation mistakes we do not assume this.
235        let mut res = Fp::conditional_select(&self.x, &Fp::zero(), self.infinity).to_bytes();
236
237        // This point is in compressed form, so we set the most significant bit.
238        res[0] |= 1u8 << 7;
239
240        // Is this point at infinity? If so, set the second-most significant bit.
241        res[0] |= u8::conditional_select(&0u8, &(1u8 << 6), self.infinity);
242
243        // Is the y-coordinate the lexicographically largest of the two associated with the
244        // x-coordinate? If so, set the third-most significant bit so long as this is not
245        // the point at infinity.
246        res[0] |= u8::conditional_select(
247            &0u8,
248            &(1u8 << 5),
249            (!self.infinity) & self.y.lexicographically_largest(),
250        );
251
252        res
253    }

255    /// Serializes this element into uncompressed form. See [`notes::serialization`](crate::notes::serialization)
256    /// for details about how group elements are serialized.
257    pub fn to_uncompressed(&self) -> [u8; 96] {
258        let mut res = [0; 96];
259
260        res[0..48].copy_from_slice(
261            &Fp::conditional_select(&self.x, &Fp::zero(), self.infinity).to_bytes()[..],
262        );
263        res[48..96].copy_from_slice(
264            &Fp::conditional_select(&self.y, &Fp::zero(), self.infinity).to_bytes()[..],
265        );
266
267        // Is this point at infinity? If so, set the second-most significant bit.
268        res[0] |= u8::conditional_select(&0u8, &(1u8 << 6), self.infinity);
269
270        res
271    }

280    /// Attempts to deserialize an uncompressed element, not checking if the
281    /// element is on the curve and not checking if it is in the correct subgroup.
282    /// **This is dangerous to call unless you trust the bytes you are reading; otherwise,
283    /// API invariants may be broken.** Please consider using `from_uncompressed()` instead.
284    pub fn from_uncompressed_unchecked(bytes: &[u8; 96]) -> CtOption<Self> {
285        // Obtain the three flags from the start of the byte sequence
286        let compression_flag_set = Choice::from((bytes[0] >> 7) & 1);
287        let infinity_flag_set = Choice::from((bytes[0] >> 6) & 1);
288        let sort_flag_set = Choice::from((bytes[0] >> 5) & 1);
289
290        // Attempt to obtain the x-coordinate
291        let x = {
292            let mut tmp = [0; 48];
293            tmp.copy_from_slice(&bytes[0..48]);
294
295            // Mask away the flag bits
296            tmp[0] &= 0b0001_1111;
297
298            Fp::from_bytes(&tmp)
299        };
300
301        // Attempt to obtain the y-coordinate
302        let y = {
303            let mut tmp = [0; 48];
304            tmp.copy_from_slice(&bytes[48..96]);
305
306            Fp::from_bytes(&tmp)
307        };
308
309        x.and_then(|x| {
310            y.and_then(|y| {
311                // Create a point representing this value
312                let p = G1Affine::conditional_select(
313                    &G1Affine {
314                        x,
315                        y,
316                        infinity: infinity_flag_set,
317                    },
318                    &G1Affine::identity(),
319                    infinity_flag_set,
320                );
321
322                CtOption::new(
323                    p,
324                    // If the infinity flag is set, the x and y coordinates should have been zero.
325                    ((!infinity_flag_set) | (infinity_flag_set & x.is_zero() & y.is_zero())) &
326                    // The compression flag should not have been set, as this is an uncompressed element
327                    (!compression_flag_set) &
328                    // The sort flag should not have been set, as this is an uncompressed element
329                    (!sort_flag_set),
330                )
331            })
332        })
333    }

344    /// Attempts to deserialize an uncompressed element, not checking if the
345    /// element is in the correct subgroup.
346    /// **This is dangerous to call unless you trust the bytes you are reading; otherwise,
347    /// API invariants may be broken.** Please consider using `from_compressed()` instead.
348    pub fn from_compressed_unchecked(bytes: &[u8; 48]) -> CtOption<Self> {
349        // Obtain the three flags from the start of the byte sequence
350        let compression_flag_set = Choice::from((bytes[0] >> 7) & 1);
351        let infinity_flag_set = Choice::from((bytes[0] >> 6) & 1);
352        let sort_flag_set = Choice::from((bytes[0] >> 5) & 1);
353
354        // Attempt to obtain the x-coordinate
355        let x = {
356            let mut tmp = [0; 48];
357            tmp.copy_from_slice(&bytes[0..48]);
358
359            // Mask away the flag bits
360            tmp[0] &= 0b0001_1111;
361
362            Fp::from_bytes(&tmp)
363        };
364
365        x.and_then(|x| {
366            // If the infinity flag is set, return the value assuming
367            // the x-coordinate is zero and the sort bit is not set.
368            //
369            // Otherwise, return a recovered point (assuming the correct
370            // y-coordinate can be found) so long as the infinity flag
371            // was not set.
372            CtOption::new(
373                G1Affine::identity(),
374                infinity_flag_set & // Infinity flag should be set
375                compression_flag_set & // Compression flag should be set
376                (!sort_flag_set) & // Sort flag should not be set
377                x.is_zero(), // The x-coordinate should be zero
378            )
379            .or_else(|| {
380                // Recover a y-coordinate given x by y = sqrt(x^3 + 4)
381                ((x.square() * x) + B).sqrt().and_then(|y| {
382                    // Switch to the correct y-coordinate if necessary.
383                    let y = Fp::conditional_select(
384                        &y,
385                        &-y,
386                        y.lexicographically_largest() ^ sort_flag_set,
387                    );
388
389                    CtOption::new(
390                        G1Affine {
391                            x,
392                            y,
393                            infinity: infinity_flag_set,
394                        },
395                        (!infinity_flag_set) & // Infinity flag should not be set
396                        compression_flag_set, // Compression flag should be set
397                    )
398                })
399            })
400        })
401    }

409    /// Returns true if this point is free of an $h$-torsion component, and so it
410    /// exists within the $q$-order subgroup $\mathbb{G}_1$. This should always return true
411    /// unless an "unchecked" API was used.
412    pub fn is_torsion_free(&self) -> Choice {
413        // Algorithm from Section 6 of https://eprint.iacr.org/2021/1130
414        // Updated proof of correctness in https://eprint.iacr.org/2022/352
415        //
416        // Check that endomorphism_p(P) == -[x^2] P
417
418        let minus_x_squared_times_p = G1Projective::from(self).mul_by_x().mul_by_x().neg();
419        let endomorphism_p = endomorphism(self);
420        minus_x_squared_times_p.ct_eq(&G1Projective::from(endomorphism_p))
421    }

423    /// Returns true if this point is on the curve. This should always return
424    /// true unless an "unchecked" API was used.
425    pub fn is_on_curve(&self) -> Choice {
426        // y^2 - x^3 ?= 4
427        (self.y.square() - (self.x.square() * self.x)).ct_eq(&B) | self.infinity
428    }

430    /// Adds two affine points together.
431    /// This function assumes that both values are on the curve.
432    /// In the zkvm context, this is accelerated with precompiles. In regular rust, this entails
433    /// converting one of the points to projective coordinates and then converting the output back.
434    pub fn add_affine(&self, rhs: &Self) -> Self {
435        if self.is_identity().into() {
436            return *rhs;
437        } else if rhs.is_identity().into() {
438            return *self;
439        }
440
441        cfg_if::cfg_if! {
442            if #[cfg(target_os = "zkvm")] {
443                // The add precompile only works when P != Q and P != -Q
444                if self.x != rhs.x {
445                    // In this case, we know that P != Q and P != -Q, since both Q and -Q have the same `x` coordinate
446                    let mut res = self.clone();
447                    res.x.mul_r_inv_internal();
448                    res.y.mul_r_inv_internal();
449                    let mut other = rhs.clone();
450                    other.x.mul_r_inv_internal();
451                    other.y.mul_r_inv_internal();
452                    unsafe {
453                        syscall_bls12381_add(res.x.0.as_mut_ptr() as *mut [u32; 24], other.x.0.as_ptr() as *const [u32; 24]);
454                    }
455                    res.x.mul_r_internal();
456                    res.y.mul_r_internal();
457                    res
458                } else if self.y == rhs.y {
459                    // In this case, we know that P == Q, since both `x` and `y` are equal, so use the double precompile instead
460                    let mut res = self.clone();
461                    res.x.mul_r_inv_internal();
462                    res.y.mul_r_inv_internal();
463                    unsafe {
464                        syscall_bls12381_double(res.x.0.as_mut_ptr() as *mut [u32; 24]);
465                    }
466                    res.x.mul_r_internal();
467                    res.y.mul_r_internal();
468                    res
469                } else {
470                    // In this case, we know that P == -Q, since `x` is equal but `y` is different, so we can just return the identity
471                    assert!(self.y + rhs.y == Fp::zero());
472                    Self::identity()
473                }
474            } else {
475                let proj = G1Projective::from(rhs);
476                let res = proj + self;
477                G1Affine::from(res)
478            }
479        }
480    }

482    #[inline(always)]
483    #[allow(unused_mut)]
484    pub fn double(mut self) -> Self {
485        if self.is_identity().into() {
486            return self;
487        }
488        cfg_if::cfg_if! {
489            if #[cfg(target_os = "zkvm")] {
490                self.x.mul_r_inv_internal();
491                self.y.mul_r_inv_internal();
492                unsafe {
493                    syscall_bls12381_double(self.x.0.as_mut_ptr() as *mut [u32; 24]);
494                }
495                self.x.mul_r_internal();
496                self.y.mul_r_internal();
497                self
498            } else {
499                let proj = G1Projective::from(self);
500                let res = proj.double();
501                G1Affine::from(res)
502            }
503        }
504    }

507/// A nontrivial third root of unity in Fp
508pub const BETA: Fp = Fp::from_raw_unchecked([
509    0x30f1_361b_798a_64e8,
510    0xf3b8_ddab_7ece_5a2a,
511    0x16a8_ca3a_c615_77f7,
512    0xc26a_2ff8_74fd_029b,

517pub fn endomorphism(p: &G1Affine) -> G1Affine {
518    // Endomorphism of the points on the curve.
519    // endomorphism_p(x,y) = (BETA * x, y)
520    // where BETA is a non-trivial cubic root of unity in Fq.
521    let mut res = *p;
522    res.x *= BETA;
523    res
524}

566impl ConstantTimeEq for G1Projective {
567    fn ct_eq(&self, other: &Self) -> Choice {
568        // Is (xz, yz, z) equal to (x'z', y'z', z') when converted to affine?
569
570        let x1 = self.x * other.z;
571        let x2 = other.x * self.z;
572
573        let y1 = self.y * other.z;
574        let y2 = other.y * self.z;
575
576        let self_is_zero = self.z.is_zero();
577        let other_is_zero = other.z.is_zero();
578
579        (self_is_zero & other_is_zero) // Both point at infinity
580            | ((!self_is_zero) & (!other_is_zero) & x1.ct_eq(&x2) & y1.ct_eq(&y2))
581        // Neither point at infinity, coordinates are the same
582    }
583}

724    /// Computes the doubling of this point.
725    pub fn double(&self) -> G1Projective {
726        // Algorithm 9, https://eprint.iacr.org/2015/1060.pdf
727
728        let t0 = self.y.square();
729        let z3 = t0 + t0;
730        let z3 = z3 + z3;
731        let z3 = z3 + z3;
732        let t1 = self.y * self.z;
733        let t2 = self.z.square();
734        let t2 = mul_by_3b(t2);
735        let x3 = t2 * z3;
736        let y3 = t0 + t2;
737        let z3 = t1 * z3;
738        let t1 = t2 + t2;
739        let t2 = t1 + t2;
740        let t0 = t0 - t2;
741        let y3 = t0 * y3;
742        let y3 = x3 + y3;
743        let t1 = self.x * self.y;
744        let x3 = t0 * t1;
745        let x3 = x3 + x3;
746
747        let tmp = G1Projective {
748            x: x3,
749            y: y3,
750            z: z3,
751        };
752
753        G1Projective::conditional_select(&tmp, &G1Projective::identity(), self.is_identity())
754    }

756    /// Adds this point to another point.
757    pub fn add(&self, rhs: &G1Projective) -> G1Projective {
758        // Algorithm 7, https://eprint.iacr.org/2015/1060.pdf
759
760        let t0 = self.x * rhs.x;
761        let t1 = self.y * rhs.y;
762        let t2 = self.z * rhs.z;
763        let t3 = self.x + self.y;
764        let t4 = rhs.x + rhs.y;
765        let t3 = t3 * t4;
766        let t4 = t0 + t1;
767        let t3 = t3 - t4;
768        let t4 = self.y + self.z;
769        let x3 = rhs.y + rhs.z;
770        let t4 = t4 * x3;
771        let x3 = t1 + t2;
772        let t4 = t4 - x3;
773        let x3 = self.x + self.z;
774        let y3 = rhs.x + rhs.z;
775        let x3 = x3 * y3;
776        let y3 = t0 + t2;
777        let y3 = x3 - y3;
778        let x3 = t0 + t0;
779        let t0 = x3 + t0;
780        let t2 = mul_by_3b(t2);
781        let z3 = t1 + t2;
782        let t1 = t1 - t2;
783        let y3 = mul_by_3b(y3);
784        let x3 = t4 * y3;
785        let t2 = t3 * t1;
786        let x3 = t2 - x3;
787        let y3 = y3 * t0;
788        let t1 = t1 * z3;
789        let y3 = t1 + y3;
790        let t0 = t0 * t3;
791        let z3 = z3 * t4;
792        let z3 = z3 + t0;
793
794        G1Projective {
795            x: x3,
796            y: y3,
797            z: z3,
798        }
799    }

802    pub fn add_mixed(&self, rhs: &G1Affine) -> G1Projective {
803        // Algorithm 8, https://eprint.iacr.org/2015/1060.pdf
804
805        let t0 = self.x * rhs.x;
806        let t1 = self.y * rhs.y;
807        let t3 = rhs.x + rhs.y;
808        let t4 = self.x + self.y;
809        let t3 = t3 * t4;
810        let t4 = t0 + t1;
811        let t3 = t3 - t4;
812        let t4 = rhs.y * self.z;
813        let t4 = t4 + self.y;
814        let y3 = rhs.x * self.z;

817        let t0 = x3 + t0;

819        let z3 = t1 + t2;
820        let t1 = t1 - t2;
821        let y3 = mul_by_3b(y3);
822        let x3 = t4 * y3;
823        let t2 = t3 * t1;

833            x: x3,

841    fn multiply(&self, by: &[u8; 32]) -> G1Projective {
842        let mut acc = G1Projective::identity();
843
844        // This is a simple double-and-add implementation of point
845        // multiplication, moving from most significant to least
846        // significant bit of the scalar.
847        //
848        // We skip the leading bit because it's always unset for Fq
849        // elements.
850        for bit in by
851            .iter()
852            .rev()
853            .flat_map(|byte| (0..8).rev().map(move |i| Choice::from((byte >> i) & 1u8)))
854            .skip(1)
855        {
856            acc = acc.double();
857            acc = G1Projective::conditional_select(&acc, &(acc + self), bit);
858        }
859
860        acc
861    }

863    /// Multiply `self` by `crate::BLS_X`, using double and add.
864    #[cfg(not(target_os = "zkvm"))]
865    fn mul_by_x(&self) -> G1Projective {
866        let mut xself = G1Projective::identity();
867        // NOTE: in BLS12-381 we can just skip the first bit.
868        let mut x = crate::BLS_X >> 1;
869        let mut tmp = *self;
870        while x != 0 {
871            tmp = tmp.double();
872
873            if x % 2 == 1 {
874                xself += tmp;
875            }
876            x >>= 1;
877        }
878        // finally, flip the sign
879        if crate::BLS_X_IS_NEGATIVE {
880            xself = -xself;
881        }
882        xself
883    }

885    #[cfg(target_os = "zkvm")]
886    fn mul_by_x(&self) -> G1Projective {
887        let mut xself = G1Affine::identity();
888
889        let mut x = crate::BLS_X >> 1;
890        let mut tmp = G1Affine::from(*self);
891        while x != 0 {
892            tmp = tmp.double();
893
894            if x % 2 == 1 {
895                xself = xself.add_affine(&tmp);
896            }
897            x >>= 1;
898        }
899        // finally, flip the sign
900        if crate::BLS_X_IS_NEGATIVE {
901            xself = -xself;
902        }
903        xself.into()
904    }

913    /// Converts a batch of `G1Projective` elements into `G1Affine` elements. This
914    /// function will panic if `p.len() != q.len()`.
915    pub fn batch_normalize(p: &[Self], q: &mut [G1Affine]) {
916        assert_eq!(p.len(), q.len());
917
918        let mut acc = Fp::one();
919        for (p, q) in p.iter().zip(q.iter_mut()) {
920            // We use the `x` field of `G1Affine` to store the product
921            // of previous z-coordinates seen.
922            q.x = acc;
923
924            // We will end up skipping all identities in p
925            acc = Fp::conditional_select(&(acc * p.z), &acc, p.is_identity());
926        }
927
928        // This is the inverse, as all z-coordinates are nonzero and the ones
929        // that are not are skipped.
930        acc = acc.invert().unwrap();
931
932        for (p, q) in p.iter().rev().zip(q.iter_mut().rev()) {
933            let skip = p.is_identity();
934
935            // Compute tmp = 1/z
936            let tmp = q.x * acc;
937
938            // Cancel out z-coordinate in denominator of `acc`
939            acc = Fp::conditional_select(&(acc * p.z), &acc, skip);
940
941            // Set the coordinates to the correct value
942            q.x = p.x * tmp;
943            q.y = p.y * tmp;
944            q.infinity = Choice::from(0u8);
945
946            *q = G1Affine::conditional_select(q, &G1Affine::identity(), skip);
947        }
948    }

965    /// Performs a Variable Base Multiscalar Multiplication.
966    pub fn msm_variable_base(points: &[G1Projective], scalars: &[Scalar]) -> G1Projective {
967        let c = if scalars.len() < 32 {
968            3
969        } else {
970            ln_without_floats(scalars.len()) + 2
971        };
972
973        let num_bits = 255usize;
974        let fr_one = Scalar::one();
975
976        let zero = G1Projective::identity();
977        let window_starts: Vec<_> = (0..num_bits).step_by(c).collect();
978
979        let window_starts_iter = window_starts.into_iter();
980
981        // Each window is of size `c`.
982        // We divide up the bits 0..num_bits into windows of size `c`, and
983        // in parallel process each such window.
984        let window_sums: Vec<_> = window_starts_iter
985            .map(|w_start| {
986                let mut res = zero;
987                // We don't need the "zero" bucket, so we only have 2^c - 1 buckets
988                let mut buckets = vec![zero; (1 << c) - 1];
989                scalars
990                    .iter()
991                    .zip(points)
992                    .filter(|(s, _)| !(*s == &Scalar::zero()))
993                    .for_each(|(&scalar, base)| {
994                        if scalar == fr_one {
995                            // We only process unit scalars once in the first window.
996                            if w_start == 0 {
997                                res = res.add(base);
998                            }
999                        } else {
1000                            let mut scalar = Scalar::montgomery_reduce(
1001                                scalar.0[0],
1002                                scalar.0[1],
1003                                scalar.0[2],
1004                                scalar.0[3],
1005                                0,
1006                                0,
1007                                0,
1008                                0,
1009                            );
1010
1011                            // We right-shift by w_start, thus getting rid of the
1012                            // lower bits.
1013                            scalar = scalar.divn(w_start as u32);
1014                            // We mod the remaining bits by the window size.
1015                            let scalar = scalar.0[0] % (1 << c);
1016
1017                            // If the scalar is non-zero, we update the corresponding
1018                            // bucket.
1019                            // (Recall that `buckets` doesn't have a zero bucket.)
1020                            if scalar != 0 {
1021                                buckets[(scalar - 1) as usize] =
1022                                    buckets[(scalar - 1) as usize].add(base);
1023                            }
1024                        }
1025                    });
1026
1027                let mut running_sum = G1Projective::identity();
1028                for b in buckets.into_iter().rev() {
1029                    running_sum += b;
1030                    res += &running_sum;
1031                }
1032
1033                res
1034            })
1035            .collect();
1036
1037        // We store the sum for the lowest window.
1038        let lowest = *window_sums.first().unwrap();
1039        // We're traversing windows from high to low.
1040        window_sums[1..]
1041            .iter()
1042            .rev()
1043            .fold(zero, |mut total, sum_i| {
1044                total += sum_i;
1045                for _ in 0..c {
1046                    total = total.double();
1047                }
1048                total
1049            })
1050            + lowest
1051    }
1052}
