21/// Represents results of a Miller loop, one of the most expensive portions
22/// of the pairing function. `MillerLoopResult`s cannot be compared with each
23/// other until `.final_exponentiation()` is called, which is also expensive.
24#[cfg_attr(docsrs, doc(cfg(feature = "pairings")))]
25#[derive(Copy, Clone, Debug)]
26pub struct MillerLoopResult(pub(crate) Fp12);

30        MillerLoopResult(Fp12::one())
31    }
32}

43impl MillerLoopResult {
44    /// This performs a "final exponentiation" routine to convert the result
45    /// of a Miller loop into an element of `Gt` with help of efficient squaring
46    /// operation in the so-called `cyclotomic subgroup` of `Fq6` so that
47    /// it can be compared with other elements of `Gt`.
48    pub fn final_exponentiation(&self) -> Gt {
49        #[must_use]
50        fn fp4_square(a: Fp2, b: Fp2) -> (Fp2, Fp2) {
51            cfg_if::cfg_if! {
52                if #[cfg(target_os = "zkvm")] {
53                    // c0 = b.square().mul_by_nonresidue() + a.square()
54                    // c1 = (a + b).square() - a.square() - b.square()
55                    let mut t0 = a;
56                    t0.square_inp();
57                    let mut c0 = b;
58                    c0.square_inp();
59                    let mut c1 = a;
60                    c1.add_inp(&b);
61                    c1.square_inp();
62                    c1.sub_inp(&t0);
63                    c1.sub_inp(&c0);
64                    c0.mul_by_nonresidue_inp();
65                    c0.add_inp(&t0);
66                    (c0, c1)
67                } else {
68                    let t0 = a.square();
69                    let t1 = b.square();
70                    let mut t2 = t1.mul_by_nonresidue();
71                    let c0 = t2 + t0;
72                    t2 = a + b;
73                    t2 = t2.square();
74                    t2 -= t0;
75                    let c1 = t2 - t1;
76                    (c0, c1)
77                }
78            }
79        }
80        // Adaptation of Algorithm 5.5.4, Guide to Pairing-Based Cryptography
81        // Faster Squaring in the Cyclotomic Subgroup of Sixth Degree Extensions
82        // https://eprint.iacr.org/2009/565.pdf
83        #[must_use]
84        fn cyclotomic_square(f: Fp12) -> Fp12 {
85            let Fp12 {
86                c0:
87                    Fp6 {
88                        c0: mut z0,
89                        c1: mut z4,
90                        c2: mut z3,
91                    },
92                c1:
93                    Fp6 {
94                        c0: mut z2,
95                        c1: mut z1,
96                        c2: mut z5,
97                    },
98            } = f;
99
100            let (t0, t1) = fp4_square(z0, z1);
101
102            // For A
103            cfg_if::cfg_if! {
104                if #[cfg(target_os = "zkvm")] {
105                    z0 = -z0;
106                    z0.add_inp(&t0);
107                    z0.double_inp();
108                    z0.add_inp(&t0);
109
110                    z1.add_inp(&t1);
111                    z1.double_inp();
112                    z1.add_inp(&t1);
113                } else {
114                    z0 = t0 - z0;
115                    z0 = z0 + z0 + t0;
116
117                    z1 = t1 + z1;
118                    z1 = z1 + z1 + t1;
119                }
120            }
121
122            cfg_if::cfg_if! {
123                if #[cfg(target_os = "zkvm")] {
124                    let (t0, t1) = fp4_square(z2, z3);
125                    let (t2, mut t3) = fp4_square(z4, z5);
126                } else {
127                    let (mut t0, t1) = fp4_square(z2, z3);
128                    let (t2, t3) = fp4_square(z4, z5);
129                }
130            }
131
132            // For C
133            cfg_if::cfg_if! {
134                if #[cfg(target_os = "zkvm")] {
135                    z4 = -z4;
136                    z4.add_inp(&t0);
137                    z4.double_inp();
138                    z4.add_inp(&t0);
139
140                    z5.add_inp(&t1);
141                    z5.double_inp();
142                    z5.add_inp(&t1);
143                } else {
144                    z4 = t0 - z4;
145                    z4 = z4 + z4 + t0;
146
147                    z5 = t1 + z5;
148                    z5 = z5 + z5 + t1;
149                }
150            }
151
152            // For B
153            cfg_if::cfg_if! {
154                if #[cfg(target_os = "zkvm")] {
155                    t3.mul_by_nonresidue_inp();
156                    z2.add_inp(&t3);
157                    z2.double_inp();
158                    z2.add_inp(&t3);
159
160                    z3 = -z3;
161                    z3.add_inp(&t2);
162                    z3.double_inp();
163                    z3.add_inp(&t2);
164                } else {
165                    t0 = t3.mul_by_nonresidue();
166                    z2 = t0 + z2;
167                    z2 = z2 + z2 + t0;
168
169                    z3 = t2 - z3;
170                    z3 = z3 + z3 + t2;
171                }
172            }
173
174            Fp12 {
175                c0: Fp6 {
176                    c0: z0,
177                    c1: z4,
178                    c2: z3,
179                },
180                c1: Fp6 {
181                    c0: z2,
182                    c1: z1,
183                    c2: z5,
184                },
185            }
186        }
187        #[cfg(target_os = "zkvm")]
188        fn cyclotomic_square_inp(f: &mut Fp12) {
189            // z0: f.c0.c0
190            // z1: f.c1.c1
191            // z2: f.c1.c0
192            // z3: f.c0.c2
193            // z4: f.c0.c1
194            // z5: f.c1.c2
195
196            let (t0, t1) = fp4_square(f.c0.c0, f.c1.c1);
197
198            // For A
199            f.c0.c0 = -f.c0.c0;
200            f.c0.c0.add_inp(&t0);
201            f.c0.c0.double_inp();
202            f.c0.c0.add_inp(&t0);
203
204            f.c1.c1.add_inp(&t1);
205            f.c1.c1.double_inp();
206            f.c1.c1.add_inp(&t1);
207
208            let (t0, t1) = fp4_square(f.c1.c0, f.c0.c2);
209            let (t2, mut t3) = fp4_square(f.c0.c1, f.c1.c2);
210
211            // For C
212            f.c0.c1 = -f.c0.c1;
213            f.c0.c1.add_inp(&t0);
214            f.c0.c1.double_inp();
215            f.c0.c1.add_inp(&t0);
216
217            f.c1.c2.add_inp(&t1);
218            f.c1.c2.double_inp();
219            f.c1.c2.add_inp(&t1);
220
221            // For B
222            t3.mul_by_nonresidue_inp();
223            f.c1.c0.add_inp(&t3);
224            f.c1.c0.double_inp();
225            f.c1.c0.add_inp(&t3);
226
227            f.c0.c2 = -f.c0.c2;
228            f.c0.c2.add_inp(&t2);
229            f.c0.c2.double_inp();
230            f.c0.c2.add_inp(&t2);
231        }
232        #[must_use]
233        fn cycolotomic_exp(f: &Fp12) -> Fp12 {
234            let x = BLS_X;
235            let mut tmp = Fp12::one();
236            let mut found_one = false;
237            for i in (0..64).rev().map(|b| ((x >> b) & 1) == 1) {
238                if found_one {
239                    cfg_if::cfg_if! {
240                        if #[cfg(target_os = "zkvm")] {
241                            cyclotomic_square_inp(&mut tmp);
242                        } else {
243                            tmp = cyclotomic_square(tmp);
244                        }
245                    }
246                } else {
247                    found_one = i;
248                }
249
250                if i {
251                    cfg_if::cfg_if! {
252                        if #[cfg(target_os = "zkvm")] {
253                            tmp.mul_inp(f);
254                        } else {
255                            tmp *= f;
256                        }
257                    }
258                }
259            }
260
261            tmp.conjugate()
262        }
263
264        cfg_if::cfg_if! {
265            if #[cfg(target_os = "zkvm")] {
266                let mut t0 = self.0;
267                t0.frobenius_map_inp();
268                t0.frobenius_map_inp();
269                t0.frobenius_map_inp();
270                t0.frobenius_map_inp();
271                t0.frobenius_map_inp();
272                t0.frobenius_map_inp();
273
274                let mut t2 = self.0.invert().unwrap();
275                t2.mul_inp(&t0);
276                let mut t1 = t2;
277                t2.frobenius_map_inp();
278                t2.frobenius_map_inp();
279                t2.mul_inp(&t1);
280                t1 = cyclotomic_square(t2);
281                t1.conjugate_inp();
282                let mut t3 = cycolotomic_exp(&t2);
283                let mut t4 = cyclotomic_square(t3);
284                let mut t5 = t1;
285                t5.mul_inp(&t3);
286                t1 = cycolotomic_exp(&t5);
287                t0 = cycolotomic_exp(&t1);
288                let mut t6 = cycolotomic_exp(&t0);
289                t6.mul_inp(&t4);
290                t4 = cycolotomic_exp(&t6);
291                t5.conjugate_inp();
292                t4.mul_inp(&t5);
293                t4.mul_inp(&t2);
294                t1.mul_inp(&t2);
295                t2.conjugate_inp();
296                t1.frobenius_map_inp();
297                t1.frobenius_map_inp();
298                t1.frobenius_map_inp();
299                t6.mul_inp(&t2);
300                t6.frobenius_map_inp();
301                t3.mul_inp(&t0);
302                t3.frobenius_map_inp();
303                t3.frobenius_map_inp();
304                t3.mul_inp(&t1);
305                t3.mul_inp(&t6);
306                t3.mul_inp(&t4);
307                Gt(t3)
308            } else {
309                let mut t0 = self
310                    .0
311                    .frobenius_map()
312                    .frobenius_map()
313                    .frobenius_map()
314                    .frobenius_map()
315                    .frobenius_map()
316                    .frobenius_map();
317
318                // We unwrap() because `MillerLoopResult` can only be constructed
319                // by a function within this crate, and we uphold the invariant
320                // that the enclosed value is nonzero.
321                let mut t1 = self.0.invert().unwrap();
322
323                let mut t2 = t0 * t1;
324                t1 = t2;
325                t2 = t2.frobenius_map().frobenius_map();
326                t2 *= t1;
327                t1 = cyclotomic_square(t2).conjugate();
328                let mut t3 = cycolotomic_exp(&t2);
329                let mut t4 = cyclotomic_square(t3);
330                let mut t5 = t1 * t3;
331                t1 = cycolotomic_exp(&t5);
332                t0 = cycolotomic_exp(&t1);
333                let mut t6 = cycolotomic_exp(&t0);
334                t6 *= t4;
335                t4 = cycolotomic_exp(&t6);
336                t5 = t5.conjugate();
337                t4 *= t5 * t2;
338                t5 = t2.conjugate();
339                t1 *= t2;
340                t1 = t1.frobenius_map().frobenius_map().frobenius_map();
341                t6 *= t5;
342                t6 = t6.frobenius_map();
343                t3 *= t0;
344                t3 = t3.frobenius_map().frobenius_map();
345                t3 *= t1;
346                t3 *= t6;
347                Gt(t3 * t4)
348            }
349        }
350    }

471impl<'a> Mul<&'a Scalar> for &Gt {
472    type Output = Gt;
473
474    fn mul(self, other: &'a Scalar) -> Self::Output {
475        let mut acc = Gt::identity();
476
477        // This is a simple double-and-add implementation of group element
478        // multiplication, moving from most significant to least
479        // significant bit of the scalar.
480        //
481        // We skip the leading bit because it's always unset for Fq
482        // elements.
483        for bit in other
484            .to_bytes()
485            .iter()
486            .rev()
487            .flat_map(|byte| (0..8).rev().map(move |i| Choice::from((byte >> i) & 1u8)))
488            .skip(1)
489        {
490            acc = acc.double();
491            acc = Gt::conditional_select(&acc, &(acc + self), bit);
492        }
493
494        acc
495    }
496}

664/// This structure contains cached computations pertaining to a $\mathbb{G}_2$
665/// element as part of the pairing function (specifically, the Miller loop) and
666/// so should be computed whenever a $\mathbb{G}_2$ element is being used in
667/// multiple pairings or is otherwise known in advance. This should be used in
668/// conjunction with the [`multi_miller_loop`](crate::multi_miller_loop)
669/// function provided by this crate.
670///
671/// Requires the `alloc` and `pairing` crate features to be enabled.
672pub struct G2Prepared {
673    infinity: Choice,
674    coeffs: Vec<(Fp2, Fp2, Fp2)>,
675}
676
677#[cfg(feature = "alloc")]
678impl From<G2Affine> for G2Prepared {
679    fn from(q: G2Affine) -> G2Prepared {
680        struct Adder {
681            cur: G2Projective,
682            base: G2Affine,
683            coeffs: Vec<(Fp2, Fp2, Fp2)>,
684        }
685
686        impl MillerLoopDriver for Adder {
687            type Output = ();
688
689            fn doubling_step(&mut self, _: &mut Self::Output) {
690                let coeffs = doubling_step(&mut self.cur);
691                self.coeffs.push(coeffs);
692            }
693            fn addition_step(&mut self, _: &mut Self::Output) {
694                let coeffs = addition_step(&mut self.cur, &self.base);
695                self.coeffs.push(coeffs);
696            }
697            fn square_output(_: &mut Self::Output) {}
698            fn conjugate(_: &mut Self::Output) {}
699            fn one() -> Self::Output {}
700        }
701
702        let is_identity = q.is_identity();
703        let q = G2Affine::conditional_select(&q, &G2Affine::generator(), is_identity);
704
705        let mut adder = Adder {
706            cur: G2Projective::from(q),
707            base: q,
708            coeffs: Vec::with_capacity(68),
709        };
710
711        miller_loop(&mut adder);
712
713        assert_eq!(adder.coeffs.len(), 68);
714
715        G2Prepared {
716            infinity: is_identity,
717            coeffs: adder.coeffs,
718        }
719    }
720}

724/// Computes $$\sum_{i=1}^n \textbf{ML}(a_i, b_i)$$ given a series of terms
725/// $$(a_1, b_1), (a_2, b_2), ..., (a_n, b_n).$$
726///
727/// Requires the `alloc` and `pairing` crate features to be enabled.
728pub fn multi_miller_loop(terms: &[(&G1Affine, &G2Prepared)]) -> MillerLoopResult {
729    struct Adder<'a, 'b, 'c> {
730        terms: &'c [(&'a G1Affine, &'b G2Prepared)],
731        index: usize,
732    }
733
734    impl MillerLoopDriver for Adder<'_, '_, '_> {
735        type Output = Fp12;
736
737        fn doubling_step(&mut self, f: &mut Self::Output) {
738            let index = self.index;
739            for term in self.terms {
740                let either_identity = term.0.is_identity() | term.1.infinity;
741
742                let new_f = ell(f, &term.1.coeffs[index], term.0);
743                *f = Fp12::conditional_select(&new_f, f, either_identity);
744            }
745            self.index += 1;
746        }
747        fn addition_step(&mut self, f: &mut Self::Output) {
748            let index = self.index;
749            for term in self.terms {
750                let either_identity = term.0.is_identity() | term.1.infinity;
751
752                let new_f = ell(f, &term.1.coeffs[index], term.0);
753                *f = Fp12::conditional_select(&new_f, f, either_identity);
754            }
755            self.index += 1;
756        }
757        fn square_output(f: &mut Self::Output) {
758            *f = f.square();
759        }
760        fn conjugate(f: &mut Self::Output) {
761            f.conjugate_inp();
762        }
763        fn one() -> Self::Output {
764            Fp12::one()
765        }
766    }
767
768    let mut adder = Adder { terms, index: 0 };
769
770    MillerLoopResult(miller_loop(&mut adder))
771}

773/// Invoke the pairing function without the use of precomputation and other optimizations.
774#[cfg_attr(docsrs, doc(cfg(feature = "pairings")))]
775pub fn pairing(p: &G1Affine, q: &G2Affine) -> Gt {
776    struct Adder {
777        cur: G2Projective,
778        base: G2Affine,
779        p: G1Affine,
780    }
781
782    impl MillerLoopDriver for Adder {
783        type Output = Fp12;
784
785        fn doubling_step(&mut self, f: &mut Self::Output) {
786            let coeffs = doubling_step(&mut self.cur);
787            *f = ell(f, &coeffs, &self.p);
788        }
789        fn addition_step(&mut self, f: &mut Self::Output) {
790            let coeffs = addition_step(&mut self.cur, &self.base);
791            *f = ell(f, &coeffs, &self.p);
792        }
793        fn square_output(f: &mut Self::Output) {
794            *f = f.square();
795        }
796        fn conjugate(f: &mut Self::Output) {
797            f.conjugate_inp();
798        }
799        fn one() -> Self::Output {
800            Fp12::one()
801        }
802    }
803
804    let either_identity = p.is_identity() | q.is_identity();
805    let p = G1Affine::conditional_select(p, &G1Affine::generator(), either_identity);
806    let q = G2Affine::conditional_select(q, &G2Affine::generator(), either_identity);
807
808    let mut adder = Adder {
809        cur: G2Projective::from(q),
810        base: q,
811        p,
812    };
813
814    let tmp = miller_loop(&mut adder);
815    let tmp = MillerLoopResult(Fp12::conditional_select(
816        &tmp,
817        &Fp12::one(),
818        either_identity,
819    ));
820    tmp.final_exponentiation()
821}

823trait MillerLoopDriver {
824    type Output;
825
826    fn doubling_step(&mut self, f: &mut Self::Output);
827    fn addition_step(&mut self, f: &mut Self::Output);
828    fn square_output(f: &mut Self::Output);
829    fn conjugate(f: &mut Self::Output);
830    fn one() -> Self::Output;
831}

833/// This is a "generic" implementation of the Miller loop to avoid duplicating code
834/// structure elsewhere; instead, we'll write concrete instantiations of
835/// `MillerLoopDriver` for whatever purposes we need (such as caching modes).
836fn miller_loop<D: MillerLoopDriver>(driver: &mut D) -> D::Output {
837    let mut f = D::one();
838
839    let mut found_one = false;
840    for i in (0..64).rev().map(|b| (((BLS_X >> 1) >> b) & 1) == 1) {
841        if !found_one {
842            found_one = i;
843            continue;
844        }
845
846        driver.doubling_step(&mut f);
847
848        if i {
849            driver.addition_step(&mut f);
850        }
851
852        D::square_output(&mut f);
853    }
854
855    driver.doubling_step(&mut f);
856
857    if BLS_X_IS_NEGATIVE {
858        D::conjugate(&mut f);
859    }
860
861    f
862}

864#[cfg(target_os = "zkvm")]
865fn ell(f: &Fp12, coeffs: &(Fp2, Fp2, Fp2), p: &G1Affine) -> Fp12 {
866    let mut c0 = coeffs.0;
867    let mut c1 = coeffs.1;
868
869    c0.c0.mul_inp(&p.y);
870    c0.c1.mul_inp(&p.y);
871
872    c1.c0.mul_inp(&p.x);
873    c1.c1.mul_inp(&p.x);
874
875    f.mul_by_014(&coeffs.2, &c1, &c0)
876}

879fn ell(f: &Fp12, coeffs: &(Fp2, Fp2, Fp2), p: &G1Affine) -> Fp12 {
880    let mut c0 = coeffs.0;
881    let mut c1 = coeffs.1;
882
883    c0.c0 *= p.y;
884    c0.c1 *= p.y;
885
886    c1.c0 *= p.x;
887    c1.c1 *= p.x;
888
889    f.mul_by_014(&coeffs.2, &c1, &c0)
890}

892#[cfg(target_os = "zkvm")]
893fn doubling_step(r: &mut G2Projective) -> (Fp2, Fp2, Fp2) {
894    // Adaptation of Algorithm 26, https://eprint.iacr.org/2010/354.pdf
895    let mut tmp0 = r.x;
896    tmp0.square_inp();
897    let mut tmp1 = r.y;
898    tmp1.square_inp();
899    let mut tmp2 = tmp1;
900    tmp2.square_inp();
901    let mut tmp3 = tmp1;
902    tmp3.add_inp(&r.x);
903    tmp3.square_inp();
904    tmp3.sub_inp(&tmp0);
905    tmp3.sub_inp(&tmp2);
906    tmp3.double_inp();
907    let mut tmp4 = tmp0;
908    tmp4.add_inp(&tmp0);
909    tmp4.add_inp(&tmp0);
910    let mut tmp6 = r.x;
911    tmp6.add_inp(&tmp4);
912    let mut tmp5 = tmp4;
913    tmp5.square_inp();
914    let mut zsquared = r.z;
915    zsquared.square_inp();
916    r.x = tmp5;
917    r.x.sub_inp(&tmp3);
918    r.x.sub_inp(&tmp3);
919    r.z.add_inp(&r.y);
920    r.z.square_inp();
921    r.z.sub_inp(&tmp1);
922    r.z.sub_inp(&zsquared);
923    r.y = tmp3;
924    r.y.sub_inp(&r.x);
925    r.y.mul_inp(&tmp4);
926    tmp2.double_inp();
927    tmp2.double_inp();
928    tmp2.double_inp();
929    r.y.sub_inp(&tmp2);
930    let mut tmp3 = tmp4;
931    tmp3.mul_inp(&zsquared);
932    tmp3.double_inp();
933    let tmp3 = -tmp3;
934    tmp6.square_inp();
935    tmp6.sub_inp(&tmp0);
936    tmp6.sub_inp(&tmp5);
937    tmp1.double_inp();
938    tmp1.double_inp();
939    tmp6.sub_inp(&tmp1);
940    let mut tmp0 = r.z;
941    tmp0.mul_inp(&zsquared);
942    tmp0.double_inp();
943
944    (tmp0, tmp3, tmp6)
945}

947#[cfg(not(target_os = "zkvm"))]
948fn doubling_step(r: &mut G2Projective) -> (Fp2, Fp2, Fp2) {
949    // Adaptation of Algorithm 26, https://eprint.iacr.org/2010/354.pdf
950    let tmp0 = r.x.square();
951    let tmp1 = r.y.square();
952    let tmp2 = tmp1.square();
953    let tmp3 = (tmp1 + r.x).square() - tmp0 - tmp2;
954    let tmp3 = tmp3 + tmp3;
955    let tmp4 = tmp0 + tmp0 + tmp0;
956    let tmp6 = r.x + tmp4;
957    let tmp5 = tmp4.square();
958    let zsquared = r.z.square();
959    r.x = tmp5 - tmp3 - tmp3;
960    r.z = (r.z + r.y).square() - tmp1 - zsquared;
961    r.y = (tmp3 - r.x) * tmp4;
962    let tmp2 = tmp2 + tmp2;
963    let tmp2 = tmp2 + tmp2;
964    let tmp2 = tmp2 + tmp2;
965    r.y -= tmp2;
966    let tmp3 = tmp4 * zsquared;
967    let tmp3 = tmp3 + tmp3;
968    let tmp3 = -tmp3;
969    let tmp6 = tmp6.square() - tmp0 - tmp5;
970    let tmp1 = tmp1 + tmp1;
971    let tmp1 = tmp1 + tmp1;
972    let tmp6 = tmp6 - tmp1;
973    let tmp0 = r.z * zsquared;
974    let tmp0 = tmp0 + tmp0;
975
976    (tmp0, tmp3, tmp6)
977}

979#[cfg(target_os = "zkvm")]
980fn addition_step(r: &mut G2Projective, q: &G2Affine) -> (Fp2, Fp2, Fp2) {
981    // Adaptation of Algorithm 27, https://eprint.iacr.org/2010/354.pdf
982    let mut zsquared = r.z;
983    zsquared.square_inp();
984    let mut ysquared = q.y;
985    ysquared.square_inp();
986    let mut t0 = q.y;
987    t0.add_inp(&r.z);
988    t0.square_inp();
989    t0.sub_inp(&ysquared);
990    t0.sub_inp(&zsquared);
991    t0.mul_inp(&zsquared);
992    let mut t1 = zsquared;
993    t1.mul_inp(&q.x);
994    t1.sub_inp(&r.x);
995    let mut t2 = t1;
996    t2.square_inp();
997    let mut t3 = t2;
998    t3.double_inp();
999    t3.double_inp();
1000    let mut t4 = t3;
1001    t4.mul_inp(&t1);
1002    t0.sub_inp(&r.y);
1003    t0.sub_inp(&r.y);
1004    let mut t5 = t0;
1005    t5.mul_inp(&q.x);
1006    t3.mul_inp(&r.x);
1007    r.x = t0;
1008    r.x.square_inp();
1009    r.x.sub_inp(&t4);
1010    r.x.sub_inp(&t3);
1011    r.x.sub_inp(&t3);
1012    r.z.add_inp(&t1);
1013    r.z.square_inp();
1014    r.z.sub_inp(&zsquared);
1015    r.z.sub_inp(&t2);
1016    let mut t6 = q.y;
1017    t6.add_inp(&r.z);
1018    t3.sub_inp(&r.x);
1019    t3.mul_inp(&t0);
1020    t4.mul_inp(&r.y);
1021    t4.double_inp();
1022    r.y = t3;
1023    r.y.sub_inp(&t4);
1024    t6.square_inp();
1025    t6.sub_inp(&ysquared);
1026    let mut ztsquared = r.z;
1027    ztsquared.square_inp();
1028    t6.sub_inp(&ztsquared);
1029    t5.double_inp();
1030    t5.sub_inp(&t6);
1031    let mut t6 = r.z;
1032    t6.double_inp();
1033    let mut t0 = -t0;
1034    t0.double_inp();
1035
1036    (t6, t0, t5)
1037}

1039#[cfg(not(target_os = "zkvm"))]
1040fn addition_step(r: &mut G2Projective, q: &G2Affine) -> (Fp2, Fp2, Fp2) {
1041    // Adaptation of Algorithm 27, https://eprint.iacr.org/2010/354.pdf
1042    let zsquared = r.z.square();
1043    let ysquared = q.y.square();
1044    let t0 = zsquared * q.x;
1045    let t1 = ((q.y + r.z).square() - ysquared - zsquared) * zsquared;
1046    let t2 = t0 - r.x;
1047    let t3 = t2.square();
1048    let t4 = t3 + t3;
1049    let t4 = t4 + t4;
1050    let t5 = t4 * t2;
1051    let t6 = t1 - r.y - r.y;
1052    let t9 = t6 * q.x;
1053    let t7 = t4 * r.x;
1054    r.x = t6.square() - t5 - t7 - t7;
1055    r.z = (r.z + t2).square() - zsquared - t3;
1056    let t10 = q.y + r.z;
1057    let t8 = (t7 - r.x) * t6;
1058    let t0 = r.y * t5;
1059    let t0 = t0 + t0;
1060    r.y = t8 - t0;
1061    let t10 = t10.square() - ysquared;
1062    let ztsquared = r.z.square();
1063    let t10 = t10 - ztsquared;
1064    let t9 = t9 + t9 - t10;
1065    let t10 = r.z + r.z;
1066    let t6 = -t6;
1067    let t1 = t6 + t6;
1068
1069    (t10, t1, t9)
1070}
