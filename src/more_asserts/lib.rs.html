1//! Small library providing some macros helpful for asserting. The API is very
2//! similar to the API provided by the stdlib's own `assert_eq!`, `assert_ne!`,
3//! `debug_assert_eq!`, or `debug_assert_ne!`.
4//!
5//! | Name                 | Enabled                       | Equivalent to                                |
6//! | -------------------- | ----------------------------- | -------------------------------------------- |
7//! | `assert_le!`         | Always                        | `assert!(a <= b)`                            |
8//! | `assert_lt!`         | Always                        | `assert!(a < b)`                             |
9//! | `assert_ge!`         | Always                        | `assert!(a >= b)`                            |
10//! | `assert_gt!`         | Always                        | `assert!(a > b)`                             |
11//! | `debug_assert_le!`   | `if cfg!(debug_assertions)`   | `debug_assert!(a <= b)`                      |
12//! | `debug_assert_lt!`   | `if cfg!(debug_assertions)`   | `debug_assert!(a < b)`                       |
13//! | `debug_assert_ge!`   | `if cfg!(debug_assertions)`   | `debug_assert!(a >= b)`                      |
14//! | `debug_assert_gt!`   | `if cfg!(debug_assertions)`   | `debug_assert!(a > b)`                       |
15//! | `debug_unreachable!` | `if cfg!(debug_assertions)`   | `unreachable!` when debug_assertions are on. |
16//!
17//! When one of the assertions fails, it prints out a message like the following:
18//!
19//! ```text
20//! thread 'main' panicked at 'assertion failed: `left < right`
21//!  left: `4`,
22//! right: `3`', src/main.rs:47:5
23//! note: Run with `RUST_BACKTRACE=1` for a backtrace.
24//! ```
25//!
26//! # Example
27//!
28//! ```rust
29//! #[macro_use]
30//! extern crate more_asserts;
31//!
32//! #[derive(Debug, PartialEq, PartialOrd)]
33//! enum Example { Foo, Bar }
34//!
35//! fn main() {
36//!     assert_le!(3, 4);
37//!     assert_ge!(10, 10,
38//!                "You can pass a message too (just like `assert_eq!`)");
39//!     debug_assert_lt!(1.3, 4.5,
40//!         "Format syntax is supported ({}).", "also like `assert_eq!`");
41//!
42//!     assert_gt!(Example::Bar, Example::Foo,
43//!         "It works on anything that implements PartialOrd, PartialEq, and Debug!");
44//! }
45//! ```

53/// Panics if the first expression is not strictly less than the second.
54/// Requires that the values be comparable with `<`.
55///
56/// On failure, panics and prints the values out in a manner similar to
57/// prelude's `assert_eq!`.
58///
59/// Optionally may take an additional message to display on failure, which
60/// is formatted using standard format syntax.
61///
62/// # Example
63///
64/// ```rust
65/// #[macro_use]
66/// extern crate more_asserts;
67///
68/// fn main() {
69///     assert_lt!(3, 4);
70///     assert_lt!(3, 4, "With a message");
71///     assert_lt!(3, 4, "With a formatted message: {}", "oh no");
72/// }
73/// ```
74#[macro_export]
75macro_rules! assert_lt {
76    ($left:expr, $right:expr) => ({
77        let (left, right) = (&($left), &($right));
78        if !(left < right) {
79            panic!("assertion failed: `(left < right)`\n  left: `{:?}`,\n right: `{:?}`",
80                   left, right);
81        }
82    });
83    ($left:expr, $right:expr, ) => ({
84        assert_lt!($left, $right);
85    });
86    ($left:expr, $right:expr, $($msg_args:tt)+) => ({
87        let (left, right) = (&($left), &($right));
88        if !(left < right) {
89            panic!("assertion failed: `(left < right)`\n  left: `{:?}`,\n right: `{:?}`: {}",
90                   left, right, format_args!($($msg_args)+));
91        }
92    })
93}

223/// Same as `assert_lt!` in debug builds or release builds where the
224/// `-C debug-assertions` was provided to the compiler. For all other builds,
225/// vanishes without a trace.
226///
227/// Optionally may take an additional message to display on failure, which
228/// is formatted using standard format syntax.
229///
230/// # Example
231///
232/// ```rust
233/// #[macro_use]
234/// extern crate more_asserts;
235///
236/// fn main() {
237///     // These are compiled to nothing if debug_assertions are off!
238///     debug_assert_lt!(3, 4);
239///     debug_assert_lt!(3, 4, "With a message");
240///     debug_assert_lt!(3, 4, "With a formatted message: {}", "oh no");
241/// }
242/// ```
243#[macro_export]
244macro_rules! debug_assert_lt {
245    ($($arg:tt)+) => {
246        if cfg!(debug_assertions) {
247            assert_lt!($($arg)+);
248        }
249    }
250}

341/// Panics if reached. This is a variant of the standard library's `unreachable!`
342/// macro that is controlled by `cfg!(debug_assertions)`.
343///
344/// Same as prelude's `unreachable!` in debug builds or release builds where the
345/// `-C debug-assertions` was provided to the compiler. For all other builds,
346/// vanishes without a trace.
347///
348/// # Example
349///
350/// ```rust
351/// #[macro_use]
352/// extern crate more_asserts;
353///
354/// fn main() {
355///     // You probably wouldn't actually use this here
356///     let mut value = 0.5;
357///     if value < 0.0 {
358///         debug_unreachable!("Value out of range {}", value);
359///         value = 0.0;
360///     }
361/// }
362/// ```
363#[macro_export]
364macro_rules! debug_unreachable {
365    ($($arg:tt)*) => {
366        if cfg!(debug_assertions) {
367            unreachable!($($arg)*);
368        }
369    }
370}

376    #[derive(PartialOrd, PartialEq, Debug)]

378        Foo, Bar, Baz
379    }

381    #[test]
382    fn test_assert_lt() {
383        assert_lt!(3, 4);
384        assert_lt!(4.0, 4.5);
385        assert_lt!("a string", "b string");
386        assert_lt!(DummyType::Foo, DummyType::Bar,
387                   "Message with {}", "cool formatting");
388
389        let a = &DummyType::Foo;
390        let b = &DummyType::Baz;
391        assert_lt!(a, b);
392
393        assert!(catch_unwind(|| assert_lt!(5, 3)).is_err());
394        assert!(catch_unwind(|| assert_lt!(5, 5)).is_err());
