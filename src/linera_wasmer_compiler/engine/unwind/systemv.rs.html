4//! Module for System V ABI unwind registry.
5
6use core::sync::atomic::{AtomicUsize, Ordering::Relaxed};
7
8use wasmer_types::CompiledFunctionUnwindInfoReference;
9
10/// Represents a registry of function unwind information for System V ABI.
11pub struct UnwindRegistry {
12    registrations: Vec<usize>,
13    published: bool,
14}
15
16extern "C" {
17    // libunwind import
18    fn __register_frame(fde: *const u8);
19    fn __deregister_frame(fde: *const u8);
20}
21
22/// There are two primary unwinders on Unix platforms: libunwind and libgcc.
23///
24/// Unfortunately their interface to `__register_frame` is different. The
25/// libunwind library takes a pointer to an individual FDE while libgcc takes a
26/// null-terminated list of FDEs. This means we need to know what unwinder
27/// is being used at runtime.
28///
29/// This detection is done currently by looking for a libunwind-specific symbol.
30/// This specific symbol was somewhat recommended by LLVM's
31/// "RTDyldMemoryManager.cpp" file which says:
32///
33/// > We use the presence of __unw_add_dynamic_fde to detect libunwind.
34///
35/// I'll note that there's also a different libunwind project at
36/// https://www.nongnu.org/libunwind/ but that doesn't appear to have
37/// `__register_frame` so I don't think that interacts with this.
38fn using_libunwind() -> bool {
39    static USING_LIBUNWIND: AtomicUsize = AtomicUsize::new(LIBUNWIND_UNKNOWN);
40
41    const LIBUNWIND_UNKNOWN: usize = 0;
42    const LIBUNWIND_YES: usize = 1;
43    const LIBUNWIND_NO: usize = 2;
44
45    // On macOS the libgcc interface is never used so libunwind is always used.
46    if cfg!(target_os = "macos") {
47        return true;
48    }
49
50    // On other platforms the unwinder can vary. Sometimes the unwinder is
51    // selected at build time and sometimes it differs at build time and runtime
52    // (or at least I think that's possible). Fall back to a `libc::dlsym` to
53    // figure out what we're using and branch based on that.
54    //
55    // Note that the result of `libc::dlsym` is cached to only look this up
56    // once.
57    match USING_LIBUNWIND.load(Relaxed) {
58        LIBUNWIND_YES => true,
59        LIBUNWIND_NO => false,
60        LIBUNWIND_UNKNOWN => {
61            let looks_like_libunwind = unsafe {
62                !libc::dlsym(
63                    std::ptr::null_mut(),
64                    "__unw_add_dynamic_fde\0".as_ptr().cast(),
65                )
66                .is_null()
67            };
68            USING_LIBUNWIND.store(
69                if looks_like_libunwind {
70                    LIBUNWIND_YES
71                } else {
72                    LIBUNWIND_NO
73                },
74                Relaxed,
75            );
76            looks_like_libunwind
77        }
78        _ => unreachable!(),
79    }
80}
81
82impl UnwindRegistry {
83    /// Creates a new unwind registry with the given base address.
84    pub fn new() -> Self {
85        Self {
86            registrations: Vec::new(),
87            published: false,
88        }
89    }
90
91    /// Registers a function given the start offset, length, and unwind information.
92    pub fn register(
93        &mut self,
94        _base_address: usize,
95        _func_start: u32,
96        _func_len: u32,
97        info: &CompiledFunctionUnwindInfoReference,
98    ) -> Result<(), String> {
99        match info {
100            CompiledFunctionUnwindInfoReference::Dwarf => {}
101            _ => return Err(format!("unsupported unwind information {info:?}")),
102        };
103        Ok(())
104    }
105
106    /// Publishes all registered functions.
107    pub fn publish(&mut self, eh_frame: Option<&[u8]>) -> Result<(), String> {
108        if self.published {
109            return Err("unwind registry has already been published".to_string());
110        }
111
112        if let Some(eh_frame) = eh_frame {
113            unsafe {
114                self.register_frames(eh_frame);
115            }
116        }
117
118        self.published = true;
119
120        Ok(())
121    }
122
123    #[allow(clippy::cast_ptr_alignment)]
124    unsafe fn register_frames(&mut self, eh_frame: &[u8]) {
125        if !using_libunwind() {
126            // Registering an empty `eh_frame` (i.e. which
127            // contains empty FDEs) cause problems on Linux when
128            // deregistering it. We must avoid this
129            // scenario. Usually, this is handled upstream by the
130            // compilers.
131            debug_assert_ne!(
132                eh_frame,
133                &[0, 0, 0, 0],
134                "`eh_frame` seems to contain empty FDEs"
135            );
136
137            // On gnu (libgcc), `__register_frame` will walk the FDEs until an entry of length 0
138            let ptr = eh_frame.as_ptr();
139            __register_frame(ptr);
140            self.registrations.push(ptr as usize);
141        } else {
142            // For libunwind, `__register_frame` takes a pointer to a single FDE
143            let start = eh_frame.as_ptr();
144            let end = start.add(eh_frame.len());
145            let mut current = start;
146
147            // Walk all of the entries in the frame table and register them
148            while current < end {
149                let len = std::ptr::read::<u32>(current as *const u32) as usize;
150
151                // Skip over the CIE and zero-length FDEs.
152                // LLVM's libunwind emits a warning on zero-length FDEs.
153                if current != start && len != 0 {
154                    __register_frame(current);
155                    self.registrations.push(current as usize);
156                }
157
158                // Move to the next table entry (+4 because the length itself is not inclusive)
159                current = current.add(len + 4);
160            }
161        }
162    }
163}
164
165impl Drop for UnwindRegistry {
166    fn drop(&mut self) {
167        if self.published {
168            unsafe {
169                // libgcc stores the frame entries as a linked list in decreasing sort order
170                // based on the PC value of the registered entry.
171                //
172                // As we store the registrations in increasing order, it would be O(N^2) to
173                // deregister in that order.
174                //
175                // To ensure that we just pop off the first element in the list upon every
176                // deregistration, walk our list of registrations backwards.
177                for fde in self.registrations.iter().rev() {
178                    __deregister_frame(*fde as *const _);
179                }
180            }
181        }
182    }
183}
