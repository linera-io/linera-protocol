1use super::OtlpHttpClient;
2use http::{header::CONTENT_TYPE, Method};
3use opentelemetry::otel_debug;
4use opentelemetry_sdk::error::{OTelSdkError, OTelSdkResult};
5use opentelemetry_sdk::logs::{LogBatch, LogExporter};
6use std::time;
7
8impl LogExporter for OtlpHttpClient {
9    async fn export(&self, batch: LogBatch<'_>) -> OTelSdkResult {
10        let client = self
11            .client
12            .lock()
13            .map_err(|e| OTelSdkError::InternalFailure(format!("Mutex lock failed: {}", e)))?
14            .clone()
15            .ok_or(OTelSdkError::AlreadyShutdown)?;
16
17        let (body, content_type) = self
18            .build_logs_export_body(batch)
19            .map_err(OTelSdkError::InternalFailure)?;
20
21        let mut request = http::Request::builder()
22            .method(Method::POST)
23            .uri(&self.collector_endpoint)
24            .header(CONTENT_TYPE, content_type)
25            .body(body.into())
26            .map_err(|e| OTelSdkError::InternalFailure(e.to_string()))?;
27
28        for (k, v) in &self.headers {
29            request.headers_mut().insert(k.clone(), v.clone());
30        }
31
32        let request_uri = request.uri().to_string();
33        otel_debug!(name: "HttpLogsClient.CallingExport");
34        let response = client
35            .send_bytes(request)
36            .await
37            .map_err(|e| OTelSdkError::InternalFailure(format!("{e:?}")))?;
38        if !response.status().is_success() {
39            let error = format!(
40                "OpenTelemetry logs export failed. Url: {}, Status Code: {}, Response: {:?}",
41                request_uri,
42                response.status().as_u16(),
43                response.body()
44            );
45            return Err(OTelSdkError::InternalFailure(error));
46        }
47        Ok(())
48    }
49
50    fn shutdown_with_timeout(&self, _timeout: time::Duration) -> OTelSdkResult {
51        let mut client_guard = self.client.lock().map_err(|e| {
52            OTelSdkError::InternalFailure(format!("Failed to acquire client lock: {}", e))
53        })?;
54
55        if client_guard.take().is_none() {
56            return Err(OTelSdkError::AlreadyShutdown);
57        }
58
59        Ok(())
60    }
61
62    fn set_resource(&mut self, resource: &opentelemetry_sdk::Resource) {
63        self.resource = resource.into();
64    }
65}
