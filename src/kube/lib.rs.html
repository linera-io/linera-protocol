1//! Kube is an umbrella-crate for interacting with [Kubernetes](http://kubernetes.io) in Rust.
2//!
3//! # Overview
4//!
5//! Kube contains a Kubernetes client, a controller runtime, a custom resource derive, and various tooling
6//! required for building applications or controllers that interact with Kubernetes.
7//!
8//! The main modules are:
9//!
10//! - [`client`] with the Kubernetes [`Client`] and its layers
11//! - [`config`] for cluster [`Config`]
12//! - [`api`] with the generic Kubernetes [`Api`]
13//! - [`derive`](kube_derive) with the [`CustomResource`] derive for building controllers types
14//! - [`runtime`] with a [`Controller`](crate::runtime::Controller) / [`watcher`](crate::runtime::watcher()) / [`reflector`](crate::runtime::reflector::reflector) / [`Store`](crate::runtime::reflector::Store)
15//! - [`core`] with generics from `apimachinery`
16//!
17//! You can use each of these as you need with the help of the [exported features](https://github.com/kube-rs/kube/blob/main/kube/Cargo.toml#L18).
18//!
19//! # Using the Client
20//! ```no_run
21//! use futures::{StreamExt, TryStreamExt};
22//! use kube::{Client, api::{Api, ResourceExt, ListParams, PostParams}};
23//! use k8s_openapi::api::core::v1::Pod;
24//!
25//! #[tokio::main]
26//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
27//!     // Infer the runtime environment and try to create a Kubernetes Client
28//!     let client = Client::try_default().await?;
29//!
30//!     // Read pods in the configured namespace into the typed interface from k8s-openapi
31//!     let pods: Api<Pod> = Api::default_namespaced(client);
32//!     for p in pods.list(&ListParams::default()).await? {
33//!         println!("found pod {}", p.name_any());
34//!     }
35//!     Ok(())
36//! }
37//! ```
38//!
39//! For details, see:
40//!
41//! - [`Client`](crate::client) for the extensible Kubernetes client
42//! - [`Api`] for the generic api methods available on Kubernetes resources
43//! - [k8s-openapi](https://docs.rs/k8s-openapi/*/k8s_openapi/) for documentation about the generated Kubernetes types
44//!
45//! # Using the Runtime with the Derive macro
46//!
47//! ```no_run
48//! use schemars::JsonSchema;
49//! use serde::{Deserialize, Serialize};
50//! use serde_json::json;
51//! use futures::{StreamExt, TryStreamExt};
52//! use k8s_openapi::apiextensions_apiserver::pkg::apis::apiextensions::v1::CustomResourceDefinition;
53//! use kube::{
54//!     api::{Api, DeleteParams, PatchParams, Patch, ResourceExt},
55//!     core::CustomResourceExt,
56//!     Client, CustomResource,
57//!     runtime::{watcher, WatchStreamExt, wait::{conditions, await_condition}},
58//! };
59//!
60//! // Our custom resource
61//! #[derive(CustomResource, Deserialize, Serialize, Clone, Debug, JsonSchema)]
62//! #[kube(group = "clux.dev", version = "v1", kind = "Foo", namespaced)]
63//! pub struct FooSpec {
64//!     info: String,
65//!     #[schemars(length(min = 3))]
66//!     name: String,
67//!     replicas: i32,
68//! }
69//!
70//! #[tokio::main]
71//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
72//!     let client = Client::try_default().await?;
73//!     let crds: Api<CustomResourceDefinition> = Api::all(client.clone());
74//!
75//!     // Apply the CRD so users can create Foo instances in Kubernetes
76//!     crds.patch("foos.clux.dev",
77//!         &PatchParams::apply("my_manager"),
78//!         &Patch::Apply(Foo::crd())
79//!     ).await?;
80//!
81//!     // Wait for the CRD to be ready
82//!     tokio::time::timeout(
83//!         std::time::Duration::from_secs(10),
84//!         await_condition(crds, "foos.clux.dev", conditions::is_crd_established())
85//!     ).await?;
86//!
87//!     // Watch for changes to foos in the configured namespace
88//!     let foos: Api<Foo> = Api::default_namespaced(client.clone());
89//!     let wc = watcher::Config::default();
90//!     let mut apply_stream = watcher(foos, wc).applied_objects().boxed();
91//!     while let Some(f) = apply_stream.try_next().await? {
92//!         println!("saw apply to {}", f.name_any());
93//!     }
94//!     Ok(())
95//! }
96//! ```
97//!
98//! For details, see:
99//!
100//! - [`CustomResource`] for documentation how to configure custom resources
101//! - [`runtime::watcher`](crate::runtime::watcher()) for how to long-running watches work and why you want to use this over [`Api::watch`](crate::Api::watch)
102//! - [`runtime`] for abstractions that help with more complicated Kubernetes application
103//!
104//! # Examples
105//! A large list of complete, runnable examples with explainations are available in the [examples folder](https://github.com/kube-rs/kube/tree/main/examples).

238    async fn derived_resource_queriable_and_has_subresources() -> Result<(), Box<dyn std::error::Error>> {
239        use crate::runtime::wait::{await_condition, conditions};
240
241        use serde_json::json;
242        let client = Client::try_default().await?;
243        let ssapply = PatchParams::apply("kube").force();
244        let crds: Api<CustomResourceDefinition> = Api::all(client.clone());
245        // Server-side apply CRD and wait for it to get ready
246        crds.patch("foos.clux.dev", &ssapply, &Patch::Apply(Foo::crd()))
247            .await?;
248        let establish = await_condition(crds.clone(), "foos.clux.dev", conditions::is_crd_established());
249        let _ = tokio::time::timeout(std::time::Duration::from_secs(10), establish).await?;
250        // Use it
251        let foos: Api<Foo> = Api::default_namespaced(client.clone());
252        // Apply from generated struct
253        {
254            let foo = Foo::new("baz", FooSpec {
255                name: "baz".into(),
256                info: Some("old baz".into()),
257                replicas: 1,
258            });
259            let o = foos.patch("baz", &ssapply, &Patch::Apply(&foo)).await?;
260            assert_eq!(o.spec.name, "baz");
261            let oref = o.object_ref(&());
262            assert_eq!(oref.name.unwrap(), "baz");
263            assert_eq!(oref.uid, o.uid());
264        }
265        // Apply from partial json!
266        {
267            let patch = json!({
268                "apiVersion": "clux.dev/v1",
269                "kind": "Foo",
270                "spec": {
271                    "name": "foo",
272                    "replicas": 2
273                }
274            });
275            let o = foos.patch("baz", &ssapply, &Patch::Apply(patch)).await?;
276            assert_eq!(o.spec.replicas, 2, "patching spec updated spec.replicas");
277        }
278        // check subresource
279        {
280            let scale = foos.get_scale("baz").await?;
281            assert_eq!(scale.spec.unwrap().replicas, Some(2));
282            let status = foos.get_status("baz").await?;
283            assert!(status.status.is_none(), "nothing has set status");
284        }
285        // set status subresource
286        {
287            let fs = serde_json::json!({"status": FooStatus { is_bad: false, replicas: 1 }});
288            let o = foos
289                .patch_status("baz", &Default::default(), &Patch::Merge(&fs))
290                .await?;
291            assert!(o.status.is_some(), "status set after patch_status");
292        }
293        // set scale subresource
294        {
295            let fs = serde_json::json!({"spec": { "replicas": 3 }});
296            let o = foos
297                .patch_scale("baz", &Default::default(), &Patch::Merge(&fs))
298                .await?;
299            assert_eq!(o.status.unwrap().replicas, 1, "scale replicas got patched");
300            let linked_replicas = o.spec.unwrap().replicas.unwrap();
301            assert_eq!(linked_replicas, 3, "patch_scale updates linked spec.replicas");
302        }
303
304        // cleanup
305        foos.delete_collection(&DeleteParams::default(), &Default::default())
306            .await?;
307        crds.delete("foos.clux.dev", &DeleteParams::default()).await?;
308        Ok(())
309    }

313    async fn custom_serialized_objects_are_queryable_and_iterable() -> Result<(), Box<dyn std::error::Error>>
314    {
315        use crate::core::{
316            object::{HasSpec, HasStatus, NotUsed, Object},
317            ApiResource,
318        };
319        use k8s_openapi::api::core::v1::Pod;
320        #[derive(Clone, Deserialize, Debug)]
321        struct PodSpecSimple {
322            containers: Vec<ContainerSimple>,
323        }
324        #[derive(Clone, Deserialize, Debug)]
325        struct ContainerSimple {
326            #[allow(dead_code)]
327            image: String,
328        }
329        type PodSimple = Object<PodSpecSimple, NotUsed>;
330
331        // use known type information from pod (can also use discovery for this)
332        let ar = ApiResource::erase::<Pod>(&());
333
334        let client = Client::try_default().await?;
335        let api: Api<PodSimple> = Api::default_namespaced_with(client, &ar);
336        let mut list = api.list(&Default::default()).await?;
337        // check we can mutably iterate over ObjectList
338        for pod in &mut list {
339            pod.spec_mut().containers = vec![];
340            *pod.status_mut() = None;
341            pod.annotations_mut()
342                .entry("kube-seen".to_string())
343                .or_insert_with(|| "yes".to_string());
344            pod.labels_mut()
345                .entry("kube.rs".to_string())
346                .or_insert_with(|| "hello".to_string());
347            pod.finalizers_mut().push("kube-finalizer".to_string());
348            pod.managed_fields_mut().clear();
349            // NB: we are **not** pushing these back upstream - (Api::apply or Api::replace needed for it)
350        }
351        // check we can iterate over ObjectList normally - and check the mutations worked
352        for pod in list {
353            assert!(pod.annotations().get("kube-seen").is_some());
354            assert!(pod.labels().get("kube.rs").is_some());
355            assert!(pod.finalizers().contains(&"kube-finalizer".to_string()));
356            assert!(pod.spec().containers.is_empty());
357            assert!(pod.managed_fields().is_empty());
358        }
359        Ok(())
360    }

364    #[cfg(feature = "derive")]
365    async fn derived_resources_discoverable() -> Result<(), Box<dyn std::error::Error>> {
366        use crate::{
367            core::{DynamicObject, GroupVersion, GroupVersionKind},
368            discovery::{self, verbs, ApiGroup, Discovery, Scope},
369            runtime::wait::{await_condition, conditions, Condition},
370        };
371
372        #[derive(CustomResource, Deserialize, Serialize, Clone, Debug, JsonSchema)]
373        #[kube(group = "kube.rs", version = "v1", kind = "TestCr", namespaced)]
374        #[kube(crates(kube_core = "crate::core"))] // for dev-dep test structure
375        struct TestCrSpec {}
376
377        let client = Client::try_default().await?;
378
379        // install crd is installed
380        let crds: Api<CustomResourceDefinition> = Api::all(client.clone());
381        let ssapply = PatchParams::apply("kube").force();
382        crds.patch("testcrs.kube.rs", &ssapply, &Patch::Apply(TestCr::crd()))
383            .await?;
384        let establish = await_condition(crds.clone(), "testcrs.kube.rs", conditions::is_crd_established());
385        let crd = tokio::time::timeout(std::time::Duration::from_secs(10), establish).await??;
386        assert!(conditions::is_crd_established().matches_object(crd.as_ref()));
387        tokio::time::sleep(std::time::Duration::from_secs(2)).await; // Established condition is actually not enough for api discovery :(
388
389        // create partial information for it to discover
390        let gvk = GroupVersionKind::gvk("kube.rs", "v1", "TestCr");
391        let gv = GroupVersion::gv("kube.rs", "v1");
392
393        // discover by both (recommended kind on groupversion) and (pinned gvk) and they should equal
394        let apigroup = discovery::oneshot::pinned_group(&client, &gv).await?;
395        let (ar1, caps1) = apigroup.recommended_kind("TestCr").unwrap();
396        let (ar2, caps2) = discovery::pinned_kind(&client, &gvk).await?;
397        assert_eq!(caps1.operations.len(), caps2.operations.len(), "unequal caps");
398        assert_eq!(ar1, ar2, "unequal apiresource");
399        assert_eq!(DynamicObject::api_version(&ar2), "kube.rs/v1", "unequal dynver");
400
401        // run (almost) full discovery
402        let discovery = Discovery::new(client.clone())
403            // skip something in discovery (clux.dev crd being mutated in other tests)
404            .exclude(&["rbac.authorization.k8s.io", "clux.dev"])
405            .run()
406            .await?;
407
408        // check our custom resource first by resolving within groups
409        assert!(discovery.has_group("kube.rs"), "missing group kube.rs");
410        let (ar, _caps) = discovery.resolve_gvk(&gvk).unwrap();
411        assert_eq!(ar.group, gvk.group, "unexpected discovered group");
412        assert_eq!(ar.version, gvk.version, "unexcepted discovered ver");
413        assert_eq!(ar.kind, gvk.kind, "unexpected discovered kind");
414
415        // check all non-excluded groups that are iterable
416        let mut groups = discovery.groups_alphabetical().into_iter();
417        let firstgroup = groups.next().unwrap();
418        assert_eq!(firstgroup.name(), ApiGroup::CORE_GROUP, "core not first");
419        for group in groups {
420            for (ar, caps) in group.recommended_resources() {
421                if !caps.supports_operation(verbs::LIST) {
422                    continue;
423                }
424                let api: Api<DynamicObject> = if caps.scope == Scope::Namespaced {
425                    Api::default_namespaced_with(client.clone(), &ar)
426                } else {
427                    Api::all_with(client.clone(), &ar)
428                };
429                api.list(&Default::default()).await?;
430            }
431        }
432
433        // cleanup
434        crds.delete("testcrs.kube.rs", &DeleteParams::default()).await?;
435        Ok(())
436    }

439    #[ignore = "needs cluster (will create await a pod)"]
440    #[cfg(feature = "runtime")]
441    async fn pod_can_await_conditions() -> Result<(), Box<dyn std::error::Error>> {
442        use crate::{
443            api::{DeleteParams, PostParams},
444            runtime::wait::{await_condition, conditions, delete::delete_and_finalize, Condition},
445            Api, Client,
446        };
447        use k8s_openapi::api::core::v1::Pod;
448        use std::time::Duration;
449        use tokio::time::timeout;
450
451        let client = Client::try_default().await?;
452        let pods: Api<Pod> = Api::default_namespaced(client);
453
454        // create busybox pod that's alive for at most 20s
455        let data: Pod = serde_json::from_value(serde_json::json!({
456            "apiVersion": "v1",
457            "kind": "Pod",
458            "metadata": {
459                "name": "busybox-kube4",
460                "labels": { "app": "kube-rs-test" },
461            },
462            "spec": {
463                "terminationGracePeriodSeconds": 1,
464                "restartPolicy": "Never",
465                "containers": [{
466                  "name": "busybox",
467                  "image": "busybox:1.34.1",
468                  "command": ["sh", "-c", "sleep 20"],
469                }],
470            }
471        }))?;
472
473        let pp = PostParams::default();
474        assert_eq!(
475            data.name_unchecked(),
476            pods.create(&pp, &data).await?.name_unchecked()
477        );
478
479        // Watch it phase for a few seconds
480        let is_running = await_condition(pods.clone(), "busybox-kube4", conditions::is_pod_running());
481        let _ = timeout(Duration::from_secs(15), is_running).await?;
482
483        // Verify we can get it
484        let pod = pods.get("busybox-kube4").await?;
485        assert_eq!(pod.spec.as_ref().unwrap().containers[0].name, "busybox");
486
487        // Wait for a more complicated condition: ContainersReady AND Initialized
488        // TODO: remove these once we can write these functions generically
489        fn is_each_container_ready() -> impl Condition<Pod> {
490            |obj: Option<&Pod>| {
491                if let Some(o) = obj {
492                    if let Some(s) = &o.status {
493                        if let Some(conds) = &s.conditions {
494                            if let Some(pcond) = conds.iter().find(|c| c.type_ == "ContainersReady") {
495                                return pcond.status == "True";
496                            }
497                        }
498                    }
499                }
500                false
501            }
502        }
503        let is_fully_ready = await_condition(
504            pods.clone(),
505            "busybox-kube4",
506            conditions::is_pod_running().and(is_each_container_ready()),
507        );
508        let _ = timeout(Duration::from_secs(10), is_fully_ready).await?;
509
510        // Delete it - and wait for deletion to complete
511        let dp = DeleteParams::default();
512        delete_and_finalize(pods.clone(), "busybox-kube4", &dp).await?;
513
514        // verify it is properly gone
515        assert!(pods.get("busybox-kube4").await.is_err());
516
517        Ok(())
518    }
