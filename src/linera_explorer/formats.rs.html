4//! Fetch BCS serde formats from the local linera-service node and decode
5//! user-operation bytes against them. The [`Formats`] type and the decode
6//! helpers come from `linera_sdk::formats`.
7
8use anyhow::{anyhow, Context as _, Result};
9pub use linera_sdk::formats::Formats;
10use serde_json::Value;
11
12use crate::{js_utils::log_str, reqwest_client};
13
14/// Issues `query { applicationFormats(chainId: ..., formatsBlobHash: ...) }`
15/// against the linera-service node and returns the bytes registered for the
16/// given application's module. Returns `Ok(None)` if the local node has no
17/// such blob.
18pub async fn fetch_formats(
19    node: &str,
20    chain_id_hex: &str,
21    formats_blob_hash_hex: &str,
22) -> Result<Option<Formats>> {
23    let url = node.to_string();
24    let query = format!(
25        r#"{{"query":"query {{ applicationFormats(chainId: \"{chain_id_hex}\", formatsBlobHash: \"{formats_blob_hash_hex}\") }}"}}"#
26    );
27    log_str(&format!("fetch_formats: POST {url}"));
28    log_str(&format!("fetch_formats: body {query}"));
29    let response = reqwest_client()
30        .post(&url)
31        .header("Content-Type", "application/json")
32        .body(query)
33        .send()
34        .await?
35        .text()
36        .await?;
37    log_str(&format!("fetch_formats: response {response}"));
38    let response: Value = serde_json::from_str(&response)
39        .with_context(|| format!("invalid JSON from node: {response}"))?;
40    if let Some(errors) = response.get("errors") {
41        return Err(anyhow!("application formats query failed: {errors}"));
42    }
43    let bytes_value = &response["data"]["applicationFormats"];
44    if bytes_value.is_null() {
45        log_str("fetch_formats: node returned null (formats blob not present locally)");
46        return Ok(None);
47    }
48    let bytes: Vec<u8> = bytes_value
49        .as_array()
50        .ok_or_else(|| anyhow!("application formats query returned non-array bytes"))?
51        .iter()
52        .map(|v| {
53            v.as_u64()
54                .and_then(|n| u8::try_from(n).ok())
55                .ok_or_else(|| anyhow!("application formats query returned non-u8 byte"))
56        })
57        .collect::<Result<_>>()?;
58    let formats: Formats = linera_sdk::bcs::from_bytes(&bytes)
59        .context("formats blob did not deserialize as Formats BCS")?;
60    Ok(Some(formats))
61}
