4use anyhow::Result;
5use serde_json::Value;
6
7use crate::reqwest_client;
8
9pub async fn introspection(url: &str) -> Result<Value> {
10    let client = reqwest_client();
11    let graphql_query =
12        "query { \
13           __schema { \
14             queryType { name } \
15             mutationType { name } \
16             subscriptionType { name } \
17             types { ...FullType } \
18             directives { name description locations args { ...InputValue } } } } \
19         fragment FullType on __Type { \
20           kind name description \
21           fields(includeDeprecated:true) { \
22             name description \
23             args { ...InputValue } \
24             type{ ...TypeRef } \
25             isDeprecated deprecationReason } \
26           inputFields { ...InputValue } \
27           interfaces { ...TypeRef } \
28           enumValues(includeDeprecated:true) { name description isDeprecated deprecationReason } \
29           possibleTypes { ...TypeRef } } \
30         fragment InputValue on __InputValue { \
31           name description \
32           type { ...TypeRef } \
33           defaultValue } \
34         fragment TypeRef on __Type { \
35           kind name \
36           ofType { kind name ofType { kind name ofType { kind name ofType { kind name ofType { kind name ofType { kind name ofType {kind name} } } } } } } }";
37    let res = client
38        .post(url)
39        .body(format!("{{\"query\":\"{}\"}}", graphql_query))
40        .send()
41        .await?
42        .text()
43        .await?;
44    Ok(serde_json::from_str::<Value>(&res)?)
45}
