89    indexer: String,
90    node: String,
91    tls: bool,

94impl Config {
95    /// Loads config from local storage.
96    fn load() -> Self {
97        let default = Config {
98            indexer: "localhost:8081".to_string(),
99            node: "localhost:8080".to_string(),
100            tls: false,
101        };
102        // Return default if window doesn't exist (e.g., in test environment).
103        let Some(window) = web_sys::window() else {
104            return default;
105        };
106        match window.local_storage() {
107            Ok(Some(st)) => match st.get_item("config") {
108                Ok(Some(s)) => serde_json::from_str::<Config>(&s).unwrap_or(default),
109                _ => default,
110            },
111            _ => default,
112        }
113    }
114}

161fn url(config: &Config, protocol: Protocol, kind: AddressKind) -> String {
162    let protocol = match protocol {
163        Protocol::Http => "http",
164        Protocol::Websocket => "ws",
165    };
166    let tls = if config.tls { "s" } else { "" };
167    let address = match kind {
168        AddressKind::Node => &config.node,
169        AddressKind::Indexer => &config.indexer,
170    };
171    format!("{}{}://{}", protocol, tls, address)
172}

174async fn get_chain(node: &str, chain_id: ChainId) -> Result<Box<Chain>> {
175    let client = reqwest::Client::new();
176    let variables = chain::Variables {
177        chain_id,
178        inboxes_input: None,
179        outboxes_input: None,
180        previous_message_blocks_input: None,
181        previous_event_blocks_input: None,
182    };
183    let chain = request::<gql_service::Chain, _>(&client, node, variables)
184        .await?

186    log_str(&serde_json::to_string_pretty(&chain).unwrap());
187    Ok(Box::new(chain))

352/// Lists entrypoints for GraphQL queries, mutations or subscriptions.
353fn list_entrypoints(types: &[Value], name: &Value) -> Option<Value> {
354    types
355        .iter()
356        .find(|x: &&Value| &x["name"] == name)
357        .map(|x| x["fields"].clone())
358}

360/// Fills recursively GraphQL objects with their type definitions.
361fn fill_type(element: &Value, types: &Vec<Value>) -> Value {
362    match element {
363        Value::Array(array) => Value::Array(
364            array
365                .iter()
366                .map(|elt: &Value| fill_type(elt, types))
367                .collect(),
368        ),
369        Value::Object(object) => {
370            let mut object = object.clone();
371            let name = element["name"].as_str();
372            let kind = element["kind"].as_str();
373            let of_type = &element["ofType"];
374            match (kind, name, of_type) {
375                (Some("OBJECT"), Some(name), _) => {
376                    match types.iter().find(|elt: &&Value| elt["name"] == name) {
377                        None => (),
378                        Some(element_definition) => {
379                            let fields = element_definition["fields"]
380                                .as_array()
381                                .unwrap()
382                                .iter()
383                                .map(|elt| fill_type(elt, types))
384                                .collect::<Vec<_>>();
385                            object.insert("fields".to_string(), Value::Array(fields));
386                        }
387                    }
388                }
389                (Some("INPUT_OBJECT"), Some(name), _) => {
390                    match types.iter().find(|elt: &&Value| elt["name"] == name) {
391                        None => (),
392                        Some(element_definition) => {
393                            let fields = element_definition["inputFields"]
394                                .as_array()
395                                .unwrap()
396                                .iter()
397                                .map(|elt| fill_type(elt, types))
398                                .collect::<Vec<_>>();
399                            object.insert("inputFields".to_string(), Value::Array(fields));
400                        }
401                    }
402                }
403                (Some("ENUM"), Some(name), _) => {
404                    match types.iter().find(|elt: &&Value| elt["name"] == name) {
405                        None => (),
406                        Some(element_definition) => {
407                            let values = element_definition["enumValues"]
408                                .as_array()
409                                .unwrap()
410                                .iter()
411                                .map(|elt| fill_type(elt, types))
412                                .collect::<Vec<_>>();
413                            object.insert("enumValues".to_string(), Value::Array(values));
414                        }
415                    }
416                }
417                (Some("LIST" | "NON_NULL"), Some(name), Value::Null) => {
418                    match types.iter().find(|elt: &&Value| elt["name"] == name) {
419                        None => (),
420                        Some(element_definition) => {
421                            object
422                                .insert("ofType".to_string(), fill_type(element_definition, types));
423                        }
424                    }
425                }
426                _ => (),
427            };
428            object.insert("ofType".to_string(), fill_type(&element["ofType"], types));
429            object.insert("type".to_string(), fill_type(&element["type"], types));
430            object.insert("args".to_string(), fill_type(&element["args"], types));
431            if let Some("LIST") = kind {
432                object.insert("_input".to_string(), Value::Array(Vec::new()));
433            }
434            if let Some("SCALAR" | "ENUM" | "OBJECT") = kind {
435                object.insert("_include".to_string(), Value::Bool(true));
436            }
437            Value::Object(object)
438        }
439        elt => elt.clone(),
440    }
441}

443/// Returns the application page.
444async fn application(app: Application) -> Result<(Page, String)> {
445    let schema = graphql::introspection(&app.link).await?;
446    let sch = &schema["data"]["__schema"];
447    let types = sch["types"]
448        .as_array()
449        .expect("introspection types is not an array")
450        .clone();
451    let queries =
452        list_entrypoints(&types, &sch["queryType"]["name"]).unwrap_or(Value::Array(Vec::new()));
453    let queries = fill_type(&queries, &types);
454    let mutations =
455        list_entrypoints(&types, &sch["mutationType"]["name"]).unwrap_or(Value::Array(Vec::new()));
456    let mutations = fill_type(&mutations, &types);
457    let subscriptions = list_entrypoints(&types, &sch["subscriptionType"]["name"])
458        .unwrap_or(Value::Array(Vec::new()));
459    let subscriptions = fill_type(&subscriptions, &types);
460    let pathname = format!("/application/{}", app.id.as_str());
461    Ok((
462        Page::Application {
463            app: Box::new(app),
464            queries,
465            mutations,
466            subscriptions,
467        },
468        pathname,
469    ))
470}

495fn format_bytes(value: &JsValue) -> JsValue {
496    let modified_value = value.clone();
497    if let Some(object) = js_sys::Object::try_from(value) {
498        js_sys::Object::keys(object)
499            .iter()
500            .for_each(|k: JsValue| match k.as_string() {
501                None => (),
502                Some(key_str) => {
503                    if &key_str == "bytes" {
504                        let array: Vec<u8> =
505                            js_sys::Uint8Array::from(getf(&modified_value, "bytes")).to_vec();
506                        let array_hex = hex::encode(array);
507                        let hex_len = array_hex.len();
508                        let hex_elided = if hex_len > 128 {
509                            // don't show all hex digits if the bytes array is too long
510                            format!("{}..{}", &array_hex[0..4], &array_hex[hex_len - 4..])
511                        } else {
512                            array_hex
513                        };
514                        setf(&modified_value, "bytes", &JsValue::from_str(&hex_elided))
515                    } else {
516                        setf(
517                            &modified_value,
518                            &key_str,
519                            &format_bytes(&getf(&modified_value, &key_str)),
520                        )
521                    }
522                }
523            });
524    };
525    modified_value
526}

528fn page_name_and_args(page: &Page) -> (&str, Vec<(String, String)>) {
529    match page {
530        Page::Unloaded | Page::Home { .. } => ("", Vec::new()),
531        Page::Block(b) => ("block", vec![("block".to_string(), b.hash.to_string())]),
532        Page::Blocks { .. } => ("blocks", Vec::new()),
533        Page::Applications(_) => ("applications", Vec::new()),
534        Page::Application { app, .. } => (
535            "application",
536            vec![("app".to_string(), stringify(&app.serialize(&SER).unwrap()))],
537        ),
538        Page::Operations(_) => ("operations", Vec::new()),
539        Page::Operation(op) => (
540            "operation",
541            vec![
542                ("height".to_string(), op.key.height.to_string()),
543                ("index".to_string(), op.key.index.to_string()),
544            ],
545        ),
546        Page::Plugin { name, .. } => ("plugin", vec![("plugin".to_string(), name.to_string())]),
547        Page::Error(_) => ("error", Vec::new()),
548    }
549}

551fn find_arg(args: &[(String, String)], key: &str) -> Option<String> {
552    args.iter()
553        .find_map(|(k, v)| if k == key { Some(v.clone()) } else { None })
554}
555
556fn find_arg_map<T, F, E>(args: &[(String, String)], key: &str, f: F) -> Result<Option<T>>
557where
558    F: FnOnce(&str) -> Result<T, E>,
559    E: std::error::Error + Send + Sync + 'static,
560{
561    match args
562        .iter()
563        .find_map(|(k, v)| if k == key { Some(v) } else { None })
564    {
565        None => Ok(None),
566        Some(v) => Ok(Some(f(v)?)),
567    }
568}

570fn chain_id_from_args(
571    app: &JsValue,
572    data: &Data,
573    args: &[(String, String)],
574    init: bool,
575) -> Result<(ChainId, bool)> {
576    match find_arg(args, "chain") {
577        None => Ok((data.chain, init)),
578        Some(chain_id) => {
579            let chain_js: JsValue = chain_id
580                .serialize(&SER)
581                .expect("failed to serialize ChainId");
582            setf(app, "chain", &chain_js);
583            Ok(ChainId::from_str(&chain_id).map(|id| (id, id != data.chain || init))?)
584        }
585    }
586}

588async fn page(
589    page_name: &str,
590    node: &str,
591    indexer: &str,
592    chain_id: ChainId,
593    args: &[(String, String)],
594) -> Result<(Page, String)> {
595    match page_name {
596        "" => home(node, chain_id).await,
597        "block" => {
598            let hash = find_arg_map(args, "block", CryptoHash::from_str)?;
599            block(node, chain_id, hash).await
600        }
601        "blocks" => blocks(node, chain_id, None, Some(20)).await,
602        "applications" => applications(node, chain_id).await,
603        "application" => {
604            let app_arg = find_arg(args, "app").context("unknown application")?;
605            let app =
606                from_value::<Application>(parse(&app_arg)).expect("cannot parse applications");
607            application(app).await
608        }
609        "operation" => {
610            let height = find_arg_map(args, "height", BlockHeight::from_str)?;
611            let index = find_arg_map(args, "index", usize::from_str)?;
612            match (height, index) {
613                (None, _) | (_, None) => operation(indexer, None, chain_id).await,
614                (Some(height), Some(index)) => {
615                    let key = OperationKey {
616                        chain_id,
617                        height,
618                        index,
619                    };
620                    operation(indexer, Some(key), chain_id).await
621                }
622            }
623        }
624        "operations" => operations(indexer, chain_id).await,
625        "plugin" => {
626            let name = find_arg(args, "plugin").context("unknown plugin")?;
627            plugin(&name, indexer).await
628        }
629        "error" => {
630            let msg = find_arg(args, "msg").unwrap_or("unknown error".to_string());
631            Err(anyhow::Error::msg(msg))
632        }
633        _ => Err(anyhow!("unknown page")),
634    }
635}

637/// Main function to switch between Vue pages.
638async fn route_aux(
639    app: &JsValue,
640    data: &Data,
641    path: &Option<String>,
642    args: &[(String, String)],
643    init: bool,
644) {
645    let chain_info = chain_id_from_args(app, data, args, init);
646    let (page_name, args): (&str, Vec<(String, String)>) = match (path, &data.page) {
647        (Some(p), _) => (p, args.to_vec()),
648        (_, p) => page_name_and_args(p),
649    };
650    let node = url(&data.config, Protocol::Http, AddressKind::Node);
651    let indexer = url(&data.config, Protocol::Http, AddressKind::Indexer);
652    let result = match chain_info {
653        Err(e) => Err(e),
654        Ok((chain_id, chain_changed)) => {
655            let page_result = page(page_name, &node, &indexer, chain_id, &args).await;
656            if chain_changed {
657                if let Some(ws) = WEBSOCKET.get() {
658                    let _ = ws.close().await;
659                }
660                let address = url(&data.config, Protocol::Websocket, AddressKind::Node);
661                subscribe_chain(app, &address, chain_id).await;
662            };
663            page_result
664        }
665    };
666    let (page, new_path) = result.unwrap_or_else(|e| error(&e));
667    let page_js = format_bytes(&page.serialize(&SER).unwrap());
668    setf(app, "page", &page_js);
669    web_sys::window()
670        .expect("window object not found")
671        .history()
672        .expect("history object not found")
673        .push_state_with_url(&page_js, &new_path, Some(&new_path))
674        .expect("push_state failed");
675}

677#[wasm_bindgen]
678pub async fn route(app: JsValue, path: JsValue, args: JsValue) {
679    let path = path.as_string();
680    let args = from_value::<Vec<(String, String)>>(args).unwrap_or_default();
681    let msg = format!("route: {} {:?}", path.as_deref().unwrap_or("none"), args);
682    log_str(&msg);
683    let data = from_value::<Data>(app.clone()).expect("cannot parse Vue data");
684    route_aux(&app, &data, &path, &args, false).await
685}

708/// Subscribes to notifications for one chain
709async fn subscribe_chain(app: &JsValue, address: &str, chain: ChainId) {
710    let (ws, mut wsio) = WsMeta::connect(
711        &format!("{}/ws", address),
712        Some(vec!["graphql-transport-ws"]),
713    )
714    .await
715    .expect("cannot connect to websocket");
716    wsio.send(WsMessage::Text(
717        "{\"type\": \"connection_init\", \"payload\": {}}".to_string(),
718    ))
719    .await
720    .expect("cannot send to websocket");
721    wsio.next().await;
722    let uuid = Uuid::new_v3(&Uuid::NAMESPACE_DNS, b"linera.dev");
723    let payload_query = format!(
724        r#"subscription {{ notifications(chainId: \"{}\") }}"#,
725        chain
726    );
727    let query = format!(
728        r#"{{ "id": "{}", "type": "subscribe", "payload": {{"query": "{}"}} }}"#,
729        uuid, payload_query
730    );
731    wsio.send(WsMessage::Text(query))
732        .await
733        .expect("cannot send to websocket");
734    let app = app.clone();
735    spawn_local(async move {
736        while let Some(evt) = wsio.next().await {
737            match evt {
738                WsMessage::Text(message) => {
739                    let graphql_message = serde_json::from_str::<
740                        GQuery<Response<notifications::ResponseData>>,
741                    >(&message)
742                    .expect("unexpected websocket response");
743                    if let Some(payload) = graphql_message.payload {
744                        if let Some(message_data) = payload.data {
745                            let data =
746                                from_value::<Data>(app.clone()).expect("cannot parse vue data");
747                            if let Reason::NewBlock { .. } = message_data.notifications.reason {
748                                if message_data.notifications.chain_id == chain {
749                                    route_aux(&app, &data, &None, &Vec::new(), false).await
750                                }
751                            }
752                        }
753                        if let Some(errors) = payload.errors {
754                            errors.iter().for_each(|e| log_str(&e.to_string()));
755                            break;
756                        }
757                    };
758                }
759                WsMessage::Binary(_) => (),
760            }
761        }
762    });
763    let _ = WEBSOCKET.set(ws);
764}

766/// Initializes pages and subscribes to notifications.
767#[wasm_bindgen]
768pub async fn start(app: JsValue) {
769    console_error_panic_hook::set_once();
770    set_onpopstate(app.clone());
771    let data = from_value::<Data>(app.clone()).expect("cannot parse vue data");
772    let address = url(&data.config, Protocol::Http, AddressKind::Node);
773    let default_chain = chains(&app, &address).await;
774    match default_chain {
775        Err(e) => {
776            route_aux(
777                &app,
778                &data,
779                &Some("error".to_string()),
780                &[("msg".to_string(), e.to_string())],
781                true,
782            )
783            .await
784        }
785        Ok(default_chain) => {
786            let indexer = url(&data.config, Protocol::Http, AddressKind::Indexer);
787            let _ = plugins(&app, &indexer).await;
788            let uri = web_sys::window()
789                .expect("window object not found")
790                .location()
791                .href()
792                .unwrap();
793            let uri = Url::parse(&uri).expect("failed to parse url");
794            let pathname = uri.path();
795            let mut args = uri.query_pairs().into_owned().collect::<Vec<_>>();
796            args.push(("chain".to_string(), default_chain.to_string()));
797            let path = match pathname {
798                "/blocks" => Some("blocks".to_string()),
799                "/applications" => Some("applications".to_string()),
800                "/operations" => Some("operations".to_string()),
801                "/operation" => Some("operation".to_string()),
802                "/plugin" => Some("plugin".to_string()),
803                pathname => match (
804                    pathname.strip_prefix("/block/"),
805                    pathname.strip_prefix("/application/"),
806                ) {
807                    (Some(hash), _) => {
808                        args.push(("block".to_string(), hash.to_string()));
809                        Some("block".to_string())
810                    }
811                    (_, Some(app_id)) => {
812                        let link = format!("{}/applications/{}", address, app_id);
813                        let app =
814                            serde_json::json!({"id": app_id, "link": link, "description": ""})
815                                .to_string();
816                        args.push(("app".to_string(), app));
817                        Some("application".to_string())
818                    }
819                    _ => None,
820                },
821            };
822            route_aux(&app, &data, &path, &args, true).await;
823        }
824    }
825}

827/// Saves config to local storage.
828#[wasm_bindgen]
829pub fn save_config(app: JsValue) {
830    let data = from_value::<Data>(app).expect("cannot parse vue data");
831    if let Ok(Some(storage)) = web_sys::window()
832        .expect("window object not found")
833        .local_storage()
834    {
835        storage
836            .set_item(
837                "config",
838                &serde_json::to_string::<Config>(&data.config)
839                    .expect("cannot parse localstorage config"),
840            )
841            .expect("cannot set config");
842    }
