46/// Calculate the iteration count for the modexp precompile.
47pub fn calculate_iteration_count<const MULTIPLIER: u64>(exp_length: u64, exp_highp: &U256) -> u64 {
48    let mut iteration_count: u64 = 0;
49
50    if exp_length <= 32 && exp_highp.is_zero() {
51        iteration_count = 0;
52    } else if exp_length <= 32 {
53        iteration_count = exp_highp.bit_len() as u64 - 1;
54    } else if exp_length > 32 {
55        iteration_count = (MULTIPLIER.saturating_mul(exp_length - 32))
56            .saturating_add(max(1, exp_highp.bit_len() as u64) - 1);
57    }
58
59    max(iteration_count, 1)
60}

62/// Run the modexp precompile.
63pub fn run_inner<F, const OSAKA: bool>(
64    input: &[u8],
65    gas_limit: u64,
66    min_gas: u64,
67    calc_gas: F,
68) -> PrecompileResult
69where
70    F: FnOnce(u64, u64, u64, &U256) -> u64,
71{
72    // If there is no minimum gas, return error.
73    if min_gas > gas_limit {
74        return Err(PrecompileError::OutOfGas);
75    }
76
77    // The format of input is:
78    // <length_of_BASE> <length_of_EXPONENT> <length_of_MODULUS> <BASE> <EXPONENT> <MODULUS>
79    // Where every length is a 32-byte left-padded integer representing the number of bytes
80    // to be taken up by the next value.
81    const HEADER_LENGTH: usize = 96;
82
83    // Extract the header
84    let base_len = U256::from_be_bytes(right_pad_with_offset::<32>(input, 0).into_owned());
85    let exp_len = U256::from_be_bytes(right_pad_with_offset::<32>(input, 32).into_owned());
86    let mod_len = U256::from_be_bytes(right_pad_with_offset::<32>(input, 64).into_owned());
87
88    // Cast base and modulus to usize, it does not make sense to handle larger values
89    let base_len =
90        usize::try_from(base_len).map_err(|_| PrecompileError::ModexpEip7823LimitSize)?;
91    let mod_len = usize::try_from(mod_len).map_err(|_| PrecompileError::ModexpEip7823LimitSize)?;
92    // cast exp len to the max size, it will fail later in gas calculation if it is too large.
93    let exp_len = usize::try_from(exp_len).unwrap_or(usize::MAX);
94
95    // for EIP-7823 we need to check size of imputs
96    if OSAKA
97        && (base_len > eip7823::INPUT_SIZE_LIMIT
98            || mod_len > eip7823::INPUT_SIZE_LIMIT
99            || exp_len > eip7823::INPUT_SIZE_LIMIT)
100    {
101        return Err(PrecompileError::ModexpEip7823LimitSize);
102    }
103
104    // special case for both base and mod length being 0.
105    if base_len == 0 && mod_len == 0 {
106        return Ok(PrecompileOutput::new(min_gas, Bytes::new()));
107    }
108
109    // Used to extract ADJUSTED_EXPONENT_LENGTH.
110    let exp_highp_len = min(exp_len, 32);
111
112    // Throw away the header data as we already extracted lengths.
113    let input = input.get(HEADER_LENGTH..).unwrap_or_default();
114
115    let exp_highp = {
116        // Get right padded bytes so if data.len is less then exp_len we will get right padded zeroes.
117        let right_padded_highp = right_pad_with_offset::<32>(input, base_len);
118        // If exp_len is less then 32 bytes get only exp_len bytes and do left padding.
119        let out = left_pad::<32>(&right_padded_highp[..exp_highp_len]);
120        U256::from_be_bytes(out.into_owned())
121    };
122
123    // Check if we have enough gas.
124    let gas_cost = calc_gas(base_len as u64, exp_len as u64, mod_len as u64, &exp_highp);
125    if gas_cost > gas_limit {
126        return Err(PrecompileError::OutOfGas);
127    }
128
129    // Padding is needed if the input does not contain all 3 values.
130    let input_len = base_len.saturating_add(exp_len).saturating_add(mod_len);
131    let input = right_pad_vec(input, input_len);
132    let (base, input) = input.split_at(base_len);
133    let (exponent, modulus) = input.split_at(exp_len);
134    debug_assert_eq!(modulus.len(), mod_len);
135
136    // Call the modexp.
137    let output = modexp(base, exponent, modulus);
138
139    // Left pad the result to modulus length. bytes will always by less or equal to modulus length.
140    Ok(PrecompileOutput::new(
141        gas_cost,
142        left_pad_vec(&output, mod_len).into_owned().into(),
143    ))
144}

146/// Calculate the gas cost for the modexp precompile with BYZANTIUM gas rules.
147pub fn byzantium_gas_calc(base_len: u64, exp_len: u64, mod_len: u64, exp_highp: &U256) -> u64 {
148    gas_calc::<0, 8, 20, _>(base_len, exp_len, mod_len, exp_highp, |max_len| -> U256 {
149        // Output of this function is bounded by 2^128
150        if max_len <= 64 {
151            U256::from(max_len * max_len)
152        } else if max_len <= 1_024 {
153            U256::from(max_len * max_len / 4 + 96 * max_len - 3_072)
154        } else {
155            // Up-cast to avoid overflow
156            let x = U256::from(max_len);
157            let x_sq = x * x; // x < 2^64 => x*x < 2^128 < 2^256 (no overflow)
158            x_sq / U256::from(16) + U256::from(480) * x - U256::from(199_680)
159        }
160    })
161}

163/// Calculate gas cost according to EIP 2565:
164/// <https://eips.ethereum.org/EIPS/eip-2565>
165pub fn berlin_gas_calc(base_len: u64, exp_len: u64, mod_len: u64, exp_highp: &U256) -> u64 {
166    gas_calc::<200, 8, 3, _>(base_len, exp_len, mod_len, exp_highp, |max_len| -> U256 {
167        let words = U256::from(max_len.div_ceil(8));
168        words * words
169    })
170}

172/// Calculate gas cost according to EIP-7883:
173/// <https://eips.ethereum.org/EIPS/eip-7883>
174///
175/// There are three changes:
176/// 1. Increase minimal price from 200 to 500
177/// 2. Increase cost when exponent is larger than 32 bytes
178/// 3. Increase cost when base or modulus is larger than 32 bytes
179pub fn osaka_gas_calc(base_len: u64, exp_len: u64, mod_len: u64, exp_highp: &U256) -> u64 {
180    gas_calc::<500, 16, 3, _>(base_len, exp_len, mod_len, exp_highp, |max_len| -> U256 {
181        let words = U256::from(max_len.div_ceil(8));
182        let words_square = words * words;
183        if max_len > 32 {
184            return words_square * U256::from(2);
185        }
186        words_square
187    })
188}

190/// Calculate gas cost.
191pub fn gas_calc<const MIN_PRICE: u64, const MULTIPLIER: u64, const GAS_DIVISOR: u64, F>(
192    base_len: u64,
193    exp_len: u64,
194    mod_len: u64,
195    exp_highp: &U256,
196    calculate_multiplication_complexity: F,
197) -> u64
198where
199    F: Fn(u64) -> U256,
200{
201    let multiplication_complexity = calculate_multiplication_complexity(max(base_len, mod_len));
202    let iteration_count = calculate_iteration_count::<MULTIPLIER>(exp_len, exp_highp);
203    let gas = (multiplication_complexity * U256::from(iteration_count)) / U256::from(GAS_DIVISOR);
204    max(MIN_PRICE, gas.saturating_to())
205}
