8/// Error of parsing character reference (`&#<dec-number>;` or `&#x<hex-number>;`).
9#[derive(Clone, Debug, PartialEq)]
10pub enum ParseCharRefError {
11    /// Number contains sign character (`+` or `-`) which is not allowed.
12    UnexpectedSign,
13    /// Number cannot be parsed due to non-number characters or a numeric overflow.
14    InvalidNumber(ParseIntError),
15    /// Character reference represents not a valid unicode codepoint.
16    InvalidCodepoint(u32),
17    /// Character reference expanded to a not permitted character for an XML.
18    ///
19    /// Currently, only `0x0` character produces this error.
20    IllegalCharacter(u32),
21}
22
23impl std::fmt::Display for ParseCharRefError {
24    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
25        match self {
26            Self::UnexpectedSign => f.write_str("unexpected number sign"),
27            Self::InvalidNumber(e) => e.fmt(f),
28            Self::InvalidCodepoint(n) => write!(f, "`{}` is not a valid codepoint", n),
29            Self::IllegalCharacter(n) => write!(f, "0x{:x} character is not permitted in XML", n),
30        }
31    }
32}
33
34impl std::error::Error for ParseCharRefError {
35    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
36        match self {
37            Self::InvalidNumber(e) => Some(e),
38            _ => None,
39        }
40    }
41}

43/// Error for XML escape / unescape.
44#[derive(Clone, Debug, PartialEq)]
45pub enum EscapeError {
46    /// Referenced entity in unknown to the parser.
47    UnrecognizedEntity(Range<usize>, String),
48    /// Cannot find `;` after `&`
49    UnterminatedEntity(Range<usize>),
50    /// Attempt to parse character reference (`&#<dec-number>;` or `&#x<hex-number>;`)
51    /// was unsuccessful, not all characters are decimal or hexadecimal numbers.
52    InvalidCharRef(ParseCharRefError),
53}
54
55impl std::fmt::Display for EscapeError {
56    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
57        match self {
58            Self::UnrecognizedEntity(rge, res) => {
59                write!(f, "at {:?}: unrecognized entity `{}`", rge, res)
60            }
61            Self::UnterminatedEntity(e) => write!(
62                f,
63                "Error while escaping character at range {:?}: Cannot find ';' after '&'",
64                e
65            ),
66            Self::InvalidCharRef(e) => {
67                write!(f, "invalid character reference: {}", e)
68            }
69        }
70    }
71}

75        match self {
76            Self::InvalidCharRef(e) => Some(e),
77            _ => None,

79    }

82/// Escapes an `&str` and replaces all xml special characters (`<`, `>`, `&`, `'`, `"`)
83/// with their corresponding xml escaped value.
84///
85/// This function performs following replacements:
86///
87/// | Character | Replacement
88/// |-----------|------------
89/// | `<`       | `&lt;`
90/// | `>`       | `&gt;`
91/// | `&`       | `&amp;`
92/// | `'`       | `&apos;`
93/// | `"`       | `&quot;`
94///
95/// This function performs following replacements:
96///
97/// | Character | Replacement
98/// |-----------|------------
99/// | `<`       | `&lt;`
100/// | `>`       | `&gt;`
101/// | `&`       | `&amp;`
102/// | `'`       | `&apos;`
103/// | `"`       | `&quot;`
104pub fn escape<'a>(raw: impl Into<Cow<'a, str>>) -> Cow<'a, str> {
105    _escape(raw, |ch| matches!(ch, b'<' | b'>' | b'&' | b'\'' | b'\"'))
106}

133/// XML standard [requires] that only `<` and `&` was escaped in text content or
134/// attribute value. All other characters not necessary to be escaped, although
135/// for compatibility with SGML they also should be escaped. Practically, escaping
136/// only those characters is enough.
137///
138/// This function performs following replacements:
139///
140/// | Character | Replacement
141/// |-----------|------------
142/// | `<`       | `&lt;`
143/// | `&`       | `&amp;`
144///
145/// [requires]: https://www.w3.org/TR/xml11/#syntax
146pub fn minimal_escape<'a>(raw: impl Into<Cow<'a, str>>) -> Cow<'a, str> {
147    _escape(raw, |ch| matches!(ch, b'<' | b'&'))
148}

150/// Escapes an `&str` and replaces a subset of xml special characters (`<`, `>`,
151/// `&`, `'`, `"`) with their corresponding xml escaped value.
152pub(crate) fn _escape<'a, F: Fn(u8) -> bool>(
153    raw: impl Into<Cow<'a, str>>,
154    escape_chars: F,
155) -> Cow<'a, str> {
156    let raw = raw.into();
157    let bytes = raw.as_bytes();
158    let mut escaped = None;
159    let mut iter = bytes.iter();
160    let mut pos = 0;
161    while let Some(i) = iter.position(|&b| escape_chars(b)) {
162        if escaped.is_none() {
163            escaped = Some(Vec::with_capacity(raw.len()));
164        }
165        let escaped = escaped.as_mut().expect("initialized");
166        let new_pos = pos + i;
167        escaped.extend_from_slice(&bytes[pos..new_pos]);
168        match bytes[new_pos] {
169            b'<' => escaped.extend_from_slice(b"&lt;"),
170            b'>' => escaped.extend_from_slice(b"&gt;"),
171            b'\'' => escaped.extend_from_slice(b"&apos;"),
172            b'&' => escaped.extend_from_slice(b"&amp;"),
173            b'"' => escaped.extend_from_slice(b"&quot;"),
174
175            // This set of escapes handles characters that should be escaped
176            // in elements of xs:lists, because those characters works as
177            // delimiters of list elements
178            b'\t' => escaped.extend_from_slice(b"&#9;"),
179            b'\n' => escaped.extend_from_slice(b"&#10;"),
180            b'\r' => escaped.extend_from_slice(b"&#13;"),
181            b' ' => escaped.extend_from_slice(b"&#32;"),
182            _ => unreachable!(
183                "Only '<', '>','\', '&', '\"', '\\t', '\\r', '\\n', and ' ' are escaped"
184            ),
185        }
186        pos = new_pos + 1;
187    }
188
189    if let Some(mut escaped) = escaped {
190        if let Some(raw) = bytes.get(pos..) {
191            escaped.extend_from_slice(raw);
192        }
193        // SAFETY: we operate on UTF-8 input and search for an one byte chars only,
194        // so all slices that was put to the `escaped` is a valid UTF-8 encoded strings
195        // TODO: Can be replaced with `unsafe { String::from_utf8_unchecked() }`
196        // if unsafe code will be allowed
197        Cow::Owned(String::from_utf8(escaped).unwrap())
198    } else {
199        raw
200    }
201}

214/// Unescape an `&str` and replaces all xml escaped characters (`&...;`) into
215/// their corresponding value, using a resolver function for custom entities.
216///
217/// If feature [`escape-html`] is enabled, then recognizes all [HTML5 escapes].
218///
219/// Predefined entities will be resolved _after_ trying to resolve with `resolve_entity`,
220/// which allows you to override default behavior which required in some XML dialects.
221///
222/// Character references (`&#hh;`) cannot be overridden, they are resolved before
223/// calling `resolve_entity`.
224///
225/// Note, that entities will not be resolved recursively. In order to satisfy the
226/// XML [requirements] you should unescape nested entities by yourself.
227///
228/// # Example
229///
230/// ```
231/// use quick_xml::escape::resolve_xml_entity;
232/// # use quick_xml::escape::unescape_with;
233/// # use pretty_assertions::assert_eq;
234/// let override_named_entities = |entity: &str| match entity {
235///     // Override standard entities
236///     "lt" => Some("FOO"),
237///     "gt" => Some("BAR"),
238///     // Resolve custom entities
239///     "baz" => Some("&lt;"),
240///     // Delegate other entities to the default implementation
241///     _ => resolve_xml_entity(entity),
242/// };
243///
244/// assert_eq!(
245///     unescape_with("&amp;&lt;test&gt;&baz;", override_named_entities).unwrap(),
246///     "&FOOtestBAR&lt;"
247/// );
248/// ```
249///
250/// [`escape-html`]: ../index.html#escape-html
251/// [HTML5 escapes]: https://dev.w3.org/html5/html-author/charref
252/// [requirements]: https://www.w3.org/TR/xml11/#intern-replacement
253pub fn unescape_with<'input, 'entity, F>(
254    raw: &'input str,
255    mut resolve_entity: F,
256) -> Result<Cow<'input, str>, EscapeError>
257where
258    // the lifetime of the output comes from a capture or is `'static`
259    F: FnMut(&str) -> Option<&'entity str>,
260{
261    let bytes = raw.as_bytes();
262    let mut unescaped = None;
263    let mut last_end = 0;
264    let mut iter = memchr2_iter(b'&', b';', bytes);
265    while let Some(start) = iter.by_ref().find(|p| bytes[*p] == b'&') {
266        match iter.next() {
267            Some(end) if bytes[end] == b';' => {
268                // append valid data
269                if unescaped.is_none() {
270                    unescaped = Some(String::with_capacity(raw.len()));
271                }
272                let unescaped = unescaped.as_mut().expect("initialized");
273                unescaped.push_str(&raw[last_end..start]);
274
275                // search for character correctness
276                let pat = &raw[start + 1..end];
277                if let Some(entity) = pat.strip_prefix('#') {
278                    let codepoint = parse_number(entity).map_err(EscapeError::InvalidCharRef)?;
279                    unescaped.push_str(codepoint.encode_utf8(&mut [0u8; 4]));
280                } else if let Some(value) = resolve_entity(pat) {
281                    unescaped.push_str(value);
282                } else {
283                    return Err(EscapeError::UnrecognizedEntity(
284                        start + 1..end,
285                        pat.to_string(),
286                    ));
287                }
288
289                last_end = end + 1;
290            }
291            _ => return Err(EscapeError::UnterminatedEntity(start..raw.len())),
292        }
293    }
294
295    if let Some(mut unescaped) = unescaped {
296        if let Some(raw) = raw.get(last_end..) {
297            unescaped.push_str(raw);
298        }
299        Ok(Cow::Owned(unescaped))
300    } else {
301        Ok(Cow::Borrowed(raw))
302    }
303}

323/// Resolves predefined XML entities. If specified entity is not a predefined XML
324/// entity, `None` is returned.
325///
326/// The complete list of predefined entities are defined in the [specification].
327///
328/// ```
329/// # use quick_xml::escape::resolve_xml_entity;
330/// # use pretty_assertions::assert_eq;
331/// assert_eq!(resolve_xml_entity("lt"), Some("<"));
332/// assert_eq!(resolve_xml_entity("gt"), Some(">"));
333/// assert_eq!(resolve_xml_entity("amp"), Some("&"));
334/// assert_eq!(resolve_xml_entity("apos"), Some("'"));
335/// assert_eq!(resolve_xml_entity("quot"), Some("\""));
336///
337/// assert_eq!(resolve_xml_entity("foo"), None);
338/// ```
339///
340/// [specification]: https://www.w3.org/TR/xml11/#sec-predefined-ent
341pub const fn resolve_xml_entity(entity: &str) -> Option<&'static str> {
342    // match over strings are not allowed in const functions
343    let s = match entity.as_bytes() {
344        b"lt" => "<",
345        b"gt" => ">",
346        b"amp" => "&",
347        b"apos" => "'",
348        b"quot" => "\"",
349        _ => return None,
350    };
351    Some(s)
352}

1823fn parse_number(num: &str) -> Result<char, ParseCharRefError> {
1824    let code = if let Some(hex) = num.strip_prefix('x') {
1825        from_str_radix(hex, 16)?
1826    } else {
1827        from_str_radix(num, 10)?
1828    };
1829    if code == 0 {
1830        return Err(ParseCharRefError::IllegalCharacter(code));
1831    }
1832    match std::char::from_u32(code) {
1833        Some(c) => Ok(c),
1834        None => Err(ParseCharRefError::InvalidCodepoint(code)),
1835    }
1836}
