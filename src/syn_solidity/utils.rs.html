28pub(crate) fn tts_until_semi(input: ParseStream<'_>) -> TokenStream {
29    let mut tts = TokenStream::new();
30    while !input.is_empty() && !input.peek(Token![;]) {
31        let tt = input.parse::<TokenTree>().unwrap();
32        tts.extend(std::iter::once(tt));
33    }
34    tts
35}
