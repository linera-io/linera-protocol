1macro_rules! impl_database_ext {
2    (
3        $database:path,
4        row: $row:path,
5        $(describe-blocking: $describe:path,)?
6    ) => {
7        impl $crate::database::DatabaseExt for $database {
8            const DATABASE_PATH: &'static str = stringify!($database);
9            const ROW_PATH: &'static str = stringify!($row);
10            impl_describe_blocking!($database, $($describe)?);
11        }
12    }
13}

15macro_rules! impl_describe_blocking {
16    ($database:path $(,)?) => {
17        fn describe_blocking(
18            query: &str,
19            database_url: &str,
20        ) -> sqlx_core::Result<sqlx_core::describe::Describe<Self>> {
21            use $crate::database::CachingDescribeBlocking;
22
23            // This can't be a provided method because the `static` can't reference `Self`.
24            static CACHE: CachingDescribeBlocking<$database> = CachingDescribeBlocking::new();
25
26            CACHE.describe(query, database_url)
27        }
28    };
29    ($database:path, $describe:path) => {
30        fn describe_blocking(
31            query: &str,
32            database_url: &str,
33        ) -> sqlx_core::Result<sqlx_core::describe::Describe<Self>> {
34            $describe(query, database_url)
35        }
36    };
37}
