15pub fn expand_derive_encode(input: &DeriveInput) -> syn::Result<TokenStream> {
16    let args = parse_container_attributes(&input.attrs)?;
17
18    match &input.data {
19        Data::Struct(DataStruct {
20            fields: Fields::Unnamed(FieldsUnnamed { unnamed, .. }),
21            ..
22        }) if unnamed.len() == 1 => {
23            expand_derive_encode_transparent(input, unnamed.first().unwrap())
24        }
25        Data::Enum(DataEnum { variants, .. }) => match args.repr {
26            Some(_) => expand_derive_encode_weak_enum(input, variants),
27            None => expand_derive_encode_strong_enum(input, variants),
28        },
29        Data::Struct(DataStruct {
30            fields: Fields::Named(FieldsNamed { named, .. }),
31            ..
32        }) => expand_derive_encode_struct(input, named),
33        Data::Union(_) => Err(syn::Error::new_spanned(input, "unions are not supported")),
34        Data::Struct(DataStruct {
35            fields: Fields::Unnamed(..),
36            ..
37        }) => Err(syn::Error::new_spanned(
38            input,
39            "structs with zero or more than one unnamed field are not supported",
40        )),
41        Data::Struct(DataStruct {
42            fields: Fields::Unit,
43            ..
44        }) => Err(syn::Error::new_spanned(
45            input,
46            "unit structs are not supported",
47        )),
48    }
49}

51fn expand_derive_encode_transparent(
52    input: &DeriveInput,
53    field: &Field,
54) -> syn::Result<TokenStream> {
55    check_transparent_attributes(input, field)?;
56
57    let ident = &input.ident;
58    let ty = &field.ty;
59
60    // extract type generics
61    let generics = &input.generics;
62    let (_, ty_generics, _) = generics.split_for_impl();
63
64    // add db type for impl generics & where clause
65    let lifetime = Lifetime::new("'q", Span::call_site());
66    let mut generics = generics.clone();
67    generics
68        .params
69        .insert(0, LifetimeParam::new(lifetime.clone()).into());
70
71    generics
72        .params
73        .insert(0, parse_quote!(DB: ::sqlx::Database));
74    generics
75        .make_where_clause()
76        .predicates
77        .push(parse_quote!(#ty: ::sqlx::encode::Encode<#lifetime, DB>));
78    let (impl_generics, _, where_clause) = generics.split_for_impl();
79
80    Ok(quote!(
81        #[automatically_derived]
82        impl #impl_generics ::sqlx::encode::Encode<#lifetime, DB> for #ident #ty_generics
83        #where_clause
84        {
85            fn encode_by_ref(
86                &self,
87                buf: &mut <DB as ::sqlx::database::Database>::ArgumentBuffer<#lifetime>,
88            ) -> ::std::result::Result<::sqlx::encode::IsNull, ::sqlx::error::BoxDynError> {
89                <#ty as ::sqlx::encode::Encode<#lifetime, DB>>::encode_by_ref(&self.0, buf)
90            }
91
92            fn produces(&self) -> Option<DB::TypeInfo> {
93                <#ty as ::sqlx::encode::Encode<#lifetime, DB>>::produces(&self.0)
94            }
95
96            fn size_hint(&self) -> usize {
97                <#ty as ::sqlx::encode::Encode<#lifetime, DB>>::size_hint(&self.0)
98            }
99        }
100    ))
101}

103fn expand_derive_encode_weak_enum(
104    input: &DeriveInput,
105    variants: &Punctuated<Variant, Comma>,
106) -> syn::Result<TokenStream> {
107    let attr = check_weak_enum_attributes(input, variants)?;
108    let repr = attr.repr.unwrap();
109    let ident = &input.ident;
110
111    let mut values = Vec::new();
112
113    for v in variants {
114        let id = &v.ident;
115        values.push(quote!(#ident :: #id => (#ident :: #id as #repr),));
116    }
117
118    Ok(quote!(
119        #[automatically_derived]
120        impl<'q, DB: ::sqlx::Database> ::sqlx::encode::Encode<'q, DB> for #ident
121        where
122            #repr: ::sqlx::encode::Encode<'q, DB>,
123        {
124            fn encode_by_ref(
125                &self,
126                buf: &mut <DB as ::sqlx::database::Database>::ArgumentBuffer<'q>,
127            ) -> ::std::result::Result<::sqlx::encode::IsNull, ::sqlx::error::BoxDynError> {
128                let value = match self {
129                    #(#values)*
130                };
131
132                <#repr as ::sqlx::encode::Encode<DB>>::encode_by_ref(&value, buf)
133            }
134
135            fn size_hint(&self) -> usize {
136                <#repr as ::sqlx::encode::Encode<DB>>::size_hint(&Default::default())
137            }
138        }
139    ))
140}

142fn expand_derive_encode_strong_enum(
143    input: &DeriveInput,
144    variants: &Punctuated<Variant, Comma>,
145) -> syn::Result<TokenStream> {
146    let cattr = check_strong_enum_attributes(input, variants)?;
147
148    let ident = &input.ident;
149
150    let mut value_arms = Vec::new();
151
152    for v in variants {
153        let id = &v.ident;
154        let attributes = parse_child_attributes(&v.attrs)?;
155
156        if let Some(rename) = attributes.rename {
157            value_arms.push(quote!(#ident :: #id => #rename,));
158        } else if let Some(pattern) = cattr.rename_all {
159            let name = rename_all(&id.to_string(), pattern);
160
161            value_arms.push(quote!(#ident :: #id => #name,));
162        } else {
163            let name = id.to_string();
164            value_arms.push(quote!(#ident :: #id => #name,));
165        }
166    }
167
168    Ok(quote!(
169        #[automatically_derived]
170        impl<'q, DB: ::sqlx::Database> ::sqlx::encode::Encode<'q, DB> for #ident
171        where
172            &'q ::std::primitive::str: ::sqlx::encode::Encode<'q, DB>,
173        {
174            fn encode_by_ref(
175                &self,
176                buf: &mut <DB as ::sqlx::database::Database>::ArgumentBuffer<'q>,
177            ) -> ::std::result::Result<::sqlx::encode::IsNull, ::sqlx::error::BoxDynError> {
178                let val = match self {
179                    #(#value_arms)*
180                };
181
182                <&::std::primitive::str as ::sqlx::encode::Encode<'q, DB>>::encode(val, buf)
183            }
184
185            fn size_hint(&self) -> ::std::primitive::usize {
186                let val = match self {
187                    #(#value_arms)*
188                };
189
190                <&::std::primitive::str as ::sqlx::encode::Encode<'q, DB>>::size_hint(&val)
191            }
192        }
193    ))
194}

196fn expand_derive_encode_struct(
197    input: &DeriveInput,
198    fields: &Punctuated<Field, Comma>,
199) -> syn::Result<TokenStream> {
200    check_struct_attributes(input, fields)?;
201
202    let mut tts = TokenStream::new();
203
204    if cfg!(feature = "postgres") {
205        let ident = &input.ident;
206        let column_count = fields.len();
207
208        let (_, ty_generics, where_clause) = input.generics.split_for_impl();
209
210        let mut generics = input.generics.clone();
211
212        // add db type for impl generics & where clause
213        for type_param in &mut generics.type_params_mut() {
214            type_param.bounds.extend::<[TypeParamBound; 2]>([
215                parse_quote!(for<'encode> ::sqlx::encode::Encode<'encode, ::sqlx::Postgres>),
216                parse_quote!(::sqlx::types::Type<::sqlx::Postgres>),
217            ]);
218        }
219
220        generics.params.push(parse_quote!('q));
221
222        let (impl_generics, _, _) = generics.split_for_impl();
223
224        let writes = fields.iter().map(|field| -> Stmt {
225            let id = &field.ident;
226
227            parse_quote!(
228                encoder.encode(&self. #id)?;
229            )
230        });
231
232        let sizes = fields.iter().map(|field| -> Expr {
233            let id = &field.ident;
234            let ty = &field.ty;
235
236            parse_quote!(
237                <#ty as ::sqlx::encode::Encode<::sqlx::Postgres>>::size_hint(&self. #id)
238            )
239        });
240
241        tts.extend(quote!(
242            #[automatically_derived]
243            impl #impl_generics ::sqlx::encode::Encode<'_, ::sqlx::Postgres> for #ident #ty_generics
244            #where_clause
245            {
246                fn encode_by_ref(
247                    &self,
248                    buf: &mut ::sqlx::postgres::PgArgumentBuffer,
249                ) -> ::std::result::Result<::sqlx::encode::IsNull, ::sqlx::error::BoxDynError> {
250                    let mut encoder = ::sqlx::postgres::types::PgRecordEncoder::new(buf);
251
252                    #(#writes)*
253
254                    encoder.finish();
255
256                    ::std::result::Result::Ok(::sqlx::encode::IsNull::No)
257                }
258
259                fn size_hint(&self) -> ::std::primitive::usize {
260                    #column_count * (4 + 4) // oid (int) and length (int) for each column
261                        + #(#sizes)+* // sum of the size hints for each column
262                }
263            }
264        ));
265    }
266
267    Ok(tts)
268}
