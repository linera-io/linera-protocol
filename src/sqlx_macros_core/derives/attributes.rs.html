8macro_rules! assert_attribute {
9    ($e:expr, $err:expr, $input:expr) => {
10        if !$e {
11            return Err(syn::Error::new_spanned($input, $err));
12        }
13    };
14}

16macro_rules! fail {
17    ($t:expr, $m:expr) => {
18        return Err(syn::Error::new_spanned($t, $m))
19    };

22macro_rules! try_set {
23    ($i:ident, $v:expr, $t:expr) => {
24        match $i {
25            None => $i = Some($v),
26            Some(_) => fail!($t, "duplicate attribute"),
27        }
28    };
29}

78pub fn parse_container_attributes(input: &[Attribute]) -> syn::Result<SqlxContainerAttributes> {
79    let mut transparent = None;
80    let mut repr = None;
81    let mut type_name = None;
82    let mut rename_all = None;
83    let mut no_pg_array = None;
84    let mut default = None;
85
86    for attr in input {
87        if attr.path().is_ident("sqlx") {
88            attr.parse_nested_meta(|meta| {
89                if meta.path.is_ident("transparent") {
90                    try_set!(transparent, true, attr);
91                } else if meta.path.is_ident("no_pg_array") {
92                    try_set!(no_pg_array, true, attr);
93                } else if meta.path.is_ident("default") {
94                    try_set!(default, true, attr);
95                } else if meta.path.is_ident("rename_all") {
96                    meta.input.parse::<Token![=]>()?;
97                    let lit: LitStr = meta.input.parse()?;
98
99                    let val = match lit.value().as_str() {
100                        "lowercase" => RenameAll::LowerCase,
101                        "snake_case" => RenameAll::SnakeCase,
102                        "UPPERCASE" => RenameAll::UpperCase,
103                        "SCREAMING_SNAKE_CASE" => RenameAll::ScreamingSnakeCase,
104                        "kebab-case" => RenameAll::KebabCase,
105                        "camelCase" => RenameAll::CamelCase,
106                        "PascalCase" => RenameAll::PascalCase,
107                        _ => fail!(lit, "unexpected value for rename_all"),
108                    };
109
110                    try_set!(rename_all, val, lit)
111                } else if meta.path.is_ident("type_name") {
112                    meta.input.parse::<Token![=]>()?;
113                    let lit: LitStr = meta.input.parse()?;
114                    let name = TypeName {
115                        val: lit.value(),
116                        span: lit.span(),
117                    };
118
119                    try_set!(type_name, name, lit)
120                } else {
121                    fail!(meta.path, "unexpected attribute")
122                }
123
124                Ok(())
125            })?;
126        } else if attr.path().is_ident("repr") {
127            let list: Punctuated<Meta, Token![,]> =
128                attr.parse_args_with(<Punctuated<Meta, Token![,]>>::parse_terminated)?;
129
130            if let Some(path) = list.iter().find_map(|f| f.require_path_only().ok()) {
131                try_set!(repr, path.get_ident().unwrap().clone(), list);
132            }
133        }
134    }
135
136    Ok(SqlxContainerAttributes {
137        transparent: transparent.unwrap_or(false),
138        repr,
139        type_name,
140        rename_all,
141        no_pg_array: no_pg_array.unwrap_or(false),
142        default: default.unwrap_or(false),
143    })
144}

146pub fn parse_child_attributes(input: &[Attribute]) -> syn::Result<SqlxChildAttributes> {
147    let mut rename = None;
148    let mut default = false;
149    let mut try_from = None;
150    let mut flatten = false;
151    let mut skip: bool = false;
152    let mut json = None;
153
154    for attr in input.iter().filter(|a| a.path().is_ident("sqlx")) {
155        attr.parse_nested_meta(|meta| {
156            if meta.path.is_ident("rename") {
157                meta.input.parse::<Token![=]>()?;
158                let val: LitStr = meta.input.parse()?;
159                try_set!(rename, val.value(), val);
160            } else if meta.path.is_ident("try_from") {
161                meta.input.parse::<Token![=]>()?;
162                let val: LitStr = meta.input.parse()?;
163                try_set!(try_from, val.parse()?, val);
164            } else if meta.path.is_ident("default") {
165                default = true;
166            } else if meta.path.is_ident("flatten") {
167                flatten = true;
168            } else if meta.path.is_ident("skip") {
169                skip = true;
170            } else if meta.path.is_ident("json") {
171                if meta.input.peek(syn::token::Paren) {
172                    let content;
173                    parenthesized!(content in meta.input);
174                    let literal: Ident = content.parse()?;
175                    assert_eq!(literal.to_string(), "nullable", "Unrecognized `json` attribute. Valid values are `json` or `json(nullable)`");
176                    json = Some(JsonAttribute::Nullable);
177                } else {
178                    json = Some(JsonAttribute::NonNullable);
179                }
180            }
181
182            Ok(())
183        })?;
184
185        if json.is_some() && flatten {
186            fail!(
187                attr,
188                "Cannot use `json` and `flatten` together on the same field"
189            );
190        }
191    }
192
193    Ok(SqlxChildAttributes {
194        rename,
195        default,
196        flatten,
197        try_from,
198        skip,
199        json,
200    })
201}

203pub fn check_transparent_attributes(
204    input: &DeriveInput,
205    field: &Field,
206) -> syn::Result<SqlxContainerAttributes> {
207    let attributes = parse_container_attributes(&input.attrs)?;
208
209    assert_attribute!(
210        attributes.rename_all.is_none(),
211        "unexpected #[sqlx(rename_all = ..)]",
212        field
213    );
214
215    let ch_attributes = parse_child_attributes(&field.attrs)?;
216
217    assert_attribute!(
218        ch_attributes.rename.is_none(),
219        "unexpected #[sqlx(rename = ..)]",
220        field
221    );
222
223    Ok(attributes)
224}

238pub fn check_weak_enum_attributes(
