45pub enum QueryDataSource<'a> {
46    Live {
47        database_url: &'a str,
48        database_url_parsed: Url,
49    },
50    Cached(DynQueryData),
51}
52
53impl<'a> QueryDataSource<'a> {
54    pub fn live(database_url: &'a str) -> crate::Result<Self> {
55        Ok(QueryDataSource::Live {
56            database_url,
57            database_url_parsed: database_url.parse()?,
58        })
59    }
60
61    pub fn matches_driver(&self, driver: &QueryDriver) -> bool {
62        match self {
63            Self::Live {
64                database_url_parsed,
65                ..
66            } => driver.url_schemes.contains(&database_url_parsed.scheme()),
67            Self::Cached(dyn_data) => dyn_data.db_name == driver.db_name,
68        }
69    }
70}

72struct Metadata {
73    #[allow(unused)]
74    manifest_dir: PathBuf,
75    offline: bool,
76    database_url: Option<String>,
77    offline_dir: Option<String>,
78    workspace_root: Arc<Mutex<Option<PathBuf>>>,
79}
80
81impl Metadata {
82    pub fn workspace_root(&self) -> PathBuf {
83        let mut root = self.workspace_root.lock().unwrap();
84        if root.is_none() {
85            use serde::Deserialize;
86            use std::process::Command;
87
88            let cargo = env("CARGO").expect("`CARGO` must be set");
89
90            let output = Command::new(cargo)
91                .args(["metadata", "--format-version=1", "--no-deps"])
92                .current_dir(&self.manifest_dir)
93                .env_remove("__CARGO_FIX_PLZ")
94                .output()
95                .expect("Could not fetch metadata");
96
97            #[derive(Deserialize)]
98            struct CargoMetadata {
99                workspace_root: PathBuf,
100            }
101
102            let metadata: CargoMetadata =
103                serde_json::from_slice(&output.stdout).expect("Invalid `cargo metadata` output");
104
105            *root = Some(metadata.workspace_root);
106        }
107        root.clone().unwrap()
108    }
109}

111static METADATA: Lazy<Mutex<HashMap<String, Metadata>>> = Lazy::new(Default::default);

113// If we are in a workspace, lookup `workspace_root` since `CARGO_MANIFEST_DIR` won't
114// reflect the workspace dir: https://github.com/rust-lang/cargo/issues/3946
115fn init_metadata(manifest_dir: &String) -> Metadata {
116    let manifest_dir: PathBuf = manifest_dir.into();
117
118    let (database_url, offline, offline_dir) = load_dot_env(&manifest_dir);
119
120    let offline = env("SQLX_OFFLINE")
121        .ok()
122        .or(offline)
123        .map(|s| s.eq_ignore_ascii_case("true") || s == "1")
124        .unwrap_or(false);
125
126    let database_url = env("DATABASE_URL").ok().or(database_url);
127
128    Metadata {
129        manifest_dir,
130        offline,
131        database_url,
132        offline_dir,
133        workspace_root: Arc::new(Mutex::new(None)),
134    }
135}

137pub fn expand_input<'a>(
138    input: QueryMacroInput,
139    drivers: impl IntoIterator<Item = &'a QueryDriver>,
140) -> crate::Result<TokenStream> {
141    let manifest_dir = env("CARGO_MANIFEST_DIR").expect("`CARGO_MANIFEST_DIR` must be set");
142
143    let mut metadata_lock = METADATA
144        .lock()
145        // Just reset the metadata on error
146        .unwrap_or_else(|poison_err| {
147            let mut guard = poison_err.into_inner();
148            *guard = Default::default();
149            guard
150        });
151
152    let metadata = metadata_lock
153        .entry(manifest_dir)
154        .or_insert_with_key(init_metadata);
155
156    let data_source = match &metadata {
157        Metadata {
158            offline: false,
159            database_url: Some(db_url),
160            ..
161        } => QueryDataSource::live(db_url)?,
162        Metadata { offline, .. } => {
163            // Try load the cached query metadata file.
164            let filename = format!("query-{}.json", hash_string(&input.sql));
165
166            // Check SQLX_OFFLINE_DIR, then local .sqlx, then workspace .sqlx.
167            let dirs = [
168                |meta: &Metadata| meta.offline_dir.as_deref().map(PathBuf::from),
169                |meta: &Metadata| Some(meta.manifest_dir.join(".sqlx")),
170                |meta: &Metadata| Some(meta.workspace_root().join(".sqlx")),
171            ];
172            let Some(data_file_path) = dirs
173                .iter()
174                .filter_map(|path| path(metadata))
175                .map(|path| path.join(&filename))
176                .find(|path| path.exists())
177            else {
178                return Err(
179                    if *offline {
180                        "`SQLX_OFFLINE=true` but there is no cached data for this query, run `cargo sqlx prepare` to update the query cache or unset `SQLX_OFFLINE`"
181                    } else {
182                        "set `DATABASE_URL` to use query macros online, or run `cargo sqlx prepare` to update the query cache"
183                    }.into()
184                );
185            };
186            QueryDataSource::Cached(DynQueryData::from_data_file(&data_file_path, &input.sql)?)
187        }
188    };
189
190    for driver in drivers {
191        if data_source.matches_driver(driver) {
192            return (driver.expand)(input, data_source);
193        }
194    }
195
196    match data_source {
197        QueryDataSource::Live {
198            database_url_parsed,
199            ..
200        } => Err(format!(
201            "no database driver found matching URL scheme {:?}; the corresponding Cargo feature may need to be enabled", 
202            database_url_parsed.scheme()
203        ).into()),
204        QueryDataSource::Cached(data) => {
205            Err(format!(
206                "found cached data for database {:?} but no matching driver; the corresponding Cargo feature may need to be enabled",
207                data.db_name
208            ).into())
209        }
210    }
211}

213fn expand_with<DB: DatabaseExt>(
214    input: QueryMacroInput,
215    data_source: QueryDataSource,
216) -> crate::Result<TokenStream>
217where
218    Describe<DB>: DescribeExt,
219{
220    let (query_data, offline): (QueryData<DB>, bool) = match data_source {
221        QueryDataSource::Cached(dyn_data) => (QueryData::from_dyn_data(dyn_data)?, true),
222        QueryDataSource::Live { database_url, .. } => {
223            let describe = DB::describe_blocking(&input.sql, database_url)?;
224            (QueryData::from_describe(&input.sql, describe), false)
225        }
226    };
227
228    expand_with_data(input, query_data, offline)
229}

239fn expand_with_data<DB: DatabaseExt>(
240    input: QueryMacroInput,
241    data: QueryData<DB>,
242    offline: bool,
243) -> crate::Result<TokenStream>
244where
245    Describe<DB>: DescribeExt,
246{
247    // validate at the minimum that our args match the query's input parameters
248    let num_parameters = match data.describe.parameters() {
249        Some(Either::Left(params)) => Some(params.len()),
250        Some(Either::Right(num)) => Some(num),
251
252        None => None,
253    };
254
255    if let Some(num) = num_parameters {
256        if num != input.arg_exprs.len() {
257            return Err(
258                format!("expected {} parameters, got {}", num, input.arg_exprs.len()).into(),
259            );
260        }
261    }
262
263    let args_tokens = args::quote_args(&input, &data.describe)?;
264
265    let query_args = format_ident!("query_args");
266
267    let output = if data
268        .describe
269        .columns()
270        .iter()
271        .all(|it| it.type_info().is_void())
272    {
273        let db_path = DB::db_path();
274        let sql = &input.sql;
275
276        quote! {
277            ::sqlx::__query_with_result::<#db_path, _>(#sql, #query_args)
278        }
279    } else {
280        match input.record_type {
281            RecordType::Generated => {
282                let columns = output::columns_to_rust::<DB>(&data.describe)?;
283
284                let record_name: Type = syn::parse_str("Record").unwrap();
285
286                for rust_col in &columns {
287                    if rust_col.type_.is_wildcard() {
288                        return Err(
289                            "wildcard overrides are only allowed with an explicit record type, \
290                             e.g. `query_as!()` and its variants"
291                                .into(),
292                        );
293                    }
294                }
295
296                let record_fields = columns
297                    .iter()
298                    .map(|output::RustColumn { ident, type_, .. }| quote!(#ident: #type_,));
299
300                let mut record_tokens = quote! {
301                    #[derive(Debug)]
302                    #[allow(non_snake_case)]
303                    struct #record_name {
304                        #(#record_fields)*
305                    }
306                };
307
308                record_tokens.extend(output::quote_query_as::<DB>(
309                    &input,
310                    &record_name,
311                    &query_args,
312                    &columns,
313                ));
314
315                record_tokens
316            }
317            RecordType::Given(ref out_ty) => {
318                let columns = output::columns_to_rust::<DB>(&data.describe)?;
319
320                output::quote_query_as::<DB>(&input, out_ty, &query_args, &columns)
321            }
322            RecordType::Scalar => {
323                output::quote_query_scalar::<DB>(&input, &query_args, &data.describe)?
324            }
325        }
326    };
327
328    let ret_tokens = quote! {
329        {
330            #[allow(clippy::all)]
331            {
332                use ::sqlx::Arguments as _;
333
334                #args_tokens
335
336                #output
337            }
338        }
339    };
340
341    // Store query metadata only if offline support is enabled but the current build is online.
342    // If the build is offline, the cache is our input so it's pointless to also write data for it.
343    if !offline {
344        // Only save query metadata if SQLX_OFFLINE_DIR is set manually or by `cargo sqlx prepare`.
345        // Note: in a cargo workspace this path is relative to the root.
346        if let Ok(dir) = env("SQLX_OFFLINE_DIR") {
347            let path = PathBuf::from(&dir);
348
349            match fs::metadata(&path) {
350                Err(e) => {
351                    if e.kind() != io::ErrorKind::NotFound {
352                        // Can't obtain information about .sqlx
353                        return Err(format!("{e}: {dir}").into());
354                    }
355                    // .sqlx doesn't exist.
356                    return Err(format!("sqlx offline path does not exist: {dir}").into());
357                }
358                Ok(meta) => {
359                    if !meta.is_dir() {
360                        return Err(format!(
361                            "sqlx offline path exists, but is not a directory: {dir}"
362                        )
363                        .into());
364                    }
365
366                    // .sqlx exists and is a directory, store data.
367                    data.save_in(path)?;
368                }
369            }
370        }
371    }
372
373    Ok(ret_tokens)
374}

376/// Get the value of an environment variable, telling the compiler about it if applicable.
377fn env(name: &str) -> Result<String, std::env::VarError> {
378    #[cfg(procmacro2_semver_exempt)]
379    {
380        proc_macro::tracked_env::var(name)
381    }
382
383    #[cfg(not(procmacro2_semver_exempt))]
384    {
385        std::env::var(name)
386    }
387}

389/// Get `DATABASE_URL`, `SQLX_OFFLINE` and `SQLX_OFFLINE_DIR` from the `.env`.
390fn load_dot_env(manifest_dir: &Path) -> (Option<String>, Option<String>, Option<String>) {
391    let mut env_path = manifest_dir.join(".env");
392
393    // If a .env file exists at CARGO_MANIFEST_DIR, load environment variables from this,
394    // otherwise fallback to default dotenv file.
395    #[cfg_attr(not(procmacro2_semver_exempt), allow(unused_variables))]
396    let env_file = if env_path.exists() {
397        let res = dotenvy::from_path_iter(&env_path);
398        match res {
399            Ok(iter) => Some(iter),
400            Err(e) => panic!("failed to load environment from {env_path:?}, {e}"),
401        }
402    } else {
403        #[allow(unused_assignments)]
404        {
405            env_path = PathBuf::from(".env");
406        }
407        dotenvy::dotenv_iter().ok()
408    };
409
410    let mut offline = None;
411    let mut database_url = None;
412    let mut offline_dir = None;
413
414    if let Some(env_file) = env_file {
415        // tell the compiler to watch the `.env` for changes.
416        #[cfg(procmacro2_semver_exempt)]
417        if let Some(env_path) = env_path.to_str() {
418            proc_macro::tracked_path::path(env_path);
419        }
420
421        for item in env_file {
422            let Ok((key, value)) = item else {
423                continue;
424            };
425
426            match key.as_str() {
427                "DATABASE_URL" => database_url = Some(value),
428                "SQLX_OFFLINE" => offline = Some(value),
429                "SQLX_OFFLINE_DIR" => offline_dir = Some(value),
430                _ => {}
431            };
432        }
433    }
434
435    (database_url, offline, offline_dir)
