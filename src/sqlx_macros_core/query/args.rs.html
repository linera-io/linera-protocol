10/// Returns a tokenstream which typechecks the arguments passed to the macro
11/// and binds them to `DB::Arguments` with the ident `query_args`.
12pub fn quote_args<DB: DatabaseExt>(
13    input: &QueryMacroInput,
14    info: &Describe<DB>,
15) -> crate::Result<TokenStream> {
16    let db_path = DB::db_path();
17
18    if input.arg_exprs.is_empty() {
19        return Ok(quote! {
20            let query_args = ::core::result::Result::<_, ::sqlx::error::BoxDynError>::Ok(<#db_path as ::sqlx::database::Database>::Arguments::<'_>::default());
21        });
22    }
23
24    let arg_names = (0..input.arg_exprs.len())
25        .map(|i| format_ident!("arg{}", i))
26        .collect::<Vec<_>>();
27
28    let arg_name = &arg_names;
29    let arg_expr = input.arg_exprs.iter().cloned().map(strip_wildcard);
30
31    let arg_bindings = quote! {
32        #(let #arg_name = &(#arg_expr);)*
33    };
34
35    let args_check = match info.parameters() {
36        None | Some(Either::Right(_)) => {
37            // all we can do is check arity which we did
38            TokenStream::new()
39        }
40
41        Some(Either::Left(_)) if !input.checked => {
42            // this is an `*_unchecked!()` macro invocation
43            TokenStream::new()
44        }
45
46        Some(Either::Left(params)) => {
47            params
48                .iter()
49                .zip(arg_names.iter().zip(&input.arg_exprs))
50                .enumerate()
51                .map(|(i, (param_ty, (name, expr)))| -> crate::Result<_> {
52                    if get_type_override(expr).is_some() {
53                        // cast will fail to compile if the type does not match
54                        // and we strip casts to wildcard
55                        return Ok(quote!());
56                    }
57
58                    let param_ty =
59                        DB::param_type_for_id(param_ty)
60                            .ok_or_else(|| {
61                                if let Some(feature_gate) = DB::get_feature_gate(param_ty) {
62                                    format!(
63                                        "optional sqlx feature `{}` required for type {} of param #{}",
64                                        feature_gate,
65                                        param_ty,
66                                        i + 1,
67                                    )
68                                } else {
69                                    format!(
70                                        "no built in mapping found for type {} for param #{}; \
71                                        a type override may be required, see documentation for details",
72                                        param_ty,
73                                        i + 1
74                                    )
75                                }
76                            })?
77                            .parse::<TokenStream>()
78                            .map_err(|_| format!("Rust type mapping for {param_ty} not parsable"))?;
79
80                    Ok(quote_spanned!(expr.span() =>
81                        // this shouldn't actually run
82                        #[allow(clippy::missing_panics_doc, clippy::unreachable)]
83                        if false {
84                            use ::sqlx::ty_match::{WrapSameExt as _, MatchBorrowExt as _};
85
86                            // evaluate the expression only once in case it contains moves
87                            let expr = ::sqlx::ty_match::dupe_value(#name);
88
89                            // if `expr` is `Option<T>`, get `Option<$ty>`, otherwise `$ty`
90                            let ty_check = ::sqlx::ty_match::WrapSame::<#param_ty, _>::new(&expr).wrap_same();
91
92                            // if `expr` is `&str`, convert `String` to `&str`
93                            let (mut _ty_check, match_borrow) = ::sqlx::ty_match::MatchBorrow::new(ty_check, &expr);
94
95                            _ty_check = match_borrow.match_borrow();
96
97                            // this causes move-analysis to effectively ignore this block
98                            ::std::unreachable!();
99                        }
100                    ))
101                })
102                .collect::<crate::Result<TokenStream>>()?
103        }
104    };
105
106    let args_count = input.arg_exprs.len();
107
108    Ok(quote! {
109        #arg_bindings
110
111        #args_check
112
113        let mut query_args = <#db_path as ::sqlx::database::Database>::Arguments::<'_>::default();
114        query_args.reserve(
115            #args_count,
116            0 #(+ ::sqlx::encode::Encode::<#db_path>::size_hint(#arg_name))*
117        );
118        let query_args = ::core::result::Result::<_, ::sqlx::error::BoxDynError>::Ok(query_args)
119        #(.and_then(move |mut query_args| query_args.add(#arg_name).map(move |()| query_args) ))*;
120    })
121}

123fn get_type_override(expr: &Expr) -> Option<&Type> {
124    match expr {
125        Expr::Group(group) => get_type_override(&group.expr),
126        Expr::Cast(cast) => Some(&cast.ty),
127        _ => None,
128    }
129}

131fn strip_wildcard(expr: Expr) -> Expr {
132    match expr {
133        Expr::Group(ExprGroup {
134            attrs,
135            group_token,
136            expr,
137        }) => Expr::Group(ExprGroup {
138            attrs,
139            group_token,
140            expr: Box::new(strip_wildcard(*expr)),
141        }),
142        // we want to retain casts if they semantically matter
143        Expr::Cast(ExprCast {
144            attrs,
145            expr,
146            as_token,
147            ty,
148        }) => match *ty {
149            // cast to wildcard `_` will produce weird errors; we interpret it as taking the value as-is
150            Type::Infer(_) => *expr,
151            _ => Expr::Cast(ExprCast {
152                attrs,
153                expr,
154                as_token,
155                ty,
156            }),
157        },
158        _ => expr,
159    }
