68static OFFLINE_DATA_CACHE: Lazy<Mutex<HashMap<PathBuf, DynQueryData>>> =
69    Lazy::new(Default::default);
70
71/// Offline query data
72#[derive(Clone, serde::Deserialize)]
73pub struct DynQueryData {
74    pub db_name: String,
75    pub query: String,
76    pub describe: serde_json::Value,
77    pub hash: String,
78}
79
80impl DynQueryData {
81    /// Loads a query given the path to its "query-<hash>.json" file. Subsequent calls for the same
82    /// path are retrieved from an in-memory cache.
83    pub fn from_data_file(path: impl AsRef<Path>, query: &str) -> crate::Result<Self> {
84        let path = path.as_ref();
85
86        let mut cache = OFFLINE_DATA_CACHE
87            .lock()
88            // Just reset the cache on error
89            .unwrap_or_else(|poison_err| {
90                let mut guard = poison_err.into_inner();
91                *guard = Default::default();
92                guard
93            });
94        if let Some(cached) = cache.get(path).cloned() {
95            if query != cached.query {
96                return Err("hash collision for saved query data".into());
97            }
98            return Ok(cached);
99        }
100
101        #[cfg(procmacro2_semver_exempt)]
102        {
103            let path = path.as_ref().canonicalize()?;
104            let path = path.to_str().ok_or_else(|| {
105                format!(
106                    "query-<hash>.json path cannot be represented as a string: {:?}",
107                    path
108                )
109            })?;
110
111            proc_macro::tracked_path::path(path);
112        }
113
114        let offline_data_contents = fs::read_to_string(path)
115            .map_err(|e| format!("failed to read saved query path {}: {}", path.display(), e))?;
116        let dyn_data: DynQueryData = serde_json::from_str(&offline_data_contents)?;
117
118        if query != dyn_data.query {
119            return Err("hash collision for saved query data".into());
120        }
121
122        let _ = cache.insert(path.to_owned(), dyn_data.clone());
123        Ok(dyn_data)
124    }
125}

127impl<DB: DatabaseExt> QueryData<DB>
128where
129    Describe<DB>: serde::Serialize + serde::de::DeserializeOwned,
130{
131    pub fn from_dyn_data(dyn_data: DynQueryData) -> crate::Result<Self> {
132        assert!(!dyn_data.db_name.is_empty());
133        assert!(!dyn_data.hash.is_empty());
134
135        if DB::NAME == dyn_data.db_name {
136            let describe: Describe<DB> = serde_json::from_value(dyn_data.describe)?;
137            Ok(QueryData {
138                db_name: SerializeDbName::default(),
139                query: dyn_data.query,
140                describe,
141                hash: dyn_data.hash,
142            })
143        } else {
144            Err(format!(
145                "expected query data for {}, got data for {}",
146                DB::NAME,
147                dyn_data.db_name
148            )
149            .into())
150        }
151    }
152
153    pub(super) fn save_in(&self, dir: impl AsRef<Path>) -> crate::Result<()> {
154        use std::io::ErrorKind;
155
156        let path = dir.as_ref().join(format!("query-{}.json", self.hash));
157        match std::fs::remove_file(&path) {
158            Ok(()) => {}
159            Err(err)
160                if matches!(
161                    err.kind(),
162                    ErrorKind::NotFound | ErrorKind::PermissionDenied,
163                ) => {}
164            Err(err) => return Err(format!("failed to delete {path:?}: {err:?}").into()),
165        }
166        let mut file = match std::fs::OpenOptions::new()
167            .write(true)
168            .create_new(true)
169            .open(&path)
170        {
171            Ok(file) => file,
172            // We overlapped with a concurrent invocation and the other one succeeded.
173            Err(err) if matches!(err.kind(), ErrorKind::AlreadyExists) => return Ok(()),
174            Err(err) => {
175                return Err(format!("failed to exclusively create {path:?}: {err:?}").into())
176            }
177        };
178
179        let data = serde_json::to_string_pretty(self)
180            .map_err(|err| format!("failed to serialize query data: {err:?}"))?;
181        file.write_all(data.as_bytes())
182            .map_err(|err| format!("failed to write query data to file: {err:?}"))?;
183
184        // Ensure there is a newline at the end of the JSON file to avoid
185        // accidental modification by IDE and make github diff tool happier.
186        file.write_all(b"\n")
187            .map_err(|err| format!("failed to append a newline to file: {err:?}"))?;
188
189        Ok(())
190    }
191}

193pub(super) fn hash_string(query: &str) -> String {
194    // picked `sha2` because it's already in the dependency tree for both MySQL and Postgres
195    use sha2::{Digest, Sha256};
196
197    hex::encode(Sha256::digest(query.as_bytes()))
198}
