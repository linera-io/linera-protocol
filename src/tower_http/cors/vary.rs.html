1use std::array;

3use http::header::{self, HeaderName, HeaderValue};
4
5use super::preflight_request_headers;

7/// Holds configuration for how to set the [`Vary`][mdn] header.
8///
9/// See [`CorsLayer::vary`] for more details.
10///
11/// [mdn]: https://developer.mozilla.org/en-US/docs/Web/HTTP/Headers/Vary
12/// [`CorsLayer::vary`]: super::CorsLayer::vary
13#[derive(Clone, Debug)]
14pub struct Vary(Vec<HeaderValue>);
15
16impl Vary {
17    /// Set the list of header names to return as vary header values
18    ///
19    /// See [`CorsLayer::vary`] for more details.
20    ///
21    /// [`CorsLayer::vary`]: super::CorsLayer::vary
22    pub fn list<I>(headers: I) -> Self
23    where
24        I: IntoIterator<Item = HeaderName>,
25    {
26        Self(headers.into_iter().map(Into::into).collect())
27    }
28
29    pub(super) fn to_header(&self) -> Option<(HeaderName, HeaderValue)> {
30        let values = &self.0;
31        let mut res = values.first()?.as_bytes().to_owned();
32        for val in &values[1..] {
33            res.extend_from_slice(b", ");
34            res.extend_from_slice(val.as_bytes());
35        }
36
37        let header_val = HeaderValue::from_bytes(&res)
38            .expect("comma-separated list of HeaderValues is always a valid HeaderValue");
39        Some((header::VARY, header_val))
40    }
41}
42
43impl Default for Vary {
44    fn default() -> Self {
45        Self::list(preflight_request_headers())
46    }
47}
48
49impl<const N: usize> From<[HeaderName; N]> for Vary {
50    fn from(arr: [HeaderName; N]) -> Self {
51        #[allow(deprecated)] // Can be changed when MSRV >= 1.53
52        Self::list(array::IntoIter::new(arr))
53    }
54}
55
56impl From<Vec<HeaderName>> for Vary {
57    fn from(vec: Vec<HeaderName>) -> Self {
58        Self::list(vec)
59    }
