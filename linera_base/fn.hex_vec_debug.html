Applies hex_debug to a slice of byte vectors.

ยงExamples

use custom_debug_derive::Debug;

#[derive(Debug)]
struct Messages {
    #[debug(with = "hex_vec_debug")]
    byte_vecs: Vec<Vec<u8>>,
}

let msgs = Messages {
    byte_vecs: vec![vec![0x12, 0x34, 0x56, 0x78], vec![0x9A]],
};

assert_eq!(
    format!("{:?}", msgs),
    "Messages { byte_vecs: [12345678, 9a] }"
);
