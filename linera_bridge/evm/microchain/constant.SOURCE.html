This is safe because `ConfirmedBlockCertificate`\n    /// implies BFT-finalized canonicality \u{2014} a quorum of validators signed this\n    /// specific block at this height, so no conflicting block can exist.\n    /// If this assumption ever changes at the protocol layer, a\n    /// `previous_block_hash` check should be added here.\n    function addBlock(bytes calldata data) external {\n        (BridgeTypes.Block memory blockValue, bytes32 signedHash) = lightClient.verifyBlock(data);\n\n        require(!verifiedBlocks[signedHash], \"block already verified\");\n        require(blockValue.header.chain_id.value.value == chainId, \"chain id mismatch\");\n        require(blockValue.header.height.value == nextExpectedHeight, \"block height must be sequential\");\n\n        nextExpectedHeight = blockValue.header.height.value + 1;\n        verifiedBlocks[signedHash] = true;\n        _onBlock(blockValue);\n    }\n\n    /// Called after a block has been verified and accepted. Subcontracts implement\n    /// this to extract and store application-specific data from the block.\n    function _onBlock(BridgeTypes.Block memory blockValue) internal virtual;\n}\n";
