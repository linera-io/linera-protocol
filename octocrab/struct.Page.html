Expand description

A Page of GitHub results, with links to the next and previous page.

let octocrab = octocrab::instance();

// Print the titles of the first page of issues.
for issue in octocrab.issues("rust-lang", "rust").list().send().await? {
    println!("{}", issue.title);
}
